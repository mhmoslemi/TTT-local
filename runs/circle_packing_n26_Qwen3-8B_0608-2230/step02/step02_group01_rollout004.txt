import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based placement (hexagonal-like within the square)
    n = 26
    centers = np.zeros((n, 2))
    
    # First pass: place circles in a grid-like pattern
    # We'll distribute them approximately equally across the square
    x = np.linspace(0.1, 0.9, n)
    y = np.linspace(0.1, 0.9, n)
    centers = np.array([x, y]).T
    centers = np.reshape(centers, (n, 2))
    
    # Randomize for better exploration
    np.random.shuffle(centers)
    
    # Initial guess for radii
    radii = np.full(n, 0.03)  # Small starting radii
    
    # Define objective function to maximize sum of radii
    def objective(radii, centers, n):
        return -np.sum(radii)  # Minimize negative sum for maximization

    # Define constraints (non-overlapping, within bounds)
    def constraint_overlap(radii, centers):
        dist = cdist(centers, centers)
        for i in range(n):
            for j in range(i + 1, n):
                if dist[i, j] < radii[i] + radii[j] - 1e-8:
                    return -1  # Violation
        return 0

    def constraint_boundary(radii, centers):
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-8 or x + r > 1.0 + 1e-8 or
                y - r < -1e-8 or y + r > 1.0 + 1e-8):
                return -1  # Violation
        return 0

    # Combine constraints
    def constraint(radii, centers):
        return constraint_overlap(radii, centers) + constraint_boundary(radii, centers)

    # Define optimization bounds and constraints
    bounds = [(0.0, 0.1) for _ in range(n)]

    # Optimization problem
    result = minimize(
        fun=lambda x: -objective(x, centers, n),
        x0=radii,
        bounds=bounds,
        method='L-BFGS-B',
        constraints={'type': 'ineq', 'fun': lambda x: -constraint(x, centers)},
        options={'maxiter': 500, 'disp': False}

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
