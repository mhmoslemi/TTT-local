def create_hexagonal_grid(n: int) -> Tuple[np.ndarray, np.ndarray]:
    """
    Creates a hexagonal grid of circles packed within a unit square.
    """
    # Determine number of rows and columns for hexagonal packing
    if n == 1:
        return np.array([[0.5, 0.5]]), np.array([0.5])
    
    # Determine number of rows and columns for hexagonal packing
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Determine spacing
    spacing = 1.0 / (cols + 1)
    
    # Create grid
    centers = np.zeros((n, 2))
    index = 0
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                x = j * spacing * 2
            else:
                x = (j + 0.5) * spacing * 2
            y = i * spacing * np.sqrt(3)
            centers[index] = [x, y]
            index += 1
            if index >= n:
                break
        if index >= n:
            break
    
    # Resize to fit in the unit square and adjust radii to be uniform
    max_distance = np.sqrt((spacing * 2)**2 + (spacing * np.sqrt(3))**2)
    radii = np.ones(n) * (max_distance / 2.0) * 0.9
    radii[0] = 0.4  # Adjust to allow more circles near the edges

    return centers, radii
