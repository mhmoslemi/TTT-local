import numpy as np
import scipy.optimize as opt
import scipy.spatial.distance as dist

def run_packing():
    # Initial guess: a 5x5 grid with slight perturbation for hexagonal optimization
    n = 26
    grid_rows, grid_cols = 5, 5
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Generate a hexagonal grid
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j + (i % 2) * 0.5
            y = i * np.sqrt(3) / 2
            if i + j >= n:  # Ensure we don't exceed the number of circles
                break
            centers[i * grid_cols + j] = [x, y]

    # Adjust positions for the unit square
    centers[:, 0] /= np.max(centers[:, 0])
    centers[:, 1] /= np.max(centers[:, 1])

    # Initial uniform radii
    min_radius = 0.01
    max_radius = 0.1
    radii = np.full(n, (max_radius - min_radius) * np.random.rand(n) + min_radius)

    # Define objective function to maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    # Define constraints
    def constraint_positions(x, centers):
        return x - np.concatenate(centers).ravel()

    def constraint_radii(x):
        return x

    def constraint_boundary(x):
        return np.concatenate([x - 1e-12, 1 - x - 1e-12])

    def constraint_overlap(x):
        idx = np.arange(n)
        radii = x[idx]
        centers = x[idx + n]
        dist_matrix = dist.cdist(centers, centers)
        overlap = np.where(dist_matrix < radii[:, np.newaxis] + radii[np.newaxis, :], 1, 0)
        return np.sum(overlap)

    # Flatten the variables: [radii, centers]
    x0 = np.concatenate([radii, centers.ravel()])

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': lambda x: constraint_positions(x, centers)},
        {'type': 'ineq', 'fun': lambda x: constraint_radii(x)},
        {'type': 'ineq', 'fun': lambda x: constraint_boundary(x)},
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x)}
    ]

    # Bounds
    bounds = []
    for _ in range(n):
        bounds.append((0, 1))
    for _ in range(2 * n):
        bounds.append((0, 1))

    # Run optimization
    result = opt.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        options={'maxiter': 1000, 'ftol': 1e-9}
    )

    # Extract results
    radii_opt = result.x[:n]
    centers_opt = result.x[n:].reshape(n, 2)

    # Validate the final packing
    success, msg = validate_packing(centers_opt, radii_opt)
    if not success:
        print(f"Validation failed: {msg}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
