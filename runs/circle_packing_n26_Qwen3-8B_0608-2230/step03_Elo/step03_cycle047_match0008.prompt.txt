import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    # Initial guess: square grid with 5x5 = 25 circles and one extra in the center
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Create 5x5 grid for the first 25 circles
    for i in range(5):
        for j in range(5):
            idx = i * 5 + j
            x = j / 4.0 + 1e-3  # Slight offset to ensure circles fit
            y = i / 4.0 + 1e-3
            centers[idx] = [x, y]

    # One extra circle in the center
    centers[25] = [0.5, 0.5]

    # Initial radii: small value for all circles
    radii[:] = 0.05

    # Objective function: maximize the sum of radii
    def objective(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        return -np.sum(radii_flat)  # Minimize negative of sum for maximization

    # Constraints: boundary and non-overlap
    def constraints(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]

        # Boundary constraints
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
                return np.inf
            if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf

        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf

        return 0

    # Optimization bounds: centers in [0, 1], radii >= 0
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
    for i in range(n):
        bounds.append((0.0, 1.0))  # radius

    # Optimization options
    options = {
        'maxiter': 1000,
        'ftol': 1e-8,
        'gtol': 1e-8,
        'disp': False
    }

    # Solve the optimization problem
    result = optimize.minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options=options
    )

    # Extract the optimal solution
    optimal_centers = result.x[:n*2].reshape(n, 2)
    optimal_radii = result.x[n*2:]
    sum_radii = np.sum(optimal_radii)

    # Validate the solution
    valid, message = validate_packing(optimal_centers, optimal_radii)
    if not valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return optimal_centers, optimal_radii, sum_radii

import numpy as np
import scipy.optimize as opt
from typing import Tuple, List

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial guess for circle positions and radii
    # Based on a hybrid hexagonal and boundary-optimized packing
    centers = np.zeros((26, 2))
    radii = np.ones(26) * 0.05  # Initial guess for radii
    
    # Place 10 circles in a hexagonal pattern near the bottom
    for i in range(10):
        row = i // 5
        col = i % 5
        x = col * (1.0 / 3.0) + 0.1
        y = row * (np.sqrt(3) / 3.0) + 0.1
        centers[i] = [x, y]
        radii[i] = 0.08

    # Place 10 circles in a hexagonal pattern near the top
    for i in range(10):
        row = i // 5
        col = i % 5
        x = col * (1.0 / 3.0) + 0.1
        y = 1.0 - row * (np.sqrt(3) / 3.0) - 0.1
        centers[10 + i] = [x, y]
        radii[10 + i] = 0.08

    # Place 6 circles near the corners
    for i in range(6):
        if i == 0:
            centers[20 + i] = [0.1, 0.1]
        elif i == 1:
            centers[20 + i] = [0.9, 0.1]
        elif i == 2:
            centers[20 + i] = [0.1, 0.9]
        elif i == 3:
            centers[20 + i] = [0.9, 0.9]
        elif i == 4:
            centers[20 + i] = [0.1, 0.5]
        elif i == 5:
            centers[20 + i] = [0.5, 0.1]
        radii[20 + i] = 0.07

    # Define the objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)  # minimize the negative sum to maximize the sum

    # Define constraints for circle-circle and circle-boundary non-overlap
    def constraint_circle_circle(x, i, j):
        # Extract centers and radii
        radii = x[0:26]
        centers = x[26:].reshape(26, 2)

        # Extract centers for circle i and j
        center_i = centers[i]
        center_j = centers[j]

        # Extract radii
        r_i = radii[i]
        r_j = radii[j]

        # Compute distance between centers
        dist = np.linalg.norm(center_i - center_j)

        # Constraint: dist >= r_i + r_j - 1e-12
        return dist - r_i - r_j + 1e-12

    def constraint_circle_boundary(x, i):
        # Extract centers and radii
        radii = x[0:26]
        centers = x[26:].reshape(26, 2)

        # Extract center and radius
        center = centers[i]
        r = radii[i]

        # Constraint: x - r >= 0 (left boundary)
        left = center[0] - r
        # Constraint: 1 - x - r >= 0 (right boundary)
        right = 1.0 - center[0] - r
        # Constraint: y - r >= 0 (bottom boundary)
        bottom = center[1] - r
        # Constraint: 1 - y - r >= 0 (top boundary)
        top = 1.0 - center[1] - r

        # We will use the minimum of all four constraints
        return np.min([left, right, bottom, top])

    # Add constraints for circle-circle overlaps
    cons = []
    for i in range(26):
        for j in range(i + 1, 26):
            cons.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_circle_circle(x, i, j)})

    # Add constraints for circle-boundary overlaps
    for i in range(26):
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_circle_boundary(x, i)})

    # Initial guess: combine radii and centers
    x0 = np.concatenate((radii, centers.flatten()))

    # Solve the constrained optimization problem
    result = opt.minimize(
        fun=objective,
        x0=x0,
        method='SLSQP',
        constraints=cons,
        bounds=[(0, 1) for _ in range(26)] + [(0, 1) for _ in range(52)]
    )

    # Extract optimized radii and centers
    optimized_radii = result.x[0:26]
    optimized_centers = result.x[26:].reshape(26, 2)

    # Validate the packing
    is_valid, message = validate_packing(optimized_centers, optimized_radii)
    if not is_valid:
        raise ValueError(f"Invalid packing: {message}")

    # Return the result
    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
