import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
from scipy.optimize import Bounds, LinearConstraint
import itertools

def run_packing():
    # Initial guess for centers and radii using a hexagonal packing
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    # Start with a hexagonal packing
    # First row
    centers[0] = [0.5, 1.0]
    radii[0] = 0.2
    # Second row
    centers[1] = [0.25, 0.75]
    radii[1] = 0.2
    centers[2] = [0.75, 0.75]
    radii[2] = 0.2
    # Continue building the hexagonal grid
    for i in range(3, n):
        if i % 2 == 0:
            row = i // 2
            col = i % 2
            x = (col + 0.5) / 2.0
            y = 1.0 - row * (np.sqrt(3)/2.0)
            centers[i] = [x, y]
            radii[i] = 0.2
        else:
            row = (i - 1) // 2
            col = 1
            x = (col + 0.5) / 2.0
            y = 1.0 - row * (np.sqrt(3)/2.0) - (np.sqrt(3)/4.0)
            centers[i] = [x, y]
            radii[i] = 0.2

    # Optimization setup
    def objective(x):
        # x is a flat array [c1x, c1y, r1, c2x, c2y, r2, ..., cnx, cny, rn]
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Negative because we minimize

    def constraint_overlap(x):
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        dist = pdist(centers_flat)
        min_dist = np.min(dist)
        min_radius_sum = np.min(radii_flat + radii_flat[np.newaxis, :])
        return min_dist - min_radius_sum + 1e-12

    def constraint_boundary(x):
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
                return -1
            if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    bounds = Bounds(
        lb=np.zeros(2*n + n),
        ub=np.array([1.0 + 1e-12] * 2*n + [1.0 + 1e-12] * n)
    )

    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary}
    ]

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'maxiter': 1000, 'ftol': 1e-12}
    )

    # Extract results
    centers_opt = result.x[:2*n].reshape(n, 2)
    radii_opt = result.x[2*n:]
    sum_radii = -result.fun

    # Final validation
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        print(f"Validation failed: {msg}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return centers_opt, radii_opt, sum_radii
