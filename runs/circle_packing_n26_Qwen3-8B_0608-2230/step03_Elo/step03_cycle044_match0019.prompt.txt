import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initial guess: hexagonal packing with some adjustments near boundaries
    # Let's try a hexagonal packing pattern with some circles near the edges
    spacing = 0.2  # Initial spacing to start with
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * spacing + 1e-3
        y = row * spacing * np.sqrt(3) / 2 + 1e-3
        if row % 2 == 1:
            x += spacing / 2
        centers[i] = [x, y]

    # Ensure circles are within the unit square and slightly adjust positions
    centers = np.clip(centers, 0 + 1e-3, 1 - 1e-3)

    # Initialize radii to a small value
    radii[:] = 0.05

    # Objective function: maximize the sum of radii
    def objective(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        return -np.sum(radii_flat)  # Minimize negative of sum for maximization

    # Constraints: boundary and non-overlap
    def constraints(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]

        # Boundary constraints
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
                return np.inf
            if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf

        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf

        return 0

    # Optimization bounds: centers in [0, 1], radii >= 0
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
    for i in range(n):
        bounds.append((0.0, 1.0))  # radius

    # Optimization options
    options = {
        'maxiter': 2000,
        'ftol': 1e-9,
        'gtol': 1e-9,
        'disp': False
    }

    # Solve the optimization problem
    result = optimize.minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options=options
    )

    # Extract the optimal solution
    optimal_centers = result.x[:n*2].reshape(n, 2)
    optimal_radii = result.x[n*2:]
    sum_radii = np.sum(optimal_radii)

    # Validate the solution
    valid, message = validate_packing(optimal_centers, optimal_radii)
    if not valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return optimal_centers, optimal_radii, sum_radii
