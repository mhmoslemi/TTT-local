import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Hexagonal packing for the first 25 circles with slight offsets
    # This is a hexagonal grid with 5 rows of 5, 5, 5, 5, and 4 circles
    for i in range(5):
        for j in range(5 - (i % 2)):
            idx = i * 5 + j
            x = j / 4.0 + 1e-3
            y = i / 4.0 + 1e-3
            if i % 2 == 0:
                x += (j + 1) / 16.0  # slight horizontal offset for even rows
            centers[idx] = [x, y]

    # One extra circle near the center
    centers[25] = [0.5, 0.5]

    # Initial radii: slightly larger for grid circles
    radii[:25] = 0.1
    radii[25] = 0.1

    # Objective function: maximize the sum of radii
    def objective(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        return -np.sum(radii_flat)  # Minimize negative of sum for maximization

    # Constraints: boundary and non-overlap
    def constraints(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]

        # Boundary constraints
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or
                y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return np.inf

        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf

        return 0

    # Optimization bounds: centers in [0, 1], radii >= 0
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
    for i in range(n):
        bounds.append((0.0, 1.0))  # radius

    # Optimization options
    options = {
        'maxiter': 1000,
        'ftol': 1e-8,
        'gtol': 1e-8,
        'disp': False
    }

    # Solve the optimization problem
    result = optimize.minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options=options
    )

    # Extract the optimal solution
    optimal_centers = result.x[:n*2].reshape(n, 2)
    optimal_radii = result.x[n*2:]
    sum_radii = np.sum(optimal_radii)

    # Validate the solution
    valid, message = validate_packing(optimal_centers, optimal_radii)
    if not valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return optimal_centers, optimal_radii, sum_radii
