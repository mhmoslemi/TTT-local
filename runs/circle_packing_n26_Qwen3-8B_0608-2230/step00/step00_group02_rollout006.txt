import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    # Initial guess: square grid with 5x5 = 25 circles and one extra in the center
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    for i in range(5):
        for j in range(5):
            idx = i * 5 + j
            x = j / 4.0 + 1e-3  # Slight offset to ensure circles fit
            y = i / 4.0 + 1e-3
            centers[idx] = [x, y]

    # One extra circle in the center
    centers[25] = [0.5, 0.5]

    # Initial radii: small value for all circles
    radii[:] = 0.05

    # Objective function: maximize the sum of radii
    def objective(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        return -np.sum(radii_flat)  # Minimize negative of sum for maximization

    # Constraints: boundary and non-overlap
    def constraints(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]

        # Boundary constraints
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:

        # Non-overlap constraints

                dist = np.sqrt(dx*dx + dy*dy)

        return 0

    # Optimization bounds: centers in [0, 1], radii >= 0
    bounds = []

    options = {

        'ftol': 1e-8,

    result = optimize.minimize(
