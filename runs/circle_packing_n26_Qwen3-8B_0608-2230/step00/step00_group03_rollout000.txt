from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing():

    num_circles = 26
    rows = 6
    cols = 5
    hex_radius = 1 / (2 * np.sqrt(3))  # Hexagonal packing radius within square
    grid_centers = []

    for i in range(rows):
        for j in range(cols):
            x = (j + 0.5 * (i % 2)) * (1 - 2 * hex_radius) / (cols - 1)
            y = i * (1 - 2 * hex_radius) / (rows - 1)
            grid_centers.append([x, y])

    centers = np.array(grid_centers)
    radii = np.full(num_circles, hex_radius)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

        return np.min(centers - radii, axis=1) >= -1e-12
    constraint_center_bounds = NonlinearConstraint(constraint_center_bounds, 0, np.inf)

    def constraint_boundary(radii):
        return np.sum((centers - radii) ** 2) >= 1e-12
    constraint_boundary = NonlinearConstraint(constraint_boundary, 0, np.inf)

        dists = np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2)
        min_dist = np.min(dists - (radii[:, np.newaxis] + radii[np.newaxis, :]) ** 2)
        return min_dist
    constraint_overlap = NonlinearConstraint(constraint_overlap, 0, np.inf)

    # Bounds for radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
