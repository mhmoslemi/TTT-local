from scipy.optimize import minimize

def run_packing():

    n_circles = 26

        centers = []

            row = i // 4
            col = i % 4

            y = 0.1 + row * 0.25
            centers.append([x, y])
        return np.array(centers)

    initial_radii = np.full(n_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Optimization constraints: non-overlap and boundary constraints
    def constraints(radii, centers):
        constraints = []
        # Boundary constraints
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r: 1.0 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r: 1.0 - y - r})
        # Non-overlap constraints
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                def dist_constraint(radii):
                    ri = radii[i]
                    rj = radii[j]
                    xi, yi = centers[i]
                    xj, yj = centers[j]
                    dist = np.sqrt((xi - xj)**2 + (yi - yj)**2)
                    return dist - ri - rj
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        return constraints

    centers = initial_centers(n_circles)

    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',

    )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
