from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing():
    n = 26
    # Initialize centers with a hexagonal grid layout
    # Start with a hexagonal grid and perturb positions to encourage even distribution
    grid_size = int(np.sqrt(n))
    spacing = 1.0 / (grid_size + 1)
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

            idx = i * grid_size + j

            radii[idx] = spacing / 2

        return -np.sum(radii_flat)  # Negative for minimization

    # Constraints: circles inside the unit square and non-overlapping

        # x is a concatenated array of centers and radii
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]

        # Check if circles are within the unit square
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or

                return np.inf

        return 0

    bounds = Bounds(np.zeros(2*n + n), np.ones(2*n + n))

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for j in range(i + 1, n):

            if dist < radii[i] + radii[j] - 1e-12:

    return True, "ok"
