import numpy as np
import scipy.optimize as opt
from typing import Tuple

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_hexagonal_layout(num_circles):
    # Create an initial hexagonal arrangement with some rows and columns
    # We'll use 5 rows to fit 26 circles
    # 5 rows: rows 0-4
    # Columns: 5 or 6 per row (alternate)
    radii = np.full(num_circles, 0.01)
    centers = np.zeros((num_circles, 2))

    # Calculate spacing based on circle radius
    spacing = 2 * radii[0]

    for i in range(num_circles):
        row = i // 5
        col = i % 5
        if row % 2 == 1:
            col += 1
        if col >= 5:
            col = 0
        x = col * spacing
        y = row * spacing
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = 0.01

    return centers, radii

def objective_function(params, centers, radii, num_circles):
    num_params = params.size
    if num_params % 2 != 0:
        raise ValueError("Number of parameters must be even")
    x = params.reshape((num_params // 2, 2))
    new_centers = np.vstack((x, centers[num_params // 2:]))
    new_radii = np.concatenate((params[num_params // 2:], radii[num_params // 2:]))

    # Calculate the sum of radii
    sum_radii = np.sum(new_radii)

    # Ensure all circles are within the unit square
    for i in range(num_circles):
        x, y = new_centers[i]
        r = new_radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return -np.inf  # Invalid configuration

    # Ensure no overlapping circles
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.sqrt(np.sum((new_centers[i] - new_centers[j]) ** 2))
            if dist < new_radii[i] + new_radii[j] - 1e-12:
                return -np.inf  # Invalid configuration

    return -sum_radii  # Negative because we are minimizing

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    num_circles = 26
    centers, radii = initial_hexagonal_layout(num_circles)

    # Flatten the centers and radii for optimization
    initial_params = np.concatenate((centers.reshape(-1), radii))

    # Define optimization bounds
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
        bounds.append((0, 1))  # radius

    # Optimization options
    options = {
        'method': 'L-BFGS-B',
        'disp': False,
        'ftol': 1e-6,
        'maxiter': 1000
    }

    # Run gradient-based optimization
    result = opt.minimize(
        fun=lambda x: objective_function(x, centers, radii, num_circles),
        x0=initial_params,
        bounds=bounds,
        method='L-BFGS-B',
        options=options
    )

    optimized_params = result.x
    centers_opt = optimized_params[:num_circles * 2].reshape((num_circles, 2))
    radii_opt = optimized_params[num_circles * 2:]

    # Validate final packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        raise ValueError(f"Invalid packing: {message}")

    total_sum = np.sum(radii_opt)

    return centers_opt, radii_opt, total_sum
