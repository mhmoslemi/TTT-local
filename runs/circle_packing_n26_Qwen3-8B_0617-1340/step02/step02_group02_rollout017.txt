import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Ensure all circles are within [0,1]^2
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
                return np.inf  # Constraint violation
        # Ensure no overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf  # Constraint violation
        return 0

    # Initial guess for circle positions using a hexagonal grid
    # Hexagonal grid with 5 rows
    num_circles = 26
    row_height = np.sqrt(3) / 2  # vertical distance between rows in hexagonal packing
    row_width = 1  # horizontal distance between centers in same row
    num_rows = int(np.ceil(np.sqrt(num_circles)))  # approximate number of rows
    num_cols = int(np.ceil(num_circles / num_rows))  # approximate number of columns

    centers = np.zeros((num_circles, 2))
    idx = 0
    for row in range(num_rows):
        for col in range(num_cols):
            if idx >= num_circles:
                break
            x = col * row_width
            y = row * row_height
            # Ensure y is within [0, 1]

                y = 1 - (y - 1)
            # Ensure x is within [0, 1]

                x = 1 - (x - 1)

            idx += 1

    bounds = [(0, 1) for _ in range(26)]

    result = opt.minimize(
        fun=lambda x: objective_function(x, centers),
        x0=initial_radii,
        bounds=bounds,
        constraints=opt.SLSQP(fun=lambda x: constraints(x, centers), args=(centers,)),

    sum_radii = np.sum(radii)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
