import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    def generate_hexagonal_grid(num_circles: int, square_size: float = 1.0) -> np.ndarray:
        """
        Generate a hexagonal grid of centers for a given number of circles inside a unit square.
        """
        # Hexagonal grid parameters
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        row_spacing = np.sqrt(3) / 2 * square_size
        col_spacing = square_size

        centers = np.zeros((num_circles, 2))
        idx = 0
        for r in range(rows):
            for c in range(cols):
                if r % 2 == 0:
                    x = c * col_spacing
                else:
                    x = (c + 0.5) * col_spacing
                y = r * row_spacing
                centers[idx] = [x, y]
                idx += 1
        return centers

    def objective_function(x: np.ndarray, centers: np.ndarray, num_circles: int) -> float:
        """
        Objective function to maximize the total sum of radii.
        """
        # x is a vector of radii
        radii = x
        return -np.sum(radii)  # Minimize negative sum to maximize the sum

    def constraints(x: np.ndarray, centers: np.ndarray, num_circles: int) -> np.ndarray:
        """
        Constraints to ensure circles remain within the square and do not overlap.
        """
        radii = x
        constraints = []

        # Boundary constraints: each circle must stay within [0,1] × [0,1]
        for i in range(num_circles):
            r = radii[i]
            x_center, y_center = centers[i]
            constraints.append((x_center - r >= -1e-12))
            constraints.append((x_center + r <= 1 + 1e-12))
            constraints.append((y_center - r >= -1e-12))
            constraints.append((y_center + r <= 1 + 1e-12))

        # Non-overlapping constraints: distance between centers >= sum of radii
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                constraints.append((dist - (radii[i] + radii[j]) >= -1e-12))

        return np.array(constraints)

    num_circles = 26
    centers = generate_hexagonal_grid(num_circles)

    # Initial guess for radii (based on hexagonal grid cell size)
    initial_radii = np.ones(num_circles) * 0.1

    # Define optimization bounds: radii must be non-negative
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Setup optimization problem
    x0 = initial_radii
    res = opt.minimize(
        objective_function,
        x0,
        args=(centers, num_circles),
        method="SLSQP",
        bounds=bounds,
        constraints=opt.constraints.Constraints(lambda x: constraints(x, centers, num_circles)),
        tol=1e-8
    )

    if not res.success:
        raise ValueError("Optimization failed.")

    radii = res.x

    if not valid:
        raise ValueError(f"Invalid packing due to {message}")

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
