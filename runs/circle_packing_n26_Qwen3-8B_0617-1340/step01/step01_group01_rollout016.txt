import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial hexagonal packing in the unit square
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Hexagonal grid with spacing based on a radius of 0.08
    r_initial = 0.08
    rows = int(math.sqrt(n))
    cols = int(math.ceil(n / rows))
    
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * (2 * r_initial * math.sqrt(3)) + r_initial
        y = row * (2 * r_initial) + r_initial
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = r_initial
    
    # Optimization problem: maximize sum of radii subject to constraints
    # Define objective function (negative because we'll minimize)
    def objective(x):
        # x contains radii to optimize
        return -np.sum(x)
    
    # Define constraints
    def constraints(x):
        # Constraint 1: Each circle must be within the unit square
        # Define circles with updated radii
        # First, extract the initial centers and update radii
        x_centers = centers.copy()
        x_radii = x
        # Check each circle
        for i in range(n):
            x_i, y_i = x_centers[i]
            r_i = x_radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.array([1.0])  # Violation
        # Constraint 2: No overlapping circles
        # Check each pair
        for i in range(n):
            for j in range(i + 1, n):
                dx = x_centers[i, 0] - x_centers[j, 0]
                dy = x_centers[i, 1] - x_centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < x_radii[i] + x_radii[j] - 1e-8:
                    return np.array([1.0])  # Violation
        return np.array([0.0])  # No violation
    
    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(n)]
    
    # Minimization with constraints
    result = optimize.minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
