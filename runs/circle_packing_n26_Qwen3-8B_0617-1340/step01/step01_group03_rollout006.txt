import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds, minimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing
    num_circles = 26
    centers = []
    radii = []

    # Define initial hexagonal grid within unit square
    rows = int(np.sqrt(num_circles))
    cols = (num_circles + rows - 1) // rows

    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * i
            y = i
            if x < 1 and y < 1:
                centers.append([x, y])
                radii.append(0.1)  # initial guess for radius

    centers = np.array(centers)
    radii = np.array(radii)

    # Objective function to maximize the total radius sum
    def objective(radii_vec):
        return -np.sum(radii_vec)  # minimize negative sum, which is maximizing the sum

    # Constraints:
    # 1. Circles must be within the unit square
    # 2. Circles must not overlap

    def constraints(radii_vec, centers):
        n = len(centers)
        constraints = []
        for i in range(n):
            # Constraint: circle i must be within the unit square
            def in_square(radii_vec, i):
                r = radii_vec[i]
                x, y = centers[i]
                return [x - r + 1e-12, 1 - x - r + 1e-12, y - r + 1e-12, 1 - y - r + 1e-12]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: in_square(r, i)})

        # Constraint: all pairs of circles must not overlap
        for i in range(n):
            for j in range(i + 1, n):
                def no_overlap(radii_vec, i, j):
                    r_i = radii_vec[i]
                    r_j = radii_vec[j]
                    dx = centers[i][0] - centers[j][0]
                    dy = centers[i][1] - centers[j][1]
                    dist_sq = dx**2 + dy**2
                    return np.sqrt(dist_sq) - r_i - r_j + 1e-12
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: no_overlap(r, i, j)})
        return constraints

    # Set up the optimization problem
    bounds = [(0, 0.5) for _ in range(num_circles)]  # radius can't exceed 0.5
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints(radii, centers)
    )

    optimized_radii = result.x

    if not is_valid:

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
