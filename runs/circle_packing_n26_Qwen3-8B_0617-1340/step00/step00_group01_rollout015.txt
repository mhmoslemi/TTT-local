def optimize_packing(centers: np.ndarray, radii: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
    # Define the variables
    n = centers.shape[0]
    x = cp.Variable((n, 2))
    r = cp.Variable(n)

    # Objective: maximize the sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # Boundary constraints
    for i in range(n):
        constraints.append(x[i, 0] >= r[i])
        constraints.append(x[i, 0] <= 1 - r[i])
        constraints.append(x[i, 1] >= r[i])
        constraints.append(x[i, 1] <= 1 - r[i])

    # Non-overlapping constraints
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.sqrt((x[i, 0] - x[j, 0])**2 + (x[i, 1] - x[j, 1])**2)
            constraints.append(dist >= r[i] + r[j])

    # Initial guess
    x.initial_value = centers
    r.initial_value = radii

    # Solve the optimization problem
    problem = cp.Problem(objective, constraints)
    problem.solve(solver=cp.SCS, eps=1e-6)

    if problem.status != 'optimal':
        raise ValueError(f"Optimization failed with status: {problem.status}")

    # Extract the solution
    centers_opt = x.value
    radii_opt = r.value

    return centers_opt, radii_opt

def validate_packing(centers: np.ndarray, radii: np.ndarray) -> Tuple[bool, str]:
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        if (x - r < -1e-12 or x + r > 1 + 1e-12

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
