import numpy as np
import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_positioning(n_circles):
    grid_size = int(np.sqrt(n_circles))
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        x = i % grid_size / grid_size
        y = i // grid_size / grid_size
        centers[i] = [x, y]
    return centers

def get_distances(centers):
    n = centers.shape[0]
    distances = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            distances[i, j] = np.sqrt(np.sum((centers[i] - centers[j])**2))
    return distances

def optimize_packing(n_circles=26, max_iter=100, tol=1e-6):
    # Initial placement
    centers = initial_positioning(n_circles)
    radii = np.ones(n_circles) * 0.05  # Small initial radii

    # Convert to vector for optimization
    x = np.concatenate([centers.flatten(), radii])

    # Objective function: maximize sum of radii
    def objective(x_flat):
        centers_flat = x_flat[:n_circles*2].reshape((n_circles, 2))
        radii_flat = x_flat[n_circles*2:]
        return -np.sum(radii_flat)  # Negative for minimization

    # Constraints
    def constraints(x_flat):
        centers_flat = x_flat[:n_circles*2].reshape((n_circles, 2))
        radii_flat = x_flat[n_circles*2:]
        # Check boundaries
        for i in range(n_circles):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return np.inf
        # Check overlap
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dist = np.sqrt(np.sum((centers_flat[i] - centers_flat[j])**2))
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf
        return 0

    # Optimization parameters
    bounds = []
    for _ in range(n_circles):
        bounds.extend([(0, 1), (0, 1)])  # x and y for each center
        bounds.extend([(0, 1)])  # radius for each circle

    # Optimization
    result = opt.minimize(
        objective,
        x,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},

        maxiter=max_iter

    # Extract results

    else:

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
