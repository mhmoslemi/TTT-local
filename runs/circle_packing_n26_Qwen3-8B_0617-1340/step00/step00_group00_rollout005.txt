import scipy.optimize as optimize
from typing import Tuple, Any

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: hexagonal packing with 6 rows
    num_circles = 26
    num_rows = 6
    num_cols = 5  # 5 columns per row, with 1 extra circle in some rows

    # Initial center positions
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

            if i % 2 == 0:
                x = j * (1 + np.sqrt(3)/2) / num_cols
            else:
                x = (j + 0.5) * (1 + np.sqrt(3)/2) / num_cols
            y = i * (1 / num_rows)
            centers[i * num_cols + j] = np.array([x, y])

    # Initial radius
    initial_radius = 0.05
    radii[:] = initial_radius

    # Objective function to maximize the sum of radii
    def objective(x: np.ndarray) -> float:
        centers[:], radii[:] = x[:2*num_circles].reshape(num_circles, 2), x[2*num_circles:]
        # Enforce non-negativity of radii
        radii[radii < 0] = 0
        # Calculate sum of radii
        return -np.sum(radii)  # Negative for minimization

    # Constraints
    constraints = []

    def constraint_centers_inside(x: np.ndarray) -> np.ndarray:
        centers[:], radii[:] = x[:2*num_circles].reshape(num_circles, 2), x[2*num_circles:]
        # Check if all circles are inside the unit square

            centers[:, 0] - radii + 1e-12,
            1 - centers[:, 0] - radii + 1e-12,
            centers[:, 1] - radii + 1e-12,
            1 - centers[:, 1] - radii + 1e-12
        ])

    constraints.append({'type': 'ineq', 'fun': constraint_centers_inside})

    def constraint_no_overlap(x: np.ndarray) -> np.ndarray:

        distances = np.zeros((num_circles, num_circles))
        for i in range(num_circles):

                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                distances[i, j] = distance - (radii[i] + radii[j]) + 1e-12

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
