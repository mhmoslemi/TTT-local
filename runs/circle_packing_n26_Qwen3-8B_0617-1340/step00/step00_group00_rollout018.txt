def run_packing():

    """Objective function to minimize: negative of total radii (to maximize radii)."""
    centers = x[:n_circles*2].reshape((n_circles, 2))
    radii = x[n_circles*2:]
    return -np.sum(radii)

def validate_packing(centers, radii):
    """
    Validate the packing solution against constraints.
    Returns:
        (is_valid, message): tuple indicating validity and a message
    """
