import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Step 1: Hexagonal grid initialization
    num_circles = 26
    num_rows = 5  # Sufficient rows for 26 circles
    num_cols = 6  # Sufficient columns for 26 circles
    # Hexagonal grid setup
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.05  # Initial guess for radii
    index = 0
    for row in range(num_rows):
        for col in range(num_cols):
            if row % 2 == 0:
                x = col * (1.0 / (num_cols - 1)) + 0.5 / (num_cols - 1)
            else:
                x = col * (1.0 / (num_cols - 1)) + 0.5 / (num_cols - 1)
            y = row * (1.0 / (num_rows - 1)) + 0.5 / (num_rows - 1)
            centers[index, 0] = x
            centers[index, 1] = y
            index += 1

    # Function to calculate total sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Define constraints
    def constraint_overlap(radii):
        # Constraint: non-overlapping circles
        # Constraint format: (c, x) where c is the constraint function and x is the argument
        # This function will return the constraint value
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                # Constraint: dist >= radii[i] + radii[j]
                constraint_value = dist - radii[i] - radii[j]
                if constraint_value < -1e-12:
                    return constraint_value
        return 0.0  # No overlap

    def constraint_boundary(radii):
        # Constraint: circles must be within the unit square
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return x - r or x + r - 1 or y - r or y + r - 1
        return 0.0  # All within the square

    # Set up optimization problem
    bounds = [(0, 1.0) for _ in range(num_circles)]  # Radii between 0 and 1
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary}

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
