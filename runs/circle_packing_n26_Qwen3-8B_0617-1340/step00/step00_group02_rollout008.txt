import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    n = 26
    # Generate an initial hexagonal grid of centers
    centers = []
    for i in range(n):
        row = i // 5 + 1
        col = i % 5
        x = col / 5.0 + 0.1  # Offset to prevent edge touching
        y = row / 5.0 + 0.1
        centers.append([x, y])
    centers = np.array(centers)
    
    # Initial radii (same for all)
    radii = np.full(n, 0.05)

    # Define variables and bounds
    x_bounds = [(0.0, 1.0) for _ in range(n * 2)]
    r_bounds = [(0.0, 1.0) for _ in range(n)]

    # Flatten centers and radii for optimization
    x = np.concatenate((centers.flatten(), radii))
    
    def objective(x_flat):
        # Reshape to centers and radii
        centers_flat = x_flat[:n * 2].reshape(n, 2)
        radii_flat = x_flat[n * 2:].reshape(n)
        return -np.sum(radii_flat)  # Minimize negative sum (maximize sum)

    def constraint_boundary(x_flat):
        centers_flat = x_flat[:n * 2].reshape(n, 2)
        radii_flat = x_flat[n * 2:].reshape(n)
        for i in range(n):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return -1.0
        return 0.0

    def constraint_overlap(x_flat):
        centers_flat = x_flat[:n * 2].reshape(n, 2)
        radii_flat = x_flat[n * 2:].reshape(n)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1.0
        return 0.0

    # Set up optimization problem
    cons = [
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Run optimization
    result = optimize.minimize(
        fun=objective,
        x0=x,
        bounds=zip(x_bounds, r_bounds),
        constraints=cons,
        method='SLSQP',
        tol=1e-8,
        options={
            'maxiter': 1000,
            'ftol': 1e-8,
            'gtol': 1e-8
        }
    )

    if not result.success:
        raise ValueError("Optimization failed with message: " + result.message)

    # Extract results
    optimal_x = result.x
    centers_opt = optimal_x[:n*2].reshape(n, 2)
    radii_opt = optimal_x[n*2:].reshape(n)
    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
