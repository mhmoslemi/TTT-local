import scipy.optimize as optimize
import math

    n_circles = 26

    # Use a grid of m rows and n columns, adjusting for the hexagonal spacing

    m = int(math.ceil(math.sqrt(n_circles)))
    n = int(math.ceil(n_circles / m))

    #   (0, 0), (1, 0), (2, 0), ..., (n-1, 0)

    centers /= np.max(centers)

        # Check that for all i, x_i - r_i >= 0 and x_i + r_i <= 1
        # We will use a soft constraint with penalty

        # Check that for all i < j, ||centers[i] - centers[j]|| >= r_i + r_j

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
