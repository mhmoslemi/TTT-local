import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define initial centers for a hexagonal-like grid configuration
    num_circles = 26
    # Start with a hexagonal grid configuration
    def hexagonal_grid(n, spacing):
        centers = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * (spacing + 1e-6)
            y = row * (spacing * np.sqrt(3) + 1e-6)
            centers.append([x, y])
        return np.array(centers)
    
    # Initial spacing based on number of circles
    initial_spaced = 0.3
    initial_centers = hexagonal_grid(num_circles, initial_spaced)
    initial_radii = np.full(num_circles, initial_spaced / 2)

    # Objective function for optimization
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraint function: distance between centers must be >= sum of radii
    def constraint_dist(radii, centers):
        n = len(radii)
        dists = np.zeros(n)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[i] = max(dists[i], radii[i] + radii[j] - dist)
        return dists

    # Constraint: circles must fit within the unit square
    def constraint_square(radii, centers):
        dists = np.zeros(len(radii))
        for i in range(len(radii)):
            x, y = centers[i]
            r = radii[i]
            dists[i] = max(0, x - r - 1e-6, 1 + 1e-6 - x - r, y - r - 1e-6, 1 + 1e-6 - y - r)
        return dists

    # Define bounds for radii (non-negative)
    bounds = Bounds(np.zeros(num_circles), np.inf)

    # Optimization setup
    def optimize_centers_and_radii():
        # Use a local optimization approach
        result = opt.minimize(
            lambda x: -objective(x[:num_circles], x[num_circles:2*num_circles]),
            x0=np.concatenate((initial_radii, initial_centers.flatten())),
            bounds=bounds,
            constraints=[
                opt.Constraint(
                    lambda x: constraint_dist(x[:num_circles], x[num_circles:2*num_circles]),
                    bounds=np.zeros(num_circles),
                    name="non_overlap"
                ),
                opt.Constraint(
                    lambda x: constraint_square(x[:num_circles], x[num_circles:2*num_circles]),
                    bounds=np.zeros(num_circles),
                    name="square_bounds"
                )
            ],
            method="SLSQP"
        )
        # Extract optimized radii and centers
        radii = result.x[:num_circles]
        centers = result.x[num_circles:2*num_circles].reshape(num_circles, 2)
        return centers, radii

    # Run the optimization
    centers, radii = optimize_centers_and_radii()
    sum_radii = np.sum(radii)

    # Validate the packing
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Optimized packing is invalid: {msg}")

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
