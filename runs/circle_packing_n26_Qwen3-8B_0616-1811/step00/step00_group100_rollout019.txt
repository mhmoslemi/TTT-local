        centers = x_vec[:, :2]
        radii = x_vec[:, 2]

        return -np.sum(radii)  # Negative for minimization

    def constraints(x):
        x_vec = x.reshape((n, 3))
        centers = x_vec[:, :2]
        radii = x_vec[:, 2]
        
        constraints = []
        # Boundary constraints: ensure circles are within [0, 1]x[0, 1]
        for i in range(n):
            x, y, r = x_vec[i]
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i, x=x, y=y, r=r: x - r})
