import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial hexagonal packing in the lower half of the square
    # Number of circles per row
    n_rows = 5
    n_cols = 5
    # Hexagonal offset
    hex_offset = np.sqrt(3) / 2
    # Initial positions
    centers = np.zeros((n_rows * n_cols, 2))
    for i in range(n_rows):
        for j in range(n_cols):
            x = j + 0.5
            y = i * hex_offset + 0.5
            if i % 2 == 1:
                x += 0.5
            centers[i * n_cols + j] = [x, y]
    
    # Adjust the initial positions to fit within the square and near the bottom
    centers[:, 1] = centers[:, 1] * 0.8  # Bring down the circles to the bottom

    # Initial radii

    # Function to compute objective (maximize sum of radii) with constraints
    def objective(radii):
        return -np.sum(radii)  # Negative since we minimize

    # Function to compute constraints
    def constraints(radii):
        # Non-overlapping constraint
        n = len(radii)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (radii[i] + radii[j]) + 1e-8)
        # Boundary constraints

            x, y = centers[i]
            r = radii[i]
            constraints.append(x - r + 1e-8)
            constraints.append(x + r - 1.0 - 1e-8)
            constraints.append(y - r + 1e-8)
            constraints.append(y + r - 1.0 - 1e-8)
        return constraints

    # Optimization
    bounds = [(0, 0.5) for _ in range(26)]  # Bounds for radii
    result = optimize.minimize(

        radii,
        method='SLSQP',
        bounds=bounds,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
