import numpy as np
import scipy.optimize as optimize
import random

def run_packing():
    # Initial configuration: Hexagonal packing
    n_circles = 26
    max_radius_initial = 0.1  # Initial guess for radius, can be tuned

    # Step 1: Generate initial centers in a hexagonal grid
    centers = np.zeros((n_circles, 2))
    radius = max_radius_initial * np.ones(n_circles)

    # Try to create a hexagonal grid
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))

    for i in range(n_circles):
        row = i // cols
        col = i % cols
        x = col + 0.5 * row
        y = row
        # Scale to unit square
        x = x / (cols - 0.5)
        y = y / rows
        centers[i, :] = [x, y]

    # Step 2: Optimize positions and radii using simulated annealing
    def objective_function(params):
        # params: flattened array of positions and radii
        # Reconstruct positions and radii
        flat_centers = params[:n_circles*2].reshape(n_circles, 2)
        flat_radii = params[n_circles*2:].reshape(n_circles)

        # Check if all radii are non-negative
        if np.any(flat_radii < 0):
            return float('inf')

        # Check if inside the square
        for i in range(n_circles):
            x, y = flat_centers[i]
            r = flat_radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
                return float('inf')

        for i in range(n_circles):

                dx = flat_centers[i, 0] - flat_centers[j, 0]
                dy = flat_centers[i, 1] - flat_centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < flat_radii[i] + flat_radii[j] - 1e-12:
                    return float('inf')

        # Objective: maximize sum of radii
        return -np.sum(flat_radii)

    # Initial guess for optimization
    initial_guess = np.concatenate([centers.flatten(), radius])

    # Simulated annealing optimization
    bounds = []
    for i in range(n_circles):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        bounds.append((0.0, 1.0))  # radius

    result = optimize.minimize(
        objective_function,
        initial_guess,
        method='L-BFGS-B',

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
