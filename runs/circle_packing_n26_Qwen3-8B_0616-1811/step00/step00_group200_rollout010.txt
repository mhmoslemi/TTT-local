import scipy.optimize as opt
from scipy.optimize import Bounds, LinearConstraint, NonlinearConstraint
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal grid packing inside the unit square
    num_circles = 26
    # Approximate hexagonal grid configuration
    max_rows = int(math.ceil(math.sqrt(num_circles)))
    radius_initial = 0.02  # Initial guess for radius
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, radius_initial)
    
    index = 0
    for row in range(max_rows):
        for col in range(max_rows):
            if index >= num_circles:
                break
            # Hexagonal grid offset
            offset_x = (col % 2) * (np.sqrt(3) / 2 * radius_initial)
            x = (col + offset_x) / 1.0  # Normalize to [0, 1]
            y = (row + (row <= max_rows - 1)) * (np.sqrt(3) / 2 * radius_initial) / 1.0
            centers[index, :] = [x, y]
            index += 1

    # Define the bounds for centers (must be within unit square)
    bounds_centers = Bounds([0, 0], [1, 1])
    
    # Define the total radius as the objective function to maximize
    def objective(radii):
        return -np.sum(radii)  # Negative because scipy minimizes
    
    # Constraints:
    # 1. All centers within [0,1]x[0,1]
    def center_bounds(x):
        return np.concatenate([x - np.array([0, 0]), np.array([1, 1]) - x]) >= 0
    constraint_center_bounds = NonlinearConstraint(center_bounds, 0, np.inf)
    
    # 2. No two circles may overlap (distance between centers >= sum of radii)
    def non_overlap_constraint(x):
        # x has shape (26 * 2 + 26,) => [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers = x[:26*2].reshape((26, 2))
        radii = x[26*2:]
        distances = np.sqrt(np.sum((centers[:, np.newaxis] - centers[np.newaxis, :])**2, axis=-1))
        # Avoid division by zero when i == j
        np.fill_diagonal(distances, np.inf)
        # Constraint: distance >= sum of radii
        return distances - (radii[:, np.newaxis] + radii[np.newaxis, :]) + 1e-8
    constraint_non_overlap = NonlinearConstraint(non_overlap_constraint, 0, np.inf)

        return x[26*2:] >= 0
    constraint_radii_non_negative = NonlinearConstraint(radii_non_negative, 0, np.inf)

    method = 'SLSQP'
    options = {
        'ftol': 1e-9,
        'maxiter': 1000,
        'disp': False
    }

        x0=x0,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
