import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # We'll distribute them in a hexagonal grid across the unit square.
    # Start with 6 rows, each containing 4-5 circles with some offset for spacing.

    # Define initial positions for a hexagonal grid
    rows = 6
    cols = 5
    hex_grid_centers = np.zeros((rows * cols, 2))
    hex_grid_radii = np.zeros(rows * cols)
    for i in range(rows):
        for j in range(cols):
            x = j * (1 / (cols + 1))
            y = i * (1 / (rows + 1))
            # Shift circles near the boundaries to fill the space more efficiently
            if i % 2 == 0:
                x += 0.1 * (j % 2)

                x += 0.1 * ((j + 1) % 2)
            if i == 0 or i == rows - 1:
                y += 0.1 * (i % 2)
            hex_grid_centers[i * cols + j] = [x, y]
            hex_grid_radii[i * cols + j] = 0.03  # Initial small radius

    # Objective function to maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative because we minimize

    # Constraints for optimization
    def constraints(radii, centers):
        constraints = []
        # Boundary constraint: circle must fit within [0,1]^2
        for i in range(len(radii)):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r >= -1e-12) + (x + r <= 1 + 1e-12) + (y - r >= -1e-12) + (y + r <= 1 + 1e-12))
        # Non-overlap constraint: distance between any two centers >= sum of radii

            for j in range(i + 1, len(radii)):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
