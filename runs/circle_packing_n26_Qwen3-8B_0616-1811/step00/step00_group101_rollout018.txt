import numpy as np
import scipy.optimize as opt
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: Hexagonal grid
    n = 26
    max_radii = 0.02  # Start with small radii
    # Grid setup for hexagonal packing
    rows = int(np.sqrt(n)) + 1
    cols = int(np.sqrt(n)) + 1
    centers = []
    radii = []
    for i in range(n):
        j = i % cols
        k = i // cols
        x = j / (cols - 1) if cols > 1 else 0.5
        y = k / (rows - 1) if rows > 1 else 0.5
        centers.append([x, y])
        radii.append(max_radii)
    centers = np.array(centers)
    radii = np.array(radii)

    # Define the optimization problem
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum (equivalent to maximizing sum)

    def constraint_center_bounds(radii_vec):
        return np.concatenate([radii_vec, np.zeros_like(radii_vec)])

    def constraint_boundary(radii_vec):
        # Constraint: centers + radii should be within [0,1]^2
        return np.concatenate([radii_vec, np.zeros_like(radii_vec)])

    def constraint_non_overlap(radii_vec):
        # Constraint: distance between centers >= r_i + r_j
        centers_opt = np.array(centers)
        dist = np.sqrt(np.sum((centers_opt[:, np.newaxis] - centers_opt) ** 2, axis=2))
        return dist - (radii_vec[:, np.newaxis] + radii_vec[np.newaxis, :])

    # Optimization settings
    bounds = [(0, 1) for _ in range(n)]  # All radii are non-negative
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_bounds},
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_non_overlap}
    ]

    # Perform optimization
    result = opt.minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    centers_opt = np.array(centers)
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)

    # Validate the packing
    valid, message = validate_packing(centers_opt, radii_opt)
    if not valid:
        print(f"Invalid packing: {message}")
        return None, None, 0.0

    # Plot the packing
    plt.figure(figsize=(6, 6))
    for i in range(n):
        x, y = centers_opt[i]
        r = radii_opt[i]
        plt.Circle((x, y), r, color='blue', fill=True, edgecolor='black')
    plt.xlim(0, 1)
    plt.ylim(0, 1)
    plt.gca().set_aspect('equal')
    plt.title(f"Circle Packing (Sum of Radii: {sum_radii:.4f})")
    plt.xlabel("x")
    plt.ylabel("y")
    plt.grid(True)
    plt.show()

    return centers_opt, radii_opt, sum_radii


def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
