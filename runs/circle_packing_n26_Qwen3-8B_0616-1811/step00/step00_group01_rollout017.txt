import scipy.optimize as optimize

def run_packing():

    num_cols = int(np.ceil(n / num_rows))

    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col + 0.5 * (row % 2)

        centers[i] = [x, y]

    max_x = np.max(centers[:, 0])
    max_y = np.max(centers[:, 1])
    centers /= np.max([max_x, max_y])

    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize
    
    # Constraints
    def constraints_func(radii):
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]) >= -1e-8)
        return constraints
    
    # Initial guess for radii
    initial_guess = initial_radii.copy()
