import numpy as np
import scipy.optimize as opt
from typing import Tuple

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based packing
    num_circles = 26
    grid_rows = 5
    grid_cols = 6
    grid_spacing = 1.0 / grid_rows  # Vertical spacing
    grid_spacing_h = 1.0 / grid_cols  # Horizontal spacing

    centers = np.zeros((num_circles, 2))
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j * grid_spacing_h
            y = i * grid_spacing
            centers[i * grid_cols + j] = [x, y]

    # Initial radii: all circles have the same initial radius
    initial_radius = 0.05
    radii = np.full(num_circles, initial_radius)

    # Objective function to maximize: sum of radii
    # Since we use minimization, we'll minimize -sum(radii)
    def objective(x: np.ndarray) -> float:
        # x is a flat array: [centers..., radii...]
        centers_flat = x[:2 * num_circles]
        centers = centers_flat.reshape((num_circles, 2))
        radii_flat = x[2 * num_circles:]
        radii = radii_flat.reshape(num_circles)

        # Check validity
        validity, msg = validate_packing(centers, radii)
        if not validity:
            return np.inf  # Penalize invalid configurations

        # Objective: minimize -sum(radii)
        return -np.sum(radii)

    # Bounds: centers in [0,1]^2, radii ≥ 0, and max radius such that circles fit
    bounds = []
    for i in range(num_circles):
        # Center bounds
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        # Radius bounds: minimum radius 0.0 to maximum such that circle fits in the square
        # Initial guess for radius is 0.05
        bounds.append((0.0, 1.0 - 2 * 0.05))  # Max radius is 1 - 2 * initial_radius

    # Initial guess
    x_initial = np.concatenate([
        centers.flatten(),
        radii.flatten()
    ])

    # Optimization
    result = opt.minimize(
        objective,
        x_initial,
        bounds=bounds,
        method="L-BFGS-B"
    )

    # Extract the optimal centers and radii
    optimal_centers = result.x[:2 * num_circles].reshape((num_circles, 2))
    optimal_radii = result.x[2 * num_circles:].reshape(num_circles)

    # Validate again

    if not validity:
        raise ValueError(f"Optimization failed with message: {msg}")

    sum_radii = np.sum(optimal_radii)
    return optimal_centers, optimal_radii, sum_radii
