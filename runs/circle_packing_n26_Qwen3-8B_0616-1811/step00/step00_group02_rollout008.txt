import numpy as np
from scipy.optimize import minimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based arrangement of circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Create a 5x5 grid with one circle in each cell
    for i in range(5):
        for j in range(5):
            idx = i * 5 + j
            x, y = i / 4.0, j / 4.0
            centers[idx] = [x, y]

    # Add one more circle in a corner to complete 26
    centers[25] = [0.0, 0.0]

    # Initial radii
    initial_radii = np.ones(num_circles) * 0.1

    # Objective function: maximize the sum of radii (equivalent to minimizing negative sum)
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraint: each circle must stay inside the unit square
    def square_bound(radii, centers):
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return False
        return True

    # Constraint: circles must not overlap
    def no_overlap(radii, centers):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    # Define optimization bounds
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Optimization setup
    result = minimize(
        fun=objective,
        x0=initial_radii,
        args=(centers,),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: square_bound(r, centers)},
            {'type': 'ineq', 'fun': lambda r: no_overlap(r, centers)}
        ]
    )

    # Extract results
    optimized_radii = result.x
    centers_optim = centers.copy()
    sum_radii = np.sum(optimized_radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
