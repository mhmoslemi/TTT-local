import cvxpy as cp
import matplotlib.pyplot as plt

def run_packing():
    # Initial hexagonal grid configuration (approximate)
    rows = 6  # Number of rows
    cols = 5  # Number of columns
    spacing = 0.3  # Initial spacing between centers
    angle = np.pi / 3  # 60 degrees for hexagonal packing
    centers = np.zeros((rows * cols, 2))
    radii = np.zeros(rows * cols)
    
    for i in range(rows):
        for j in range(cols):
            x = j * spacing + 0.5 * spacing
            y = i * spacing * np.sin(angle) + 0.5 * spacing * np.cos(angle)
            centers[i * cols + j] = [x, y]
    
    # Convert to variables
    n = len(centers)
    x = cp.Variable((n, 2))
    r = cp.Variable(n)

    # Objective: maximize the sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints: circles must be inside the unit square
    constraints = []
    for i in range(n):
        x_i, y_i = x[i]
        constraints.append(x_i - r[i] >= 0)
        constraints.append(x_i + r[i] <= 1)
        constraints.append(y_i - r[i] >= 0)
        constraints.append(y_i + r[i] <= 1)
    
    # Constraints: circles must not overlap
    for i in range(n):
        for j in range(i + 1, n):
            dx = x[i][0] - x[j][0]
            dy = x[i][1] - x[j][1]
            dist = cp.sqrt(dx**2 + dy**2)
            constraints.append(dist >= r[i] + r[j] - 1e-6)

    # Additional constraints to prevent negative radii
    constraints.append(r >= 0)

    # Solve the optimization problem
    prob = cp.Problem(objective, constraints)
    prob.solve(solver=cp.SCS, eps=1e-6)

    # Extract the optimal solution
    if prob.status not in ["OPTIMAL", "OPTIMAL_INACCURATE"]:
        raise RuntimeError("Optimization failed")

    centers_opt = x.value

    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
