import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial seed configuration (hexagonal packing with some adjustments)
    # We'll use a 5x5 grid with some rows offset for hexagonal packing
    centers = []
    radii = []
    row_count = 5
    col_count = 5
    offset = 0.5 * np.sqrt(3)  # Vertical spacing for hexagonal packing
    spacing = 0.4

    for row in range(row_count):
        for col in range(col_count):
            x = col * spacing
            y = row * spacing + (row % 2) * offset
            centers.append([x, y])
            radii.append(0.03)

    centers = np.array(centers)
    radii = np.array(radii)

    # Objective function: maximize sum of radii
    def objective(x):
        # x contains all radii in a flat array
        return -np.sum(x)

    # Constraint: circles do not overlap
    def constraint_overlap(x):
        # Convert x to radii array
        radii_flat = x
        for i in range(len(radii_flat)):
            for j in range(i + 1, len(radii_flat)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1.0  # Not feasible
        return 0.0

    # Constraint: circles are within the unit square
    def constraint_boundary(x):
        radii_flat = x
        for i in range(len(radii_flat)):
            x_center, y_center = centers[i]
            r = radii_flat[i]
            if x_center - r < -1e-8 or x_center + r > 1 + 1e-8:
                return -1.0
            if y_center - r < -1e-8 or y_center + r > 1 + 1e-8:
                return -1.0
        return 0.0

    # Optimization bounds: radii must be non-negative
    bounds = [(0.0, 1.0) for _ in range(len(radii))]

    # Create optimization problem
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary}
    ]

    # Run optimization

        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
