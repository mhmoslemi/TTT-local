from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26
    num_rows = int(np.sqrt(n))  # Approximate rows for hexagonal packing
    num_cols = (n + num_rows - 1) // num_rows  # Number of columns

    # Create initial hexagonal grid
    centers = np.zeros((n, 2))
    radius = 0.1  # Initial guess for radius

    # Generate the hexagonal grid
    for i in range(num_rows):

            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            if i % 2 == 1:
                x += 0.5
            centers[i * num_cols + j] = [x, y]
    centers /= max(centers.max(), 1.0)  # Normalize centers to fit in [0,1]^2

    # Initial guess for radii
    radii = np.full(n, radius)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints: circles are within the unit square

        for i in range(n):

                return np.inf
        return 0

    # Constraints: circles do not overlap
    def constraint_overlap(radii):

            for j in range(i + 1, n):

                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
