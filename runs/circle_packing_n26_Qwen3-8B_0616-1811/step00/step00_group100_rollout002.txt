import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Define the number of circles
    n = 26
    
    # Step 1: Create a hexagonal grid in the unit square
    # Determine the number of rows and columns in the hexagonal grid
    # Start with rows and alternate the column offsets
    num_rows = 6
    num_cols_per_row = 4
    cols_per_row = [4, 4, 4, 4, 4, 4]
    
    # Define initial grid points
    centers = []
    for row in range(num_rows):
        col_offsets = [0.5 * (num_cols_per_row - 1) * 1 / (num_cols_per_row - 1) * (i - (num_cols_per_row - 1) / 2) for i in range(num_cols_per_row)]
        for col in range(num_cols_per_row):
            col_offset = col_offsets[col]
            x = col_offset
            y = row * np.sqrt(3) / 2
            if row % 2 == 1:  # offset for even rows
                x += 0.5
            centers.append([x, y])
    
    # Convert to numpy arrays
    centers = np.array(centers)
    
    # Step 2: Compute initial radii
    # Radius is the distance between adjacent centers in the grid
    radius = np.min(np.sqrt(np.sum((centers[1:] - centers[:-1])**2, axis=1))) / 2
    radii = np.full(n, radius)
    
    # Step 3: Use scipy.optimize to refine the packing
    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum
    
    # Create bounds for radii (must be non-negative)
    bounds = [(0, 1) for _ in range(n)]
    
    # Constraints: circles must be within the square and not overlap
    def constraints(radii):
        centers = np.zeros((n, 2))
        for i in range(n):
            # Place each circle optimally within the square
            x = np.random.uniform(1e-6, 1 - 1e-6)
            y = np.random.uniform(1e-6, 1 - 1e-6)
            centers[i] = [x, y]
        
        # Check boundaries
        def boundary_constraints(radii):
            for i in range(n):
                x, y = centers[i]
                r = radii[i]
                if (x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12):
                    return False
            return True
        
        # Check circle-circle overlaps
        def overlap_constraints(radii):
            for i in range(n):
                for j in range(i + 1, n):
                    dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                    if dist < radii[i] + radii[j] - 1e-12:
                        return False
            return True
        
        return boundary_constraints(radii) and overlap_constraints(radii)
    
    # Use sequential quadratic programming (SLSQP) for constrained optimization
    result = optimize.minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')
    
    # Extract the optimized radii and centers
    optimized_radii = result.x
    optimized_centers = centers
    
    # Step 4: Validate the solution

    if not success:
        raise ValueError(f"Packing failed: {message}")
    
    sum_radii = np.sum(optimized_radii)
    return optimized_centers, optimized_radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
