import scipy.optimize as opt

def run_packing():

    num_circles = 26
    hex_grid = np.array([[i + j*0.5, j*sqrt(3)/2] for j in range(6) for i in range(5)])
    hex_grid = hex_grid[:num_circles]
    
    # Scale to fit within [0,1]^2
    def scale_to_unit_square(grid):
        x_min, x_max = np.min(grid[:, 0]), np.max(grid[:, 0])
        y_min, y_max = np.min(grid[:, 1]), np.max(grid[:, 1])
        scale_x = 1.0 / (x_max - x_min)
        scale_y = 1.0 / (y_max - y_min)
        scaled_grid = np.column_stack((grid[:, 0] * scale_x, grid[:, 1] * scale_y))
        return scaled_grid
    
    centers = scale_to_unit_square(hex_grid)
    radii = np.full(num_circles, 1.0)  # Initial estimate
    
    # Objective function to maximize total radius
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize total
    
    # Constraint functions
    def constraint_bounds(radii):
        return [radii[i] >= 0 for i in range(num_circles)]
    
    def constraint_boundary(radii):
        return [np.all(centers[i][0] - radii[i] >= -1e-12) and
                np.all(centers[i][0] + radii[i] <= 1 + 1e-12) and
                np.all(centers[i][1] - radii[i] >= -1e-12) and
                np.all(centers[i][1] + radii[i] <= 1 + 1e-12) for i in range(num_circles)]
    
    def constraint_overlap(radii):
        overlaps = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    overlaps.append(False)
                else:
                    overlaps.append(True)
        return overlaps
    
    # Combine all constraints
    constraints = []

    result = opt.minimize(

        method='SLSQP',
