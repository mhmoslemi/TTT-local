import numpy as np
import scipy.optimize as opt
import math

def run_packing():
    # Step 1: Initial hexagonal grid placement
    n = 26
    rows = math.ceil(math.sqrt(n))  # Determine number of rows in hexagonal grid
    cols = math.ceil(n / rows)

    # Estimate maximum radius based on packing density
    initial_max_radius = 0.25  # Initial estimate

    # Initial hexagonal grid: place circles in a hexagonal grid
    centers = np.zeros((n, 2))
    radii = np.zeros(n) + initial_max_radius

    for i in range(n):
        col = i % cols
        row = i // cols

        centers[i, 0] = x
        centers[i, 1] = y

    # Normalize to fit in unit square
    centers = centers / np.max(centers)

        return -np.sum(radii)  # Negative for minimization

        radii = x[2*n:]

        return 0.0

        return 0.0

    bounds = []

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
