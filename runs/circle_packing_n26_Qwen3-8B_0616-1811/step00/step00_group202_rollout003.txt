import numpy as np
import cvxpy as cp

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define the number of circles
    n = 26

    # Step 1: Initial Hexagonal Packing

    # Initial hexagonal packing in a unit square
    # We'll use two rows for a hexagonal lattice
    # Adjust this to fit in [0,1]^2

    # Number of rows
    rows = 2
    # Number of circles per row (we'll adjust to get 26 circles)
    cols = n // rows
    # Adjust to fit 26 circles
    cols = 13
    rows = 2

    # Generate hexagonal lattice points
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * i
            y = i * np.sqrt(3) / 2
            # Scale to [0, 1] square
            x = x / (cols - 1) if cols > 1 else 0
            y = y / (rows - 1) if rows > 1 else 0
            centers.append([x, y])

    # Convert to numpy array
    centers = np.array(centers)

    # Initial guess for radii (uniform)
    radii = np.ones(n) * 0.1

    # Step 2: Optimization using CVXPY

    # Define variables
    r = cp.Variable(n)
    x = cp.Variable((n, 2))

    # Objective: Maximize sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # All circles must be within the unit square
    for i in range(n):
        constraints.append(x[i, 0] - r[i] >= -1e-8)
        constraints.append(x[i, 0] + r[i] <= 1 + 1e-8)
        constraints.append(x[i, 1] - r[i] >= -1e-8)
        constraints.append(x[i, 1] + r[i] <= 1 + 1e-8)

    # No overlaps between circles
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.sqrt((x[i, 0] - x[j, 0])**2 + (x[i, 1] - x[j, 1])**2)
            constraints.append(dist >= r[i] + r[j] - 1e-8)

    # Initial guess for x (centers)
    x_initial = centers

    # Solve the optimization problem
    problem = cp.Problem(objective, constraints)
    result = problem.solve(solver=cp.ECOS, verbose=False, warm_start=True, x_initial=x_initial, r_initial=radii)

    if result is None or result != 0:
        # Fallback to initial packing if optimization fails
        print("Optimization failed, using initial packing")
        return centers, radii, np.sum(radii)

    # Extract results
    optimized_centers = x.value
    optimized_radii = r.value

    # Step 3: Validate the packing
    is_valid, message = validate_packing(optimized_centers, optimized_radii)
    if not is_valid:
        print(f"Validation failed: {message}")
        return centers, radii, np.sum(radii)

    # Return the optimized centers, radii, and sum of radii
    return optimized_centers, optimized_radii, np.sum(optimized_radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
