import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial setup: create a hexagonal grid
    num_circles = 26
    radius_initial = 0.05  # Initial guess for radii
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, radius_initial)

    # Generate an initial hexagonal packing in a square
    # We will create a hexagonal grid inside the square
    # Let's use a 5x5 grid with some rows shifted
    rows = 5
    cols = 5
    offset = 0.5  # horizontal offset for alternate rows
    spacing = 1.2  # spacing between centers
    radius_min = 0.02

    for i in range(rows):
        for j in range(cols):
            x = j * spacing
            y = i * spacing
            if i % 2 == 1:
                x += offset
            # Scale to fit in [0,1] square
            x = x / (rows * spacing)
            y = y / (rows * spacing)
            centers[i * cols + j] = [x, y]
            # Adjust radius based on spacing
            radii[i * cols + j] = min(radius_initial, spacing / 2 - 1e-5)

    # Define objective function
    def objective(radii_flat):
        return -np.sum(radii_flat)

    # Define constraints
    def constraint_circle_inside(x):
        for i in range(num_circles):
            x_i, y_i = x[i*2], x[i*2+1]
            r_i = radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1  # Violation
        return 0

    def constraint_no_overlap(x):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i = x[i*2], x[i*2+1]
                x_j, y_j = x[j*2], x[j*2+1]
                r_i = radii[i]
                r_j = radii[j]
                dist_sq = (x_i - x_j)**2 + (y_i - y_j)**2
                if np.sqrt(dist_sq) < r_i + r_j - 1e-8:
                    return -1  # Violation
        return 0

    # Flatten centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Set bounds for radii: [radius_min, ...]
    bounds = [(radius_min, 1.0) for _ in range(num_circles * 2)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap},
    ]

    # Optimize
    result = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-5
    )

    # Extract best solution
    best_centers = result.x[0::2].reshape(-1, 2)
    best_radii = result.x[1::2]
    best_sum = np.sum(best_radii)

    # Validate
    valid, msg = validate_packing(best_centers, best_radii)
    if not valid:
        raise ValueError(f"Validation failed: {msg}")

    return best_centers, best_radii, best_sum
