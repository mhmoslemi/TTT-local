            y = row * (1.0 / (5)) * (1.0 + 0.1) + np.random.uniform(-0.02, 0.02)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    def nested_positions(n):

            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    def grid_positions(n):

            row = i // 5
