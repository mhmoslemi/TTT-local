            row = i // 6
            col = i % 6
            x = col * (side_length * np.sqrt(3))
            y = row * (side_length * 1.5)
            positions.append([x, y])
        return np.array(positions)

    # Step 2: Try to find an optimal packing
    # Generate initial positions and radii
    n = 26
    side_length = 0.15  # Initial side length in hexagonal packing
    initial_positions = hexagonal_positions(n, side_length)
    initial_radii = np.full(n, side_length)

    # Normalize positions to fit within the unit square
    min_x, min_y = initial_positions.min(axis=0)
    max_x, max_y = initial_positions.max(axis=0)
    scale = min(1.0 / max_x, 1.0 / max_y)
    initial_positions *= scale
    initial_radii *= scale

    # Define objective function

        # x is a flattened array of [pos1_x, pos1_y, r1, pos2_x, pos2_y, r2, ...]

        radii = x[2*n:]

        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([centers[:, 0] - radii - 1e-8, 1 - centers[:, 0] - radii - 1e-8,

        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-8
                idx += 1

    x0 = np.concatenate([initial_positions.flatten(), initial_radii])

import numpy as np

from scipy.optimize import minimize

def run_packing():
    # Initial hexagonal grid with 26 circles
    n = 26
    max_radius = 0.05  # Start with small radii
    max_rows = int(np.sqrt(n)) + 1
    cols = n // max_rows + 1 if n % max_rows != 0 else n // max_rows
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initialize hexagonal grid
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col + 0.5 * (row % 2)
        y = row * np.sqrt(3) / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = max_radius

    # Rescale to fit in unit square
    max_center = np.max(centers)
    centers /= max_center
    radii *= 1.0 / max_center

    # Objective function to maximize sum of radii
    def objective(x):
        # x is a flat array with (n*2) elements for centers and (n) elements for radii
        centers_flat = x[:n*2].reshape((n, 2))
        radii_flat = x[n*2:]

        # Ensure radii are nonnegative
        radii_flat = np.maximum(radii_flat, 0)

        # Update centers and radii
        centers = centers_flat
        radii = radii_flat

        # Validate packing
        valid, _ = validate_packing(centers, radii)
        if not valid:
            return -np.inf

        # Compute sum of radii
        sum_radii = np.sum(radii)

        return sum_radii

    # Initial guess for optimization
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization settings
    bounds = []
    for _ in range(n*2):
        bounds.append((0.0, 1.0))  # Centers must be in [0,1]^2
    for _ in range(n):
        bounds.append((0.0, 1.0))  # Radii must be nonnegative

    # Optimize
    result = minimize(
        fun=objective,
        x0=initial_guess,
        method='L-BFGS-B',
        bounds=bounds,
        options={'maxiter': 500, 'ftol': 1e-8}
    )

    # Extract final centers and radii
    centers = result.x[:n*2].reshape((n, 2))
    radii = result.x[n*2:]

    # Re-validate
    valid, _ = validate_packing(centers, radii)
    if not valid:
        return np.zeros((n, 2)), np.zeros(n), 0.0

    sum_radii = np.sum(radii)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
