import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial positions using a hexagonal grid pattern
    n = 26
    rows = int(np.ceil(n / 6))  # Adjusting for hexagonal packing
    cols = 6
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col + (row % 2) * 0.5
        y = row * np.sqrt(3) / 2
        centers[i, 0] = x / 6  # Scale to fit in [0, 1]
        centers[i, 1] = y / (rows * np.sqrt(3) / 2)

    # Initial guess for radii (equal radii for simplicity)
    initial_radius = 0.1
    radii = np.full(n, initial_radius)

    # Define optimization bounds: centers within [0, 1]^2, radii non-negative
    bounds_centers = [(0, 1) for _ in range(2 * n)]
    bounds_radii = [(0, 1) for _ in range(n)]

    # Objective function to maximize the sum of radii
    def objective(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:].reshape(n)
        return -np.sum(radii_flat)  # Minimize negative of sum

    # Constraints: circles within square and no overlap
    def constraints(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:].reshape(n)
        constraints = []

        # Circle within square
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12
                or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return np.inf  # Penalize violations
    
        # Circle-to-circle distances
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf  # Penalize violations

        return 0  # No constraints violated

    # Optimization setup
    x0 = np.concatenate([centers.flatten(), radii.flatten()])
    result = optimize.minimize(
        objective,
        x0,
        bounds=np.concatenate([bounds_centers, bounds_radii]),
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
        options={"maxiter": 500, "ftol": 1e-8, "eps": 1e-8}
    )

    # Extract final centers and radii
    centers_opt = result.x[:2 * n].reshape(n, 2)
    radii_opt = result.x[2 * n:].reshape(n)
    sum_radii = np.sum(radii_opt)

    # Validate packing for safety
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        print("Invalid packing:", msg)
        return np.zeros((n, 2)), np.zeros(n), 0.0

Approach 2:

import numpy as np
import scipy.optimize as optimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing
    num_circles = 26
    rows = int(np.sqrt(num_circles))
    cols = (num_circles + rows - 1) // rows

    # Hexagonal packing parameters
    hex_radius = 1.0 / (2 * np.sqrt(3))  # Radius to fill the unit square
    spacing = 2 * hex_radius  # Distance between centers in hexagonal pattern

    # Initial positions
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col * spacing
        y = row * spacing * np.sqrt(3) / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = hex_radius

    # Adjust positions to fit in unit square
    scale = 1.0 / (np.max(centers) + np.max(radii))
    centers *= scale
    radii *= scale

    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Define constraints: circles within square and non-overlapping
    def constraint_within_square(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1
        return 0

    def constraint_no_overlap(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1
        return 0

    # Optimization variables:
    # - First 2*num_circles entries: coordinates of centers
    # - Last num_circles entries: radii of circles
    x0 = np.concatenate((centers.flatten(), radii))

    # Optimization bounds: circles are within [0,1]^2, radii >= 0
    bounds = [(0, 1) for _ in range(2 * num_circles)] + [(0, 1) for _ in range(num_circles)]

    # Optimization constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_within_square},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8,
        options={'maxiter': 500}
    )

    # Extract results
    centers_opt = result.x[:2*num_circles].reshape((num_circles, 2))
    radii_opt = result.x[2*num_circles:]
    sum_radii_opt = np.sum(radii_opt)

    # Validate the packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers_opt, radii_opt, sum_radii_opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
