=== SYSTEM ===
You are an expert computational geometer and numerical optimization specialist evaluating two algorithmic strategies for the Circle Packing problem. The objective is to pack N circles within a [0,1]x[0,1] unit square to maximize the sum of their radii without any overlap.

A messy, hardcoded, or unconventional script that uses superior geometric heuristics to achieve a tighter packing is strictly better than a clean, standard script that gets trapped in a local minimum.

=== USER ===
Goal: Discover a program whose output maximizes the metric 'sum of radii' (target 2.636). We search by repeatedly expanding promising programs sampled from a buffer.

Evaluate the structural potential of the two approaches based purely on these mathematical criteria:
* Initialization Strategy: Does the algorithm seed the centers intelligently (e.g., hexagonal lattices, staggered grids, boundary-hugging) rather than relying on pure uniform random placement?
* Optimization Engine: Does it utilize a robust method for non-convex spaces (e.g., SLSQP with strict bounds, simulated annealing, basin-hopping, or physics-based repulsion/expansion)?
* Local Optima Escape: Does the algorithm include perturbation mechanics (jiggling, swapping, targeted radius inflation) to break out of jammed configurations?
* Constraint Handling: How mathematically sound is the enforcement of the `x, y +/- r <= 1` boundaries and the `dist >= r_i + r_j` overlap rules?

Judge only the merit and headroom of each approach. Do not assume that the
already-measured score decides the comparison.

```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
from itertools import combinations

def run_packing():
    # Initial hexagonal grid packing
    num_circles = 26
    num_rows = 5
    circles_per_row = 5
    spacing = 1.0 / (num_rows + 1)  # vertical spacing
    horizontal_shift = 0.5 / (circles_per_row + 1)  # horizontal shift for hexagonal packing

    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Initial guess for radii
    for i in range(num_rows):
        for j in range(circles_per_row):
            row_idx = i * circles_per_row + j
            x = j * (1.0 / circles_per_row) + horizontal_shift
            y = i * spacing + spacing
            centers[row_idx] = [x, y]
            radii[row_idx] = 0.1  # Initial small radius

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraint functions
    def constraint1(radii):
        return np.sum(radii) - 2.636  # Ensure the sum is at least 2.636

    def constraint2(radii):
        return np.max(radii) - 0.1  # Ensure no radius exceeds 0.1 initially

    # Bounds for radii: non-negative
    bounds = [(0, 0.1) for _ in range(num_circles)]

    # Initial radius guess
    initial_radii = np.full(num_circles, 0.1)

    # Optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(
        objective,
        initial_radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    optimal_radii = result.x
    optimal_centers = centers.copy()

    # Local refinement: move circles slightly to fill gaps
    for _ in range(5):  # Run local refinement multiple times
        for i in range(num_circles):
            # Try moving this circle slightly
            dx = np.random.uniform(-0.01, 0.01)
            dy = np.random.uniform(-0.01, 0.01)
            new_center = optimal_centers[i] + [dx, dy]

            # Check if new center is within bounds
            if (new_center[0] < -1e-12 or new_center[0] > 1 + 1e-12 or
                new_center[1] < -1e-12 or new_center[1] > 1 + 1e-12):
                continue

            # Check for overlap with other circles
            overlap = False
            for j in range(num_circles):
                if i == j:
                    continue
                dist = np.hypot(new_center[0] - optimal_centers[j][0],
                                new_center[1] - optimal_centers[j][1])
                if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                    overlap = True
                    break

            if not overlap:
                optimal_centers[i] = new_center

    # Final check for boundary and overlap
    for i in range(num_circles):
        x, y = optimal_centers[i]
        r = optimal_radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return np.zeros((26, 2)), np.zeros(26), 0.0

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.hypot(optimal_centers[i][0] - optimal_centers[j][0],
                            optimal_centers[i][1] - optimal_centers[j][1])
            if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                return np.zeros((26, 2)), np.zeros(26), 0.0

    return optimal_centers, optimal_radii, np.sum(optimal_radii)
```

Approach 2:
```python
import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial placement of 25 circles in a grid
    num_circles = 26
    grid_size = 5
    grid_spacing = 1.0 / (grid_size - 1)
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # First 25 circles in grid
    for i in range(grid_size):
        for j in range(grid_size):
            if i * grid_size + j < 25:
                centers[i * grid_size + j, 0] = j * grid_spacing
                centers[i * grid_size + j, 1] = i * grid_spacing
                radii[i * grid_size + j] = 0.1  # Initial radius

    # Place the 26th circle at a central position
    centers[25, 0] = 0.5
    centers[25, 1] = 0.5
    radii[25] = 0.1

    def objective(radii_vector):
        return -np.sum(radii_vector)

    def constraints_func(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:]
        centers = centers_flat
        radii = radii_flat

        # Boundary constraints
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf  # Violates boundary

        # Non-overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf  # Overlap

        # Return negative sum of radii (since we're minimizing)
        return -np.sum(radii)

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Bounds: center coordinates between 0 and 1, radii non-negative
    bounds = []
    for _ in range(num_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    # Optimization settings
    result = opt.minimize(
        fun=constraints_func,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        options={
            'ftol': 1e-9,
            'maxiter': 1000,
            'disp': False
        }
    )

    # Extract the solution
    optimized_x = result.x
    centers_opt = optimized_x[:2 * num_circles].reshape(num_circles, 2)
    radii_opt = optimized_x[2 * num_circles:]

    # Check validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Optimization failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
