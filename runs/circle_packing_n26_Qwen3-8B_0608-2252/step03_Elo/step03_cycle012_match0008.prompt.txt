import numpy as np
import random
import scipy.optimize as optimize
import math
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initial placement using a hybrid of hexagonal and nested grid
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    spacing = 1.0 / (cols + 1)

    for i in range(n):
        row = i // cols
        col = i % cols
        # Use a hexagonal pattern for better packing
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = spacing / 2

    # Define objective function for optimization
    def objective(x):
        centers_opt = x[:2*n].reshape(n, 2)
        radii_opt = x[2*n:]

        # Check boundary constraints
        for i in range(n):
            x, y = centers_opt[i]
            r = radii_opt[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return float('inf')

        # Check circle overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_opt[i] + radii_opt[j] - 1e-12:
                    return float('inf')

        # Objective is to maximize total radius (negative for minimization)
        return -np.sum(radii_opt)

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: centers in [0,1], radii >= 0
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Use a more global optimization method (simulated annealing)
    result = optimize.minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        options={'maxiter': 1000, 'ftol': 1e-6}
    )

    # If optimization failed, use a random perturbation
    if result.success:
        centers_opt = result.x[:2*n].reshape(n, 2)
        radii_opt = result.x[2*n:]
    else:
        # Fallback to perturbed initial guess
        centers_opt = centers + np.random.rand(n, 2) * 0.1
        radii_opt = radii + np.random.rand(n) * 0.1

        # Ensure centers and radii are within bounds
        for i in range(n):
            centers_opt[i, 0] = np.clip(centers_opt[i, 0], 0, 1)
            centers_opt[i, 1] = np.clip(centers_opt[i, 1], 0, 1)
            radii_opt[i] = np.clip(radii_opt[i], 0, 1)

    # Final check and validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Warning: Packing validation failed: {message}")
        return centers, radii, np.sum(radii)

    return centers_opt, radii_opt, np.sum(radii_opt)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as optimize
import math

def run_packing():

    # Create initial centers based on a hexagonal close packing within the square
    # Initial radius guess based on the target sum

        row = i // max_cols

            x = col * (2 * radius_guess) + radius_guess
        else:
            x = col * (2 * radius_guess) + radius_guess + radius_guess * math.sqrt(3) / 2
        y = row * (2 * radius_guess) + radius_guess
        
        centers[i] = [x, y]
    
    # Initial guess for radii
    radii_initial = np.full(n, radius_guess)
    
    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize
    
    # Constraints function
    def constraints(radii):
        constraints = []
        
        # Bounds: radii >= 0
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Distance constraints: distance between centers >= sum of radii
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: dist - (r[i] + r[j])})
        
        # Boundary constraints: circles must lie within the unit square
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] + x - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - x - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] + y - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - y - 1e-12})
        
        return constraints
    
    # Optimization options
    options = {
        'maxiter': 1000,
        'ftol': 1e-6,
        'disp': False
    }
    
    # Use a better optimization method
    result = optimize.minimize(
        fun=objective,
        x0=radii_initial,
        constraints=constraints(radii_initial),
        method='SLSQP',
        options=options
    )
    
    # Extract the result
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)
    
    # Post-processing to optimize boundary placements
    def optimize_boundary_circles(centers, radii):
        new_centers = np.copy(centers)
        new_radii = np.copy(radii)
        
        # Re-center circles near the boundaries
        for i in range(n):
            x, y = new_centers[i]
            r = new_radii[i]
            
            # If the circle is too close to the boundary, re-center it
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                # Move the circle toward the nearest boundary
                dx = max(0, x - r) - x
                dy = max(0, y - r) - y
                if dx < dy:
                    dx = -min(x - r, 0)
                    dy = 0
                else:
                    dy = -min(y - r, 0)
                    dx = 0
                new_centers[i] = [x + dx, y + dy]
        
        return new_centers, new_radii
    
    centers, radii = optimize_boundary_circles(centers, radii)
    
    # Validate the result using the provided function
    is_valid, message = validate_packing(centers, radii)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0
    
    return centers, radii, sum_radii
