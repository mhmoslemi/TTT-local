import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    n = 26
    centers = np.zeros((n, 2))
    
    # Step 1: Create an optimized initial layout using hexagonal packing
    # Approximate number of rows for hexagonal packing
    rows = int(math.ceil(math.sqrt(n)))
    cols = int(math.ceil(n / rows))
    
    for i in range(n):
        row = i // cols
        col = i % cols
        # Hexagonal grid: even rows offset by 0.5 in x-direction
        if row % 2 == 0:
            x = col * (1.0 / cols) + (1.0 / (2 * cols))
        else:
            x = col * (1.0 / cols) + 0.5 / cols
        y = row * (1.0 / rows) + 0.5 / rows
        centers[i] = [x, y]
    
    # Step 2: Estimate initial radius based on area
    # Use a more optimistic radius that allows for better packing
    initial_radius = np.sqrt(1 / (n * np.pi)) * 1.1  # Slightly increase to allow for optimization
    
    # Initial guess for radii
    radii_initial = np.full(n, initial_radius)
    
    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize
    
    # Constraints: non-overlapping and bounded within the unit square
    def constraints(radii):
        constraints = []
        
        # Radii must be non-negative
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Distance between centers must be >= sum of radii
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: dist - (r[i] + r[j])})
        
        # Boundary constraints: must lie within the unit square
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] + x - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - x - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] + y - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - y - 1e-12})
        
        return constraints
    
    # Optimization options
    options = {
        'maxiter': 1500,  # Increase the maximum number of iterations for better convergence
        'ftol': 1e-10,    # Very tight tolerance for precision
        'disp': False
    }
    
    # Run optimization
    result = optimize.minimize(
        fun=objective,
        x0=radii_initial,
        constraints=constraints(radii_initial),
        method='SLSQP',
        options=options
    )
    
    # Extract the result
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)
    
    # Validate the result using the provided function
    is_valid, message = validate_packing(centers, radii)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0
    
    return centers, radii, sum_radii
