import numpy as np
import scipy.optimize as optimize
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: hexagonal arrangement of circles
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # First, place in a hexagonal grid
    rows = int(np.sqrt(n))
    cols = n // rows + (1 if n % rows != 0 else 0)
    spacing = 1.0 / (cols + 1)

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i, 0] = x
        centers[i, 1] = y
        # Assume a minimum radius to ensure spacing
        radii[i] = spacing / 2 - 1e-4

    # Optimize the centers and radii
    def objective(x):
        centers_opt = x[:2*n].reshape(n, 2)
        radii_opt = x[2*n:]

        # Ensure constraints: radii >= 0
        if np.any(radii_opt < 0):
            return float('inf')

        # Ensure circles are inside the unit square
        for i in range(n):
            x, y = centers_opt[i]
            r = radii_opt[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return float('inf')

        # Ensure circles do not overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_opt[i] + radii_opt[j] - 1e-12:
                    return float('inf')

        # Objective is to maximize sum of radii (negative for minimization)
        return -np.sum(radii_opt)

    # Initial guess for the optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization bounds: centers must be in [0, 1] and radii >= 0
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        options={'maxiter': 1000, 'ftol': 1e-6}
    )

    # Extract optimized centers and radii
    centers_opt = result.x[:2*n].reshape(n, 2)
    radii_opt = result.x[2*n:]

    # Final check and validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Warning: Packing validation failed: {message}")
        return centers, radii, np.sum(radii)

    return centers_opt, radii_opt, np.sum(radii_opt)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    n = 26
    # Initial guess for centers using hexagonal packing
    radius_guess = 0.05  # Initial guess for radius
    
    # Estimate rows and columns for hexagonal packing
    max_cols = int(math.ceil(math.sqrt(n)))
    max_rows = int(math.ceil(n / max_cols))
    
    centers = np.zeros((n, 2))
    for i in range(n):
        col = i % max_cols
        row = i // max_cols
        
        # Hexagonal packing: alternate row offsets
        if row % 2 == 0:
            x = col * (2 * radius_guess) + radius_guess
        else:
            x = col * (2 * radius_guess) + radius_guess + radius_guess * math.sqrt(3) / 2
        y = row * (2 * radius_guess) + radius_guess
        
        centers[i] = [x, y]
    
    # Initial guess for radii
    radii_initial = np.full(n, radius_guess)
    
    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize
    
    # Constraints function
    def constraints(radii):
        constraints = []
        
        # Bounds: radii >= 0
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Distance constraints: distance between centers >= sum of radii
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: dist - (r[i] + r[j])})
        
        # Boundary constraints: circles must lie within the unit square
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] + x - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - x - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] + y - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - y - 1e-12})
        
        return constraints
    
    # Optimization options
    options = {
        'maxiter': 500,
        'ftol': 1e-6,
        'disp': False
    }
    
    # Use a better optimization method
    result = optimize.minimize(
        fun=objective,
        x0=radii_initial,
        constraints=constraints(radii_initial),
        method='COBYLA',
        options=options
    )
    
    # Extract the result
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)
    
    # Validate the result using the provided function
    is_valid, message = validate_packing(centers, radii)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0
    
    return centers, radii, sum_radii
