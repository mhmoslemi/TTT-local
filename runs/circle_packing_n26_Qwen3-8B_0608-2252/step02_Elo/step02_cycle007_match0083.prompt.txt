import scipy.optimize as optimize
import math

def run_packing():
    n = 26
    centers = np.zeros((n, 2))
    
    # Step 1: Create an optimized initial layout
    # First, approximate square packing
    max_rows = int(math.ceil(math.sqrt(n)))
    max_cols = int(math.ceil(n / max_rows))

        row = i // max_cols
        col = i % max_cols
        x = col * (1.0 / max_cols) + 0.5 / max_cols

        centers[i] = [x, y]
    
    # Step 2: Estimate initial radius based on area
    # Maximum possible radius if all circles are equally sized and packed into the square
    # Use area = n * π * r^2 <= 1

    initial_radius = np.sqrt(1 / (n * np.pi))
    
    # Initial guess for radii
    radii_initial = np.full(n, initial_radius)
    
    # Objective function: maximize the sum of radii

        return -np.sum(radii)  # Minimize negative sum to maximize

    def constraints(radii):
        constraints = []
        
        # Radii must be non-negative
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Distance between centers must be >= sum of radii
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: dist - (r[i] + r[j])})

            x, y = centers[i]

            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - y - 1e-12})
        
        return constraints

    result = optimize.minimize(
        fun=objective,
        x0=radii_initial,

        method='SLSQP',
