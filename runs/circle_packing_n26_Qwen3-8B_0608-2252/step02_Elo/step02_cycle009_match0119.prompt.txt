import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial hexagonal grid arrangement
    def hexagonal_grid(n_circles):
        r = 0.05  # Initial radius (to be optimized)
        centers = []
        for i in range(n_circles):
            row = i // 3
            col = i % 3
            x = col * (2 * r) + r
            y = row * (np.sqrt(3) * r) + r
            centers.append([x, y])
        centers = np.array(centers)
        return centers, np.full(n_circles, r)

    # Objective function to maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    # Constraints for optimization
    def constraints(radii, centers):
        n = len(radii)
        constraints = []
        # Constraint 1: All circles must fit in the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - y - r})
        # Constraint 2: No overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                def dist_constraint(radii, i=i, j=j):
                    r_i = radii[i]
                    r_j = radii[j]
                    dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                    return dist - r_i - r_j
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        return constraints

    # Initial guess for radii
    centers, initial_radii = hexagonal_grid(26)
    bounds = [(0, 1) for _ in range(26)]  # Radii must be between 0 and 1

    # Set up optimization
    result = minimize(
        fun=objective,
        x0=initial_radii,
        args=(centers,),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints(initial_radii, centers)
    )

    # Extract the solution
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Check if the solution is valid
    valid, message = validate_packing(centers, radii)
    if not valid:
        print(f"Optimization failed: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import math

def run_packing():

    # Initial guess for centers using hexagonal packing
    radius_guess = 0.05  # Initial guess for radius

        col = i % max_cols
        row = i // max_cols

        else:
            x = col * (2 * radius_guess) + radius_guess + radius_guess * math.sqrt(3) / 2
        y = row * (2 * radius_guess) + radius_guess
        
        centers[i] = [x, y]
    
    # Initial guess for radii
    radii_initial = np.full(n, radius_guess)
    
    # Objective function to maximize the sum of radii

        constraints = []
        
        # Bounds: radii >= 0

            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Distance constraints: distance between centers >= sum of radii

            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: dist - (r[i] + r[j])})

            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] + x - 1e-12})
