import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

        row_count = 5
        col_count = 5
        positions = []
        for i in range(n):
            row = i // col_count
            col = i % col_count
            # Use hexagonal packing
            x = col * (1 / (col_count - 1)) + np.random.uniform(-0.02, 0.02)
            y = row * (1 / (row_count - 1)) + np.random.uniform(-0.02, 0.02)
            # Add slight random offset to spread out
            x += np.random.uniform(-0.03, 0.03)
            y += np.random.uniform(-0.03, 0.03)
            # Ensure coordinates are within the unit square
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)

    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1 - centers[:, 1] - radii - 1e-6
        ])

    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist
