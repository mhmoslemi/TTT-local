def generate_initial_config(n):
    """Generate an initial configuration with a mix of hexagonal and square packing."""
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        h = 1.0 / np.sqrt(3)
        v = h * np.sqrt(3) / 2
        if row % 2 == 1:
            y_offset = v / 2
        else:
            y_offset = 0
        x = col * h
        y = row * v + y_offset
        positions.append([x, y])
    # Introduce some randomness to avoid perfect hexagonal symmetry
    np.random.seed(42)
    for i in range(n):
        if np.random.rand() < 0.1:
            positions[i][0] += np.random.uniform(-0.1, 0.1)
            positions[i][1] += np.random.uniform(-0.1, 0.1)
    return np.array(positions)

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
