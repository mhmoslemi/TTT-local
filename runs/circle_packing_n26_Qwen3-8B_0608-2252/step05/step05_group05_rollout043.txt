import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    def hexagonal_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.3
            y = row * 0.3 + (col % 2) * 0.15
            positions.append([x, y])
        return np.array(positions)

    def grid_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.2 + 0.1
            y = row * 0.2 + 0.1
            positions.append([x, y])
        return np.array(positions)

    def random_positions(n):
        return np.random.rand(n, 2)

    def perturb_positions(positions, scale=0.1):
        return positions + np.random.uniform(-scale, scale, size=positions.shape)

    def normalize_positions(positions):
        if positions.size == 0:
            return np.zeros((n, 2))
        max_x, max_y = positions.max(axis=0)
        scale = min(1.0 / max_x, 1.0 / max_y)
        return positions * scale

    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        sum_r = np.sum(radii)
        return -sum_r

    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-8,
            1.0 - centers[:, 0] - radii - 1e-8,
            centers[:, 1] - radii - 1e-8,
            1.0 - centers[:, 1] - radii - 1e-8
        ])

    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-8
                idx += 1
        return dist

    def validate_packing(centers, radii):
        n = centers.shape[0]

        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"

        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))

            best_centers = centers

    return (best_centers, best_radii, best_sum)
