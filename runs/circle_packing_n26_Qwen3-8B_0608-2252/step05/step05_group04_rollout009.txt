    best_radii = np.zeros(n)

        x0 = np.concatenate([positions.flatten(), initial_radii])

        def objective(x):

            radii = x[2*n:]
            return -np.sum(radii)

            centers = x[:2*n].reshape(n, 2)
            radii = x[2*n:]
            return np.concatenate([
