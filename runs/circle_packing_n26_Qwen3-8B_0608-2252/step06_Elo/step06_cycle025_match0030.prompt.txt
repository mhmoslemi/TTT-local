            y_offset = v / 2
        else:
            y_offset = 0
        x = col * h
        y = row * v + y_offset
        positions.append([x, y])
    
    # Add some circles near the corners and edges
    corner_positions = [
        [0.1, 0.1], [0.1, 0.9], [0.9, 0.9], [0.9, 0.1],

        [0.1, 0.2], [0.1, 0.8], [0.9, 0.2], [0.9, 0.8]
    ]
    for pos in corner_positions:
        if len(positions) < n:
            positions.append(pos)

    if len(positions) > n:

    return np.array(positions)
