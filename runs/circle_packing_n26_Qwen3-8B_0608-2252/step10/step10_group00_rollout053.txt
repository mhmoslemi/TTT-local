            row = i // cols
            col = i % cols
            x = col * (1.0 / (cols - 1)) + np.random.uniform(-0.01, 0.01)
            y = row * (1.0 / (rows - 1)) + np.random.uniform(-0.01, 0.01)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

        radii = x[2*n:]
        return -np.sum(radii)

        radii = x[2*n:]
        return np.concatenate([

            1 - centers[:, 0] - radii - 1e-6,
