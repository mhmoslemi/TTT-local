import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial configuration: square grid of 26 circles
    # We divide the square into a 5x5 grid (25 circles) and one extra circle
    n = 26
    row, col = np.meshgrid(np.linspace(0.1, 0.9, 5), np.linspace(0.1, 0.9, 5))
    centers = np.column_stack((row.flatten(), col.flatten()))
    centers = np.vstack((centers, [0.5, 0.5]))  # Add one extra circle at the center
    radii = np.array([0.1] * n)  # Initial equal radii

    # Objective function: maximize the sum of radii
    def objective(x):
        # x contains radii
        return -np.sum(x)  # Minimize the negative sum

    # Constraints: centers must be within [0,1]^2 and no circles can overlap
    def constraint_centers(radii):
        # Ensure centers are within [0,1]^2
        for i in range(n):
            x, y = centers[i]
            if (x - radii[i] < 0 or x + radii[i] > 1 or
                y - radii[i] < 0 or y + radii[i] > 1):
                return False
        return True

    def constraint_overlap(radii):
        # Ensure circles do not overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx * dx + dy * dy)
                if distance < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    # Initial guess: radii
    initial_guess = radii.copy()

    # Define constraints for the optimization
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_centers(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
    ]

    # Perform optimization
    result = opt.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1)] * n
    )

    # Extract results
    optimized_radii = result.x
    optimized_centers = centers.copy()
    sum_radii = np.sum(optimized_radii)

    # Return the optimized configuration
    return (optimized_centers, optimized_radii, sum_radii)
