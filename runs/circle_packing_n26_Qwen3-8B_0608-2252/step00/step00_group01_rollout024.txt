import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess for centers in a 5x5 grid with an extra circle
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Create a 5x5 grid
    for i in range(5):
        for j in range(5):
            centers[i * 5 + j] = [j / 4.0, i / 4.0]

    # Place the 26th circle in a hexagonal-like position
    idx = 25
    centers[idx] = [0.5, 0.25]  # Example position
    radii[idx] = 0.1  # Small radius to avoid overlap

    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Define constraints
    def constraint_overlap(x):
        x_vec = x[:num_circles]
        y_vec = x[num_circles:]
        centers = x_vec.reshape((num_circles, 2))
        radii = y_vec
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -1.0  # Violates non-overlap constraint
        return 0.0

    def constraint_boundary(x):
        x_vec = x[:num_circles]
        y_vec = x[num_circles:]

                return -1.0  # Violates boundary constraint
        return 0.0

    def constraint_radius(x):
        x_vec = x[num_circles:]
        for r in x_vec:
            if r < 0:
                return -1.0
        return 0.0

    # Prepare optimization variables
    x0 = np.concatenate([centers.flatten(), radii])

    # Define bounds: centers must be in [0, 1], radii must be >= 0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
