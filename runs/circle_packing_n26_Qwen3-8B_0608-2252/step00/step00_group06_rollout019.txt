import numpy as np
import random
import math

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initialize_packing(num_circles, population_size=100):
    centers = np.random.rand(population_size, num_circles, 2)
    radii = np.random.rand(population_size, num_circles) * 0.1
    return centers, radii

def evaluate_fitness(centers, radii):
    valid, _ = validate_packing(centers, radii)
    if not valid:
        return -np.inf
    return np.sum(radii)

def mutate_config(config, mutation_rate=0.1, mutation_strength=0.1):
    centers = config[0]
    radii = config[1]
    new_centers = centers.copy()
    new_radii = radii.copy()

    for i in range(centers.shape[0]):
        for j in range(centers.shape[1]):
            if random.random() < mutation_rate:
                new_centers[i][j][0] += np.random.uniform(-mutation_strength, mutation_strength)
                new_centers[i][j][1] += np.random.uniform(-mutation_strength, mutation_strength)
                new_radii[i][j] += np.random.uniform(-mutation_strength, mutation_strength)

    new_centers = np.clip(new_centers, 0, 1)
    new_radii = np.clip(new_radii, 0, 1)
    return new_centers, new_radii

def crossover(parent1, parent2):

    centers2, radii2 = parent2

                new_centers[i][j] = centers2[i][j]
                new_radii[i][j] = radii2[i][j]

    return new_centers, new_radii
