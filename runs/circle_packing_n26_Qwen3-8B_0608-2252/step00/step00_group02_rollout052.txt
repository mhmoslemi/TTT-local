import scipy.optimize as opt
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: hexagonal packing
    num_circles = 26
    rows = int(math.ceil(math.sqrt(num_circles)))
    cols = int(math.ceil(num_circles / rows))
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col + 0.5 * (1 - 1 / cols) * (1 - row % 2)
        y = row * (1 + 1 / cols)
        centers[i] = [x, y]
        radii[i] = 0.01  # Initial small radius

    # Define the optimization problem
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        constraints = []
        for i in range(num_circles):
            constraints.append((radii[i] >= 0, "radius non-negative"))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                def circle_dist(radii):
                    if radii[i] < 1e-12 or radii[j] < 1e-12:
                        return np.inf
                    dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                    return dist - radii[i] - radii[j]
                constraints.append((circle_dist(radii) >= -1e-12, "circles not overlapping"))
        for i in range(num_circles):
            def circle_boundary(radii):
                x, y = centers[i]

                    return np.inf
                return (x - radii[i] >= -1e-12) + (x + radii[i] <= 1 + 1e-12) + (y - radii[i] >= -1e-12) + (y + radii[i] <= 1 + 1e-12)
            constraints.append((circle_boundary(radii) >= 0, "circle within bounds"))
        return constraints

    # Optimization

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
