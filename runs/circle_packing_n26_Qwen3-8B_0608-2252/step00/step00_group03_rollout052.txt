import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial number of circles
    n = 26
    
    # Initial positions: grid with some spacing
    grid_size = 5
    grid_spacing_x = 1.0 / (grid_size - 1)
    grid_spacing_y = 1.0 / (grid_size - 1)
    
    # Create a basic grid layout
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // (grid_size - 1)
        col = i % (grid_size - 1)
        centers[i, 0] = col * grid_spacing_x
        centers[i, 1] = row * grid_spacing_y
    
    # Initial equal radii
    radii = np.full(n, 0.05)
    
    # Function to evaluate the objective (negative of sum of radii)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraint 1: No circle outside the unit square
    def bounds_func(radii):
        constraints = []
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return False
        return True
    
    # Constraint 2: No overlapping circles
    def overlap_func(radii):
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True
    
    # Optimization function
    def optimize_packing(centers, radii):
        # Optimization bounds: all radii must be non-negative
        bounds = [(0.0, 1.0) for _ in range(n)]
        
        # Optimization constraints
        constraints = []
        constraints.append({'type': 'ineq', 'fun': lambda r: bounds_func(r)})
        constraints.append({'type': 'ineq', 'fun': lambda r: overlap_func(r)})
        
        # Optimization
        result = optimize.minimize(
            objective,
            radii,
            bounds=bounds,
            constraints=constraints,
            method='SLSQP',
            tol=1e-8
        )
        return result.x
    
    # Run optimization
    optimized_radii = optimize_packing(centers, radii)
    optimized_centers = centers.copy()
    
    # Final check

    if not valid:
        raise ValueError(error_msg)
    
    # Return results
    return (optimized_centers, optimized_radii, np.sum(optimized_radii))

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
