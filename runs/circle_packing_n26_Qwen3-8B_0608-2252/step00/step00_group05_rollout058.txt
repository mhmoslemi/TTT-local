from scipy.optimize import minimize
import math

    rows = 5
    cols = 5

                x = (j + 0.5) * grid_spacing

            radii[i * cols + j] = radius_initial

    # Objective function: maximize the sum of radii
    def objective(x):
        # x is a flat array where the first half is the centers (2 per circle), then the radii
        centers_flat = x[:2 * num_circles].reshape((num_circles, 2))
        radii_flat = x[2 * num_circles:]
        # We want to maximize the sum of radii, so minimize the negative of the sum
        return -np.sum(radii_flat)

    # Constraints
    def constraint_boundary(x):
        # Check if any circle is outside the unit square
        centers_flat = x[:2 * num_circles].reshape((num_circles, 2))

            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or
