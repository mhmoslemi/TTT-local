import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
from scipy.optimize import Bounds

def create_hexagonal_grid(num_circles):
    # Create a hexagonal grid in a square
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    centers = np.zeros((num_circles, 2))
    index = 0
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            if i % 2 == 0:
                x = j
            centers[index] = [x, y]
            index += 1
            if index >= num_circles:
                break
        if index >= num_circles:
            break
    return centers

def objective_function(radii, centers):
    return -np.sum(radii)  # Minimize negative sum to maximize sum of radii

def constraint_dist(centers, radii, i, j):
    dist = np.linalg.norm(centers[i] - centers[j])
    return dist - radii[i] - radii[j]

def constraint_boundary(centers, radii, i):
    x, y = centers[i]
    r = radii[i]
    return x - r - 1e-12  # x - r >= -1e-12
def constraint_boundary_2(centers, radii, i):

def constraint_boundary_3(centers, radii, i):

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
