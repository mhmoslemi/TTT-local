import numpy as np
import scipy.optimize as opt
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial random centers
    centers = np.random.rand(26, 2)
    # Initialize radii with some randomness (not exceeding 0.1 for safety)
    radii = np.random.uniform(0.01, 0.1, 26)
    # Initial sum of radii
    sum_radii = np.sum(radii)

    # Define bounds for centers and radii
    bounds = []
    for _ in range(26):
        bounds.append((0.0, 1.0))  # x, y
        bounds.append((0.0, 0.1))  # radius

    # Define objective function to minimize (negative of sum of radii)
    def objective(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        sum_r = np.sum(radii_flat)
        return -sum_r  # We minimize the negative to maximize the sum

    # Define constraints
    constraints = []

    # Constraint: All circles must be inside the unit square
    def inside_square(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (0 - 1e-12 <= x_i - r_i <= 1 + 1e-12 and
                    0 - 1e-12 <= y_i - r_i <= 1 + 1e-12 and
                    0 - 1e-12 <= x_i + r_i <= 1 + 1e-12 and
                    0 - 1e-12 <= y_i + r_i <= 1 + 1e-12):
                return -1
        return 0

    constraints.append({'type': 'ineq', 'fun': inside_square})

    # Constraint: No two circles should overlap
    def no_overlap(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return -1
        return 0

    constraints.append({'type': 'ineq', 'fun': no_overlap})

    # Optimization
    result = opt.minimize(

        bounds=bounds,
        constraints=constraints,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
