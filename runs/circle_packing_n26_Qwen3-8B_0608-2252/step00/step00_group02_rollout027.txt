import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # This is a rough starting point to avoid degenerate solutions
    initial_centers = np.zeros((n, 2))
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * 0.4
        y = row * np.sqrt(3) * 0.4

        initial_centers[i, 1] = y

    # Initial guess for radii: all radii are 0.15
    initial_radii = np.full(n, 0.15)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    def inequality_constraints(x):
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:]
        # Non-overlap constraint

            for j in range(i+1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        # Boundary constraint
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-8 or x + r > 1 + 1e-8 or y - r < -1e-8 or y + r > 1 + 1e-8:
                return np.inf
        return 0.0

    # Bounds: radii must be non-negative

    }

    result = opt.minimize(

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
