import numpy as np
import scipy.optimize as optimize
from typing import Tuple

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    n_circles = 26
    radius_initial = 0.05  # initial guess for radius
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    # Create a hexagonal packing pattern
    # For simplicity, arrange in a 6x5 grid (with one row shifted for hexagonal)
    grid_rows = 6
    grid_cols = 5
    spacing = 1.0 / (grid_rows + 1)
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j * spacing
            y = i * spacing
            if i % 2 == 1:
                x += spacing / 2
            centers[i * grid_cols + j] = np.array([x, y])
            radii[i * grid_cols + j] = radius_initial

    # Flatten the centers and radii into a single vector for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Define the objective function
    def objective(x):
        centers_flat = x[:n_circles * 2].reshape((n_circles, 2))
        radii_flat = x[n_circles * 2:]
        sum_radii = np.sum(radii_flat)
        constraint_penalty = 0.0

        # Add penalty for boundary violations
        for i in range(n_circles):
            x_center, y_center = centers_flat[i]
            r = radii_flat[i]
            if x_center - r < 0 or x_center + r > 1 or y_center - r < 0 or y_center + r > 1:
                constraint_penalty += 1000 * (max(0, x_center - r - 1e-12) + max(0, x_center + r - 1 - 1e-12) +

        # Add penalty for overlapping circles

                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    constraint_penalty += 1000 * (radii_flat[i] + radii_flat[j] - dist - 1e-12)

        # Maximize the sum of radii by minimizing -sum_radii + penalty
        return -sum_radii + constraint_penalty

    # Set bounds for the parameters
    bounds = []
    for i in range(n_circles):
        # Bounds for centers [0,1]x[0,1]
        bounds.append((0.0, 1.0))
        bounds.append((0.0, 1.0))
        # Bounds for radii [0, 0.5]
        bounds.append((0.0, 0.5))

    # Run the optimization
    result = optimize.minimize(

        x0=x0,
        method='nelder-mead',
        bounds=bounds,
        tol=1e-6

    x_opt = result.x
