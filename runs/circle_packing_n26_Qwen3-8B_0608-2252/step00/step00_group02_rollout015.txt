import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: 5x5 grid
    n_circles = 26
    num_rows = 5
    num_cols = 5
    dx = 1.0 / num_cols
    dy = 1.0 / num_rows
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    for i in range(n_circles):
        row = i // num_cols
        col = i % num_cols
        centers[i, 0] = col * dx + dx / 2
        centers[i, 1] = row * dy + dy / 2
        radii[i] = 0.05  # Initial guess

    # Define the objective function to maximize
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define the constraints
    def constraint_overlap(x):
        centers_flat = x[:n_circles*2].reshape((n_circles, 2))
        radii_flat = x[n_circles*2:]
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1000000
        return 0

    def constraint_boundary(x):
        centers_flat = x[:n_circles*2].reshape((n_circles, 2))
        radii_flat = x[n_circles*2:]
        for i in range(n_circles):
            x_center, y_center = centers_flat[i]
            r = radii_flat[i]
            if x_center - r < 0 or x_center + r > 1 or y_center - r < 0 or y_center + r > 1:
                return -1000000
        return 0

    def constraint_radius(x):
        radii_flat = x[n_circles*2:]
        for r in radii_flat:
            if r < 0:
                return -1000000
        return 0

    # Pack the variables: centers (2n_circles) and radii (n_circles)
    x0 = np.concatenate([centers.flatten(), radii])
    bounds = Bounds(np.zeros(2 * n_circles + n_circles), np.inf * np.ones(2 * n_circles + n_circles))

    # Set up constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_radius}
    ]

    # Run the optimization
    result = opt.minimize(
        fun=objective,
        x0=x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'maxiter': 1000, 'ftol': 1e-8, 'eps': 1e-8}
    )

    centers_opt = result.x[:2 * n_circles].reshape((n_circles, 2))
    radii_opt = result.x[2 * n_circles:]

    # Validate the packing
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        print(f"Invalid packing: {msg}")
        return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0

    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
