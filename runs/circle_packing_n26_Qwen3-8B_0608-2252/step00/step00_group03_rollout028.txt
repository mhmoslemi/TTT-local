from scipy.optimize import minimize, Bounds
import random

    num_circles = 26
    grid_size = int(np.sqrt(num_circles))
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Generate hexagonal grid points with a small spacing
    spacing = 0.25  # Adjust this to fit more or fewer circles
    i = 0

            y = row * spacing * np.sqrt(3) / 2

    initial_radius = 0.02
    radii[:] = initial_radius

    def objective(x):

        # Reshape x into centers and radii

        radii_flat = x[2*num_circles:]
        return -np.sum(radii_flat)  # Minimize negative sum to maximize sum

                dx = centers_flat[i, 0] - centers_flat[j, 0]
