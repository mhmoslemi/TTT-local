import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Define the number of circles
    n = 26
    # Initial guess for centers (hexagonal grid)
    centers = np.zeros((n, 2))
    r = np.sqrt(3) / 3  # Radius of one circle in a hexagonal grid
    
    # Generate hexagonal grid in unit square
    cols = int(np.sqrt(n))
    rows = (n + cols - 1) // cols
    for i in range(rows):
        for j in range(cols):
            # Offset for staggered hexagonal grid
            x = j + (i % 2) * 0.5
            y = i * np.sqrt(3) / 2
            # Normalize to unit square
            x = x / (cols - 1) if cols > 1 else 0.5
            y = y / (rows - 1) if rows > 1 else 0.5
            centers[i * cols + j] = [x, y]
    
    # Initial guess for radii
    radii = np.full(n, r)
    
    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum
    
    # Constraint 1: circles within unit square
    def constraint_boundary(radii):
        return np.array([r for r in radii])
    
    # Constraint 2: no overlap between circles
    def constraint_overlap(radii):
        centers = np.zeros((n, 2))
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return np.inf
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0
    
    # Optimization
    result = optimize.minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=[(0, 1)] * n,
        constraints=[
            {'type': 'ineq', 'fun': constraint_overlap},
            {'type': 'ineq', 'fun': constraint_boundary}
        ]
    )
    
    # Extract results
    centers_opt = centers
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)
    
    return centers_opt, radii_opt, sum_radii
