import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial guess: place circles in a 5x5 grid with one extra
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # 5x5 grid with a center row offset
    for i in range(5):
        for j in range(5):
            idx = i * 5 + j
            if idx < num_circles:
                x = i / 4.0 + 0.5 / 8.0
                y = j / 4.0 + 0.5 / 8.0
                centers[idx] = [x, y]
                radii[idx] = 0.1  # Initial radius guess

    # Add one extra circle in the center
    if num_circles > 25:
        idx = 25
        centers[idx] = [0.5, 0.5]
        radii[idx] = 0.1

    # Define variables
    n = num_circles
    x = np.concatenate([centers.reshape(n, 2), radii.reshape(n, 1)], axis=1)
    x_initial = x.copy()

    # Objective function: maximize sum of radii (minimize -sum)
    def objective(x):
        return -np.sum(x[:, 2])

    # Bounds: x_i, y_i ∈ [0, 1], r_i ≥ 0
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # x_i
        bounds.append((0.0, 1.0))  # y_i
        bounds.append((0.0, 1.0))  # r_i

    # Constraints: circles must not overlap and stay within the square
    def constraint_overlap(x):
        centers = x[:, :2].reshape(n, 2)
        radii = x[:, 2].reshape(n)
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j] - 1e-8:
                    return dist - (radii[i] + radii[j] - 1e-8)
        return 0.0

    def constraint_boundary(x):
        centers = x[:, :2].reshape(n, 2)
        radii = x[:, 2].reshape(n)
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-8 or x_i + r_i > 1.0 + 1e-8:
                return x_i - r_i - (-1e-8)
            if y_i - r_i < -1e-8 or y_i + r_i > 1.0 + 1e-8:
                return y_i - r_i - (-1e-8)
        return 0.0

    cons = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary}
    ]

    # Run optimization
    result = optimize.minimize(
        objective,
        x_initial.flatten(),
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Extract results
    x_opt = result.x.reshape(n, 3)
    centers_opt = x_opt[:, :2]
    radii_opt = x_opt[:, 2]

    # Compute sum of radii
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
