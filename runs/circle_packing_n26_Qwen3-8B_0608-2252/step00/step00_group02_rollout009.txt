import scipy.optimize as opt

def run_packing():

    # This is a dense but suboptimal initial guess; we'll refine it

    num_circles = 26
    row_count = 5
    col_count = 5
    radius_initial = 0.1  # Initial estimate for radius

    # Define a hexagonal packing grid
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles) + radius_initial

            col = j
            # Use hexagonal spacing
            if i % 2 == 1:
                col += 0.5  # offset for hexagonal packing

            y = row / (row_count - 1) if row_count > 1 else 0.5
            centers[i * col_count + j] = [x, y]
            radii[i * col_count + j] = radius_initial

    # Optimization parameters
    bounds = []
    for _ in range(num_circles):
        bounds.append((0.0, 1.0))  # x-position
        bounds.append((0.0, 1.0))  # y-position
        bounds.append((0.0, 1.0))  # radius

    # Objective function: maximize sum of radii
    def objective(x):
        centers_flat = x[:num_circles * 2].reshape((num_circles, 2))
        radii_flat = x[num_circles * 2:].reshape((num_circles,))
        return -np.sum(radii_flat)  # minimize negative sum to maximize sum

    # Constraints: no circle can go out of bounds
    def bounds_constraint(x):
        centers_flat = x[:num_circles * 2].reshape((num_circles, 2))
        radii_flat = x[num_circles * 2:].reshape((num_circles,))

            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-8 or x_i + r_i > 1 + 1e-8:
                return -1
            if y_i - r_i < -1e-8 or y_i + r_i > 1 + 1e-8:
                return -1
        return 0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
