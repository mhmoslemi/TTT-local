import numpy as np
import scipy.optimize as optimize
import random
from typing import Tuple, List
import matplotlib.pyplot as plt

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Helper function to generate random hexagonal positions
    def hexagonal_positions(n, spacing):
        positions = []
        row = 0
        col = 0
        for i in range(n):
            x = col * spacing
            y = row * spacing * np.sqrt(3)/2
            if col % 2 == 1:
                y += spacing * np.sqrt(3)/2
            positions.append([x, y])
            col += 1
            if col >= 5:
                row += 1
                col = 0
        return np.array(positions)

    # Helper function to generate grid positions
    def grid_positions(n, spacing):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * spacing + spacing / 2
            y = row * spacing + spacing / 2
            positions.append([x, y])
        return np.array(positions)

    # Helper function to generate staggered positions
    def staggered_positions(n, spacing):
        positions = []
        row = 0
        col = 0
        for i in range(n):
            x = col * spacing + spacing / 2
            y = row * spacing * np.sqrt(3)/2
            if col % 2 == 1:
                y += spacing * np.sqrt(3)/2
            positions.append([x, y])
            col += 1
            if col >= 5:
                row += 1
                col = 0
        return np.array(positions)

    # Helper function to generate random grid positions
    def random_grid_positions(n, spacing):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = np.random.uniform(col * spacing, (col + 1) * spacing)
            y = np.random.uniform(row * spacing, (row + 1) * spacing)
            positions.append([x, y])
        return np.array(positions)

    # Helper function to generate corner-based grid positions
    def corner_grid_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.2 + random.uniform(0, 0.05)
            y = row * 0.2 + random.uniform(0, 0.05)
            positions.append([x, y])
        return np.array(positions)

    # Generate a variety of initial guesses
    initial_positions_list = [
        hexagonal_positions(n, 0.25),
        grid_positions(n, 0.2),
        staggered_positions(n, 0.25),
        corner_grid_positions(n),
        random_grid_positions(n, 0.2),
        np.random.rand(n, 2)
    ]

    # Define objective function to maximize the sum of radii
    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        sum_r = np.sum(radii)
        return -sum_r  # Minimize the negative sum to maximize sum

    # Define boundary constraint: circles must remain inside the unit square
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1.0 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1.0 - centers[:, 1] - radii - 1e-6
        ])

    # Define distance constraint: circles must not overlap
    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(int(n * (n - 1) / 2))
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

    def validate_packing(centers, radii):
        n = centers.shape[0]

        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"

        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

    best_sum = 0
    best_centers = None
    best_radii = None

    def evolve_positions(population, n_pop, n_gen, mutation_rate):

            parents = random.sample(best_pop, n_pop)
            # Crossover
            children = []
            for i in range(0, n_pop, 2):

                    parent1 = parents[i]
                    parent2 = parents[i + 1]
                    child1 = np.random.rand(n, 2) * 0.5 + 0.5 * parent1[:2*n].reshape(n, 2)
                    child2 = np.random.rand(n, 2) * 0.5 + 0.5 * parent2[:2*n].reshape(n, 2)

            for child in children:

                        child[i, 1] += np.random.normal(0, 0.05)
                        child[i, 0] = np.clip(child[i, 0], 0, 1)
                        child[i, 1] = np.clip(child[i, 1], 0, 1)
            # Replace old population
            best_pop = children
        return best_pop

    population_size = 10
    population = [np.concatenate([pos.flatten(), np.full(n, 0.1)]) for pos in initial_positions_list[:population_size]]
    for i in range(population_size - len(initial_positions_list)):
        pos = np.random.rand(n, 2)
        population.append(np.concatenate([pos.flatten(), np.full(n, 0.1)]))
    population = evolve_positions(population, population_size, 100, 0.1)

    for pos in population:

        max_x, max_y = pos[:2*n].reshape(n, 2).max(axis=0)
        scale = min(1.0 / max_x, 1.0 / max_y)
        positions = pos[:2*n].reshape(n, 2) * scale
        initial_radii = np.full(n, 0.1 * scale)
        initial_radii = np.clip(initial_radii, 0, 1.0)

        # Apply small variations to positions to avoid symmetrical overlaps
        perturbations = np.random.uniform(-0.03, 0.03, size=positions.shape)
        positions += perturbations

        # Initial guess
        x0 = np.concatenate([positions.flatten(), initial_radii])

        # Bounds: [0, 1] for positions and [0, 1] for radii
        bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

        # Constraints
        cons = [
            {'type': 'ineq', 'fun': boundary_constraint},
            {'type': 'ineq', 'fun': distance_constraint}
        ]

        # Run optimization
        result = optimize.minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=bounds,
            constraints=cons,
            tol=1e-8,
            options={'maxiter': 1000, 'ftol': 1e-8, 'epsilon': 1e-8}
        )

        # Check if result is valid
        centers = result.x[:2*n].reshape(n, 2)
        radii = result.x[2*n:]
        sum_r = np.sum(radii)

        # Validate result
        valid, msg = validate_packing(centers, radii)
        if valid and sum_r > best_sum:
            best_sum = sum_r
            best_centers = centers
            best_radii = radii

    return (best_centers, best_radii, best_sum)
