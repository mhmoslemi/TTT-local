            positions.append([x, y])
        return np.array(positions)

    initial_position_strategies = [hexagonal_positions, nested_positions, random_positions, initial_positions]

        initial_radii = np.full(n, 0.04)

            radii = x[2*n:]
            return -np.sum(radii)
