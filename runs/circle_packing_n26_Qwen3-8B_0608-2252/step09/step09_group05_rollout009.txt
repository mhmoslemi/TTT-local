        for i in range(n):
            row = i // 5
            col = i % 5

            y = row * (1.0 / (5)) * (1.0 + 0.1) + np.random.uniform(-0.02, 0.02)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    def grid_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.2 + np.random.uniform(-0.02, 0.02)
            y = row * 0.2 + np.random.uniform(-0.02, 0.02)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    def nested_positions(n):
        positions = []
        for i in range(n):
            cluster = i // 5
            x = (i % 5) * 0.2 + np.random.uniform(-0.02, 0.02)
            y = cluster * 0.2 + np.random.uniform(-0.02, 0.02)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)
