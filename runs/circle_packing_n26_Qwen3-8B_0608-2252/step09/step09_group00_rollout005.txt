def hexagonal_grid(n, spacing, rows, cols):

        x = col * spacing

        # Add small random perturbation to avoid symmetric configurations
        x += np.random.uniform(-spacing * 0.1, spacing * 0.1)
        y += np.random.uniform(-spacing * 0.1, spacing * 0.1)
        positions.append([x, y])
    return np.array(positions)
