    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]

    # Initial guess for radii (uniform)
    radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Bounds for radii
    bounds = Bounds(0, 1.0)

    # Define constraints
    def constraint_boundary(x):
        # Check if circle is within unit square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        # Check if any circles overlap

            for j in range(i + 1, num_circles):

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    def hexagonal_positions(n, side_length):

            y = row * (side_length * 1.5)
            positions.append([x, y])

    # Step 2: Try to find an optimal packing

    n = 26

    max_x, max_y = initial_positions.max(axis=0)
    scale = min(1.0 / max_x, 1.0 / max_y)
    initial_positions *= scale
    initial_radii *= scale

    def objective(x):
        # x is a flattened array of [pos1_x, pos1_y, r1, pos2_x, pos2_y, r2, ...]

    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([centers[:, 0] - radii - 1e-8, 1 - centers[:, 0] - radii - 1e-8,

        bounds=bounds,
        constraints=cons,
        tol=1e-6

    radii = result.x[2*n:]
    sum_radii = np.sum(radii)
