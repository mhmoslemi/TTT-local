```python
import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: 5 rows with 5 or 6 circles
    num_circles = 26
    rows = 5
    cols = 6
    centers = np.zeros((num_circles, 2))
    
    # Initial placement in a grid
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]

    # Initial guess for radii (uniform)
    radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Bounds for radii
    bounds = Bounds(0, 1.0)

    # Define constraints
    def constraint_boundary(x):
        # Check if circle is within unit square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        # Check if any circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i = centers[i]
                r_i = x[i]
                x_j, y_j = centers[j]
                r_j = x[j]
                dist_sq = (x_i - x_j) ** 2 + (y_i - y_j) ** 2
                if dist_sq < (r_i + r_j) ** 2 - 1e-12:
                    return -1
        return 0

    # Nonlinear constraints
    cons = [
        NonlinearConstraint(constraint_boundary, -np.inf, 0),
        NonlinearConstraint(constraint_overlap, -np.inf, 0)
    ]

    # Set up optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    if not result.success:
        print("Optimization failed:", result.message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    optimized_radii = result.x
    sum_radii = np.sum(optimized_radii)

    # Validate the solution
    valid, message = validate_packing(centers, optimized_radii)
    if not valid:
        print("Validation failed:", message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, optimized_radii, sum_radii
```

```python
import numpy as np
import scipy.optimize as optimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing
    num_circles = 26
    rows = int(np.sqrt(num_circles))
    cols = (num_circles + rows - 1) // rows

    # Hexagonal packing parameters
    hex_radius = 1.0 / (2 * np.sqrt(3))  # Radius to fill the unit square
    spacing = 2 * hex_radius  # Distance between centers in hexagonal pattern

    # Initial positions
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col * spacing
        y = row * spacing * np.sqrt(3) / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = hex_radius

    # Adjust positions to fit in unit square
    scale = 1.0 / (np.max(centers) + np.max(radii))
    centers *= scale
    radii *= scale

    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Define constraints: circles within square and non-overlapping
    def constraint_within_square(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1
        return 0

    def constraint_no_overlap(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1
        return 0

    # Optimization variables:
    # - First 2*num_circles entries: coordinates of centers
    # - Last num_circles entries: radii of circles
    x0 = np.concatenate((centers.flatten(), radii))

    # Optimization bounds: circles are within [0,1]^2, radii >= 0
    bounds = [(0, 1) for _ in range(2 * num_circles)] + [(0, 1) for _ in range(num_circles)]

    # Optimization constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_within_square},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8,
        options={'maxiter': 500}
    )

    # Extract results
    centers_opt = result.x[:2*num_circles].reshape((num_circles, 2))
    radii_opt = result.x[2*num_circles:]
    sum_radii_opt = np.sum(radii_opt)

    # Validate the packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers_opt, radii_opt, sum_radii_opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
