import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial positions using a hexagonal grid pattern
    n = 26
    rows = int(np.ceil(n / 6))  # Adjusting for hexagonal packing
    cols = 6
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col + (row % 2) * 0.5
        y = row * np.sqrt(3) / 2
        centers[i, 0] = x / 6  # Scale to fit in [0, 1]
        centers[i, 1] = y / (rows * np.sqrt(3) / 2)

    # Initial guess for radii (equal radii for simplicity)
    initial_radius = 0.1
    radii = np.full(n, initial_radius)

    # Define optimization bounds: centers within [0, 1]^2, radii non-negative
    bounds_centers = [(0, 1) for _ in range(2 * n)]
    bounds_radii = [(0, 1) for _ in range(n)]

    # Objective function to maximize the sum of radii
    def objective(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:].reshape(n)
        return -np.sum(radii_flat)  # Minimize negative of sum

    # Constraints: circles within square and no overlap
    def constraints(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:].reshape(n)
        constraints = []

        # Circle within square
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12
                or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return np.inf  # Penalize violations
    
        # Circle-to-circle distances
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf  # Penalize violations

        return 0  # No constraints violated

    # Optimization setup
    x0 = np.concatenate([centers.flatten(), radii.flatten()])
    result = optimize.minimize(
        objective,
        x0,
        bounds=np.concatenate([bounds_centers, bounds_radii]),
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
        options={"maxiter": 500, "ftol": 1e-8, "eps": 1e-8}
    )

    # Extract final centers and radii
    centers_opt = result.x[:2 * n].reshape(n, 2)
    radii_opt = result.x[2 * n:].reshape(n)
    sum_radii = np.sum(radii_opt)

    # Validate packing for safety
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        print("Invalid packing:", msg)
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return centers_opt, radii_opt, sum_radii

import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial placement of 25 circles in a grid
    num_circles = 26
    grid_size = 5
    grid_spacing = 1.0 / (grid_size - 1)
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # First 25 circles in grid
    for i in range(grid_size):
        for j in range(grid_size):
            if i * grid_size + j < 25:
                centers[i * grid_size + j, 0] = j * grid_spacing
                centers[i * grid_size + j, 1] = i * grid_spacing
                radii[i * grid_size + j] = 0.1  # Initial radius

    # Place the 26th circle at a central position
    centers[25, 0] = 0.5
    centers[25, 1] = 0.5
    radii[25] = 0.1

    def objective(radii_vector):
        return -np.sum(radii_vector)

    def constraints_func(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:]
        centers = centers_flat
        radii = radii_flat

        # Boundary constraints
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf  # Violates boundary

        # Non-overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf  # Overlap

        # Return negative sum of radii (since we're minimizing)
        return -np.sum(radii)

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Bounds: center coordinates between 0 and 1, radii non-negative
    bounds = []
    for _ in range(num_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    # Optimization settings
    result = opt.minimize(
        fun=constraints_func,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        options={
            'ftol': 1e-9,
            'maxiter': 1000,
            'disp': False
        }
    )

    # Extract the solution
    optimized_x = result.x
    centers_opt = optimized_x[:2 * num_circles].reshape(num_circles, 2)
    radii_opt = optimized_x[2 * num_circles:]

    # Check validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Optimization failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
