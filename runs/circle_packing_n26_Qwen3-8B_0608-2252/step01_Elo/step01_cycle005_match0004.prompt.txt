import numpy as np
import scipy.optimize as optimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing
    num_circles = 26
    rows = int(np.sqrt(num_circles))
    cols = (num_circles + rows - 1) // rows

    # Hexagonal packing parameters
    hex_radius = 1.0 / (2 * np.sqrt(3))  # Radius to fill the unit square
    spacing = 2 * hex_radius  # Distance between centers in hexagonal pattern

    # Initial positions
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col * spacing
        y = row * spacing * np.sqrt(3) / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = hex_radius

    # Adjust positions to fit in unit square
    scale = 1.0 / (np.max(centers) + np.max(radii))
    centers *= scale
    radii *= scale

    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Define constraints: circles within square and non-overlapping
    def constraint_within_square(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1
        return 0

    def constraint_no_overlap(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1
        return 0

    # Optimization variables:
    # - First 2*num_circles entries: coordinates of centers
    # - Last num_circles entries: radii of circles
    x0 = np.concatenate((centers.flatten(), radii))

    # Optimization bounds: circles are within [0,1]^2, radii >= 0
    bounds = [(0, 1) for _ in range(2 * num_circles)] + [(0, 1) for _ in range(num_circles)]

    # Optimization constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_within_square},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8,
        options={'maxiter': 500}
    )

    # Extract results
    centers_opt = result.x[:2*num_circles].reshape((num_circles, 2))
    radii_opt = result.x[2*num_circles:]
    sum_radii_opt = np.sum(radii_opt)

    # Validate the packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers_opt, radii_opt, sum_radii_opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial guess for circle positions and radii
    num_circles = 26
    centers_initial = np.zeros((num_circles, 2))
    radii_initial = np.zeros(num_circles)
    
    # Define hexagonal packing parameters
    # For 26 circles, we can use a 5x5 grid (25 circles) with one extra near the center
    # We'll start with 25 in a hexagonal grid and one at the center
    for i in range(5):
        for j in range(5):
            x = j / 4 + 0.1
            y = i / 4 + 0.1
            centers_initial[i*5 + j] = [x, y]
    # Add one more circle at the center
    centers_initial[25] = [0.5, 0.5]
    radii_initial[:] = 0.08  # Initial radius guess

    # Define the optimization function: maximize the total radius sum
    def objective(radii):
        return -np.sum(radii)  # Minimize the negative sum to maximize the sum

    # Define constraints
    def circle_bounds(radii, centers):
        # Ensure all circles are within the unit square
        for i in range(num_circles):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        return True

    def non_overlap(radii, centers):
        # Ensure circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    # Set up the optimization problem
    bounds = [(0, 0.2) for _ in range(num_circles)]

    def constraint_func(radii):
        centers = centers_initial
        if not circle_bounds(radii, centers):
            return np.array([1.0])  # Constraint violation
        if not non_overlap(radii, centers):
            return np.array([1.0])  # Constraint violation
        return np.array([0.0])

    # Initial guess for radii
    radii_guess = radii_initial.copy()

    # Solve the optimization problem
    result = optimize.minimize(
        objective,
        radii_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraint_func}],
        tol=1e-6
    )

    # Extract final results
    radii = result.x
    centers = centers_initial.copy()
    sum_radii = np.sum(radii)

    # Validate the packing
    valid, message = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers, radii, sum_radii
