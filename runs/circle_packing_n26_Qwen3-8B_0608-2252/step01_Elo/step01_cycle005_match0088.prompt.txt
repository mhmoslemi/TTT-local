import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: 5 rows with 5 or 6 circles
    num_circles = 26
    rows = 5
    cols = 6
    centers = np.zeros((num_circles, 2))
    
    # Initial placement in a grid
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]

    # Initial guess for radii (uniform)
    radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Bounds for radii
    bounds = Bounds(0, 1.0)

    # Define constraints
    def constraint_boundary(x):
        # Check if circle is within unit square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        # Check if any circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i = centers[i]
                r_i = x[i]
                x_j, y_j = centers[j]
                r_j = x[j]
                dist_sq = (x_i - x_j) ** 2 + (y_i - y_j) ** 2
                if dist_sq < (r_i + r_j) ** 2 - 1e-12:
                    return -1
        return 0

    # Nonlinear constraints
    cons = [
        NonlinearConstraint(constraint_boundary, -np.inf, 0),
        NonlinearConstraint(constraint_overlap, -np.inf, 0)
    ]

    # Set up optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    if not result.success:
        print("Optimization failed:", result.message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    optimized_radii = result.x
    sum_radii = np.sum(optimized_radii)

    if not valid:

import scipy.optimize as opt

def run_packing():

    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))

    spacing = 1.0 / (cols + 1)
    offset = spacing / 2
    centers = np.zeros((num_circles, 2))
    radius = np.ones(num_circles) * 0.05  # Initial guess for radii
    
    idx = 0
    for row in range(rows):
        for col in range(cols):
            x = col * spacing + offset
            y = row * spacing * np.sqrt(3) / 2 + offset
            if idx < num_circles:
                centers[idx] = [x, y]
                idx += 1

    # Optimization objective: minimize sum of squared radii (equivalent to maximizing sum of radii under constraint sum(r) = constant)
    def optimize_objective(x):
        global centers, radius
        new_centers = x[:2*num_circles].reshape(num_circles, 2)
        new_radii = x[2*num_circles:]
        sum_radii = np.sum(new_radii)
        penalty = 0
        # Constraint: All circles must be within the square
        for i in range(num_circles):
            if new_centers[i, 0] - new_radii[i] < 0 or new_centers[i, 0] + new_radii[i] > 1:
                penalty += 1e6
            if new_centers[i, 1] - new_radii[i] < 0 or new_centers[i, 1] + new_radii[i] > 1:
                penalty += 1e6
        # Constraint: No overlaps
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dist = np.linalg.norm(new_centers[i] - new_centers[j])
                if dist < new_radii[i] + new_radii[j] - 1e-8:
                    penalty += 1e6
        # Objective: Minimize sum of squared radii (equivalent to maximizing sum of radii due to constraint)
        return np.sum(new_radii ** 2) + penalty

    # Bounds on positions and radii
    bounds = []
    for i in range(num_circles):
        # Bounds for (x, y) coordinates
        x_bounds = (0.0, 1.0)
        y_bounds = (0.0, 1.0)
        bounds.extend([x_bounds, y_bounds])
        # Bounds for radius
        radius_bounds = (0.0, 0.5)
        bounds.extend([radius_bounds])

    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radius])

    # Run optimization
    result = opt.minimize(optimize_objective, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-8)

    # Extract result

        optimized_centers = centers
        optimized_radii = radius
        sum_radii = np.sum(optimized_radii)

    return optimized_centers, optimized_radii, sum_radii
