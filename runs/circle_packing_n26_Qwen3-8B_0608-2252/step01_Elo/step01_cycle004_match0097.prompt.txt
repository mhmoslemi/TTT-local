import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: 5 rows with 5 or 6 circles
    num_circles = 26
    rows = 5
    cols = 6
    centers = np.zeros((num_circles, 2))
    
    # Initial placement in a grid
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]

    # Initial guess for radii (uniform)
    radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Bounds for radii
    bounds = Bounds(0, 1.0)

    # Define constraints
    def constraint_boundary(x):
        # Check if circle is within unit square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        # Check if any circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i = centers[i]
                r_i = x[i]
                x_j, y_j = centers[j]
                r_j = x[j]
                dist_sq = (x_i - x_j) ** 2 + (y_i - y_j) ** 2
                if dist_sq < (r_i + r_j) ** 2 - 1e-12:
                    return -1
        return 0

    # Nonlinear constraints
    cons = [
        NonlinearConstraint(constraint_boundary, -np.inf, 0),
        NonlinearConstraint(constraint_overlap, -np.inf, 0)
    ]

    # Set up optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    if not result.success:
        print("Optimization failed:", result.message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    optimized_radii = result.x
    sum_radii = np.sum(optimized_radii)

    # Validate the solution
    valid, message = validate_packing(centers, optimized_radii)
    if not valid:
        print("Validation failed:", message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, optimized_radii, sum_radii

import numpy as np
import scipy.optimize as optimize
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: hexagonal arrangement of circles
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # First, place in a hexagonal grid
    rows = int(np.sqrt(n))
    cols = n // rows + (1 if n % rows != 0 else 0)
    spacing = 1.0 / (cols + 1)

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i, 0] = x
        centers[i, 1] = y
        # Assume a minimum radius to ensure spacing
        radii[i] = spacing / 2 - 1e-4

    # Optimize the centers and radii
    def objective(x):
        centers_opt = x[:2*n].reshape(n, 2)
        radii_opt = x[2*n:]

        # Ensure constraints: radii >= 0
        if np.any(radii_opt < 0):
            return float('inf')

        # Ensure circles are inside the unit square
        for i in range(n):
            x, y = centers_opt[i]
            r = radii_opt[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return float('inf')

        # Ensure circles do not overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_opt[i] + radii_opt[j] - 1e-12:
                    return float('inf')

        # Objective is to maximize sum of radii (negative for minimization)
        return -np.sum(radii_opt)

    # Initial guess for the optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization bounds: centers must be in [0, 1] and radii >= 0
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        options={'maxiter': 1000, 'ftol': 1e-6}
    )

    # Extract optimized centers and radii
    centers_opt = result.x[:2*n].reshape(n, 2)
    radii_opt = result.x[2*n:]

    # Final check and validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Warning: Packing validation failed: {message}")
        return centers, radii, np.sum(radii)

    return centers_opt, radii_opt, np.sum(radii_opt)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
