from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_circles = 26
    rows = 5
    cols = 6
    centers = np.zeros((num_circles, 2))
    
    # Initial placement in a grid
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]

    # Initial guess for radii (uniform)
    radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Bounds for radii
    bounds = Bounds(0, 1.0)

    # Define constraints
    def constraint_boundary(x):
        # Check if circle is within unit square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        # Check if any circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i = centers[i]
                r_i = x[i]
                x_j, y_j = centers[j]
                r_j = x[j]
                dist_sq = (x_i - x_j) ** 2 + (y_i - y_j) ** 2
                if dist_sq < (r_i + r_j) ** 2 - 1e-12:
                    return -1
        return 0

    # Nonlinear constraints
    cons = [
        NonlinearConstraint(constraint_boundary, -np.inf, 0),
        NonlinearConstraint(constraint_overlap, -np.inf, 0)
    ]

    # Set up optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}

    if not result.success:
        print("Optimization failed:", result.message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    sum_radii = np.sum(optimized_radii)

import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial guess for circle positions and radii
    num_circles = 26
    centers_initial = np.zeros((num_circles, 2))
    radii_initial = np.zeros(num_circles)
    
    # Define hexagonal packing parameters
    # For 26 circles, we can use a 5x5 grid (25 circles) with one extra near the center
    # We'll start with 25 in a hexagonal grid and one at the center
    for i in range(5):
        for j in range(5):
            x = j / 4 + 0.1
            y = i / 4 + 0.1
            centers_initial[i*5 + j] = [x, y]
    # Add one more circle at the center
    centers_initial[25] = [0.5, 0.5]
    radii_initial[:] = 0.08  # Initial radius guess

    # Define the optimization function: maximize the total radius sum
    def objective(radii):
        return -np.sum(radii)  # Minimize the negative sum to maximize the sum

    # Define constraints
    def circle_bounds(radii, centers):
        # Ensure all circles are within the unit square
        for i in range(num_circles):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        return True

    def non_overlap(radii, centers):
        # Ensure circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    # Set up the optimization problem
    bounds = [(0, 0.2) for _ in range(num_circles)]

    def constraint_func(radii):
        centers = centers_initial
        if not circle_bounds(radii, centers):
            return np.array([1.0])  # Constraint violation
        if not non_overlap(radii, centers):
            return np.array([1.0])  # Constraint violation
        return np.array([0.0])

    # Initial guess for radii
    radii_guess = radii_initial.copy()

    # Solve the optimization problem
    result = optimize.minimize(
        objective,
        radii_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraint_func}],
        tol=1e-6
    )

    # Extract final results
    radii = result.x
    centers = centers_initial.copy()
    sum_radii = np.sum(radii)

    # Validate the packing
    valid, message = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers, radii, sum_radii
