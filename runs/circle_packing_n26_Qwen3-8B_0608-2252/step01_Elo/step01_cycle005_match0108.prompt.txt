import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial hexagonal grid arrangement
    def hexagonal_grid(n_circles):
        r = 0.05  # Initial radius (to be optimized)
        centers = []
        for i in range(n_circles):
            row = i // 3
            col = i % 3
            x = col * (2 * r) + r
            y = row * (np.sqrt(3) * r) + r
            centers.append([x, y])
        centers = np.array(centers)
        return centers, np.full(n_circles, r)

    # Objective function to maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    # Constraints for optimization
    def constraints(radii, centers):
        n = len(radii)
        constraints = []
        # Constraint 1: All circles must fit in the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - y - r})
        # Constraint 2: No overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                def dist_constraint(radii, i=i, j=j):
                    r_i = radii[i]
                    r_j = radii[j]
                    dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                    return dist - r_i - r_j
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        return constraints

    # Initial guess for radii
    centers, initial_radii = hexagonal_grid(26)
    bounds = [(0, 1) for _ in range(26)]  # Radii must be between 0 and 1

    # Set up optimization
    result = minimize(
        fun=objective,
        x0=initial_radii,
        args=(centers,),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints(initial_radii, centers)
    )

    # Extract the solution
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Check if the solution is valid
    valid, message = validate_packing(centers, radii)
    if not valid:
        print(f"Optimization failed: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: 5 rows with 5 or 6 circles
    num_circles = 26
    rows = 5
    cols = 6
    centers = np.zeros((num_circles, 2))
    
    # Initial placement in a grid
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]

    # Initial guess for radii (uniform)
    radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Bounds for radii
    bounds = Bounds(0, 1.0)

    # Define constraints
    def constraint_boundary(x):
        # Check if circle is within unit square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        # Check if any circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i = centers[i]
                r_i = x[i]
                x_j, y_j = centers[j]
                r_j = x[j]
                dist_sq = (x_i - x_j) ** 2 + (y_i - y_j) ** 2
                if dist_sq < (r_i + r_j) ** 2 - 1e-12:
                    return -1
        return 0

    # Nonlinear constraints
    cons = [
        NonlinearConstraint(constraint_boundary, -np.inf, 0),
        NonlinearConstraint(constraint_overlap, -np.inf, 0)
    ]

    # Set up optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    if not result.success:
        print("Optimization failed:", result.message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    optimized_radii = result.x
    sum_radii = np.sum(optimized_radii)

    # Validate the solution
    valid, message = validate_packing(centers, optimized_radii)
    if not valid:
        print("Validation failed:", message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, optimized_radii, sum_radii
