import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial hexagonal grid arrangement
    def hexagonal_grid(n_circles):
        r = 0.05  # Initial radius (to be optimized)
        centers = []
        for i in range(n_circles):
            row = i // 3
            col = i % 3
            x = col * (2 * r) + r
            y = row * (np.sqrt(3) * r) + r
            centers.append([x, y])
        centers = np.array(centers)
        return centers, np.full(n_circles, r)

    # Objective function to maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    # Constraints for optimization
    def constraints(radii, centers):
        n = len(radii)
        constraints = []
        # Constraint 1: All circles must fit in the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - y - r})
        # Constraint 2: No overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                def dist_constraint(radii, i=i, j=j):
                    r_i = radii[i]
                    r_j = radii[j]
                    dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                    return dist - r_i - r_j
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        return constraints

    # Initial guess for radii
    centers, initial_radii = hexagonal_grid(26)
    bounds = [(0, 1) for _ in range(26)]  # Radii must be between 0 and 1

    # Set up optimization
    result = minimize(
        fun=objective,
        x0=initial_radii,
        args=(centers,),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints(initial_radii, centers)
    )

    # Extract the solution
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Check if the solution is valid
    valid, message = validate_packing(centers, radii)
    if not valid:
        print(f"Optimization failed: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
from itertools import combinations

def run_packing():
    # Initial hexagonal grid packing
    num_circles = 26
    num_rows = 5
    circles_per_row = 5
    spacing = 1.0 / (num_rows + 1)  # vertical spacing
    horizontal_shift = 0.5 / (circles_per_row + 1)  # horizontal shift for hexagonal packing

    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Initial guess for radii
    for i in range(num_rows):
        for j in range(circles_per_row):
            row_idx = i * circles_per_row + j
            x = j * (1.0 / circles_per_row) + horizontal_shift
            y = i * spacing + spacing
            centers[row_idx] = [x, y]
            radii[row_idx] = 0.1  # Initial small radius

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraint functions
    def constraint1(radii):
        return np.sum(radii) - 2.636  # Ensure the sum is at least 2.636

    def constraint2(radii):
        return np.max(radii) - 0.1  # Ensure no radius exceeds 0.1 initially

    # Bounds for radii: non-negative
    bounds = [(0, 0.1) for _ in range(num_circles)]

    # Initial radius guess
    initial_radii = np.full(num_circles, 0.1)

    # Optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(
        objective,
        initial_radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    optimal_radii = result.x
    optimal_centers = centers.copy()

    # Local refinement: move circles slightly to fill gaps
    for _ in range(5):  # Run local refinement multiple times
        for i in range(num_circles):
            # Try moving this circle slightly
            dx = np.random.uniform(-0.01, 0.01)
            dy = np.random.uniform(-0.01, 0.01)
            new_center = optimal_centers[i] + [dx, dy]

            # Check if new center is within bounds
            if (new_center[0] < -1e-12 or new_center[0] > 1 + 1e-12 or
                new_center[1] < -1e-12 or new_center[1] > 1 + 1e-12):
                continue

            # Check for overlap with other circles
            overlap = False
            for j in range(num_circles):
                if i == j:
                    continue
                dist = np.hypot(new_center[0] - optimal_centers[j][0],
                                new_center[1] - optimal_centers[j][1])
                if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                    overlap = True
                    break

            if not overlap:
                optimal_centers[i] = new_center

    # Final check for boundary and overlap
    for i in range(num_circles):
        x, y = optimal_centers[i]
        r = optimal_radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return np.zeros((26, 2)), np.zeros(26), 0.0

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.hypot(optimal_centers[i][0] - optimal_centers[j][0],
                            optimal_centers[i][1] - optimal_centers[j][1])
            if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                return np.zeros((26, 2)), np.zeros(26), 0.0

    return optimal_centers, optimal_radii, np.sum(optimal_radii)
