from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal grid packing
    n = 26
    rows = 5
    cols = 6  # 6, 5, 6, 5, 6
    grid = []
    radii = []

    # Define the hexagonal grid with spacing
    spacing = 1 / (rows + 1)  # Spacing between rows and columns

    for row in range(rows):
        for col in range(cols - row % 2):
            x = col * (spacing * np.sqrt(3))
            y = row * (spacing * 2)
            grid.append(np.array([x, y]))
            radii.append(spacing)

    grid = np.array(grid)
    radii = np.array(radii)

    # Optimization function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_overlap(x, r):
        dist = cdist(x, x)
        for i in range(x.shape[0]):
            for j in range(i + 1, x.shape[0]):
                if dist[i, j] < r[i] + r[j] - 1e-12:
                    return False
        return True

    def constraint_boundaries(x, r):
        for i in range(x.shape[0]):
            x_i, y_i = x[i]
            r_i = r[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return False
        return True

    def constraint_sum(r):
        return np.sum(r)

    # Optimize
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraint_sum(r) - 1.0},
            {'type': 'ineq', 'fun': lambda r: constraint_overlap(grid, r)},
            {'type': 'ineq', 'fun': lambda r: constraint_boundaries(grid, r)},
        ],
        bounds=[(0, 1)] * n
    )

    # Final configuration
    centers = grid
    radii = result.x
    sum_radii = np.sum(radii)

    is_valid, message = validate_packing(centers, radii)
    if not is_valid:

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
