import numpy as np
import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        x, y = centers[i]
        r = radii[i]

    return True, "ok"

def initial_grid_packing(n):
    """
    Generate a square grid arrangement of n circles.
    """
    row = int(np.ceil(np.sqrt(n)))
    col = int(np.ceil(n / row))
    
    centers = np.zeros((n, 2))
    for i in range(n):
        x = i % col / col
        y = i // col / row
        centers[i] = [x, y]
    return centers

def objective_function(params, n):
    """
    Objective function to maximize the sum of radii.
    """
    centers = params[:2*n].reshape(n, 2)
    radii = params[2*n:]
    
    # Compute the sum of radii
    sum_radii = np.sum(radii)
    return -sum_radii  # Minimize negative sum to maximize sum

def constraints_function(params, n):
    """
    Constraints for the optimization:
    1. All circles are within the unit square.
    2. No circles overlap.
    """
    centers = params[:2*n].reshape(n, 2)
    radii = params[2*n:]
    
    # Constraint 1: Within unit square
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if not (x - r >= -1e-12 and x + r <= 1 + 1e-12 and y - r >= -1e-12 and y + r <= 1 + 1e-12):

    return np.array([])

def run_packing():
    n = 26
    initial_centers = initial_grid_packing(n)

    initial_params = np.concatenate((initial_centers.flatten(), initial_radii))

    # Bounds for centers (x and y between 0 and 1)
    # Bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(2 * n)] + [(0, 1) for _ in range(n)]

    # Define the optimization problem
    result = opt.minimize(

        x0=initial_params,
        method='SLSQP',
