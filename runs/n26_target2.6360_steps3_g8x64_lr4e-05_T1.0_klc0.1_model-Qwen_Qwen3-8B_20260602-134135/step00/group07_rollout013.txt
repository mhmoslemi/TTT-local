import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based arrangement for 26 circles (5 rows x 6 columns)
    num_circles = 26
    num_rows = 5
    num_cols = 6
    
    # Initial positions in a grid
    centers_initial = np.zeros((num_circles, 2))
    for i in range(num_rows):
        for j in range(num_cols):
            idx = i * num_cols + j
            if idx >= num_circles:
                break
            x = j / (num_cols - 1)
            y = i / (num_rows - 1)
            centers_initial[idx] = [x, y]
    
    # Initial radii: assuming unit spacing between centers
    # The spacing between centers is 1/(num_cols - 1) horizontally and 1/(num_rows - 1) vertically
    spacing_h = 1 / (num_cols - 1)
    spacing_v = 1 / (num_rows - 1)
    spacing = min(spacing_h, spacing_v)
    radii_initial = np.full(num_circles, spacing / 2)
    
    # Optimization bounds and constraints
    bounds = [(0.0, 1.0) for _ in range(2 * num_circles)]
    constraints = []
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            # Add non-overlap constraint between circles i and j
            constraints.append({
                'type': 'ineq',

            })
    
    def objective(x):
        # Objective function: maximize sum of radii
        return -np.sum(x[num_circles:])
    
    # Initial guess combining centers and radii
    x0 = np.concatenate([centers_initial.flatten(), radii_initial])
    
    # Optimization with bounds and constraints
    result = optimize.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-8
    )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
