import numpy as np
import scipy.optimize as opt
import scipy.spatial.distance as dist

def run_packing():

    n_circles = 26
    centers = np.zeros((n_circles, 2))
    
    # Grid layout with some adjustments for better spacing
    for i in range(n_circles):
        row = i // 6
        col = i % 6
        x = col / 5.0

        centers[i] = [x, y]

    # Objective function: maximize the sum of radii
    def objective(x):

        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -np.inf  # Penalize if any circle is out of bounds

        for i in range(26):

                dx = centers_flat[i, 0] - centers_flat[j, 0]

                    return -np.inf  # Penalize if any circles overlap

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            if dist < radii[i] + radii[j] - 1e-12:

    return True, "ok"
