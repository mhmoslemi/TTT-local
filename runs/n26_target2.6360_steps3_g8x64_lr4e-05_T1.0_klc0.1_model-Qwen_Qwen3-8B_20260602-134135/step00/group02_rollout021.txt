def calculate_distances(centers):
    n = centers.shape[0]
    dists = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            dists[i, j] = np.linalg.norm(centers[i] - centers[j])
    return dists

def compute_overlap_penalty(centers, radii, dists, eps=1e-12):
    n = centers.shape[0]
    penalty = 0.0
    for i in range(n):
        for j in range(i + 1, n):
            if dists[i, j] < radii[i] + radii[j] - eps:
                penalty += 1e4 * (radii[i] + radii[j] - dists[i, j])
    return penalty

def compute_boundary_penalty(centers, radii, eps=1e-12):

    penalty = 0.0
    for i in range(n):
        x, y = centers[i]
        r = radii[i]

            penalty += 1e4 * max(0, x - r + eps, x + r - 1 - eps)
        if y - r < -eps or y + r > 1 + eps:
            penalty += 1e4 * max(0, y - r + eps, y + r - 1 - eps)
    return penalty

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
