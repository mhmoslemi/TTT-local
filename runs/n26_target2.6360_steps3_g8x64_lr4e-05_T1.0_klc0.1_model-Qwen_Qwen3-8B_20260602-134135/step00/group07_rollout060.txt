import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define constants
    n = 26
    max_radius = 0.5  # Initial guess for max radius
    
    # Initialize centers in a hexagonal lattice within the unit square
    # Start by placing circles in two rows for demonstration purposes
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Hexagonal grid configuration with 5 rows (3 rows of 6 circles and 2 rows of 5 circles)
    # To simplify, we'll place circles in 2 rows for now
    num_circles = n
    row_count = 2
    col_count = num_circles // row_count
    col_count = max(1, col_count)
    
    for i in range(row_count):
        for j in range(col_count):
            x = j / (col_count - 1)
            y = i / (row_count - 1)
            centers[i * col_count + j] = [x, y]
            radii[i * col_count + j] = max_radius
    
    # Define objective function to maximize sum of radii
    def objective(x):
        # x contains flattened centers and radii
        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:].reshape(n)
        
        # Calculate sum of radii
        sum_radii = np.sum(radii_flat)
        
        # Add small perturbations to prevent degeneracy
        centers_flat += np.random.normal(0, 1e-6, centers_flat.shape)
        radii_flat += np.random.normal(0, 1e-6, radii_flat.shape)
        
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraint_centers(x):
        centers_flat = x[:n * 2].reshape(n, 2)
        return np.array([np.all(centers_flat[:, 0] >= 0) and np.all(centers_flat[:, 0] <= 1),
                         np.all(centers_flat[:, 1] >= 0) and np.all(centers_flat[:, 1] <= 1)])

    def constraint_radii(x):
        radii_flat = x[n * 2:].reshape(n)
        return np.array([np.all(radii_flat >= 0)])

        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:].reshape(n)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
