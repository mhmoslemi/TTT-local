import numpy as np
import math
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based placement
    n = 26
    grid = np.zeros((n, 2))
    i, j = 0, 0
    for idx in range(n):
        grid[idx, 0] = i / 5.0
        grid[idx, 1] = j / 5.0
        if idx % 5 == 4:
            i += 1
            j += 1
        else:
            j += 1
    initial_centers = grid

    # Optimization function to maximize sum of radii
    def objective(x):
        r = x[:n]
        centers = x[n:]
        centers = centers.reshape((n, 2))
        obj = np.sum(r)
        return -obj  # Minimize negative of the objective to maximize

    # Constraint: Ensure all circles are within the unit square
    def constraint1(x, eps=1e-12):
        r = x[:n]
        centers = x[n:].reshape((n, 2))
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = r[i]
            if (x_i - r_i < -eps or x_i + r_i > 1 + eps
                or y_i - r_i < -eps or y_i + r_i > 1 + eps):
                return -np.inf
        return 0.0

    # Constraint: Ensure no two circles overlap
    def constraint2(x, eps=1e-12):
        r = x[:n]
        centers = x[n:].reshape((n, 2))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j] - eps:
                    return -np.inf
        return 0.0

    # Initial guess: Radii are set initially based on spacing
    initial_r = np.ones(n) * 0.05
    x0 = np.concatenate((initial_r, initial_centers.flatten()))

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Optimization with bounds
    bounds = []
    for _ in range(n):
        bounds.append((0.0, 1.0))  # Radii bounds
    for _ in range(2 * n):
        bounds.append((0.0, 1.0))  # Center coordinates bounds

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Extract results
    radii = result.x[:n]
    centers = result.x[n:].reshape((n, 2))
    sum_radii = np.sum(radii)

    # Check constraints for validation
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Packing is invalid: {msg}")

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
