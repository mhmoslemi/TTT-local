import scipy.optimize as optimize

def generate_hexagonal_packing(n=26, width=1.0, height=1.0):

    # Return initial centers and dummy radii
    centers = []
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    spacing = np.sqrt(3) / 2
    radius = 0.05

            break

    centers = np.array(centers) / max(max(centers[:, 0]), max(centers[:, 1]))
    return np.array(centers), np.full(n, radius)
