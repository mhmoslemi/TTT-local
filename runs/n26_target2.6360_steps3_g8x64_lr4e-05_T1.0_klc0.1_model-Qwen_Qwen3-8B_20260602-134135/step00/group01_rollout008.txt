import numpy as np
import scipy.optimize as optimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # --- Step 1: Initial Hexagonal Grid Layout ---
    # Define parameters
    num_circles = 26
    max_radius = 0.3  # Start with an initial guess for radius

    # Create initial positions using hexagonal grid
    centers = np.zeros((num_circles, 2))
    radius = np.full(num_circles, max_radius)
    pos = 0

    # Define rows and columns for hexagonal packing
    row_spacing = 2 * max_radius
    col_spacing = np.sqrt(3) * max_radius

    # Create initial layout
    for row in range(4):
        for col in range(6):
            x = col * col_spacing
            y = row * row_spacing + (col % 2) * (row_spacing / 2)
            if x < 1 - 1e-6 and y < 1 - 1e-6:
                centers[pos, 0] = x
                centers[pos, 1] = y
                pos += 1

    # Trim layout to exactly 26 circles
    centers = centers[:26]
    radius = radius[:26]

    # --- Step 2: Define Objective and Constraints ---
    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize

    # Constraints:
    # 1. Each circle must be inside the square
    # 2. Each circle must have non-negative radius
    # 3. No two circles overlap

    # Constraint for circle within square
    def inside_box(x, y, r):
        return [
            x - r >= -1e-6,
            x + r <= 1 + 1e-6,
            y - r >= -1e-6,
            y + r <= 1 + 1e-6
        ]

    # Constraint for non-overlapping circles
    def no_overlap(i, j, x_i, y_i, r_i, x_j, y_j, r_j):
        dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
        return dist >= r_i + r_j - 1e-6

    # Define constraints for optimization
    constraints = []

    # Constraint 1: All circles must be within the square
    for i in range(num_circles):
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: inside_box(centers[i, 0], centers[i, 1], r[i])
        })

    # Constraint 2: All radii must be non-negative
    for i in range(num_circles):
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: r[i]
        })

    # Constraint 3: All circles must not overlap
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            constraints.append({
                'type': 'ineq',
                'fun': lambda r, i=i, j=j: no_overlap(i, j, centers[i, 0], centers[i, 1], r[i], centers[j, 0], centers[j, 1], r[j])
            })

    # --- Step 3: Optimize ---
    # Initial guess for radii
    initial_guess = np.full(num_circles, max_radius)

    # Solve the optimization problem
    result = optimize.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1) for _ in range(num_circles)]
    )

    # Extract the optimized centers and radii
    optimized_radii = result.x
    optimized_centers = np.copy(centers)
    sum_radii = np.sum(optimized_radii)

    # Validate the result
    valid, message = validate_packing(optimized_centers, optimized_radii)
    if not valid:
        raise ValueError(f"Invalid packing: {message}")

    return optimized_centers, optimized_radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
