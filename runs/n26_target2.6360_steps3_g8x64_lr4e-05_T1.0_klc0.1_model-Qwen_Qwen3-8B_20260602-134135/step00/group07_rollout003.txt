import numpy as np
import scipy.optimize as optimize

def run_packing():

    n_circles = 26

    # Adjust the number of concentric layers and spacing for 26 circles

    total_circles = 0

    for layer in range(num_layers):

        angle_step = 2 * np.pi / num_circles_in_layer

            r = layer * 0.2  # Radius increases with layer number
            x = 0.5 + r * np.cos(angle)
            y = 0.5 + r * np.sin(angle)
            positions.append((x, y))

    centers = np.array(positions)
    radii = np.full(n_circles, 0.1)  # Initial guess for radii

    def objective_function(radii):
        return -np.sum(radii)  # Minimize negative sum (equivalent to maximizing sum)

    def constraint_overlap(i, j, radii):
        dx = centers[i, 0] - centers[j, 0]
        dy = centers[i, 1] - centers[j, 1]
        distance = np.sqrt(dx ** 2 + dy ** 2)
        return distance - (radii[i] + radii[j]) + 1e-8

    def constraint_boundary(i, radii):
        x, y = centers[i]
        return (x - radii[i]) - 0 + 1e-8  # left boundary
    def constraint_boundary_right(i, radii):
        x, y = centers[i]
        return (1.0 - x - radii[i]) + 1e-8  # right boundary
