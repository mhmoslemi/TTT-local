import numpy as np
import scipy.optimize as opt
from scipy.spatial.distance import pdist, squareform
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Generate a regular grid of centers
    n = 26
    grid_size = int(np.ceil(np.sqrt(n)))
    centers = np.zeros((n, 2))

    # Generate grid of points
    for i in range(n):
        row = i // grid_size
        col = i % grid_size
        centers[i, 0] = col / (grid_size - 1)
        centers[i, 1] = row / (grid_size - 1)

    initial_radius = 0.05
    radii = np.full(n, initial_radius)

    def sum_radii(x):
        return np.sum(x)

    # Function to compute the objective with penalty for overlaps
    def objective(x, centers, radii, eps=1e-12):
        # x is the vector of radii (same length as radii)
        # Compute pairwise distances between centers
        dists = squareform(pdist(centers))
        # Compute pairwise radii sums
        radii_sums = radii[:, np.newaxis] + radii[np.newaxis, :]
        # Compute constraint violations
        violations = np.sum((dists < radii_sums - eps) * (dists - radii_sums + eps))
        # Maximize sum of radii, so minimize negative of sum of radii
        return -sum_radii(x) + 1000 * violations

    # Constraints: radii must be positive and circles within unit square
    def constraint_radius(x):
        return x

    def constraint_boundary(x, centers):

        for i in range(n):
            x_i, y_i = centers[i]
            r_i = x[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or

        {'type': 'ineq', 'fun': lambda x: constraint_boundary(x, centers)}
    ]

    result = opt.minimize(
        fun=lambda x: objective(x, centers, radii),
        x0=x0,
        method='SLSQP',
        constraints=cons,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
