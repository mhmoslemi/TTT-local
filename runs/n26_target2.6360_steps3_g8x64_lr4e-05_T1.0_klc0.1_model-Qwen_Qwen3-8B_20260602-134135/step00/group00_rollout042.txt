import numpy as np
import cvxpy as cp

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Initialize grid positions
    grid_size = 5
    grid_spacing = 1.0 / (grid_size - 1)
    centers = np.zeros((grid_size * grid_size, 2))
    index = 0
    for i in range(grid_size):
        for j in range(grid_size):
            x = i * grid_spacing
            y = j * grid_spacing
            centers[index, 0] = x + 0.05  # Add small margin
            centers[index, 1] = y + 0.05
            index += 1

    # Add the 26th circle at a corner for better use of space
    centers[25, 0] = 0.05  # Corner near (0, 0)
    centers[25, 1] = 0.05

    # Step 2: Define optimization variables
    num_circles = centers.shape[0]
    x = cp.Variable((num_circles, 2))
    r = cp.Variable(num_circles, non_neg=True)

    # Objective: Maximize sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # Constraint 1: All circles must be inside the unit square
    for i in range(num_circles):
        constraints.append(x[i, 0] - r[i] >= -1e-8)
        constraints.append(x[i, 0] + r[i] <= 1 + 1e-8)
        constraints.append(x[i, 1] - r[i] >= -1e-8)
        constraints.append(x[i, 1] + r[i] <= 1 + 1e-8)

    # Constraint 2: All circles must be non-overlapping
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = cp.sqrt((x[i, 0] - x[j, 0])**2 + (x[i, 1] - x[j, 1])**2)
            constraints.append(dist >= r[i] + r[j] - 1e-8)

    # Constraint 3: Initial centers close to grid
    for i in range(num_circles):
        constraints.append(cp.norm(x[i] - centers[i], 2) <= 0.1)

    # Solve the optimization problem
    problem = cp.Problem(objective, constraints)
    problem.solve(solver=cp.SCS, eps=1e-6)

    # Check if solution is feasible
    if problem.status not in [cp.OPTIMAL, cp.OPTIMAL_INFEASIBLE]:
        raise ValueError(f"Optimization failed with status {problem.status}")

    # Extract the solution
    centers_result = x.value
    radii_result = r.value

    # Final validation
    is_valid, msg = validate_packing(centers_result, radii_result)
    if not is_valid:
        raise ValueError(f"Validation failed: {msg}")

    sum_radii = np.sum(radii_result)
    return centers_result, radii_result, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
