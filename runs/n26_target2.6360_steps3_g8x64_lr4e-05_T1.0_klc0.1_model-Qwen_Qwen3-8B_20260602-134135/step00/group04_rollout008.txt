import cvxpy as cp

def run_packing():

    num_circles = 26
    grid_rows = 5
    grid_cols = 5
    spacing = 1.0 / (grid_cols + 1)
    offset = 0.5 * spacing  # Slight hexagonal offset

    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Create an initial grid-like arrangement with slight hexagonal offset
    for i in range(grid_rows):
        for j in range(grid_cols):
            col = j * spacing
            row = i * spacing
            if i % 2 == 1:
                col += offset  # Hexagonal offset
            index = i * grid_cols + j
            centers[index, 0] = col
            centers[index, 1] = row
            radii[index] = 0.05  # Initial guess for radius

    # Define optimization variables
    x = cp.Variable((num_circles, 2))
    r = cp.Variable(num_circles)

    # Objective: maximize sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []
    for i in range(num_circles):
        # Boundary constraints
        constraints.append(0 <= x[i, 0] - r[i])
        constraints.append(x[i, 0] + r[i] <= 1)
        constraints.append(0 <= x[i, 1] - r[i])
        constraints.append(x[i, 1] + r[i] <= 1)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
