import scipy.optimize as optimize

    def objective(radii_flat):

    # Define constraints
    def constraint_overlap(radii_flat):
        centers_flat = centers.flatten()
        radii = radii_flat.reshape(-1, 1)
        distances = np.zeros((num_circles, num_circles))
