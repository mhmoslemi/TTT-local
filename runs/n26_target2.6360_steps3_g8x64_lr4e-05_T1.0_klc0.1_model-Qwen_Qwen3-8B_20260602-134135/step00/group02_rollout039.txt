import numpy as np
import cvxpy as cp

def run_packing():
    # Define the number of circles
    n = 26

    # Initial hexagonal grid configuration
    # Let's create a hexagonal grid with rows and columns arranged in a way that fits within the unit square
    # We'll use a staggered approach for rows
    rows = 5
    cols = 5
    offset = 0.5  # Horizontal offset for staggered rows

    # Define initial positions and radii (start with a uniform radius)
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    radius_guess = 0.04  # Initial guess for radius
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * (1.0 / (cols - 1)) + offset * (row % 2)
        y = row * (1.0 / (rows - 1)) * np.sqrt(3) / 2
        centers[i] = np.array([x, y])
        radii[i] = radius_guess

    # Use CVXPY to define and solve the optimization problem
    # Define variables
    x = cp.Variable((n, 2))
    r = cp.Variable(n)

    # Objective: Maximize the sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # Boundary constraints: circles must be within [0,1]x[0,1]
    for i in range(n):
        constraints += [
            x[i, 0] - r[i] >= 0,
            x[i, 0] + r[i] <= 1,
            x[i, 1] - r[i] >= 0,
            x[i, 1] + r[i] <= 1
        ]

    # Non-overlapping constraint: distance between centers >= sum of radii
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.sqrt((x[i, 0] - x[j, 0])**2 + (x[i, 1] - x[j, 1])**2)
            constraints += [dist >= r[i] + r[j]]

    # Solve the problem
    prob = cp.Problem(objective, constraints)
    prob.solve(solver=cp.SCS, eps=1e-6)  # Use SCS solver with tighter tolerance

    # Check for success
    if prob.status != cp.OPTIMAL:
        raise RuntimeError(f"Optimization failed with status: {prob.status}")

    # Extract the solution
    centers_opt = x.value
    radii_opt = r.value
    sum_radii = np.sum(radii_opt)

    # Ensure all variables are non-negative
    radii_opt = np.maximum(radii_opt, 0)
    centers_opt = np.clip(centers_opt, 0, 1)

    # Final validation
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        raise RuntimeError(f"Validation failed: {msg}")

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
