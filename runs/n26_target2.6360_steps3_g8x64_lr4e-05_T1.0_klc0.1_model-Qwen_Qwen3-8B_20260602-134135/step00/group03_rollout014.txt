import numpy as np
from scipy.optimize import minimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: grid-based arrangement
    n = 26
    centers_initial = []
    radii_initial = []

    # Evenly distribute circles in a grid pattern
    for i in range(n):
        row = i // 5  # 5 rows
        col = i % 5   # 5 columns
        # Adjust for 6 circles per row (not exact, but for starting point)
        x = col / 4.5
        y = row / 4.5
        radius = 0.05
        centers_initial.append((x, y))
        radii_initial.append(radius)

    # Convert to numpy arrays
    centers = np.array(centers_initial)
    radii = np.array(radii_initial)

    # Objective function to maximize sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Negative for minimization

    # Constraint: circle must be within the unit square
    def square_boundaries(x):
        for i in range(n):
            x_i, y_i = x[i*2], x[i*2+1]
            r_i = x[n*2 + i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf
        return -np.sum(x[n*2:])  # Negative for minimization

    # Constraint: non-overlapping circles
    def non_overlap(x):
        for i in range(n):
            for j in range(i+1, n):
                x_i, y_i = x[i*2], x[i*2+1]
                x_j, y_j = x[j*2], x[j*2+1]
                r_i = x[n*2 + i]
                r_j = x[n*2 + j]
                dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
                if dist < r_i + r_j - 1e-8:
                    return np.inf
        return -np.sum(x[n*2:])  # Negative for minimization

    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Bounds: positions in [0,1] and radii >= 0
    bounds = []
    for i in range(n):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
        bounds.append((0, 0.5))  # radius (smaller max to prevent overcrowding)

    # Optimization
    result = minimize(
        fun=lambda x: objective(x) + square_boundaries(x) + non_overlap(x),
        x0=initial_guess,
        bounds=bounds,
        method='L-BFGS-B',

    # Extract optimal centers and radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
