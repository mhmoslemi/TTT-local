import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26
    rows = 5
    cols_per_row = [6, 7, 7, 7, 6]  # Rows with varying number of circles
    grid = []
    
    # Set initial hexagonal grid positions
    for row in range(rows):
        for col in range(cols_per_row[row]):
            x_center = col + 0.5 * (row % 2)
            y_center = row * (np.sqrt(3)/2)
            grid.append([x_center, y_center])
    
    # Initialize centers and radii
    centers = np.array(grid)
    radii = np.full(n, 0.1)  # Start with small radii
    
    # Normalize the grid to fit in the unit square
    max_x = np.max(centers[:, 0])
    max_y = np.max(centers[:, 1])
    centers /= max_x if max_x > 0 else 1
    radii *= 0.5  # Adjust radii to fit better in the unit square
    
    # Define objective function to maximize the sum of radii
    def objective_func(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    # Define constraints
    def constraint_overlap(x, y, r1, r2):
        return np.sqrt((x - y)**2) - (r1 + r2)
    
    # Function to check if a radius change is valid
    def is_valid(radii):

        for i in range(n):

                    return False

                return False
        return True

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
