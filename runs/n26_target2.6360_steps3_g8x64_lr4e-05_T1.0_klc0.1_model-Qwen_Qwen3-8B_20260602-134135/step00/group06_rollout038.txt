import numpy as np
import scipy.optimize as optimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: grid-based placement with hexagonal arrangement
    n = 26
    rows = 6
    cols = 5
    centers = np.zeros((n, 2))
    radii = np.ones(n) * 0.05  # Initial radius estimate

    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            centers[idx] = [x, y]
    
    # Normalize to unit square
    centers[:, 0] = centers[:, 0] / (cols - 1) if cols > 1 else 0
    centers[:, 1] = centers[:, 1] / (rows - 1) if rows > 1 else 0

    def objective_function(x):
        # x is a flat array containing centers and radii
        x_flat = x.reshape((n, 4))
        centers_new = x_flat[:, :2]
        radii_new = x_flat[:, 2]
        sum_radii = np.sum(radii_new)
        return -sum_radii  # Minimize negative to maximize

    def constraint_non_overlap(x):
        x_flat = x.reshape((n, 4))
        centers_new = x_flat[:, :2]
        radii_new = x_flat[:, 2]
        distances = cdist(centers_new, centers_new)
        for i in range(n):
            for j in range(i + 1, n):
                if distances[i, j] < radii_new[i] + radii_new[j] - 1e-8:
                    return distances[i, j] - (radii_new[i] + radii_new[j] - 1e-8)
        return 0

    def constraint_boundary(x):
        x_flat = x.reshape((n, 4))
        centers_new = x_flat[:, :2]
        radii_new = x_flat[:, 2]

            x, y = centers_new[i]
            r = radii_new[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return 1e-8 - (x - r) if x - r < 0 else (x + r - 1) if x + r > 1 else (y - r) if y - r < 0 else (y + r - 1)
        return 0

    # Initial guess for centers and radii
    x0 = np.zeros(n * 4)
    x0[:2*n] = centers.flatten()
    x0[2*n:] = radii

    # Set up optimization
    cons = [
        {'type': 'ineq', 'fun': constraint_non_overlap},
        {'type': 'ineq', 'fun': constraint_boundary}
    ]
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    result = optimize.minimize(

        method='SLSQP',
        bounds=bounds,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
