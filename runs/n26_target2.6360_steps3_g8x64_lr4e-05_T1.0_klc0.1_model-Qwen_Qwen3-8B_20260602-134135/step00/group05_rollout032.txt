import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial grid-based layout
    num_circles = 26
    grid_rows = 5
    grid_cols = 5
    margin = 0.05  # small margin for initial padding

    # Initial grid of 5x5 circles
    grid_centers = []
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j * (1 - 2 * margin) / (grid_cols - 1) + margin
            y = i * (1 - 2 * margin) / (grid_rows - 1) + margin
            grid_centers.append([x, y])
    
    # Add one extra circle at the center
    grid_centers.append([0.5, 0.5])
    
    # Initial radii (all same initially)
    initial_radii = np.full(num_circles, 0.03)

    # Function to compute the objective (sum of radii)
    def objective(params, grid_centers, num_circles):
        centers = params[:num_circles * 2].reshape(num_circles, 2)
        radii = params[num_circles * 2:]
        return -np.sum(radii)  # minimize negative sum, i.e., maximize sum

    # Function to compute the penalty for constraints
    def penalty(params, grid_centers, num_circles):
        centers = params[:num_circles * 2].reshape(num_circles, 2)
        radii = params[num_circles * 2:]
        penalty = 0.0

        # Check boundaries
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                penalty += 1e6 * max(0, x - r, x + r - 1, y - r, y + r - 1)

        # Check overlaps
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    penalty += 1e6 * (radii[i] + radii[j] - dist)

        return penalty

    # Combine objective and penalty
    def total_objective(params):
        return objective(params, grid_centers, num_circles) + penalty(params, grid_centers, num_circles)

    # Initial guess
    initial_guess = np.concatenate([np.array(grid_centers), initial_radii])

    # Optimization
    result = optimize.minimize(
        total_objective,
        initial_guess,
        method='L-BFGS-B',
        bounds=[(0, 1) for _ in range(num_circles * 2)] + [(0, 1) for _ in range(num_circles)]
    )

    # Extract results
    centers = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii = result.x[num_circles * 2:]
    sum_radii = np.sum(radii)

    # Final validation (not used in function, but included in the return for completeness)
    # validate_packing(centers, radii)

    return centers, radii, sum_radii
