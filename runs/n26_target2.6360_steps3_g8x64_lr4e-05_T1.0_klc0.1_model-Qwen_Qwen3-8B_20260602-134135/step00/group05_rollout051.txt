def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_cols = 5

    bounds = [(0, 1) for _ in range(n_circles)]

        # Objective is to maximize sum of radii
        return -np.sum(radii)

    def constraint_overlap(radii, grid):
