def run_packing():

    num_circles = 26
    num_rows = 5
    num_cols = 6
    gap = 0.5  # Initial spacing between circles
    radius = 0.15  # Initial radius for each circle

    # Set up initial positions in hexagonal grid
    centers = np.zeros((num_circles, 2))

            col = j
            x = col * gap + 0.5
            y = row * (np.sqrt(3) * gap / 2) + 0.5
            if i % 2 == 1:  # Offset even rows
                x -= gap / 2
            centers[i * num_cols + j] = [x, y]

    # Initial radii
    radii = np.full(num_circles, radius)

    centers_flat = centers.reshape(-1)

        return -np.sum(radii)  # Minimize the negative sum to maximize the sum

    constraints = []

    # Constraint: centers within the unit square
    def in_bounds(x):
        centers_flat = x[:2 * num_circles]
        centers = centers_flat.reshape(-1, 2)
        return np.all(centers[:, 0] >= 0 - 1e-12) and np.all(centers[:, 0] <= 1 + 1e-12) and \

    constraints.append({'type': 'ineq', 'fun': in_bounds})

    # Constraint: circles not overlapping

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
