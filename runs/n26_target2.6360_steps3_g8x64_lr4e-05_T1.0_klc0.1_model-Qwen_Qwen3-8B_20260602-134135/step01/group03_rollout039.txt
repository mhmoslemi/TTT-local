import scipy.optimize as opt

def run_packing():

    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Initial hexagonal packing with 5 rows
    # First row: 5 circles
    row_radius = 0.1
    num_rows = 5
    for i in range(num_rows):
        num_circles_in_row = 5 if i % 2 == 0 else 4
        y_pos = i * (np.sqrt(3) * row_radius) + row_radius
        x_positions = np.linspace(0.1, 0.9, num_circles_in_row)
        for j in range(num_circles_in_row):
            centers[i * num_circles_in_row + j] = [x_positions[j], y_pos]
            radii[i * num_circles_in_row + j] = row_radius

    # Define the objective function to maximize sum of radii (minimize negative sum)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Define the constraints
    def constraint_overlap(i, j):
        return lambda radii_vec: np.sqrt(np.sum((centers[i] - centers[j])**2)) - (radii_vec[i] + radii_vec[j]) + 1e-12

    def constraint_boundary(i):
        return lambda radii_vec: (centers[i][0] - radii_vec[i]) - 1e-12
    def constraint_boundary2(i):
        return lambda radii_vec: (1.0 - centers[i][0] - radii_vec[i]) - 1e-12
    def constraint_boundary3(i):
        return lambda radii_vec: (centers[i][1] - radii_vec[i]) - 1e-12
    def constraint_boundary4(i):
        return lambda radii_vec: (1.0 - centers[i][1] - radii_vec[i]) - 1e-12

    # Flatten the centers and radii for the optimization
    flat_radii = radii

    # Define the constraints and bounds
    constraints = []
    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i)})
        constraints.append({'type': 'ineq', 'fun': constraint_boundary2(i)})
        constraints.append({'type': 'ineq', 'fun': constraint_boundary3(i)})
        constraints.append({'type': 'ineq', 'fun': constraint_boundary4(i)})

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            constraints.append({'type': 'ineq', 'fun': constraint_overlap(i, j)})

    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    result = opt.minimize(

        flat_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    sum_radii = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
