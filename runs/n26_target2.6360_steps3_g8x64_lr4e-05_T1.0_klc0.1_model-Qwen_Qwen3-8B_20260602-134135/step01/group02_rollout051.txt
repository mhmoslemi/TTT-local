import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))

    spacing = 0.95 / (num_cols - 1) if num_cols > 1 else 0.5
    vertical_spacing = 0.95 / (num_rows - 1) if num_rows > 1 else 0.5
    offset = vertical_spacing / 2  # Offset for staggered rows

    def objective(radii):
        return -np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
