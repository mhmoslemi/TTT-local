import numpy as np
import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Calculate initial hexagonal grid
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    spacing = 1.0 / (cols + 1)
    
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + 0.5 * spacing
        y = row * spacing + 0.5 * spacing + (row % 2) * spacing * 0.5
        centers[i] = [x, y]
        radii[i] = 0.04  # Initial guess for radius
    
    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization
    
    # Constraints to ensure circles are within the unit square and don't overlap
    def constraint_boundary(i, centers, radii):
        return lambda x: np.min([
            centers[i, 0] - x[i],
            1 - centers[i, 0] - x[i],
            centers[i, 1] - x[i],
            1 - centers[i, 1] - x[i]
        ]) + 1e-6
    
    def constraint_overlap(i, j, centers):
        return lambda x: np.sqrt(np.sum((centers[i] - centers[j])**2)) - (x[i] + x[j]) + 1e-6

    constraints = []

        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i, centers, radii)})

    # Optimization setup

    result = opt.minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
