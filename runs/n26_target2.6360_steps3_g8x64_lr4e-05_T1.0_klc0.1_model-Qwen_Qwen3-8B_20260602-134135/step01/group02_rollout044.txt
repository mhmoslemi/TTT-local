import numpy as np
import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    spacing = 1.0 / (cols + 1)
    offset = 0.5 * spacing

        row = i // cols
        col = i % cols

        y = row * spacing + offset
        centers[i] = [x, y]

    # Function to calculate total sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    def distance(c1, c2):

        return 1e6

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
