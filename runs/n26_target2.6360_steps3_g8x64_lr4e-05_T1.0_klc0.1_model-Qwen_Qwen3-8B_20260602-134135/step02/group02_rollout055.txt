import numpy as np
import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Initial random grid-based placement
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Randomly place centers in a grid with some perturbation
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    spacing_x = 1.0 / (cols + 1)
    spacing_y = 1.0 / (rows + 1)
    offset_x = spacing_x * 0.5
    offset_y = spacing_y * 0.5

    # Random perturbation factor
    perturb = 0.15
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing_x + offset_x + np.random.uniform(-perturb, perturb)
        y = row * spacing_y + offset_y + np.random.uniform(-perturb, perturb)
        centers[i, 0] = np.clip(x, 0, 1)
        centers[i, 1] = np.clip(y, 0, 1)

    # Initial radius based on available space per circle
    # Use a more aggressive initial radius for better exploration
    radius_initial = (1.0 / np.sqrt(n)) * 0.95
    radii = np.full(n, radius_initial)

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint to ensure circles are within the unit square
    def constraint_boundary(i, centers):
        def constraint(x):
            x_i = x[i]
            cx, cy = centers[i]
            # Left boundary
            left = cx - x_i
            # Right boundary
            right = 1 - cx - x_i
            # Top boundary
            top = 1 - cy - x_i
            # Bottom boundary
            bottom = cy - x_i
            return np.min([left, right, top, bottom])
        return constraint

    # Constraint to ensure no two circles overlap
    def constraint_overlap(i, j, centers):
        def constraint(x):
            x_i = x[i]
            x_j = x[j]
            cx_i, cy_i = centers[i]
            cx_j, cy_j = centers[j]
            dist = np.sqrt((cx_i - cx_j)**2 + (cy_i - cy_j)**2)
            return dist - (x_i + x_j)
        return constraint

    # Constraints for the optimization
    constraints = []
    for i in range(n):
        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i, centers)})

    bounds = [(0.0, 0.5) for _ in range(n)]

    result = opt.minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
