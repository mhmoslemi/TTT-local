import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Hexagonal grid with tighter spacing
    row_count = 4
    col_count = 7
    spacing_x = 1.0 / (col_count + 1)
    spacing_y = np.sqrt(3) / 2 * spacing_x

    for i in range(n):
        row = i // col_count
        col = i % col_count
        x = col * spacing_x + 0.5 * spacing_x
        y = row * spacing_y + 0.5 * spacing_y

        radii[i] = spacing_x * 0.5

    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraint: circles must stay inside the unit square
    def constraint_boundary(i, centers):
        def cons(x):
            x_i, y_i = centers[i]
            r = x[i]
            # Calculate distance to each edge
            left = x_i - r
            right = 1 - x_i - r
            bottom = y_i - r
            top = 1 - y_i - r
            return np.min([left, right, bottom, top])
        return cons

    # Constraint: circles must not overlap
    def constraint_overlap(i, j, centers):
        def cons(x):
            x_i, y_i = centers[i]
            x_j, y_j = centers[j]
            r_i = x[i]
            r_j = x[j]
            dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
            return dist - r_i - r_j
        return cons

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
