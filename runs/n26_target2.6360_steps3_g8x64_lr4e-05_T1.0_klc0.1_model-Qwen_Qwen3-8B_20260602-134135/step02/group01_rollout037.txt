import numpy as np
import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Improved initial placement: hexagonal close packing with compact grid
    # This grid is designed to evenly distribute 26 circles in the square
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Define a more compact hexagonal grid
    rows = 6
    cols = 5
    spacing = 1.0 / (cols + 1)
    offset = np.sqrt(3) / 2 * spacing  # Height of hexagon

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + 0.5 * spacing
        y = row * offset + spacing * 0.5
        # Bias towards the edges for better corner utilization
        if row == 0 or row == rows - 1:
            x += spacing * 0.2
        if col == 0 or col == cols - 1:
            y += spacing * 0.2
        centers[i] = [x, y]
        radii[i] = 0.1  # Initial radius for better convergence

    # Define objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraint: circles must stay inside the unit square
    def constraint_boundary(i, centers):
        def cons(x):
            x_i, y_i = centers[i]
            r = x[i]
            # Calculate distance to each edge
            left = x_i - r
            right = 1 - x_i - r
            bottom = y_i - r
            top = 1 - y_i - r
            return np.min([left, right, bottom, top])
        return cons

    # Constraint: circles must not overlap
    def constraint_overlap(i, j, centers):
        def cons(x):
            x_i, y_i = centers[i]
            x_j, y_j = centers[j]
            r_i = x[i]
            r_j = x[j]
            dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
            return dist - r_i - r_j
        return cons

    # Define optimization problem
    bounds = [(0, 0.5) for _ in range(n)]  # Radii are constrained to be less than 0.5

    # Set up constraints
    constraints = []
    for i in range(n):
        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i, centers)})
    
    for i in range(n):
        for j in range(i + 1, n):
            constraints.append({'type': 'ineq', 'fun': constraint_overlap(i, j, centers)})

    # Use COBYLA optimization for better convergence with bounds
    result = opt.minimize(
        objective,
        radii,
        method='COBYLA',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8,
        options={'disp': False}
    )

    if not result.success:
        warnings.warn("Optimization did not converge")
        return centers, radii, np.sum(radii)

    best_centers = centers
    best_radii = result.x
    best_sum = np.sum(best_radii)

    # Final validation

    if not valid:
        warnings.warn(f"Validation failed: {msg}")
        return centers, radii, np.sum(radii)

    return best_centers, best_radii, best_sum

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
