import numpy as np
from scipy.optimize import differential_evolution
from scipy.spatial.distance import pdist, squareform
import random

def run_packing():
    num_circles = 26
    # Initial positions using hexagonal grid
    grid_rows = 4
    grid_cols = 7
    spacing_x = 1.0 / grid_cols
    spacing_y = 1.0 / grid_rows
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    idx = 0
    
    # Create initial hexagonal grid with some randomness
    for row in range(grid_rows):
        for col in range(grid_cols):
            if idx < num_circles:
                x = col * spacing_x + spacing_x / 2
                y = row * spacing_y + spacing_y / 2
                # Add small perturbation to avoid exact same centers
                perturbation = np.random.uniform(-0.01, 0.01, size=2)
                centers[idx] = np.array([x + perturbation[0], y + perturbation[1]])
                radii[idx] = 0.05  # Initial radius
                idx += 1

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Return negative sum to minimize

    # Constraint for circles not overlapping
    def constraint_non_overlap(radii):
        dist_matrix = squareform(pdist(centers))
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                if dist_matrix[i, j] < radii[i] + radii[j] - 1e-12:
                    return -1
        return 0

    # Constraint for circles within the unit square
    def constraint_boundary(radii):
        for i in range(len(radii)):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return -1
        return 0

    # Constraint for non-negative radii
    def constraint_non_negative(radii):
        for r in radii:
            if r < 0:
                return -1
        return 0

    # Bounds for radii
    bounds = [(0, 0.5) for _ in range(num_circles)]

    # Optimization using differential evolution
    result = differential_evolution(

        bounds,
