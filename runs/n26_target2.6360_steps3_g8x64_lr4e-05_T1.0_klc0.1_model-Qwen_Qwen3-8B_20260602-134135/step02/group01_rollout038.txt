import numpy as np
import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Create a more refined initial placement with variable spacing and corner bias
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initial placement logic
    row_count = 4
    col_count = 7
    total_rows = 5
    spacing = 0.8 / (col_count + 1)
    offset = 0.0

    for i in range(n):
        row = i // col_count
        col = i % col_count
        x = col * spacing + 0.5 * spacing
        y = row * spacing + offset
        # Bias circles near the corners to make better use of space
        if (row == 0 or row == total_rows - 1) or (col == 0 or col == col_count - 1):
            x += spacing * 0.15
            y += spacing * 0.15
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial radius

    # Define objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraint functions
    def constraint_boundary(i, centers):
        def cons(x):
            x_i, y_i = centers[i]
            r = x[i]
            # Calculate distance to each edge
            left = x_i - r
            right = 1 - x_i - r
            bottom = y_i - r
            top = 1 - y_i - r
            return np.min([left, right, bottom, top])
        return cons

    def constraint_overlap(i, j, centers):
        def cons(x):
            x_i, y_i = centers[i]
            x_j, y_j = centers[j]
            r_i = x[i]
            r_j = x[j]
            dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
            return dist - r_i - r_j
        return cons

    # Setup bounds
    bounds = [(0, 0.5) for _ in range(n)]

    # Setup constraints
    constraints = []
    for i in range(n):
        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i, centers)})

    for i in range(n):
        for j in range(i + 1, n):
            constraints.append({'type': 'ineq', 'fun': constraint_overlap(i, j, centers)})

    # Perform optimization
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8,
        options={'maxiter': 1000, 'ftol': 1e-9}
    )

    if not result.success:
        warnings.warn("Optimization did not converge")
        return centers, radii, np.sum(radii)

    best_centers = centers
    best_radii = result.x
    best_sum = np.sum(best_radii)

    # Final validation
    valid, msg = validate_packing(best_centers, best_radii)
    if not valid:
        warnings.warn(f"Validation failed: {msg}")
        return centers, radii, np.sum(radii)

    return best_centers, best_radii, best_sum

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
