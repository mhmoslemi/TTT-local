import numpy as np
import scipy.optimize as opt
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Initial placement: use hexagonal grid with tighter spacing and better boundary utilization
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Hexagonal grid with tighter spacing and better boundary utilization
    row_count = 5
    col_count = 6
    spacing = 1.0 / (col_count + 2)
    offset = 0.5 * spacing

    for i in range(n):
        row = i // col_count
        col = i % col_count
        x = col * spacing + 0.5 * spacing
        y = row * spacing + offset
        # Adjust x and y near the boundaries to better utilize corner and edge space
        if row == 0 or row == row_count - 1:
            x += spacing * 0.15
        if col == 0 or col == col_count - 1:
            y += spacing * 0.15
        centers[i] = [x, y]
        radii[i] = 0.04  # Reduced initial radius to allow more room for optimization

    # Define objective function: maximize sum of radii
    def objective(radii_vector):
        return -np.sum(radii_vector)  # Negative for minimization

    # Define constraints
    def constraint_boundary(i, centers):
        def cons(radii):
            x, y = centers[i]
            r = radii[i]
            # Calculate minimum distance to any edge
            return np.min([x - r, 1 - x - r, y - r, 1 - y - r])
        return cons

    def constraint_overlap(i, j, centers):
        def cons(radii):
            x_i, y_i = centers[i]
            x_j, y_j = centers[j]
            r_i = radii[i]
            r_j = radii[j]
            # Distance between centers
            dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
            return dist - r_i - r_j
        return cons

    # Define optimization bounds
    bounds = [(0, 0.5) for _ in range(n)]  # Radii bounded between 0 and 0.5

    # Set up constraints
    constraints = []
    for i in range(n):
        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i, centers)})

        for j in range(i + 1, n):
            constraints.append({'type': 'ineq', 'fun': constraint_overlap(i, j, centers)})

    # Perform optimization using a more robust method

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
