import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n_circles = 26

    # Objective function: maximize sum of radii
    def objective(x):
        # x is flattened array: [c1x, c1y, r1, c2x, c2y, r2, ..., cnx, cny, rn]
        flattened = x.reshape(-1, 3)
        centers = flattened[:, :2]
        radii = flattened[:, 2]

        # Calculate sum of radii
        sum_radii = np.sum(radii)

        # Add a small penalty for overlapping circles
        penalty = 0.0
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    penalty += (radii[i] + radii[j] - dist) ** 2

        return -sum_radii - penalty  # Minimize negative sum and penalty

    # Constraints: circles must stay within the unit square
    def constraint_boundary(x):
        flattened = x.reshape(-1, 3)
        centers = flattened[:, :2]

    )
