import scipy.optimize as opt
from scipy.optimize import LinearConstraint, Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based circle positions
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Create a grid of positions with spacing (approximate hexagonal)
    positions = np.zeros((n, 2))
    count = 0
    for i in range(rows):

            y = i / (rows - 1) if rows > 1 else 0.5
            positions[count] = [x, y]
            count += 1

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

            for i in range(n):
                x, y = positions[i]
                r = radii[i]
                if (x - r < 0 or x + r > 1 or
                    y - r < 0 or y + r > 1):
                    return -1
            return 0
        return constraint

    # Constraints: circles must not overlap
    def no_overlap(radii):

            for i in range(n):
                for j in range(i + 1, n):
                    dx = positions[i][0] - positions[j][0]
                    dy = positions[i][1] - positions[j][1]

                        return -1
            return 0
        return constraint

    ]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
