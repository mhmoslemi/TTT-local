import numpy as np
import scipy.optimize as optimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial arrangement: Hexagonal packing
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Hexagonal grid setup for initial arrangement
    rows = int(np.sqrt(n))
    cols = n // rows + 1
    
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 1:
                x = j + 0.5
            else:
                x = j
            y = i
            if i * cols + j < n:
                centers[i * cols + j] = np.array([x / (cols - 1), y / (rows - 1)])
                radii[i * cols + j] = 0.05  # Initial radius

    # Objective function to maximize sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    # Constraint: Distance between centers >= sum of radii
    def constraint_dist(x, centers):
        radii = x[:n]
        centers = centers.reshape((n, 2))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    return -1
        return 0

    # Constraint: Circles must stay within the unit square
    def constraint_boundary(x, centers):
        radii = x[:n]
        centers = centers.reshape((n, 2))
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1
        return 0

    # Combine constraints into a single constraint function
    def constraints(x, centers):
        # Radii must be non-negative
        if np.any(x[:n] < 0):
            return -1
        # Check distance constraints
        if constraint_dist(x, centers) < 0:
            return -1
        # Check boundary constraints
        if constraint_boundary(x, centers) < 0:
            return -1
        return 0

    # Initial guess: radii values
    x0 = np.concatenate([radii, np.zeros(0)])  # Radii as initial guess, nothing else needed

    bounds = [(0, 1.0) for _ in range(n)]  # Radii can range from 0 to 1.0

        bounds=bounds,
        method="SLSQP"
    )

    result = optimize.minimize(
        fun=objective,
        x0=x0,
        constraints=constraints_func,
        method="SLSQP",

    )

    best_centers = centers
    sum_radii = np.sum(best_radii)
