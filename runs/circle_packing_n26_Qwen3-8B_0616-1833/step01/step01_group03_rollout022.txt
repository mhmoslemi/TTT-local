import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial positions for 26 circles in a grid-like arrangement
    n = 26
    grid_size = int(np.ceil(np.sqrt(n)))  # 6 rows, 5 columns or 5 rows, 6 columns
    centers = np.zeros((n, 2))
    index = 0
    for i in range(grid_size):
        for j in range(grid_size):
            if index < n:
                # Place circles with spacing to ensure minimal overlap
                x = j * 1.0 / (grid_size - 1)
                y = i * 1.0 / (grid_size - 1)
                centers[index, 0] = x
                centers[index, 1] = y
                index += 1
    
    # Initial radii (same for all circles at the start)
    initial_radii = np.full(n, 0.05)

    def total_radius(radii):
        return -np.sum(radii)  # Minimize the negative sum for maximization

    def constraints(centers, radii):
        # Ensure each circle stays within the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-8 or x + r > 1 + 1e-8 or y - r < -1e-8 or y + r > 1 + 1e-8:
                return False
        # Ensure no overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    # Prepare for optimization
    x_bounds = []
    for i in range(n):
        x_bounds.append((0.0, 1.0))
        x_bounds.append((0.0, 1.0))
        x_bounds.append((0.0, 0.5))  # Radius is always non-negative
    x_bounds = np.array(x_bounds).reshape((n, 3))

    # Optimization
    x0 = np.concatenate((centers.flatten(), initial_radii))
    result = minimize(
        total_radius,
        x0,
        method='SLSQP',
        bounds=Bounds(x_bounds),
        constraints={'type': 'ineq', 'fun': lambda x: constraints(x[:n*2].reshape(n, 2), x[n*2:])},
        options={'ftol': 1e-10, 'maxiter': 500}
    )

    # Extract results
    centers_opt = result.x[:n*2].reshape(n, 2)
    radii_opt = result.x[n*2:]
    sum_radii = np.sum(radii_opt)

    # Validate the final result
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        raise ValueError(msg)

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
