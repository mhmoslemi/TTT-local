import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based configuration of centers
    num_circles = 26
    grid_rows, grid_cols = 5, 5
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // grid_cols
        col = i % grid_cols
        centers[i, 0] = col / (grid_cols - 1)
        centers[i, 1] = row / (grid_rows - 1)

    # Initial guessed radii (all equal for simplicity)
    radii = np.full(num_circles, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x contains radii in order
        return -np.sum(x)

    # Box constraint: radii >= 0 (since we're minimizing negative sum)
    bounds = [(0, np.inf) for _ in range(num_circles)]

    # Constraint function: distance between centers >= r_i + r_j
    def constraint_dist(x, centers, i, j):
        r_i = x[i]
        r_j = x[j]
        dx = centers[i, 0] - centers[j, 0]
        dy = centers[i, 1] - centers[j, 1]
        dist = np.sqrt(dx**2 + dy**2)
        return dist - r_i - r_j

    # Build constraint list
    constraints = []
    for i in range(num_circles):

            constraints.append({
                'type': 'ineq',
                'fun': lambda x, i=i, j=j: constraint_dist(x, centers, i, j)
            })

    # Optimization
    result = opt.minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
