import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define constants
    n = 26
    max_radius = 1.0
    initial_radius = 0.05

    # Initialize centers using a hexagonal close-packed (HCP) configuration
    def hexagonal_grid(num_rows, num_cols):
        """
        Generate a hexagonal grid of centers.
        """
        centers = []
        for i in range(num_rows):
            for j in range(num_cols):
                x = j + 0.5 * i
                y = i * np.sqrt(3) / 2
                centers.append([x, y])
        return np.array(centers)

    # Generate an initial grid for 26 circles
    # Try to fit in 5 rows with 5, 5, 5, 5, 6 columns
    centers = hexagonal_grid(5, 5)
    # Trim and add more if needed
    centers = centers[:n]
    radii = np.full(n, initial_radius)

    # Define objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative because we're minimizing

    # Define constraints
    def constraint_centers(centers):
        return np.all((centers >= 0) & (centers <= 1))

    def constraint_radii(radii):
        return np.all(radii >= 0)

    def constraint_overlap(centers, radii):
        n = len(centers)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-6:
                    return False
        return True

    # Define bounds for centers (must be within [0,1]^2)
    bounds_centers = [(0, 1) for _ in range(n * 2)]

    # Define bounds for radii (must be non-negative)
    bounds_radii = [(0, max_radius) for _ in range(n)]

    # Define optimization variables
    # We'll optimize centers and radii simultaneously
    # Flatten centers and radii into a single vector for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Define constraints in terms of the variable vector
    def constraint_centers_flat(x):
        centers = x[:n * 2].reshape((n, 2))
        return constraint_centers(centers)

    def constraint_radii_flat(x):
        radii = x[n * 2:]
        return constraint_radii(centers, radii)

    def constraint_overlap_flat(x):
        centers = x[:n * 2].reshape((n, 2))
        radii = x[n * 2:]
        return constraint_overlap(centers, radii)

    # Optimization setup
    constraints = [
        {'type': 'ineq', 'fun': constraint_centers_flat},
        {'type': 'ineq', 'fun': constraint_radii_flat},
        {'type': 'ineq', 'fun': constraint_overlap_flat}
    ]

    # Run optimization
    result = optimize.minimize(
        fun=objective,
        x0=x0,
        bounds=[
            (0, 1) for _ in range(n * 2)
        ] + [(0, max_radius) for _ in range(n)],
        constraints=constraints,
        method='SLSQP',
        tol=1e-6

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
