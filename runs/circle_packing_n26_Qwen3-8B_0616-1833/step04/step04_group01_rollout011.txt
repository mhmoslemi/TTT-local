import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def objective_function(params, n_circles):

    radii = params[n_circles * 2:].reshape((n_circles,))

    sum_radii = np.sum(radii)

        edge_penalty += np.maximum(0, r - (1 - y - 1e-12))

def constraint_function(params, n_circles):

    radii = params[n_circles * 2:].reshape((n_circles,))

            # Constraint: distance >= r_i + r_j
            constraints.append(dist - (radii[i] + radii[j] - 1e-8))
    return constraints

def initialize_centers(n_circles):

    spacing = 0.4
    for r in range(6):
        for c in range(6):

    corner_centers = [
        [0.1, 0.1],
        [0.1, 0.9],
        [0.9, 0.1],
        [0.9, 0.9]
    ]
    return np.array(hex_centers + corner_centers)[:n_circles]

def optimize_centers(radii, n_circles, initial_centers):
    bounds = []
    for _ in range(n_circles * 2):
        bounds.append((0, 1))
    for _ in range(n_circles):
        bounds.append((0, 1))
    result = optimize.minimize(
        fun=lambda x: objective_function(x, n_circles),
        x0=np.concatenate([initial_centers.flatten(), radii]),
        bounds=bounds,
        constraints={"type": "ineq", "fun": lambda x: constraint_function(x, n_circles)},
        method="SLSQP"
    )
    return result.x[:n_circles*2].reshape((n_circles, 2)), result.x[n_circles*2:]
