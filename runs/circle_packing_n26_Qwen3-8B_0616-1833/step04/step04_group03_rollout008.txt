import scipy.optimize as opt

        centers[i] = [x, y]

        return -np.sum(radii)

            r = radii[i]

            return (x - r)  # x - r >= 0
        return {'type': 'ineq', 'fun': cons}

    def constraint_boundary_right(i, centers):

            x, y = centers[i]
            return (1 - x - r)  # 1 - x - r >= 0

            dist = np.sqrt(dx**2 + dy**2)
