import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def objective_function(params, n_circles):
    centers = params[:n_circles * 2].reshape((n_circles, 2))
    radii = params[n_circles * 2:].reshape((n_circles,))
    sum_radii = np.sum(radii)
    radius_penalty = np.sum(np.maximum(0, -radii))
    edge_penalty = 0
    for i in range(n_circles):
        x, y = centers[i]
        r = radii[i]
        edge_penalty += max(0, r - (1 - x - 1e-12))
        edge_penalty += max(0, r - (1 - (1 - x) - 1e-12))
        edge_penalty += max(0, r - (1 - y - 1e-12))
        edge_penalty += max(0, r - (1 - (1 - y) - 1e-12))
    overlap_penalty = 0
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                overlap_penalty += (radii[i] + radii[j] - dist)
    return -sum_radii + radius_penalty + edge_penalty + overlap_penalty

def constraint_function(params, n_circles):
    centers = params[:n_circles * 2].reshape((n_circles, 2))
    radii = params[n_circles * 2:].reshape((n_circles,))
    constraints = []
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            constraints.append(dist - (radii[i] + radii[j] - 1e-8))
    return constraints

def initialize_centers(n_circles):
    # Create a more efficient initial layout with hexagonal arrangement
    hex_centers = []
    for r in range(5):
        for c in range(6):
            if r % 2 == 0:
                x = c * 0.4 + 0.1
            else:
                x = (c + 0.5) * 0.4 + 0.1
            y = r * 0.4 + 0.1
            hex_centers.append([x, y])
    corner_centers = [
        [0.1, 0.1],
        [0.1, 0.9],
        [0.9, 0.1],
        [0.9, 0.9]
    ]
    return np.array(hex_centers + corner_centers)[:n_circles]

def optimize_radii(centers, n_circles):
    bounds = []
    for _ in range(n_circles * 2):
        bounds.append((0, 1))
    for _ in range(n_circles):
        bounds.append((0, 1))
    result = optimize.minimize(
        fun=lambda x: objective_function(x, n_circles),
        x0=np.concatenate([centers.flatten(), np.ones(n_circles) * 0.05]),
        bounds=bounds,
        constraints={"type": "ineq", "fun": lambda x: constraint_function(x, n_circles)},
        method="SLSQP"
    )
    return result.x[n_circles * 2:]

def optimize_centers(radii, n_circles, initial_centers):
    bounds = []
    for _ in range(n_circles * 2):
        bounds.append((0, 1))
    for _ in range(n_circles):
        bounds.append((0, 1))
    result = optimize.minimize(
        fun=lambda x: objective_function(x, n_circles),
        x0=np.concatenate([initial_centers.flatten(), radii]),
        bounds=bounds,
        constraints={"type": "ineq", "fun": lambda x: constraint_function(x, n_circles)},
        method="SLSQP"
    )
    return result.x[:n_circles*2].reshape((n_circles, 2)), result.x[n_circles*2:]

def optimize_combined(centers, radii, n_circles):
    bounds = [(0, 1)] * (n_circles * 2) + [(0, 1)] * n_circles
    result = optimize.minimize(
        fun=lambda x: objective_function(x, n_circles),
        x0=np.concatenate([centers.flatten(), radii]),
        bounds=bounds,
        constraints={"type": "ineq", "fun": lambda x: constraint_function(x, n_circles)},
        method="Cobyla"
    )
    return result.x[:n_circles*2].reshape((n_circles, 2)), result.x[n_circles*2:]

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    initial_centers = initialize_centers(n_circles)
    initial_radii = optimize_radii(initial_centers, n_circles)

    optimized_centers, optimized_radii = optimize_centers(initial_radii, n_circles, initial_centers)
    optimized_centers, optimized_radii = optimize_combined(optimized_centers, optimized_radii, n_circles)

    sum_radii = np.sum(optimized_radii)

    # Validate the packing
    is_valid, message = validate_packing(optimized_centers, optimized_radii)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.array([]), np.array([]), 0.0

    return optimized_centers, optimized_radii, sum_radii
