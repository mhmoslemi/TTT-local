import scipy.optimize as opt

def run_packing():

    grid = np.linspace(0, 1, 6)  # 6 points in each direction to allow for 25 circles
    centers = []
    for i in range(n):
        row = i // 5
        col = i % 5
        x = grid[row] + np.random.uniform(-0.05, 0.05)
        y = grid[col] + np.random.uniform(-0.05, 0.05)
        centers.append([x, y])
    centers = np.array(centers)

    # Initialize radii by considering distance to nearest boundary
    radii = np.zeros(n)
    for i in range(n):
        x, y = centers[i]
        dist_to_boundary = min(x, 1 - x, y, 1 - y)
        radii[i] = dist_to_boundary * 0.6  # Safe radius based on distance to boundary

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize the sum

    # Define constraints for boundary and non-overlap
    constraints = []

    def constraint_boundary(i, centers):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (x - r)  # Ensure x - r >= 0
        return {'type': 'ineq', 'fun': cons}

    def constraint_boundary_right(i, centers):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (1 - x - r)  # Ensure 1 - x - r >= 0
        return {'type': 'ineq', 'fun': cons}

    def constraint_boundary_top(i, centers):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (1 - y - r)  # Ensure 1 - y - r >= 0
        return {'type': 'ineq', 'fun': cons}

    def constraint_boundary_bottom(i, centers):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (y - r)  # Ensure y - r >= 0
        return {'type': 'ineq', 'fun': cons}

    def constraint_overlap(i, j, centers):
        def cons(radii):
            r_i = radii[i]
            r_j = radii[j]
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            return dist - (r_i + r_j)  # Ensure dist >= r_i + r_j
        return {'type': 'ineq', 'fun': cons}

    # Add boundary constraints for all circles
    for i in range(n):
        constraints.append(constraint_boundary(i, centers))

        constraints.append(constraint_boundary_top(i, centers))

    # Add non-overlap constraints for all pairs
    for i in range(n):

            constraints.append(constraint_overlap(i, j, centers))

    bounds = [(0, None) for _ in range(n)]

    # Solve the optimization problem
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
