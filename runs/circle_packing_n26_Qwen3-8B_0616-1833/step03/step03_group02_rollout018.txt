import scipy.optimize as opt

def run_packing():

    # This forms a grid-like arrangement with some flexibility
    positions = np.zeros((n, 2))
    k = 0
    for i in range(5):
        for j in range(5):
            if i == 4 and j == 4:
                # Extra circle at bottom right corner
                positions[k] = np.array([0.7, 0.7])
            else:
                positions[k] = np.array([(j + 0.5) / 5, (i + 0.5) / 5])
            k += 1

    # Initial guess for radii
    radii = np.full(n, 0.05)

    # Objective function: maximize the sum of radii (converted to minimization)
    def objective(radii):
        return -np.sum(radii)

    # Constraints to ensure circles stay within the unit square
    def constraint_boundary(x, y, r):
        return x - r
    def constraint_boundary_right(x, y, r):
        return 1 - x - r
    def constraint_boundary_top(x, y, r):
        return 1 - y - r
    def constraint_boundary_bottom(x, y, r):
        return y - r

    # Constraints to ensure circles do not overlap
    def constraint_overlap(x1, y1, r1, x2, y2, r2):
        dx = x1 - x2
        dy = y1 - y2
        dist = np.sqrt(dx**2 + dy**2)
        return dist - r1 - r2

    # Define all constraints
    constraints = []
    for i in range(n):
        x, y = positions[i]
        constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: constraint_boundary(x, y, r[i])})
