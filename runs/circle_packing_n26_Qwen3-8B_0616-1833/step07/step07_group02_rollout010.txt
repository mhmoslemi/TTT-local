import scipy.optimize as opt

def run_packing():

    cols = 5

    radii = np.full(n, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize the sum
    
    # Constraints for boundary and non-overlap
    constraints = []
    
    def constraint_boundary(i, centers, radii):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (x - r)  # Ensure x - r >= 0
        return {'type': 'ineq', 'fun': cons}
    
    def constraint_boundary_right(i, centers, radii):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (1 - x - r)  # Ensure 1 - x - r >= 0
        return {'type': 'ineq', 'fun': cons}
    
    def constraint_boundary_top(i, centers, radii):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (1 - y - r)  # Ensure 1 - y - r >= 0
        return {'type': 'ineq', 'fun': cons}
    
    def constraint_boundary_bottom(i, centers, radii):
        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (y - r)  # Ensure y - r >= 0

    def constraint_overlap(i, j, centers, radii):

            r_j = radii[j]
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            return dist - (r_i + r_j)  # Ensure dist >= r_i + r_j
