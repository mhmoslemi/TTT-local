import scipy.optimize as opt

    constraints = []

            return (x - r)  # Ensure x - r >= 0
        return {'type': 'ineq', 'fun': cons}

        def cons(radii):
            r = radii[i]
            x, y = centers[i]
            return (1 - x - r)  # Ensure 1 - x - r >= 0
        return {'type': 'ineq', 'fun': cons}

            r = radii[i]
            x, y = centers[i]

            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
