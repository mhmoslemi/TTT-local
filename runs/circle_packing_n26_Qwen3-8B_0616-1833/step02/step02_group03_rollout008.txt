import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial grid-based configuration: 5 rows of 5 circles (25 circles), plus one additional circle
    # Grid spacing can be adjusted to provide room for expansion
    grid_spacing = 0.26  # Adjust spacing to allow for some flexibility

    # Initialize centers in a grid pattern
    centers = []
    for i in range(5):
        for j in range(5):
            x = j * grid_spacing
            y = i * grid_spacing
            centers.append([x, y])

    # Add one more circle near the center to allow for optimization
    centers.append([0.5, 0.5])
    centers = np.array(centers)

    # Initial radii - all initially set to grid spacing divided by 2 to fit in the grid
    radii = np.full(26, grid_spacing / 2)

    # Define objective function: maximize sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Negative for minimization

    # Define constraints
    constraints = []

    # Boundary constraints: each circle must be within [0, 1] x [0, 1]
    for i in range(26):
        def boundary_constraint(radii_flat):
            x = centers[i, 0] - radii_flat[i]
            y = centers[i, 1] - radii_flat[i]
            if x < -1e-12:
                return -1e-12
            if x > 1 + 1e-12:
                return -1e-12
            if y < -1e-12:
                return -1e-12
            if y > 1 + 1e-12:
                return -1e-12
            return 0

        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: boundary_constraint(r)})

    # Non-overlap constraints: distance between centers >= sum of radii
    for i in range(26):
        for j in range(i + 1, 26):
            def overlap_constraint(radii_flat):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                return dist - (radii_flat[i] + radii_flat[j]) + 1e-12

            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: overlap_constraint(r)})

    # Optimization
    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1) for _ in range(26)]
    )

    # Final centers and radii
    final_centers = centers
    final_radii = result.x
    sum_radii = np.sum(final_radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
