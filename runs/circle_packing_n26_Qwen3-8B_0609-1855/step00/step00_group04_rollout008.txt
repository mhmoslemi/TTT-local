import numpy as np
import scipy.optimize as optimize

def run_packing():

    n = 26

    num_rows = 2
    num_cols = 13

    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col / (num_cols - 1)
        y = row / (num_rows - 1)
        centers[i] = [x, y]
    
    # Initial radii: all equal for simplicity
    radii = np.full(n, 1.0 / (num_rows * num_cols))

    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        # x contains positions and radii: [x1, y1, r1, x2, y2, r2, ...]
        # Reshape into centers and radii
        positions = x[:2 * n].reshape(n, 2)
        radii_vals = x[2 * n:]
        # Calculate sum of radii
        sum_radii = np.sum(radii_vals)
        # Return negative sum to maximize
        return -sum_radii

    # Constraints: all circles within [0,1]x[0,1] and non-overlapping
    def constraint_positions(x):
        positions = x[:2 * n].reshape(n, 2)
        return np.concatenate([
