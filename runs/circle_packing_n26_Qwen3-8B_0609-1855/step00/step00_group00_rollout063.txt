import numpy as np
import scipy.optimize as opt
from typing import Tuple, List, Union

def validate_packing(centers: np.ndarray, radii: np.ndarray) -> Tuple[bool, str]:
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_hexagonal_grid(n: int, side_length: float = 0.1) -> Tuple[np.ndarray, np.ndarray]:
    centers = np.zeros((n, 2))
    radii = np.full(n, side_length)
    row = 0
    col = 0
    for i in range(n):
        x = col * (1 + side_length * np.sqrt(3))
        y = row * (1 + side_length * 2)
        centers[i, 0] = x
        centers[i, 1] = y
        col += 1
        if col % 2 == 0:
            row += 1
        if i == n - 1:
            break
    return centers, radii

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26
    side_length_min = 0.01
    num_restarts = 3
    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)
    best_sum = 0.0

    for _ in range(num_restarts):
        centers, radii = initial_hexagonal_grid(n, side_length_min)
        centers = np.clip(centers, 0, 1)
        radii = np.clip(radii, 0, 1)

        # Flatten centers and radii for optimization
        flat_centers = centers.reshape(-1)
        flat_radii = radii

        def objective(params):
            centers_new = params[:2*n].reshape(n, 2)
            radii_new = params[2*n:]
            return -np.sum(radii_new)

        def constraint_center_bounds(params):
            centers_new = params[:2*n].reshape(n, 2)
            return np.concatenate([centers_new.flatten(), params[2*n:]])

        def constraint_circle_bounds(params):
            centers_new = params[:2*n].reshape(n, 2)
            radii_new = params[2*n:]
            for i in range(n):
                x, y = centers_new[i]
                r = radii_new[i]
                if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                    return np.inf
            return 0

        def constraint_non_overlap(params):
            centers_new = params[:2*n].reshape(n, 2)
            radii_new = params[2*n:]
            for i in range(n):
                for j in range(i + 1, n):
                    dx = centers_new[i, 0] - centers_new[j, 0]
                    dy = centers_new[i, 1] - centers_new[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    if dist < radii_new[i] + radii_new[j] - 1e-8:
                        return np.inf
            return 0

        bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

        # Optimize
        result = opt.minimize(
            objective,
            np.concatenate([flat_centers, flat_radii]),
            method='SLSQP',
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_center_bounds},
                {'type': 'ineq', 'fun': constraint_circle_bounds},
                {'type': 'ineq', 'fun': constraint_non_overlap}
            ],
            options={'maxiter': 100, 'ftol': 1e-8}
        )

        if not result.success:
            continue

        centers_opt = result.x[:2*n].reshape(n, 2)
        radii_opt = result.x[2*n:]
        sum_radii = -result.fun

        # Validate and update best
        valid, msg = validate_packing(centers_opt, radii_opt)
        if valid and sum_radii > best_sum:
            best_centers = centers_opt
            best_radii = radii_opt
            best_sum = sum_radii

    return best_centers, best_radii, best_sum
