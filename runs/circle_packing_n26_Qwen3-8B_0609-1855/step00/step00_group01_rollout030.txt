from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import BoundsError

def run_packing():
    # Initial guess: hexagonal packing with some adjustment for square boundary
    num_circles = 26
    cols = int(np.ceil(np.sqrt(num_circles)))
    rows = int(np.ceil(num_circles / cols))

    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        col = i % cols
        row = i // cols
        x = col + 0.5 * (row % 2)  # Offset for hexagonal packing
        y = row * (np.sqrt(3) / 2)  # Height of a hexagon
        centers[i] = [x, y]
    centers = centers - np.min(centers, axis=0)  # Shift to fit in [0,1]^2
    centers = centers / np.max(centers, axis=0)  # Normalize to unit square

    # Initial radii: uniform
    radii = np.ones(num_circles) * 0.05

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints: circle within square, no overlap
    def constraint1(radii):
        return np.array([r for r in radii if r < 0])  # Negative radii

    def constraint2(radii, centers):

        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if (x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1):
                return np.inf, f"Circle {i} is outside the square"
        return 0

    def constraint3(radii, centers):
        # Check no overlaps
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx * dx + dy * dy)
                if distance < radii[i] + radii[j] - 1e-12:
                    return np.inf, f"Circles {i} and {j} overlap"
        return 0

    # Convert to scipy optimization format
    bounds = Bounds(0, 1)  # Radii can't be negative or exceed 1

    result = minimize(
