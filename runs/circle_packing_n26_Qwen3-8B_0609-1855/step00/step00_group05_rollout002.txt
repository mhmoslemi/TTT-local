def generate_hexagonal_layout(num_circles, square_size=1.0):

    # Returns (centers, radii)
    centers = []
    radii = []
    
    # Calculate number of rows and columns
    num_rows = int(np.ceil(math.sqrt(num_circles)))
    num_cols = int(np.ceil(num_circles / num_rows))
    
    # Calculate spacing between centers
    spacing = square_size / (num_cols - 1) if num_cols > 1 else 1.0
    spacing_y = square_size / (num_rows - 1) if num_rows > 1 else 1.0
    
    # Generate centers in a hexagonal pattern
    for row in range(num_rows):
        for col in range(num_cols):
            # Hexagonal offset
            x = col * spacing
            y = row * spacing_y
            if row % 2 == 1:
                x += spacing / 2  # shift for hexagonal pattern
            
            # Place in square
            x = min(max(x, 0), square_size)
            y = min(max(y, 0), square_size)

            # This will be refined in optimization
            r = spacing / (2 * np.sqrt(3)) if num_rows > 1 else spacing / 2

            radii.append(r)
    
    return np.array(centers), np.array(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
