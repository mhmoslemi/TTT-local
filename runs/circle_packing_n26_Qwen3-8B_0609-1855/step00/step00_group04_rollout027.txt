import numpy as np
import math
import random
from scipy.optimize import minimize
from typing import Tuple, Any, List, Union

def initial_hexagonal_packing(n: int) -> Tuple[np.ndarray, np.ndarray]:
    # Hexagonal grid pattern
    rows = int(math.ceil(n / 3))
    cols = int(math.ceil(n / rows))
    
    centers = []
    radii = []
    
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)
            y = i + 0.5
            if i % 2 == 1:
                x += 0.5
            
            if len(centers) < n:
                centers.append([x, y])
                radii.append(1.0 / (2 * math.sqrt(3)))
    
    # Trim if necessary
    centers = np.array(centers, dtype=np.float64)
    radii = np.array(radii, dtype=np.float64)
    
    # Scale down slightly to ensure fitting in square
    scale = 0.9
    centers *= scale
    radii *= scale
    
    return centers, radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def objective_function(x: np.ndarray, n_circles: int = 26) -> float:
    # x is a 1D array with shape (26 * 3) = [x1, y1, r1, x2, y2, r2, ...]
    centers = x.reshape((n_circles, 2))
    radii = x[n_circles:].reshape(n_circles)
    
    # Ensure non-negative radii
    radii = np.maximum(radii, 0)
    
    # Validate constraints
    valid, _ = validate_packing(centers, radii)
    if not valid:
        return -np.inf  # Penalize invalid configurations
    
    return np.sum(radii)

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n_circles = 26
    initial_centers, initial_radii = initial_hexagonal_packing(n_circles)
    initial_x = np.concatenate([initial_centers.flatten(), initial_radii])
    
    # Optimize using scipy's minimize
    result = minimize(
        objective_function,
        initial_x,
        method="L-BFGS-B",
        bounds=[(0, 1) for _ in range(2 * n_circles)] + [(0, 1) for _ in range(n_circles)]
    )
    
    # Extract the best solution
    x_best = result.x
    centers_best = x_best[:2 * n_circles].reshape((n_circles, 2))
    radii_best = x_best[2 * n_circles:].reshape(n_circles)
    
    # Final validation to ensure the best solution is valid
    valid, _ = validate_packing(centers_best, radii_best)
    if not valid:
        return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
    
    sum_radii = np.sum(radii_best)
    return centers_best, radii_best, sum_radii
