def run_packing():
    # Initial layout: hexagonal grid
    # Number of rows and columns to fit ~26 circles
    # Use a hexagonal grid with rows offset by half circle diameter
    # Start with some initial guess for spacing and radii

    n = 26

    # Start with a hexagonal grid placement
    # Let's distribute them in 5 rows, with 5, 6, 6, 6, 5 circles
    # The grid will be slightly more compact than square
    row_height = 0.5  # Vertical spacing between rows
    col_width = 0.4   # Horizontal spacing between columns

    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initialize with a hexagonal grid
    for i in range(n):
        row = i // 6
        col = i % 6
        x = col * col_width
        y = row * row_height

            x += col_width / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = 0.05  # Initial small radius

    # While ensuring no overlap and all circles within the square

    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize sum

    # Constraints

        # For all i < j

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
