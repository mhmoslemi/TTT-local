import numpy as np
import cvxpy as cp

def run_packing():
    n = 26
    # Define the optimization variables
    centers = cp.Variable((n, 2))
    radii = cp.Variable(n, non_negative=True)

    # Objective: maximize sum of radii
    objective = cp.Maximize(cp.sum(radii))

    # Constraints
    constraints = []

    # Boundary constraints for each circle
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        constraints.append(x - r >= -1e-12)
        constraints.append(x + r <= 1 + 1e-12)
        constraints.append(y - r >= -1e-12)
        constraints.append(y + r <= 1 + 1e-12)

    # Non-overlap constraints for all pairs of circles
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.norm(centers[i] - centers[j])
            constraints.append(dist >= radii[i] + radii[j] - 1e-12)

    # Solve the optimization problem
    problem = cp.Problem(objective, constraints)
    try:
        problem.solve(solver=cp.SCS, max_num_iter=1000, eps=1e-6)
    except:

def generate_hexagonal_layout(N):
    # Hexagonal grid in unit square
    rows = int(np.sqrt(N))
    cols = int(np.ceil(N / rows))
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            if x + 0.5 > 1 or y + 1 > 1:
                continue
            centers.append([x, y])
    # Adjust to fit exactly N circles
    while len(centers) < N:
        x = np.random.rand()
        y = np.random.rand()
        centers.append([x, y])
    return np.array(centers)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
