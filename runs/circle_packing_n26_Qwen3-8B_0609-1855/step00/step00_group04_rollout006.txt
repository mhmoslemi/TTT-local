import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial guess for center positions based on hexagonal packing
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    
    centers = np.zeros((num_circles, 2))
    radius_guess = 0.02  # Initial guess for radius
   
    # Place circles in a hexagonal grid
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col + 0.5 * (row % 2)
        y = row * np.sqrt(3) / 2
        if y > 1 - 1e-6:
            # Adjust if circle would exceed square
            y = 1 - 1e-6
        centers[i] = np.array([x, y])
    
    # Initial guess for radii (we'll optimize this)
    radii = np.full(num_circles, radius_guess)

    # Parameters for optimization
    tol = 1e-6
    maxiter = 1000

    # Optimize the radii and positions
    def objective(x):
        # x contains all parameters: first 2*num_circles are centers, next num_circles are radii
        centers_flat = x[:2*num_circles].reshape(num_circles, 2)
        radii_flat = x[2*num_circles:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraint_overlap(x):
        centers_flat = x[:2*num_circles].reshape(num_circles, 2)
        radii_flat = x[2*num_circles:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1  # Violation of non-overlap constraint
        return 0

    def constraint_bounds(x):
        centers_flat = x[:2*num_circles].reshape(num_circles, 2)
        for i in range(num_circles):
            if not (0 - 1e-8 <= centers_flat[i, 0] <= 1 + 1e-8 and
                    0 - 1e-8 <= centers_flat[i, 1] <= 1 + 1e-8):
                return -1
        return 0

        bounds.append((0, 1))  # radii

    # Optimization
    result = optimize.minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_overlap},
            {'type': 'ineq', 'fun': constraint_bounds}

        maxiter=maxiter

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
