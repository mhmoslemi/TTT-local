import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid layout with 5x5 grid (25 circles) and an extra circle
    grid_size = 5
    grid = np.zeros((grid_size, grid_size, 2))
    
    # Regular grid with spacing
    dx = 1.0 / (grid_size + 1)
    dy = 1.0 / (grid_size + 1)

        for j in range(grid_size):
            grid[i, j, 0] = (i + 1) * dx
            grid[i, j, 1] = (j + 1) * dy

    extra_center = np.array([0.05, 0.05])
    grid = np.vstack((grid.reshape(-1, 2), extra_center))

    radii = np.full(26, 0.05)  # Initial radius

    def objective(radii_flat):
        return -np.sum(radii_flat)  # Negative for minimization

    def constraints(radii_flat):
        # Convert to array
        radii = radii_flat.reshape(-1)
        centers = grid  # Use the initial grid layout as centers (can be adjusted)
        constraints = []

        constraints.append({'type': 'ineq', 'fun': lambda x: x})

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
