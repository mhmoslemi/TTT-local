import numpy as np
from scipy.optimize import minimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial configuration: hexagonal packing with adjusted spacing for square
    n = 26
    # Start with a 5x6 grid (5 rows, 6 columns), slightly spaced for square
    x = np.linspace(0.1, 0.9, 6)
    y = np.linspace(0.1, 0.9, 5)
    centers = np.array([[x[i], y[j]] for j in range(5) for i in range(6)])
    
    # Initial guess for radii: uniform scaling to fit in square
    # Estimate a rough radius assuming hexagonal packing
    # The hexagonal spacing is the same as radius in packed configuration
    # Since we fit to square, radius will be less than 0.5
    initial_radii = np.full(n, 0.15)
    
    # Define the optimization problem
    def objective(params, n):
        # params includes positions (2n) and radii (n)
        # Split params into positions and radii
        flat_params = params.reshape(-1)
        positions = flat_params[:2*n].reshape(n, 2)
        radii = flat_params[2*n:].reshape(n)
        
        # Compute total sum of radii
        total_radius = np.sum(radii)
        
        # Penalize boundary violations
        boundary_penalty = 0
        for i in range(n):
            x, y = positions[i]
            r = radii[i]
            if x - r < -1e-9 or x + r > 1 + 1e-9 or y - r < -1e-9 or y + r > 1 + 1e-9:
                boundary_penalty += 1000000 * (max(0, (x - r + 1e-9)) + max(0, (x + r - 1 + 1e-9)) + max(0, (y - r + 1e-9)) + max(0, (y + r - 1 + 1e-9)))
        
        # Penalize circle-circle overlaps
        overlap_penalty = 0
        for i in range(n):
            for j in range(i+1, n):
                dx = positions[i][0] - positions[j][0]
                dy = positions[i][1] - positions[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-9:
                    overlap_penalty += 1000000 * (radii[i] + radii[j] - dist)
        
        # Total objective (minimize penalty, maximize radii)
        return -(total_radius - 0.1 * (boundary_penalty + overlap_penalty))
    
    # Initial guess
    initial_params = np.concatenate([centers.reshape(-1), initial_radii])
    
    # Optimization bounds: positions between 0 and 1, radii between 0 and 0.5
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 0.5) for _ in range(n)]
    
    # Optimization with L-BFGS-B (bounded)
    result = minimize(
        lambda params: objective(params, n),
        initial_params,
        method='L-BFGS-B',
        bounds=bounds,
        options={'maxiter': 200, 'ftol': 1e-8}
    )
    
    # Extract final positions and radii
    final_params = result.x
    final_positions = final_params[:2*n].reshape(n, 2)
    final_radii = final_params[2*n:].reshape(n)
    
    # Final sum of radii
    final_sum = np.sum(final_radii)
    
    # Return the solution
    return final_positions, final_radii, final_sum
