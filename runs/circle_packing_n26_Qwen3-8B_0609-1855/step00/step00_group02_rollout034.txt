import scipy.optimize as optimize

def run_packing():

    dy = 1.0 / (row_count - 1)

    centers = np.zeros((n, 2))
    idx = 0

        for j in range(col_count):
            x = j * dx
            y = i * dy
            centers[idx] = [x, y]
            idx += 1

    # Initial radius assuming circles just touch their neighbors
    r_initial = 1.0 / (2 * np.sqrt(3))  # For hexagonal grid

        return -np.sum(radii)

    # Constraints:
    # 1. Circle centers within [0,1]^2
    # 2. Circle radii non-negative
    # 3. No overlapping between circles
    # 4. Circle edges within [0,1]^2

    def constraints(radii, centers):
        constraints = []

        # Constraint 1: Radii >= 0
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})

        # Constraint 2: Circle edge within square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y, r=r: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y, r=r: x + r - 1.0})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y, r=r: y - r})

                    dist = np.sqrt(dx ** 2 + dy ** 2)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
