import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Number of circles
    n = 26

    # We start with a hexagonal grid that fits within the unit square
    # This is a rough guess, will be optimized later
    centers = np.zeros((n, 2))
    radii = np.ones(n) * 0.05  # Initial guess for radii

    # Generate a hexagonal grid
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= n:
                break
            x = j + 0.5 * i
            y = i
            # Normalize to fit within the unit square
            x = x / (cols - 1)
            y = y / (rows - 1)
            centers[idx] = [x, y]

    # Define the objective function to maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize the negative sum of radii

    # Define the constraints
    def constraint_center_bounds(x):
        # x[0::2] are center x-coordinates, x[1::2] are center y-coordinates
        # x[2*i] = x_i, x[2*i+1] = r_i
        centers = x[0::2].reshape(-1, 2)
        radii = x[1::2]
        return np.concatenate([
            centers[:, 0] - radii + 1e-12,
            1 - centers[:, 0] - radii + 1e-12,
            centers[:, 1] - radii + 1e-12,
            1 - centers[:, 1] - radii + 1e-12
        ])

    def constraint_circle_overlap(x):
        # x[0::2] are center x-coordinates, x[1::2] are center y-coordinates
        # x[2*i] = x_i, x[2*i+1] = r_i
        centers = x[0::2].reshape(-1, 2)
        radii = x[1::2]
        distances = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return distances - (radii[:, np.newaxis] + radii[np.newaxis, :]) + 1e-12

    # Initial guess: centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: x-coordinates between [0,1], y-coordinates between [0,1], radii >= 0
    bounds = [(0, 1) for _ in range(n * 2)]
    for i in range(n):
        bounds[2 * i + 1] = (0, 1)  # Radii must be non-negative (we're maximizing, so lower bound is 0)

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_bounds},
        {'type': 'ineq', 'fun': constraint_circle_overlap}
    ]

    # Optimization
    result = optimize.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract optimized centers and radii
    centers_opt = result.x[0::2].reshape(-1, 2)

    if not valid:
        raise ValueError(f"Invalid packing: {msg}")

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
