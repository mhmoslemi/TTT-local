import scipy.optimize as opt
import matplotlib.pyplot as plt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal grid layout
    num_circles = 26
    grid_size = 4
    radius_initial = 0.04  # Starting radius guess
    radii = np.full(num_circles, radius_initial)
    
    # Generate initial centers using hexagonal grid in a unit square
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // grid_size
        col = i % grid_size
        x = col / (grid_size - 1) if grid_size > 1 else 0.5
        y = row / (grid_size - 1) if grid_size > 1 else 0.5
        centers[i] = [x, y]
    
    # Optimization function to maximize the sum of radii
    def objective(radii_vec):
        # Make sure radii are non-negative
        radii_vec = np.clip(radii_vec, 0, None)
        return -np.sum(radii_vec)  # Negative because we want to minimize the negative sum

    # Constraints: circle does not overlap with others and stays in bounds
    def constraint_overlap(i, j, radii_vec):
        def cons(radii_vec):
            r_i = radii_vec[i]
            r_j = radii_vec[j]
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < r_i + r_j - 1e-12:
                return r_i + r_j - dist - 1e-12
            return 0
        return cons

    def constraint_boundary(i, radii_vec):
        r = radii_vec[i]
        x, y = centers[i]
        if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
            return r - (1 - np.max([x - r, y - r, 1 - (x + r), 1 - (y + r)])) - 1e-12
        return 0

    constraints = []
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: constraint_overlap(i, j, r)})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_boundary(i, r)})

    # Initial guess
    initial_guess = radii.copy()

    bounds = [(0, None) for _ in range(num_circles)]

    return final_centers, final_radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
