import scipy.optimize as optimize

def run_packing():

    n = 26
    centers = np.zeros((n, 2))
    # Place circles in a grid pattern
    for i in range(n):

        y = i // 5 + 0.1  # 5 rows
        centers[i] = [x, y]

    # Estimate initial radii based on spacing
    radii = np.zeros(n)
    for i in range(n):
        dist = np.inf
        for j in range(n):
            if i != j:
                dist = min(dist, np.linalg.norm(centers[i] - centers[j]))
        radii[i] = dist / 2.0

    # Define objective function to maximize sum of radii
    def objective(x):
        # Convert x to centers and radii

        radii_flat = x[n*2:].reshape(n)
        # Compute sum of radii
        return -np.sum(radii_flat)

        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        for i in range(n):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return 1.0  # Not satisfied
        return 0.0  # Satisfied

    def constraint_overlap(x):
        centers_flat = x[:n*2].reshape(n, 2)
