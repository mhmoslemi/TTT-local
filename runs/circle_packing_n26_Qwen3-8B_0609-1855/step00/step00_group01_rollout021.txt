import numpy as np
import scipy.optimize as optimize
import math

def generate_hexagonal_grid(n, spacing):
    centers = []
    for i in range(n):
        row = i // 2
        col = i % 2
        x = col * spacing
        y = row * (np.sqrt(3) / 2) * spacing
        centers.append([x, y])
    return np.array(centers)

def objective_function(params, n):
    # params is a flat array of [x1, y1, r1, x2, y2, r2, ..., xn, yn, rn]
    centers = params[:2*n].reshape((n, 2))
    radii = params[2*n:]
    
    # Initialize loss
    loss = 0.0
    # Penalize boundary violation
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
            loss += 1e6  # Large penalty for boundary violations
    
    # Penalize overlaps
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = math.hypot(dx, dy)
            if dist < radii[i] + radii[j] - 1e-12:
                loss += 1e6  # Large penalty for overlaps
    
    # Penalize negative radii
    for r in radii:
        if r < 0:
            loss += 1e6
    
    return loss

def run_packing():
    n = 26
    initial_spacing = 0.2  # Initial spacing to fit into the unit square
    initial_centers = generate_hexagonal_grid(n, initial_spacing)
    initial_radii = np.full(n, 0.1)  # Initial guess for radii
    
    # Flatten the parameters

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
