import cvxpy as cp

def run_packing():

    centers = np.zeros((n, 2))

    row_spacing = 0.5  # vertical spacing between rows
    col_spacing = 0.5  # horizontal spacing between columns

    def generate_hex_grid(num_rows, num_cols, row_spacing, col_spacing):
        grid = []
        for i in range(num_rows):
            for j in range(num_cols):
                x = j * col_spacing
                y = i * row_spacing
                grid.append((x, y))
        return np.array(grid)

    grid = generate_hex_grid(6, 5, row_spacing, col_spacing)
