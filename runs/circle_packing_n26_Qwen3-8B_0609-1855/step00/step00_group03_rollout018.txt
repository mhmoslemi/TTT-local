        centers[i] = [x, y]
    
    # Initial guess for radii: same as the grid spacing
    radii = np.full(n, 0.15)

    def cost_function(x):

        Cost function to minimize - sum of radii (to be maximized).

        """

        x_flat = x.copy()

        centers_x = x_flat[:, 0]
        centers_y = x_flat[:, 1]
        radii = x_flat[:, 2]

        if np.any(radii < 0):
            return np.inf

        for i in range(n):
            x, y, r = centers_x[i], centers_y[i], radii[i]
            if x - r < -1e-6 or x + r > 1 + 1e-6 or y - r < -1e-6 or y + r > 1 + 1e-6:
                return np.inf

        # Compute sum of radii
        total_radius = np.sum(radii)
