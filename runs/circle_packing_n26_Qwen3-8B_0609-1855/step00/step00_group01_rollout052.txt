import numpy as np
import scipy.optimize as optimize

def run_packing():

    n = 26

    # Initial hexagonal grid configuration
    def initial_grid(n):
        # Hexagonal grid with spacing
        spacing = 0.1
        rows = int(np.ceil(np.sqrt(n)))
        cols = int(np.ceil(np.sqrt(n)))
        centers = []
        for i in range(rows):
            for j in range(cols):
                x = j * spacing + 0.5 * spacing
                y = i * spacing * np.sqrt(3)
                centers.append((x, y))
        # Trim to exactly n circles
        centers = centers[:n]
        return np.array(centers)

    centers = initial_grid(n)
    radii = np.ones(n) * 0.05

        return -np.sum(radii)

    def constraint_boundary(radii):
        return np.sum(radii) - 2.636  # We will try to maximize this value

    def constraint_non_overlap(radii, centers):

        return np.sum(dist - (radii[:, np.newaxis] + radii[np.newaxis, :])) - 1e-12

    # Setup optimization
    bounds = [(0, 0.1) for _ in range(n)]  # Radii are between 0 and 0.1
    constraints = [

        {'type': 'ineq', 'fun': lambda r: constraint_non_overlap(r, centers)}
    ]

    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,

    return optimal_centers, optimal_radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
