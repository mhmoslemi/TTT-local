import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint
from scipy.optimize import NonlinearConstraint
from scipy.optimize import minimize

def run_packing():
    # Initial configuration: place 26 circles in a grid-like pattern with hexagonal spacing
    # First, define a grid with rows and columns
    num_circles = 26
    cols = int(np.sqrt(num_circles)) + 1
    rows = (num_circles + cols - 1) // cols

    # Initial centers and radii
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Place them in a hexagonal grid
    for i in range(num_circles):
        col = i % cols
        row = i // cols
        x = col + 0.5 * (row % 2)
        y = row * np.sqrt(3) / 2
        centers[i] = [x / (cols + 1), y / (rows + 1)]
        radii[i] = 0.01  # Initial small radius, will be optimized

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints: circles must not overlap
    def constraint_1(x):
        # x is flattened radii vector
        radii = x
        # Check circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance_sq = dx**2 + dy**2
                if distance_sq < (radii[i] + radii[j])**2 - 1e-12:
                    return np.inf  # Penalize overlap
        # Check circles are inside the unit square
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if not (0 - r <= x <= 1 + r and 0 - r <= y <= 1 + r):
                return np.inf  # Penalize circles outside the square
        return 0  # No penalty

    # Set up optimization problem
    bounds = Bounds(0, np.inf, shape=num_circles)  # Radii must be non-negative
    cons = NonlinearConstraint(constraint_1, 0, 0, jac=None)

    # Run optimization
    result = minimize(
        objective,
        radii,
        method="SLSQP",
        bounds=bounds,
        constraints=cons,
        options={"ftol": 1e-8, "maxiter": 1000, "disp": False}
    )

    # Extract optimal radii and centers
    optimal_radii = result.x
    sum_radii = np.sum(optimal_radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
