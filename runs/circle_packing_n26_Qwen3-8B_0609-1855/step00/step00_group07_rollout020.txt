import numpy as np
import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    # Number of circles
    n = 26

    # Initial guess for center positions (hexagonal packing)
    # We'll use a hexagonal grid with some spacing
    grid_size = 5
    spacing = 0.25
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // grid_size
        col = i % grid_size
        x = col * spacing + 0.25 * (1 - col % 2)
        y = row * spacing * np.sqrt(3) + 0.25
        centers[i] = [x, y]

    # Initial guess for radii (equal for all)
    radii = np.full(n, 0.05)

    # Function to compute constraints
    def constraints(x):

        return [valid]

    # Convert centers and radii to a flat array for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimize using SLSQP
    bounds = []
    for i in range(n):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])
        bounds.extend([(0.0, 0.1)])  # Limit radii to avoid overlapping

    result = opt.minimize(

        x0,
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        method='SLSQP',
        tol=1e-6

    # Extract optimized centers and radii
    optimized_centers = result.x[:n*2].reshape(n, 2)
    optimized_radii = result.x[n*2:]

    # Return the result
    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
