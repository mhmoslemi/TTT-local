import scipy.optimize as optimize

def run_packing():

    radii = np.full(n_circles, 0.05)  # Initial guess for radii

    # The packing will be arranged in rows with increasing number of circles

        max_rows += 1

    for row in range(max_rows):

    centers = centers / np.max(centers) * 1.0

    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    constraints = []

    # Constraint: All circles within the unit square
    def in_bounds(radii, centers):
        for i in range(n_circles):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:

        return True

    def no_overlap(radii, centers):
        for i in range(n_circles):

                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

    # Constraint wrapper for scipy.optimize

        if not in_bounds(radii, centers):
            return np.inf

            return np.inf
        return 0

    bounds = [(0.0, 1.0) for _ in range(n_circles)]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
