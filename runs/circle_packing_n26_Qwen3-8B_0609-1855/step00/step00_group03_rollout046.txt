import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

            if dist < radii[i] + radii[j] - 1e-12:

    return True, "ok"

def initial_circle_centers(n):
    """
    Generate an initial layout of circle centers using hexagonal packing.
    """
    # Create a hexagonal grid with 4 rows, each with 7 circles (adjustable based on n)
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(rows):
        for c in range(cols):
            if idx >= n:
                break
            x = c + 0.5 * (r % 2)
            y = r * np.sqrt(3) / 2
            centers[idx] = [x, y]
            idx += 1
    return centers

def objective_function(params, n):
    """
    Objective function to maximize the sum of radii.
    """
    centers = params[:n * 2].reshape(n, 2)
    radii = params[n * 2:]
    return -np.sum(radii)  # We minimize the negative sum to maximize the sum

def constraint_square_inside(params, n):
    """
    Constraint: All circles must lie within the unit square.
    """
    centers = params[:n * 2].reshape(n, 2)
    radii = params[n * 2:]

        x, y = centers[i]
        r = radii[i]
        if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:

def constraint_no_overlap(params, n):
    """
    Constraint: No two circles can overlap.

    radii = params[n * 2:]
    for i in range(n):

def constraint_non_negative_radii(params, n):
    """
    Constraint: All radii must be non-negative.

    return 0
