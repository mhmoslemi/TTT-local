import scipy.optimize as opt
from scipy.spatial.distance import pdist, squareform
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: grid-based packing with spacing
    num_circles = 26
    grid_rows = 5
    grid_cols = 5
    spacing = 0.2

    # Initial centers (slightly offset for better boundary utilization)
    centers = np.zeros((num_circles, 2))
    idx = 0

        for j in range(grid_cols):

                break
            x = j * spacing + 0.05
            y = i * spacing + 0.05

            idx += 1

    # Initial radii: assume all circles have the same radius for a starting point
    # This will be optimized later
    radii = np.full(num_circles, 0.05)

        return -np.sum(radii)

    def constraints(radii):
        # Constraints: centers must lie within [0,1]^2
        for i in range(num_circles):
            if centers[i, 0] - radii[i] < -1e-12 or centers[i, 0] + radii[i] > 1 + 1e-12:
                return np.inf
            if centers[i, 1] - radii[i] < -1e-12 or centers[i, 1] + radii[i] > 1 + 1e-12:
                return np.inf

        return 0

        objective,
        radii,
        method='SLSQP',

        constraints={'type': 'ineq', 'fun': constraints}

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
