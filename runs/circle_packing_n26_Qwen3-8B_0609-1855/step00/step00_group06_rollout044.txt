import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial configuration: hexagonal packing with small circles
    def create_hexagonal_packing(num_circles, spacing=0.15):
        centers = []
        for i in range(num_circles):
            row = i // 2
            col = i % 2
            x = col * spacing
            y = row * (spacing * np.sqrt(3))
            centers.append([x, y])
        centers = np.array(centers)
        # Normalize to [0,1]^2
        centers = (centers - centers.min(axis=0)) / (centers.max(axis=0) - centers.min(axis=0))
        return centers

    # Initial guess for centers and radii
    centers = create_hexagonal_packing(26)
    radii = np.full(26, 0.05)

    # Objective function to maximize sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Negative because we are minimizing

    # Constraints for non-overlapping and in-boundary
    def constraint_overlap(radii_flat, centers):
        # Convert to numpy array
        radii = radii_flat.reshape(-1)
        # Bound constraints
        for i in range(len(radii)):
            if radii[i] < 0:
                return -1
        # Check boundaries
        for i in range(len(radii)):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return -1
        # Check overlaps
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1
        return 0

    # Define bounds for radii
    bounds = [(0.0, 1.0) for _ in range(26)]

    # Initial guess for radii
    initial_radii = np.full(26, 0.05)

    # Optimization setup
    result = opt.minimize(
        objective,
        initial_radii,
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers)}],
        method='SLSQP'
    )

    # Extract result
    radii_opt = result.x
    centers_opt = centers.copy()
    total_radius = np.sum(radii_opt)

    # Validate the solution
    valid, msg = validate_packing(centers_opt, radii_opt)
