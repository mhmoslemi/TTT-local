import cvxpy as cp

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n_circles = 26
    # Determine rows and columns for hexagonal packing
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))
    # Offset between rows
    row_offset = np.sqrt(3) / 2 * 0.9  # Adjust to fit within the square

        y = row * row_offset

    # Define variables
    r = cp.Variable(n_circles, nonneg=True)
    # Objective: maximize sum of radii
    obj = cp.Maximize(cp.sum(r))
