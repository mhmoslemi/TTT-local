import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    rows = 5
    cols = 5
    r_initial = (1.0 / (np.sqrt(3) * rows)) * 0.9  # Initial radius based on hexagonal packing

    for i in range(num_circles):
        row = i // cols
        col = i % cols
        # Shift every other row
        if row % 2 == 1:
            col += 0.5

        x = col * (2 * r_initial) + r_initial

        x = max(r_initial, min(1.0 - r_initial, x))
        y = max(r_initial, min(1.0 - r_initial, y))

        radii[i] = r_initial

    # Step 2: Optimization setup
    x0 = np.concatenate((centers.flatten(), radii))
    bounds = []

    for i in range(num_circles):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
        bounds.append((0, 1))  # radius

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x[num_circles * 2:])

        return x[2 * i] - centers[i][0], x[2 * i + 1] - centers[i][1]

    def constraint_rad(x, i):
        return x[2 * num_circles + i] - radii[i]

    # Build constraints
    constraints = []
    for i in range(num_circles):
        # Position constraint
        constraints.append({
            'type': 'eq',
            'fun': lambda x, i=i: constraint_pos(x, i),
            'jac': lambda x, i=i: np.zeros_like(x)
        })
        # Radius constraint
        constraints.append({
            'type': 'eq',
            'fun': lambda x, i=i: constraint_rad(x, i),
            'jac': lambda x, i=i: np.zeros_like(x)
        })

    # Optimization with SLSQP
    result = opt.minimize(
        fun=objective,
        x0=x0,
        method='SLSQP',

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
