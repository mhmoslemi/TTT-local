def optimize_packing(centers, radii):
    num_circles = centers.shape[0]

    # Flatten for optimization
    x0 = np.concatenate((centers.flatten(), radii))

    # Objective: Maximize sum of radii
    def objective(x):
        return -np.sum(x[num_circles * 2:])  # Negative for minimization

    # Bounds for centers and radii
    bounds = []
    for i in range(num_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    # Constraints for circle positions and radii
    def constraint_positions(x, i):
        return x[i * 2: i * 2 + 2] - centers[i]

    def constraint_radii(x, i):
        return x[i * 2 + 2] - radii[i]

    # Define non-linear constraints for circle-to-circle overlap and boundary
    def constraint_overlap(x):
        centers_opt = x[:num_circles * 2].reshape((num_circles, 2))
        radii_opt = x[num_circles * 2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_opt[i] + radii_opt[j] - 1e-8:
                    return dist - radii_opt[i] - radii_opt[j] + 1e-8
        return 0.0

    def constraint_boundary(x):
        centers_opt = x[:num_circles * 2].reshape((num_circles, 2))
        radii_opt = x[num_circles * 2:]
        for i in range(num_circles):
            x_pos, y_pos = centers_opt[i]
            r = radii_opt[i]
            if x_pos - r < 0 or x_pos + r > 1 or y_pos - r < 0 or y_pos + r > 1:
                return 1.0
        return 0.0

    # Optimization with SLSQP
    result = opt.minimize(
        fun=objective,
        x0=x0,
        method='SLSQP',
        jac=lambda x: np.zeros_like(x),
        bounds=bounds,
        constraints=[
            {'type': 'eq', 'fun': lambda x, i=i: constraint_positions(x, i), 'args': (i,)} for i in range(num_circles)
        ] + [
            {'type': 'eq', 'fun': lambda x, i=i: constraint_radii(x, i), 'args': (i,)} for i in range(num_circles)
        ] + [
            {'type': 'ineq', 'fun': constraint_overlap},
            {'type': 'ineq', 'fun': constraint_boundary}
        ],
        options={
            'maxiter': 1000,
            'ftol': 1e-8,
            'disp': False
        }
    )

    centers_opt = result.x[:num_circles * 2].reshape((num_circles, 2))
    radii_opt = result.x[num_circles * 2:]

    # Validate the optimized packing

    if not valid:
        print(f"Validation failed: {message}")
        return None

    # Calculate sum of radii
    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
