import numpy as np
import scipy.optimize as opt

def generate_candidate_centers(n: int) -> np.ndarray:
    """
    Generate multiple candidate center configurations to explore different packing arrangements.
    """
    candidates = []
    
    # Base grid-like packing
    grid_centers = np.zeros((n, 2))
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col / (num_cols - 1) if num_cols > 1 else 0.5
        y = row / (num_rows - 1) if num_rows > 1 else 0.5
        grid_centers[i] = [x, y]
    candidates.append(grid_centers)
    
    # Hexagonal packing
    hex_centers = np.zeros((n, 2))
    col_count = int(np.ceil(np.sqrt(n)))
    row_count = int(np.ceil(n / col_count))
    
    for i in range(n):
        col = i % col_count
        row = i // col_count
        col_offset = 0.5 if col % 2 == 1 else 0.0
        x = 0.5 + (col - (col_count - 1) / 2) * (1 / (col_count - 1)) * 0.8 + col_offset
        y = 0.5 + (row - (row_count - 1) / 2) * (1 / (row_count - 1)) * 0.8
        hex_centers[i] = [x, y]
    candidates.append(hex_centers)
    
    # Staggered grid variations
    for _ in range(3):
        staggered_centers = np.zeros((n, 2))
        num_rows = int(np.ceil(np.sqrt(n)))
        num_cols = int(np.ceil(n / num_rows))

            # Add small variations

            y += np.random.uniform(-0.03, 0.03)
        
            staggered_centers[i] = [x, y]

        r = 0.35  # radius for the outer ring

    candidates.append(spiral_centers)
    
    return candidates

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    candidates = generate_candidate_centers(n)
    
    best_centers = None
    best_radii = None
    best_sum_radii = 0.0
    
    for centers in candidates:
        # Initial guess for radii based on spacing
        distances = np.zeros(n)
        for i in range(n):
            dist = np.inf
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = min(dist, np.sqrt(dx*dx + dy*dy))
            distances[i] = dist
        
        # Calculate initial radii based on the distances
        initial_radii = np.zeros(n)
        for i in range(n):
            initial_radii[i] = distances[i] / 2
        
        # Optimization bounds
        bounds = [(0.0, 1.0) for _ in range(n * 2)]  # For x and y coordinates
        radius_bounds = [(0.0, 0.2) for _ in range(n)]  # Radii allow some flexibility
        
        # Objective function: maximize sum of radii
        def objective(x):
            centers = x[:n*2].reshape(n, 2)
            radii = x[n*2:]
            return -np.sum(radii)  # Negative for minimization
        
        # Constraints: ensure each circle is within the unit square
        def square_constraints(x):
            centers = x[:n*2].reshape(n, 2)
            radii = x[n*2:]
            return np.concatenate([
                centers[:, 0] - radii + 1e-12,  # x - r >= 0
                1 - centers[:, 0] - radii + 1e-12,  # 1 - x - r >= 0
                centers[:, 1] - radii + 1e-12,  # y - r >= 0
                1 - centers[:, 1] - radii + 1e-12  # 1 - y - r >= 0
            ])
        
        def circle_overlap_constraints(x):
            centers = x[:n*2].reshape(n, 2)
            radii = x[n*2:]
            constraints = []
            for i in range(n):
                for j in range(i + 1, n):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    constraints.append(dist - radii[i] - radii[j] + 1e-12)
            return constraints
        
        # Combine all constraints
        cons = [
            {'type': 'ineq', 'fun': square_constraints},
            {'type': 'ineq', 'fun': circle_overlap_constraints}
        ]
        
        # Initial guess
        x0 = np.concatenate([centers.flatten(), initial_radii])
        
        # Optimize under constraints
        result = opt.minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=[*bounds, *radius_bounds],
            constraints=cons,
            tol=1e-10,
            options={'maxiter': 5000, 'ftol': 1e-10, 'eps': 1e-8, 'disp': False}
        )
        
        if not result.success:
            continue

            best_radii = radii_opt
            best_sum_radii = sum_radii
    
    if best_centers is None:
        raise RuntimeError("No valid packing found")
    
    return best_centers, best_radii, best_sum_radii
