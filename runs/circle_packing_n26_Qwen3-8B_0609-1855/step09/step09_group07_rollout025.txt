    Generate centers for a grid-like packing with optimized spacing for boundaries.
    """

    num_cols = int(np.ceil(n / num_rows))
    
    centers = np.zeros((n, 2))

        row = i // num_cols
        col = i % num_cols

        h_spacing = 1.0 / (num_cols - 1)

    return centers
