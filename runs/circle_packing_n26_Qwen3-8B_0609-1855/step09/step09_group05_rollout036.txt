def generate_initial_centers(n: int) -> np.ndarray:
    """
    Generate a structured initial configuration of centers based on a hexagonal grid with boundary consideration.
    """
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        
        # Calculate x and y positions using hexagonal grid
        x = 0.5 + (col - (num_cols - 1) / 2) * (1 / (num_cols - 1)) * 0.8
        y = 0.5 + (row - (num_rows - 1) / 2) * (1 / (num_rows - 1)) * 0.8

            y -= 0.05

        if col == 0:

            y += 0.1

        x += np.random.uniform(-0.03, 0.03)

        centers[i] = [x, y]
    
    return centers
