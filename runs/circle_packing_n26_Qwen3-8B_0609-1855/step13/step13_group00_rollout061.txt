def generate_hexagonal_centers(n: int) -> np.ndarray:
    """
    Generate a hexagonal grid center configuration for n circles.
    """
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = (col + 0.5) * (1 / num_cols)  # Adjust for hexagonal spacing
        y = (row + 0.5) * (1 / num_rows)
        centers[i] = [x, y]
    return centers
