import numpy as np
import scipy.optimize as opt
from typing import Tuple

def generate_initial_centers_hexagonal(n: int) -> np.ndarray:
    """
    Generate a hexagonal initial configuration of centers with boundary consideration.
    """
    if n == 0:
        return np.array([])
    
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        
        x = 0.5 + (col - (cols - 1) / 2) * (1 / (cols - 1)) * 0.9
        y = 0.5 + (row - (rows - 1) / 2) * (1 / (rows - 1)) * 0.9
        
        if row % 2 == 0:
            y -= 0.05
        
        if col == 0:
            x -= 0.1
        elif col == cols - 1:
            x += 0.1
        if row == 0:
            y -= 0.1
        elif row == rows - 1:
            y += 0.1
        
        x += np.random.uniform(-0.01, 0.01)
        y += np.random.uniform(-0.01, 0.01)
        
        centers[i] = [x, y]
    
    return centers

def generate_initial_centers_nested(n: int) -> np.ndarray:
    """
    Generate a nested initial configuration of centers with boundary consideration.
    """
    if n == 0:
        return np.array([])
    
    layers = int(np.ceil(np.sqrt(n)))
    centers = np.zeros((n, 2))
    for i in range(n):
        layer = i // (layers ** 2)
        pos_in_layer = i % (layers ** 2)
        
        r = 0.5 + (layer / (layers - 1)) * 0.45
        theta = 2 * np.pi * pos_in_layer / (layers ** 2)
        
        x = 0.5 + r * np.cos(theta)
        y = 0.5 + r * np.sin(theta)
        
        if theta < np.pi / 2:
            x -= 0.1
        elif theta < np.pi:
            y -= 0.1
        elif theta < 3 * np.pi / 2:
            x += 0.1
        else:
            y += 0.1
        
        x += np.random.uniform(-0.01, 0.01)
        y += np.random.uniform(-0.01, 0.01)
        
        centers[i] = [x, y]
    
    return centers

def generate_initial_centers_random(n: int) -> np.ndarray:
    """
    Generate a random initial configuration of centers with boundary consideration.
    """
    if n == 0:
        return np.array([])
    
    centers = np.zeros((n, 2))
    for i in range(n):
        x = np.random.uniform(0.1, 0.9)
        y = np.random.uniform(0.1, 0.9)
        x += np.random.uniform(-0.01, 0.01)
        y += np.random.uniform(-0.01, 0.01)
        centers[i] = [x, y]
    
    return centers

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26
    best_centers = None
    best_radii = None
    best_sum_radii = 0.0

    bounds = [(0.0, 1.0) for _ in range(n * 2)]
    radius_bounds = [(0.0, 0.25) for _ in range(n)]

    def objective(x):
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:]
        return -np.sum(radii)  # Negative for minimization

    def square_constraints(x):
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:]
        return np.concatenate([
            centers[:, 0] - radii + 1e-12,  # x - r >= 0
            1 - centers[:, 0] - radii + 1e-12,  # 1 - x - r >= 0
            centers[:, 1] - radii + 1e-12,  # y - r >= 0
            1 - centers[:, 1] - radii + 1e-12  # 1 - y - r >= 0
        ])

    def circle_overlap_constraints(x):
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:]
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - radii[i] - radii[j] + 1e-12)
        return constraints

    cons = [
        {'type': 'ineq', 'fun': square_constraints},
        {'type': 'ineq', 'fun': circle_overlap_constraints}
    ]

    for _ in range(100):
        # Try multiple initial configurations
        initial_centers = generate_initial_centers_hexagonal(n)
        # initial_centers = generate_initial_centers_nested(n)

        initial_radii = np.full(n, 0.095)
        x0 = np.concatenate([initial_centers.flatten(), initial_radii])

        result = opt.minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=[*bounds, *radius_bounds],
            constraints=cons,
            tol=1e-10,
            options={'maxiter': 10000, 'ftol': 1e-10, 'eps': 1e-8, 'disp': False}
        )

        if not result.success:

            best_radii = radii_opt
            best_sum_radii = sum_radii

    if best_centers is None:
        raise RuntimeError("No valid packing found")

    return best_centers, best_radii, best_sum_radii
