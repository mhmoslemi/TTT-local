def generate_initial_centers(n: int) -> np.ndarray:

    Generate an initial configuration of centers using a more optimized hexagonal grid layout.
    """

    max_row = int(np.ceil(n / max_col))
    
    # Generate centers in a hexagonal grid
    centers = np.zeros((n, 2))
    idx = 0

                break

            centers[idx] = [x, y]
            idx += 1
    return centers
