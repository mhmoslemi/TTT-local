def generate_hexagonal_centers(n: int) -> np.ndarray:
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        
        x_center = 0.5 + (col - num_cols / 2) * (1 / (num_cols - 1)) * 0.85
        y_center = 0.5 + (row - num_rows / 2) * (1 / (num_rows - 1)) * 0.85
        
        if col == 0:
            x_center -= 0.1
        elif col == num_cols - 1:
            x_center += 0.1

            y_center -= 0.1
        elif row == num_rows - 1:
            y_center += 0.1
        
        x_center += np.random.uniform(-0.02, 0.02)
        y_center += np.random.uniform(-0.02, 0.02)
        
        centers[i] = [x_center, y_center]
    
    return centers

def generate_spiral_centers(n: int) -> np.ndarray:
    centers = np.zeros((n, 2))
    for i in range(n):
        rad = 0.5 + (i / n) * 0.3
        angle = 2 * np.pi * i / n
        x = 0.5 + rad * np.cos(angle)
        y = 0.5 + rad * np.sin(angle)
        centers[i] = [x, y]
    return centers

def generate_clustered_centers(n: int) -> np.ndarray:

        centers[i] = [x_center, y_center]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j])**2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
