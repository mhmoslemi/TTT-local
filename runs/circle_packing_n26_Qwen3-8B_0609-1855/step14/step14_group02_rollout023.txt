import scipy.optimize as opt

def generate_initial_centers_hex(n: int) -> np.ndarray:
    """Generate a hexagonal pattern with boundary adjustment."""
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        
        x = 0.5 + (col - (num_cols - 1) / 2) * (1 / (num_cols - 1))
        y = 0.5 + (row - (num_rows - 1) / 2) * (1 / (num_rows - 1))
        
        # Adjust to corner and boundary circles
        if row == 0 or row == num_rows - 1:
            y = 0.5 + (row - (num_rows - 1) / 2) * (1 / (num_rows - 1)) * 0.9
        if col == 0 or col == num_cols - 1:
            x = 0.5 + (col - (num_cols - 1) / 2) * (1 / (num_cols - 1)) * 0.9
        
        # Add small random variation for diversity
        x += np.random.uniform(-0.02, 0.02)
        y += np.random.uniform(-0.02, 0.02)
        
        centers[i] = [x, y]
    
    return centers

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    best_centers = None
    best_radii = None
    best_sum_radii = 0.0
    
    def objective(x):
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:]
        return -np.sum(radii)  # Negative for minimization

        centers = x[:n*2].reshape(n, 2)

            centers[:, 0] - radii + 1e-12,  # x - r >= 0

            centers[:, 1] - radii + 1e-12,  # y - r >= 0

        ])

    def circle_overlap_constraints(x):
        centers = x[:n*2].reshape(n, 2)
