import scipy.optimize as opt
from typing import Tuple

def generate_initial_centers_hexagonal(n: int) -> np.ndarray:
    """
    Generate a hexagonal initial configuration of centers with boundary consideration.
    """
    if n == 0:
        return np.array([])
    
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        
        x = 0.5 + (col - (cols - 1) / 2) * (1 / (cols - 1)) * 0.9
        y = 0.5 + (row - (rows - 1) / 2) * (1 / (rows - 1)) * 0.9

            y -= 0.05

            x += 0.1

            y -= 0.1
        elif row == rows - 1:
            y += 0.1
        
        x += np.random.uniform(-0.01, 0.01)
        y += np.random.uniform(-0.01, 0.01)
        
        centers[i] = [x, y]
    
    return centers

def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"
