def generate_hexagonal_initial_centers(n: int) -> np.ndarray:

        row = i // num_cols
        col = i % num_cols

        y = 0.5 + (row - (num_rows - 1)/2) * (1 / (num_rows - 1)) * 0.9

        x += np.random.uniform(-0.03, 0.03)
        y += np.random.uniform(-0.03, 0.03)

    return centers
