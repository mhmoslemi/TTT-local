def generate_hexagonal_centers(n: int) -> np.ndarray:

    Generate a hexagonal packing layout for n circles within the unit square.
    """

    cols = int(np.ceil(n / rows))
    spacing = 1.0 / (cols - 1)
    hex_spacing = spacing * np.sqrt(3) / 2
    centers = np.zeros((n, 2))
    count = 0
    for i in range(rows):

            if i % 2 == 0:

            centers[count] = [x, y]
            count += 1
            if count == n:
                break
        if count == n:
            break
    return centers
