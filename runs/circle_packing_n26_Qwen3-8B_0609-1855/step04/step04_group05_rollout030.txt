def compute_hybrid_hexagonal_centers(n: int) -> np.ndarray:

    Generate a hybrid hexagonal packing with better boundary and corner placement.
    """
    centers = np.zeros((n, 2))
    col_per_row = int(np.ceil(np.sqrt(n)))
    row_per_col = int(np.ceil(n / col_per_row))
    
    for i in range(n):
        row = i // col_per_row
        col = i % col_per_row
        
        # Hexagonal grid with spacing
        spacing = 1 / (row_per_col + 1)  # Distance between rows
        offset = 0.5 * spacing  # Horizontal offset for hexagonal layout

        if row % 2 == 1:
            # Offset for even rows in hex grid
            x = col * spacing + offset
        else:
            x = col * spacing
        y = row * spacing

        # Adjust positions near the boundaries and corners
        if row == 0 and col == 0:
            x = 0.1
        elif row == 0 and col == col_per_row - 1:
            x = 1.0 - 0.1
        elif row == row_per_col - 1 and col == 0:
            y = 1.0 - 0.1
        elif row == row_per_col - 1 and col == col_per_row - 1:
            x = 1.0 - 0.1
            y = 1.0 - 0.1

        centers[i] = [x, y]
    return centers

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
