import numpy as np
import scipy.optimize as optimize
import random

def run_packing():
    num_circles = 26
    num_rows = 6
    num_cols = 5
    # Generate initial centers with a hybrid grid and corner circles
    centers = np.zeros((num_circles, 2))
    for i in range(num_rows):
        for j in range(num_cols):
            # Main grid
            x = j / (num_cols - 1) if num_cols > 1 else 0
            y = i / (num_rows - 1) if num_rows > 1 else 0
            centers[i * num_cols + j] = [x, y]
    
    # Add corner circles (adjust manually)
    extra_circles = [
        [0.1, 0.1], [0.9, 0.1], [0.1, 0.9], [0.9, 0.9],
        [0.15, 0.15], [0.85, 0.15], [0.15, 0.85], [0.85, 0.85]
    ]
    for idx, pos in enumerate(extra_circles):
        centers[num_rows * num_cols + idx] = pos

    # Define the objective function to maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    # Define constraint for radius non-negativity
    def constraint_radius_bounds(x):
        return -x

    # Define constraint for boundary
    def constraint_boundary(x, centers):
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if (x_i - r_i < 0 or x_i + r_i > 1 or
                y_i - r_i < 0 or y_i + r_i > 1):
                return np.inf
        return 0.0

    # Define constraint for non-overlap
    def constraint_overlap(x, centers):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < x[i] + x[j] - 1e-6:
                    return np.inf
        return 0.0

    # Create constraints for the optimization problem
    constraints = [
        {'type': 'ineq', 'fun': constraint_radius_bounds},
        {'type': 'ineq', 'fun': lambda x: constraint_boundary(x, centers)},
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers)}
    ]

    # Initial guess for radii
    radii_initial = np.full(num_circles, 0.1)

    # Optimize the radii using SLSQP
    result = optimize.minimize(objective, radii_initial, constraints=constraints, method='SLSQP', tol=1e-8)

    # Extract the optimized radii and centers
    radii = result.x

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
