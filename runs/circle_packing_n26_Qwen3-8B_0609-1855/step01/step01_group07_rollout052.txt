import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():

    np.random.seed(42)  # For reproducibility
    centers = np.random.uniform(low=0.05, high=0.95, size=(26, 2))  # Avoid edge collision
    radii = np.full(26, 0.02)  # Initial small radii to avoid initial overlaps

    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x[len(centers)*2:])  # Minimize negative sum of radii

    # Constraint: each circle stays within the unit square
    def constraint_boundary(x):
        for i in range(len(centers)):
            x_i, y_i = centers[i]
            r_i = x[len(centers)*2 + i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or

                return -1
        return 0

    # Constraint: no overlap between circles
    def constraint_overlap(x):
        for i in range(len(centers)):
            for j in range(i+1, len(centers)):
                x_i, y_i = centers[i]
                r_i = x[len(centers)*2 + i]
                x_j, y_j = centers[j]

                dist = np.hypot(x_i - x_j, y_i - y_j)
                if dist < r_i + r_j - 1e-12:

        return 0

    x0 = np.concatenate((centers.flatten(), radii))

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        r = radii[i]

            if dist < radii[i] + radii[j] - 1e-12:
