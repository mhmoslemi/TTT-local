import scipy.optimize as optimize

def run_packing():

    num_rows = 5
    num_cols = 5
    row_offset = np.sqrt(3) / 2  # vertical offset between rows

    row_spacing = 1.0 / (num_rows - 1)  # vertical spacing

            row = i * row_spacing

    def objective(x):
        return -np.sum(x)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):

        for j in range(i + 1, n):

            if dist < radii[i] + radii[j] - 1e-12:

    return True, "ok"
