import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    # Initialize random centers within the unit square
    np.random.seed(42)  # For reproducibility
    centers = np.random.rand(26, 2)
    
    # Initial guess for radii - small values to prevent overlap
    radii = np.full(26, 0.02)

    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x[len(centers)*2:])  # Minimize negative sum of radii

    # Bounds on the centers and radii
    center_bounds = np.array([[0.0, 1.0] for _ in range(len(centers)*2)])
    radius_bounds = np.array([0.0 for _ in range(len(radii))])
    bounds = Bounds(center_bounds, center_bounds)
    radius_bounds = Bounds(radius_bounds, np.inf * np.ones_like(radius_bounds))

    # Constraints
    def constraint_boundary(x):
        for i in range(len(centers)):
            x_i, y_i = centers[i]
            r_i = x[len(centers)*2 + i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or
                y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return -1
        return 0

    def constraint_overlap(x):
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                x_i, y_i = centers[i]
                r_i = x[len(centers)*2 + i]
                x_j, y_j = centers[j]
                r_j = x[len(centers)*2 + j]
                dist = np.hypot(x_i - x_j, y_i - y_j)
                if dist < r_i + r_j - 1e-12:
                    return -1
        return 0

    # Flatten centers and radii into a single vector for optimization
    x0 = np.concatenate((centers.flatten(), radii))

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Run optimization
    result = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    if not result.success:

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
