import scipy.optimize as optimize

def run_packing():

    radii = np.zeros(num_circles)

    # Number of rows and columns such that the pattern fits in a unit square
    num_rows = 5
    num_cols = 6  # More columns for better packing
    spacing = 0.25  # Initial spacing for hexagonal packing
    hex_radius = 0.1  # Initial radius for all circles
    offset = np.sqrt(3) / 2 * spacing  # Vertical offset for staggered rows

    # Place circles in a staggered hexagonal pattern
    for i in range(num_rows):
        for j in range(num_cols):
            x = j * spacing
            y = i * spacing + (i % 2) * offset
            centers[i * num_cols + j] = [x, y]

    # Define the objective function (negative sum of radii for minimization)
    def objective(radii_vals):
        return -np.sum(radii_vals)

    # Constraint: circles must not exceed the unit square
    def constraint_boundary(radii_vals):
        for i in range(num_circles):
            x, y = centers[i]
            r = radii_vals[i]

        return 0.0

    def constraint_overlap(radii_vals):
        for i in range(num_circles):

                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_vals[i] + radii_vals[j] - 1e-6:
                    return np.inf
        return 0.0

    bounds = [(0.0, 1.0) for _ in range(num_circles)]

def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
