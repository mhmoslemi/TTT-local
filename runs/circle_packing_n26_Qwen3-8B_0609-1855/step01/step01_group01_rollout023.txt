import scipy.optimize as opt
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26
    centers = np.random.rand(n, 2)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    # Define constraints
    constraints = []

    for i in range(n):
        # Constraint for radius >= 0
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: r[i] - 1e-12  # Ensure r >= 0
        })
        # Constraint for x - r >= 0
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: centers[i][0] - r[i] - 1e-12
        })
        # Constraint for x + r <= 1
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: 1.0 - centers[i][0] - r[i] - 1e-12
        })
        # Constraint for y - r >= 0
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: centers[i][1] - r[i] - 1e-12
        })
        # Constraint for y + r <= 1
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: 1.0 - centers[i][1] - r[i] - 1e-12
        })
    
    # Add constraints to ensure circles do not overlap
    for i in range(n):

                r_j = radii[j]
                dx = centers[i][0] - centers[j][0]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
