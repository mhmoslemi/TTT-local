import scipy.optimize as optimize

def run_packing():

    radii = np.zeros(num_circles)

    # We will place the centers in a staggered grid with some flexibility
    num_rows = 5

            x = j / (num_cols - 1) if num_cols > 1 else 0
            y = i / (num_rows - 1) if num_rows > 1 else 0

                y += np.sqrt(3) / 4

    def constraint_boundary(x, centers):

            if (x_i - r_i < -1e-8 or x_i + r_i > 1 + 1e-8 or

        method='SLSQP',

        options={'maxiter': 1000}

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):

        for j in range(i + 1, n):

            if dist < radii[i] + radii[j] - 1e-12:
