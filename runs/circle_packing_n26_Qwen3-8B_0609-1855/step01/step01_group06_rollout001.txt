import numpy as np
import scipy.optimize as optimize

def run_packing():

    # Define a hexagonal grid with 5 rows and 5 columns, adjusted to fit in a unit square
    centers = np.zeros((num_circles, 2))
    radius_initial = 0.15  # Initial radius estimate

    # Generate hexagonal packing in a unit square
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        x = col / 4.0
        y = row / 4.0 + (row % 2) * (np.sqrt(3) / 2) * radius_initial / radius_initial
        centers[i] = np.array([x, y])

    # Define the objective function to maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    # Define the constraint to ensure circles are within the square
    def constraint_boundary(x):

            x_i, y_i = centers[i]
            r_i = x[i]

                y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return np.inf
        return 0.0

    # Define the constraint to ensure circles do not overlap
    def constraint_overlap(x):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < x[i] + x[j] - 1e-6:
                    return np.inf
        return 0.0

    # Constraints for the optimization problem
    constraints = [
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = optimize.minimize(

        radii_initial,
        constraints=constraints,
        method='SLSQP',
        tol=1e-8,
        options={'maxiter': 1000}
    )

    # Extract the optimized radii and centers
    radii = result.x

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
