def compute_hexagonal_centers(n: int) -> np.ndarray:
    """
    Generate centers for a hexagonal close packing with optimal spacing.
    """
    row_count = int(np.ceil(n / 2))
    col_count = int(np.ceil(n / row_count))
    
    centers = np.zeros((n, 2))
    
    for i in range(n):
        row = i // col_count
        col = i % col_count
        
        h_spacing = 1.0 / (col_count - 1)
        v_spacing = 1.0 / (row_count - 1)

        x = 0.5 + (col - (col_count - 1) / 2) * h_spacing * 0.8

            x += 0.1

            y -= 0.1
        elif row == row_count - 1:
            y += 0.1
        
        centers[i] = [x, y]
    
    return centers

    """

            dy = centers[i, 1] - centers[j, 1]
