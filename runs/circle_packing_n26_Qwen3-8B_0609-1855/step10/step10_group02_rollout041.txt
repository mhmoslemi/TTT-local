def compute_grid_centers(n: int) -> np.ndarray:
    """
    Generate centers for a grid-like packing with optimized spacing.
    """
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    centers = np.zeros((n, 2))
    
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col / (num_cols - 1)
        y = row / (num_rows - 1)
        
        # Apply boundary padding
        if col == 0:
            x -= 0.1
        elif col == num_cols - 1:
            x += 0.1
        if row == 0:
            y -= 0.1
        elif row == num_rows - 1:
            y += 0.1
        
        centers[i] = [x, y]
    
    return centers
