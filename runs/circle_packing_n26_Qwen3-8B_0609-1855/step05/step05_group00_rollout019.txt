import scipy.optimize as opt
from typing import Tuple

def compute_hexagonal_grid_centers(n: int) -> np.ndarray:
    """
    Generate a hexagonal grid for n circles, adjusted for the unit square.
    """
    centers = np.zeros((n, 2))
    row_count = int(np.ceil(n / 6))
    col_count = 6
    spacing = 1 / (col_count + 1)
    for i in range(n):
        row = i // col_count
        col = i % col_count
        x_offset = spacing * (col + 1)
        y_offset = spacing * (row + 1)
        if row % 2 == 1:  # Alternate rows for hexagonal layout
            y_offset += spacing / 2

        # Adjust for boundary by shifting slightly
        x_adjust = 0.0
        y_adjust = 0.0
        if x_offset < 0.0:
            x_adjust = -x_offset
        elif x_offset > 1.0:
            x_adjust = x_offset - 1.0

        if y_offset < 0.0:
            y_adjust = -y_offset
        elif y_offset > 1.0:
            y_adjust = y_offset - 1.0

        x = x_offset - x_adjust
        y = y_offset - y_adjust
        centers[i] = [x, y]
    return centers

def validate_packing(centers, radii):
