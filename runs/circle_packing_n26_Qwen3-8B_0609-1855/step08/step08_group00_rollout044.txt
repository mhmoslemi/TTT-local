def generate_hexagonal_centers(n: int) -> np.ndarray:
    """
    Generate a hexagonal grid layout of centers within the unit square.
    """
    # Determine hexagonal grid dimensions
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Calculate the spacing to fit within the unit square
    spacing = 1.0 / (cols - 1) if cols > 1 else 1.0
    row_spacing = 1.0 / (rows - 1) if rows > 1 else 1.0
    
    # Generate centers based on hexagonal grid
    centers = np.zeros((n, 2))
    idx = 0
    for row in range(rows):
        for col in range(cols):
            x = col * spacing
            y = row * row_spacing
            if row % 2 == 1:
                x += spacing / 2
            centers[idx] = [x, y]
            idx += 1
    
    # Adjust boundary positions to stay within [0,1] and increase spacing slightly
    centers -= np.min(centers, axis=0)  # Normalize to start from 0
    centers /= np.max(centers, axis=0)  # Normalize to fit within [0,1]
    centers += np.random.uniform(-0.03, 0.03, size=centers.shape)  # Add small random perturbation
    
    return centers

def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
