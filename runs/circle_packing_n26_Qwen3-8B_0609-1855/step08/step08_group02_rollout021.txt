def generate_candidate_centers(n: int) -> np.ndarray:

    Generate multiple candidate center configurations to explore different packing arrangements.

    grid_centers = np.zeros((n, 2))

    num_cols = int(np.ceil(n / num_rows))

        row = i // num_cols
        col = i % num_cols
        x = col / (num_cols - 1) if num_cols > 1 else 0.5
        y = row / (num_rows - 1) if num_rows > 1 else 0.5

        if col == 0:
            x -= 0.1
        elif col == num_cols - 1:
            x += 0.1
        if row == 0:
            y -= 0.1
        elif row == num_rows - 1:
            y += 0.1
        
        grid_centers[i] = [x, y]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
