def compute_optimal_centers_hexagonal(n: int) -> np.ndarray:

    row, col = 0, 0

    return centers

    perturbation = np.random.rand(n, 2) * 0.1 - 0.05

    return centers

def compute_optimal_centers_random(n: int) -> np.ndarray:

def validate_packing(centers, radii):

        return False, "NaN values present"

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
