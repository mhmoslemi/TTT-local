import numpy as np
import scipy.optimize as opt
from typing import Tuple

def generate_candidate_centers(n: int) -> np.ndarray:
    """
    Generate several candidate center configurations to explore.
    """
    n_candidates = 10
    candidates = []

    # Start with a grid-like packing
    grid_centers = np.zeros((n, 2))
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col / (num_cols - 1)  # Uniform x distribution
        y = row / (num_rows - 1)  # Uniform y distribution
        grid_centers[i] = [x, y]
    candidates.append(grid_centers)

    # Start with a hexagonal packing in the center
    hex_centers = np.zeros((n, 2))
    row_count = int(np.ceil(n / 2))
    col_count = int(np.ceil(n / row_count))

        row = i // col_count

        x = 0.5 + (col - (col_count - 1) / 2) * (1 / (col_count - 1)) * 0.8

def optimize_packing(centers: np.ndarray, n: int) -> Tuple[np.ndarray, np.ndarray, float]:
    """
    Optimize the radii of the circles given their centers to maximize the sum of radii.
    """
    # Objective function: maximize sum of radii
    def objective(x):
        radii = x
        return -np.sum(radii)  # Negative for minimization

    # Constraints: ensure each circle is within the unit square and no overlap
    def square_constraints(radii):
        return np.concatenate([
            centers[:, 0] - radii + 1e-12,  # x - r >= 0
            1 - centers[:, 0] - radii + 1e-12,  # 1 - x - r >= 0
            centers[:, 1] - radii + 1e-12,  # y - r >= 0
            1 - centers[:, 1] - radii + 1e-12  # 1 - y - r >= 0
        ])

    def circle_overlap_constraints(radii):
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - radii[i] - radii[j] + 1e-12)
        return constraints

    # Initial guess
    initial_radii = np.full(n, 0.1)

    # Optimization bounds and constraints
    bounds = [(0.0, 0.15) for _ in range(n)]
    cons = [
        {'type': 'ineq', 'fun': square_constraints},
        {'type': 'ineq', 'fun': circle_overlap_constraints}
    ]

    # Optimization
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-10,
        options={'maxiter': 5000, 'ftol': 1e-10, 'eps': 1e-8, 'disp': False}
    )

    if not result.success:
        raise RuntimeError("Optimization failed")
    
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)
    return centers, radii_opt, sum_radii

def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
