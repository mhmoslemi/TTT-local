import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles placed on a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1  # Initial guess for radii

    # Define the objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape((10, 2))
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Define constraints
    def constraints(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape((10, 2))
        radii_flat = x[60:]
        # Constraint 1: Each circle is inside the unit square
        for i in range(10):
            r = radii_flat[i]
            x_center, y_center = centers_flat[i]
            if not (r <= x_center <= 1 - r and r <= y_center <= 1 - r):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0  # All constraints satisfied

    # Define the bounds: for each circle, x, y in [0, 1], r >= 0 and <= 0.5
    bounds = []
    for _ in range(10):
        bounds += [(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)]

    # Set up the optimization problem
    x0 = np.concatenate([centers.flatten(), radii])
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract result
    best_centers = result.x[:60].reshape((10, 2))
    best_radii = result.x[60:]
    sum_radii = -result.fun

    # Ensure all constraints are satisfied
    # Check if any circle is outside the square
    for i in range(10):
        r = best_radii[i]
        x, y = best_centers[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            raise ValueError("At least one circle is outside the unit square.")
    # Check if any circles overlap
    for i in range(10):
        for j in range(i + 1, 10):
            dx = best_centers[i, 0] - best_centers[j, 0]
            dy = best_centers[i, 1] - best_centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < best_radii[i] + best_radii[j]:
                raise ValueError("At least two circles overlap.")

    return (best_centers, best_radii, sum_radii)
