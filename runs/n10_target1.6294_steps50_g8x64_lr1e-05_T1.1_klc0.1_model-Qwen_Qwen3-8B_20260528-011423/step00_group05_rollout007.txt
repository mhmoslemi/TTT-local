import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.random.uniform(0.05, 0.15, 10)

    def objective_function(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize sum_radii

    def constraints(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        constraints = []

        # Constraint: each circle is within the unit square
        for i in range(10):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x < r or x > 1 - r or y < r or y > 1 - r:
                return np.inf  # Invalid configuration
        constraints.append({'type': 'ineq', 'fun': lambda p: 0})

        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.hypot(dx, dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 1e-6:
                    return np.inf  # Too close
        constraints.append({'type': 'ineq', 'fun': lambda p: 0})

        # Constraint: all radii are non-negative
        for r in radii_flat:
            if r < 0:
                return np.inf
        constraints.append({'type': 'ineq', 'fun': lambda p: 0})

        return 0

    # Initial parameters: flatten centers and radii
    initial_params = np.concatenate([centers.reshape(10 * 2), radii])

    # Optimization
    result = minimize(
        objective_function,
        initial_params,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0.0, 1.0) for _ in range(60)] + [(0.0, 0.5) for _ in range(10)]
    )

    # Extract and reshape results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
