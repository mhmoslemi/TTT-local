import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(centers, radii):
        n = len(radii)
        cons = []
        
        # Constraint: All circles are fully inside the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            cons.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: x - r})
            cons.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1 - x - r})
            cons.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: y - r})
            cons.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1 - y - r})
        
        # Constraint: No two circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                cons.append({'type': 'ineq', 'fun': lambda r, r_i=radii[i], r_j=radii[j], d=dist: d - r_i - r_j})
        
        return cons
    
    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            constraints=constraints(centers, radii),
            bounds=[(0, 1)] * 10,
            options={'ftol': 1e-9, 'maxiter': 1000}
        )
    
    centers_result = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii_result = result.x
    sum_radii = -result.fun
    
    return (centers_result, radii_result, sum_radii)
