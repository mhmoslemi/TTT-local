import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        radii = radii.reshape(-1)
        constraints = []
        # Ensure circles are within the unit square
        for r in radii:
            constraints.append({'type': 'ineq', 'fun': lambda r: r})
            constraints.append({'type': 'ineq', 'fun': lambda r: 1 - r - 2 * r})  # x >= r, x <= 1 - r
        # Ensure no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                def constraint(radii):
                    r1 = radii[i]
                    r2 = radii[j]
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    return np.sqrt(dx**2 + dy**2) - r1 - r2
                constraints.append({'type': 'ineq', 'fun': constraint})
        return constraints

    result = minimize(objective, radii, constraints=constraints(radii), method='SLSQP', bounds=[(0, 1)]*10)
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
