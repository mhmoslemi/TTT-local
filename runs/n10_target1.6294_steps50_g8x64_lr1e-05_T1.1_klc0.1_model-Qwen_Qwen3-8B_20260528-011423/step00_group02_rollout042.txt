import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for the centers and radii (random positions with small radii)
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.full(10, 0.05)

    def objective(x):
        # x is a 22-dimensional vector: 10 centers (2D) + 10 radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        sum_radii = np.sum(radii)
        # Constraint 1: All circles are inside the unit square
        in_square = np.all((radii <= (centers[:, 0] - 0.0)) & (centers[:, 0] + radii <= 1.0) &
                          (radii <= (centers[:, 1] - 0.0)) & (centers[:, 1] + radii <= 1.0))
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return float('inf')
        # Objective is to maximize sum_radii, so minimize -sum_radii
        return -sum_radii

    # Bounds for centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0.1, 0.9), (0.1, 0.9)])  # centers
        bounds.append((0.0, 0.9))                # radii

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds, method='L-BFGS-B')

    # Extract result
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
