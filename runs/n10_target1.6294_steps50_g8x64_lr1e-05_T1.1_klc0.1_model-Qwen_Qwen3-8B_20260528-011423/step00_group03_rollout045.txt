import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints_centered(radii):
        return np.sum(radii) - 1.6294

    def constraints_bounds(radii):
        return radii

    def constraints_non_overlap(x):
        dist_sq = np.sum((x[:, np.newaxis, :] - x[np.newaxis, :, :]) ** 2, axis=-1)
        r_sum = np.sum(radii[:, np.newaxis] + radii[np.newaxis, :], axis=-1)
        return dist_sq - r_sum

    def bounds():
        return [(0, 1) for _ in range(20)]

    def initial_guess():
        angles = np.linspace(0, 2 * np.pi, 10, endpoint=False)
        radii = np.full(10, 0.14)
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = 0.5 + 0.4 * np.cos(angles[i])
            centers[i, 1] = 0.5 + 0.4 * np.sin(angles[i])
        return centers, radii

    def optimize():
        centers, radii = initial_guess()
        x0 = np.concatenate([centers.flatten(), radii])
        cons = [
            {'type': 'ineq', 'fun': constraints_bounds},
            {'type': 'ineq', 'fun': lambda x: constraints_non_overlap(x[:20].reshape(10, 2))},
            {'type': 'ineq', 'fun': lambda x: constraints_centered(x[20:])}
        ]
        bounds = bounds()
        res = minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=bounds,
            constraints=cons,
            tol=1e-6,
            options={'maxiter': 1000}
        )
        centers_opt = res.x[:20].reshape(10, 2)
        radii_opt = res.x[20:]
        return centers_opt, radii_opt, np.sum(radii_opt)

    centers, radii, sum_radii = optimize()
    return (centers, radii, sum_radii)
