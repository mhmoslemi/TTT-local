import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radii: all same
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        # All circles are within the unit square
        radii_bounds = [(0, 1 - x) for x in centers]
        for i in range(10):
            if radii[i] > radii_bounds[i]:
                return False
        # No overlapping between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    bounds = [(0, 1) for _ in range(10)]
    result = minimize(
        objective,
        radii,
        constraints={"type": "ineq", "fun": constraints},
        bounds=bounds,
        method="SLSQP"
    )

    # Extract solution
    optimized_radii = result.x
    optimized_centers = copy.deepcopy(centers)
    sum_radii = np.sum(optimized_radii)

    # Final check to ensure constraints are met
    assert all(optimized_radii[i] <= 1 - optimized_centers[i, 0] for i in range(10))
    assert all(optimized_radii[i] <= 1 - optimized_centers[i, 1] for i in range(10))
    for i in range(10):
        for j in range(i + 1, 10):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            assert dist >= optimized_radii[i] + optimized_radii[j]

    return (optimized_centers, optimized_radii, sum_radii)
