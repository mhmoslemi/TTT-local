import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (random points in the square)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Constraint 1: All circles must be inside the square
        # r_i <= x_i - r_i <= 1 - r_i and same for y
        # So, x_i - r_i >= r_i => x_i >= 2*r_i
        # And y_i - r_i >= r_i => y_i >= 2*r_i
        # Also, x_i + r_i <= 1 and y_i + r_i <= 1
        
        # We can write the constraints as:
        # x_i - r_i >= r_i => x_i >= 2*r_i
        # y_i - r_i >= r_i => y_i >= 2*r_i
        # x_i + r_i <= 1
        # y_i + r_i <= 1
        # For all i
        
        # Constraint 2: Distance between any two centers >= sum of their radii
        # For all i < j: ||centers[i] - centers[j]|| >= radii[i] + radii[j]
        
        # Flatten centers and radii for optimization
        centers_flat = centers.reshape(-1)
        radii_flat = radii.flatten()
        
        # Constraint 1: x >= 2*r and y >= 2*r
        constraint1 = np.zeros(10)
        for i in range(10):
            constraint1[i] = centers_flat[2*i] - 2*radii_flat[i]
            constraint1[i] = max(constraint1[i], 0)  # Ensure non-negative
        # Constraint 1: x + r <= 1 and y + r <= 1
        constraint2 = np.zeros(10)
        for i in range(10):
            constraint2[i] = 1 - (centers_flat[2*i] + radii_flat[i])
            constraint2[i] = max(constraint2[i], 0)
        # Constraint 1: y >= 2*r
        constraint3 = np.zeros(10)
        for i in range(10):
            constraint3[i] = centers_flat[2*i + 1] - 2*radii_flat[i]
            constraint3[i] = max(constraint3[i], 0)
        # Constraint 1: y + r <= 1
        constraint4 = np.zeros(10)
        for i in range(10):
            constraint4[i] = 1 - (centers_flat[2*i + 1] + radii_flat[i])
            constraint4[i] = max(constraint4[i], 0)
        
        # Constraint 2: Distance between centers >= sum of radii
        # Flatten all pairwise distances and compare with sum of radii
        dists = np.zeros(0)
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists = np.append(dists, dist - radii[i] - radii[j])
        
        # Combine all constraints
        return np.concatenate((constraint1, constraint2, constraint3, constraint4, dists))
    
    # Define bounds for radii
    bounds = [(0, 0.5) for _ in range(10)]  # Max radius is 0.5 to stay inside the square
    
    # Define constraints for the optimizer
    def constraint_func(radii):
        return constraints(radii)
    
    constraints_opt = [{'type': 'ineq', 'fun': constraint_func}]
    
    # Perform optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints_opt, tol=1e-6)
    
    # Extract final result
    final_radii = result.x
    final_centers = centers  # Keep initial centers as they are, but they can be optimized in more advanced methods
    
    # Return the result
    sum_radii = np.sum(final_radii)
    return (final_centers, final_radii, sum_radii)
