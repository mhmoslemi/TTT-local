import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[2:])  # radii are x[2:]
    
    # Constraints: 1. All circles are inside the square
    #              2. No two circles overlap
    #              3. All radii are >= 0
    
    # Flatten centers and radii for optimization
    x0 = np.concatenate((centers.reshape(10, 2), radii.reshape(-1, 1)), axis=1).reshape(-1)
    
    def constraint_inside(x):
        # Unpack
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        
        # Check if all circles are inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return -1
        return 0
    
    def constraint_overlap(x):
        # Unpack
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0
    
    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Bounds: coordinates in [0, 1], radii >= 0
    bounds = [(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]
    
    # Set up optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=bounds,
            constraints=cons,
            tol=1e-6,
            options={'maxiter': 1000}
        )
    
    # Extract result
    x_result = result.x
    centers_result = x_result[:60].reshape(10, 2)
    radii_result = x_result[60:]
    sum_radii = -result.fun
    
    return (centers_result, radii_result, sum_radii)
