import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_center_in_square(x, r):
        x = np.reshape(x, (10, 2))
        return np.concatenate([x[:, 0] - r, 1 - x[:, 0] - r, x[:, 1] - r, 1 - x[:, 1] - r])

    def constraint_no_overlap(x):
        x = np.reshape(x, (10, 2))
        dists = np.sqrt(np.sum((x[:, np.newaxis] - x[np.newaxis, :])**2, axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    bounds = [(0, 1 - 1e-6) for _ in range(20)]  # each center has x and y coordinate

    # Optimize
    result = minimize(
        fun=objective,
        x0=np.concatenate([initial_centers.reshape(-1), initial_radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraint_center_in_square(x, x[10:])},
            {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x)}
        ]
    )

    centers = np.reshape(result.x[:20], (10, 2))
    radii = result.x[10:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
