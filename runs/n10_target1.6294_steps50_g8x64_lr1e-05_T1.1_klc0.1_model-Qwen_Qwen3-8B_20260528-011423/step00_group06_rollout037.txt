import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced points with small radii
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 10) for y in np.linspace(0.1, 0.9, 10)])
    radii = np.full(10, 0.05)

    def objective(radii_flat):
        return -np.sum(radii_flat)

    def constraints(radii_flat):
        radii = radii_flat.reshape(10)
        # Constraint 1: Each circle must be inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - 2 * np.min(np.abs(centers[i, 0] - np.array([0, 1])))
                return np.inf
            if radii[i] > 1 - 2 * np.min(np.abs(centers[i, 1] - np.array([0, 1])))
                return np.inf
        # Constraint 2: No overlapping between circles
        dists = squareform(pdist(centers))
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Prepare radii_flat as a list for optimization
    radii_flat = radii.copy()
    res = minimize(
        objective,
        radii_flat,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1 - 2 * np.min(np.abs(centers[i, 0] - np.array([0, 1])))) for i in range(10)]
    )

    # Extract results
    radii = res.x
    centers = centers
    sum_radii = -res.fun

    return (centers, radii, sum_radii)
