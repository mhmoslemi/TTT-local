import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: spread 10 circles in a grid-like pattern
    centers = np.array([[i * 0.25, j * 0.25] for i in range(10) for j in range(10)])
    centers = centers[:10]  # Take only 10 circles
    radii = np.ones(10)  # Initial radius for each circle

    def objective(radii, centers):
        return -np.sum(radii)  # We want to maximize the sum, so minimize -sum

    def inequality_constraints(radii, centers):
        n = len(radii)
        cons = []
        # Constraint 1: Each circle is within the unit square
        for i in range(n):
            r = radii[i]
            x, y = centers[i]
            cons.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: r - (1 - r - x) + r - (1 - r - y)})
        # Constraint 2: No two circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                def constraint(radii):
                    r_i = radii[i]
                    r_j = radii[j]
                    x_i, y_i = centers[i]
                    x_j, y_j = centers[j]
                    dx = x_i - x_j
                    dy = y_i - y_j
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - r_i - r_j
                cons.append({'type': 'ineq', 'fun': constraint})
        return cons

    # Optimization
    res = minimize(
        fun=objective,
        x0=radii,
        args=(centers,),
        method='SLSQP',
        constraints=inequality_constraints(radii, centers),
        bounds=[(0, 1)] * 10,
        tol=1e-6
    )

    if res.success:
        sum_radii = -res.fun
        centers = np.array(res.x).reshape(-1, 2)
        return (centers, res.x, sum_radii)
    else:
        warnings.warn("Optimization did not converge")
        return (centers, radii, np.sum(radii))
