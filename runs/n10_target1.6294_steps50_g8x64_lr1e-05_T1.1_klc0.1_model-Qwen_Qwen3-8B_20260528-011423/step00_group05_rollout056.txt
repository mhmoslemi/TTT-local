import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(x):
        # x is flattened array of centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return -sum(radii)  # minimize negative sum to maximize sum

    def constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]

        # Box constraints: circles must be inside the unit square
        constraints_box = []
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            constraints_box.append(x_i - r_i >= 0)
            constraints_box.append(1 - x_i - r_i >= 0)
            constraints_box.append(y_i - r_i >= 0)
            constraints_box.append(1 - y_i - r_i >= 0)
        constraints_box.append(np.sum(radii) <= 1.6294)  # Target sum constraint

        # Pairwise distance constraints: no overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints_box.append(dist >= radii[i] + radii[j])

        return constraints_box

    # Convert constraints to a list for the optimize function
    def constraint_func(x):
        res = constraints(x)
        return res

    # Optimization
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1)] * 2)  # centers
        bounds.extend([0, 1.0])      # radii (non-negative)
    result = minimize(
        objective, 
        x0=np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_func},
        tol=1e-8
    )

    # Extract results
    x_opt = result.x
    centers_opt = x_opt[:20].reshape(10, 2)
    radii_opt = x_opt[20:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
