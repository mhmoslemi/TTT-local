import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - 2 * centers[i, 0] or radii[i] > 1 - 2 * centers[i, 1]:
                return np.inf

        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    result = minimize(objective, radii, constraints=lambda x: constraints(x), method='SLSQP', bounds=[(0, 1) for _ in range(10)])
    radii = result.x
    sum_radii = -result.fun

    # Apply constraints again to enforce them
    def apply_constraints(radii):
        for i in range(10):
            if radii[i] < 0:
                radii[i] = 0
            if radii[i] > 1 - 2 * centers[i, 0] or radii[i] > 1 - 2 * centers[i, 1]:
                radii[i] = min(1 - 2 * centers[i, 0], 1 - 2 * centers[i, 1])
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    radii[i] = (dist - radii[j]) / 2
        return radii

    radii = apply_constraints(radii)
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
