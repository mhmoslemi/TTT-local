import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.array([0.1] * 10)

    def objective_function(x):
        # Flatten x into centers and radii
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:].reshape(10)
        
        # Objective: maximize the sum of radii (minimize negative sum)
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    def constraints(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:].reshape(10)

        # Constraint 1: All circles are inside the square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -np.inf
        # Constraint 2: No overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -np.inf
        # Constraint 3: All radii non-negative
        for r in radii_flat:
            if r < 0:
                return -np.inf
        return 0

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization setup
    result = minimize(
        objective_function,
        x0,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    )

    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:].reshape(10)
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
