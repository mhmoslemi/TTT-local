import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 10
    initial_centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    initial_radii = np.full(num_circles, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraint_center_inside(radii, centers):
        # Ensure each circle is fully inside the unit square
        return np.sum((centers - radii[:, np.newaxis])[:, :, 0] >= 0) + \
               np.sum((1 - centers - radii[:, np.newaxis])[:, :, 0] >= 0) + \
               np.sum((centers - radii[:, np.newaxis])[:, :, 1] >= 0) + \
               np.sum((1 - centers - radii[:, np.newaxis])[:, :, 1] >= 0)

    def constraint_no_overlap(radii, centers):
        # Check that the distance between centers >= sum of radii
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraints(radii, centers):
        return [constraint_center_inside(radii, centers), constraint_no_overlap(radii, centers)]

    def objective_with_constraints(radii, centers):
        res = minimize(lambda r: -np.sum(r), radii, method='SLSQP',
                       constraints=[{'type': 'ineq', 'fun': lambda r: -r[i] for i in range(num_circles)}])
        if res.success:
            return res.fun, res.x
        else:
            return float('inf'), np.zeros(num_circles)

    # Run optimization
    result = minimize(lambda x: -np.sum(x), initial_radii, method='SLSQP',
                      constraints=[{'type': 'ineq', 'fun': constraint_center_inside, 'args': (initial_centers,)},
                                   {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (initial_centers,)}])

    if not result.success:
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    # Extract best solution
    best_radii = result.x
    best_centers = initial_centers

    return best_centers, best_radii, np.sum(best_radii)
