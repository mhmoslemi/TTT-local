import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((10, 2))
    radii = np.ones(10) * 0.1  # Initial guess for radii

    def objective(radii):
        return -np.sum(radii)  # We minimize the negative sum to maximize it

    def constraints(radii):
        centers = np.zeros((10, 2))
        # Place circles in a grid-like configuration
        for i in range(10):
            x = i % 5 / 4
            y = i // 5 / 4
            centers[i] = [x, y]
        # Check that all circles are within the square
        for i in range(10):
            r = radii[i]
            if not (r <= centers[i][0] <= 1 - r and r <= centers[i][1] <= 1 - r):
                return np.inf
        # Check that circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': constraints, 'jac': None}
        ],
        bounds=[(0, 1.0) for _ in range(10)],
        options={'disp': False, 'ftol': 1e-8}
    )

    # Extract result
    final_radii = result.x
    final_centers = np.zeros((10, 2))
    for i in range(10):
        x = i % 5 / 4
        y = i // 5 / 4
        final_centers[i] = [x, y]

    # Final check and small perturbation to avoid exact overlap
    for i in range(10):
        r = final_radii[i]
        final_centers[i] = [np.clip(final_centers[i][0], r, 1 - r), np.clip(final_centers[i][1], r, 1 - r)]

    for i in range(10):
        for j in range(i + 1, 10):
            dx = final_centers[i][0] - final_centers[j][0]
            dy = final_centers[i][1] - final_centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < final_radii[i] + final_radii[j] - 1e-6:
                final_centers[j] += np.random.uniform(-1e-4, 1e-4, 2)

    final_sum = np.sum(final_radii)

    return (final_centers, final_radii, final_sum)
