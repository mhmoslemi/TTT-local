import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (random points in the unit square)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Initial guess for radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Convert radii to a flat array for optimization
        radii_flat = radii.flatten()
        # Constraints: radii must be non-negative
        bounds = [(0, 1) for _ in range(10)]
        # Constraint: distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return False
        # Constraint: circles must be fully inside the unit square
        for i in range(10):
            if not (radii_flat[i] <= centers[i, 0] <= 1 - radii_flat[i] and
                    radii_flat[i] <= centers[i, 1] <= 1 - radii_flat[i]):
                return False
        return True

    # Optimize radii
    result = minimize(objective, radii, method='L-BFGS-B', bounds=[(0, 1) for _ in range(10)])
    optimized_radii = result.x
    if not constraints(optimized_radii):
        optimized_radii = np.full(10, 0.1)  # Fallback to initial guess if constraints violated

    sum_radii = np.sum(optimized_radii)
    return (centers, optimized_radii, sum_radii)
