import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers: evenly spaced grid
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    centers = np.vstack([centers, [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5]])  # Add 5 more centers near center
    radii = np.ones(10) * 0.1  # Initial radius for all circles

    def objective(radii, centers):
        # Objective: minimize negative sum of radii (i.e. maximize sum of radii)
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint 1: All circles are within the unit square
        # x - r >= 0, x + r <= 1, same for y
        # x[i] - r[i] >= 0
        # 1 - x[i] - r[i] >= 0
        # y[i] - r[i] >= 0
        # 1 - y[i] - r[i] >= 0
        # We can rephrase this using bounds on r[i]
        return (radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]) & (radii >= 0)

    def non_overlap_constr(radii, centers):
        # Constraint 2: No two circles overlap
        # distance between center[i] and center[j] >= r[i] + r[j]
        # We compute all pairwise distances and ensure they are >= sum of radii
        diff = centers[:, np.newaxis, :] - centers[np.newaxis, :, :]
        dist = np.sqrt(np.sum(diff**2, axis=2))
        return dist - (radii[np.newaxis, :] + radii[:, np.newaxis])

    def wrapped_objective(x):
        # x is a flat array of radii and centers
        radii = x[:10]
        centers = x[10:].reshape(10, 2)
        # Check constraints and return objective
        if not np.all(radii >= 0) or not np.all(radii <= 1 - centers[:, 0]) or not np.all(radii <= 1 - centers[:, 1]):
            return np.inf
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        if np.any(dist < (radii[np.newaxis, :] + radii[:, np.newaxis])):
            return np.inf
        return -np.sum(radii)

    def wrapped_constraints(x):
        # x is a flat array of radii and centers
        radii = x[:10]
        centers = x[10:].reshape(10, 2)
        # All radii must be non-negative and not exceed square limits
        if not np.all(radii >= 0) or not np.all(radii <= 1 - centers[:, 0]) or not np.all(radii <= 1 - centers[:, 1]):
            return np.inf
        # No two circles can overlap
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        if np.any(dist < (radii[np.newaxis, :] + radii[:, np.newaxis])):
            return np.inf
        return 0

    # Initial guess flattened
    x0 = np.concatenate([radii, centers.flatten()])

    # Optimization settings
    bounds = [(0, 1 - centers[i, 0]) for i in range(10)] + [(0, 1) for _ in range(20)]  # Radii bounds and center bounds

    # Optimization
    result = minimize(wrapped_objective, x0, bounds=bounds, method='SLSQP', constraints={'type': 'ineq', 'fun': wrapped_constraints})

    # Extract results
    radii = result.x[:10]
    centers = result.x[10:].reshape(10, 2)
    sum_radii = np.sum(radii)

    # Ensure all constraints are met (just in case)
    if not np.all(radii >= 0) or not np.all(radii <= 1 - centers[:, 0]) or not np.all(radii <= 1 - centers[:, 1]):
        print("Warning: Some radii or centers violate square constraints.")
    dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
    if np.any(dist < (radii[np.newaxis, :] + radii[:, np.newaxis])):
        print("Warning: Some circles overlap.")

    return (centers, radii, sum_radii)
