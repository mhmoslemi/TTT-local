import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import warnings

def run_packing():
    def objective(params):
        # Reshape params into centers and radii
        centers = params[:60].reshape((10, 2))
        radii = params[60:]
        # Check if all circles are inside the unit square
        if np.any(radii < 0):
            return float('inf')
        if np.any(centers[:, 0] < radii) or np.any(centers[:, 0] > 1 - radii):
            return float('inf')
        if np.any(centers[:, 1] < radii) or np.any(centers[:, 1] > 1 - radii):
            return float('inf')
        # Compute sum of radii
        sum_radii = np.sum(radii)
        # Check if any circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return float('inf')
        return -sum_radii  # We minimize -sum_radii to maximize sum_radii

    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)
    initial_params = np.concatenate([initial_centers.flatten(), initial_radii])

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=OptimizeResult)
        result = minimize(objective, initial_params, method='L-BFGS-B', bounds=[(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)])

    # Extract results
    centers = result.x[:60].reshape((10, 2))
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
