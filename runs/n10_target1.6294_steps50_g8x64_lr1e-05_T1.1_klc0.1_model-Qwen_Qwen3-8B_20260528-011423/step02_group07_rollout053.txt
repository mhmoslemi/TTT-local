import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Check if all radii are non-negative
        if any(r < 0 for r in radii):
            return np.inf
        # Check if all circles are within the unit square
        for i in range(10):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf
        # Check if circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization bounds
    bounds = [(0, 1) for _ in range(10)]

    # Run the optimization
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    return (centers, res.x, -res.fun)
