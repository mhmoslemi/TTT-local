import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import math

def run_packing():
    # Initial guess for circle centers (randomly placed)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    
    def objective(radii):
        return -np.sum(radii)  # We minimize the negative sum to maximize the sum

    def constraint_center_inside(radii):
        return np.sum((radii <= 1 - centers[:, 0]) * (radii <= 1 - centers[:, 1]) * (radii >= 0))

    def constraint_no_overlap(radii):
        dists = pdist(centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.sum(dists >= radii_sum)

    # Objective function
    def objective_func(x):
        radii = x
        return -np.sum(radii)

    # Constraints
    def constraint_1(x):
        # Ensure circles are within the unit square
        dist = np.sum((x <= 1 - centers[:, 0]) * (x <= 1 - centers[:, 1]) * (x >= 0))
        return dist

    def constraint_2(x):
        # Ensure no two circles overlap
        dists = pdist(centers)
        radii_sum = x[:, np.newaxis] + x[np.newaxis, :]
        return np.sum(dists >= radii_sum)

    # Bounds for each radius: [0, 1]
    bounds = [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(objective_func, radii, method='SLSQP',
                      bounds=bounds,
                      constraints=[{'type': 'ineq', 'fun': constraint_1},
                                   {'type': 'ineq', 'fun': constraint_2}])

    optimal_radii = result.x
    sum_radii = -result.fun
    return (centers, optimal_radii, sum_radii)
