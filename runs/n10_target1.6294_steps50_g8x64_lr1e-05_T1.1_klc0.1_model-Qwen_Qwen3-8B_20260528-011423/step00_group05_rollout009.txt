import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 10 circles placed in a grid pattern
    def initial_guess():
        centers = np.zeros((10, 2))
        for i in range(10):
            x = i % 5 / 4.0 + 0.125
            y = i // 5 / 4.0 + 0.125
            centers[i] = [x, y]
        return centers

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: Circles must be fully inside the unit square
    def constraint_inside(radii, centers):
        return np.min(centers - radii) >= 0.0

    # Constraint: Circles must not overlap
    def constraint_overlap(radii, centers):
        distances = np.sqrt(np.sum((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2, axis=2))
        return np.min(distances - radii[:, np.newaxis] - radii[np.newaxis, :]) >= 0.0

    # Optimization problem setup
    def optimize():
        centers = initial_guess()
        radii = np.full(10, 0.1)
        bounds = [(0.0, 1.0) for _ in range(20)]  # Bounds for x and y of each center
        constraints = []

        # Constraint to ensure circles are inside the square
        constraints.append({
            'type': 'ineq',
            'fun': lambda x: np.min(np.concatenate((x[::2] - x[1::2], 1.0 - x[::2] - x[1::2]))),
        })

        # Constraint to ensure no overlap between circles
        def constraint_overlap_func(x):
            centers = x.reshape((10, 2))
            distances = np.sqrt(np.sum((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2, axis=2))
            return np.min(distances - x[::2] - x[1::2]) >= 0.0

        constraints.append({
            'type': 'ineq',
            'fun': constraint_overlap_func,
        })

        # Optimization problem with bounds and constraints
        result = minimize(
            fun=objective,
            x0=np.concatenate((centers.flatten(), radii)),
            bounds=bounds,
            constraints=constraints,
            method='SLSQP',
            tol=1e-8
        )

        centers_opt = result.x[:20].reshape((10, 2))
        radii_opt = result.x[20:]
        sum_radii = np.sum(radii_opt)

        # Validate constraints
        if np.min(centers_opt - radii_opt) < 0 or np.min(1.0 - centers_opt - radii_opt) < 0:
            raise ValueError("Some circles are not fully inside the unit square.")

        if np.min(np.sqrt(np.sum((centers_opt[:, np.newaxis] - centers_opt[np.newaxis, :]) ** 2, axis=2)) - radii_opt[:, np.newaxis] - radii_opt[np.newaxis, :]) < 0:
            raise ValueError("Some circles are overlapping.")

        return (centers_opt, radii_opt, sum_radii)

    try:
        from scipy.optimize import minimize
        return optimize()
    except ImportError:
        raise ImportError("scipy.optimize is required for this optimization.")
