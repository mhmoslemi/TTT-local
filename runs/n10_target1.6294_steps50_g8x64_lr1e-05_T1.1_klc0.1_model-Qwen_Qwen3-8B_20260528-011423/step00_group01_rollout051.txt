import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    initial_centers = np.random.rand(10, 2)
    initial_radii = np.random.rand(10) * 0.1

    def objective_function(params):
        # Unpack parameters
        centers = params[:10*2].reshape(10, 2)
        radii = params[10*2:]

        # Objective: maximize sum of radii (minimize negative sum)
        sum_radii = -np.sum(radii)

        # Constraints
        constraints = []

        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r >= 0) * 1e6)
            constraints.append((x + r <= 1) * 1e6)
            constraints.append((y - r >= 0) * 1e6)
            constraints.append((y + r <= 1) * 1e6)

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append((dist - (radii[i] + radii[j])) * 1e6)

        # Convert to scalar objective
        return sum_radii

    # Define bounds for parameters
    bounds = []
    for _ in range(10*2):
        bounds.append((0, 1))
    for _ in range(10):
        bounds.append((0, 1))

    # Optimization
    result = minimize(objective_function, x0=np.concatenate([initial_centers.flatten(), initial_radii]),
                      bounds=bounds, method='SLSQP', tol=1e-6, options={'maxiter': 500})

    # Extract results
    centers = result.x[:10*2].reshape(10, 2)
    radii = result.x[10*2:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
