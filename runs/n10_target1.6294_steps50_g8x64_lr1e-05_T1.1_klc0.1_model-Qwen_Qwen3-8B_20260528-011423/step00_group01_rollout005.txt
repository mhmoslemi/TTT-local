import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with some radius
    centers = np.zeros((10, 2))
    radii = np.zeros(10)
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
        radii[i] = 0.1
    
    # Objective function: maximize sum of radii (minimize -sum)
    def objective(radii_flat):
        return -np.sum(radii_flat)
    
    # Constraint: circles are inside the square [0, 1] x [0, 1]
    def square_constraint(radii_flat):
        return np.array([radii_flat[i] for i in range(10)])
    
    # Constraint: no two circles overlap
    def circle_overlap_constraint(radii_flat):
        centers_flat = np.reshape(centers, (10 * 2,))
        dists = []
        for i in range(10):
            for j in range(i + 1, 10):
                x1, y1 = centers_flat[2*i], centers_flat[2*i + 1]
                x2, y2 = centers_flat[2*j], centers_flat[2*j + 1]
                dist = math.hypot(x1 - x2, y1 - y2)
                dists.append(dist - (radii_flat[i] + radii_flat[j]))
        return np.array(dists)
    
    # Bounds: all radii must be >= 0
    bounds = [(0, 1) for _ in range(10)]
    
    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': lambda x: square_constraint(x)},
        {'type': 'ineq', 'fun': lambda x: circle_overlap_constraint(x)}
    ]
    
    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000}
    )
    
    # Update centers and radii once optimization is done
    radii = result.x
    # Update centers to fit radii (small perturbation to centers to avoid edge cases)
    for i in range(10):
        dx = (np.random.rand() - 0.5) * 0.05
        dy = (np.random.rand() - 0.5) * 0.05
        centers[i, 0] += dx
        centers[i, 1] += dy
        # Clamp centers to stay inside the square
        centers[i, 0] = np.clip(centers[i, 0], radii[i], 1 - radii[i])
        centers[i, 1] = np.clip(centers[i, 1], radii[i], 1 - radii[i])
    
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
