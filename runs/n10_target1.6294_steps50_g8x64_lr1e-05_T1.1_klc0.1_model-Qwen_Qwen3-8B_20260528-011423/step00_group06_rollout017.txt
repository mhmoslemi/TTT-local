import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective_function(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        # Reshape to centers and radii
        centers = x[:60].reshape((10, 2))
        radii = x[60:].reshape((10,))

        # Constraint 1: All circles are inside the unit square
        constraints = []
        for i in range(10):
            if radii[i] < 0:
                return float('inf')
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return float('inf')

        # Constraint 2: No two circles overlap
        dists = pdist(centers)
        for d, (i, j) in zip(dists, [(a, b) for a in range(10) for b in range(a+1, 10)]):
            if d < radii[i] + radii[j]:
                return float('inf')

        # Objective: maximize sum of radii, so minimize negative sum
        return -np.sum(radii)

    # Set up bounds for variables: x and y in [0, 1], radii in [0, 0.5]
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    
    # Optimizer settings
    result = minimize(objective_function, x0=np.ravel(np.vstack((centers, radii))),
                      bounds=bounds, method='SLSQP', tol=1e-6, options={'maxiter': 1000})
    
    # Extract results
    centers_result = result.x[:60].reshape((10, 2))
    radii_result = result.x[60:].reshape((10,))
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
