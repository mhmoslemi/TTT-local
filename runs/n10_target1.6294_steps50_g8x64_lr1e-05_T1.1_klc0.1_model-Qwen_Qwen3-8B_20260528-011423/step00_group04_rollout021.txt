import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.15

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        # Ensure centers are within [r, 1-r]
        center_bounds = np.array([radii[i] <= centers[i, 0] <= 1 - radii[i] for i in range(10)])
        return center_bounds

    def constraint_no_overlap(radii):
        # Ensure distance between centers >= sum of radii
        dist_matrix = cdist(centers, centers)
        dist_matrix[np.triu_indices_from(dist_matrix, 1)] = 0
        return dist_matrix - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization setup
    bounds = [(0, 0.5) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
