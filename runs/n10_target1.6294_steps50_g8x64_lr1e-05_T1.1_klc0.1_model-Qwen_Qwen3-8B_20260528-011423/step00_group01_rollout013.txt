import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 10 equally spaced circles in a grid
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    centers = np.vstack((centers, [0.5, 0.5]))  # add one more circle in the center
    radii = np.full(10, 0.1)  # initial radius

    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraints(radii, centers):
        constraints = []
        for i in range(10):
            r_i = radii[i]
            x_i, y_i = centers[i]
            # Ensure circle i is inside the unit square
            if r_i > 0:
                constraints.append({'type': 'ineq', 'fun': lambda r: r - x_i})
                constraints.append({'type': 'ineq', 'fun': lambda r: 1 - r - x_i})
                constraints.append({'type': 'ineq', 'fun': lambda r: r - y_i})
                constraints.append({'type': 'ineq', 'fun': lambda r: 1 - r - y_i})
            # Ensure circle i doesn't overlap with other circles
            for j in range(i + 1, 10):
                r_j = radii[j]
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraints.append({'type': 'ineq', 'fun': lambda r: dist - r_i - r_j})
        return constraints

    # Optimize
    res = minimize(objective, radii, args=(centers,), method='SLSQP', constraints=constraints(radii, centers), tol=1e-6)

    # Extract results
    centers_opt = np.array(centers)
    radii_opt = res.x
    sum_radii = -res.fun

    return (centers_opt, radii_opt, sum_radii)
