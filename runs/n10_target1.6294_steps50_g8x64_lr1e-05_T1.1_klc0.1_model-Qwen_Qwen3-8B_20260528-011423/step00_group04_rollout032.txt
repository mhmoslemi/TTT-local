import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: spread circles evenly
    centers = np.array([[0.1 * i, 0.1 * j] for i in range(10) for j in range(10)])
    centers = centers[:10]  # Take first 10
    radii = np.full(10, 0.1)  # Initial radius for each circle

    def objective(radii):
        return -np.sum(radii)  # Maximize the sum of radii

    def constraint_circle_inside(radii, centers):
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return 1.0
        return 0.0

    def constraint_no_overlap(radii, centers):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return 1.0
        return 0.0

    # Optimization bounds: radius >= 0 and <= 0.5
    bounds = [(0.0, 0.5) for _ in range(10)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
    ]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract final solution
    final_radii = result.x
    final_centers = centers  # Centers remain the same during optimization

    # Return required result
    return (final_centers, final_radii, -result.fun)
