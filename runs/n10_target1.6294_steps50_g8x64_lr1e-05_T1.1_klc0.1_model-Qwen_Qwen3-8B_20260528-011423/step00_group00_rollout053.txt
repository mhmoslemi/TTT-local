import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess for centers: spread out points in the square
    centers = np.random.rand(10, 2) * 0.8 + 0.1  # Avoid edges initially
    radii = np.full(10, 0.1)  # Initial guess for radii

    def objective(radii):
        return -sum(radii)  # We want to maximize the sum, so minimize the negative

    def constraints(radii, centers):
        # Bounds on radii
        bounds = [(0, 1 - np.min(centers, axis=0))]  # Radii must not exceed minimum margin
        bounds = [(0, min(1 - centers[i, 0], 1 - centers[i, 1])) for i in range(10)]  # Ensure circles fit in square

        # Distance constraints between centers
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_func(x):
        # x is a vector of (centers + radii)
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -sum(radii)

    def constraint_func(x):
        # x is a vector of (centers + radii)
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        for i in range(10):
            if radii[i] > 1 - np.min(centers[i]) or radii[i] > 1 - np.max(centers[i]):
                return np.inf  # Violates square boundary
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return np.inf  # Violates non-overlapping constraint
        return 0

    # Initial guess
    x0 = np.concatenate([centers.reshape(10 * 2), radii])

    # Optimization
    result = minimize(objective_func, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': constraint_func}, bounds=[(0, 1) for _ in range(60 + 10)])

    # Extract result
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return centers, radii, sum_radii
