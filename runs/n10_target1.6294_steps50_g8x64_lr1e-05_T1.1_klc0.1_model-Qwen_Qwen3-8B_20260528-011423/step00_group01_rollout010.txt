import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced centers
    centers = np.array([[i * 0.25 + 0.05, j * 0.25 + 0.05] for i in range(10) for j in range(10)])
    centers = centers[:10]  # Take first 10 centers
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are inside the unit square
        if np.any(radii < 0):
            return np.inf
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    res = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        bounds=[(0, 1) for _ in range(10)],
        options={'disp': False, 'ftol': 1e-9, 'maxiter': 1000}
    )

    centers = np.array([[i * 0.25 + 0.05, j * 0.25 + 0.05] for i in range(10) for j in range(10)][:10])
    radii = res.x
    sum_radii = -res.fun

    # Ensure all circles are inside the unit square
    for i in range(10):
        if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
            radii[i] = 1 - max(centers[i, 0], centers[i, 1])
            sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
