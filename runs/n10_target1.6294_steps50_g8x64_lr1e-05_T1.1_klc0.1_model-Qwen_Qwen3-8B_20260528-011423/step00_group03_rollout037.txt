import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: spread circles in a grid pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25

    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraint_bounds(radii):
        # Each circle must be fully inside the unit square
        return np.all(radii <= 0.5 - 1e-8)

    def constraint_overlap(radii):
        # No two circles can overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1
        return 0

    bounds = [(0, 0.5) for _ in range(10)]

    constraints = [
        {
            'type': 'ineq',
            'fun': constraint_bounds
        },
        {
            'type': 'ineq',
            'fun': constraint_overlap
        }
    ]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
