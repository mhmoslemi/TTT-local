import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds, LinearConstraint

def run_packing():
    # Initial guess for centers (random within [0,1]^2)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial radii small enough to not overlap

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum of radii to maximize it

    def constraints(radii, centers):
        # Non-overlapping constraint: distance between centers >= sum of radii
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                idx += 1
        return dists

    def bounds(radii):
        return [(0, 1 - 1e-6) for _ in range(len(radii))]  # Ensure circles fit inside the unit square

    def square_bounds(centers):
        lb = np.zeros_like(centers)
        ub = np.ones_like(centers)
        lb[:, 0] = radii
        lb[:, 1] = radii
        ub[:, 0] = 1 - radii
        ub[:, 1] = 1 - radii
        return Bounds(lb, ub)

    # Optimization setup
    res = optimize.minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        bounds=bounds(radii),
        constraints=[{'type': 'ineq', 'fun': lambda x: constraints(x, centers)}],
        method='SLSQP'
    )

    # Update centers to satisfy square bounds
    centers = np.clip(centers, radii, 1 - radii)

    # Return the result
    return (centers, res.x, -res.fun)
