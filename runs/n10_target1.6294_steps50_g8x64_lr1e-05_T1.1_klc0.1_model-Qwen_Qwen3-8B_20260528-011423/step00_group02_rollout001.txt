import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equal radii and equally spaced positions
    n_circles = 10
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    # Initial estimate: place circles in a square grid
    for i in range(n_circles):
        x = (i % 3) / 2.5 + 0.1
        y = (i // 3) / 2.5 + 0.1
        centers[i] = [x, y]
        radii[i] = 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        radii = radii.reshape((n_circles,))  # Ensure it's a 1D array
        constraints = []
        for i in range(n_circles):
            # Constraint for circle i to be inside the unit square
            r = radii[i]
            if r < 0:
                return False
            if r > 1 - 0.1:
                return False
            constraints.append((r, r, 0, 0, 1 - r, 1 - r))
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                r1 = radii[i]
                r2 = radii[j]
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < r1 + r2 - 1e-8:
                    return False
                constraints.append((r1 + r2, dist, 0, 0, 0, 0))
        return True

    # Bounds for optimization: radii >= 0
    bounds = [(0, 1) for _ in range(n_circles)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{
            'type': 'ineq',
            'fun': lambda x: constraints(x)
        }],
        bounds=bounds
    )

    optimal_radii = result.x
    sum_radii = np.sum(optimal_radii)

    # Reconstruct centers based on the optimal radii
    # For simplicity, we will keep initial centers but allow optimization
    # This is a basic approach; in practice, you would optimize centers too
    # However, this is a simplified version to demonstrate the idea
    return (centers, optimal_radii, sum_radii)
