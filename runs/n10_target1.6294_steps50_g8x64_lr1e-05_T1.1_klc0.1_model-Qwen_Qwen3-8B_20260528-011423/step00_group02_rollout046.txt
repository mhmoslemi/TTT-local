import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles at evenly spaced points
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraint_center_bounds(centers, radii):
        # All circles are fully inside the unit square
        return np.min(np.column_stack((centers - radii[:, np.newaxis], 1 - centers - radii[:, np.newaxis])), axis=1)

    def constraint_no_overlap(centers, radii):
        # No two circles overlap
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2).sum(axis=2))
        return np.min(dist - radii[:, np.newaxis] - radii[np.newaxis, :], axis=1)

    def objective_with_constraints(radii):
        centers = np.random.rand(10, 2)
        # Use a more reasonable initial guess for centers
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = 0.1 + i * 0.18
            centers[i, 1] = 0.5

        bounds = [(0, 0.5) for _ in range(10)]  # Set upper bound on radii for better optimization

        constraints = []
        # Add constraint for centers within square
        constraints.append({
            'type': 'ineq',
            'fun': lambda r: constraint_center_bounds(centers, r)
        })
        # Add constraint for no overlap
        constraints.append({
            'type': 'ineq',
            'fun': lambda r: constraint_no_overlap(centers, r)
        })

        result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')
        return result.x, -result.fun

    # Run optimization
    radii_opt, sum_radii_opt = objective_with_constraints(radii)

    # Adjust centers to satisfy constraints and bounds
    centers_opt = np.random.rand(10, 2)
    for i in range(10):
        centers_opt[i, 0] = 0.1 + i * 0.18
        centers_opt[i, 1] = 0.5

    # Return the result
    return centers_opt, radii_opt, sum_radii_opt
