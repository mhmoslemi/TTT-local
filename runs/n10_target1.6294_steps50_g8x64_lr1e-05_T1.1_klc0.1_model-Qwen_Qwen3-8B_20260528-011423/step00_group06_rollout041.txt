import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square
    def square_constraint(radii, centers):
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0]) - 1e-8

    # Constraints: circles must not overlap
    def no_overlap_constraint(radii, centers):
        n = len(radii)
        dx = centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0]
        dy = centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1]
        dists = np.sqrt(dx**2 + dy**2)
        return np.minimum(dists - (radii[:, np.newaxis] + radii[np.newaxis, :]), 1e-8)

    # Optimization problem
    def optimize_packing():
        bounds = Bounds(0, 1 - 1e-8)
        cons = []

        # Add square constraint
        cons.append({
            'type': 'ineq',
            'fun': lambda r: square_constraint(r, centers)
        })

        # Add non-overlapping constraint
        cons.append({
            'type': 'ineq',
            'fun': lambda r: no_overlap_constraint(r, centers)
        })

        result = optimize.minimize(
            objective,
            radii,
            method='SLSQP',
            bounds=bounds,
            constraints=cons,
            tol=1e-8
        )

        # Update centers based on the optimal radii
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        # Optional: refine centers with more advanced optimization (not implemented here)
        return (centers, result.x, -result.fun)

    centers, radii, sum_radii = optimize_packing()
    return (centers, radii, sum_radii)
