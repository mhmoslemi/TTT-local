import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints_func(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x.reshape(10, 3)[:, :2]
        radii = x.reshape(10, 3)[:, 2]
        # Ensure circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Ensure circles do not overlap
        dists = pdist(centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        for i, d in enumerate(dists):
            if d < np.min(radii_sum[i]):
                return np.inf
        return 0.0

    def constraints(x):
        return constraints_func(x)

    # Set up bounds: x and y in [0, 1], radii >= 0
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    result = minimize(objective, x0=np.hstack((centers.reshape(10, 2).ravel(), radii)), 
                      bounds=bounds, constraints={"type": "ineq", "fun": constraints}, 
                      method='SLSQP', options={"disp": False, "ftol": 1e-9, "maxiter": 1000})
    
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:30]
    sum_radii = -result.fun
    
    return (centers_result, radii_result, sum_radii)
