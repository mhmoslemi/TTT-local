import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles evenly in a grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.ones(10) * 0.1
    
    # Objective function to maximize the sum of radii (minimize negative sum)
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraints
    def constraint_center_inside(radii, centers):
        return np.all(radii <= 1 - centers[:, 0]) & np.all(radii <= 1 - centers[:, 1])
    
    def constraint_no_overlap(radii, centers):
        n = len(radii)
        distances = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                distances[i, j] = dist - (radii[i] + radii[j])
        return np.min(distances)
    
    def constraint_all_non_negative(radii):
        return np.all(radii >= 0)
    
    # Optimization problem
    bounds = [(0, 1 - 0.01) for _ in range(10)]  # Slight buffer to ensure circles are inside
    
    def constraint_center_inside_func(x):
        return constraint_center_inside(x, initial_centers)
    
    def constraint_no_overlap_func(x):
        return constraint_no_overlap(x, initial_centers)
    
    def constraint_all_non_negative_func(x):
        return constraint_all_non_negative(x)
    
    # Combine constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside_func},
        {'type': 'ineq', 'fun': constraint_no_overlap_func},
        {'type': 'ineq', 'fun': constraint_all_non_negative_func}
    ]
    
    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            lambda x: objective(x, initial_centers),
            initial_radii,
            bounds=bounds,
            constraints=constraints,
            method='SLSQP'
        )
    
    # Extract result
    radii = result.x
    centers = initial_centers
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
