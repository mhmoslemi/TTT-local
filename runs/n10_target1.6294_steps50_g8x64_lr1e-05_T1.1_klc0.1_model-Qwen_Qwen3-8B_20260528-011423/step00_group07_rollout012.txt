import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if (centers[i, 0] - radii[i] < 0) or (centers[i, 0] + radii[i] > 1):
                return False
            if (centers[i, 1] - radii[i] < 0) or (centers[i, 1] + radii[i] > 1):
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    def packing_func(x):
        # x is the vector of radii
        radii = x
        if not constraints(radii):
            return np.inf  # Penalize invalid configurations
        return -np.sum(radii)  # Negative sum for maximization
    
    # Optimize
    result = minimize(packing_func, radii, method='SLSQP', bounds=[(0, 1)] * 10)
    optimal_radii = result.x
    optimal_centers = centers.copy()
    
    # Check if the optimization succeeded and the constraints are satisfied
    if not constraints(optimal_radii):
        optimal_radii = np.array([0.1] * 10)
    
    # Return the result
    return (optimal_centers, optimal_radii, -result.fun)
