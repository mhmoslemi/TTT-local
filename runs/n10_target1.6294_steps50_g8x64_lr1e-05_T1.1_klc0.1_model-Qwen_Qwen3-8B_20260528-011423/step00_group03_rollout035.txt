import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(centers, radii):
        n = centers.shape[0]
        cons = []
        # Boundary constraint: each circle must be inside the unit square
        for i in range(n):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - (1 - centers[i, 0] - centers[i, 1])})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 0]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - (1 - centers[i, 1])})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - (1 - centers[i, 1] - centers[i, 0])})
        # Pairwise non-overlapping constraint
        for i in range(n):
            for j in range(i + 1, n):
                def dist_constraint(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - (radii[i] + radii[j])
                cons.append({'type': 'ineq', 'fun': dist_constraint})
        return cons

    def bounds(radii):
        return [(0, 1) for _ in range(10)]

    def objective_wrapper(radii):
        return -np.sum(radii)

    result = minimize(objective_wrapper, radii, method='SLSQP',
                      bounds=bounds(radii), constraints=constraints(centers, radii))
    
    centers_result = centers
    radii_result = result.x
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
