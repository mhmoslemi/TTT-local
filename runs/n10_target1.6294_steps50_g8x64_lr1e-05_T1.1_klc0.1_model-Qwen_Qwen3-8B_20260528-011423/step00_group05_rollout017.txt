import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles evenly spaced in a grid
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more center
    centers = centers[:10]  # Keep only 10 centers
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.array([r <= 1 - x - r and r <= x + r for x, r in zip(centers, radii)])

    def constraint_no_overlap(radii):
        dists = np.empty(0)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists = np.append(dists, dist - (radii[i] + radii[j]))
        return dists

    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1 - np.sqrt(2) * 0.1) for _ in range(10)]
    )

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
