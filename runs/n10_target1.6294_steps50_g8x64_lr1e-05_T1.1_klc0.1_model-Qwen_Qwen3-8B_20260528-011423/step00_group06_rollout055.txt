import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Each circle must be within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - np.min(centers[i])
                return False
        # No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx*dx + dy*dy)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    def constraint_func(x):
        # x is a flat array of [r0, r1, ..., r9]
        radii = x.reshape(10)
        return constraints(radii)

    def objective_func(x):
        # x is a flat array of [r0, r1, ..., r9]
        radii = x.reshape(10)
        return -np.sum(radii)

    # Initial guess for radii
    initial_radii = np.array([0.1] * 10)

    # Optimization setup
    result = minimize(objective_func, initial_radii, method='L-BFGS-B', bounds=[(0, 1)] * 10,
                      constraints={'type': 'ineq', 'fun': constraint_func},
                      tol=1e-6, options={'maxiter': 500, 'disp': False})

    radii_opt = result.x
    sum_radii_opt = -result.fun

    # Ensure all constraints are satisfied
    if not constraints(radii_opt):
        radii_opt = np.array([0.1] * 10)
        sum_radii_opt = 1.0

    # Format the output
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    return (centers, radii_opt, sum_radii_opt)
