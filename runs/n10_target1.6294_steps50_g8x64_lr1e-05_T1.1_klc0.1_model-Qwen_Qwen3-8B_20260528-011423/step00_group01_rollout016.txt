import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum
    
    def constraints_func(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        
        # Constraint 1: Circles must be inside the unit square
        inside = np.all((radii <= 1 - centers) & (centers <= 1 - radii), axis=1)
        if not np.all(inside):
            return -np.inf  # Penalize if any circle is outside
        
        # Constraint 2: Circles must not overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -np.inf  # Penalize if circles overlap
        
        return -np.sum(radii)
    
    x0 = np.concatenate([centers.flatten(), radii])
    bounds = [(0, 1) for _ in range(20)] + [(0, 0.5) for _ in range(10)]
    
    result = minimize(
        fun=constraints_func,
        x0=x0,
        bounds=bounds,
        method='SLSQP',
        options={'maxiter': 1000, 'ftol': 1e-9}
    )
    
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
