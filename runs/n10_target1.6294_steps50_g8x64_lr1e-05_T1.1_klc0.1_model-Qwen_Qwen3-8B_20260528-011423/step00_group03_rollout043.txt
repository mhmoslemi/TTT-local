```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: randomly placed circles
    np.random.seed(42)
    centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    radii = np.random.uniform(low=0.05, high=0.15, size=10)
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize the negative sum to maximize the sum
    
    # Constraints
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        inside = np.all((centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1) & \
                 np.all((1 - centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1)
        return np.sum(inside)  # Return 10 if all are inside
    
    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        dists = np.linalg.norm(centers_flat[:, np.newaxis] - centers_flat[np.newaxis, :], axis=2)
        sum_radii = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        return np.min(dists - sum_radii)  # Ensure min distance >= sum of radii
    
    # Bounds for each parameter
    bounds = []
    for _ in range(10):
        bounds.extend([(0.1, 0.9), (0.1, 0.9), (0.05, 0.15)])
    
    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}
        ],
        options={'ftol': 1e-6, 'maxiter': 1000}
    )
    
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun  # Since we maximized by minimizing negative sum
    
    return (centers_opt, radii_opt, sum_radii_opt)
```
