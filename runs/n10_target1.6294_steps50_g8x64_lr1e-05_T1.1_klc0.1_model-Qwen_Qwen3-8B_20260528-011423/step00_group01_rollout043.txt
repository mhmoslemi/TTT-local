import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(radii, centers):

    def constraints(centers, radii):
        # Constraint 1: All circles are inside the unit square
        lb = np.zeros(centers.shape)
        ub = np.ones(centers.shape)
        for i in range(10):
            lb[i] = radii[i]
            ub[i] = 1 - radii[i]
        lb = lb.reshape(-1)
        ub = ub.reshape(-1)
        return np.concatenate((lb, ub))

    def non_overlap(centers, radii):
        # Constraint 2: No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return dists - radii[:, np.newaxis] - radii[np.newaxis, :]

    def bounds_func(x):
        # Bounds for each center (x, y) and radius
        bounds = []
        for i in range(10):
            bounds.extend([(0, 1), (0, 1), (0, 1 - 0)])  # x, y, r
        return bounds

    # Initial guess: 10 circles arranged in a grid
    centers = np.array([(i / 3, j / 3) for i in range(3) for j in range(3)])
    radii = np.array([0.1] * 10)  # initial guess for radii

    # Flatten centers and radii for optimization
    x0 = np.concatenate((centers.reshape(-1), radii))

    # Optimization problem
    def objective_func(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)

    def constraint_func(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Constraint 1: circles inside the square
        lb = radii_flat
        ub = 1 - radii_flat
        lb = lb.reshape(-1)
        ub = ub.reshape(-1)
        # Constraint 2: no overlap
        dists = np.sqrt(np.sum((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2, axis=2))
        return dists - radii_flat[:, np.newaxis] - radii_flat[np.newaxis, :]

    # Setup constraints
    cons = ({'type': 'ineq', 'fun': lambda x: constraint_func(x)},
            {'type': 'ineq', 'fun': lambda x: constraint_func(x) + np.finfo(float).eps})

    # Optimization
    result = minimize(objective_func, x0, method='SLSQP', bounds=bounds_func(x0), constraints=cons, tol=1e-8)

    # Extract result
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
