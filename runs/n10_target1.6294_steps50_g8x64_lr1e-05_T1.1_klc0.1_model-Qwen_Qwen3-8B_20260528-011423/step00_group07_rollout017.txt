import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers_guess = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for j in range(5) for i in range(2)])
    radii_guess = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = 0.1 + (i // 5) * 0.2 + np.random.uniform(-0.05, 0.05)
            centers[i, 1] = 0.1 + (i % 5) * 0.2 + np.random.uniform(-0.05, 0.05)
        return np.array([radii[i] + radii[j] <= np.hypot(centers[i, 0] - centers[j, 0], centers[i, 1] - centers[j, 1]) for i in range(10) for j in range(i+1)])

    bounds = [(0, 0.5) for _ in range(10)]

    result = minimize(
        objective,
        radii_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    radii = result.x
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = np.random.uniform(radii[i], 1 - radii[i])
        centers[i, 1] = np.random.uniform(radii[i], 1 - radii[i])

    # Ensure circles are non-overlapping and fit inside the square
    # This is a simple heuristic and may not produce the optimal solution
    # For better results, a more sophisticated optimization is needed

    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
