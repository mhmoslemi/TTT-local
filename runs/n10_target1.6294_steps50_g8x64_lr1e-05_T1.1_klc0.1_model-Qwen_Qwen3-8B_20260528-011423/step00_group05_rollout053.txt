import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place 10 circles evenly spaced in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii, centers):
        constraints = []
        # Constraint: All circles are within the unit square
        for i in range(10):
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (1 - r - centers[i, 0])})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (1 - r - centers[i, 1])})
        # Constraint: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                def cons(x):
                    return np.linalg.norm(centers[i] - centers[j]) - (radii[i] + radii[j])
                constraints.append({'type': 'ineq', 'fun': cons})
        return constraints

    # Set up the optimization problem
    bounds = [(0, 1.0) for _ in range(10)]
    res = minimize(lambda r: objective(r, centers), radii, method='SLSQP',
                   bounds=bounds, constraints=constraints(radii, centers))

    # Extract results
    centers = np.array(res.x[:10].reshape(10, 2))
    radii = res.x[:10]
    sum_radii = -objective(radii, centers)

    # Ensure constraints are satisfied (for safety, adjust if necessary)
    for i in range(10):
        r = radii[i]
        if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
            raise ValueError("Not all circles are within the unit square")
    for i in range(10):
        for j in range(i + 1, 10):
            dist = np.linalg.norm(centers[i] - centers[j])
            if dist < radii[i] + radii[j]:
                raise ValueError("Two circles overlap")

    return centers, radii, sum_radii
