import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles placed in a grid pattern
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(centers, radii):
        cons = []
        # Boundary constraint: r <= x, y <= 1 - r
        for i in range(10):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 0]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 1]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 0] - r[i]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 1] - r[i]})
        # Distance constraint: distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                def dist_constraint(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.hypot(dx, dy)
                    return dist - (r[i] + r[j])
                cons.append({'type': 'ineq', 'fun': dist_constraint})
        return cons

    def wrapper(x):
        radii[:] = x
        return objective(x)

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(10)]
    result = minimize(wrapper, radii, bounds=bounds, constraints=constraints(centers, radii), method='SLSQP')
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
