import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.zeros((10, 2))
    for i in range(10):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]
    radii = np.full(10, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def bounds(radii):
        return [(0, 1 - np.min(centers, axis=0)), ] * 10

    def constraints(radii, centers):
        cons = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                cons.append((dist - (radii[i] + radii[j]), 0))
        return cons

    def objective_func(x):
        radii = x
        return -np.sum(radii)

    def constraint_func(x):
        radii = x
        cons = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                cons.append(dist - (radii[i] + radii[j]))
        return np.array(cons)

    # Optimize
    result = minimize(objective_func, radii, method='SLSQP',
                      constraints={'type': 'ineq', 'fun': constraint_func},
                      bounds=bounds(radii),
                      options={'ftol': 1e-8, 'disp': False})
    optimized_radii = result.x
    sum_radii = -result.fun

    # Ensure circles are within the unit square
    for i in range(10):
        r = optimized_radii[i]
        x, y = centers[i]
        if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
            optimized_radii[i] = min(r, 1 - np.min([x, 1 - x, y, 1 - y]))

    return (centers, optimized_radii, sum_radii)
