import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for circle centers (randomly placed)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Start with small random radii

    def objective(radii):
        # Objective is to minimize the negative sum of radii
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        # Ensure circles are inside the unit square
        x, y = centers[:, 0], centers[:, 1]
        r = radii
        return r - (x - r) * (1 - x - r) * (y - r) * (1 - y - r)

    def constraint_no_overlap(radii):
        # Ensure no two circles overlap
        r = radii
        distances = cdist(centers, centers)
        return distances - (r[:, np.newaxis] + r[np.newaxis, :])

    # Optimization bounds and constraints
    bounds = [(0, 0.5) for _ in range(10)]  # Max radius is 0.5
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Perform optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract the result
    centers_result = centers
    radii_result = result.x
    sum_radii_result = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii_result)
