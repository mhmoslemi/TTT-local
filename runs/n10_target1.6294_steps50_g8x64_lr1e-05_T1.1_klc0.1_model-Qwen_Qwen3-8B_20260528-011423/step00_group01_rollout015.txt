import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: 10 circles at random positions inside the unit square
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial radii small to avoid overlap

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraints: circle inside unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        # Constraints: no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Define bounds for radii
    bounds = [(0, 1) for _ in range(10)]

    # Setup optimization problem
    result = minimize(
        fun=lambda x: objective(x),
        x0=radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": lambda x: constraints(x)},
        method="SLSQP"
    )

    # Extract result
    radii_opt = result.x
    sum_radii_opt = -result.fun
    centers_opt = centers

    return (centers_opt, radii_opt, sum_radii_opt)
