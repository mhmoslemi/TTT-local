import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place 10 circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for j in range(5) for i in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraint 1: All circles are within the unit square
        def in_unit_square(r):
            return np.all((radii <= 1 - radii) & (radii >= 0))
        if not in_unit_square(radii):
            return np.inf

        # Constraint 2: No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        radii_sum = radii[np.newaxis, :] + radii[:, np.newaxis]
        overlap = np.any(dists < radii_sum - 1e-8)
        if overlap:
            return np.inf

        return 0

    # Optimization
    bounds = [(0, 1 - 1e-8) for _ in range(10)]  # Ensure radii don't exceed unit square
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    radii_opt = result.x
    sum_radii = -result.fun
    centers_opt = centers.copy()

    return (centers_opt, radii_opt, sum_radii)
