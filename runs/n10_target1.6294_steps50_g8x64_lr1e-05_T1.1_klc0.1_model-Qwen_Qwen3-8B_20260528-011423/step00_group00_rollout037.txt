import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(radii_and_centers):
        centers = radii_and_centers[10:]
        radii = radii_and_centers[:10]
        constraints = []
        # Box constraints: 0 <= r <= 0.5
        for i in range(10):
            constraints.append({'type': 'ineq', 'fun': lambda x: x[i]})
            constraints.append({'type': 'ineq', 'fun': lambda x: 0.5 - x[i]})
        # Boundary constraints: r <= x - 0 and 1 - x - r >= 0
        for i in range(10):
            for j in range(10):
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.sqrt(dx*dx + dy*dy)
                    constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: dist - (x[i] + x[j])})
        return constraints

    def bounds():
        return [(0, 0.5)] * 10 + [(0, 1)] * 20

    def initial_guess():
        # Initial guess: 10 circles arranged in a grid
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        radii = np.full(10, 0.05)
        return np.concatenate((radii, centers.flatten()))

    # Initial guess
    x0 = initial_guess()

    # Define problem
    res = minimize(objective, x0, method='SLSQN', jac=None, constraints=constraints(x0), bounds=bounds())

    # Extract result
    radii = res.x[:10]
    centers = res.x[10:].reshape(10, 2)

    # Calculate sum of radii
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
