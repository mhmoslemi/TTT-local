import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: 10 circles evenly spaced
    centers = np.array([[0.1 + i * 0.2, 0.1] for i in range(10)])
    radii = np.array([0.05] * 10)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraint_circle_inside(radii, centers):
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -1
        return 0
    
    def constraint_no_overlap(radii, centers):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0
    
    bounds = [(0, 1) for _ in range(10)]
    
    # Prepare initial guess
    x0 = np.concatenate((centers.flatten(), radii))
    
    # Define objective function with radii
    def objective_with_radii(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return -np.sum(radii)
    
    # Define constraints
    def constraint_circle_inside_with_x(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1
        return 0
    
    def constraint_no_overlap_with_x(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0
    
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside_with_x},
        {'type': 'ineq', 'fun': constraint_no_overlap_with_x}
    ]
    
    result = minimize(objective_with_radii, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-8)
    
    final_centers = result.x[:20].reshape(10, 2)
    final_radii = result.x[20:]
    final_sum_radii = np.sum(final_radii)
    
    return (final_centers, final_radii, final_sum_radii)
