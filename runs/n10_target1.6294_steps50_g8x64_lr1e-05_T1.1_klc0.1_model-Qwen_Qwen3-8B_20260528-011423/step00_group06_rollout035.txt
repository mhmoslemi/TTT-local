import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles at evenly spaced grid points
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 3) for y in np.linspace(0.1, 0.9, 3)])
    radii = np.full(9, 0.1)  # 9 circles for the 3x3 grid (excluding one to make 10 circles)
    radii = np.append(radii, 0.1)  # Add one more circle with small radius for initial guess

    def objective(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:2*10].reshape(10, 2)
        radii_flat = x[2*10:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers_flat = x[:2*10].reshape(10, 2)
        radii_flat = x[2*10:]
        constraints = []

        # Constraint: all circles must be inside the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append(x_i - r_i >= 0)
            constraints.append(1 - x_i - r_i >= 0)
            constraints.append(y_i - r_i >= 0)
            constraints.append(1 - y_i - r_i >= 0)

        # Constraint: no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.sqrt(dx*dx + dy*dy)
                constraints.append(dist >= radii_flat[i] + radii_flat[j])

        return constraints

    # Combine into a single constraint function for scipy.optimize
    def constraint_func(x):
        return constraints(x)

    # Initial guess for optimization
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Set bounds for each variable: center coordinates [0, 1], radii [0, 0.5]
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # x coordinate
        bounds.append((0, 1))  # y coordinate
        bounds.append((0, 0.5))  # radius

    # Setup for optimization
    res = minimize(objective, initial_guess, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_func})

    # Extract results
    best_centers = res.x[:2*10].reshape(10, 2)
    best_radii = res.x[2*10:]
    best_sum = -res.fun

    return (best_centers, best_radii, best_sum)
