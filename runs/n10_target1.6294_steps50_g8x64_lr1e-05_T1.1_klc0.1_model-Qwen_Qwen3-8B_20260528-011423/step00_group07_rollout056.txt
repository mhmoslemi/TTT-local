import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii, centers):
        # Check if all circles are inside the unit square
        x, y = centers[:, 0], centers[:, 1]
        return (x - radii) >= 0.0 and (1.0 - x - radii) >= 0.0 and (y - radii) >= 0.0 and (1.0 - y - radii) >= 0.0

    def constraint_no_overlap(radii, centers):
        # Check if all circles are non-overlapping
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 + 
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        return dists >= radii[:, np.newaxis] + radii[np.newaxis, :]

    def constraint_all_non_negative(radii):
        return radii >= 0.0

    def bounds(radii):
        return [(0.0, 1.0) for _ in range(10)]

    # Setup optimization problem
    n = 10
    x0 = radii

    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_all_non_negative(r)}
    ]

    res = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    if not res.success:
        raise ValueError("Optimization failed.")

    centers_opt = centers.copy()
    radii_opt = res.x
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
