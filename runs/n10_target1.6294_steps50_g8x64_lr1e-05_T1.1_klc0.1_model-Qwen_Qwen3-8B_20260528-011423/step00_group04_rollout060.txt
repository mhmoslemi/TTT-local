import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as dist

def run_packing():
    np.random.seed(42)
    initial_centers = np.random.rand(10, 2)
    initial_radii = np.random.rand(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dist_centers = dist.pdist([centers[i], centers[j]])
                constraints.append(dist_centers - (radii[i] + radii[j]))
        return np.array(constraints)

    def bounds(radii):
        return [(0, 1 - 1e-6) for _ in radii]

    def objective_func(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)

    def constraint_func(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dist_centers = dist.pdist([centers[i], centers[j]])
                constraints.append(dist_centers - (radii[i] + radii[j]))
        return np.array(constraints)

    def bounds_func(x):
        return [(0, 1 - 1e-6) for _ in x]

    x0 = np.random.rand(60 + 10)
    x0[:60] = initial_centers.reshape(60)
    x0[60:] = initial_radii

    result = minimize(objective_func, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': constraint_func}, bounds=bounds_func, tol=1e-8)

    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -objective_func(result.x)

    return (centers, radii, sum_radii)
