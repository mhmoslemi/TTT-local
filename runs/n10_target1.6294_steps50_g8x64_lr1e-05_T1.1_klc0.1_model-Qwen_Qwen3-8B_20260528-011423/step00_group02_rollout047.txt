import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (random points inside the unit square)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Initial guess for radii

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            r = radii[i]
            if r > 1 - 1e-6:  # Allow a small tolerance
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-6:  # Allow a small tolerance
                    return False
        return True

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{"type": "ineq", "fun": lambda x: constraints(x)}],
        options={'disp': False}
    )

    radii_opt = result.x
    sum_radii = -result.fun

    # Ensure all circles are within the square
    for i in range(10):
        r = radii_opt[i]
        centers[i, 0] = max(1e-6, min(1 - 1e-6, centers[i, 0]))
        centers[i, 1] = max(1e-6, min(1 - 1e-6, centers[i, 1]))

    return (centers, radii_opt, sum_radii)
