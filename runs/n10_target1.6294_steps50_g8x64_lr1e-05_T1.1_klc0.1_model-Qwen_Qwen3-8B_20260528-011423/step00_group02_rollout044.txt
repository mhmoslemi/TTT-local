import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 10 circles
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    centers = centers[:10]  # Take first 10 points

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints_func(x):
        # x contains all parameters: centers and radii
        # Unpack variables
        centers = x[:10*2].reshape(10, 2)
        radii = x[10*2:]

        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist_sq = dx ** 2 + dy ** 2
                if dist_sq < (radii[i] + radii[j]) ** 2:
                    return np.inf

        # All constraints are satisfied
        return 0

    # Flatten centers and radii into a single vector for optimization
    initial_guess = np.concatenate((centers.ravel(), np.zeros(10)))

    # Define constraints for optimization
    constraints = ({
        'type': 'ineq',
        'fun': constraints_func
    })

    # Bounds for radii: must be non-negative
    bounds = [(0, 1) for _ in range(10)]

    # Optimize
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract result
    centers_opt = result.x[:10*2].reshape(10, 2)
    radii_opt = result.x[10*2:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
