import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: spread circles evenly in the unit square
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial small radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(circles):
        # Extract centers and radii
        centers = circles[:10, :2]
        radii = circles[10:]

        # Constraint 1: All circles are within the unit square
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False

        return True

    # Flatten centers and radii into a single vector for optimization
    x0 = np.r_[centers.reshape(-1), radii]

    # Define bounds for centers (0 to 1) and radii (0 to 0.5)
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])
    for _ in range(10):
        bounds.extend([(0.0, 0.5)])

    # Set up optimization problem
    def wrapped_objective(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return -np.sum(radii)

    def wrapped_constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        if not constraints([centers, radii]):
            return np.inf
        return 0

    # Optimization settings
    options = {'disp': False, 'ftol': 1e-8, 'maxiter': 1000}
    result = minimize(wrapped_objective, x0, method='SLSQP', bounds=bounds,
                      constraints={'type': 'eq', 'fun': wrapped_constraints},
                      options=options)

    # Extract final solution
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
