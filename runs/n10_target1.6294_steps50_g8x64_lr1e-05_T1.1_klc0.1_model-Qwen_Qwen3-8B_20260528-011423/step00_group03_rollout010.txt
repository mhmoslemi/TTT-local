import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: centers evenly spaced in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
    radii = np.ones(10) * 0.1  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_with_constraints(x):
        radii = x
        if not constraints(radii):
            return np.inf
        return -np.sum(radii)

    result = minimize(objective_with_constraints, radii, method='Nelder-Mead', bounds=[(0, 1) for _ in range(10)])
    if not result.success:
        return centers, radii, np.sum(radii)
    optimal_radii = result.x
    optimal_centers = centers.copy()

    return optimal_centers, optimal_radii, np.sum(optimal_radii)
