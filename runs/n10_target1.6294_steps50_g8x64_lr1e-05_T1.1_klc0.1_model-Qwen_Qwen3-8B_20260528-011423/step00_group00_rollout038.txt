import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = 0.1 + 0.2 * (i % 5)
        centers[i, 1] = 0.1 + 0.2 * (i // 5)
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: All circles are fully inside the unit square
        for r in radii:
            if not (r <= 1 - r):
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Set up optimization problem
    bounds = [(0, 1 - 1e-6) for _ in range(10)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=lambda x: {'type': 'ineq', 'fun': lambda x: 1 - constraints(x)})

    # Extract results
    radii_opt = result.x
    sum_radii = -result.fun
    centers_opt = centers.copy()

    return (centers_opt, radii_opt, sum_radii)
