import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced centers
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_inside(radii):
        for i in range(10):
            if radii[i] < 0:
                return -np.inf
            if radii[i] + centers[i, 0] > 1 or radii[i] + centers[i, 1] > 1:
                return -np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 1] - radii[i] < 0:
                return -np.inf
        return 0

    def constraint_overlap(radii):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    # Set up constraints for optimization
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimize using sequential least squares programming
    try:
        result = minimize(
            fun=objective,
            x0=radii,
            method='SLSQP',
            constraints=cons,
            options={'ftol': 1e-8, 'maxiter': 1000}
        )
        radii = result.x
    except:
        radii = np.full(10, 0.1)

    # Ensure non-negative radii
    radii = np.maximum(radii, 0)

    # Re-check constraints for final validation
    for i in range(10):
        if radii[i] < 0 or radii[i] + centers[i, 0] > 1 or radii[i] + centers[i, 1] > 1 or centers[i, 0] - radii[i] < 0 or centers[i, 1] - radii[i] < 0:
            radii[i] = 0

    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
