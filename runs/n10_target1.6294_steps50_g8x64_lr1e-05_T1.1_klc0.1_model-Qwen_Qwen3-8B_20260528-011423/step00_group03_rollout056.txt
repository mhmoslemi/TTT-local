import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: randomly placed circles with small radii
    np.random.seed(0)
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.random.uniform(0.05, 0.1, 10)
    
    # Objective function to minimize: negative sum of radii
    def objective(x):
        # Reshape x into centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize -sum => maximize sum

    # Constraints: circle must be within the unit square
    def constraint_square(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.concatenate([
            centers_flat[:, 0] - radii_flat,
            1 - centers_flat[:, 0] - radii_flat,
            centers_flat[:, 1] - radii_flat,
            1 - centers_flat[:, 1] - radii_flat
        ])

    # Constraint: circles must not overlap
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distances = cdist(centers_flat, centers_flat)
        radii_sum = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        return distances - radii_sum

    # Initial x: flatten centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: centers must be in [0.1, 0.9], radii >= 0
    bounds = [(0.1, 0.9) for _ in range(60)] + [(0, 1) for _ in range(10)]

    # Constraints: square and overlap
    cons = [
        {'type': 'ineq', 'fun': constraint_square},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    # Extract result
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
