import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: centers spread out evenly in the square, radii initialized to 0.1
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i, j in itertools.product(range(5), repeat=2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints_func(x):
        # x is a vector of 20 elements: 10 centers (each 2D) and 10 radii
        # Unpack centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]

        # Constraint 1: All circles are inside the unit square
        in_square = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                          (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii))
        if not in_square:
            return np.inf

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return np.inf

        return 0.0

    # Flatten centers and radii for optimization
    x0 = np.concatenate([centers.ravel(), radii])

    # Define bounds for x (centers and radii)
    bounds = []
    for i in range(10):
        for j in range(2):
            bounds.append((0.0, 1.0))  # center coordinates
        bounds.append((0.0, 1.0))  # radius

    # Define the constraint as an inequality
    constraints = [{'type': 'ineq', 'fun': constraints_func}]

    # Run the optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-6)

    # Extract results
    x_result = result.x
    centers_result = x_result[:20].reshape(10, 2)
    radii_result = x_result[20:]
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
