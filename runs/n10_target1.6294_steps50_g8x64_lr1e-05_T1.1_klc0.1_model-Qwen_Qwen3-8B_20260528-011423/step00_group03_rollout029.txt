import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced in a grid
    centers = np.array([[i * 0.25 + 0.125, j * 0.25 + 0.125] for i in range(10) for j in range(1)])
    centers = centers.reshape(-1, 2)
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - 0.0001:  # Tiny buffer for floating point precision
                return False
        # Check overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:  # Small buffer for numerical stability
                    return False
        return True

    # Optimization with constraints
    def constraint_func(radii_vec):
        radii = radii_vec
        if not constraints(radii):
            return np.inf
        return 0

    # Define bounds for the radii
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_func},
        options={'ftol': 1e-9, 'disp': False}
    )

    # Extract results
    optimal_radii = result.x
    optimal_centers = centers.copy()
    total_sum = np.sum(optimal_radii)

    return (optimal_centers, optimal_radii, total_sum)
