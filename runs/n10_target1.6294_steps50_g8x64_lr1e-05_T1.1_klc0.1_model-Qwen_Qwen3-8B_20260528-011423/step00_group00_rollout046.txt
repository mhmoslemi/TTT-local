```python
import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (randomly placed in the unit square)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Initial small radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraints: each circle is inside the unit square
        in_square = np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]), axis=1)
        # Constraints: no overlapping between circles
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        non_overlapping = np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :], axis=1)
        return np.concatenate((in_square, non_overlapping))

    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimization problem
    result = minimize(objective, radii, method='SLSQP', bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints})

    centers_result = centers
    radii_result = result.x
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
```
