```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    centers = np.vstack([centers[:10], np.array([0.5, 0.5])])
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Constraints
    def constraint_circle_inside(x):
        # x has shape (20,)
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return np.min(radii) >= 0

    def constraint_no_overlap(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        distances = cdist(centers, centers)
        radius_sum = radii.reshape(-1, 1) + radii.reshape(1, -1)
        return np.min(distances - radius_sum)

    # Bounds: centers in [0,1], radii >= 0
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]

    # Constraints dictionary
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap},
    ]

    # Initial guess in flat format
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimize
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8,
        options={'maxiter': 500}
    )

    # Extract result
    final_centers = result.x[:20].reshape(10, 2)
    final_radii = result.x[20:]
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
```
