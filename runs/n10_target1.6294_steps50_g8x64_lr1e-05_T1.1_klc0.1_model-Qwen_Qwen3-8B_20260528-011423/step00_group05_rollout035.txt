import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: place circles in grid
    centers = np.array([[i / 3 + 0.1, j / 3 + 0.1] for j in range(3) for i in range(3)])
    centers = np.vstack((centers, [0.5, 0.1], [0.5, 0.9], [0.1, 0.5], [0.9, 0.5]))
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint: each circle is inside the square
        r = radii.reshape(-1, 1)
        inside = np.all((r <= 1 - r) & (r >= r), axis=1)
        if not np.all(inside):
            return float('inf')
        
        # Constraint: no overlapping between circles
        dist = pdist(centers)
        r_sum = np.outer(radii, radii).flatten()
        overlap = np.any(dist < r_sum)
        if overlap:
            return float('inf')
        return 0

    def bounds(radii):
        return [(0, 1 - 0.1) for _ in range(10)]

    result = minimize(objective, radii, method='SLSQP',
                      constraints={'type': 'eq', 'fun': constraints},
                      bounds=bounds, tol=1e-6)

    centers = np.vstack((centers, [0.5, 0.5]))
    centers[-1] = [0.5, 0.5]
    if result.success:
        radii = result.x
        sum_radii = np.sum(radii)
    else:
        radii = np.full(10, 0.1)
        sum_radii = 1.0

    return (centers, radii, sum_radii)
