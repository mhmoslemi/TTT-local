import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for circle centers and radii
    # Start with small radii and spread out centers
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(x):
        # Flatten x into centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # All circles must be inside the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False
        # No two circles overlap
        dists = cdist(centers_flat, centers_flat)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii_flat[i] + radii_flat[j]:
                    return False
        return True

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Optimization bounds
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x and y coordinates
        bounds.extend([(0, 0.5)])       # radius (limited to 0.5 to stay in square)

    # Run optimization
    result = minimize(
        objective,
        x0,
        bounds=bounds,
        constraints={"type": "soft_bound", "fun": constraints},
        method="SLSQP"
    )

    if not result.success:
        raise RuntimeError("Optimization failed to find a valid packing")

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun  # Objective was negative of sum

    return (centers_opt, radii_opt, sum_radii_opt)
