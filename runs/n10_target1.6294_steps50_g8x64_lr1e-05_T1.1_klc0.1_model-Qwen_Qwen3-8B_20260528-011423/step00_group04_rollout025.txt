import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: evenly spaced grid of 10 circles
    centers = np.zeros((10, 2))
    radii = np.zeros(10)
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
        radii[i] = 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[::3]
        radii_flat = x[2::3]

        # Constraint 1: Each circle is inside the unit square
        for i in range(10):
            r = radii_flat[i]
            x_i = centers_flat[2*i]
            y_i = centers_flat[2*i + 1]
            if not (r <= x_i <= 1 - r and r <= y_i <= 1 - r):
                return False

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[2*i] - centers_flat[2*j]
                dy = centers_flat[2*i + 1] - centers_flat[2*j + 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 1e-6:
                    return False

        return True

    def objective_func(x):
        centers_flat = x[::3]
        radii_flat = x[2::3]
        return -np.sum(radii_flat)

    def bounds_func(x):
        # Bounds for each circle: [x, y, r]
        # x and y range from 0 to 1, r from 0 to 0.5 (since r <= 1 - r => r <= 0.5)
        bounds = []
        for i in range(10):
            bounds.append((0, 1))  # x
            bounds.append((0, 1))  # y
            bounds.append((0, 0.5))  # r
        return bounds

    # Flatten the initial guess
    x0 = []
    for i in range(10):
        x0.append(centers[i, 0])
        x0.append(centers[i, 1])
        x0.append(radii[i])
    x0 = np.array(x0)

    # Set up optimization
    result = minimize(
        objective_func,
        x0,
        method='SLSQP',
        bounds=bounds_func(x0),
        constraints={
            'type': 'ineq',
            'fun': constraints,
            'jac': None
        },
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    # Extract the result
    x_opt = result.x
    centers_opt = x_opt[::3].reshape(10, 2)
    radii_opt = x_opt[2::3]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
