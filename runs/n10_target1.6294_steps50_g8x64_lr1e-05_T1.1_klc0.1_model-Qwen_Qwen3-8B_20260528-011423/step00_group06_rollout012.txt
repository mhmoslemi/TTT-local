import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii):
        return np.array([r <= 0.5 for r in radii])  # Ensures r <= 0.5 for centers to fit in [r, 1-r]

    def constraint_no_overlap(radii):
        n = len(radii)
        dist = np.zeros(n)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i, j] = math.hypot(dx, dy)
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization problem setup
    bounds = [(0, 0.5) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_center_inside(r)},
        {'type': 'ineq', 'fun': lambda r: np.max(constraint_no_overlap(r))}
    ]

    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-5,
        options={'disp': False}
    )

    # Extract results
    best_radii = result.x
    best_centers = centers.copy()
    sum_radii = np.sum(best_radii)

    return (best_centers, best_radii, sum_radii)
