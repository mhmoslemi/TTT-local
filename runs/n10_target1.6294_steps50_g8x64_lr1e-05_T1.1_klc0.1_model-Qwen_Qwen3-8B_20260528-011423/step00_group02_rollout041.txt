import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)
    
    # Objective function to minimize (negative sum of radii)
    def objective(x):
        radii = x[::2]
        centers = x[1::2].reshape(-1, 2)
        return -np.sum(radii)
    
    # Constraints
    def constraint1(x):
        radii = x[::2]
        centers = x[1::2].reshape(-1, 2)
        return np.array([r <= 1 - r for r in radii])  # r <= 1 - r => 2r <= 1 => r <= 0.5
        
    def constraint2(x):
        radii = x[::2]
        centers = x[1::2].reshape(-1, 2)
        dists = np.abs(centers[:, np.newaxis, :] - centers[np.newaxis, :, :])
        dists = np.sqrt(np.sum(dists**2, axis=2))
        return np.array([dists[i, j] >= radii[i] + radii[j] for i in range(10) for j in range(i+1)])
    
    # Bounds
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 0.5), (0, 1), (0, 1)])
    
    # Set up optimization problem
    cons1 = {'type': 'ineq', 'fun': constraint1}
    cons2 = {'type': 'ineq', 'fun': constraint2}
    cons = [cons1, cons2]
    
    # Run optimization
    result = minimize(objective, x0=np.concatenate([initial_radii, initial_centers.flatten()]),
                      method='SLSQP', bounds=bounds, constraints=cons, options={'maxiter': 1000})
    
    # Extract result
    x = result.x
    radii = x[::2]
    centers = x[1::2].reshape(-1, 2)
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
