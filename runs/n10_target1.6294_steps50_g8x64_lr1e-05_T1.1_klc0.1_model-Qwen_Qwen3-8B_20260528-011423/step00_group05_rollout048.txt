import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.array([(i*0.25 + 0.125, j*0.25 + 0.125) for i in range(2) for j in range(5)])
    radii = np.array([0.1] * 10)

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: all circles inside the unit square
    def square_constraints(radii):
        return np.concatenate([radii, radii])

    # Constraints: no two circles overlap
    def no_overlap_constraints(radii):
        centers = np.reshape(np.array([0.1 + i*0.25 for i in range(10)] + [0.1 + j*0.25 for j in range(10)]), (10, 2))
        dist = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                idx += 1
        return dist

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: r},
            {'type': 'ineq', 'fun': lambda r: np.array([1 - r[i] - r[i] for i in range(10)])},
            {'type': 'ineq', 'fun': lambda r: no_overlap_constraints(r)},
        ],
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    centers = np.reshape(np.array([0.1 + i*0.25 for i in range(10)] + [0.1 + j*0.25 for j in range(10)]), (10, 2))
    sum_radii = result.fun * -1
    radii = result.x

    return (centers, radii, sum_radii)
