import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            x, y = np.random.uniform(0.25, 0.75, 2)
            centers[i] = x, y

        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    bounds = [(0.0, 0.5) for _ in range(10)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: r[0]},
            {'type': 'ineq', 'fun': lambda r: r[1]},
            {'type': 'ineq', 'fun': lambda r: r[2]},
            {'type': 'ineq', 'fun': lambda r: r[3]},
            {'type': 'ineq', 'fun': lambda r: r[4]},
            {'type': 'ineq', 'fun': lambda r: r[5]},
            {'type': 'ineq', 'fun': lambda r: r[6]},
            {'type': 'ineq', 'fun': lambda r: r[7]},
            {'type': 'ineq', 'fun': lambda r: r[8]},
            {'type': 'ineq', 'fun': lambda r: r[9]},
        ]
    )

    if result.success:
        # Use a better heuristic for centers based on optimized radii
        centers = np.zeros((10, 2))
        for i in range(10):
            x = np.random.uniform(0.25, 0.75)
            y = np.random.uniform(0.25, 0.75)
            centers[i] = x, y
        sum_radii = result.fun
        return (centers, np.abs(result.x), sum_radii)
    else:
        # Fallback to initial guess
        sum_radii = np.sum(radii)
        return (centers, radii, sum_radii)
