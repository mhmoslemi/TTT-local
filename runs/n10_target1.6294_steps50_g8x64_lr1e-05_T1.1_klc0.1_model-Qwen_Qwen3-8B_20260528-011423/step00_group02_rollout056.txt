from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    radii = np.full(10, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii, centers):
        # Constraint: radii >= 0
        cons = []
        for i in range(10):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        # Constraint: circles are inside the square
        for i in range(10):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i, c=centers[i]: 1 - r[i] - c[0]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i, c=centers[i]: 1 - r[i] - c[1]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i, c=centers[i]: r[i] - c[0]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i, c=centers[i]: r[i] - c[1]})
        # Constraint: circles do not overlap
        dists = pdist(centers)
        for i in range(10):
            for j in range(i + 1, 10):
                idx = squareform(dists)[i, j]
                cons.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, idx=idx: r[i] + r[j] - np.sqrt(idx)})
        return cons

        bounds = [(0, 1.0) for _ in range(10)]
        result = minimize(
            fun=lambda x: objective(x, centers),
            x0=radii,
            bounds=bounds,
            constraints=constraints(radii, centers),
            method='SLSQP',
            tol=1e-6
        )
        return result
