import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: spread out centers randomly in the unit square
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1  # Start with small radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(centers, radii):
        # Constraint: all circles are inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii), axis=1)
        if not np.all(inside):
            return np.inf  # Penalize if any circle is outside

        # Constraint: no two circles overlap
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        overlaps = (dists < radii[:, np.newaxis] + radii[np.newaxis, :])
        if np.any(overlaps):
            return np.inf  # Penalize if any circles overlap

        return 0

    def objective_with_constraints(radii_flat):
        nonlocal centers
        radii = radii_flat.reshape(10)
        obj = objective(radii)
        con = constraints(centers, radii)
        return obj + 1000 * con  # Add penalty for constraint violations

    # Define bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(10)]

    # Initial guess for radii
    initial_guess = np.random.rand(10)

    # Optimization
    result = minimize(objective_with_constraints, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-6)

    # Extract results
    radii_opt = result.x
    centers_opt = centers  # We fix centers to avoid too many variables

    # Check if optimization succeeded
    if result.success:
        sum_radii = np.sum(radii_opt)
    else:
        sum_radii = np.sum(radii)

    return (centers_opt, radii_opt, sum_radii)
