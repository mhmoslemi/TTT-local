import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: randomly placed circles
    np.random.seed(42)
    centers = np.random.uniform(
        low=np.array([0.1, 0.1]), 
        high=np.array([0.9, 0.9]),
        size=(10, 2)
    )
    radii = np.random.uniform(low=0.05, high=0.1, size=10)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii):
        return np.sum((radii <= 1 - centers[:, 0]) * (radii <= 1 - centers[:, 1]))

    def constraint_no_overlap(radii):
        dist = np.sqrt(np.sum((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2, axis=2))
        return np.sum((dist >= radii[:, np.newaxis] + radii[np.newaxis, :]) * 1)

    def constraint_box(radii):
        return np.sum((radii <= 1 - centers[:, 0]) * (radii <= 1 - centers[:, 1]))

    # Define bounds for radii
    bounds = [(0.0, 1.0 - np.min(centers, axis=0))] * 10

    # Set up constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_box},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Run optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii = -result.fun

    return (centers_final, radii_final, sum_radii)
