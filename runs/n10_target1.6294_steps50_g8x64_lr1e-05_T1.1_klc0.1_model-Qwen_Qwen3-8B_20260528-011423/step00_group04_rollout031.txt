import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            x, y = centers[i]
            if radii[i] < 0:
                return False
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Initial constraints
    if not constraints(radii):
        return np.array([]), np.array([]), 0.0

    # Optimize with bounds and constraints
    bounds = [(0, 0.5) for _ in range(10)]

    def wrapper(x):
        radii = x
        if not constraints(radii):
            return np.inf
        return -np.sum(radii)

    result = minimize(wrapper, radii, bounds=bounds, method='L-BFGS-B', options={'ftol': 1e-9, 'maxiter': 1000})

    # Reconstruct centers based on optimized radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    sum_radii = -result.fun
    return centers, result.x, sum_radii
