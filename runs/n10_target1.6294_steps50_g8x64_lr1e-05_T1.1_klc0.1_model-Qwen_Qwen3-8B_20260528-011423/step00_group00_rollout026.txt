import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    radii *= 0.15  # Initial guess for radii is small to avoid overlap

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:].reshape(10)
        return -np.sum(radii_flat)

    # Constraints
    def constraint_centers(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:].reshape(10)
        return np.concatenate([centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat,
                               centers_flat[:, 1] - radii_flat, 1 - centers_flat[:, 1] - radii_flat])

    def constraint_overlap(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:].reshape(10)
        dist = np.sqrt(((centers_flat[:, np.newaxis] - centers_flat[np.newaxis, :]) ** 2).sum(axis=2))
        return dist - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Bounds for parameters
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1)] * 2)  # for centers
        bounds.extend([(0, 0.5)] * 1)  # for radii (maximum radius is 0.5)

    # Initial parameter vector
    initial_params = np.concatenate([centers.flatten(), radii.flatten()])

    # Optimization with constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_centers},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(objective, initial_params, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract result
    final_centers = result.x[:60].reshape(10, 2)
    final_radii = result.x[60:].reshape(10)
    final_sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, final_sum_radii)
