import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(radii):
        return -sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.zeros((10, 2))
        # Initialize centers in a grid to start with
        for i in range(10):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
        # Constraint: each circle is within the unit square
        for i in range(10):
            r_i = radii[i]
            if not (r_i <= centers[i, 0] <= 1 - r_i and r_i <= centers[i, 1] <= 1 - r_i):
                return np.inf
        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -sum(radii)

    # Initial guess: uniform radii
    initial_radii = np.full(10, 0.15)
    # Set bounds for each radius
    bounds = [(0.0, 0.5) for _ in range(10)]

    # Use SLSQP method for constrained optimization
    result = minimize(objective, initial_radii, bounds=bounds, constraints={"type": "eq", "fun": constraints})

    centers = np.zeros((10, 2))
    for i in range(10):
        # Move centers to avoid overlap (simple grid-based movement)
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    # Use final radii from optimization
    radii = result.x
    # Final check for validity
    for i in range(10):
        r_i = radii[i]
        if not (r_i <= centers[i, 0] <= 1 - r_i and r_i <= centers[i, 1] <= 1 - r_i):
            radii[i] = 0.0
    # Recheck for overlap
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < radii[i] + radii[j]:
                radii[i] = max(0.0, radii[i] - (dist - radii[j]) / 2)
                radii[j] = max(0.0, radii[j] - (dist - radii[i]) / 2)
    sum_radii = sum(radii)

    return (centers, radii, sum_radii)
