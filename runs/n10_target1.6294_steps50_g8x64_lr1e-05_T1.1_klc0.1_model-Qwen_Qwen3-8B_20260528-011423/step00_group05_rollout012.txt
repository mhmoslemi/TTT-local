import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: spread the circles evenly
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(radii_, centers_):
        return -np.sum(radii_)

    def constraint_inside(radii_, centers_):
        return np.all(radii_ <= 1 - np.min(np.vstack((centers_[:, 0], np.ones(10))), axis=0))

    def constraint_overlap(radii_, centers_):
        dists = np.sqrt(np.sum((centers_[:, np.newaxis, :] - centers_[np.newaxis, :, :]) ** 2, axis=2))
        return np.all(dists >= radii_[:, np.newaxis] + radii_[np.newaxis, :])

    def objective_with_constraints(radii, centers):
        # Objective function to maximize the sum of radii
        return -np.sum(radii)

    def constraint_inside_func(radii, centers):
        r = radii
        return np.all(r <= 1 - np.min(np.vstack((centers[:, 0], np.ones(10))), axis=0))

    def constraint_overlap_func(radii, centers):
        r = radii
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return np.all(dists >= r[:, np.newaxis] + r[np.newaxis, :])

    bounds = [(0, 1 - 1e-6) for _ in range(10)]  # Allow some margin for the radii

    # Initial guess for centers (uniform distribution)
    initial_centers = np.random.rand(10, 2)
    initial_radii = np.full(10, 0.1)

    # Create optimization problem
    res = minimize(
        fun=objective_with_constraints,
        x0=initial_radii,
        args=(initial_centers,),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_inside_func, 'args': (initial_centers,)},
            {'type': 'ineq', 'fun': constraint_overlap_func, 'args': (initial_centers,)}
        ],
        options={'maxiter': 1000, 'ftol': 1e-8}
    )

    if res.success:
        optimized_radii = res.x
        # Refine centers with fixed radii using optimization
        def objective_centers(centers):
            # Objective: minimize the sum of squared deviations from initial centers (not relevant)
            return np.sum((centers - initial_centers) ** 2)

        def constraint_inside_centers(centers):
            r = optimized_radii
            x = centers[:, 0]
            y = centers[:, 1]
            return np.all(r <= 1 - np.min(np.vstack((x, np.ones(10))), axis=0)) & np.all(r <= np.min(np.vstack((np.zeros(10), x)), axis=0)) & np.all(r <= 1 - np.min(np.vstack((np.zeros(10), y)), axis=0)) & np.all(r <= np.min(np.vstack((1 - np.ones(10), y)), axis=0))

        def constraint_overlap_centers(centers):
            r = optimized_radii
            dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
            return np.all(dists >= r[:, np.newaxis] + r[np.newaxis, :])

        # Initial guess for centers
        centers_guess = initial_centers.copy()

        res_centers = minimize(
            fun=objective_centers,
            x0=centers_guess.reshape(-1),
            method='SLSQP',
            bounds=[(0, 1) for _ in range(20)],
            constraints=[
                {'type': 'ineq', 'fun': constraint_inside_centers},
                {'type': 'ineq', 'fun': constraint_overlap_centers}
            ],
            options={'maxiter': 1000, 'ftol': 1e-8}
        )

        if res_centers.success:
            optimized_centers = res_centers.x.reshape(10, 2)
            sum_radii = np.sum(optimized_radii)
            return (optimized_centers, optimized_radii, sum_radii)
        else:
            return (initial_centers, initial_radii, np.sum(initial_radii))
    else:
        return (initial_centers, initial_radii, np.sum(initial_radii))
