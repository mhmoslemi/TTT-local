import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles spaced in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii <= 1 - centers) & (radii <= centers), axis=1)
        inside = np.all(inside)
        if not inside:
            return -np.inf

        # Constraint 2: No overlap between circles
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        non_overlap = np.all(dists >= radii_sum)
        if not non_overlap:
            return -np.inf

        return np.sum(radii)

    # Objective and constraints
    def objective_func(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return -np.sum(radii)

    def constraint_func(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii <= 1 - centers) & (radii <= centers), axis=1)
        inside = np.all(inside)
        if not inside:
            return -np.inf

        # Constraint 2: No overlap between circles
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        non_overlap = np.all(dists >= radii_sum)
        if not non_overlap:
            return -np.inf

        return np.sum(radii)

    # Optimize
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x, y for each center
        bounds.extend([(0.0, 0.5)])  # r for each circle (max radius to fit inside square)

    result = minimize(objective_func, x0=np.concatenate([centers.ravel(), radii]), bounds=bounds, method='SLSQP', constraints={'type': 'eq', 'fun': constraint_func})

    if result.success:
        centers_opt = result.x[:20].reshape(10, 2)
        radii_opt = result.x[20:]
        sum_radii = np.sum(radii_opt)
        return (centers_opt, radii_opt, sum_radii)
    else:
        # Fallback to initial guess
        centers_opt = centers
        radii_opt = radii
        sum_radii = np.sum(radii_opt)
        return (centers_opt, radii_opt, sum_radii)
