import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: centers placed in a grid, radii set to 0.1
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.1)
    
    def objective(x):
        # x is a flattened array of centers and radii
        x_flat = x.reshape((10, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative to maximize sum_radii

    def constraint_bounds(x):
        x_flat = x.reshape((10, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        # Ensure each circle is inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return 1.0
        return 0.0

    def constraint_overlap(x):
        x_flat = x.reshape((10, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return 1.0
        return 0.0

    # Bounds for each variable
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0 - 1e-6)])  # Radius can't be more than 1 - 1e-6

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.reshape(-1), radii]), 
                      bounds=bounds, method='SLSQP',
                      constraints=[
                          {'type': 'ineq', 'fun': constraint_bounds},
                          {'type': 'ineq', 'fun': constraint_overlap}
                      ])

    x_opt = result.x
    centers_opt = x_opt.reshape((10, 3))[:, :2]
    radii_opt = x_opt.reshape((10, 3))[:, 2]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
