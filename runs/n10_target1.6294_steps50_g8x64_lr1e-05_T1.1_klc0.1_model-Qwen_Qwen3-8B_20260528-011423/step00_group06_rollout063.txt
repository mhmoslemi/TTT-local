import numpy as np
from scipy.optimize import minimize
import numpy.linalg as la

def run_packing():
    # Initial guess: distribute centers evenly and assign equal radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Each circle fits inside the square
        # x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1
        # So, for each center (x, y), we have x - r >= 0 => x >= r, and similar for others
        # This is handled by the bounds on centers and radii, but here for feasibility
        # We only need to check if the radii are small enough to fit in the square
        return [r for r in radii if r > 1.0]

    def bounds(radii):
        return [(-1.0, 1.0) for _ in range(10)]

    def update_centers(radii):
        # Update centers based on radii to ensure they fit in the square
        # Move centers outward to maximize radius while staying within [0, 1 - r]
        centers[:, 0] = np.clip(centers[:, 0], radii, 1 - radii)
        centers[:, 1] = np.clip(centers[:, 1], radii, 1 - radii)
        return centers

    # Objective function that depends on centers, radii, and constraints
    def objective_with_centers(radii_flat):
        nonlocal centers
        radii = radii_flat.reshape(-1)
        centers = update_centers(radii)
        return -np.sum(radii)

    # Constraint function that ensures circles are non-overlapping
    def constraints_with_centers(radii_flat):
        nonlocal centers
        radii = radii_flat.reshape(-1)
        centers = update_centers(radii)
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dists[i, j] = la.norm(centers[i] - centers[j])
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Bounds for the radii
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Initial guess for radii
    initial_guess = np.full(10, 0.1)

    # Optimization
    result = minimize(
        fun=objective_with_centers,
        x0=initial_guess,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints_with_centers}
    )

    radii = result.x
    centers = update_centers(radii)
    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
