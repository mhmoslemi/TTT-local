import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective(params):
        # Reshape parameters into centers and radii
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:].reshape(10)

        # Objective: minimize negative sum of radii
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    def constraints(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:].reshape(10)

        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf

        # Constraint 2: No two circles overlap
        for i, j in itertools.combinations(range(10), 2):
            dx = centers_flat[i, 0] - centers_flat[j, 0]
            dy = centers_flat[i, 1] - centers_flat[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii_flat[i] + radii_flat[j]:
                return np.inf

        return 0

    # Combine objective and constraints into a single function for minimize
    def full_objective(params):
        # Constraints are enforced by returning np.inf when violated
        # Objective is the negative of sum of radii
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Set bounds for centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0)] * 2)  # x and y for each center
        bounds.extend([(0.0, 0.5)] * 2)  # radii (upper bound is 0.5 to ensure circles fit)
    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radii.flatten()])

    # Use L-BFGS-B optimization
    result = minimize(
        full_objective,
        initial_guess,
        bounds=bounds,
        method='L-BFGS-B',
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    # Extract final centers, radii, and sum
    final_centers = result.x[:60].reshape(10, 2)
    final_radii = result.x[60:].reshape(10)
    final_sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, final_sum_radii)
