import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_square(radii):
        centers = np.vstack([np.array([0.5, 0.5]), np.array([0.5, 0.5])])
        for i in range(10):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return -np.inf
        return 0

    def constraint_overlap(radii):
        centers = np.vstack([np.array([0.5, 0.5]), np.array([0.5, 0.5])])
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    bounds = [(0, 0.5) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_square},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Update centers with the optimized radii
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    return (centers, result.x, np.sum(result.x))
