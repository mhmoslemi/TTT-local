import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: spread the circles evenly
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii, centers):
        # 1. All circles are inside the unit square
        # 2. No two circles overlap
        # Return list of constraints (inequality constraints)
        n = len(radii)
        con = []
        # Constraint for circle i to be inside the square
        for i in range(n):
            con.append({'type': 'ineq', 'fun': lambda r, c, i=i: c[i, 0] - r[i]})
            con.append({'type': 'ineq', 'fun': lambda r, c, i=i: 1 - c[i, 0] - r[i]})
            con.append({'type': 'ineq', 'fun': lambda r, c, i=i: c[i, 1] - r[i]})
            con.append({'type': 'ineq', 'fun': lambda r, c, i=i: 1 - c[i, 1] - r[i]})
        # Constraint for circle i and circle j to not overlap
        for i in range(n):
            for j in range(i + 1, n):
                def constraint(radii_arr):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    return dist - (radii[i] + radii[j])
                con.append({'type': 'ineq', 'fun': constraint})
        return con

    # Define bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Set up the optimization
    result = minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        bounds=bounds,
        constraints=constraints(radii, centers),
        method='SLSQP',
        tol=1e-6
    )

    # Update centers and radii based on the result
    best_radii = result.x
    # Use the same initial centers (could be refined in a more advanced setup)
    best_centers = centers.copy()

    # Compute sum of radii
    sum_radii = np.sum(best_radii)

    return (best_centers, best_radii, sum_radii)
