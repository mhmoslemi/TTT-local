import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(0)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -sum(radii)

    def constraints(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        constraints = []
        for i in range(10):
            # Ensure the circle is inside the unit square
            constraints.append((centers[i, 0] - radii[i]) >= 0)
            constraints.append((centers[i, 1] - radii[i]) >= 0)
            constraints.append((1 - centers[i, 0] - radii[i]) >= 0)
            constraints.append((1 - centers[i, 1] - radii[i]) >= 0)
            # Ensure non-negativity of radii
            constraints.append((radii[i] >= 0))
        # Ensure no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append((dist - (radii[i] + radii[j])) >= 0)
        return constraints

    # Initial guess as a flat array
    x0 = np.hstack((centers.flatten(), radii))

    # Optimization
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1) for _ in range(60)] + [(0, 0.5) for _ in range(10)]
    )

    # Extract results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
