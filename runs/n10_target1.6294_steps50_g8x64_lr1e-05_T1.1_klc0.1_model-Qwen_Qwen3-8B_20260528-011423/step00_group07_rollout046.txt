import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.ones(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)

    def constraint_square(radii, centers):
        for i in range(10):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return 1
        return 0

    def constraint_overlap(radii, centers):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return 1
        return 0

    def constraints(radii, centers):
        return [constraint_square(radii, centers), constraint_overlap(radii, centers)]

    bounds = [(0.0, 0.5) for _ in range(10)]

    result = minimize(
        lambda r: objective(r, centers),
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': lambda r: -constraint_square(r, centers)}, 
                     {'type': 'ineq', 'fun': lambda r: -constraint_overlap(r, centers)}]
    )

    best_radii = result.x
    best_centers = centers.copy()
    sum_radii = np.sum(best_radii)

    return (best_centers, best_radii, sum_radii)
