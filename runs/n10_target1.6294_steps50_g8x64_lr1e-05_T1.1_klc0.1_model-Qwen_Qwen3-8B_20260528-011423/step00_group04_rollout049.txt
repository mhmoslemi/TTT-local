from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = np.random.rand()
            centers[i, 1] = np.random.rand()

        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        
        # Ensure circles do not overlap
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i+1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        
        return -np.sum(radii)

    # Initial guess: random centers and small radii
    centers = np.random.rand(10, 2)

    # Optimization
    result = minimize(objective, radii, method='L-BFGS-B', bounds=[(0, 0.5)]*10, constraints={'type': 'eq', 'fun': constraints})
