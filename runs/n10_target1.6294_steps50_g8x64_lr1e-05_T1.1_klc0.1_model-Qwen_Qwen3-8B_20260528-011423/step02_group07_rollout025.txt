import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Check if all radii are non-negative
        for r in radii:
            if r < 0:
                return np.inf
        # Get centers based on initial grid
        centers_grid = np.vstack([np.array([0.1 + i*0.2, 0.1 + j*0.2]) for i in range(5) for j in range(2)])
        # Check if all circles are inside the square
        for i in range(10):
            x, y = centers_grid[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
        # Check if all circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_grid[i, 0] - centers_grid[j, 0]
                dy = centers_grid[i, 1] - centers_grid[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    res = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1) for _ in range(10)],
        constraints={'type': 'ineq', 'fun': constraints},
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    # Return the result
    centers = np.vstack([np.array([0.1 + i*0.2, 0.1 + j*0.2]) for i in range(5) for j in range(2)])
    return (centers, res.x, -res.fun)
