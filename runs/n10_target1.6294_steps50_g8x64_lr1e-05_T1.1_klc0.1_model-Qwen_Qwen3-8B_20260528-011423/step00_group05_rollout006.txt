import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.ones(10) * 0.1  # Initial small radius

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define bounds for each radius
    bounds = [(0, 0.5) for _ in range(10)]

    # Set up the optimization problem
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-8,
        options={'maxiter': 500, 'ftol': 1e-8}
    )

    # Check if the optimization was successful
    if not result.success:
        raise RuntimeError("Optimization failed")

    centers = np.array(centers)
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
