import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25

    # Initial radii
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_1(radii):
        # Ensure each circle is fully inside the unit square
        return np.array([r <= x - r and r <= 1 - x - r and r <= y - r and r <= 1 - y - r for x, y, r in zip(centers[:, 0], centers[:, 1], radii)])

    def constraint_2(radii):
        # Ensure circles do not overlap
        dist = np.zeros(10)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i] += dx * dx + dy * dy - (radii[i] + radii[j]) ** 2
        return dist

    # Optimization
    bounds = [(0, 1.0) for _ in range(10)]
    cons = [
        {'type': 'ineq', 'fun': constraint_1},
        {'type': 'ineq', 'fun': constraint_2}
    ]

    result = minimize(objective, radii, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-6)

    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
