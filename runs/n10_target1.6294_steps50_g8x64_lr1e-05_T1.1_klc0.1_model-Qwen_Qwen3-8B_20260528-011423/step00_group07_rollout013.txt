import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)

    # Objective function to minimize negative sum of radii
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints: circles must be inside the unit square
    def square_constraint(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.sum((centers_flat - radii_flat[:, np.newaxis])[:, :, 0] >= 0) + \
               np.sum((centers_flat + radii_flat[:, np.newaxis])[:, :, 0] <= 1) + \
               np.sum((centers_flat - radii_flat[:, np.newaxis])[:, :, 1] >= 0) + \
               np.sum((centers_flat + radii_flat[:, np.newaxis])[:, :, 1] <= 1)

    # Constraints: no overlapping between circles
    def non_overlap_constraint(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distances = np.sqrt(np.sum((centers_flat[:, np.newaxis] - centers_flat[np.newaxis, :]) ** 2, axis=2))
        return np.sum(distances >= radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Bounds for centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # centers x, y
        bounds.extend([(0.0, 1.0)])  # radii

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii]),
        constraints=[
            {'type': 'ineq', 'fun': square_constraint},
            {'type': 'ineq', 'fun': non_overlap_constraint}
        ],
        bounds=bounds,
        method='SLSQP'
    )

    # Extract solution
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
