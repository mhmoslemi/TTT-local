```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    
    # Initial radii: uniform
    radii = np.full(10, 0.1)
    
    def objective(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape((10, 2))
        radii_flat = x[60:].reshape(10)
        
        # Objective: maximize sum of radii (minimize negative sum)
        sum_radii = np.sum(radii_flat)
        return -sum_radii
    
    def constraints(x):
        centers_flat = x[:60].reshape((10, 2))
        radii_flat = x[60:].reshape(10)
        
        # Ensure each circle is within unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i][0], centers_flat[i][1], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        
        # No negative radii
        for r in radii_flat:
            if r < 0:
                return np.inf
        
        return 0
    
    # Optimize using scipy.optimize.minimize
    result = minimize(
        fun=objective,
        x0=np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}]
    )
    
    # Extract results
    centers_opt = result.x[:60].reshape((10, 2))
    radii_opt = result.x[60:].reshape(10)
    sum_radii_opt = -result.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
```
