import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid with some spacing
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.array([radii[i] <= 1 - np.linalg.norm(centers[i]) for i in range(10)])

    def constraint_no_overlap(radii):
        dist = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dist[i, j] = np.linalg.norm(centers[i] - centers[j])
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x).flatten()}
    ]

    bounds = [(0, 1) for _ in range(10)]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    return (centers, result.x, np.sum(result.x))
