import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import warnings

def run_packing():
    # Initial guess: random positions within the square and equal radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) / 10.0

    def objective(x):
        # x is a flat array of centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize it

    def constraints(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Constraint 1: Each circle is inside the unit square
        lb = np.array([r for r in radii])
        ub = np.array([1 - r for r in radii])
        in_square = np.all((centers >= lb) & (centers <= ub), axis=1)
        # Constraint 2: No two circles overlap
        distances = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        min_dist = np.triu(distances, k=1)
        dists = min_dist - (radii[:, np.newaxis] + radii[np.newaxis, :])
        overlap = np.any(dists < 0)
        return [in_square, overlap]

    # Setup bounds
    bounds = []
    for i in range(10):
        # Center bounds for x and y of each circle
        bounds.extend([(0, 1) for _ in range(2)])
        # Radius bounds
        bounds.extend([(-np.inf, 1 - 1e-6)])  # Radius can't exceed 1 - small value to stay inside

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Objective function
    def objective_func(x):
        return objective(x)

    # Constraints
    def constraint_func(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        in_square = np.all((centers >= radii[:, np.newaxis]) & (centers <= 1 - radii[:, np.newaxis]), axis=1)
        distances = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        min_dist = np.triu(distances, k=1)
        dists = min_dist - (radii[:, np.newaxis] + radii[np.newaxis, :])
        overlap = np.any(dists < 0)
        if not in_square.all() or overlap:
            return np.inf  # Penalize invalid configurations
        return 0

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective_func, x0, bounds=bounds, constraints={"type": "ineq", "fun": constraint_func}, method="SLSQP", tol=1e-6)

    # Extract results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
