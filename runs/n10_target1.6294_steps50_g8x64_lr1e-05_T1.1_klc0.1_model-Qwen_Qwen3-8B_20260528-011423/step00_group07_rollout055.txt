import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[i * 0.25 + 0.1, j * 0.25 + 0.1] for i in range(4) for j in range(3)])
    initial_radii = np.ones(10) * 0.1  # Initial radius for each circle

    def objective_function(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints_func(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Constraint 2: Distance between any two centers >= sum of their radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return 0.0

    # Flatten centers and radii for optimization
    x0 = np.concatenate((initial_centers.flatten(), initial_radii))

    # Set up optimization problem
    res = minimize(
        objective_function,
        x0,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints_func},
        bounds=[(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    )

    # Extract results
    centers = res.x[:20].reshape(10, 2)
    radii = res.x[20:]
    sum_radii = -objective_function(radii)

    return (centers, radii, sum_radii)
