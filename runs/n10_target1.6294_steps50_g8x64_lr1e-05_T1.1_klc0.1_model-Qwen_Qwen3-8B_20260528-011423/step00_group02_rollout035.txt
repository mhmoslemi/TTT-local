import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: distribute circles evenly
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraints: each circle must be fully inside the unit square
        # r <= x <= 1 - r and r <= y <= 1 - r
        # For all circles, we need x_i - r_i >= 0, 1 - x_i - r_i >= 0, same for y
        constraints = []
        for i in range(10):
            r = radii[i]
            x, y = centers[i]
            constraints.append(r - x + 0.001)  # r <= x, with some tolerance
            constraints.append(1 - x - r + 0.001)  # 1 - x >= r
            constraints.append(r - y + 0.001)
            constraints.append(1 - y - r + 0.001)
        return np.array(constraints)

    def non_overlap(radii):
        # No two circles can overlap: distance between centers >= sum of radii
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        for i in range(10):
            for j in range(i+1, 10):
                if dists[i, j] < radii[i] + radii[j] - 1e-3:
                    return False
        return True

    # Define optimization bounds: radii >= 0
    bounds = [(0, 1) for _ in range(10)]

    # Optimize
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        tol=1e-6
    )

    # Check if optimization was successful
    if not result.success:
        raise RuntimeError("Optimization failed")

    # Extract final result
    final_radii = result.x
    sum_radii = np.sum(final_radii)

    # Ensure no overlap for final configuration
    if not non_overlap(final_radii):
        raise RuntimeError("Final configuration does not satisfy non-overlap constraint")

    return (centers, final_radii, sum_radii)
