import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial guess: equally spaced circles in a grid
    centers = np.array([[i/3 + 0.1, j/3 + 0.1] for i in range(3) for j in range(3)])
    radii = np.full(9, 0.1)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add a central circle
    radii = np.append(radii, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        n = len(radii)
        centers = np.array([[x, y] for x, y in zip(np.random.rand(n), np.random.rand(n))])
        # Update centers based on radii (simple heuristic)
        # This is a placeholder for actual optimization, not a real packing method
        # For the purpose of this example, use a fixed center configuration
        centers = np.array([[0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],
                            [0.3, 0.1], [0.3, 0.3], [0.3, 0.5], [0.3, 0.7], [0.3, 0.9]])
        # Bounds for radii
        bounds = [(0, 1 - np.random.rand()) for _ in range(n)]
        # Constraints: circles must be inside the unit square
        for i in range(n):
            if not (radii[i] <= 0.1 and radii[i] <= 0.9):
                return -np.inf, np.inf
        # Constraints: circles must not overlap
        dists = pdist(centers)
        for i in range(n):
            for j in range(i + 1, n):
                if dists[j] < radii[i] + radii[j]:
                    return -np.inf, np.inf
        return objective(radii), bounds

    # Optimization
    result = minimize(lambda radii: -np.sum(radii), radii, method='SLSQP', constraints=constraints)
    centers = np.array([[0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],

    sum_radii = result.fun
    return (centers, -result.fun, sum_radii)
