import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    # Objective function: maximize the sum of radii
    def objective(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints: circles must be inside the unit square
    def circle_inside(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([r <= 1 - np.max(np.vstack((c, 1 - c)), axis=0) for c, r in zip(centers, radii)])

    # Constraint: circles do not overlap
    def no_overlap(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(60)]  # 10 centers * 2 coordinates + 10 radii

    # Optimization
    result = minimize(
        objective,
        np.concatenate((centers.ravel(), radii)),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': circle_inside},
            {'type': 'ineq', 'fun': no_overlap}
        ],
        options={'ftol': 1e-10, 'disp': False}
    )

    # Extract results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun  # Convert back to positive sum

    return (centers, radii, sum_radii)
