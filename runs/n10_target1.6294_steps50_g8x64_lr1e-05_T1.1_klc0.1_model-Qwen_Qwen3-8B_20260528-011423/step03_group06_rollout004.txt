import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: same for all
    radii = np.full(10, 0.1)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints_func(radii_vec):
        # Ensure all radii are non-negative and circles are within the unit square
        for i in range(10):
            r = radii_vec[i]
            if r < 0:
                return np.inf
            if r > 1 - np.min(centers[i, :]):
                return np.inf

        # Check circle-circle overlaps
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return np.inf
        return 0

    # Set up optimization
    bounds = [(0, 1 - 1e-6) for _ in range(10)]
    res = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints_func},
        tol=1e-6
    )

    # Extract result
    radii = res.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
