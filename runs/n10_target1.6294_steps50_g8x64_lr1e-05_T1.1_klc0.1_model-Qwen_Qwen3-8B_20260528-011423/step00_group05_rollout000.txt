import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: equally spaced grid points with small radii
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    radii = np.full(10, 0.05)
    
    def objective_function(params):
        # Flatten parameters into centers and radii
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        
        # Objective: maximize the sum of radii (minimize negative sum)
        objective = -np.sum(radii_flat)
        
        # Constraints: circles must be within the unit square
        square_constraints = [np.zeros(10) <= radii_flat, np.ones(10) * (1 - radii_flat) >= centers_flat[:, 0],
                              np.zeros(10) <= radii_flat, np.ones(10) * (1 - radii_flat) >= centers_flat[:, 1]]
        
        # Constraints: circles must not overlap
        overlap_constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                overlap_constraints.append(dist >= radii_flat[i] + radii_flat[j])
        
        # Combine all constraints
        constraints = []
        for c in square_constraints:
            constraints.append({'type': 'ineq', 'fun': lambda p, c=c: c})
        for c in overlap_constraints:
            constraints.append({'type': 'ineq', 'fun': lambda p, c=c: c})
        
        # Check if constraints are satisfied
        for c in constraints:
            if c['fun'](params) < -1e-6:
                return np.inf  # Penalize invalid configurations
        
        return objective
    
    # Optimization
    bounds = [(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]
    result = minimize(objective_function, np.concatenate([centers.reshape(10, 2).ravel(), radii]), 
                      method='SLSQP', bounds=bounds, constraints=[])
    
    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun
    
    return (centers_opt, radii_opt, sum_radii)
