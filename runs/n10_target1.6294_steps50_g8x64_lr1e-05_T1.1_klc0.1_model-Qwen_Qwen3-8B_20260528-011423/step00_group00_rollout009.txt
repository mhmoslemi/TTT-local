import numpy as np
from scipy.optimize import minimize
import time

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1  # Initial radius

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii, centers):
        cons = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                cons.append(dist - (radii[i] + radii[j]))
        return cons

    def boundary_constraints(radii, centers):
        cons = []
        for i in range(10):
            cons.append(radii[i] - (centers[i, 0] - 0))  # x - r >= 0
            cons.append(radii[i] - (1 - centers[i, 0]))  # 1 - x - r >= 0
            cons.append(radii[i] - (centers[i, 1] - 0))  # y - r >= 0
            cons.append(radii[i] - (1 - centers[i, 1]))  # 1 - y - r >= 0
        return cons

    # Bounds for radii
    bounds = [(0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraints(x, centers)},
            {'type': 'ineq', 'fun': lambda x: boundary_constraints(x, centers)}
        ],
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract final values
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
