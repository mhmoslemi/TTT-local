import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: spread circles evenly in the square
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints_sum(x):
        return np.sum(x)  # Sum of radii
    def constraints_min(x):
        return 0.0  # No constraint on minimum radius
    
    def constraint_box(r):
        # Ensure all circles are fully inside the square
        return np.sum((r - 0.0) ** 2 + (1.0 - r - 0.0) ** 2 + (1.0 - r - 0.0) ** 2 + (r - 0.0) ** 2)
    def constraint_overlap(c, r):
        # Check that no two circles overlap
        dist = np.sqrt(((c[:, 0][:, np.newaxis] - c[:, 0][np.newaxis, :]) ** 2 +
                        (c[:, 1][:, np.newaxis] - c[:, 1][np.newaxis, :]) ** 2))
        return np.sum(dist - (r[:, np.newaxis] + r[np.newaxis, :]))  # sum of min distance violations
    
    def objective_func(x):
        # x is flattened [x1, y1, r1, x2, y2, r2, ...]
        c = x[:20].reshape((10, 2))
        r = x[20:]
        # Ensure radii are non-negative and within the square
        for i in range(10):
            if r[i] < 0:
                return np.inf
            if r[i] > 1 - 2 * c[i, 0] or r[i] > 1 - 2 * c[i, 1]:
                return np.inf
        # Check for overlap between circles
        dist = np.sqrt(((c[:, 0][:, np.newaxis] - c[:, 0][np.newaxis, :]) ** 2 +
                        (c[:, 1][:, np.newaxis] - c[:, 1][np.newaxis, :]) ** 2))
        for i in range(10):
            for j in range(i + 1, 10):
                if dist[i, j] < r[i] + r[j]:
                    return np.inf
        # Compute and return the negative sum of radii for maximization
        return -np.sum(r)
    
    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])
    
    # Optimizer settings
    bounds = []
    for i in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    
    # Optimization
    result = minimize(objective_func, x0=x0, bounds=bounds, method='SLSQP', tol=1e-8)
    
    # Extract the result
    centers_result = result.x[:20].reshape((10, 2))
    radii_result = result.x[20:]
    sum_radii = -result.fun
    
    # Ensure all circles are fully inside the unit square
    for i in range(10):
        if radii_result[i] < 0 or radii_result[i] > 1 - 2 * centers_result[i, 0] or \
           radii_result[i] > 1 - 2 * centers_result[i, 1]:
            raise ValueError("One or more circles are outside the unit square.")
    
    # Ensure no two circles overlap
    dist = np.sqrt(((centers_result[:, 0][:, np.newaxis] - centers_result[:, 0][np.newaxis, :]) ** 2 +
                    (centers_result[:, 1][:, np.newaxis] - centers_result[:, 1][np.newaxis, :]) ** 2))
    for i in range(10):
        for j in range(i + 1, 10):
            if dist[i, j] < radii_result[i] + radii_result[j]:
                raise ValueError("Two circles overlap.")
    
    return (centers_result, radii_result, sum_radii)
