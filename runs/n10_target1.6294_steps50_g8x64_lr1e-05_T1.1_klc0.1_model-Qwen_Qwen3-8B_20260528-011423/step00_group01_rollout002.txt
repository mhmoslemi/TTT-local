import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: spread circles evenly
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
    radii = np.ones(10) * 0.15  # Initial guess for radii

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints_eq(radii):
        # Distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return 1
        return 0

    def constraints_ineq(radii):
        # Radii cannot exceed available space
        for i in range(10):
            if radii[i] > 1 - 2 * centers[i, 0] or radii[i] > 1 - 2 * centers[i, 1]:
                return 1
        return 0

    # Optimization bounds
    bounds = [(0.0, 0.5) for _ in range(10)]  # Radii can't be larger than 0.5

    # Optimization options
    options = {'maxiter': 1000, 'disp': False}

    # Combine constraints
    def constraints(x):
        cons_eq = constraints_eq(x)
        cons_ineq = constraints_ineq(x)
        return [cons_eq, cons_ineq]

    # Perform optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        jac=None,
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        options=options
    )

    # Extract results
    final_radii = result.x
    final_centers = centers.copy()
    final_sum_radii = np.sum(final_radii)

    # Ensure all circles are within the square
    for i in range(10):
        if final_radii[i] > 1 - 2 * final_centers[i, 0] or final_radii[i] > 1 - 2 * final_centers[i, 1]:
            final_radii[i] = min(1 - 2 * final_centers[i, 0], 1 - 2 * final_centers[i, 1])

    return (final_centers, final_radii, final_sum_radii)
