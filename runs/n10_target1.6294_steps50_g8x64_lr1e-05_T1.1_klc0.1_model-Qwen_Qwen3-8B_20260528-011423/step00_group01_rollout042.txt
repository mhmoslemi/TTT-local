import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective_function(x):
        # Flatten the input to centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize the negative to maximize the sum

    def constraint_inside(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    def constraint_boundary(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        constraints = []
        for i in range(10):
            x_center, y_center = centers[i]
            r = radii[i]
            constraints.append(x_center - r)
            constraints.append(1 - x_center - r)
            constraints.append(y_center - r)
            constraints.append(1 - y_center - r)
        return np.array(constraints)

    def constraint_non_negative(x):
        return x[60:]

    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.2, 0.2], [0.5, 0.2], [0.8, 0.2],
        [0.2, 0.5], [0.5, 0.5], [0.8, 0.5],
        [0.2, 0.8], [0.5, 0.8], [0.8, 0.8],
        [0.5, 0.3]
    ])
    initial_radii = np.array([0.1] * 10)
    x0 = np.concatenate([initial_centers.flatten(), initial_radii])

    # Set up optimization problem
    bounds = [(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]

    result = minimize(
        fun=objective_function,
        x0=x0,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun  # Convert back to maximized sum
    return (centers, radii, sum_radii)
