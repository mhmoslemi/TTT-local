import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: circles are inside the unit square
        for r in radii:
            if r < 0:
                return False
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(10):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        return True

    # Define bounds for radii (non-negative)
    bounds = [(0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        options={'disp': False, 'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    radii_opt = result.x
    sum_radii = -result.fun  # Objective was negative sum
    # Ensure constraints are satisfied (just in case)
    if not constraints(radii_opt):
        raise ValueError("Optimization did not satisfy constraints")
    
    return (centers, radii_opt, sum_radii)
