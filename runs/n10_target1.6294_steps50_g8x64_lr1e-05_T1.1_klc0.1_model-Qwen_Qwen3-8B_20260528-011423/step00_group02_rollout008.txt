import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place 10 circles in a grid pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.125
        centers[i, 1] = (i // 5) / 4.0 + 0.125

    # Initial radii
    radii = np.ones(10) * 0.15

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: each circle must be inside the square
        # r_i <= x_i - r_i <= 1 - r_i, same for y
        # Check for all circles
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Check for non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        res = minimize(objective, radii, method='SLSQP', bounds=[(0, 0.5)]*10,
                       constraints={'type': 'ineq', 'fun': constraints})

    if not res.success:
        return (centers, radii, np.sum(radii))

    final_radii = res.x
    final_sum = np.sum(final_radii)
    return (centers, final_radii, final_sum)
