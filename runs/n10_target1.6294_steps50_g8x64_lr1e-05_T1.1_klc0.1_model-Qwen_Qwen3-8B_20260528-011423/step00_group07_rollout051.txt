import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: centers at a grid and radii as 0.1
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative
    
    def inequality_constraints(centers, radii):
        # Constraints: each circle is inside the unit square
        inside = np.concatenate([radii <= centers[:, 0], radii <= 1 - centers[:, 0],
                                 radii <= centers[:, 1], radii <= 1 - centers[:, 1]])
        # Constraints: no two circles overlap
        overlaps = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                overlaps.append(radii[i] + radii[j] - dist)
        return np.concatenate([inside, overlaps])
    
    def constraint_func(x):
        # x is a flattened array: [c1x, c1y, c2x, c2y, ..., c10y, r1, r2, ..., r10]
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        return inequality_constraints(centers_flat, radii_flat)
    
    bounds = [(0, 1) for _ in range(20)] + [(0, 0.5) for _ in range(10)]
    
    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.ravel(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_func},
        options={'ftol': 1e-9, 'maxiter': 1000}
    )
    
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
