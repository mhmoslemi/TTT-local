import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.zeros(10) + 0.15  # Initial radius guess

    def objective(radii):
        return -np.sum(radii)

    def constraint_overlap(radii, centers):
        # No two circles overlap
        n = len(centers)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return 1
        return 0

    def constraint_inside(radii, centers):
        # All circles are inside the unit square
        for i in range(len(centers)):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return 1
        return 0

    # Combine the constraints
    def constraints(radii):
        # First constraint: no overlap
        if constraint_overlap(radii, centers) > 0:
            return np.inf
        # Second constraint: inside the square
        if constraint_inside(radii, centers) > 0:
            return np.inf
        return 0

    # Bounds for radii
    bounds = [(0.0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        fun=lambda r: objective(r) + 1e-4 * constraints(r),
        x0=radii,
        bounds=bounds,
        method='SLSQP',
        tol=1e-6,
        options={'maxiter': 1000, 'ftol': 1e-8, 'disp': False}
    )

    # Extract result
    radii_result = result.x
    centers_result = centers.copy()
    sum_radii_result = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii_result)
