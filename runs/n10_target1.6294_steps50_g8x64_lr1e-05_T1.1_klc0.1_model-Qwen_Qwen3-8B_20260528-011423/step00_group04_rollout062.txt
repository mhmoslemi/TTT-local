import numpy as np
from scipy.optimize import minimize
import warnings

# Define the problem
def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(x):
        # Flatten x into centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        cons = []

        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            cons.append((x_i - r_i >= 0))
            cons.append((x_i + r_i <= 1))
            cons.append((y_i - r_i >= 0))
            cons.append((y_i + r_i <= 1))

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                x_i, y_i = centers_flat[i]
                x_j, y_j = centers_flat[j]
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
                cons.append((dist >= r_i + r_j))

        return np.array(cons)

    # Bounds for x and y coordinates and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1), (0, 1)])  # x, y, radius for each circle

    # Optimize
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                      method='SLSQP', bounds=bounds, constraints={"type": "ineq", "fun": constraints})
    
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
