import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Check constraints for all circles
        # 1. Radii must be non-negative
        if np.any(radii < 0):
            return np.inf  # Penalize negative radii

        # 2. Circles must be inside the unit square
        for i in range(10):
            if radii[i] < 1e-8:  # Very small radius, skip
                continue
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return np.inf
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return np.inf

        # 3. Circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                if radii[i] < 1e-8 or radii[j] < 1e-8:  # Skip very small radii
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf

        return 0  # No constraint violation

    # Define the optimization problem
    bounds = [(0.0, 0.5) for _ in range(10)]  # Radii can't exceed 0.5

    # Use L-BFGS-B optimizer
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        method='L-BFGS-B',
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6,
        options={'disp': False}
    )

    # Extract results
    best_radii = result.x
    best_centers = centers.copy()
    sum_radii = -result.fun

    return (best_centers, best_radii, sum_radii)
