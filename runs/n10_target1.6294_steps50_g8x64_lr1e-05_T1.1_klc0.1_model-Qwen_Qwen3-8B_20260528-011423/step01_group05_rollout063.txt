import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced circles in grid
    centers = np.zeros((10, 2))
    radii = np.full(10, 0.1)
    for i in range(10):
        centers[i, 0] = 0.1 + (i % 5) * 0.2
        centers[i, 1] = 0.1 + (i // 5) * 0.2

    def objective(radii):
        return -np.sum(radii)

    def constraint_pair(i, j):
        def constraint(radii):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            return dist - (radii[i] + radii[j])
        return constraint

    # Define constraints for non-overlapping circles
    constraints = []
    for i in range(10):
        for j in range(i + 1, 10):
            constraints.append({'type': 'ineq', 'fun': constraint_pair(i, j)})

    # Define bounds for radii
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Run optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Check if optimization succeeded
    if not result.success:
        warnings.warn("Optimization did not converge.")
        return centers, radii, np.sum(radii)

    # Update centers and radii to ensure circles are fully inside the square
    new_centers = np.copy(centers)
    new_radii = result.x
    for i in range(10):
        # Ensure radii are within bounds
        new_radii[i] = min(new_radii[i], 1.0 - new_centers[i, 0])
        new_radii[i] = min(new_radii[i], 1.0 - new_centers[i, 1])
        new_radii[i] = max(new_radii[i], 0.0)
        # Adjust centers to ensure circles are inside the square
        new_centers[i, 0] = max(0.0 + new_radii[i], min(1.0 - new_radii[i], new_centers[i, 0]))
        new_centers[i, 1] = max(0.0 + new_radii[i], min(1.0 - new_radii[i], new_centers[i, 1]))

    sum_radii = np.sum(new_radii)
    return new_centers, new_radii, sum_radii
