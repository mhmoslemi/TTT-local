import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: place 10 circles in a grid pattern
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)
    
    # Objective function: minimize negative sum of radii to maximize sum
    def objective(x):
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        return -np.sum(radii)
    
    # Bounds: each radius must be between 0 and 0.5 (since circle must fit in unit square)
    bounds = Bounds(np.zeros(20), np.full(20, 0.5))
    
    # Constraints: circle must be inside the unit square
    def square_bound(x):
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        return np.concatenate([
            centers[:, 0] - radii,
            1 - centers[:, 0] - radii,
            centers[:, 1] - radii,
            1 - centers[:, 1] - radii
        ])
    
    # Constraints: no two circles can overlap
    def non_overlap(x):
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Combine all constraints
    constraints = [
        {'type': 'ineq', 'fun': square_bound},
        {'type': 'ineq', 'fun': non_overlap}
    ]
    
    # Optimization
    result = minimize(objective, x0=np.concatenate([radii, centers.ravel()]), bounds=bounds, constraints=constraints)
    
    # Extract result
    radii = result.x[::2]
    centers = result.x[1::2].reshape(10, 2)
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
