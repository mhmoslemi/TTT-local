import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective_function(radii):
        return -np.sum(radii)  # maximize sum of radii
    
    def constraints(radii):
        # Ensure circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            for j in range(10):
                if i == j:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        # Check if circles are within the square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        return True
    
    def optimize():
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            result = minimize(
                objective_function,
                radii,
                method='L-BFGS-B',
                bounds=[(0, 1) for _ in range(10)],
                constraints=[
                    {
                        'type': 'ineq',
                        'fun': lambda x: constraints(x)
                    }
                ]
            )
        return result
    
    result = optimize()
    radii_opt = result.x
    sum_radii_opt = -result.fun
    
    # Refine using local optimization
    for _ in range(5):
        result = optimize()
        radii_opt = result.x
        sum_radii_opt = -result.fun
    
    # Final check and return
    centers_final = centers
    radii_final = radii_opt
    sum_radii_final = sum_radii_opt
    
    return (centers_final, radii_final, sum_radii_final)
