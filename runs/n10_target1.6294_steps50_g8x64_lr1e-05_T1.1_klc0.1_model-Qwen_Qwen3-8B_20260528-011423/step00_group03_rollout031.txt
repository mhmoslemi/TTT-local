import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.ones(10) * 0.15

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints(radii_vec):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
        r = radii_vec
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < r[i] + r[j]:
                    return False
        for i in range(10):
            x, y = centers[i]
            if not (radii_vec[i] <= x <= 1 - radii_vec[i] and radii_vec[i] <= y <= 1 - radii_vec[i]):
                return False
        return True

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': lambda r: r},
            {'type': 'ineq', 'fun': lambda r: np.array([1 - r[i] - (i % 5 + 0.5) / 5 for i in range(10)])},
            {'type': 'ineq', 'fun': lambda r: np.array([1 - r[i] - (i // 5 + 0.5) / 5 for i in range(10)])},
            {'type': 'ineq', 'fun': lambda r: np.array([1 - r[i] - math.hypot((i % 5 + 0.5) / 5 - (j % 5 + 0.5) / 5, (i // 5 + 0.5) / 5 - (j // 5 + 0.5) / 5) - r[j] for i in range(10) for j in range(i + 1, 10)])}
        ],
        options={'ftol': 1e-9, 'disp': False}
    )

    radii = result.x
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Ensure all circles are within bounds
    for i in range(10):
        x, y = centers[i]
        if radii[i] > x or radii[i] > 1 - x or radii[i] > y or radii[i] > 1 - y:
            radii[i] = min(radii[i], x, 1 - x, y, 1 - y)

    # Final check for circle centers inside bounds
    for i in range(10):
        x, y = centers[i]
        if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
            radii[i] = min(radii[i], x, 1 - x, y, 1 - y)

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
