import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid-like formation
    centers = np.zeros((10, 2))
    radii = np.ones(10) * 0.1  # Initial guess for radii

    def objective_function(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        # Constraint 1: circles are inside the unit square
        for i in range(10):
            if not (radii_flat[i] <= centers_flat[i][0] <= 1 - radii_flat[i] and
                    radii_flat[i] <= centers_flat[i][1] <= 1 - radii_flat[i]):
                return np.inf
        # Constraint 2: circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return -sum_radii  # Minimize -sum_radii to maximize sum_radii

    # Flatten centers and radii for optimization
    x0 = np.concatenate((centers.flatten(), radii))

    # Constraints: non-overlapping, inside the square
    # We use a penalty method to handle constraints in the objective function

    # Bounds: centers between 0 and 1, radii between 0 and 0.5
    bounds = []
    for i in range(10):
        bounds.extend([(0, 1), (0, 1)])  # for centers
        bounds.append((0, 0.5))  # for radii

    # Run optimization
    result = minimize(objective_function, x0, method='L-BFGS-B', bounds=bounds)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
