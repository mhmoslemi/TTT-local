import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    def objective_function(positions):
        radii = positions[:10]
        centers = positions[10:]
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(positions):
        radii = positions[:10]
        centers = positions[10:].reshape(10, 2)
        constraints = []
        # Constraint: each circle is within the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            # Left and right boundaries
            constraints.append(2 * r - 1 + x)
            constraints.append(2 * r - 1 - x)
            # Bottom and top boundaries
            constraints.append(2 * r - 1 + y)
            constraints.append(2 * r - 1 - y)
        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    # Initial guess: randomly place 10 circles with small radii
    np.random.seed(42)
    initial_guess = np.random.rand(10 + 10 * 2)
    initial_guess[:10] = 0.1 * np.random.rand(10)  # Initial radii
    initial_guess[10:] = np.random.rand(10 * 2)  # Initial centers

    # Set up optimization problem
    res = minimize(
        objective_function,
        initial_guess,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1) for _ in range(10)] + [(0, 1) for _ in range(10 * 2)]
    )

    positions = res.x
    radii = positions[:10]
    centers = positions[10:].reshape(10, 2)
    sum_radii = -res.fun  # Convert back to sum of radii

    return (centers, radii, sum_radii)
