import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for center positions (random points in the unit square)
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1

    def objective(r):
        return -np.sum(r)

    def constraints(r):
        # Constraint 1: All circles are fully inside the unit square
        # r[i] <= x[i] <= 1 - r[i], r[i] <= y[i] <= 1 - r[i]
        # We use the square of the radius to avoid negative values
        # and ensure the constraints are satisfied
        # r[i] is the radius of the i-th circle
        # x[i] = centers[i, 0], y[i] = centers[i, 1]
        # r[i] must be <= x[i], y[i], 1 - x[i], 1 - y[i]
        # We use a margin to ensure the center is at least r[i] away from the edge
        # So, for each circle, x[i] - r[i] >= 0 and 1 - x[i] - r[i] >= 0
        # same for y[i]
        r = np.clip(r, 0, 1)  # Ensure radii stay within valid bounds

        # Check all circles are fully inside the square
        for i in range(10):
            x, y = centers[i]
            if x - r[i] < 0 or 1 - x - r[i] < 0 or y - r[i] < 0 or 1 - y - r[i] < 0:
                return np.inf

        # Constraint 2: No circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r[i] + r[j]:
                    return np.inf

        return 0

    # Define bounds for optimization
    bounds = [(0, 1) for _ in range(10)]

    # Define objective function to minimize
    def objective_func(r):
        return -objective(r)

    # Define constraints for optimization
    cons = {'type': 'ineq', 'fun': constraints}

    # Set up optimization
    result = minimize(
        fun=objective_func,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'ftol': 1e-9, 'maxiter': 500}
    )

    # Extract final values
    final_radii = result.x
    final_centers = centers.copy()  # Keep the same centers
    sum_radii = np.sum(final_radii)

    return final_centers, final_radii, sum_radii
