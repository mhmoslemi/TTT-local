import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        constraints = []

        # Constraint: circles fully inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i][0], centers_flat[i][1], radii_flat[i]
            constraints.append(x_i - r_i >= 0)
            constraints.append(1 - x_i - r_i >= 0)
            constraints.append(y_i - r_i >= 0)
            constraints.append(1 - y_i - r_i >= 0)

        # Constraint: no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                distance = np.sqrt(dx**2 + dy**2)
                distance_min = radii_flat[i] + radii_flat[j]
                constraints.append(distance >= distance_min)

        return constraints

    # Set up the optimization problem
    x0 = np.concatenate([centers.flatten(), radii])
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1)] * 2)  # x and y
        bounds.extend([(0, 0.5)] * 1)  # radius (max radius is 0.5 to fit in square)

    # Use L-BFGS-B as it handles bounds well
    result = minimize(objective, x0, method='L-BFGS-B', bounds=bounds, constraints=constraints, tol=1e-8)

    # Extract results
    centers_final = result.x[:60].reshape(10, 2)
    radii_final = result.x[60:]
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
