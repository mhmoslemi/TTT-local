import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: 10 circles placed in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are within the unit square
        for i in range(10):
            if radii[i] > 0:
                if (centers[i, 0] - radii[i] < 0 or
                    centers[i, 0] + radii[i] > 1 or
                    centers[i, 1] - radii[i] < 0 or
                    centers[i, 1] + radii[i] > 1):
                    return np.inf
        # Ensure no circles overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 0.5) for _ in range(10)],
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    # Best solution found
    optimal_radii = result.x
    optimal_centers = centers.copy()
    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
