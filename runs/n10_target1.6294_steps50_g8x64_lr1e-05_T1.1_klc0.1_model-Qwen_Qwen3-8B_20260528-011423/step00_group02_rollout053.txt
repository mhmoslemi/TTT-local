import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a square grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: All circles inside the square
        # r_i <= x_i - r_i <= 1 - r_i --> x_i >= r_i and x_i <= 1 - r_i
        # Similarly for y_i
        x_centers = centers[:, 0]
        y_centers = centers[:, 1]
        r = radii
        for i in range(10):
            if x_centers[i] < r[i] or x_centers[i] > 1 - r[i] or y_centers[i] < r[i] or y_centers[i] > 1 - r[i]:
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x_centers[i] - x_centers[j]
                dy = y_centers[i] - y_centers[j]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Define bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimization
    result = minimize(objective, radii, bounds=bounds, constraints=constraints)
    
    # Extract results
    optimal_radii = result.x
    sum_radii = -result.fun

    # Validate result
    if not constraints(optimal_radii):
        warnings.warn("Optimization did not satisfy constraints; returning initial guess")
        optimal_radii = radii
        sum_radii = np.sum(optimal_radii)

    return (centers, optimal_radii, sum_radii)
