import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import Optimizer
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    # Objective function: minimize the negative sum of radii
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints
    def constraint_center_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1
        return 0

    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 1e-8:
                    return -1
        return 0

    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Bounds for centers (0 to 1)
    bounds_centers = [(0, 1) for _ in range(60)]
    # Bounds for radii (0 to 0.5)
    bounds_radii = [(0, 0.5) for _ in range(10)]
    bounds = bounds_centers + bounds_radii

    # Optimization
    result = minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    if result.success:
        centers_opt = result.x[:60].reshape(10, 2)
        radii_opt = result.x[60:]
        sum_radii_opt = -result.fun
    else:
        centers_opt = centers
        radii_opt = radii
        sum_radii_opt = np.sum(radii)

    return (centers_opt, radii_opt, sum_radii_opt)
