import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.array([[i*0.2 + 0.1, j*0.2 + 0.1] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x[10:])  # Negative for minimization
    
    # Constraints
    def constraints(x):
        centers = x[:10*2].reshape(10, 2)
        radii = x[10:]
        # Constraint 1: All circles inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii))
        # Constraint 2: No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        no_overlap = np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
        # Constraint 3: All radii non-negative
        non_negative = np.all(radii >= 0)
        return [inside, no_overlap, non_negative]
    
    # Prepare the optimization problem
    bounds = [(0, 1) for _ in range(10*2)] + [(0, 1) for _ in range(10)]
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                      method='SLSQP', bounds=bounds, constraints=({'type': 'ineq', 'fun': constraints}))
    
    # Extract results
    centers_opt = result.x[:10*2].reshape(10, 2)
    radii_opt = result.x[10:]
    sum_radii_opt = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii_opt)
