import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.08)

    def objective_function(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        return -np.sum(radii_flat)  # We minimize the negative sum to maximize it

    def constraints(x):
        # Constraint 1: Each circle is inside the unit square
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        constraints = []
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append(x_i - r_i >= 0)
            constraints.append(1 - x_i - r_i >= 0)
            constraints.append(y_i - r_i >= 0)
            constraints.append(1 - y_i - r_i >= 0)
        # Constraint 2: Circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                constraints.append(dist - (r_i + r_j) >= 0)
        return constraints

    # Set up the optimization problem
    x0 = np.concatenate([centers.flatten(), radii])
    bounds = []
    for i in range(10):
        bounds.extend([(0, 1) for _ in range(2)])  # x and y for each center
        bounds.extend([(-np.inf, 1 - 1e-6) for _ in range(10)])  # radius must be non-negative and small enough

    # Initial constraints setup
    cons = [{'type': 'ineq', 'fun': lambda x, i=i: constraints(x)[i]} for i in range(len(constraints(x0)))]

    result = minimize(objective_function, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii = -result.fun
    return (centers_result, radii_result, sum_radii)
