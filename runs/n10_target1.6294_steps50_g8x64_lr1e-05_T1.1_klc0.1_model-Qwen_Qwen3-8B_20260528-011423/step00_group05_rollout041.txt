import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess for the centers: randomly placed points
    centers = np.random.rand(10, 2)
    # Initial guess for the radii: small values
    radii = np.full(10, 0.05)
    
    def objective_function(radii_and_centers):
        # Extract the centers and radii
        centers = radii_and_centers[:20].reshape(10, 2)
        radii = radii_and_centers[20:]
        
        # Calculate the sum of radii
        sum_r = np.sum(radii)
        
        # Check if all circles are within the square
        for i in range(10):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf  # Penalize for circles out of bounds
        
        # Check if all circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return np.inf  # Penalize for overlapping circles
        
        return -sum_r  # Maximize sum of radii by minimizing -sum_r

    # Optimization
    result = minimize(
        objective_function,
        np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=[(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    )
    
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = -result.fun  # Since we minimized -sum_r

    return (centers_opt, radii_opt, sum_radii)
