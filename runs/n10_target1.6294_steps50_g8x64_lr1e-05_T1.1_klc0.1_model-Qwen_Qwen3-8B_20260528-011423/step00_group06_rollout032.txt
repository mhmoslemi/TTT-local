import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective_function(params):
        # Flatten parameters into centers and radii
        centers = params[:60].reshape(10, 2)
        radii = params[60:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(params):
        centers = params[:60].reshape(10, 2)
        radii = params[60:]
        constraints = []

        # Constraint 1: Each circle is within the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            constraints.append(x - r >= 0)
            constraints.append(1 - x - r >= 0)
            constraints.append(y - r >= 0)
            constraints.append(1 - y - r >= 0)

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]) >= 0)

        return constraints

    # Initial guess: Place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = 0.1 + 0.2 * (i % 5)
        centers[i, 1] = 0.1 + 0.2 * (i // 5)
    initial_params = np.concatenate([centers.flatten(), np.ones(10) * 0.05])

    # Set up optimization problem
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x and y for each center
        bounds.extend([(0, 0.5)])       # radius for each circle

    res = minimize(objective_function, initial_params, method='SLSQP',
                   constraints={'type': 'ineq', 'fun': constraints},
                   bounds=bounds, tol=1e-8)

    centers = res.x[:60].reshape(10, 2)
    radii = res.x[60:]
    sum_radii = -res.fun  # Negative because we minimized negative sum

    return (centers, radii, sum_radii)
