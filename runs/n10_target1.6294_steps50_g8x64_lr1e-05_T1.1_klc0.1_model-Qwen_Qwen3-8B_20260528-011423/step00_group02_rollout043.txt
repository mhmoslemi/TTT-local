import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints_func(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Constraint 2: No overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Initial guess for the optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization settings
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1 - 1e-6) for _ in range(2)])  # x, y for each center
        bounds.extend([(0, 1 - 1e-6) for _ in range(10)])  # radii

    # Use L-BFGS-B optimizer
    result = minimize(objective, x0, bounds=bounds, constraints={"type": "ineq", "fun": constraints_func},
                      method='SLSQP')

    # Extract results
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
