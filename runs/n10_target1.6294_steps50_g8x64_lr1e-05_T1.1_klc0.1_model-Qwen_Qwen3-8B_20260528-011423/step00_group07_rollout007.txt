import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced centers
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3],
    ])
    radii = np.full(10, 0.05)

    # Objective function: minimize negative sum of radii
    def objective(x):
        global centers, radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square
    def square_constraint(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        for i in range(10):
            x_i, y_i = centers[i]
            if not (radii[i] <= x_i <= 1 - radii[i] and radii[i] <= y_i <= 1 - radii[i]):
                return -1
        return 0

    # Constraints: circles must not overlap
    def non_overlap_constraint(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0

    # Bounds: x and y coordinates must be within [0, 1], radii >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])
    bounds.extend([(0.0, 0.5) for _ in range(10)])  # Radii limited to max 0.5 to fit in square

    # Optimization setup
    cons = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': non_overlap_constraint}
    ]

    # Run optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8,
        options={'maxiter': 500}
    )

    # Extract final solution
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return centers, radii, sum_radii
