```python
import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess for circle centers and radii
    # Place 5 circles on the bottom and 5 on the top, spaced evenly
    centers = np.zeros((10, 2))
    radii = np.zeros(10)
    for i in range(10):
        x = 0.2 + 0.2 * i
        y = 0.2 if i < 5 else 0.8
        centers[i] = [x, y]
        radii[i] = 0.1

    def objective(radii):
        return -np.sum(radii)  # We minimize -sum(radii) to maximize sum(radii)

    def constraints(radii):
        # Ensure all circles are within the square
        for i in range(10):
            if radii[i] < 0:
                return False
            # Check if center is within the square
            if (centers[i, 0] - radii[i] < 0) or (centers[i, 0] + radii[i] > 1) or \
               (centers[i, 1] - radii[i] < 0) or (centers[i, 1] + radii[i] > 1):
                return False
        return True

    def overlapping(radii):
        # Check if any two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx * dx + dy * dy)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    def constraint_func(radii):
        return [constraints(radii), overlapping(radii)]

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(10)]  # Radii cannot exceed 0.5
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraint_func)

    # Extract the result
    optimal_radii = res.x
    optimal_centers = copy.deepcopy(centers)
    sum_radii = np.sum(optimal_radii)

    return (optimal_centers, optimal_radii, sum_radii)
```
