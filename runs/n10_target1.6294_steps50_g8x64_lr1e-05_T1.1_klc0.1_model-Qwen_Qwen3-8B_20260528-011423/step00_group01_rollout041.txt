import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as dist

def run_packing():
    # Initial guess: evenly spaced centers
    centers = np.array([[i / 9 + 0.05, j / 9 + 0.05] for j in range(10) for i in range(10)])
    centers = centers[:10]  # Take first 10 to get a starting point

    # Initial radii: small value to start with
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraints on radii: non-negative
        bounds = [(0, 1) for _ in range(10)]
        # Constraint: distance between centers >= sum of radii
        dists = dist.cdist(centers, centers)
        for i in range(10):
            for j in range(i+1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return False
        # Constraint: circles are inside the unit square
        for i in range(10):

                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        return True

    # Use a local optimization method to maximize the sum of radii
    result = minimize(
        objective,
        radii,
        bounds=[(0, 1) for _ in range(10)],
        constraints=[{'type': 'ineq', 'fun': constraints}],
        method='SLSQP'
    )

    # Extract final result
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun  # Since we maximized the negative sum

    return (centers, radii, sum_radii)
