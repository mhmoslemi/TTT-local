import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: randomly place centers within the square and assign equal radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(x):
        # x contains all parameters: centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_r = np.sum(radii_flat)
        return -sum_r  # Minimize negative sum to maximize sum
    
    # Define constraints
    def constraints_func(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Constraint 1: All circles must be within the unit square
        in_bounds = np.logical_and(
            np.logical_and(centers_flat[:, 0] >= radii_flat, centers_flat[:, 0] <= 1 - radii_flat),
            np.logical_and(centers_flat[:, 1] >= radii_flat, centers_flat[:, 1] <= 1 - radii_flat)
        )
        # Constraint 2: No two circles overlap
        distances = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distances[i, j] = np.sqrt(dx**2 + dy**2)
                distances[j, i] = distances[i, j]
        overlaps = np.sum(distances < radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])
        return (in_bounds.all(), overlaps == 0)
    
    # Convert constraints into a scipy constraint format
    def constraint(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Constraint 1: All circles must be within the unit square
        in_bounds = np.logical_and(
            np.logical_and(centers_flat[:, 0] >= radii_flat, centers_flat[:, 0] <= 1 - radii_flat),
            np.logical_and(centers_flat[:, 1] >= radii_flat, centers_flat[:, 1] <= 1 - radii_flat)
        )
        # Constraint 2: No two circles overlap
        distances = np.zeros((10, 10))

                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]

        bounds.append((0, 1))  # x

        bounds.append((0, 0.5))  # radius

    try:
        result = minimize(
            objective,
            x0=np.concatenate([centers.flatten(), radii]),

            bounds=bounds,

        # Fallback to simpler method if optimization fails

            bounds=bounds,

    # Extract final solution
    centers_final = result.x[:60].reshape(10, 2)
