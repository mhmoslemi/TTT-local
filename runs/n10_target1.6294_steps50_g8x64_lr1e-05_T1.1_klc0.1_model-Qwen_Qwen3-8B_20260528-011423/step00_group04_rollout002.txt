import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
    radii = np.array([0.15] * 10)

    def objective(radii):
        return -np.sum(radii)

    def constraint_square(radii):
        return np.array([r <= 0.5 for r in radii])

    def constraint_overlap(centers, radii):
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraint_overlap_func(x):
        radii = x[10:]
        centers = x[:10].reshape(10, 2)
        return np.concatenate([constraint_square(radii), constraint_overlap(centers, radii)])

    def bounds_func(x):
        return [(0.0, 0.5) for _ in range(10)] + [(0.0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        fun=lambda x: -np.sum(x[10:]),
        x0=np.concatenate([centers.reshape(10, 2).ravel(), radii]),
        bounds=bounds_func,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraint_overlap_func(x)},
            {'type': 'ineq', 'fun': lambda x: np.array([r <= 0.5 for r in x[10:]])}
        ],
        method='SLSQP'
    )

    centers_opt = result.x[:10].reshape(10, 2)
    radii_opt = result.x[10:]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
