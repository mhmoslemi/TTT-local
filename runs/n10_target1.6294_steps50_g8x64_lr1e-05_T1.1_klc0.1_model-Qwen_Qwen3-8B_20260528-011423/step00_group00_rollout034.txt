    radii = np.array([0.1] * 10)
    
    def objective(radii, centers):

    def constraints(radii, centers):
        # Constraints: all circles inside the square
        for i in range(10):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Constraints: no overlapping circles
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    def objective_func(x):
        # x is a flat array of [r0, r1, ..., r9, x0, x1, ..., x9, y0, y1, ..., y9]
        radii = x[:10]
        centers = x[10:].reshape((10, 2))
        return -np.sum(radii)
    
    def constraint_func(x):
        radii = x[:10]
        centers = x[10:].reshape((10, 2))
        # Check if all circles are inside the square
        for i in range(10):
            if radii[i] < 0:
                return -np.inf
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):

        for i in range(10):

        return np.sum(radii)

    x0 = np.concatenate((radii, centers.flatten()))
