import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (spread evenly with some padding)
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial guess for radii
    radii = np.ones(10) * 0.1

    # Objective function to minimize (negative of sum of radii)
    def objective(x):
        return -np.sum(x[10:])

    # Constraints: radius <= 0.5 (max possible radius to fit in square)
    def constraint_radius(x):
        return x[10:] - 0.5

    # Constraints: distance between centers >= sum of radii
    def constraint_distance(x):
        centers_flat = x[:10*2].reshape((10, 2))
        radii_flat = x[10:]
        dists = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii_flat[np.newaxis, :] + radii_flat[:, np.newaxis])

    # Combine all constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_radius},
        {'type': 'ineq', 'fun': constraint_distance}
    ]

    # Bounds for coordinates (0 to 1) and radii (0 to 0.5)
    bounds = [(0.0, 1.0) for _ in range(10*2)] + [(0.0, 0.5) for _ in range(10)]

    # Optimization settings
    result = minimize(
        objective,
        x0=np.concatenate((centers.flatten(), radii)),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    # Extract result
    centers_result = result.x[:10*2].reshape((10, 2))
    radii_result = result.x[10:]
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
