import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles equally spaced along a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds():
        return [(0.0, 1.0) for _ in range(10)]

    # Use SLSQP method as it handles bounds and constraints well
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds(),
        constraints={'type': 'ineq', 'fun': constraints, 'args': (radii,)},
        options={'disp': False, 'ftol': 1e-9, 'eps': 1e-8}
    )

    # Generate new centers based on optimized radii
    # Note: This is a simplified approach; in a real optimization, centers would be variables too
    # Here we fix centers but adjust radii for demonstration
    optimized_radii = result.x
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])

    # Ensure circles are within the square
    optimized_radii = np.clip(optimized_radii, 0, 0.5)

    # Ensure no overlap
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = math.hypot(dx, dy)
            if dist < optimized_radii[i] + optimized_radii[j]:
                optimized_radii[i] = (dist - optimized_radii[j]) / 2
                optimized_radii[j] = (dist - optimized_radii[i]) / 2

    sum_radii = np.sum(optimized_radii)
    return (centers, optimized_radii, sum_radii)
