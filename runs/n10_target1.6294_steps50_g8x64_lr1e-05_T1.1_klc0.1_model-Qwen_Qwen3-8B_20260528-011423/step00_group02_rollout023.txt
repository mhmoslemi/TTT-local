import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: centers at grid points, radii initialized to 0.1
    centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    radii = np.full(10, 0.1)

    def objective(x):
        # x contains all variables: centers and radii
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize

    def constraints(x):
        # Constraint: circles must stay within the unit square
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False
        return True

    def non_overlapping(x):
        # Constraint: circles must not overlap
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        distances = pdist(centers_flat)
        radii_sum = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        return distances >= radii_sum.flatten()

    # Define optimization problem
    bounds = [(0.1, 0.9) for _ in range(20)] + [(0.0, 0.5) for _ in range(10)]
    constraints_list = [
        {'type': 'ineq', 'fun': constraints},
        {'type': 'ineq', 'fun': non_overlapping}
    ]

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.reshape(20), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints_list,
        options={'ftol': 1e-6, 'disp': False}
    )

    # Extract result
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
