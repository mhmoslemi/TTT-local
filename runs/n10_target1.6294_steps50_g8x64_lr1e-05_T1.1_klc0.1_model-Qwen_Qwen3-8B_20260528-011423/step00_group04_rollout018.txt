import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (random points)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial small radii

    def objective_function(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        x_centers = x[:60].reshape(10, 2)
        x_radii = x[60:].reshape(10)
        sum_radii = np.sum(x_radii)
        # Constraint 1: All circles must be fully inside the unit square
        for i in range(10):
            if x_radii[i] < 0:
                return np.inf
            if x_centers[i][0] < x_radii[i] or x_centers[i][0] > 1 - x_radii[i]:
                return np.inf
            if x_centers[i][1] < x_radii[i] or x_centers[i][1] > 1 - x_radii[i]:
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x_centers[i][0] - x_centers[j][0]
                dy = x_centers[i][1] - x_centers[j][1]
                distance = np.sqrt(dx*dx + dy*dy)
                if distance < x_radii[i] + x_radii[j]:
                    return np.inf
        return -sum_radii  # Minimize -sum_radii to maximize sum_radii

    # Flatten the initial guess
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization options
    options = {
        'disp': False,
        'ftol': 1e-9,
        'maxiter': 1000
    }

    # Perform optimization
    result = minimize(objective_function, initial_guess, method='L-BFGS-B', bounds=[(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)], options=options)

    # Extract the result
    optimal_centers = result.x[:60].reshape(10, 2)
    optimal_radii = result.x[60:].reshape(10)
    sum_radii = -result.fun

    # Ensure all radii are non-negative
    optimal_radii = np.maximum(optimal_radii, 0)

    # Return result
    return (optimal_centers, optimal_radii, sum_radii)
