import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            for j in range(i+1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Bounds for radii: minimum 0, maximum 0.5 (since circle must be inside unit square)
    bounds = [(0.0, 0.5) for _ in range(10)]
    
    # Optimization result
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        method='SLSQP',
        constraints={
            'type': 'eq',
            'fun': lambda r: -np.sum(r),
        },
        options={
            'ftol': 1e-8,
            'maxiter': 1000,
            'disp': False
        }
    )
    
    # Final radii and centers
    final_radii = result.x
    final_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    sum_radii = np.sum(final_radii)
    
    # Ensure all circles are within the unit square
    for i in range(10):
        x, y = final_centers[i]
        r = final_radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            final_radii[i] = min(r, x - r, 1 - x - r, y - r, 1 - y - r)
    
    sum_radii = np.sum(final_radii)
    
    return (final_centers, final_radii, sum_radii)
