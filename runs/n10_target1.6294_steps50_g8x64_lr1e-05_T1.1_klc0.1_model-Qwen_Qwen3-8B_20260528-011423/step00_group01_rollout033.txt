import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place 10 circles with equal radii at random positions
    def initial_guess():
        centers = np.random.rand(10, 2)
        radii = np.array([0.1] * 10)
        return centers, radii

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints:
    # 1. Each circle is inside the unit square
    # 2. No two circles overlap
    # 3. Radii must be non-negative

    def constraint_inside(radii, centers):
        return np.all(radii <= 1 - centers[:, 0] - radii) & np.all(radii <= centers[:, 0] - radii) & \
               np.all(radii <= 1 - centers[:, 1] - radii) & np.all(radii <= centers[:, 1] - radii)

    def constraint_overlap(radii, centers):
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[i, j] = dist - (radii[i] + radii[j])
        return np.all(dists >= 0)

    # Optimization
    def optimize():
        # Initial guess
        centers, radii = initial_guess()
        # Create bounds for radii (non-negative)
        bounds = [(0, 1) for _ in range(10)]
        # Constraints
        constraints = [
            {'type': 'ineq', 'fun': lambda x: constraint_inside(x, centers)},
            {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers)}
        ]
        # Optimize
        result = minimize(objective, radii, bounds=bounds, constraints=constraints)
        return result.x, centers, -result.fun

    # Run optimization
    radii, centers, sum_radii = optimize()
    
    return (centers, radii, sum_radii)
