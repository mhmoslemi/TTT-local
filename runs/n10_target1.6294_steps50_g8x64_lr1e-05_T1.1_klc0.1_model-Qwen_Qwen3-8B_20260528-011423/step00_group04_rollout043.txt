import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # We minimize -sum to maximize sum

    def constraints(radii):
        n = len(radii)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    # Bounds for radii
    bounds = [(0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    # Extract results
    radii_opt = result.x
    sum_radii = -result.fun

    # Validate constraints (optional, but ensure the circles don't go out of bounds)
    for i in range(10):
        x, y = centers[i]
        r = radii_opt[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            raise ValueError("Circle out of bounds")

    return (centers, radii_opt, sum_radii)
