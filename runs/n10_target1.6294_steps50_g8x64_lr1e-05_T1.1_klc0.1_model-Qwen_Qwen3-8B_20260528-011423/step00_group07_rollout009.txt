```python
import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective_function(x, radii):
        return -np.sum(radii)

    def constraint_circle_bounds(x, radius):
        x_pos = x[0]
        y_pos = x[1]
        return [x_pos - radius >= 0, 1 - x_pos - radius >= 0, y_pos - radius >= 0, 1 - y_pos - radius >= 0]

    def constraint_circle_overlap(x1, x2, r1, r2):
        dx = x1[0] - x2[0]
        dy = x1[1] - x2[1]
        distance_squared = dx**2 + dy**2
        return distance_squared - (r1 + r2)**2

    def constraints(x, radii):
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                x_i = x[2 * i:2 * i + 2]
                x_j = x[2 * j:2 * j + 2]
                r_i = radii[i]
                r_j = radii[j]
                cons = {'type': 'ineq', 'fun': lambda x: constraint_circle_overlap(x_i, x_j, r_i, r_j)}
                constraints.append(cons)
        for i in range(10):
            x_i = x[2 * i:2 * i + 2]
            r_i = radii[i]
            cons = {'type': 'ineq', 'fun': lambda x: constraint_circle_bounds(x, r_i)[0]}
            constraints.append(cons)
            cons = {'type': 'ineq', 'fun': lambda x: constraint_circle_bounds(x, r_i)[1]}
            constraints.append(cons)
            cons = {'type': 'ineq', 'fun': lambda x: constraint_circle_bounds(x, r_i)[2]}
            constraints.append(cons)
            cons = {'type': 'ineq', 'fun': lambda x: constraint_circle_bounds(x, r_i)[3]}
            constraints.append(cons)
        return constraints

    initial_guess = np.random.rand(20)
    initial_radii = np.random.rand(10)
    initial_radii /= 10  # Initial guess for radii

    result = minimize(
        fun=objective_function,
        x0=initial_guess,
        args=(initial_radii,),
        method='SLSQP',
        constraints=constraints(initial_guess, initial_radii),
        bounds=[(0, 1) for _ in range(20)],
        options={'ftol': 1e-10, 'maxiter': 1000}
    )

    centers = result.x.reshape((10, 2))
    radii = result.fun / -1  # Since objective function is -sum(radii)
    return (centers, radii, sum(radii))
```
