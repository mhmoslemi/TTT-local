import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: spread circles evenly in the square
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraint_circle_inside(radii_vec):
        return np.sum((radii_vec > 0) * (radii_vec > 1 - np.linalg.norm(centers, axis=1) - np.abs(1 - np.linalg.norm(centers, axis=1)))) 

    def constraint_no_overlap(radii_vec):
        dist = np.linalg.norm(centers[:, np.newaxis, :] - centers[np.newaxis, :, :], axis=2)
        return np.sum((dist < radii_vec[:, np.newaxis] + radii_vec[np.newaxis, :]) * (dist > 0))

    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    res = minimize(objective, radii, method='SLSQP', constraints=constraints, bounds=[(0, 1 - np.linalg.norm(centers, axis=1)) for _ in range(10)])
    
    centers = np.clip(centers, radii[:, np.newaxis], 1 - radii[:, np.newaxis])
    sum_radii = np.sum(res.x)
    return (centers, res.x, sum_radii)
