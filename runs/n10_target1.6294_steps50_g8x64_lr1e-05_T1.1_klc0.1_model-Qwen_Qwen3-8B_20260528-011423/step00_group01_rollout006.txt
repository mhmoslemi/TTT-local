import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    # Start with equally spaced points and small radii
    n_circles = 10
    centers = np.random.rand(n_circles, 2)
    radii = np.full(n_circles, 0.05)

    def objective_function(x):
        centers_flat = x[:n_circles*2].reshape(n_circles, 2)
        radii_flat = x[n_circles*2:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraint_inside(x):
        centers_flat = x[:n_circles*2].reshape(n_circles, 2)
        radii_flat = x[n_circles*2:]
        for i in range(n_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1.0
        return 0.0

    def constraint_overlap(x):
        centers_flat = x[:n_circles*2].reshape(n_circles, 2)
        radii_flat = x[n_circles*2:]
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii_flat[i] + radii_flat[j]:
                    return -1.0
        return 0.0

    # Initial guess flattened
    x0 = np.concatenate([centers.flatten(), radii])

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds for centers (0 to 1) and radii (0 to 0.5)
    bounds = [(0, 1) for _ in range(n_circles * 2)] + [(0, 0.5) for _ in range(n_circles)]

    # Optimization
    result = minimize(
        fun=objective_function,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract results
    centers_result = result.x[:n_circles*2].reshape(n_circles, 2)
    radii_result = result.x[n_circles*2:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
