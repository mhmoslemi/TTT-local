import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.15)

    def objective(radii):
        return -np.sum(radii)

    def constraint_inside(radii):
        for i, r in enumerate(radii):
            x, y = centers[i]
            if not (x - r >= 0 and x + r <= 1 and y - r >= 0 and y + r <= 1):
                return -np.inf
        return 0

    def constraint_overlap(radii):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    # Optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    result = minimize(objective, radii, method='SLSQP', constraints=constraints)

    # Update centers based on optimized radii
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
