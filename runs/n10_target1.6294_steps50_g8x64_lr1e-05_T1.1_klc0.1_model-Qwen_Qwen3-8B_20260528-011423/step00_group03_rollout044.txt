import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[i * 0.25, j * 0.25] for i in range(10) for j in range(10)])
    centers = centers[:10]
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[i * 0.25, j * 0.25] for i in range(10) for j in range(10)])
        centers = centers[:10]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx ** 2 + dy ** 2)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    def bounds(radii):
        return [(0, 1 - 0.01) for _ in range(10)]

    # Initial guess for radii
    initial_guess = np.full(10, 0.1)

    # Optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': lambda r: constraints(r)}
    )

    radii = result.x
    centers = np.array([[i * 0.25, j * 0.25] for i in range(10) for j in range(10)])
    centers = centers[:10]

    # Ensure all circles are within the square
    for i in range(10):
        if radii[i] < 0:
            radii[i] = 0
        if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
            radii[i] = 0
        if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
            radii[i] = 0

    sum_radii = np.sum(radii)
    if sum_radii > 1.6294:
        sum_radii = 1.6294

    return (centers, radii, sum_radii)
