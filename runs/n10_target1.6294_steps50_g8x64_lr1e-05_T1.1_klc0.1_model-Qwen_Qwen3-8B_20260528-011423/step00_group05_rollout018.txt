import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(centers, radii):
        # Box constraints: circles must be fully inside the unit square
        box_constraints = []
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            box_constraints.append((x - r >= 0))
            box_constraints.append((x + r <= 1))
            box_constraints.append((y - r >= 0))
            box_constraints.append((y + r <= 1))
        # Non-overlapping constraint: distance between centers >= sum of radii
        dist_constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dist_constraints.append(dist - (radii[i] + radii[j]) >= 0)
        return box_constraints + dist_constraints

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=RuntimeWarning)
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            jac=lambda r: -np.ones_like(r),
            bounds=[(0, 1) for _ in range(10)],
            constraints=[{'type': 'ineq', 'fun': lambda r: constraints(centers, r)}]
        )

    centers_result = centers
    radii_result = result.x
    sum_radii_result = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii_result)
