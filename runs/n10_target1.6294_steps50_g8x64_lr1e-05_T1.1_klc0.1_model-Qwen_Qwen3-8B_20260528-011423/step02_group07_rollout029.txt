import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Check if all radii are non-negative
        for r in radii:
            if r < 0:
                return np.inf
        # Check if all circles are inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
        # Check if all circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Setup bounds and constraints for optimization
    bounds = [(0, 1) for _ in range(10)]
    cons = {'type': 'ineq', 'fun': constraints}

    # Run the optimization
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Return the centers, radii, and sum of radii
    return (centers, res.x, -res.fun)
