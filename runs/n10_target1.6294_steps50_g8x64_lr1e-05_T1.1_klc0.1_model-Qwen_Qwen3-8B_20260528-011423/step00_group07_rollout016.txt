import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(x):
        # x is flattened array of centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        
        # Constraint 1: All circles are within the unit square
        for i in range(10):
            x_i, y_i, r_i = centers[i][0], centers[i][1], radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: No circles overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        
        return 0
    
    # Flatten centers and radii
    x0 = np.concatenate([centers.ravel(), radii])
    bounds = [(0, 1) for _ in range(60)] + [(0, 0.5) for _ in range(10)]
    
    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    
    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
