import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles evenly spaced with small radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)

    def objective(r):
        return -np.sum(r)  # Minimize negative of sum to maximize

    def constraints(r, centers):
        # Ensure all circles are inside the unit square
        r = np.abs(r)
        return np.concatenate([
            r - (centers[:, 0] - r) + (1 - centers[:, 0] - r),
            r - (centers[:, 1] - r) + (1 - centers[:, 1] - r)
        ])

    def overlap_constraints(r, centers):
        # Ensure no two circles overlap
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (r[:, np.newaxis] + r[np.newaxis, :])

    # Optimization settings
    bounds = [(0, 1) for _ in range(10)]
    constraints_list = [
        {'type': 'ineq', 'fun': lambda r: constraints(r, centers)},
        {'type': 'ineq', 'fun': lambda r: overlap_constraints(r, centers).flatten()}
    ]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints_list, tol=1e-6)

    centers = np.random.rand(10, 2)  # Random initial centers for better packing
    best_radii = np.clip(result.x, 0, 1)
    sum_radii = np.sum(best_radii)
    return (centers, best_radii, sum_radii)
