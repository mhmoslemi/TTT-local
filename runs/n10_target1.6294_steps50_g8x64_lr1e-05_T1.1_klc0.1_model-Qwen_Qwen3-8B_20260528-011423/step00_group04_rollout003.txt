import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a square grid
    def initial_guess():
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
        return centers

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraint 1: Each circle is inside the unit square
    def constraint_inside(x):
        return np.array([x[i] <= 1 - x[i + 10] for i in range(10)] +
                        [x[i] <= 1 - x[i + 10] for i in range(10)])  # x and y direction

    # Constraint 2: No overlaps between any two circles
    def constraint_overlap(x):
        dists = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x[i * 2] - x[j * 2]
                dy = x[i * 2 + 1] - x[j * 2 + 1]
                dists[idx] = np.sqrt(dx * dx + dy * dy) - (x[i] + x[j])
                idx += 1
        return dists

    # Initial guess for all 20 variables (10 centers * 2 + 10 radii)
    initial_center = initial_guess()
    x0 = np.concatenate([initial_center.flatten(), np.ones(10)])

    # Bounds: radii must be >= 0 and centers must be between 0 and 1 - radius
    bounds = []
    for i in range(10):
        lb = 0.0
        ub = 1.0 - x0[i * 2] - x0[i * 2 + 1]
        bounds.append((lb, ub))
    for i in range(10):
        bounds.append((0.0, 1.0))

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    result = minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        options={'ftol': 1e-9, 'disp': False}
    )

    # Extract the result
    centers = np.reshape(result.x[:20], (10, 2))
    radii = result.x[20:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
