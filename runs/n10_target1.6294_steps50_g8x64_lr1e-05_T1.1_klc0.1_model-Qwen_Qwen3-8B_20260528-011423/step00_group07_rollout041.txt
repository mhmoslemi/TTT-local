import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints_func(x):
        # x is the flattened array of centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Check if all circles are inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers[i][0], centers[i][1], radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Check if all circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)  # Return negative of sum for minimization

    # Initial guess: Place 10 circles at the corners of a grid
    centers = np.zeros((10, 2))
    radii = np.zeros(10)
    for i in range(10):
        row = i // 3
        col = i % 3
        centers[i] = [col / 2 + 0.25, row / 2 + 0.25]
        radii[i] = 0.05
    x0 = np.concatenate((centers.flatten(), radii))

    # Bounds: x and y coordinates in [0, 1], radii >= 0
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
        bounds.append((0, 1))  # radius
    bounds = [(0, 1)] * 30  # Simplified bounds for all 30 variables

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=[{'type': 'ineq', 'fun': constraints_func}])
    x_opt = result.x
    centers_opt = x_opt[:20].reshape(10, 2)
    radii_opt = x_opt[20:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
