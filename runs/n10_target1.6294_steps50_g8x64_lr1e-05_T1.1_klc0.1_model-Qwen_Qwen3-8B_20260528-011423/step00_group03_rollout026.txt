import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([(i/3, j/3) for i in range(4) for j in range(4)])
    centers = centers[:10]
    radii = np.ones(10) * 0.1  # Initial estimate for radii
    sum_radii = 1.0

    # Define objective function: minimize negative sum of radii
    def objective(x):
        nonlocal sum_radii
        # x is a flattened array: centers are in x[0:20], radii in x[20:30]
        centers = x[:20].reshape(10, 2)
        radii = x[20:].clip(min=0)  # Ensure non-negative radii

        # Compute sum of radii
        sum_radii = np.sum(radii)
        return -sum_radii

    # Define constraints
    def constraint_centers(x):
        centers = x[:20].reshape(10, 2)
        return np.concatenate([
            np.maximum(centers[:, 0] - radii, 0),  # x >= r
            np.maximum(1 - centers[:, 0] - radii, 0),  # 1 - x >= r
            np.maximum(centers[:, 1] - radii, 0),  # y >= r
            np.maximum(1 - centers[:, 1] - radii, 0)  # 1 - y >= r
        ])

    def constraint_overlap(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Set up optimization
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    cons = [
        {'type': 'ineq', 'fun': constraint_centers},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    result = minimize(objective, x0=np.concatenate([centers.ravel(), radii]), method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    if result.success:
        centers = result.x[:20].reshape(10, 2)
        radii = result.x[20:]
        sum_radii = -result.fun
        return (centers, radii, sum_radii)
    else:
        # Fallback to initial guess if optimization fails
        return (centers, radii, sum_radii)
