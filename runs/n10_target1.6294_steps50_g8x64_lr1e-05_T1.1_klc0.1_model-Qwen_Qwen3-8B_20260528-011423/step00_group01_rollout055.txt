import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: distribute circles uniformly
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = 0.1 + 0.8 * i / 9
        centers[i, 1] = 0.5

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: circles must be inside the unit square
    def in_square(radii, centers):
        return np.all(radii <= 1 - np.min(np.vstack((centers[:, 0], np.ones(10))), axis=0))

    # Constraint: no two circles overlap
    def no_overlap(radii, centers):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Bounds: radii must be non-negative
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        np.array([0.1] * 10),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: in_square(r, centers)},  # In square
            {'type': 'ineq', 'fun': lambda r: no_overlap(r, centers)}   # No overlap
        ]
    )

    # Generate better initial centers for a more accurate optimization
    better_centers = np.zeros((10, 2))
    for i in range(10):
        angle = 2 * np.pi * i / 10
        better_centers[i, 0] = 0.5 + 0.4 * np.cos(angle)
        better_centers[i, 1] = 0.5 + 0.4 * np.sin(angle)

    # Another round of optimization with better centers
    result2 = minimize(
        objective,
        np.array([0.1] * 10),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: in_square(r, better_centers)},  # In square
            {'type': 'ineq', 'fun': lambda r: no_overlap(r, better_centers)}   # No overlap
        ]
    )

    # Final optimization with perturbed centers
    perturbed_centers = better_centers + np.random.uniform(-0.1, 0.1, size=better_centers.shape)
    result3 = minimize(
        objective,
        np.array([0.1] * 10),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: in_square(r, perturbed_centers)},  # In square
            {'type': 'ineq', 'fun': lambda r: no_overlap(r, perturbed_centers)}   # No overlap
        ]
    )

    # Return the best result
    best_result = max([result, result2, result3], key=lambda x: x.fun)
    centers_final = perturbed_centers
    radii_final = -best_result.fun * np.ones(10)
    sum_radii_final = -best_result.fun

    return (centers_final, radii_final, sum_radii_final)
