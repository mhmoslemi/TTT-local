import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.ones(10) * 0.05

    # Objective function: minimize the negative sum of radii
    def objective(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return -np.sum(radii)

    # Constraints: each circle must be within the unit square
    def square_constraint(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return np.concatenate([
            centers[:, 0] - radii,
            1 - centers[:, 0] - radii,
            centers[:, 1] - radii,
            1 - centers[:, 1] - radii
        ])

    # Constraints: circles must not overlap
    def no_overlap_constraint(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        distances = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                distances[idx] = dist - (radii[i] + radii[j])
                idx += 1
        return distances

    # Bounds: x and y coordinates between 0 and 1, radii non-negative
    bounds = []
    for i in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y of each center
        bounds.append((0.0, 1.0))  # radius of each circle

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': no_overlap_constraint}
    ]

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([initial_centers.ravel(), initial_radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract results
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
