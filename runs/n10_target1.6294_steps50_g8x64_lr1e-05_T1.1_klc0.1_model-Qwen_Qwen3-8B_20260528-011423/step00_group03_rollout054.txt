import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: all circles are within the unit square
        for i in range(10):
            if radii[i] <= 0:
                return False
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        # Constraint 2: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(10)]
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=({'type': 'ineq', 'fun': lambda r: constraints(r)}),
        tol=1e-6
    )

    # Extract the optimal solution
    optimal_radii = result.x
    sum_radii = -result.fun

    return (centers, optimal_radii, sum_radii)
