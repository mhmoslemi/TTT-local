import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1
    radii[0] = 0.15  # Start with a larger radius for the first circle

    # Objective function: minimize the negative sum of radii
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints: 1. Circles are inside the unit square
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.hstack([centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat,
                          centers_flat[:, 1] - radii_flat, 1 - centers_flat[:, 1] - radii_flat])

    # Constraints: 2. Circles are non-overlapping
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        dists = np.sqrt(np.sum((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :])**2, axis=2))
        return dists - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Constraints: 3. All radii >= 0
    def constraint_radii(x):
        return x[60:]

    # Set up optimization problem
    bounds = []
    for _ in range(10):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
    for _ in range(10):
        bounds.append((0, 0.5))  # radius

    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_radii}
    ]

    # Initial guess for the optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization settings
    options = {'ftol': 1e-6, 'maxiter': 1000}
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=RuntimeError)
        result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, options=options)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
