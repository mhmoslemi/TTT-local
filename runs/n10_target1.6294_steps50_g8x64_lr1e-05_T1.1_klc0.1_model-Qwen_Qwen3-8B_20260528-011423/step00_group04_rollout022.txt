import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return False
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define bounds for radii: [0, 0.5] (since radius can't exceed 0.5)
    bounds = [(0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        method='SLSQP'
    )

    # Extract results
    radii_result = result.x
    sum_radii = -result.fun

    # Ensure constraints are satisfied (for safety)
    if not constraints(radii_result):
        raise ValueError("Constraints not satisfied after optimization")

    return (centers, radii_result, sum_radii)
