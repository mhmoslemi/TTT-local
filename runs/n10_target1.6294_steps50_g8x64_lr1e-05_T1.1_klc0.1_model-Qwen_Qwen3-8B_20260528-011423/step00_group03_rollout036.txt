import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly spread in the unit square)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial radii

    # Objective function to minimize: negative of sum of radii
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraint: each circle is inside the unit square
    def in_square(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.concatenate([centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat,
                               centers_flat[:, 1] - radii_flat, 1 - centers_flat[:, 1] - radii_flat])

    # Constraint: circles do not overlap
    def no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distances = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distances[i, j] = np.sqrt(dx * dx + dy * dy)
                distances[j, i] = distances[i, j]
        return distances - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Bounds for centers: [0, 1] for x and y
    bounds_centers = [(0, 1) for _ in range(60)]

    # Bounds for radii: [0, 0.5] (since the maximum radius for a circle to fit is 0.5)
    bounds_radii = [(0, 0.5) for _ in range(10)]

    # Combine bounds
    bounds = bounds_centers + bounds_radii

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': in_square},
        {'type': 'ineq', 'fun': lambda x: no_overlap(x).flatten()}
    ]

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                      bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun  # Since we minimized the negative sum

    return (centers_opt, radii_opt, sum_radii)
