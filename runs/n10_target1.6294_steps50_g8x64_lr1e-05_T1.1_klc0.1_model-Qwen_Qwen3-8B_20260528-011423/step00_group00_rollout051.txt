import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.array([0.1] * 10)

    def objective_function(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return radii <= 1 - 0.1  # Assuming initial guess has radius 0.1

    def constraint_no_overlap(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        distances = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return distances - radii_sum

    # Optimization bounds
    bounds = [(0, 1 - 0.1) for _ in range(10)]

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization
    result = minimize(objective_function, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Final centers (same as initial for simplicity; more advanced code would optimize them too)
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
