import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective_function(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1.0
        return 0.0

    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1.0
        return 0.0

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii.flatten()])

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds: x and y must be in [0,1], radii must be non-negative
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])  # x, y, r

    # Optimization
    result = minimize(objective_function, x0, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
