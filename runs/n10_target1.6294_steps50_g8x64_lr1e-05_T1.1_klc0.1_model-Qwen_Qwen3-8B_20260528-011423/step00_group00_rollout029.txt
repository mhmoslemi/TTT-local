import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced circles in grid
    centers = np.zeros((10, 2))
    radii = np.full(10, 0.1)
    for i in range(10):
        centers[i, 0] = 0.1 + (i % 5) * 0.2
        centers[i, 1] = 0.1 + (i // 5) * 0.2

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        n = len(radii)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    def bounds(radii):
        return [(0.0, 1.0) for _ in range(len(radii))]

    result = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        bounds=bounds(radii),
        constraints=[
            {'type': 'ineq', 'fun': lambda x: np.array(constraints(x, centers))},
        ],
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Check if optimization succeeded
    if not result.success:
        warnings.warn("Optimization did not converge.")
        return centers, radii, np.sum(radii)

    # Update centers and radii
    new_centers = np.copy(centers)
    new_radii = result.x
    for i in range(10):
        # Ensure circles are fully inside the square
        new_radii[i] = min(new_radii[i], 1.0 - new_centers[i, 0])
        new_radii[i] = min(new_radii[i], 1.0 - new_centers[i, 1])
        new_radii[i] = max(new_radii[i], 0.0)
        # Adjust center to ensure circle is inside
        new_centers[i, 0] = max(0.0 + new_radii[i], min(1.0 - new_radii[i], new_centers[i, 0]))
        new_centers[i, 1] = max(0.0 + new_radii[i], min(1.0 - new_radii[i], new_centers[i, 1]))

    sum_radii = np.sum(new_radii)
    return new_centers, new_radii, sum_radii
