import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (random points in the unit square)
    centers = np.random.uniform(0, 1, (10, 2))
    radii = np.full(10, 0.1)  # Initial radius guess
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize radii
    
    def constraint_circle_inside(radii, centers):
        return np.sum((radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) & 
                     (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1]))
    
    def constraint_no_overlap(centers, radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Optimization bounds
    bounds = [(0, 1) for _ in range(10)]  # Radii must be between 0 and 1
    
    # Set up the optimization problem
    res = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        bounds=bounds,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraint_circle_inside(x, centers)},
            {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(centers, x)}
        ]
    )
    
    # Extract results
    radii = res.x
    centers = np.random.uniform(0, 1, (10, 2))  # Re-initialize centers as the solution is not tracked in the optimization
    sum_radii = np.sum(radii)
    
    return centers, radii, sum_radii
