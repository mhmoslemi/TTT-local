import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraints for being inside the unit square
        inside = np.all((radii <= 1 - initial_centers[:, 0]) & (radii <= 1 - initial_centers[:, 1]), axis=1)
        # Constraint for non-overlapping
        dists = cdist(initial_centers, initial_centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        non_overlap = np.all(dists >= radii_sum, axis=1)
        return np.concatenate((inside, non_overlap))

    # Prepare variables for optimization
    bounds = [(0, 1) for _ in range(10)]
    res = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints)

    centers = initial_centers
    radii = res.x
    sum_radii = -res.fun

    return (centers, radii, sum_radii)
