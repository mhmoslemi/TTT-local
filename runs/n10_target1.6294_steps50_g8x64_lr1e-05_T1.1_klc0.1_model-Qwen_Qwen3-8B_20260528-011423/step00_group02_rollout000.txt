import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.full(10, 0.08)

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: each circle must be inside the unit square
    def square_constraint(radii):
        return np.sum((radii <= 0.5) * 1)  # Only consider radii <= 0.5

    # Constraint 2: no overlap between circles
    def no_overlap_constraint(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        distances = np.zeros((10, 10))
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                distances[i, j] = dist - (radii[i] + radii[j])
        return np.min(distances)

    # Optimization settings
    bounds = [(0, 0.5) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': no_overlap_constraint}
    ]

    # Run optimization
    result = minimize(objective, initial_radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract result
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
