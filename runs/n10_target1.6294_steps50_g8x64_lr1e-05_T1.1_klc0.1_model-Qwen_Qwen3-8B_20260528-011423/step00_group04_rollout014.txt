import numpy as np
from scipy.optimize import minimize
from itertools import combinations

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    radii = np.ones(10) * 0.1
    for i in range(10):
        centers[i, 0] = (i % 5) * 0.3
        centers[i, 1] = (i // 5) * 0.3

    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            # Ensure circles are inside the unit square
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
            # Ensure circles do not overlap
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)  # Return negative sum for optimization

    bounds = [(0, 0.5) for _ in range(10)]

    result = minimize(objective, radii, bounds=bounds, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}])

    centers = np.zeros((10, 2))
    for i in range(10):
        x, y = centers[i]
        r = result.x[i]
        # Use a simple grid to place circles for visualization
        centers[i, 0] = (i % 5) * 0.3
        centers[i, 1] = (i // 5) * 0.3
        # Adjust the centers to prevent overlap (simplified for demonstration)
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < result.x[i] + result.x[j]:
                # Move circles apart by half the overlap
                overlap = (result.x[i] + result.x[j]) - dist
                dir_x = dx / dist if dist > 0 else 0
                dir_y = dy / dist if dist > 0 else 0
                centers[i, 0] += dir_x * overlap / 2
                centers[i, 1] += dir_y * overlap / 2
                centers[j, 0] -= dir_x * overlap / 2
                centers[j, 1] -= dir_y * overlap / 2

    # Final check: ensure all circles are inside the square
    for i in range(10):
        if not (result.x[i] <= centers[i, 0] <= 1 - result.x[i] and result.x[i] <= centers[i, 1] <= 1 - result.x[i]):
            result.x[i] = min(result.x[i], 1 - result.x[i])  # Adjust radius to ensure inside
            centers[i, 0] = np.clip(centers[i, 0], result.x[i], 1 - result.x[i])
            centers[i, 1] = np.clip(centers[i, 1], result.x[i], 1 - result.x[i])

    return (centers, result.x, -result.fun)
