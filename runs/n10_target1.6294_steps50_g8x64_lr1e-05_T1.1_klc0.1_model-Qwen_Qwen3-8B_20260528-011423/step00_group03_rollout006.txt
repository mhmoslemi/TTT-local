import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(centers, radii):
        # Constraint 1: circles must be inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii), axis=1)
        if not np.all(inside):
            return np.inf

        # Constraint 2: circles must not overlap
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf

        return 0

    def objective_with_constraints(x):
        # x has shape (20,)
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Check constraints
        if np.any(radii < 0):
            return np.inf
        if not np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                      (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii)):
            return np.inf
        # Check overlaps
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Define bounds for the parameters: centers (x, y) and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    
    # Initial guess
    initial_guess = []
    for i in range(10):
        initial_guess.extend([0.1 + i*0.2, 0.1 + (i//5)*0.2, 0.05])

    # Run optimization
    result = minimize(objective_with_constraints, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-6)
    
    # Extract the result
    centers = np.array(result.x[:20]).reshape(10, 2)
    radii = result.x[20:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
