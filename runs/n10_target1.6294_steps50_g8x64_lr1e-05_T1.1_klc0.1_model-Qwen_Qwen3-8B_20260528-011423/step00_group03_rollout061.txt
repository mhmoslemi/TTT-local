import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced circles with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i+1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={"type": "success", "fun": constraints})

    # Check if optimization succeeded
    if not result.success:
        return np.array([[-1, -1]] * 10), np.zeros(10), 0.0

    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
