import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[i / 9 + 0.05, j / 9 + 0.05] for i in range(10) for j in range(10)][:10])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # We minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        def inside(r):
            return np.all((radii <= (1 - radii)) & (radii > 0))

        # Constraint 2: No two circles overlap
        def no_overlap(r):
            dists = cdist(centers, centers)
            return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

        return np.array([inside(radii), no_overlap(radii)])

    bounds = [(0, 1 - 1e-6) for _ in range(10)]  # Ensure circles fit in the unit square

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    return (centers, result.x, -result.fun)
