import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)

    def constraint_circle_inside(radii, centers):
        # Ensure circles are inside the unit square
        return np.sum((centers - radii[:, np.newaxis]) >= 0) + np.sum((1 - centers - radii[:, np.newaxis]) >= 0)

    def constraint_no_overlap(radii, centers):
        # Ensure circles do not overlap
        distances = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum(distances >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Define optimization problem
    bounds = [(0, 0.5) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
    ]

    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Update centers to be consistent with the final radii
    # Use a simple approach to adjust centers without overlapping
    # This is a simplification and may not yield the optimal result
    # For the purpose of this example, we assume it's acceptable
    new_centers = np.copy(centers)
    for i in range(10):
        for j in range(10):
            if i != j:
                dx = new_centers[i, 0] - new_centers[j, 0]
                dy = new_centers[i, 1] - new_centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < result.fun + result.x[i] + result.x[j]:
                    overlap = dist - (result.x[i] + result.x[j])
                    # Move the circle with smaller radius more
                    if result.x[i] < result.x[j]:
                        new_centers[i, 0] += overlap / 2
                        new_centers[i, 1] += overlap / 2
                    else:
                        new_centers[j, 0] -= overlap / 2
                        new_centers[j, 1] -= overlap / 2

    return new_centers, result.x, np.sum(result.x)
