import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced circles in grid
    centers = np.zeros((10, 2))
    radii = np.full(10, 0.1)
    for i in range(10):
        centers[i, 0] = 0.1 + (i % 5) * 0.2
        centers[i, 1] = 0.1 + (i // 5) * 0.2

    def objective(radii):
        return -np.sum(radii)

    def constraint_func(radii):
        n = len(radii)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    def boundary_constraint(radii):
        bounds = []
        for i in range(len(radii)):
            # Ensure circles are within the square
            bounds.append(radii[i] - (1.0 - centers[i, 0]))
            bounds.append(radii[i] - (1.0 - centers[i, 1]))
            bounds.append(radii[i] - centers[i, 0])
            bounds.append(radii[i] - centers[i, 1])
        return np.array(bounds)

    # Define constraints for optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint_func},
        {'type': 'ineq', 'fun': boundary_constraint}
    ]

    # Define bounds for radii
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Run optimization
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Check if optimization succeeded
    if not result.success:
        warnings.warn("Optimization did not converge.")
        return centers, radii, np.sum(radii)

    # Update centers and radii
    new_radii = result.x
    new_centers = np.copy(centers)
    for i in range(10):
        # Ensure circles are fully inside the square
        new_radii[i] = min(new_radii[i], 1.0 - new_centers[i, 0])
        new_radii[i] = min(new_radii[i], 1.0 - new_centers[i, 1])
        new_radii[i] = max(new_radii[i], 0.0)

        # Adjust center to ensure circle is inside
        new_centers[i, 0] = max(0.0 + new_radii[i], min(1.0 - new_radii[i], new_centers[i, 0]))
        new_centers[i, 1] = max(0.0 + new_radii[i], min(1.0 - new_radii[i], new_centers[i, 1]))

    sum_radii = np.sum(new_radii)
    return new_centers, new_radii, sum_radii
