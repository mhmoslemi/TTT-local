import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([
            [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
            [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
        ])
        # Constraint: all circles are within the square
        for i in range(10):
            if not (radii[i] <= 0.1 and radii[i] <= 0.9):
                return np.inf
        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    res = minimize(objective, initial_radii, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints})
    radii = res.x
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
