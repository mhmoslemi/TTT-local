import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

    def constraints_func(c):
        # Constraint 1: All circles are inside the unit square
        # x - r >= 0, 1 - x - r >= 0, same for y
        r = c[2:]
        x = c[::2]
        y = c[1::2]
        inside_sq = np.all((x - r >= 0) & (1 - x - r >= 0) & (y - r >= 0) & (1 - y - r >= 0))
        if not inside_sq:
            return np.inf  # Penalize invalid configurations

        # Constraint 2: No two circles overlap
        # Extract centers and radii
        cx = c[::2]
        cy = c[1::2]
        r = c[2:]
        dists = pdist(np.column_stack((cx, cy)))
        dists = squareform(dists)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < r[i] + r[j]:
                    return np.inf  # Penalize overlapping circles

        return 0.0

    # Define bounds: centers between 0 and 1, radii >= 0
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 0.5))  # radius (max radius is 0.5)
    
    # Initial guess: flatten centers and radii
    initial_guess = np.array([centers[i, 0] for i in range(10)] + [centers[i, 1] for i in range(10)] + list(radii))

    # Set up optimization
    result = minimize(
        fun=objective,
        x0=initial_guess,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints_func},
        method="SLSQP",
        options={"ftol": 1e-7, "maxiter": 200}
    )

    # Extract final centers, radii, and sum
    final_centers = np.array([result.x[::2], result.x[1::2]]).T
    final_radii = result.x[2:]
    sum_radii = -result.fun

    # Ensure constraints are satisfied (for validation)
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        valid = constraints_func(result.x) == 0

    return (final_centers, final_radii, sum_radii)
