import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[i / 3 + 0.1, j / 3 + 0.1] for i in range(3) for j in range(3)])
    radii = np.full(9, 0.1)
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = np.append(radii, 0.1)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        n = len(radii)
        cons = []
        # Constraint: each circle is inside the square
        for i in range(n):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - 0.0})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1.0 - r[i] - centers[i, 0]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1.0 - r[i] - centers[i, 1]})
        
        # Constraint: no two circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                def dist_con(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx * dx + dy * dy)
                    return dist - (radii[i] + radii[j])
                cons.append({'type': 'ineq', 'fun': dist_con})
        
        return cons
    
    # Optimize
    result = minimize(objective, radii, method='SLSQP', constraints=constraints(radii))
    
    # Extract results
    centers_result = centers
    radii_result = result.x
    sum_radii_result = np.sum(radii_result)
    
    return (centers_result, radii_result, sum_radii_result)
