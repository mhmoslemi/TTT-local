import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: place 10 circles evenly spaced
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i + 0.5) / 10
        centers[i, 1] = 0.5
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: all circles fit inside the unit square
        for i in range(10):
            if radii[i] > 1 - 2 * centers[i, 0] or radii[i] > 1 - 2 * centers[i, 1]:
                return False
        # Constraint 2: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimize
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1 - 2 * centers[i, 0]) if i < 10 else (0, 1 - 2 * centers[i, 1]) for i in range(10)],
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r)}]
    )

    # Extract solution
    optimal_radii = result.x
    sum_radii = np.sum(optimal_radii)
    optimal_centers = centers.copy()

    return (optimal_centers, optimal_radii, sum_radii)
