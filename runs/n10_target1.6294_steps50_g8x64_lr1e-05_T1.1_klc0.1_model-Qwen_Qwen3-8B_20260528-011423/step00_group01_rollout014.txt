import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: spread circles evenly
    centers = np.array([(i * 0.2 + 0.1, j * 0.2 + 0.1) for i, j in itertools.product(range(10), repeat=2)])
    radii = np.full(10, 0.1)

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[10:])

    # Constraints: 1. Each circle is inside the square
    #              2. No two circles overlap
    def constraints(x):
        centers = x[:10*2].reshape(10, 2)
        radii = x[10:]
        # Check if all circles are inside the square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Check if all circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Bounds: centers in [0, 1], radii >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0)] * 2)
    bounds.extend([(0.0, 1.0)] * 10)

    # Optimization
    result = minimize(objective, np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints=constraints)

    centers = result.x[:10*2].reshape(10, 2)
    radii = result.x[10:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
