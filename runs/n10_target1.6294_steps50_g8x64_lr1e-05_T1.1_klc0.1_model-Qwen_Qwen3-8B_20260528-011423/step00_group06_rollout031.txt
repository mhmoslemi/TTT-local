import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers (randomly placed within the unit square)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1  # Initial radii

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii, centers):
        # Constraint: all circles are within the unit square
        for i in range(10):
            if radii[i] > 1 - 2 * 1e-6:
                return False
        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx * dx + dy * dy)
                if distance < radii[i] + radii[j] - 1e-6:
                    return False
        return True

    def objective_func(x):
        radii = x[:10]
        centers = x[10:]
        # Validate constraints
        if not constraints(radii, centers.reshape(10, 2)):
            return np.inf
        return -np.sum(radii)

    # Flatten centers and radii for optimization
    initial_guess = np.concatenate((radii, centers.flatten()))

    # Bounds: radii >= 0, centers in [0, 1]
    bounds = [(0, 1) for _ in range(10)] + [(0, 1) for _ in range(20)]

    # Optimization
    result = minimize(objective_func, initial_guess, method='L-BFGS-B', bounds=bounds)

    # Extract result
    final_radii = result.x[:10]
    final_centers = result.x[10:].reshape(10, 2)

    # Calculate sum of radii
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
