import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints
    def constraint1(radii):
        # All circles are inside the unit square
        return np.minimum(radii, np.array([1 - centers[i, 0] for i in range(10)])) - radii

    def constraint2(radii):
        # No two circles overlap
        distance_matrix = cdist(centers, centers)
        sum_radii_matrix = radii[:, np.newaxis] + radii[np.newaxis, :]
        return distance_matrix - sum_radii_matrix

    # Bounds on radii
    bounds = [(0, 1 - np.min(np.array([centers[i, 0], centers[i, 1]])) - 1e-6) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        x0=np.ones(10),
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': constraint1},
            {'type': 'ineq', 'fun': constraint2}
        ],
        bounds=bounds
    )

    radii = result.x
    sum_radii = np.sum(radii)
    centers = np.array(centers)
    return (centers, radii, sum_radii)
