import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii, centers):
        # Ensure all circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Ensure circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    # Flatten parameters for optimization
    initial_params = np.concatenate([centers.ravel(), radii])

    # Define objective and constraints for scipy.optimize.minimize
    def objective_func(params):
        centers = params[:60].reshape(10, 2)
        radii = params[60:]
        return -np.sum(radii)

    def constraint_func(params):
        centers = params[:60].reshape(10, 2)
        radii = params[60:]
        for i in range(10):
            if radii[i] < 0:
                return -np.inf
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return -np.inf
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return -np.inf
        return np.sum(radii)

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective_func, initial_params, method='SLSQP',
                          bounds=[(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)],
                          constraints={'type': 'ineq', 'fun': constraint_func})

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
