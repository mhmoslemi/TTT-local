import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1 + 0.05

    def objective(x):
        # Flatten input for optimization
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        constraints = []

        # 1. All circles are inside the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf  # Violation

        # 2. No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf  # Violation

        return 0  # No constraints violated

    # Set bounds for centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1) for _ in range(2)])  # x and y for each center
        bounds.extend([(0, 0.5) for _ in range(10)])  # radii cannot exceed 0.5

    # Initial guess for the optimization
    initial_guess = np.r_[centers.flatten(), radii]

    # Optimization using L-BFGS-B
    try:
        result = minimize(objective, initial_guess, bounds=bounds, constraints={"type": "ineq", "fun": constraints}, method='L-BFGS-B')
    except:
        # Fallback if scipy is not available or optimization fails
        return (np.random.rand(10, 2), np.random.rand(10) * 0.1 + 0.05, np.sum(np.random.rand(10) * 0.1 + 0.05))

    # Extract final result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
