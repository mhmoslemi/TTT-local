import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Small initial radii

    # Objective function to minimize: negative sum of radii (maximizing sum)
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraint 1: Circle is fully inside the unit square
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        return np.array([r - (x - r) for x in centers_flat for r in radii_flat if x[0] - r < 0 or x[1] - r < 0 or x[0] + r > 1 or x[1] + r > 1])

    # Constraint 2: No overlap between circles
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        dists = pdist(centers_flat)
        sum_radii = np.sum(radii_flat)
        return dists - sum_radii

    # Combine constraints
    cons = []
    cons.append({'type': 'ineq', 'fun': constraint_inside})
    cons.append({'type': 'ineq', 'fun': constraint_overlap})

    # Bounds for circle centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    bounds = bounds[:60] + [(0.0, 0.5) for _ in range(10)]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, np.concatenate([centers.flatten(), radii]), method='SLSQP', bounds=bounds, constraints=cons, tol=1e-5)

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
