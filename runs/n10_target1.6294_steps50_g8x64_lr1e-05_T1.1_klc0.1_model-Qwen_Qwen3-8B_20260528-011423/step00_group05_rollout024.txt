import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles spaced in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(params):
        # Unpack parameters into centers and radii
        flat_centers = params[:10*2].reshape(10, 2)
        flat_radii = params[10*2:].reshape(10)
        # Objective: minimize negative sum of radii (equivalent to maximizing sum)
        sum_radii = np.sum(flat_radii)
        # Constraints: circles must be within the square
        constraints = []
        for i in range(10):
            if flat_radii[i] < 0:
                constraints.append(-flat_radii[i])
        # Constraints: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = flat_centers[i, 0] - flat_centers[j, 0]
                dy = flat_centers[i, 1] - flat_centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < flat_radii[i] + flat_radii[j]:
                    constraints.append(flat_radii[i] + flat_radii[j] - dist)
        return -sum_radii

    def constraint_bounds(params):
        # Bounds for the parameters: centers are [0,1] and radii are [0, 0.5]
        bounds = []
        for i in range(10):
            for j in range(2):
                bounds.append((0.0, 1.0))
        for i in range(10):
            bounds.append((0.0, 0.5))
        return bounds

    # Optimization setup
    initial_params = np.concatenate([centers.reshape(10*2), radii])
    result = minimize(objective, initial_params, method='L-BFGS-B', bounds=constraint_bounds(initial_params), tol=1e-8)

    # Extract results
    flat_centers = result.x[:10*2].reshape(10, 2)
    flat_radii = result.x[10*2:].reshape(10)
    sum_radii = np.sum(flat_radii)

    return (flat_centers, flat_radii, sum_radii)
