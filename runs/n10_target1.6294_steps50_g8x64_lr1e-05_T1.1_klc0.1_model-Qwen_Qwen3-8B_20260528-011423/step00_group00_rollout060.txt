from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():

    centers = np.random.rand(10, 2)

        return True

    bounds = [(0, 0.5) for _ in range(10)]

    # Define the optimization problem
    def objective_func(x):
        radii = x[:10]
        centers = x[10:]
        centers = centers.reshape((10, 2))
        return -np.sum(radii)

    def constraint_func(x):
        radii = x[:10]
        centers = x[10:].reshape((10, 2))
        # Check if all circles are within the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):

        dists = cdist(centers, centers)

    result = minimize(objective_func, x0, bounds=bounds, constraints={"type": "ineq", "fun": constraint_func})
