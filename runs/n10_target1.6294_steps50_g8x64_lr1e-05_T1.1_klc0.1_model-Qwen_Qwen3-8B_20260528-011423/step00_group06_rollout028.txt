import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: random positions and equal radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective_function(x):
        # Unpack variables
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]

        # Constraint 1: All circles are within the unit square
        # r <= x <= 1 - r and r <= y <= 1 - r
        # => r + r <= 1 => r <= 0.5
        # We'll assume radii are already bounded between 0 and 0.5
        # So we just need to ensure the centers are within [r, 1-r]

        # Constraint 2: No two circles overlap
        # distance between centers >= sum of radii
        # We minimize the negative of the sum of radii
        # Objective: maximize sum of radii => minimize -sum of radii

        # Check for overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii_flat[i] + radii_flat[j]:
                    # Penalize overlap
                    return 1e6  # Large penalty
        return -np.sum(radii_flat)

    def constraints(x):
        # Constraint 1: All circles are within the unit square
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            if centers_flat[i][0] < radii_flat[i] or centers_flat[i][0] > 1 - radii_flat[i] or \
               centers_flat[i][1] < radii_flat[i] or centers_flat[i][1] > 1 - radii_flat[i]:
                return np.inf  # Large penalty for violation
        return 0

    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Bounds: centers between 0 and 1, radii between 0 and 0.5
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])
    for _ in range(10):
        bounds.extend([(0, 0.5)])

    # Optimization
    result = minimize(objective_function, initial_guess, bounds=bounds, method='L-BFGS-B', constraints={"type": "ineq", "fun": constraints})

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun
    return (centers_opt, radii_opt, sum_radii)
