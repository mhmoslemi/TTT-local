import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles arranged in a square grid with equal radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        # Objective: maximize the sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: Each circle is within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            # Maximum radius for circle at (x, y) is such that x - r >= 0 and 1 - x - r >= 0
            # Similarly for y direction
            max_r = np.minimum(centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1])
            if radii[i] > max_r:
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Prepare bounds for radii (non-negative and not exceeding feasible maximum)
    bounds = [(0, 1) for _ in range(10)]

    # Optimization: find the radii that maximize the sum while satisfying constraints
    def constraint_func(x):
        # x is the vector of radii
        if not constraints(x):
            return np.inf
        return 0

    result = minimize(objective, radii, bounds=bounds, constraints={"type": "ineq", "fun": constraint_func})

    # Extract final centers, radii, and sum
    centers_final = centers
    radii_final = result.x
    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
