import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: spread out points in the unit square
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Small initial radii

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii, centers):
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    def bounds(radii):
        return [(0, 1) for _ in range(10)]

    def bounds_centers():
        return [(0, 1) for _ in range(10 * 2)]

    # Define optimization problem
    bounds = bounds(radii)
    cons = []
    for i in range(10):
        for j in range(i + 1, 10):
            def dist_constraint(radii_all):
                r_i = radii_all[i]
                r_j = radii_all[j]
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                return np.sqrt(dx**2 + dy**2) - (r_i + r_j)
            cons.append({'type': 'ineq', 'fun': dist_constraint})

    for i in range(10):
        def x_constraint(radii_all):
            r = radii_all[i]
            return centers[i, 0] - r
        cons.append({'type': 'ineq', 'fun': x_constraint})

        def y_constraint(radii_all):
            r = radii_all[i]
            return centers[i, 1] - r
        cons.append({'type': 'ineq', 'fun': y_constraint})

        def x_upper_constraint(radii_all):
            r = radii_all[i]
            return 1 - r - centers[i, 0]
        cons.append({'type': 'ineq', 'fun': x_upper_constraint})

        def y_upper_constraint(radii_all):
            r = radii_all[i]
            return 1 - r - centers[i, 1]
        cons.append({'type': 'ineq', 'fun': y_upper_constraint})

    result = minimize(lambda x: -np.sum(x), radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract optimized values
    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
