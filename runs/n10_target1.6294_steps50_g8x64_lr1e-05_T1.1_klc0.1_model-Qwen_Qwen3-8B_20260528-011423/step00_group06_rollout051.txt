import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import itertools

def run_packing():
    # Initial guess for centers (spread out around the square)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)
    
    # Objective function: minimize the negative sum of radii
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii
    
    # Constraints
    def constraint_centers(x):
        centers_flat = x[:60].reshape(10, 2)
        return centers_flat.flatten()
    
    def constraint_radii(x):
        radii_flat = x[60:]
        return radii_flat
    
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        inside = np.all((centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1) & \
                 np.all((1 - centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1)
        return np.sum(inside) == 10
    
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        dists = pdist(centers_flat)
        dists = squareform(dists)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii_flat[i] + radii_flat[j]:
                    return -1
        return 0
    
    # Bounds for x and y coordinates of centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x and y coordinates for each center
    for _ in range(10):
        bounds.extend([(0, 0.5)])       # radius for each circle
    
    # Setup optimization
    x0 = np.concatenate((centers.flatten(), radii))
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-8)
    
    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun
    
    return (centers_opt, radii_opt, sum_radii)
