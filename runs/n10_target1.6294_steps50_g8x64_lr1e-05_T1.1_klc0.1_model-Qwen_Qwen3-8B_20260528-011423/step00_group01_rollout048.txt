import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective_function(positions):
        # positions is a 1D array of shape (20,)
        # reshape into 10 centers, each with 2 coordinates
        centers = positions.reshape((10, 2))
        # calculate the sum of radii (initial guess)
        # we can use the same positions as radii for initial guess
        radii = positions[:10]
        # enforce non-negative radii
        radii = np.clip(radii, 0, None)
        # check if all circles fit inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf  # invalid configuration
        # calculate total sum of radii
        sum_radii = np.sum(radii)
        # calculate the minimum distance between centers
        min_dist = np.inf
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf  # circles overlap
                if dist < min_dist:
                    min_dist = dist
        # return -sum_radii to maximize the sum (minimize -sum_radii)
        return -sum_radii

    # Initial guess: place circles at grid points with small radii
    # Create a 3x4 grid, then shift slightly to avoid overlap
    initial_positions = []
    for i in range(3):
        for j in range(4):
            x = 0.2 + i * 0.4
            y = 0.2 + j * 0.4
            initial_positions.append(x)
            initial_positions.append(y)
    initial_positions += [0.05] * 10  # initial radii
    initial_positions = np.array(initial_positions)

    # Optimize
    result = minimize(
        fun=objective_function,
        x0=initial_positions,
        method='L-BFGS-B',
        bounds=[(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    )

    # Extract results
    positions = result.x
    centers = positions[:20].reshape((10, 2))
    radii = np.clip(positions[20:], 0, None)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
