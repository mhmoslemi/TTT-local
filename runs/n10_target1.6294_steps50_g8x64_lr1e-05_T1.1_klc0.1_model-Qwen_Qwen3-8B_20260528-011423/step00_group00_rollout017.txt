import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: randomly placed circles with small radii
    np.random.seed(42)
    centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    radii = np.random.uniform(low=0.05, high=0.1, size=10)

    # Objective function: minimize the negative sum of radii
    def objective(x):
        # Unpack into centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraint 1: circles are inside the unit square
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        inside = np.all((radii_flat <= centers_flat[:, 0]) & (centers_flat[:, 0] <= 1 - radii_flat) &
                        (radii_flat <= centers_flat[:, 1]) & (centers_flat[:, 1] <= 1 - radii_flat))
        return inside

    # Constraint 2: circles are non-overlapping
    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distances = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :])**2).sum(axis=2))
        overlaps = np.any(distances < radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])
        return not overlaps

    # Optimization bounds
    bounds = []
    for _ in range(10):
        bounds.extend([(0.1, 0.9), (0.1, 0.9)])  # x and y for each center
    bounds.extend([(0.05, 0.15)] * 10)  # radii

    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Set up the optimization problem
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0=np.concatenate([centers.flatten(), radii]),
            method='L-BFGS-B',
            bounds=bounds,
            constraints=cons,
            tol=1e-6,
            options={'maxiter': 1000}
        )

    # Extract optimal solution
    optimal_centers = result.x[:60].reshape(10, 2)
    optimal_radii = result.x[60:]
    sum_radii = -result.fun

    return (optimal_centers, optimal_radii, sum_radii)
