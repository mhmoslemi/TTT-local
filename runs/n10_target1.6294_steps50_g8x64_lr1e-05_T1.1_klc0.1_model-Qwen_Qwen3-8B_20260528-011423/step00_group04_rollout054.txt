import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i * 0.25, j * 0.25] for i in range(10) for j in range(10)][:10])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            x, y = centers[i]
            x = max(0.0, min(1.0 - radii[i], x))
            y = max(0.0, min(1.0 - radii[i], y))
            centers[i] = [x, y]
        # Boundary constraints
        for i in range(10):
            if centers[i][0] < radii[i] or centers[i][0] > 1.0 - radii[i]:
                return np.inf
            if centers[i][1] < radii[i] or centers[i][1] > 1.0 - radii[i]:
                return np.inf
        # Overlap constraints
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Optimize
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message=".*does not have attribute.*")
        result = minimize(
            fun=constraints,
            x0=radii,
            method='L-BFGS-B',
            bounds=[(0.0, 1.0) for _ in range(10)]
        )

    # Extract results
    final_radii = result.x
    centers = np.zeros((10, 2))
    for i in range(10):
        x = max(0.0, min(1.0 - final_radii[i], np.random.uniform(0.0, 1.0)))
        y = max(0.0, min(1.0 - final_radii[i], np.random.uniform(0.0, 1.0)))
        centers[i] = [x, y]
    sum_radii = np.sum(final_radii)
    return (centers, final_radii, sum_radii)
