import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import squareform, pdist

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.08)

    def objective(radii):
        return -np.sum(radii)

    def constraint_box(radii):
        # Ensure centers are within the unit square
        # Note: centers are global, not optimized here, so need to check validity
        return np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]) &
                      (radii >= 0) & (centers[:, 0] >= radii) & (centers[:, 1] >= radii))

    def constraint_overlap(radii):
        # Calculate pairwise distances between centers
        dists = squareform(pdist(centers))
        # Ensure distance between centers is at least sum of radii
        return np.min(dists - (radii[:, np.newaxis] + radii[np.newaxis, :])) >= 0

    # Create optimization problem
    bounds = [(0, 1 - 0.1) for _ in range(10)]  # radii must be at least 0.1 to fit in unit square

    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_box},
            {'type': 'ineq', 'fun': constraint_overlap}
        ]
    )

    radii_opt = result.x
    sum_radii_opt = -result.fun

    return (centers, radii_opt, sum_radii_opt)
