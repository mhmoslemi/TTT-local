import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective_function(params):
        # Extract centers and radii from parameters
        centers = params[:10*2].reshape(10, 2)
        radii = params[10*2:]

        # Objective function to minimize: negative of sum of radii
        sum_radii = np.sum(radii)
        return -sum_radii

    def constraints(params):
        # Extract centers and radii from parameters
        centers = params[:10*2].reshape(10, 2)
        radii = params[10*2:]

        # Constraint 1: All circles must be inside the unit square
        # For each circle: r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i
        # These are implicit in the objective function, but we add them as constraints

        # Constraint 2: No two circles can overlap
        # Distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -1.0
        return 0.0

    # Initial parameters
    initial_params = np.concatenate([centers.flatten(), radii])

    # Optimization settings
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1) for _ in range(2)])  # x, y for each center
        bounds.extend([(0, 0.5) for _ in range(10)])  # radii must be <= 0.5

    # Set up constraints
    def constraint_func(params):
        return constraints(params)

    cons = {'type': 'ineq', 'fun': constraint_func}

    # Perform optimization
    result = minimize(objective_function, initial_params, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract optimal centers and radii
    centers_opt = result.x[:10*2].reshape(10, 2)
    radii_opt = result.x[10*2:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
