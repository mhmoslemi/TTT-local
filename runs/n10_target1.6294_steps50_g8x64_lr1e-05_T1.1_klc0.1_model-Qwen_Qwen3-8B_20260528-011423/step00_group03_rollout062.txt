import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced centers in a grid
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    centers = np.vstack((centers, [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5]))
    radii = np.full(10, 0.1)

    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        radii = x[:10]
        centers = x[10:]
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints: 1. Circles are inside the square
    #               2. No two circles overlap
    def constraint_inside(radii, centers):
        return np.all(radii <= 1 - centers[:, 0]) and np.all(radii <= 1 - centers[:, 1])

    def constraint_no_overlap(centers, radii):
        dists = np.sqrt(((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2).sum(axis=2))
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization setup
    bounds = [(0, 1) for _ in range(10)] + [(0, 1) for _ in range(20)]
    constraints = []

    # Constraint: circles are inside the square
    constraints.append({
        'type': 'ineq',
        'fun': lambda x: np.min([x[:10] - (1 - x[10:]), 1 - x[10:]]).min()
    })

    # Constraint: no two circles overlap
    constraints.append({
        'type': 'ineq',
        'fun': lambda x: np.min(np.sqrt(((x[10:].reshape(10, 2)[:, np.newaxis] - x[10:].reshape(10, 2)[np.newaxis, :]) ** 2).sum(axis=2)) - x[:10][:, np.newaxis] - x[:10][np.newaxis, :])
    })

    result = minimize(objective, np.concatenate((radii, centers.ravel())), bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-7)

    if not result.success:
        print("Optimization failed")
        return np.zeros((10, 2)), np.zeros(10), 0.0

    optimal_radii = result.x[:10]
    optimal_centers = result.x[10:].reshape(10, 2)

    sum_radii = np.sum(optimal_radii)

    return optimal_centers, optimal_radii, sum_radii
