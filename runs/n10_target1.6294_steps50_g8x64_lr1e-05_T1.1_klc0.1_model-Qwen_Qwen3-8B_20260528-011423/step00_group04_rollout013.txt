import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.ones(10) * 0.05

    # Objective function to minimize (negative of sum of radii)
    def objective(x):
        radii = x[:10]
        centers = x[10:]
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints: 
    def constraint_positions(x):
        radii = x[:10]
        centers = x[10:].reshape(10, 2)
        return np.array([r - (center[0] - 0.0) for r, center in zip(radii, centers)])  # x >= r
    def constraint_distances(x):
        radii = x[:10]
        centers = x[10:].reshape(10, 2)
        dists = np.zeros(45)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                dists[idx] = dist - (radii[i] + radii[j])
                idx += 1
        return dists

    # Bounds for radii and positions
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # radius for each circle
        bounds.append((0, 1))  # x position
        bounds.append((0, 1))  # y position

    # Constraints for non-overlapping and inside the square
    constraints = [
        {'type': 'ineq', 'fun': constraint_positions},
        {'type': 'ineq', 'fun': constraint_distances}
    ]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0=np.hstack((radii, centers.flatten())), 
                          method='SLSQP', bounds=bounds, constraints=constraints)

    centers_opt = result.x[10:].reshape(10, 2)
    radii_opt = result.x[:10]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
