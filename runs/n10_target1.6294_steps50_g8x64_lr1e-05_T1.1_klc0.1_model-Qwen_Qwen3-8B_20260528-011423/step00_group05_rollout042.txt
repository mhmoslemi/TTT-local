import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) * 0.4
        centers[i, 1] = (i // 5) * 0.4

    # Initial radii: all equal
    radii = np.full(10, 0.1)

    # Objective function: minimize negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    constraints = []

    # Constraint: each circle must be inside the unit square
    def square_constraint(radii):
        return np.sum((radii <= np.min([centers[:, 0], 1 - centers[:, 0]])) &
                     (radii <= np.min([centers[:, 1], 1 - centers[:, 1]])))
    constraints.append({'type': 'ineq', 'fun': square_constraint})

    # Constraint: no two circles overlap
    def no_overlap(radii):
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])
    constraints.append({'type': 'ineq', 'fun': no_overlap})

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract results
    centers = np.array(result.x).reshape(-1, 2)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
