import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[i / 9 + 0.05, j / 9 + 0.05] for j in range(10) for i in range(10)])
    radii = np.full(10, 0.05)

    def objective(r):
        return -np.sum(r)

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Constraint 1: Circles must be inside the unit square
        for i in range(10):
            if radii_flat[i] > 1 - 2 * centers_flat[i, 0] or radii_flat[i] > 1 - 2 * centers_flat[i, 1]:
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0

    # Define the optimization variables
    # First 60 variables: centers (10 circles, 2 coordinates each)
    # Next 10 variables: radii (1 for each circle)
    x0 = np.concatenate([centers.flatten(), radii])

    # Setup optimization problem
    res = minimize(
        objective,
        x0,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        bounds=[(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]
    )

    # Extract solution
    x_opt = res.x
    centers_opt = x_opt[:60].reshape(10, 2)
    radii_opt = x_opt[60:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
