import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 10 circles with small radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)
    
    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array of shape (20,)
        centers_reshaped = x[:20].reshape(10, 2)
        radii_reshaped = x[20:]
        return -np.sum(radii_reshaped)  # Minimize negative sum to maximize sum
    
    # Constraint: circles must be fully inside the unit square
    def constraint_inside(x):
        centers_reshaped = x[:20].reshape(10, 2)
        radii_reshaped = x[20:]
        for i in range(10):
            x_i, y_i = centers_reshaped[i]
            r_i = radii_reshaped[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1.0  # Violation
        return 0.0
    
    # Constraint: circles must not overlap
    def constraint_overlap(x):
        centers_reshaped = x[:20].reshape(10, 2)
        radii_reshaped = x[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_reshaped[i, 0] - centers_reshaped[j, 0]
                dy = centers_reshaped[i, 1] - centers_reshaped[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_reshaped[i] + radii_reshaped[j]:
                    return -1.0  # Violation
        return 0.0
    
    # Constraint: radii must be non-negative
    def constraint_non_negative(x):
        radii_reshaped = x[20:]
        for r in radii_reshaped:
            if r < 0:
                return -1.0  # Violation
        return 0.0
    
    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]
    
    # Initial guess flattened
    x0 = np.concatenate((centers.flatten(), radii))
    
    # Optimization
    result = minimize(objective, x0, constraints=cons, method='SLSQP', tol=1e-6)
    
    # Extract results
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = -result.fun  # Since we maximized sum by minimizing negative sum
    
    return (centers_opt, radii_opt, sum_radii)
