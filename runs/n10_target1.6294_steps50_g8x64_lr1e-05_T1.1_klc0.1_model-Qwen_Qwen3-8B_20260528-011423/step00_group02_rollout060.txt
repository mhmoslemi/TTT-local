import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    def objective(x):
        return -np.sum(radii)

    def constraints(x):
        # Extract centers and radii from x
        centers = x[:10*2].reshape(10, 2)
        radii = x[10*2:]
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            cx, cy = centers[i]
            r = radii[i]
            if not (r <= cx <= 1 - r and r <= cy <= 1 - r):
                return np.inf
        # Constraint 2: No two circles overlap
        dists = pdist(centers)
        for d, (i, j) in zip(dists, [(a, b) for a in range(10) for b in range(a+1, 10)]):
            ri = radii[i]
            rj = radii[j]
            if d < ri + rj - 1e-8:
                return np.inf
        return 0

    # Initial guess: randomly distribute centers and set radii to small values
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.zeros(10)
    for i in range(10):
        cx, cy = centers[i]
        r = min(cx, 1 - cx, cy, 1 - cy) * 0.5
        radii[i] = r
    x0 = np.concatenate((centers.ravel(), radii))

    # Optimization
    res = minimize(objective, x0, method='L-BFGS-B', bounds=[(0, 1) for _ in range(10*2)] + [(0, 1) for _ in range(10)], constraints={"type": "ineq", "fun": constraints})

    # Extract results
    centers = res.x[:10*2].reshape(10, 2)
    radii = res.x[10*2:]
    sum_radii = -res.fun

    return (centers, radii, sum_radii)
