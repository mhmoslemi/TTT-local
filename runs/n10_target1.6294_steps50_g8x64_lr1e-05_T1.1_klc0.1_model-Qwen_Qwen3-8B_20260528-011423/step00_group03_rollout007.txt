import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii (random positions and equal radii)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1 + 0.05  # Small initial radii

    def objective_function(params):
        # Flatten parameters for optimization
        params = params.reshape(-1)
        centers = params[:20].reshape(10, 2)
        radii = params[20:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraint_circle_inside(params):
        params = params.reshape(-1)
        centers = params[:20].reshape(10, 2)
        radii = params[20:]
        # Check if all circles are inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -1.0
        return 0.0

    def constraint_no_overlap(params):
        params = params.reshape(-1)
        centers = params[:20].reshape(10, 2)
        radii = params[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return -1.0
        return 0.0

    # Initial parameters for optimization
    initial_params = np.concatenate([centers.flatten(), radii])

    # Set up optimization problem
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        bounds.append((0.0, 0.5))  # radius (max radius is 0.5)
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    result = minimize(
        objective_function,
        initial_params,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract final values
    final_params = result.x
    centers = final_params[:20].reshape(10, 2)
    radii = final_params[20:]
    sum_radii = -result.fun  # Since we maximized the sum

    return (centers, radii, sum_radii)
