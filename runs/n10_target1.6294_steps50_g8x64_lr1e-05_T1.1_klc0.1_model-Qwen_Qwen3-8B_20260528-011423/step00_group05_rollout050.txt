import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equal radii and centers on a grid
    def initial_guess():
        centers = np.zeros((10, 2))
        radii = np.zeros(10)
        for i in range(10):
            x = i % 5 / 4
            y = i // 5 / 4
            centers[i] = [x, y]
            radii[i] = 0.1
        return centers, radii

    centers, radii = initial_guess()
    bounds = [(0, 1) for _ in range(20)]  # 10 centers * 2 coordinates

    # Objective function to minimize: negative sum of radii
    def objective(x):
        centers = x.reshape(10, 2)
        radii = x[20:].reshape(10)
        return -np.sum(radii)

    # Constraint: each circle is inside the unit square
    def square_inside(x):
        centers = x.reshape(10, 2)
        radii = x[20:].reshape(10)
        return np.array([r <= x - 0.0001 and r <= 1 - x - 0.0001 for x, r in zip(centers, radii)])

    # Constraint: no two circles overlap
    def no_overlap(x):
        centers = x.reshape(10, 2)
        radii = x[20:].reshape(10)
        dists = np.sqrt(((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Setup constraints
    cons = [
        {'type': 'ineq', 'fun': square_inside},
        {'type': 'ineq', 'fun': no_overlap}
    ]

    # Optimization
    try:
        result = minimize(objective, np.concatenate((centers.flatten(), radii)), bounds=bounds, constraints=cons)
    except:
        result = minimize(objective, np.concatenate((centers.flatten(), radii)), bounds=bounds)

    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:].reshape(10)
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
