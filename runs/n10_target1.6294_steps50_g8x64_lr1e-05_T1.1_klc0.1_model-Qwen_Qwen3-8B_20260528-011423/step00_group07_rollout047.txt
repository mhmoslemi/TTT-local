import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guesses: random positions with small radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    # Objective function: minimize the negative sum of radii (equivalent to maximizing sum of radii)
    def objective(x):
        # Reshape x to centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize the negative sum

    # Constraints: 1. All circles within the square, 2. No overlap between circles
    def square_constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        constraints = []
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append((x_i - r_i >= 0) * 1e6)
            constraints.append((x_i + r_i <= 1) * 1e6)
            constraints.append((y_i - r_i >= 0) * 1e6)
            constraints.append((y_i + r_i <= 1) * 1e6)
        return np.array(constraints)

    def circle_overlap_constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                constraints.append((dist - r_i - r_j >= 0) * 1e6)
        return np.array(constraints)

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii.flatten()])

    # Bounds: centers [0, 1], radii [0, 0.5]
    bounds = [(0, 1) for _ in range(60)] + [(0, 0.5) for _ in range(10)]

    # Set up constraints
    constraints = [
        {'type': 'ineq', 'fun': square_constraints},
        {'type': 'ineq', 'fun': circle_overlap_constraints}
    ]

    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract result
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:].reshape(10)
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
