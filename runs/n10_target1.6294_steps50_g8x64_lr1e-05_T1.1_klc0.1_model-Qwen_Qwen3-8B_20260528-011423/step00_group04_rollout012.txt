import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equal radii and evenly spaced centers
    initial_r = 0.1
    centers = np.zeros((10, 2))
    for i in range(10):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]
    radii = np.full(10, initial_r)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraint_circle_inside(radii, centers):
        # Check if all circles are inside the unit square
        for i in range(10):
            r = radii[i]
            x, y = centers[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -np.inf
        return 0

    def constraint_no_overlap(radii, centers):
        # Check that no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    # Define bounds for radii: they must be non-negative
    bounds = [(0.0, 0.5) for _ in range(10)]

    # Define constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (centers,)}
    ]

    # Minimize the negative sum of radii (maximize sum)
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Extract the optimized result
    optimized_radii = result.x
    optimized_centers = centers.copy()  # centers are not optimized in this approach

    # Calculate the sum of radii
    total_sum = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, total_sum)
