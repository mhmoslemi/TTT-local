import numpy as np
from scipy.optimize import minimize
import sys

def objective_function(positions):
    radii = np.array([0.1] * 10)  # Initial guess for radii
    # Initial guess for positions, spread out in the unit square
    x = positions[:20].reshape(10, 2)
    # Compute the sum of the radii
    sum_radii = np.sum(radii)
    # Constraint: each circle must be fully inside the unit square
    constraints = []
    for i in range(10):
        x_i, y_i = x[i]
        constraint = {
            'type': 'ineq',
            'fun': lambda x, i=i: x[i] - radii[i],
            'args': (i,)
        }
        constraints.append(constraint)
        constraint = {
            'type': 'ineq',
            'fun': lambda x, i=i: 1 - x[i] - radii[i],
            'args': (i,)
        }
        constraints.append(constraint)
    # Constraint: no two circles overlap
    for i in range(10):
        for j in range(i + 1, 10):
            def dist_constraint(x):
                dx = x[2*i] - x[2*j]
                dy = x[2*i + 1] - x[2*j + 1]
                dist = np.sqrt(dx**2 + dy**2)
                return dist - (radii[i] + radii[j])
            constraint = {
                'type': 'ineq',
                'fun': dist_constraint,
                'args': (i, j)
            }
            constraints.append(constraint)
    # Objective function is the negative of the sum of radii to maximize
    def obj(x):
        return -np.sum(radii)
    # Combine the objective and constraints into a single optimization problem
    result = minimize(obj, positions, method='SLSQP', constraints=constraints)
    # Extract the optimal positions
    optimal_positions = result.x[:20].reshape(10, 2)
    # Assign radii based on the distance from the edges
    optimal_radii = []
    for i in range(10):
        x_i, y_i = optimal_positions[i]
        r = min(x_i, 1 - x_i, y_i, 1 - y_i)
        optimal_radii.append(r)
    sum_radii = np.sum(optimal_radii)
    return (optimal_positions, np.array(optimal_radii), sum_radii)

def run_packing():
    # Initial guess for positions (spread out in the unit square)
    initial_positions = np.random.rand(20)
    centers, radii, sum_radii = objective_function(initial_positions)
    return (centers, radii, sum_radii)
