import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid of 10 points
    centers = np.array([[(i + 0.5) / 3, (j + 0.5) / 3] for i in range(3) for j in range(3)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[(i + 0.5) / 3, (j + 0.5) / 3] for i in range(3) for j in range(3)])
        centers_radii = np.column_stack((centers, radii))
        # Box constraints: circles must be inside the unit square
        box_cons = []
        for i in range(10):
            x, y, r = centers_radii[i]
            box_cons.append((x - r >= 0.0) * 1e6)
            box_cons.append((x + r <= 1.0) * 1e6)
            box_cons.append((y - r >= 0.0) * 1e6)
            box_cons.append((y + r <= 1.0) * 1e6)
        # Pairwise distance constraints: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                x1, y1, r1 = centers_radii[i]
                x2, y2, r2 = centers_radii[j]
                dist = np.sqrt((x1 - x2)**2 + (y1 - y2)**2)
                box_cons.append((dist - r1 - r2) * 1e6)
        return box_cons

    # Optimization problem
    bounds = [(0.0, 0.5) for _ in range(10)]
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    centers = np.array([[(i + 0.5) / 3, (j + 0.5) / 3] for i in range(3) for j in range(3)])
    radii = res.x
    sum_radii = -res.fun

    return centers, radii, sum_radii
