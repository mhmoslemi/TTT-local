import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: 10 circles evenly spaced in a grid
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    centers = np.vstack((centers, [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5]))
    radii = np.full(10, 0.1)
    
    # Objective function: minimize negative sum of radii
    def objective(x):
        # x is a flat array of 20 values (10 centers × 2 coords + 10 radii)
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints: circles must be inside the square and not overlapping
    def constraint_inside(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        return np.sum((centers_flat - radii_flat[:, np.newaxis]) < 0) + \
               np.sum((1 - centers_flat - radii_flat[:, np.newaxis]) < 0)

    def constraint_overlap(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        dist = np.sum((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2, axis=2)
        return np.sum((np.sqrt(dist) - radii_flat[:, np.newaxis] - radii_flat[np.newaxis, :]) < 0)

    # Bounds for each center coordinate and radius
    bounds = []
    for i in range(10):
        bounds.extend([(0, 1) for _ in range(2)])  # x and y coordinates
        bounds.extend([(0, 1)])  # radius

    # Define constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Initial guess flattened
    x0 = np.concatenate((centers.flatten(), radii))

    # Perform optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-8)

    # Extract results
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii_result = -result.fun  # Objective is negative of sum

    # Return as required
    return (centers_result, radii_result, sum_radii_result)
