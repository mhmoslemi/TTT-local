import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.uniform(size=(10, 2))
    radii = np.random.uniform(low=0.05, high=0.1, size=10)
    
    def objective(x):
        # Flatten input for optimization
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Ensure all radii are non-negative
        radii_flat[radii_flat < 0] = 0
        # Objective is the negative of the sum of radii (maximize)
        sum_radii = np.sum(radii_flat)
        # Apply constraints
        constraints = []
        
        # Constraint: all circles are within the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                constraints.append({'type': 'ineq', 'fun': lambda x, i=i: x[60 + i] - 0.001})
        
        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 0.001:
                    constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: x[60 + i] + x[60 + j] - dist})
        
        # If any constraint is violated, return a large penalty
        for c in constraints:
            c['fun'] = lambda x, c=c: c['fun'](x)
        
        return -sum_radii
    
    bounds = [(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]  # centers and radii bounds

    # Disable warnings to avoid optimization issues
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, np.concatenate([centers.flatten(), radii]), bounds=bounds)
    
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
