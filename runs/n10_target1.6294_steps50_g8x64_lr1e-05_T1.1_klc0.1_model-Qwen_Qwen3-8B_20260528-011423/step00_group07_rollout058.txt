import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint: circles must be inside the unit square
        for i in range(10):
            if radii[i] > 0:
                if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                    return False
                if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                    return False
        # Constraint: no overlaps between circles
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i+1, 10):
                if radii[i] + radii[j] > dists[i, j]:
                    return False
        return True

    def bounds(radii):
        return [(0.0, 1.0) for _ in range(10)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'func': constraints}]
    )

    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii = -result.fun

    # Ensure circles are within the square and no overlaps
    for i in range(10):
        if radii_opt[i] > 0:
            if centers_opt[i, 0] - radii_opt[i] < 0 or centers_opt[i, 0] + radii_opt[i] > 1:
                centers_opt[i, 0] = max(0.0 + radii_opt[i], min(1.0 - radii_opt[i], centers_opt[i, 0]))
            if centers_opt[i, 1] - radii_opt[i] < 0 or centers_opt[i, 1] + radii_opt[i] > 1:
                centers_opt[i, 1] = max(0.0 + radii_opt[i], min(1.0 - radii_opt[i], centers_opt[i, 1]))

    return (centers_opt, radii_opt, sum_radii)
