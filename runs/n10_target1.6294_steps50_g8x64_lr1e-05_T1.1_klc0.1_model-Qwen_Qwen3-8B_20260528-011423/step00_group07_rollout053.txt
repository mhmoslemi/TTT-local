import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers (random positions in the square)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial radii

    def objective(x):
        # x is a flat vector containing centers and radii
        # Reshape for easier access
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Constraint 1: All circles are inside the square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                distance = np.sqrt(dx * dx + dy * dy)
                if distance < radii_flat[i] + radii_flat[j]:
                    return False
        return True

    # Pack all parameters into a single vector
    x0 = np.concatenate((centers.reshape(10 * 2), radii))
    # Use scipy.optimize.minimize
    result = minimize(objective, x0, constraints=lambda x: constraints(x), method='SLSQP', options={'ftol': 1e-6})
    optimal_x = result.x
    centers_opt = optimal_x[:60].reshape(10, 2)
    radii_opt = optimal_x[60:]
    sum_radii = -result.fun  # Convert back to maximized sum of radii
    return (centers_opt, radii_opt, sum_radii)
