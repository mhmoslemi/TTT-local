import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.08)

    def objective(radii):
        return -np.sum(radii)

    def constraints(centers, radii):
        # Box constraints: circles are inside the unit square
        box_con = []
        for i in range(10):
            box_con.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - 1e-8})
            box_con.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - 1e-8})
        # Non-overlapping constraints
        for i in range(10):
            for j in range(i + 1, 10):
                def cons(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    return dist - (radii[i] + radii[j])
                box_con.append({'type': 'ineq', 'fun': cons})
        return box_con

    def bounds(radii):
        return [(0, 1) for _ in range(10)]

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="Convergence is not guaranteed")
        result = minimize(
            fun=objective,
            x0=radii,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints(centers, radii),
            tol=1e-6
        )

    # Refine centers slightly to improve packing
    def refine_centers(centers, radii):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    # Move centers slightly away from each other
                    direction = np.array([dx, dy]) / dist
                    centers[i] += 0.01 * direction
                    centers[j] -= 0.01 * direction
        return centers

    centers = refine_centers(centers, result.x)
    return centers, result.x, np.sum(result.x)
