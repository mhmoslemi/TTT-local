import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.125
        centers[i, 1] = (i // 5) / 4.0 + 0.125

    # Initial radii
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: circles are fully inside the unit square
    def constraint_inside(radii):
        r = radii
        return np.array([r[i] <= 1 - centers[i, 0] for i in range(10)] + 
                       [r[i] <= 1 - centers[i, 1] for i in range(10)])

    # Constraint: circles do not overlap
    def constraint_overlap(radii):
        r = radii
        dists = np.sqrt(np.sum((centers[:, np.newaxis] - centers[np.newaxis, :])**2, axis=2))
        return np.array([dists[i, j] >= r[i] + r[j] for i in range(10) for j in range(i+1)])

    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds: radii >= 0
    bounds = [(0, None) for _ in range(10)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Return result
    return (centers, result.x, np.sum(result.x))
