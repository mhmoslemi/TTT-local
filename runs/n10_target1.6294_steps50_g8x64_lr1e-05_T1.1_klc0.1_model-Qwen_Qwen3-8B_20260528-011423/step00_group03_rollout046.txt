import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.1
        centers[i, 1] = (i // 5) / 4.0 + 0.1

    # Initial radii: small value
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x[10:])  # Minimize negative sum (equivalent to maximize sum)

    # Constraints
    def constraint_center_inside(x):
        # x[0:20] are centers (x0, y0, x1, y1, ...)
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return np.array([r <= x_i for x_i, r in zip(centers[:, 0], radii)]) & np.array([r <= y_i for y_i, r in zip(centers[:, 1], radii)]) & np.array([1 - r >= x_i for x_i, r in zip(centers[:, 0], radii)]) & np.array([1 - r >= y_i for y_i, r in zip(centers[:, 1], radii)])

    def constraint_no_overlap(x):
        # x[0:20] are centers (x0, y0, x1, y1, ...)
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        dists = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx * dx + dy * dy) - (radii[i] + radii[j])
                idx += 1
        return dists

    # Set up optimization problem
    x0 = np.concatenate((centers.flatten(), radii))
    cons = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization with bounds: radii >= 0 and centers in [0, 1]
    bounds = [
        (0.0, 1.0) for _ in range(20)
    ] + [(0.0, 1.0) for _ in range(10)]

    # Optimization with L-BFGS-B method
    result = minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract final solution
    centers_final = result.x[:20].reshape(10, 2)
    radii_final = result.x[20:]
    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
