import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[i/3 + 0.1, j/3 + 0.1] for i in range(3) for j in range(3)])
    centers = np.vstack([centers, [0.5, 0.1], [0.5, 0.9], [0.1, 0.5], [0.9, 0.5]])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are fully inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if (centers[i, 0] - radii[i]) < 0 or (centers[i, 0] + radii[i]) > 1:
                return False
            if (centers[i, 1] - radii[i]) < 0 or (centers[i, 1] + radii[i]) > 1:
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    result = minimize(objective, radii, method='SLSQP', constraints=({'type': 'ineq', 'fun': constraints}))

    return (centers, result.x, np.sum(result.x))
