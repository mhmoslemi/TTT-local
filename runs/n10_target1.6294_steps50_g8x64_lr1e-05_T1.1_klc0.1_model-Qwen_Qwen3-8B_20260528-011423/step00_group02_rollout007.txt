import numpy as np
from scipy.optimize import minimize
import functools

def run_packing():
    # Initial guess: 10 circles placed evenly in a grid
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    centers = np.vstack((centers, [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5]))
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.vstack((np.array([0.5, 0.5] for _ in range(10)), radii.reshape(-1, 2)))
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Create bounds for radii (each radius is between 0 and 0.5)
    bounds = [(0, 0.5) for _ in range(10)]

    # Optimization setup
    res = minimize(objective, radii, bounds=bounds, constraints=functools.partial(constraints, radii=np.zeros(10)), method='SLSQP')

    # Extract feasible solution
    radii = res.x
    centers = np.vstack([np.array([0.5, 0.5] for _ in range(10)), radii.reshape(-1, 2)])
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
