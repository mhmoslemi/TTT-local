import numpy as np
from scipy.optimize import minimize
from numpy import sqrt, pi

def run_packing():
    # Initial guess: randomly place 10 circles with small radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1
    centers = np.column_stack((centers, radii))  # For tracking

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(x):
        centers = x[:60].reshape(10, 3)
        radii = x[60:]
        return -np.sum(radii)

    # Constraint: circles are entirely within the unit square
    def square_constraint(x):
        centers = x[:60].reshape(10, 3)
        radii = x[60:]
        return [1 - centers[i, 0] - radii[i] for i in range(10)] + \
               [centers[i, 0] - radii[i] for i in range(10)] + \
               [1 - centers[i, 1] - radii[i] for i in range(10)] + \
               [centers[i, 1] - radii[i] for i in range(10)]

    # Constraint: circles do not overlap
    def no_overlap_constraint(x):
        centers = x[:60].reshape(10, 3)
        radii = x[60:]
        distances = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = sqrt(dx*dx + dy*dy)
                distances[idx] = distance - radii[i] - radii[j]
                idx += 1
        return distances

    # Initial guess flattened
    x0 = np.hstack((centers.reshape(10 * 3), radii))

    # Bounds: x, y between 0 and 1, radii between 0 and 0.5
    bounds = [(0, 1) for _ in range(10 * 3)] + [(0, 0.5) for _ in range(10)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': no_overlap_constraint}
    ]

    # Optimize
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-8)

    # Extract results
    if result.success:
        centers_result = result.x[:60].reshape(10, 2)
        radii_result = result.x[60:]
        sum_radii_result = np.sum(radii_result)
    else:
        # Fallback to initial guess if optimization fails
        centers_result = centers[:, :2]
        radii_result = radii
        sum_radii_result = np.sum(radii)

    return (centers_result, radii_result, sum_radii_result)
