import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 equally spaced circles along diagonal
    centers = np.array([[i/9, i/9] for i in range(10)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints_fn(x):
        # x is flattened array of centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        
        # Constraint 1: Circles must be inside the unit square
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: Circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0
    
    # Flatten centers and radii for optimization
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Set up optimization problem
    res = minimize(objective, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints_fn})
    
    # Extract results
    x_opt = res.x
    centers_opt = x_opt[:20].reshape(10, 2)
    radii_opt = x_opt[20:]
    sum_radii_opt = -res.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
