import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    radii = np.full(10, 0.1)

    # Objective function to minimize: negative of sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_center_inside(radii):
        return np.ones(10) * (radii <= 0.5)

    def constraint_overlap(radii):
        n = 10
        dists = np.zeros(n * n)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[j * n + i] = math.sqrt(dx * dx + dy * dy)
                dists[i * n + j] = dists[j * n + i]
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :]).flatten()

    # Bounds for radii
    bounds = [(0.0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_center_inside},
            {'type': 'ineq', 'fun': constraint_overlap}
        ]
    )

    radii = result.x
    sum_radii = np.sum(radii)

    # Ensure all circles fit inside the square
    for i in range(10):
        x, y = centers[i]
        if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
            radii[i] = min(radii[i], 1 - max(x, y))

    # Re-calculate sum_radii
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
