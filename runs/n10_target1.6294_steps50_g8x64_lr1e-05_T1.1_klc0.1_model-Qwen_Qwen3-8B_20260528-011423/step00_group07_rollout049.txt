import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    # Objective function: minimize negative sum of radii (equivalent to maximizing sum)
    def objective(x):
        # x is flattened array of (x1, y1, r1, x2, y2, r2, ..., x10, y10, r10)
        centers_flat = x[:20].reshape((10, 2))
        radii_flat = x[20:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints: 
    # 1. Each circle must be within the unit square
    # 2. No two circles may overlap
    # 3. All radii must be non-negative

    def constraint_square(x):
        centers_flat = x[:20].reshape((10, 2))
        radii_flat = x[20:]
        return np.concatenate([
            (centers_flat[:, 0] - radii_flat) - 0.0,
            (centers_flat[:, 1] - radii_flat) - 0.0,
            (1.0 - centers_flat[:, 0] - radii_flat) - 0.0,
            (1.0 - centers_flat[:, 1] - radii_flat) - 0.0
        ])

    def constraint_overlap(x):
        centers_flat = x[:20].reshape((10, 2))
        radii_flat = x[20:]
        dist = cdist(centers_flat, centers_flat)
        return dist - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    def constraint_radii(x):
        return x[20:]

    # Set up optimization problem
    bounds = [(0.0, 1.0) for _ in range(20)] + [(0.0, 1.0) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_square},
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_radii}
    ]

    result = minimize(objective, np.r_[centers.ravel(), radii], method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract final solution
    centers_final = result.x[:20].reshape((10, 2))
    radii_final = result.x[20:]
    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
