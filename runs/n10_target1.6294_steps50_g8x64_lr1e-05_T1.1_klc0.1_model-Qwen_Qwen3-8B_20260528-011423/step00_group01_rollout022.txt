import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as dist

def run_packing():
    # Initial guess for centers (randomly placed within the square)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial radii

    def objective_function(params):
        # Unpack parameters into centers and radii
        centers_flat = params[:10*2].reshape(10, 2)
        radii_flat = params[10*2:].reshape(10)

        # Constraint: circles are inside the unit square
        inside = np.all((radii_flat <= centers_flat[:, 0]) & (centers_flat[:, 0] <= 1 - radii_flat) &
                        (radii_flat <= centers_flat[:, 1]) & (centers_flat[:, 1] <= 1 - radii_flat), axis=0)

        # Constraint: circles do not overlap
        distances = dist.cdist(centers_flat, centers_flat)
        radii_sum = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        non_overlapping = np.all(distances >= radii_sum, axis=1)

        # If any constraint is violated, penalize the objective
        if not np.all(inside) or not np.all(non_overlapping):
            return np.inf

        # Objective: minimize the negative sum of radii (maximize sum)
        return -np.sum(radii_flat)

    # Bounds for parameters: centers in [0,1], radii >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0)] * 2)  # x, y for each center
        bounds.extend([(0.0, 0.5)])     # radius for each circle

    result = minimize(objective_function, np.concatenate([centers.flatten(), radii.flatten()]),
                      method='L-BFGS-B', bounds=bounds, tol=1e-6)

    # Extract final parameters
    final_centers = result.x[:10*2].reshape(10, 2)
    final_radii = result.x[10*2:].reshape(10)
    total_sum = np.sum(final_radii)

    return (final_centers, final_radii, total_sum)
