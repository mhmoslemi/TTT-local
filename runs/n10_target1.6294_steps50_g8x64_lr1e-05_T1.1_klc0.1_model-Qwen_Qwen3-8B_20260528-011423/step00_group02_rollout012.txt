import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)

    def objective(x):
        # x is a flattened array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative to maximize sum_radii

    def constraints(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii))
        # Constraint 2: No overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return inside

    # Define bounds for each variable
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0 - 0.01)])  # r is at least 0.01 to avoid zero

    result = minimize(
        objective,
        x0=np.concatenate([initial_centers.ravel(), initial_radii]),
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=bounds,
        tol=1e-6
    )

    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun  # Objective is -sum_radii, so negate to get sum_radii

    return (centers, radii, sum_radii)
