```python
import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess: equally spaced grid points
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_bounds(radii):
        return np.abs(radii) <= 0.5

    def constraint_overlap(radii):
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx * dx + dy * dy)
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': constraint_bounds},
            {'type': 'ineq', 'fun': constraint_overlap}
        ],
        bounds=[(0, 0.5) for _ in range(10)]
    )

    return (centers, result.x, -result.fun)
```
