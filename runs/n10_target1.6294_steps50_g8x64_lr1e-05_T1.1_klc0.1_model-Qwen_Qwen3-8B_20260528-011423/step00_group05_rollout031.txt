import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly placed)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Initial radius guess

    def objective_function(radii_vec):
        # Objective: minimize the negative sum of radii (maximize sum)
        return -np.sum(radii_vec)

    def constraint_circle_inside(radii_vec, centers):
        # Each circle must be fully inside the unit square
        x, y = centers.T
        r = radii_vec
        return r - (1 - r - x) - (1 - r - y)

    def constraint_no_overlap(radii_vec, centers):
        # No two circles may overlap
        # Convert centers and radii to arrays
        r = radii_vec
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)
                dists[j, i] = dists[i, j]
        return dists - (r[:, np.newaxis] + r[np.newaxis, :])

    # Prepare optimization problem
    bounds = [(0, 1) for _ in range(10)]  # Bounds for radii

    # Constraints
    constraints = []
    # Constraint: circles inside the unit square
    constraints.append({'type': 'ineq', 'fun': lambda x: constraint_circle_inside(x, centers)})
    # Constraint: no overlap between circles
    constraints.append({'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x, centers)})

    # Prepare initial guess
    initial_guess = radii.copy()

    # Run optimization
    result = minimize(
        objective_function,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8
    )

    # Extract result
    radii_opt = result.x
    centers_opt = centers.copy()
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
