import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Number of circles
    n = 10

    # Initial guess: evenly spaced centers
    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = (i + 0.5) / n
        centers[i, 1] = 0.5

    # Initial radii: all set to 0.1
    radii = np.full(n, 0.1)

    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., xn, yn, rn]
        x_vec = x[:2*n]
        r_vec = x[2*n:]
        centers_flat = x_vec.reshape(n, 2)
        radii_flat = r_vec
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints
    def constraints(x):
        x_vec = x[:2*n]
        r_vec = x[2*n:]
        centers_flat = x_vec.reshape(n, 2)
        radii_flat = r_vec
        # Constraint 1: Each circle must be inside the unit square
        inside = np.all((radii_flat <= centers_flat[:, 0]) & (centers_flat[:, 0] <= 1 - radii_flat) &
                        (radii_flat <= centers_flat[:, 1]) & (centers_flat[:, 1] <= 1 - radii_flat))
        if not inside:
            return np.inf
        # Constraint 2: No two circles overlap
        dist = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist[i, j] = np.sqrt(dx*dx + dy*dy)
                dist[j, i] = dist[i, j]
        overlap = np.any(dist < radii_flat[np.newaxis, :] + radii_flat[:, np.newaxis])
        if overlap:
            return np.inf
        return 0

    # Bounds for x, y, and r
    bounds = []
    for i in range(n):
        bounds.extend([(0, 1), (0, 1), (0, 0.5)])  # Max radius is 0.5 to stay inside the square

    # Optimization settings
    options = {
        'ftol': 1e-6,
        'maxiter': 1000,
        'disp': False
    }

    # Run optimization
    result = minimize(
        objective,
        x0=np.concatenate((centers.flatten(), radii)),
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'eq', 'fun': constraints},
        options=options
    )

    # Extract result
    x_opt = result.x
    centers_opt = x_opt[:2*n].reshape(n, 2)
    radii_opt = x_opt[2*n:]
    sum_radii_opt = -objective(x_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
