import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[10:])

    # Constraints: 1. Circle inside the square, 2. No overlap between circles
    def constraint_inside(x):
        for i in range(10):
            r = x[10 + i]
            cx, cy = x[i], x[i + 1]
            if not (r <= cx <= 1 - r and r <= cy <= 1 - r):
                return False
        return True

    def constraint_overlap(x):
        for i in range(10):
            for j in range(i + 1, 10):
                r1 = x[10 + i]
                r2 = x[10 + j]
                cx1, cy1 = x[i], x[i + 1]
                cx2, cy2 = x[j], x[j + 1]
                dist = math.hypot(cx1 - cx2, cy1 - cy2)
                if dist < r1 + r2 - 1e-8:
                    return False
        return True

    # Define the optimization problem
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x and y coordinates
    for i in range(10):
        bounds.append((0.0, 0.5))  # radius (assuming optimal radius <= 0.5)

    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimize
    result = minimize(
        objective,
        np.concatenate((initial_centers.flatten(), initial_radii)),
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-6
    )

    # Extract results
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
