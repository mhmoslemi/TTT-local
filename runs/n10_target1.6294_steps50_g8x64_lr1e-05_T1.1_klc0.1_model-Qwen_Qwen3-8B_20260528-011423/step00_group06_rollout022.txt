import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.15

    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)

    # Constraints: 1. All circles inside the square
    #              2. No two circles overlap
    #              3. All radii non-negative
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return 1.0
        return 0.0

    def constraint_distance(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i, j in itertools.combinations(range(10), 2):
            dx = centers_flat[i][0] - centers_flat[j][0]
            dy = centers_flat[i][1] - centers_flat[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii_flat[i] + radii_flat[j]:
                return 1.0
        return 0.0

    # Non-negativity constraint
    def constraint_non_negative(x):
        return x[60:]

    # Create constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_distance},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]

    # Bounds for centers (0 <= x <= 1, 0 <= y <= 1) and radii (0 <= r <= 0.5)
    bounds = [(0, 1) for _ in range(60)] + [(0, 0.5) for _ in range(10)]

    # Optimize
    result = minimize(objective, np.concatenate([centers.flatten(), radii]), method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
