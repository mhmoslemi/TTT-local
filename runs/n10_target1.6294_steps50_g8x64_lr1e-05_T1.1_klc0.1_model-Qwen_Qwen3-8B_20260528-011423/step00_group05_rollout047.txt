import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[i*0.25 + 0.1, j*0.25 + 0.1] for i in range(10) for j in range(1)])
    radii = np.full(10, 0.1)
    
    def objective_function(x):
        # x is a flat array of centers and radii
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        constraints = []
        
        # Each circle must be inside the unit square
        for i in range(10):
            r = radii_flat[i]
            x_i, y_i = centers_flat[i]
            if x_i - r < 0 or x_i + r > 1 or y_i - r < 0 or y_i + r > 1:
                return [False]
        
        # No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.hypot(dx, dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 1e-8:
                    return [False]
        
        return [True]
    
    # Convert centers and radii to a flat array for optimization
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Set up optimization problem
    result = minimize(
        objective_function,
        x0,
        method='L-BFGS-B',
        constraints={'type': 'eq', 'fun': constraints},
        options={'ftol': 1e-8, 'maxiter': 1000}
    )
    
    # Extract results
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii_opt = -result.fun  # Since we minimized the negative sum
    
    return (centers_opt, radii_opt, sum_radii_opt)
