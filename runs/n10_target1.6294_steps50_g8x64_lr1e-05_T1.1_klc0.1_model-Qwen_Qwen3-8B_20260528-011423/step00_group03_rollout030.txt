import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: spread circles evenly in the square
    np.random.seed(0)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1 + 0.05

    # Objective function: maximize the sum of radii
    def objective(x):
        # Flatten x into centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize the negative to maximize

    # Constraints: all circles must be inside the square
    def in_square(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        return np.all((centers_flat[:, 0] - radii_flat >= 0) &
                      (centers_flat[:, 1] - radii_flat >= 0) &
                      (centers_flat[:, 0] + radii_flat <= 1) &
                      (centers_flat[:, 1] + radii_flat <= 1))

    # Constraint: no overlap between circles
    def no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return False
        return True

    # Combine constraints
    def constraints(x):
        return [in_square(x), no_overlap(x)]

    # Bounds for centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x and y for each center
        bounds.extend([(0, 0.5)])  # radius for each circle

    # Use an initial guess
    initial_guess = np.concatenate([centers.flatten(), radii.flatten()])

    # Set up the optimization problem
    result = minimize(objective, initial_guess, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:].reshape(10)
    sum_radii = -result.fun  # Objective function returns -sum_radii

    return (centers, radii, sum_radii)
