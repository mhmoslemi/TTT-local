import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.08)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Ensure no circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization with constraints
    result = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}])
    optimized_radii = result.x
    optimized_centers = centers.copy()

    # Ensure circles are inside the square
    for i in range(10):
        if optimized_radii[i] < 0:
            optimized_radii[i] = 0
        if not (optimized_radii[i] <= optimized_centers[i, 0] <= 1 - optimized_radii[i] and
                optimized_radii[i] <= optimized_centers[i, 1] <= 1 - optimized_radii[i]):
            optimized_centers[i, 0] = optimized_radii[i]
            optimized_centers[i, 1] = optimized_radii[i]

    # Final check for non-overlapping
    for i in range(10):
        for j in range(i + 1, 10):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            dist = math.sqrt(dx**2 + dy**2)
            if dist < optimized_radii[i] + optimized_radii[j]:
                print(f"Overlap between circle {i} and {j}")
                return (optimized_centers, optimized_radii, np.sum(optimized_radii))

    return (optimized_centers, optimized_radii, np.sum(optimized_radii))
