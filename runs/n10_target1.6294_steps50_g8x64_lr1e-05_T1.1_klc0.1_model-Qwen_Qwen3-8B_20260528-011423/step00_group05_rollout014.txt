import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing the negative

    def constraints(radii, centers):
        constraints = []
        n = len(radii)
        # Ensure circles are inside the unit square
        for i in range(n):
            r = radii[i]
            if r > 0:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - centers[i, 0] + r})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - centers[i, 0] + r})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - centers[i, 1] + r})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - centers[i, 1] + r})
        # Ensure no two circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                r_i = radii[i]
                r_j = radii[j]
                def dist_constraint(radii):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    return np.sqrt(dx**2 + dy**2) - r_i - r_j
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        return constraints

    def bounds(n):
        return [(0.0, 0.1)] * n  # Limit bounds to avoid unbounded optimization

    # Initial guess: place circles randomly
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial radii within a small range

    # Optimize
    result = minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        bounds=bounds(10),
        constraints=constraints(radii, centers),
        method='SLSQP'
    )

    # Extract result
    final_radii = result.x
    final_centers = centers.copy()
    sum_radii = np.sum(final_radii)
    return (final_centers, final_radii, sum_radii)
