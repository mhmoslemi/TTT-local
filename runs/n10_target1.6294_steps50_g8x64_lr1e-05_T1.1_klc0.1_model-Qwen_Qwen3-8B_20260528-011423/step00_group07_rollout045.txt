import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced points in the unit square
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1
    radii[radii < 0.01] = 0.01  # Ensure radii are not too small

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraint_center_inside(radii, centers):
        return np.array([x - r >= 0 for x, r in zip(centers[:, 0], radii)] +
                        [1 - x - r >= 0 for x, r in zip(centers[:, 0], radii)] +
                        [y - r >= 0 for y, r in zip(centers[:, 1], radii)] +
                        [1 - y - r >= 0 for y, r in zip(centers[:, 1], radii)])

    def constraint_no_overlap(radii, centers):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.array([dists[i, j] - radii[i] - radii[j] for i in range(10) for j in range(i+1)])

    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (radii, centers)}
    ]

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            args=(centers,),
            method='SLSQP',
            constraints=constraints,
            bounds=[(0, 1 - 0.01) for _ in range(10)],
            jac=lambda x, c: -np.ones_like(x)
        )

    best_radii = result.x
    best_centers = centers.copy()
    sum_radii = -result.fun

    return (best_centers, best_radii, sum_radii)
