import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers (random positions)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Initial radius guess

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints_func(centers, radii):
        constraints = []
        # Constraint 1: Each circle is fully inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1 - y - r})
        
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (radii[i] + radii[j])})
        
        # Constraint 3: Radii must be non-negative
        for i in range(10):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})

        return constraints

    # Optimization
    bounds = [(0, 1) for _ in range(10 * 2)]  # Bounds for centers
    radii_bounds = [(0, 1) for _ in range(10)]  # Bounds for radii

    # Combine center positions and radii into a single vector for optimization
    x0 = np.hstack((centers.flatten(), radii))
    res = minimize(
        lambda x: objective(x[10:], x[:20].reshape(10, 2)),
        x0,
        method='SLSQP',
        bounds=np.vstack((bounds, radii_bounds)),
        constraints=constraints_func(centers, radii)
    )

    # Extract results
    centers_opt = res.x[:20].reshape(10, 2)
    radii_opt = res.x[20:]
    sum_radii = np.sum(radii_opt)
    return (centers_opt, radii_opt, sum_radii)
