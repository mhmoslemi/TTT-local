import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    num_circles = 10
    initial_centers = np.random.rand(num_circles, 2)
    initial_radii = np.random.rand(num_circles) * 0.1

    def objective_function(params):
        centers = params[:num_circles*2].reshape(num_circles, 2)
        radii = params[num_circles*2:].reshape(num_circles)
        # Objective: maximize sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraints(params):
        centers = params[:num_circles*2].reshape(num_circles, 2)
        radii = params[num_circles*2:].reshape(num_circles)
        constraints = []
        # Constraint 1: Each circle is within the unit square
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r >= 0) * 1e6)
            constraints.append((x + r <= 1) * 1e6)
            constraints.append((y - r >= 0) * 1e6)
            constraints.append((y + r <= 1) * 1e6)
        # Constraint 2: No overlap between circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append((dist - (radii[i] + radii[j])) * 1e6)
        return np.sum(constraints)

    # Setup optimization
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0.0, 1.0)] * 2)  # x and y for each center
        bounds.extend([0.0, 0.5])        # radius between 0 and 0.5
    initial_params = np.concatenate([initial_centers.flatten(), initial_radii])
    result = minimize(objective_function, initial_params, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    
    centers = result.x[:num_circles*2].reshape(num_circles, 2)
    radii = result.x[num_circles*2:].reshape(num_circles)
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
