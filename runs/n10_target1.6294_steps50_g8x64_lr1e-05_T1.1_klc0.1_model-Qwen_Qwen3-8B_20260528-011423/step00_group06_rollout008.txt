import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Constraint 1: Each circle is inside the square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Bounds for radii (must be non-negative)
    bounds = [(0.0, 1.0) for _ in range(10)]
    
    # Optimization setup
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=[lambda r: constraints(r)])
    
    # Extract result
    radii = res.x
    sum_radii = -res.fun
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    
    return (centers, radii, sum_radii)
