import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.array([r <= 1 - r for r in radii])

    def constraint_no_overlap(centers, radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraint_sum_radii(radii):
        return np.sum(radii) - 1.6294

    bounds = [(0, 1 - r) for r in radii]

    # Optimize
    res = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: np.min(constraint_circle_inside(r))},
            {'type': 'ineq', 'fun': lambda r: np.min(constraint_no_overlap(centers, r))},
            {'type': 'eq', 'fun': constraint_sum_radii}
        ]
    )

    # Update centers based on optimal radii
    # For simplicity, we'll keep centers fixed but this could be optimized further
    centers = np.array(centers)
    return (centers, res.x, np.sum(res.x))
