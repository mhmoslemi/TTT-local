import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles arranged in grid pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
    radii = np.full(10, 0.15)
    
    def objective(x):
        return -np.sum(x)  # Maximize radii sum
    
    def constraint_center_inside(x):
        return x <= 1 - x[::2]  # x <= 1 - x for each circle (x_i <= 1 - x_i)
    
    def constraint_overlap(x):
        n = 10
        r = x[n:]
        centers = x[:n*2].reshape(n, 2)
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (r[np.newaxis, :] + r[:, np.newaxis])
    
    # Objective function
    def objective_func(x):
        return -np.sum(x[n*2:])  # Maximize radii
    
    # Constraints
    n = 10
    cons = []
    cons.append({'type': 'ineq', 'fun': constraint_center_inside})
    cons.append({'type': 'ineq', 'fun': constraint_overlap})
    
    # Initial guess and bounds
    x0 = np.concatenate((centers.flatten(), radii))
    bounds = []
    for i in range(n*2):
        bounds.append((0, 1))
    for i in range(n):
        bounds.append((0, 1))
    bounds = [(0, 1)] * (n * 2 + n)
    
    # Optimize
    result = minimize(objective_func, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)
    
    x_opt = result.x
    centers_opt = x_opt[:n*2].reshape(n, 2)
    radii_opt = x_opt[n*2:]
    sum_radii = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii)
