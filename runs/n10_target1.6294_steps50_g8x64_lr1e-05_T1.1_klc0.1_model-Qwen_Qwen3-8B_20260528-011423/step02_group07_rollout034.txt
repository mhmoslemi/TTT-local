import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles in a grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Check if all radii are non-negative
        for r in radii:
            if r < 0:
                return np.inf
        # Get current centers based on the initial grid
        centers = initial_centers.copy()
        # Check if all circles are inside the unit square
        for i in range(10):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return np.inf
        # Check if all circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    res = minimize(objective, radii, method='SLSQP', 
                   bounds=[(0, 1) for _ in range(10)], 
                   constraints={'type': 'ineq', 'fun': constraints})
    
    # Return result
    centers = initial_centers.copy()
    return (centers, res.x, -res.fun)
