import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.full(10, 0.1)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative to maximize

    # Constraints:
    # 1. Each circle stays within the unit square
    def constraint_square(radii):
        return np.array([radii[i] for i in range(10)])
    
    # 2. No two circles overlap
    def constraint_overlap(radii):
        dists = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                dists[idx] = dist - (radii[i] + radii[j])
                idx += 1
        return dists

    # 3. Non-negativity of radii
    def constraint_non_negative(radii):
        return radii

    # Setup optimization problem
    bounds = [(0.0, 0.5) for _ in range(10)]  # Upper bound on radii to ensure circles fit in square

    # Optimization using SLSQP
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_square},
            {'type': 'ineq', 'fun': constraint_overlap},
            {'type': 'ineq', 'fun': constraint_non_negative}
        ]
    )

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
