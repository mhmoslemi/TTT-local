import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    # Objective function to minimize negative sum of radii
    def objective(params):
        # Flatten the parameters for easier handling
        flat_params = params.reshape(-1)
        centers_flat = flat_params[:10*2].reshape(10, 2)
        radii_flat = flat_params[10*2:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints: circles must be inside the unit square
    def constraint_inside(params):
        flat_params = params.reshape(-1)
        centers_flat = flat_params[:10*2].reshape(10, 2)
        radii_flat = flat_params[10*2:].reshape(10)
        inside = np.all((centers_flat - radii_flat[:, np.newaxis]) >= 0)
        inside &= np.all((1 - centers_flat - radii_flat[:, np.newaxis]) >= 0)
        return inside

    # Constraints: circles must not overlap
    def constraint_no_overlap(params):
        flat_params = params.reshape(-1)
        centers_flat = flat_params[:10*2].reshape(10, 2)
        radii_flat = flat_params[10*2:].reshape(10)
        distances = np.sqrt(np.sum((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2, axis=2))
        overlap = np.all(distances >= radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])
        return overlap

    # Initial bounds for parameters
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1) for _ in range(2)])  # Centers x, y
        bounds.extend([(0, 0.5) for _ in range(10)])  # Radii (max radius is 0.5)

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii.flatten()]),
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}
        ],
        method='SLSQP',
        tol=1e-6
    )

    centers_opt = result.x[:10*2].reshape(10, 2)
    radii_opt = result.x[10*2:].reshape(10)
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
