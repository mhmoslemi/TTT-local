import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25

    # Initial radii
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: radii >= 0
        for r in radii:
            if r < 0:
                return False
        # Constraint 2: circles are inside the unit square
        for i in range(10):
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return False
        # Constraint 3: circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization setup
    bounds = [(0, 1 - 0.1) for _ in range(10)]  # Allow radii up to 0.9

    result = minimize(
        objective,
        radii,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraints(x)}
        ],
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    # Extract results
    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
