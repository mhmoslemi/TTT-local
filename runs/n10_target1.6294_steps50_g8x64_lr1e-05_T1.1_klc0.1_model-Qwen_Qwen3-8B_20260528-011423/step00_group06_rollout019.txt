import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with some initial radius
    initial_centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    initial_radii = np.full(9, 0.1)
    initial_centers = np.vstack((initial_centers, [0.5, 0.5]))
    initial_radii = np.append(initial_radii, 0.1)

    def objective_function(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = np.random.uniform(radii[i], 1 - radii[i])
            centers[i, 1] = np.random.uniform(radii[i], 1 - radii[i])

        # Constraint 1: All circles fully inside the unit square
        square_constraints = []
        for i in range(10):
            x = centers[i, 0]
            y = centers[i, 1]
            square_constraints.append((x - radii[i] >= 0) * 1e6)
            square_constraints.append((x + radii[i] <= 1) * 1e6)
            square_constraints.append((y - radii[i] >= 0) * 1e6)
            square_constraints.append((y + radii[i] <= 1) * 1e6)
        square_constraints = np.sum(square_constraints)

        # Constraint 2: No two circles overlap
        overlap_constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                overlap_constraints.append((dist - (radii[i] + radii[j])) * 1e6)
        overlap_constraints = np.sum(overlap_constraints)

        return square_constraints + overlap_constraints

    bounds = [(0, 0.5) for _ in range(10)]

    result = minimize(objective_function, initial_radii, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints}, bounds=bounds, tol=1e-6)

    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = np.random.uniform(result.x[i], 1 - result.x[i])
        centers[i, 1] = np.random.uniform(result.x[i], 1 - result.x[i])

    sum_radii = np.sum(result.x)

    return (centers, result.x, sum_radii)
