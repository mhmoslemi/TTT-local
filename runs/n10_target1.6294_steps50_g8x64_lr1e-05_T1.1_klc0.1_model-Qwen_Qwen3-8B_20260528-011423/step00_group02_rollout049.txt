import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 10 circles in the unit square
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # small initial radii

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # minimize the negative sum to maximize

    # Constraints
    def constraint_center(radii):
        # Ensure circles are within the unit square
        return np.sum((radii <= np.min(centers[:, 0]) - 1e-8) + (radii <= np.min(centers[:, 1]) - 1e-8) +
                      (radii >= np.max(centers[:, 0]) + 1e-8) + (radii >= np.max(centers[:, 1]) + 1e-8))

    def constraint_overlap(radii):
        # Ensure circles do not overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum((dists < radii[:, np.newaxis] + radii[np.newaxis, :]) * (dists > 1e-8))

    # Optimization problem
    def objective_function(x):
        # x is a flat list of [r0, r1, ..., r9]
        radii = x.reshape(10,)
        # Check if circles stay within the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf  # Penalize if any circle is out of bounds
        # Check for overlaps
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf  # Penalize if any two circles overlap
        return -np.sum(radii)  # return negative sum to minimize

    # Initial guess for radii
    initial_guess = np.random.rand(10) * 0.1
    result = minimize(objective_function, initial_guess, method='L-BFGS-B', bounds=[(0, 0.5)]*10)

    # Update centers and radii with the results
    optimal_radii = result.x
    centers = np.array([[random.uniform(r, 1 - r) for _ in range(2)] for r in optimal_radii])

    # Ensure circles stay within bounds
    for i in range(10):
        if not (optimal_radii[i] <= centers[i, 0] <= 1 - optimal_radii[i] and optimal_radii[i] <= centers[i, 1] <= 1 - optimal_radii[i]):
            optimal_radii[i] = np.min([centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1]])

    # Final check for overlaps
    dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
    for i in range(10):
        for j in range(i + 1, 10):
            if dists[i, j] < optimal_radii[i] + optimal_radii[j]:
                optimal_radii[i] = dists[i, j] - optimal_radii[j] - 1e-8
                optimal_radii[j] = dists[i, j] - optimal_radii[i] - 1e-8

    sum_radii = np.sum(optimal_radii)
    return (centers, optimal_radii, sum_radii)
