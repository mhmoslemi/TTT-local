import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(x):
        # x is a flat array of 20 elements: x0, y0, r0, x1, y1, r1, ..., x9, y9, r9
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize the negative sum to maximize it

    def constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        constraints = []

        # Boundary constraints
        for i in range(10):
            x_i, y_i, r_i = centers[i][0], centers[i][1], radii[i]
            constraints.append(r_i <= x_i - r_i)
            constraints.append(r_i <= 1 - x_i - r_i)
            constraints.append(r_i <= y_i - r_i)
            constraints.append(r_i <= 1 - y_i - r_i)

        # Pairwise non-overlapping constraints
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.sqrt(dx**2 + dy**2)
                constraints.append(dist >= radii[i] + radii[j])

        return constraints

    # Initial guess: place circles in a grid pattern
    initial_centers = np.array([[0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
                                [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]])
    initial_radii = np.ones(10) * 0.05
    initial_x = initial_centers.flatten()
    initial_x = np.concatenate((initial_x, initial_radii))

    # Set up optimization problem
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x_i
        bounds.append((0.0, 1.0))  # y_i
        bounds.append((0.0, 1.0))  # r_i

    res = minimize(objective, initial_x, method='SLSQP', bounds=bounds, constraints=constraints)

    centers = res.x[:20].reshape(10, 2)
    radii = res.x[20:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
