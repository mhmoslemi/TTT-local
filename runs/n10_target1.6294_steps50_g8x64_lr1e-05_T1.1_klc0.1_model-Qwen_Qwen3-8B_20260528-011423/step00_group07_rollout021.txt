import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.random.uniform(0.05, 0.15, 10)
    radii = np.clip(radii, 0, 0.5)

    def objective(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        radii_flat = np.clip(radii_flat, 0, 0.5)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        radii_flat = np.clip(radii_flat, 0, 0.5)
        # Boundary constraints: circle must be inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Overlap constraints: distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0

    # Bounds for centers and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0.1, 0.9) for _ in range(2)])  # x and y for each center
        bounds.extend([(0.0, 0.5)])  # radius for each circle

    # Set up optimization problem
    x0 = np.concatenate([centers.flatten(), radii])
    result = minimize(objective, x0, method='L-BFGS-B', bounds=bounds, constraints=constraints, tol=1e-5, options={'maxiter': 1000})

    # Extract final result
    centers_final = result.x[:60].reshape(10, 2)
    radii_final = result.x[60:].reshape(10)
    sum_radii_final = -result.fun  # negate to get the maximized sum

    return (centers_final, radii_final, sum_radii_final)
