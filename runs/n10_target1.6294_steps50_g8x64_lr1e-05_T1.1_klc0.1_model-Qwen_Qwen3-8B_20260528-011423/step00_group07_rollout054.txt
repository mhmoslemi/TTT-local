import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.ones(10) * 0.1

    def objective(x):
        # Flatten the input into centers and radii
        centers_flat = x[:10*2].reshape(10, 2)
        radii_flat = x[10*2:]
        sum_r = np.sum(radii_flat)
        return -sum_r  # Minimize -sum_r to maximize sum_r

    def constraints(x):
        # Flatten the input into centers and radii
        centers_flat = x[:10*2].reshape(10, 2)
        radii_flat = x[10*2:]

        # Constraint 1: All circles are within the unit square
        bounds = [(0 + r, 1 - r) for r in radii_flat]
        for i in range(10):
            if not (bounds[i][0] <= centers_flat[i, 0] <= bounds[i][1] and
                    bounds[i][0] <= centers_flat[i, 1] <= bounds[i][1]):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0

    # Initial guess for x: centers (10*2) + radii (10)
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up optimization problem
    result = minimize(objective, x0, method='SLSQP', bounds=[(0, 1) for _ in range(10*2)] + [(0, 1) for _ in range(10)],
                      constraints={'type': 'ineq', 'fun': constraints}, tol=1e-8, options={'maxiter': 500})

    # Extract the result
    centers_result = result.x[:10*2].reshape(10, 2)
    radii_result = result.x[10*2:]
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
