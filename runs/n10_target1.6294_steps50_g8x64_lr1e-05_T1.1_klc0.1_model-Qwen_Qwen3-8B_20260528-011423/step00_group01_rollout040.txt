import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25

    radii = np.ones(10) * 0.15  # Initial guess for radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(radii, centers):
        return np.min(radii - (centers[:, 0] - 0.5)**2 - (centers[:, 1] - 0.5)**2)  # Placeholder for constraints

    def constraint_no_overlap(radii, centers):
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization
    bounds = [(0, 0.5) for _ in range(10)]
    cons = (
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: np.min(constraint_no_overlap(r, centers))},
    )

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'ftol': 1e-9, 'eps': 1e-9}
    )

    centers = np.array(result.x).reshape(-1, 2)
    radii = np.array(result.x)
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
