import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    # Objective function to minimize (negative sum of radii)
    def objective(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)

    # Constraints: 
    # 1. Each circle is inside the unit square
    # 2. No two circles overlap
    # 3. Radii are non-negative

    def constraint_inside_square(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([r - x[i, 0] + r <= 0 for i, r in enumerate(radii)] +
                        [r - (1 - x[i, 0]) + r <= 0 for i, r in enumerate(radii)] +
                        [r - x[i, 1] + r <= 0 for i, r in enumerate(radii)] +
                        [r - (1 - x[i, 1]) + r <= 0 for i, r in enumerate(radii)])

    def constraint_no_overlap(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        dists = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                idx += 1
        return dists

    cons = [
        {'type': 'ineq', 'fun': constraint_inside_square},
        {'type': 'ineq', 'fun': constraint_no_overlap},
        {'type': 'ineq', 'fun': lambda x: x[60:]}
    ]

    # Bounds for centers and radii
    bounds = [(0, 1) for _ in range(60)] + [(0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(objective, np.concatenate([centers.reshape(10 * 2), radii]), 
                      method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
