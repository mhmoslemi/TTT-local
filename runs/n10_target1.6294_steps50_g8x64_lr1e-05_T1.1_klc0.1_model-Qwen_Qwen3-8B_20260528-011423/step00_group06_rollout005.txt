```python
import numpy as np
from scipy.optimize import minimize
import warnings

# Helper function to compute the sum of radii
def objective(radii):
    return -np.sum(radii)  # We minimize the negative sum to maximize the sum

# Helper function to check constraints
def constraints(centers, radii):
    n = len(centers)
    # Constraint 1: All circles are inside the unit square
    # x_i - r_i >= 0, 1 - x_i - r_i >= 0, same for y
    # We'll enforce this by minimizing the violation
    inside = np.sum((centers[:, 0] - radii) * (1 - centers[:, 0] - radii) * (centers[:, 1] - radii) * (1 - centers[:, 1] - radii))
    # Constraint 2: No circles overlap: distance between centers >= sum of radii
    # We'll minimize the violation of this constraint
    overlaps = 0.0
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            overlaps += max(0, dist - (radii[i] + radii[j]))
    return inside + overlaps

# Function to run the packing optimization
def run_packing():
    n = 10
    # Initial guess: place circles evenly in a grid
    centers = np.array([[i / (n - 1), j / (n - 1)] for i in range(n) for j in range(n)][:n])
    radii = np.full(n, 0.05)  # Small initial radii

    # Define bounds: x and y for each center are between [r, 1 - r]
    # Radii must be >= 0 and <= 0.5 (since r <= x <= 1 - r => r <= 0.5)
    bounds = [(0, 0.5) for _ in range(n)] * 2 + [(0, 0.5) for _ in range(n)]

    # Define the optimization problem
    def objective_with_centers(centers_and_radii):
        centers = centers_and_radii[:n*2].reshape(n, 2)
        radii = centers_and_radii[n*2:].reshape(n)
        return -np.sum(radii)

    def constraint_inside(centers_and_radii):
        centers = centers_and_radii[:n*2].reshape(n, 2)
        radii = centers_and_radii[n*2:].reshape(n)
        return np.sum((centers[:, 0] - radii) * (1 - centers[:, 0] - radii) * (centers[:, 1] - radii) * (1 - centers[:, 1] - radii))

    def constraint_overlap(centers_and_radii):
        centers = centers_and_radii[:n*2].reshape(n, 2)
        radii = centers_and_radii[n*2:].reshape(n)
        overlaps = 0.0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                overlaps += max(0, dist - (radii[i] + radii[j]))
        return overlaps

    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(
        objective_with_centers,
        x0=np.concatenate([centers.flatten(), radii]),
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-6
    )

    centers_opt = result.x[:n*2].reshape(n, 2)
    radii_opt = result.x[n*2:].reshape(n)
    sum_radii = -result.fun

    # Check if the result is valid
    if sum_radii < 0.1:
        warnings.warn("Optimization did not converge to a reasonable solution.")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return centers_opt, radii_opt, sum_radii
```
