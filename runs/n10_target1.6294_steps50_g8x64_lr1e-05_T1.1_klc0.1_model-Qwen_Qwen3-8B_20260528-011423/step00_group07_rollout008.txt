import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints_func(x):
        # x is a flat array of centers and radii: [x1, y1, r1, x2, y2, r2, ... x10, y10, r10]
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        sum_radii = np.sum(radii)
        # Constraint 1: All circles must be inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return -sum_radii  # Minimize negative sum to maximize sum

    # Prepare initial guess for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up optimization problem
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1)] * 2)  # x and y for each center
        bounds.extend([(0, 0.5)] * 1)  # radius for each circle (max radius is 0.5)
    constraints = [{'type': 'ineq', 'fun': constraints_func}]

    # Perform optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-8)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun  # Since we maximized sum by minimizing -sum

    return (centers_opt, radii_opt, sum_radii_opt)
