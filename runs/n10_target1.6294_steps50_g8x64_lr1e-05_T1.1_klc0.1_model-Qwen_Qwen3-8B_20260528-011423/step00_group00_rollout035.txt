import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    
    def objective(x):
        # x is a flat array of size 20 (10 centers * 2 + 10 radii)
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        # Constraint 1: All circles are inside the unit square
        # r <= x <= 1 - r and r <= y <= 1 - r
        # So, for each circle, we have 4 constraints: x - r >= 0, 1 - x - r >= 0, y - r >= 0, 1 - y - r >= 0
        # We convert these into a penalty function
        penalty = 0
        for i in range(10):
            r = radii_flat[i]
            x_i, y_i = centers_flat[i]
            penalty += max(0, r - x_i) + max(0, x_i + r - 1) + max(0, r - y_i) + max(0, y_i + r - 1)
        # Constraint 2: No two circles overlap
        # For every pair (i, j), distance between centers >= r_i + r_j
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                penalty += max(0, (radii_flat[i] + radii_flat[j]) - dist)
        # Objective is to maximize sum of radii, so we minimize negative of sum
        sum_radii = np.sum(radii_flat)
        return -sum_radii + penalty
    
    # Bounds for x, y, and r
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x, y for each center
        bounds.extend([0, 1])            # r for each circle
    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])
    # Optimization
    result = minimize(objective, x0, bounds=bounds, method='L-BFGS-B', tol=1e-6)
    
    # Extract results
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii_opt = -result.fun  # since the objective returns -sum_radii + penalty
    
    return (centers_opt, radii_opt, sum_radii_opt)
