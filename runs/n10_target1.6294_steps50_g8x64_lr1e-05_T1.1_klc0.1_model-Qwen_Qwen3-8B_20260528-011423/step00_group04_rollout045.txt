```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)
    
    def objective(x):
        # Flatten x to centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        
        # Objective: maximize sum of radii (minimize -sum)
        return -np.sum(radii_flat)
    
    def constraint_center_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1
        return 0
    
    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1
        return 0
    
    # Optimization setup
    bounds = []
    for _ in range(60):
        bounds.append((0, 1))  # x-coordinates
    for _ in range(10):
        bounds.append((0, 0.5))  # radii (max radius is 0.5)
    
    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    
    # Run optimization
    result = minimize(objective, x0=np.concatenate([centers.ravel(), radii]), 
                      bounds=bounds, constraints=constraints, method='SLSQP')
    
    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii)
```
