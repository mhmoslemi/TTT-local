import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.2 + i*0.25, 0.2 + j*0.25] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints_center(radii, centers):
        # Ensure circles are within bounds
        return np.concatenate([
            radii - (centers[:, 0] - radii) + (centers[:, 0] - radii) - 0.1,
            radii - (1 - centers[:, 0] - radii) + (1 - centers[:, 0] - radii) - 0.1,
            radii - (centers[:, 1] - radii) + (centers[:, 1] - radii) - 0.1,
            radii - (1 - centers[:, 1] - radii) + (1 - centers[:, 1] - radii) - 0.1
        ])

    def constraints_overlap(radii, centers):
        # Ensure circles do not overlap
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=-1))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    def objective_function(x):
        radii = x[:10]
        centers = x[10:]
        centers = centers.reshape((10, 2))
        return -np.sum(radii)

    def constraint_function(x):
        radii = x[:10]
        centers = x[10:].reshape((10, 2))
        # Bounds for radii: non-negative
        bounds = [(0, 1) for _ in range(10)]
        # Bounds for centers: [r, 1-r] in x and y
        for i in range(10):
            r = radii[i]
            x_min, x_max = r, 1 - r
            y_min, y_max = r, 1 - r
            if not (x_min <= centers[i, 0] <= x_max and y_min <= centers[i, 1] <= y_max):
                return np.inf

        # Check overlaps
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=-1))
        overlap = dist - (radii[:, np.newaxis] + radii[np.newaxis, :])
        if np.any(overlap < -1e-6):
            return np.inf

        return -np.sum(radii)

    # Setup optimization
    x0 = np.concatenate([radii, centers.reshape(-1)])
    res = minimize(objective_function, x0, bounds=[(0, 1) for _ in range(10)] + [(0, 1) for _ in range(20)], method='SLSQP', constraints={'type': 'ineq', 'fun': constraint_function}, tol=1e-6, options={'maxiter': 1000})

    final_radii = res.x[:10]
    final_centers = res.x[10:].reshape((10, 2))
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
