import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: centers at grid points, radii as small as possible
    centers = np.array([
        [0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],
        [0.3, 0.1], [0.3, 0.3], [0.3, 0.5], [0.3, 0.7], [0.3, 0.9]
    ])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraint_center_inside(center, radius):
        return np.array([
            center[0] - radius >= 0,
            center[1] - radius >= 0,
            1 - center[0] - radius >= 0,
            1 - center[1] - radius >= 0
        ])

    def constraint_no_overlap(center1, radius1, center2, radius2):
        dx = center1[0] - center2[0]
        dy = center1[1] - center2[1]
        dist = np.sqrt(dx**2 + dy**2)
        return dist - (radius1 + radius2)

    # Create bounds for radii
    bounds = [(0.0, 0.5) for _ in range(10)]

    # Define constraints
    constraints = []

    # Add constraints for each circle being inside the square
    for i in range(10):
        def constraint_center_inside_i(radii, i=i):
            center = centers[i]
            radius = radii[i]
            return constraint_center_inside(center, radius)
        constraints.append({'type': 'ineq', 'fun': constraint_center_inside_i})

    # Add constraints for non-overlapping between all pairs of circles
    for i in range(10):
        for j in range(i + 1, 10):
            def constraint_no_overlap_ij(radii, i=i, j=j):
                center1 = centers[i]
                radius1 = radii[i]
                center2 = centers[j]
                radius2 = radii[j]
                return constraint_no_overlap(center1, radius1, center2, radius2)
            constraints.append({'type': 'ineq', 'fun': constraint_no_overlap_ij})

    # Convert centers to a parameter vector for optimization
    x0 = radii.copy()

    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints)

    radii_opt = result.x
    centers_opt = centers.copy()
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
