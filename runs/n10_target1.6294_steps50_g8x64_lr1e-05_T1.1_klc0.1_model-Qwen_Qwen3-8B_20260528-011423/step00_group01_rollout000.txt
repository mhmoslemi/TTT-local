import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i / 3, j / 3] for i in range(4) for j in range(4)])
    centers = centers[:10]
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii, centers):
        return np.all(centers[:, 0] >= radii) and np.all(centers[:, 0] <= 1 - radii) and \
               np.all(centers[:, 1] >= radii) and np.all(centers[:, 1] <= 1 - radii)

    def constraint_no_overlap(radii, centers):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_and_constraints(radii):
        # Objective: maximize the sum of radii (minimize negative sum)
        obj = -np.sum(radii)
        # Constraints
        constraint1 = constraint_center_inside(radii, centers)
        constraint2 = constraint_no_overlap(radii, centers)
        return obj, constraint1, constraint2

    # Optimization
    result = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        bounds=[(0, 1 - centers[i, 0]) for i in range(10)] + [(0, 1 - centers[i, 1]) for i in range(10)],
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraint_center_inside(x, centers)},
            {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x, centers)}
        ]
    )

    # Extract result
    radii_opt = result.x
    centers_opt = centers
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
