import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1  # Initial radius guess

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x[10:])  # Negative because we minimize

    # Bounds: each center (x, y) and radius (r) has bounds
    bounds = []
    for i in range(10):
        x = x[2 * i]
        y = x[2 * i + 1]
        r = x[2 * i + 2]
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
        bounds.append((0, 1))  # r

    # Constraints: circle is inside the square
    def square_bound(x):
        for i in range(10):
            x_i = x[2 * i]
            y_i = x[2 * i + 1]
            r_i = x[2 * i + 2]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return 1
        return 0

    # Constraint: no two circles overlap
    def no_overlap(x):
        for i in range(10):
            for j in range(i + 1, 10):
                x_i = x[2 * i]
                y_i = x[2 * i + 1]
                r_i = x[2 * i + 2]
                x_j = x[2 * j]
                y_j = x[2 * j + 1]
                r_j = x[2 * j + 2]
                dist = np.hypot(x_i - x_j, y_i - y_j)
                if dist < r_i + r_j:
                    return 1
        return 0

    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': square_bound},
        {'type': 'ineq', 'fun': no_overlap}
    ]

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP')

    # Extract result
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
