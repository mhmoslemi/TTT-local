import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)

    # Constraints
    def constraint_center_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.sum((centers_flat[:, 0] - radii_flat) >= 0) + np.sum((1 - centers_flat[:, 0] - radii_flat) >= 0) + \
               np.sum((centers_flat[:, 1] - radii_flat) >= 0) + np.sum((1 - centers_flat[:, 1] - radii_flat) >= 0)

    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distances = np.linalg.norm(centers_flat[:, np.newaxis] - centers_flat[np.newaxis, :], axis=2)
        return np.sum(distances >= radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Bounds: radii >= 0 and centers in [0,1]
    bounds = [(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]

    # Optimize
    result = minimize(objective, np.concatenate([centers.ravel(), radii]), method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
