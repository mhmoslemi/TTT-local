import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.array([[i * 0.2, j * 0.2] for i in range(5) for j in range(2)])
    centers = np.vstack([centers, [0.2, 0.6], [0.8, 0.6], [0.2, 0.8], [0.8, 0.8]])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return False
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define the optimization problem
    bounds = [(0, 1) for _ in range(10)]
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=({'type': 'ineq', 'fun': constraints}),
        options={'disp': False}
    )

    # Extract final result
    final_radii = result.x
    sum_radii = np.sum(final_radii)
    centers_final = centers.copy()

    return (centers_final, final_radii, sum_radii)
