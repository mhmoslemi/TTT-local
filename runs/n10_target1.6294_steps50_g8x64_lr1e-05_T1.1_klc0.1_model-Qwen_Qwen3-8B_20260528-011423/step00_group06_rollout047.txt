import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (random points in the square)
    centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    radii = np.ones(10) * 0.1  # Initial radius guess

    # Objective function
    def objective(x):
        # Reshape x to (10, 4) -> (10, 2) centers and (10,) radii
        centers_flat = x[:10*2].reshape(10, 2)
        radii_flat = x[10*2:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints
    def constraint_1(x):
        # All circles are inside the unit square
        centers_flat = x[:10*2].reshape(10, 2)
        radii_flat = x[10*2:]
        return np.sum((centers_flat[:, 0] - radii_flat) > 0) + np.sum((1 - centers_flat[:, 0] - radii_flat) > 0) + \
               np.sum((centers_flat[:, 1] - radii_flat) > 0) + np.sum((1 - centers_flat[:, 1] - radii_flat) > 0)

    def constraint_2(x):
        # No two circles overlap
        centers_flat = x[:10*2].reshape(10, 2)
        radii_flat = x[10*2:]
        distances = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return np.sum(distances >= radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Bounds for all variables: x and y coordinates between 0.1 and 0.9, radii >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.1, 0.9), (0.1, 0.9), (0, np.inf)])
    
    # Initial guess flattened
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_1},
        {'type': 'ineq', 'fun': constraint_2}
    ]

    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract the result
    centers_result = result.x[:10*2].reshape(10, 2)
    radii_result = result.x[10*2:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
