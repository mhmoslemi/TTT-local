import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles evenly spaced
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint: circles must be inside the unit square
        for i in range(10):
            if radii[i] > 0:
                if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                    return False
                if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                    return False
        # Constraint: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist_sq = dx * dx + dy * dy
                if dist_sq < (radii[i] + radii[j]) ** 2:
                    return False
        return True

    # Optimization with bounds and constraints
    def objective_func(x):
        nonlocal centers
        radii = x
        return -np.sum(radii)

    bounds = [(0.0, 0.5) for _ in range(10)]
    result = minimize(objective_func, radii, bounds=bounds, constraints={"type": "ineq", "fun": constraints})

    if result.success:
        radii_opt = result.x
        sum_radii = np.sum(radii_opt)
        # Re-check constraints for safety
        for i in range(10):
            if radii_opt[i] > 0:
                if centers[i, 0] - radii_opt[i] < 0 or centers[i, 0] + radii_opt[i] > 1:
                    sum_radii = 0.0
                    break
                if centers[i, 1] - radii_opt[i] < 0 or centers[i, 1] + radii_opt[i] > 1:
                    sum_radii = 0.0
                    break
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist_sq = dx * dx + dy * dy
                if dist_sq < (radii_opt[i] + radii_opt[j]) ** 2:
                    sum_radii = 0.0
                    break
            if sum_radii == 0.0:
                break
        return (centers, radii_opt, sum_radii)
    else:
        return (centers, radii, 0.0)
