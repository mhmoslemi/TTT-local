import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.array([[i * 0.25, j * 0.25] for j in range(2) for i in range(5)])
    radii = np.array([0.1] * 10)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: All circles are inside the square
        for i in range(10):
            r = radii[i]
            if r > 1 - np.min(centers[i, :]):
                return False
            if r > 1 - np.max(centers[i, :]):
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_func(x):
        radii = x
        if not constraints(radii):
            return np.inf  # Penalize invalid solutions
        return -np.sum(radii)

    def bounds_func(x):
        return [(0, 1) for _ in range(10)]

    # Optimization
    x0 = radii
    result = minimize(objective_func, x0, bounds=bounds_func, method='L-BFGS-B')
    radii_opt = result.x
    sum_radii = -result.fun

    # Ensure constraints are satisfied
    if not constraints(radii_opt):
        sum_radii = 1.6294  # Target value as fallback if constraints are not met

    centers_opt = centers.copy()  # Keep initial positions, could be improved later

    return (centers_opt, radii_opt, sum_radii)
