import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: distribute circles in a grid
    centers = np.array([[i / 3 + 0.1, j / 3 + 0.1] for i in range(3) for j in range(3)])
    centers = np.vstack((centers, [0.5, 0.5], [0.5, 0.1], [0.5, 0.9], [0.1, 0.5], [0.9, 0.5]))
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - 2 * np.min([centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1]]):
                return False
        # Ensure no overlaps
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1) for _ in range(10)],
        constraints=[{'type': 'ineq', 'fun': constraints}],
        options={'disp': False, 'maxiter': 1000}
    )

    # Extract results
    centers = np.array([[centers[i, 0], centers[i, 1]] for i in range(10)])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
