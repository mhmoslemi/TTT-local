import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    # Objective function to minimize: negative sum of radii (maximize)
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)

    # Constraints
    def constraint1(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Each circle must be within the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1
        return 0

    def constraint2(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1
        return 0

    # Bounds
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y for each center
    for _ in range(10):
        bounds.extend([(0.0, 0.5)])  # radii must be <= 0.5

    # Optimization
    initial_guess = np.concatenate([centers.flatten(), radii])
    result = minimize(objective, initial_guess, method='SLSQP', bounds=bounds,
                      constraints=[{'type': 'ineq', 'fun': constraint1},
                                   {'type': 'ineq', 'fun': constraint2}],
                      tol=1e-8, options={'disp': False})

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
