import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - 2 * centers[i, 0] or radii[i] > 1 - 2 * centers[i, 1]:
                return False
        # Ensure no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimize
    result = minimize(objective, radii, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints}, bounds=[(0, 1.0)] * 10)

    # Extract results
    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = np.sum(radii_opt)

    # Ensure constraints are satisfied (just in case)
    if not constraints(radii_opt):
        raise ValueError("Optimization failed to satisfy constraints")

    return (centers_opt, radii_opt, sum_radii_opt)
