import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles on a grid
    centers = np.array([[i * 0.25 + 0.1, j * 0.25 + 0.1] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective(x):
        # Flatten x into centers and radii
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        
        # Objective: maximize the sum of radii (minimize negative)
        return -np.sum(radii_flat)
    
    def constraints(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        
        # Constraint 1: Each circle is fully inside the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.hypot(dx, dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 1e-8:
                    return np.inf
        
        # Constraint 3: All radii are non-negative
        for r in radii_flat:
            if r < 0:
                return np.inf
        
        return 0
    
    # Bounds for x and y coordinates (each between 0 and 1)
    x_bounds = [(0, 1) for _ in range(20)]
    # Bounds for radii (each between 0 and 0.5)
    r_bounds = [(0, 0.5) for _ in range(10)]
    
    # Combine bounds for all variables
    bounds = x_bounds + r_bounds
    
    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                      method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    
    centers_final = result.x[:20].reshape(10, 2)
    radii_final = result.x[20:]
    sum_radii = np.sum(radii_final)
    
    return (centers_final, radii_final, sum_radii)
