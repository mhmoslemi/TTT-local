import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 10 circles arranged in a grid near the center
    centers = np.array([[0.2 + i*0.2, 0.2 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = 0.2 + (i // 2) * 0.2
            centers[i, 1] = 0.2 + (i % 2) * 0.2
        radii = radii.reshape(-1)
        # Ensure each circle is inside the unit square
        for i, r in enumerate(radii):
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return np.inf
        # Ensure no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Optimization
    bounds = [(0.01, 0.5) for _ in range(10)]  # Radii must be in this range
    result = minimize(objective, radii, bounds=bounds, constraints={"type": "eq", "fun": constraints})

    # Extract the result
    centers = np.array([[0.2 + i*0.2, 0.2 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
