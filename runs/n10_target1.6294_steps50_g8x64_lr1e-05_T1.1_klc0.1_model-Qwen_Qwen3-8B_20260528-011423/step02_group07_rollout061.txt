import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraint_function(radii_vec):
        # Check if all radii are non-negative
        for r in radii_vec:
            if r < 0:
                return np.inf
        # Check if all circles are within the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii_vec[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
        # Check if all circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return np.inf
        return 0

    # Optimization
    bounds = [(0, 1) for _ in range(10)]
    constraints = {'type': 'ineq', 'fun': constraint_function}
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    return (centers, result.x, -result.fun)
