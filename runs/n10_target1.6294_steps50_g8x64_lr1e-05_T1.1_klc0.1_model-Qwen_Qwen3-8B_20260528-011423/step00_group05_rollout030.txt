import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.zeros(10) + 0.05

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii_flat):
        return -np.sum(radii_flat)

    # Constraints
    def constraint_center_inside(radii_flat):
        # Flatten radii and centers
        radii = radii_flat.reshape(-1)
        centers_flat = centers.reshape(-1)
        # Check if all centers are within [r, 1-r] for their respective radii
        return np.all(centers_flat >= radii) & np.all(centers_flat <= 1 - radii)

    def constraint_no_overlap(radii_flat):
        # Flatten radii and centers
        radii = radii_flat.reshape(-1)
        centers_flat = centers.reshape(-1)
        # Check if all pairs of circles satisfy distance >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                idx_i = i * 2
                idx_j = j * 2
                dx = centers_flat[idx_i] - centers_flat[idx_j]
                dy = centers_flat[idx_i + 1] - centers_flat[idx_j + 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    # Define constraints for the optimization
    cons = [
        {'type': 'ineq', 'fun': lambda x: constraint_center_inside(x)},
        {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x)}
    ]

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(10)]

    # Optimization: minimize negative sum of radii
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'maxiter': 1000}
    )

    # Extract and format results
    radii = result.x
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
