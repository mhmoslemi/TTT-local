import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: place 10 circles in a grid with some spacing
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4 + 0.1
        centers[i, 1] = (i // 5) / 4 + 0.1
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints_func(radii, centers):
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                dist_min = radii[i] + radii[j]
                if dist < dist_min - 1e-8:
                    # Overlap detected, add constraint to prevent it
                    constraints.append((dist - dist_min))
        for i in range(10):
            r = radii[i]
            if r > 1 - 1e-8:
                constraints.append(r - (1 - 1e-8))
        return constraints

    def wrapped_constraints(radii):
        return constraints_func(radii, centers)

    bounds = [(0, 1 - 1e-8) for _ in range(10)]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': wrapped_constraints})

    if not result.success:
        raise RuntimeError("Optimization failed")

    radii = result.x
    sum_radii = np.sum(radii)

    # Optional: Use a more refined optimization step with fixed centers
    def objective_fixed_centers(radii):
        return -np.sum(radii)

    def constraints_fixed_centers(radii):
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                dist_min = radii[i] + radii[j]
                if dist < dist_min - 1e-8:
                    constraints.append((dist - dist_min))
        for i in range(10):
            r = radii[i]
            if r > 1 - 1e-8:
                constraints.append(r - (1 - 1e-8))
        return constraints

    result_fixed = minimize(objective_fixed_centers, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints_fixed_centers})

    if not result_fixed.success:
        raise RuntimeError("Optimization failed after fixing centers")

    radii = result_fixed.x
    sum_radii = np.sum(radii)

    # Final check for constraints
    for i in range(10):
        if radii[i] > 1 - 1e-8:
            raise ValueError("Circle exceeds unit square")
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < radii[i] + radii[j] - 1e-8:
                raise ValueError("Circles overlap")

    return (centers, radii, sum_radii)
