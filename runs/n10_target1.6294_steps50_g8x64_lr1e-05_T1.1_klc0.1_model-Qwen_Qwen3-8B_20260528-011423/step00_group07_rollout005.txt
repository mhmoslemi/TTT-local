import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid centered in the square
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(centers, radii):
        # Constraint 1: All circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return False
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def constraint_func(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x.reshape(10, 2)
        radii = x[2::2]
        if not constraints(centers, radii):
            return np.inf
        return 0

    def bounds_func(x):
        # Bounds are [0, 1] for x and y, and [0, 0.5] for radii (since r <= 0.5)
        bounds = []
        for i in range(10):
            bounds.append((0.0, 1.0))  # x
            bounds.append((0.0, 1.0))  # y
            bounds.append((0.0, 0.5))  # r
        return bounds

    # Flatten centers and radii for optimization
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Set up optimization
    result = minimize(
        fun=constraint_func,
        x0=x0,
        method='SLSQP',
        bounds=bounds_func(x0),
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    # Extract result
    x_res = result.x
    centers_res = x_res[:20].reshape(10, 2)
    radii_res = x_res[2::2]
    sum_radii_res = np.sum(radii_res)

    # Return the result
    return (centers_res, radii_res, sum_radii_res)
