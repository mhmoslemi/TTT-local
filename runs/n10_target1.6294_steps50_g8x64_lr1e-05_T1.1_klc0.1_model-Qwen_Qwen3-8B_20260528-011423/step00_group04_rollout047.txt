import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        x_flat = x.reshape((10, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]

        # Objective: maximize sum of radii (minimize negative sum)
        sum_radii = -np.sum(radii)
        return sum_radii

    def constraints(x):
        x_flat = x.reshape((10, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]

        # Constraint 1: Each circle is inside the unit square
        inside = np.all((radii <= centers) & (centers <= 1 - radii), axis=1)
        if not np.all(inside):
            return np.inf

        # Constraint 2: No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        overlap = dists < radii[:, np.newaxis] + radii[np.newaxis, :]
        if np.any(overlap):
            return np.inf

        return 0

    # Optimization setup
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0 - 0.0)])  # x, y, r

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate((centers.reshape(-1), radii)),
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6,
        options={'maxiter': 1000}
    )

    x_opt = result.x
    centers_opt = x_opt[:60].reshape((10, 2))
    radii_opt = x_opt[60:].reshape((10,))

    # Ensure constraints are satisfied (sanity check)
    assert np.all((radii_opt <= centers_opt) & (centers_opt <= 1 - radii_opt))
    dists = np.sqrt(np.sum((centers_opt[:, np.newaxis, :] - centers_opt[np.newaxis, :, :]) ** 2, axis=2))
    assert np.all(dists >= radii_opt[:, np.newaxis] + radii_opt[np.newaxis, :])

    return (centers_opt, radii_opt, -result.fun)
