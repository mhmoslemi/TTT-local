import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.08)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints: all circles must be inside the unit square
    def constraint_inside(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            for coord in [0, 1]:
                if not (radii[i] <= centers[i, coord] <= 1 - radii[i]):
                    return -1
        return 0

    # Constraints: no two circles overlap
    def constraint_overlap(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        dists = pdist(centers)
        for i in range(len(dists)):
            if dists[i] < radii[i] + radii[j]:
                return -1
        return 0

    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds: radii must be non-negative
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimize
    result = minimize(
        objective,
        x0=np.concatenate([centers.ravel(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=cons
    )

    # Extract result
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
