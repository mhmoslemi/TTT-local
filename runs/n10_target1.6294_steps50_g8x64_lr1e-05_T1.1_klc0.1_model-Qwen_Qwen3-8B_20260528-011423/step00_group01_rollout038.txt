import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess: place 10 circles in a square grid
    centers = np.array([[i / 3 + 0.1, j / 3 + 0.1] for i in range(3) for j in range(3)])
    radii = np.full(10, 0.1)

    def objective_function(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - 0.2:  # allow some margin
                return False
        # Ensure circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    # Optimization bounds: radii must be non-negative and <= 1 - 0.2
    bounds = [(0, 1 - 0.2) for _ in range(10)]

    # Set up optimization
    result = minimize(
        objective_function,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=({'type': 'ineq', 'fun': constraints}),
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract result
    radii_opt = result.x
    # Check constraints to ensure valid solution
    for i in range(10):
        if radii_opt[i] < 0:
            radii_opt[i] = 0
        if radii_opt[i] > 1 - 0.2:
            radii_opt[i] = 1 - 0.2
    sum_radii = np.sum(radii_opt)

    # Return centers, radii, and sum
    return (centers, radii_opt, sum_radii)
