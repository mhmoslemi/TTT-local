import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(x):
        # x is a flat array: [x0, y0, r0, x1, y1, r1, ..., x9, y9, r9]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize it

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        constraints = []
        # Constraint: each circle is inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            constraints.append(x_i - r_i)
            constraints.append(1 - x_i - r_i)
            constraints.append(y_i - r_i)
            constraints.append(1 - y_i - r_i)
        # Constraint: no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.hypot(dx, dy)
                sum_r = radii_flat[i] + radii_flat[j]
                constraints.append(dist - sum_r)
        return constraints

    # Flatten initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up optimization problem
    res = minimize(
        objective,
        x0,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]
    )

    # Extract solution
    x_sol = res.x
    centers_sol = x_sol[:60].reshape(10, 2)
    radii_sol = x_sol[60:]
    sum_radii_sol = -res.fun  # Convert back to positive sum

    return (centers_sol, radii_sol, sum_radii_sol)
