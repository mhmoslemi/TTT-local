import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (random positions)
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii):
        return np.all((radii <= centers[:, 0] - 0.1) & (radii <= 1 - centers[:, 0] - 0.1) &
                      (radii <= centers[:, 1] - 0.1) & (radii <= 1 - centers[:, 1] - 0.1))

    def constraint_no_overlap(radii):
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.all(dist >= radii[:, np.newaxis] + radii[np.newaxis, :])

    bounds = [(0, 0.5) for _ in range(10)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_center_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}
        ],
        options={'ftol': 1e-6, 'disp': False}
    )

    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
