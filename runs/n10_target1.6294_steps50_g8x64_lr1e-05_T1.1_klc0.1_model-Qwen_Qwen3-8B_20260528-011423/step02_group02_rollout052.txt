import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: each circle must be fully inside the unit square
    def constraint_square(radii):
        return np.sum(radii)

    # Constraint: no two circles overlap
    def constraint_overlap(radii):
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        k = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[k] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                k += 1
        return dists

    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimization setup
    initial_guess = np.ones(10) * 0.1
    result = minimize(
        objective,
        initial_guess,
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_square},
            {'type': 'ineq', 'fun': constraint_overlap}
        ],
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    radii = result.x
    centers_result = centers.copy()
    sum_radii = np.sum(radii)

    # Final check: ensure all circles are within the square and non-overlapping
    for i in range(10):
        x, y = centers_result[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            warnings.warn("One or more circles are not fully inside the square.", UserWarning)

    return (centers_result, radii, sum_radii)
