import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

        return -np.sum(radii)
    
    def constraints(centers, radii, box_size=1.0):
        # Constraint 1: Circles are fully inside the square
        in_box = np.all((radii <= centers) & (centers <= box_size - radii), axis=1)
        if not np.all(in_box):
            return -np.inf
        
        # Constraint 2: No two circles overlap
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        overlap = np.any(dists < radii_sum - 1e-6)
        if overlap:
            return -np.inf
        
        return np.sum(radii)
    
    def objective_and_constraints(x):
        radii = x
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        if not np.all((radii <= 0.5) & (radii >= 0)):
            return -np.inf

        in_box = np.all((radii <= centers) & (centers <= 1 - radii), axis=1)

            return -np.inf

        if overlap:

        return -np.sum(radii)
    
    # Optimization
    result = minimize(objective_and_constraints, radii, method='L-BFGS-B', bounds=[(0, 0.5)]*10)
