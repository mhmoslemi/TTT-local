import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: randomly distribute centers in the square and assign equal radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Combine radii into flat array for optimization
    def objective_flat(x):
        return -np.sum(x)

    def constraint_flat(x):
        for i in range(10):
            if x[i] < 0:
                return np.inf
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < x[i] + x[j]:
                    return np.inf
        return 0

    # Optimization setup
    bounds = [(0, 0.5) for _ in range(10)]
    result = minimize(objective_flat, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_flat}, tol=1e-6)

    # Extract results
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)

    return (centers, radii_opt, sum_radii)
