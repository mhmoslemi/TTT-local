import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1]*10)

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(centers, radii):
        # Constraint: r_i <= x_i - 0 and 1 - x_i - r_i >= 0
        # Similarly for y_i
        left = centers[:, 0] - radii
        right = 1 - centers[:, 0] - radii
        bottom = centers[:, 1] - radii
        top = 1 - centers[:, 1] - radii
        return np.concatenate([left, right, bottom, top])

    def constraint_no_overlap(centers, radii):
        # Constraint: distance between centers >= sum of radii
        n = len(radii)
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return dists - (radii.reshape(-1, 1) + radii.reshape(1, -1))

    # Optimization
    bounds = [(0, 1) for _ in range(20)]  # 10 circles * 2 coordinates
    cons = []

    # Circle inside unit square
    cons.append({'type': 'ineq', 'fun': lambda r: constraint_circle_inside(centers, r)})

    # No overlap between circles
    cons.append({'type': 'ineq', 'fun': lambda r: constraint_no_overlap(centers, r)})

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-7)

    # Post-processing
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    centers = centers + np.random.normal(0, 0.05, centers.shape)  # Perturb initial guess

    # Re-optimizing with perturbed centers
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-7)

    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    centers = centers + np.random.normal(0, 0.03, centers.shape)  # Perturb initial guess slightly more

    # Final optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-7)

    return (centers, result.x, -result.fun)
