import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    # Start with equally spaced points
    centers = np.array([[i/9 + 0.05, j/9 + 0.05] for i in range(10) for j in range(10)])
    centers = centers[:10]  # Select first 10 points
    radii = np.array([0.05] * 10)

    def objective(x):
        # x is flattened array of [x1, y1, r1, x2, y2, r2, ... x10, y10, r10]
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        sum_radii = np.sum(radii)
        return -sum_radii  # minimize -sum_radii to maximize sum_radii

    def constraints(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Constraint 1: Each circle is inside the square
        for i in range(10):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Constraint 2: Circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Bounds: x and y between 0 and 1, radii between 0 and 0.5
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])
    # Optimization
    res = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=({'type': 'ineq', 'fun': constraints} if constraints(x0) else None))
    if not res.success or not constraints(res.x):
        # If optimization fails, return initial guess
        return (centers, radii, np.sum(radii))
    centers_opt = res.x[:60].reshape(10, 2)
    radii_opt = res.x[60:]
    sum_radii_opt = np.sum(radii_opt)
    return (centers_opt, radii_opt, sum_radii_opt)
