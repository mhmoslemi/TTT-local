import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for circle centers (randomly placed within the square)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Starting with a small radius

    def objective_function(params):
        # Flatten parameters to centers and radii
        centers_flat = params[:20].reshape(10, 2)
        radii_flat = params[20:]
        # Ensure radii are non-negative
        radii_flat = np.maximum(radii_flat, 0)
        # Enforce that circles are within the unit square
        # x - r >= 0 --> r <= x; 1 - r >= x --> r <= 1 - x; similarly for y
        # We minimize the negative sum of radii to maximize it
        sum_radii = -np.sum(radii_flat)
        # Constraints: circles are fully within the square
        for i in range(10):
            r = radii_flat[i]
            if centers_flat[i, 0] - r < 0 or centers_flat[i, 0] + r > 1:
                sum_radii -= 100  # Penalize large violations
            if centers_flat[i, 1] - r < 0 or centers_flat[i, 1] + r > 1:
                sum_radii -= 100  # Penalize large violations
        # Constraints: circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    sum_radii -= 100  # Penalize overlap
        return sum_radii

    def constraints_function(params):
        centers_flat = params[:20].reshape(10, 2)
        radii_flat = params[20:]
        # Enforce that circles are fully within the unit square
        for i in range(10):
            r = radii_flat[i]
            if centers_flat[i, 0] - r < 0 or centers_flat[i, 0] + r > 1:
                return np.inf
            if centers_flat[i, 1] - r < 0 or centers_flat[i, 1] + r > 1:
                return np.inf
        # Enforce that circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0

    # Optimization setup
    bounds = []
    for _ in range(20):  # 10 centers * 2 coordinates
        bounds.append((0, 1))
    for _ in range(10):  # 10 radii
        bounds.append((0, 1))

    # Minimize the negative of the sum of radii
    result = minimize(
        objective_function,
        x0=np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints_function},
        options={'ftol': 1e-6, 'maxiter': 500}
    )

    # Extract result
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = -objective_function(result.x)

    # Return the result
    return (centers_opt, radii_opt, sum_radii)
