```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles at a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Maximize sum by minimizing negative sum

    def constraints(centers, radii):
        # Constraint: circles must be inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii), axis=1)
        if not np.all(inside):
            return np.inf
        # Constraint: no overlapping between circles
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        overlap = np.sum((dists < radii[:, np.newaxis] + radii[np.newaxis, :]) * (dists > 0))
        if overlap > 0:
            return np.inf
        return 0

    def objective_and_constraints(x):
        # x is flattened array: [x1, y1, r1, x2, y2, r2, ...]
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        obj_val = -np.sum(radii)
        constraint_val = constraints(centers, radii)
        return obj_val, constraint_val

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Optimization setup
    bounds = []
    for i in range(10):
        bounds += [(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)]  # x, y, r

    # Use SLSQP method for constrained optimization
    result = minimize(lambda x: objective_and_constraints(x)[0],
                      x0,
                      method='SLSQP',
                      bounds=bounds,
                      constraints=[{'type': 'ineq', 'fun': lambda x: objective_and_constraints(x)[1]}],
                      tol=1e-6,
                      options={'maxiter': 1000})

    # Extract result
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii_result = -objective(radii_result)

    # Check constraints
    assert np.all(radii_result >= 0)
    assert np.all((radii_result <= centers_result[:, 0]) & (centers_result[:, 0] <= 1 - radii_result) &
                  (radii_result <= centers_result[:, 1]) & (centers_result[:, 1] <= 1 - radii_result))
    dists = np.sqrt(((centers_result[:, np.newaxis, 0] - centers_result[np.newaxis, :, 0])**2 +
                     (centers_result[:, np.newaxis, 1] - centers_result[np.newaxis, :, 1])**2))
    assert np.all(dists >= radii_result[:, np.newaxis] + radii_result[np.newaxis, :])

    return (centers_result, radii_result, sum_radii_result)
```
