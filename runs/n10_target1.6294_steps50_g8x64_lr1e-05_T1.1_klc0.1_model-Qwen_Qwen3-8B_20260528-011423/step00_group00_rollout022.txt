import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints_func(radii, centers):
        n = len(radii)
        constraints = []
        # Constraint: radii >= 0
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        # Constraint: circles are inside the unit square
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (1 - r[i] - centers[i, 0])})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (1 - r[i] - centers[i, 1])})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (r[i] - centers[i, 0])})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (r[i] - centers[i, 1])})
        # Constraint: no overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                def dist_constraint(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.hypot(dx, dy)
                    return dist - r[i] - r[j]
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: dist_constraint(r)})
        return constraints

    bounds = [(0, 1) for _ in range(10)]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints_func(radii, centers))

    centers = np.array(result.x).reshape(-1, 2)
    sum_radii = -result.fun
    return centers, result.x, sum_radii
