import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.ones(10) * 0.05

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(x):
        # x is a flat array of centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -np.inf
        return 0

    def constraint_no_overlap(x):
        # x is a flat array of centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1)] * 2)  # x and y coordinates for each center
        bounds.extend([(0, 0.5)] * 2)  # radii (maximum radius is 0.5)

    # Initial guess flattened
    x0 = centers.reshape(-1).tolist()
    x0.extend(radii.tolist())

    # Optimization with constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-8)

    # Extract result
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:]
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
