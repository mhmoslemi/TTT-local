import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        try:
            from scipy.optimize import minimize
        except:
            raise RuntimeError("scipy.optimize is required for this function")

    # Initial guess: place circles in a grid pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        row = i // 3
        col = i % 3
        centers[i, 0] = 0.1 + col * 0.25
        centers[i, 1] = 0.1 + row * 0.25

    # Define the objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Define the constraints
    def constraint1(radii):
        # Each circle must be fully inside the unit square
        r = radii
        return np.minimum(r, 1 - r)  # r <= x <= 1 - r and r <= y <= 1 - r

    def constraint2(radii):
        # No two circles overlap
        radii = radii.reshape(-1, 1)
        dist = np.linalg.norm(centers[:, np.newaxis, :] - centers[np.newaxis, :, :], axis=2)
        return dist - radii - radii[np.newaxis, :]

    # Optimization problem
    bounds = [(0, 1) for _ in range(10)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint1(r)},
        {'type': 'ineq', 'fun': lambda r: np.min(constraint2(r))}
    ]

    result = minimize(objective, centers.flatten(), bounds=bounds, constraints=cons, method='SLSQP')

    # Extract results
    radii = result.x
    centers = np.reshape(centers, (10, 2))
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
