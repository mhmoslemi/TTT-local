import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: randomly placed circles
    np.random.seed(0)
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.random.uniform(0.05, 0.1, 10)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(centers, radii):
        # Constraint 1: Circles are inside the unit square
        inside = np.all((radii <= centers) & (centers <= 1 - radii), axis=1)
        if not np.all(inside):
            return np.inf
        
        # Constraint 2: No two circles overlap
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        min_dist = np.min(dists[np.triu_indices(10, k=1)])
        if min_dist < np.min(radii) + 1e-8:
            return np.inf

        return 0

    def packing_function(x):
        # x contains all variables: centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return constraints(centers, radii)

    # Optimize
    x0 = np.concatenate((centers.flatten(), radii))
    result = minimize(objective, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': packing_function})
    x_opt = result.x
    centers_opt = x_opt[:60].reshape(10, 2)
    radii_opt = x_opt[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
