import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: circles arranged in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are fully inside the square
        for r in radii:
            if r < 0:
                return False
        return True

    def constraint_func(x):
        # x is a flat array of centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        if not constraints(radii):
            return np.inf
        # Check distance between centers
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Flatten initial guess
    x0 = np.concatenate([centers.flatten(), radii])
    res = minimize(constraint_func, x0, method='L-BFGS-B', bounds=[None]*30)

    # Extract result
    centers_res = res.x[:20].reshape(10, 2)
    radii_res = res.x[20:]
    sum_radii = -res.fun

    return (centers_res, radii_res, sum_radii)
