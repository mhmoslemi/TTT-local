import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.zeros((10, 2))
    for i in range(10):
        x = i // 2 / 4.5 + 0.05
        y = i % 2 / 4.5 + 0.05
        centers[i] = [x, y]
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii, centers):
        # Ensure circles are inside the square
        return np.array([r - (1 - r) for r in radii])

    def constraint_no_overlap(radii, centers):
        # Ensure circles do not overlap
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization setup
    bounds = [(0, 0.5) for _ in range(10)]
    cons = [
        {'type': 'ineq', 'fun': lambda x: constraint_center_inside(x, centers)},
        {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x, centers)}
    ]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Update centers based on optimized radii
    centers = np.zeros((10, 2))
    for i in range(10):
        x = i // 2 / 4.5 + 0.05
        y = i % 2 / 4.5 + 0.05
        centers[i] = [x, y]

    # Final output
    return (centers, result.x, np.sum(result.x))
