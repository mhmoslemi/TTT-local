import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        x = (i % 5) / 4.0 + 0.1
        y = (i // 5) / 4.0 + 0.1
        centers[i] = [x, y]
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Ensure no overlaps between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimize
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=[(0, 1)] * 10,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        method='SLSQP'
    )

    # Extract result
    optimized_radii = result.x
    sum_radii = -result.fun
    return (centers, optimized_radii, sum_radii)
