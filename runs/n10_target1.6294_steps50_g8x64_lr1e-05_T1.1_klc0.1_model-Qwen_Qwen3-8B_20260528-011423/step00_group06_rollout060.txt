import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles evenly in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Constraint 1: Each circle is inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1:
                return np.inf
        # Constraint 2: Distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0
    
    # Optimization
    res = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}])
    radii_opt = res.x
    sum_radii = -res.fun
    
    # Validate constraints
    for i in range(10):
        if radii_opt[i] < 0 or radii_opt[i] > 1:
            print(f"Invalid radius: {radii_opt[i]}")
            return None, None, 0.0
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < radii_opt[i] + radii_opt[j]:
                print(f"Overlap between circle {i} and {j}")
                return None, None, 0.0
    
    return centers, radii_opt, sum_radii
