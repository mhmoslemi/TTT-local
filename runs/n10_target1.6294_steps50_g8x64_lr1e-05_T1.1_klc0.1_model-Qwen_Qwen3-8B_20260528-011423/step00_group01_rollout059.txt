import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.array([
        [0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)
    ])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([
            [0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)
        ])
        if np.any(radii < 0):
            return np.inf
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        return 0

    bounds = [(0, 0.5) for _ in range(10)]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={"type": "eq", "fun": constraints})

    centers = np.array([
        [0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)
    ])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
