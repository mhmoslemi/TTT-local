import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for centers (random positions inside the unit square)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective_function(x):
        # x is flattened array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        
        # Objective: maximize sum of radii (minimize negative sum)
        sum_radii = -np.sum(radii_flat)
        
        # Constraints: all circles are inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        
        # Constraints: no two circles overlap
        dists = cdist(centers_flat, centers_flat)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii_flat[i] + radii_flat[j]:
                    return np.inf
        
        return sum_radii

    def bounds():
        # Bounds for centers (x, y) in [0, 1] and radii >= 0
        bounds = []
        for i in range(10):
            bounds.append((0.0, 1.0))  # x_i
            bounds.append((0.0, 1.0))  # y_i
            bounds.append((0.0, 1.0))  # r_i
        return bounds

    # Flatten initial guess
    x0 = np.concatenate([centers.reshape(10 * 2), radii])

    # Optimization
    result = minimize(objective_function, x0, bounds=bounds(), method='SLSQP', tol=1e-8)

    # Extract result
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:].reshape(10)
    sum_radii_result = -np.sum(radii_result)

    return (centers_result, radii_result, sum_radii_result)
