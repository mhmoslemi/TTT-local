import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[x, y] for x, y in itertools.product(np.linspace(0.1, 0.9, 5), repeat=2)])
    radii = np.full(10, 0.1)
    
    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x[2:])  # Negative for minimization

    # Bounds for centers: each center must be at least radius away from the boundary
    def bounds_func(x):
        bounds = []
        for i in range(10):
            x_i = x[2*i]
            y_i = x[2*i+1]
            r_i = x[2*i+2]
            bounds.append((0 + r_i, 1 - r_i))  # x
            bounds.append((0 + r_i, 1 - r_i))  # y
            bounds.append((0, 1))              # r
        return bounds

    # Constraints: distance between centers >= sum of radii
    def constraint_func(x):
        centers = np.reshape(x[:20], (10, 2))
        radii = x[20:]
        constraints = []
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    # Prepare initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up constraints
    cons = {'type': 'ineq', 'fun': constraint_func}

    # Set up bounds
    bounds = bounds_func(x0)

    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    centers_opt = np.reshape(result.x[:20], (10, 2))
    radii_opt = result.x[20:]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
