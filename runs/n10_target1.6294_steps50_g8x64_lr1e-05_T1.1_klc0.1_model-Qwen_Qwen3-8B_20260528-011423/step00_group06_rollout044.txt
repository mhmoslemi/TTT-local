import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.zeros((10, 2))
    for i in range(10):
        x = i // 2 / 3 + 0.5
        y = i % 2 / 2 + 0.25
        centers[i] = [x, y]
    radii = np.full(10, 0.1)

    # Objective function: minimize negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: All circles must be inside the square
    def constraint_inside(radii):
        return np.sum((radii - 0.01) < np.min(np.vstack((centers[:, 0] - radii, 1 - centers[:, 0] - radii,
                                                          centers[:, 1] - radii, 1 - centers[:, 1] - radii)), axis=0))

    # Constraint 2: Circles must not overlap
    def constraint_overlapping(radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=-1))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimizer settings
    bounds = [(0, 1) for _ in range(10)]

    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlapping(r)}
    ]

    # Optimize
    result = minimize(objective, radii, bounds=bounds, constraints=cons, method='SLSQP', options={'ftol': 1e-9, 'maxiter': 500})

    # Extract result
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
