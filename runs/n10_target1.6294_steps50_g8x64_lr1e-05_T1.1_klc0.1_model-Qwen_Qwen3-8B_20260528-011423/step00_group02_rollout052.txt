```python
import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    # Start with a grid of 10 points and small equal radii
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)

    def objective_function(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        sum_radii = np.sum(radii)
        return -sum_radii  # We minimize negative sum to maximize it

    def constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Constraint 1: Each circle is within the unit square
        for i in range(10):
            x_i, y_i, r_i = centers[i][0], centers[i][1], radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    # Set up bounds for optimization
    bounds = []
    for i in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])  # x, y, r

    # Use a local optimizer with initial guess
    result = minimize(
        objective_function,
        x0=np.concatenate([initial_centers.flatten(), initial_radii]),
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
        tol=1e-6,
        options={"maxiter": 1000, "ftol": 1e-9, "eps": 1e-9}
    )

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun  # Since we minimized negative sum

    return (centers, radii, sum_radii)
```
