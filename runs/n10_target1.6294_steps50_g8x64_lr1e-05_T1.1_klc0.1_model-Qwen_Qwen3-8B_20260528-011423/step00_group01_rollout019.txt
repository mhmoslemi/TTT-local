import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints_func(x):
        # x is flattened array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        sum_radii = np.sum(radii)
        # Boundary constraints
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        # Pairwise distance constraints
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -sum_radii  # Maximize sum_radii

    # Prepare initial guess
    x0 = np.concatenate((centers.reshape(20), radii))
    # Optimization options
    bounds = []
    for _ in range(10):
        bounds.extend([(0.1, 0.9) for _ in range(2)])  # x, y bounds
        bounds.extend([(0.0, 0.5) for _ in range(10)])  # radii bounds (upper bound to avoid overlap)
    # Run optimization
    result = minimize(constraints_func, x0, bounds=bounds, method='L-BFGS-B')
    # Extract solution
    best_x = result.x
    centers_best = best_x[:60].reshape(10, 2)
    radii_best = best_x[60:]
    sum_radii_best = -result.fun  # Since we minimized -sum_radii
    return (centers_best, radii_best, sum_radii_best)
