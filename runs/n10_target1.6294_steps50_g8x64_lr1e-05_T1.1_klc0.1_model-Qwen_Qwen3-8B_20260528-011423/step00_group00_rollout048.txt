import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_square(radii):
        for i in range(10):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return -1000
        return 0

    def constraint_overlap(radii):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1000
        return 0

    # Optimization setup
    bounds = [(0, 1) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_square},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    result = minimize(objective, radii, method='SLSQP',
                      bounds=bounds, constraints=constraints,
                      options={'ftol': 1e-9, 'disp': False})

    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
