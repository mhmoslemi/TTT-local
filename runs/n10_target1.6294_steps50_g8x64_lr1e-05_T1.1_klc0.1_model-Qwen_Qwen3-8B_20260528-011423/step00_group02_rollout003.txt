import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 equally spaced circles in grid-like pattern
    centers = np.array([[i*0.25 + 0.125, j*0.25 + 0.125] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraint 1: All circles are inside the square
        # Constraint 2: No two circles overlap
        # Constraint 3: Radii are non-negative
        n = len(radii)
        cons = []

        # Constraint 1: Radii are non-negative
        for i in range(n):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})

        # Constraint 2: All circles are inside the square
        for i in range(n):
            con = lambda r, i=i: r[i] - (1 - r[i])  # r[i] <= x <= 1 - r[i] => 2*r[i] <= 1
            cons.append({'type': 'ineq', 'fun': con})

        # Constraint 3: No two circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                def con(r, i=i, j=j):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - (radii[i] + radii[j])
                cons.append({'type': 'ineq', 'fun': con})

        return cons

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=constraints(radii), bounds=[(0, 1)]*len(radii))

    # Extract results
    centers = np.array([[i*0.25 + 0.125, j*0.25 + 0.125] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
