import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    radii = np.full(10, 0.1)

    # Objective function to maximize the sum of radii (minimize negative sum)
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)

    # Constraints: circles are inside the unit square
    def square_constraint(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.concatenate([
            centers_flat[:, 0] - radii_flat,
            1 - centers_flat[:, 0] - radii_flat,
            centers_flat[:, 1] - radii_flat,
            1 - centers_flat[:, 1] - radii_flat
        ])

    # Constraints: circles do not overlap
    def no_overlap_constraint(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distances = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return distances - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Bounds: centers between 0.1 and 0.9, radii between 0 and 0.5
    bounds = Bounds([0.1, 0.1, 0.0] * 10, [0.9, 0.9, 0.5] * 10)

    # Optimization problem
    constraints = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': no_overlap_constraint}
    ]

    # Initial guess in flattened form
    x0 = np.concatenate([centers.reshape(10 * 2), radii])

    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-8)

    # Extract results
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:]
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
