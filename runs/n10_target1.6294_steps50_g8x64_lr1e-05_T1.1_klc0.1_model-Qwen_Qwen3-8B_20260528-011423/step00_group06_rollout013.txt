import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraint 1: Radii must be non-negative
        if np.any(radii < 0):
            return np.inf
        # Constraint 2: Circles must fit in the unit square
        if np.any(radii > 1 - np.min(np.vstack([centers[:, 0], 1 - centers[:, 0], centers[:, 1], 1 - centers[:, 1]]).T)):
            return np.inf
        # Constraint 3: Circles must not overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    result = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}])

    if result.success:
        sum_radii = -result.fun
        return (centers, result.x, sum_radii)
    else:
        warnings.warn("Optimization failed to converge")
        return (centers, np.zeros(10), 0.0)
