import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: all circles must be inside the unit square
        for i in range(10):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        # Constraint 2: circles must not overlap
        distances = pdist(centers)
        for d, (i, j) in zip(distances, [(a, b) for a in range(10) for b in range(a+1, 10)]):
            if d < radii[i] + radii[j]:
                return False
        return True

    # Define the optimization problem
    bounds = [(0, 1) for _ in range(20)]  # 10 centers * 2 coordinates
    cons = (
        {'type': 'ineq', 'fun': lambda x: constraints(x[10:], x[:10])},
    )
    # Initial guess: concatenate centers and radii
    x0 = np.concatenate((centers.flatten(), radii))

    # Set up the optimization
    result = minimize(
        fun=lambda x: -np.sum(x[10:]),
        x0=x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'ftol': 1e-8, 'disp': False}
    )

    # Extract results
    optimized_centers = result.x[:20].reshape(10, 2)
    optimized_radii = result.x[10:]
    sum_radii = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
