import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 equally spaced circles along a grid
    centers = np.array([[i * 0.25, j * 0.25] for i in range(4) for j in range(4)])
    # Make sure we have exactly 10 circles
    centers = centers[:10]
    radii = np.zeros(10) + 0.1  # Starting with small radii

    def objective(radii):
        return -np.sum(radii)  # We minimize -sum to maximize sum

    def constraints(radii):
        # Constraint 1: All circles are within the unit square
        # x >= r, x <= 1 - r, y >= r, y <= 1 - r
        # So for each circle i, check x[i] - r[i] >= 0 and 1 - r[i] - x[i] >= 0, same for y[i]
        # These are all non-negativity constraints on the radii
        # Here, we use a simple check for optimization feasibility

        # Constraint 2: No two circles overlap
        # For each pair of circles, distance between centers >= sum of radii
        # This is an inequality constraint
        n = len(radii)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def constraint_func(radii):
        return 0  # No constraint on radii directly, only through objective

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1) for _ in range(10)],
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraint_func(r)},
        ],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    # Extract the best solution
    best_radii = result.x
    best_centers = centers
    sum_radii = -result.fun
    return (best_centers, best_radii, sum_radii)
