import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly distributed in the unit square)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Starting with small radii
    
    def objective_function(params):
        # Unpack parameters into centers and radii
        flat_centers = params[:20].reshape(10, 2)
        flat_radii = params[20:]
        
        # Ensure radii are non-negative
        flat_radii = np.maximum(flat_radii, 0)
        
        # Check if all circles are inside the unit square
        for i in range(10):
            x, y = flat_centers[i]
            r = flat_radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return float('inf')  # Penalize invalid positions
        
        # Calculate total sum of radii
        total_radii = np.sum(flat_radii)
        
        # Check for overlaps between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = flat_centers[i, 0] - flat_centers[j, 0]
                dy = flat_centers[i, 1] - flat_centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < flat_radii[i] + flat_radii[j]:
                    return float('inf')  # Penalize overlap
        
        return -total_radii  # Minimize negative total to maximize total
    
    # Initial parameters: flatten centers and radii
    initial_params = np.concatenate([centers.reshape(20), radii])
    
    # Perform optimization
    result = minimize(
        objective_function,
        initial_params,
        method='L-BFGS-B',
        bounds=[(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    )
    
    # Extract final result
    final_centers = result.x[:20].reshape(10, 2)
    final_radii = result.x[20:]
    final_sum_radii = -result.fun  # Convert back from negative
    
    return (final_centers, final_radii, final_sum_radii)
