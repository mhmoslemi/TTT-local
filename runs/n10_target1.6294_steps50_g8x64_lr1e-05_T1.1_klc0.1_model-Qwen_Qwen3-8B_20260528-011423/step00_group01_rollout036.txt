import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with some radius
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    centers = np.vstack([centers, [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5], [0.5, 0.5]])  # add 5 more circles at center
    radii = np.full(10, 0.1)

    def objective(x):
        # x is a flat array of 10*2 + 10 = 30 elements: first 20 are centers, last 10 are radii
        centers_flat = x[:20]
        radii_flat = x[20:]
        centers = centers_flat.reshape(10, 2)
        radii = radii_flat

        # Check if all circles are inside the unit square
        if np.any(centers[:, 0] < radii) or np.any(centers[:, 0] > 1 - radii) or \
           np.any(centers[:, 1] < radii) or np.any(centers[:, 1] > 1 - radii):
            return np.inf  # invalid configuration

        # Check if circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf  # overlapping circles

        # Maximize the sum of radii, so minimize the negative sum
        return -np.sum(radii)

    def bounds():
        # Bounds for centers: [r, 1 - r], and r >= 0
        # Bounds for radii: [0, 0.5]
        bounds = []
        for _ in range(10):
            bounds.extend([(r, 1 - r) for r in [0, 1]])
            bounds.extend([(0, 0.5)]
        return bounds

    # Initial guess as flat array
    x0 = np.concatenate([centers.ravel(), radii])

    # Optimization
    result = minimize(objective, x0, bounds=bounds(), method='L-BFGS-B', tol=1e-8, options={'maxiter': 1000})

    # Extract final values
    centers_final = result.x[:20].reshape(10, 2)
    radii_final = result.x[20:]

    # Ensure all circles are valid
    for i in range(10):
        x, y = centers_final[i]
        r = radii_final[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            radii_final[i] = min(r, x - r, 1 - x - r, y - r, 1 - y - r)

    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
