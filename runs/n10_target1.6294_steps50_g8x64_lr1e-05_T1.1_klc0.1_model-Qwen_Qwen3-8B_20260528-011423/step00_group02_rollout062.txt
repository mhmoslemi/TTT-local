import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles placed in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.1)

    def objective_function(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x[:60].reshape((10, 2))
        radii = x[60:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize the negative sum to maximize the sum

    def constraints(x):
        centers = x[:60].reshape((10, 2))
        radii = x[60:]
        # Constraint 1: Each circle is inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii))
        if not inside:
            return np.inf
        # Constraint 2: Circles do not overlap
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)
                dists[j, i] = dists[i, j]
        min_dist = np.min(dists)
        overlap = np.any(dists < radii[:, np.newaxis] + radii[np.newaxis, :])
        if overlap:
            return np.inf
        return 0

    # Set up the optimization problem
    bounds = []
    for _ in range(10):
        bounds += [(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)]  # x, y, r for each circle

    # Initial guess
    x0 = np.concatenate([initial_centers.ravel(), initial_radii])

    # Optimization
    result = minimize(objective_function, x0, method='L-BFGS-B', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints}, tol=1e-6)

    # Extract results
    centers = result.x[:60].reshape((10, 2))
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
