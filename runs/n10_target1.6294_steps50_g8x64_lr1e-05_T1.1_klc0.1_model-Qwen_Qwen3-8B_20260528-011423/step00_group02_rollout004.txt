import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    # Objective function: minimize negative sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraint: each circle is inside the unit square
    def square_constraint(x):
        x_vals = x[::2]
        y_vals = x[1::2]
        r_vals = x[2::3]
        return np.array([r - (1 - r - x) for r, x in zip(r_vals, x_vals)] +
                        [r - (1 - r - y) for r, y in zip(r_vals, y_vals)])
    
    # Constraint: no two circles overlap
    def no_overlap_constraint(x):
        x_vals = x[::2]
        y_vals = x[1::2]
        r_vals = x[2::3]
        n = len(r_vals)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = x_vals[i] - x_vals[j]
                dy = y_vals[i] - y_vals[j]
                dist = np.sqrt(dx * dx + dy * dy)
                constraints.append(dist - (r_vals[i] + r_vals[j]))
        return np.array(constraints)

    # Flatten centers and radii
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0,
            method='SLSQP',
            constraints=[
                {'type': 'ineq', 'fun': square_constraint},
                {'type': 'ineq', 'fun': no_overlap_constraint}
            ],
            bounds=[(0, 1) for _ in range(2 * 10)] + [(0, 1) for _ in range(10)]
        )

    centers_opt = result.x[::2].reshape(10, 2)
    radii_opt = result.x[2::3]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
