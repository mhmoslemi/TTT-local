import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid points
    centers = []
    for i in range(10):
        x = i / 3
        y = (i % 3) / 3
        centers.append([x, y])
    centers = np.array(centers)

    # Initial radius guess: all same
    radii = np.full(10, 0.1)

    # Define the objective function
    def objective(radii):
        return -np.sum(radii)  # Minimize -sum(radii) to maximize sum(radii)

    # Define the constraints
    def constraints(radii, centers):
        # Constraint 1: radii must be non-negative
        bounds = [(0, 1) for _ in range(10)]
        # Constraint 2: circles are fully inside the unit square
        def in_bounds(radii):
            return np.all(radii <= 1 - np.max(centers, axis=0))  # Ensure r <= x <= 1 - r and r <= y <= 1 - r
        # Constraint 3: circles are non-overlapping
        def no_overlap(radii):
            for i in range(10):
                for j in range(i + 1, 10):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    if dist < radii[i] + radii[j]:
                        return False
            return True
        # Combine all constraints
        if not (np.all(radii >= 0) and in_bounds(radii) and no_overlap(radii)):
            return np.inf
        return 0

    # Set up optimization
    result = minimize(
        fun=lambda x: objective(x),
        x0=radii,
        bounds=[(0, 1) for _ in range(10)],
        constraints=[{'type': 'ineq', 'fun': lambda x: constraints(x, centers)}],
        method='SLSQP'
    )

    # Extract results
    final_radii = result.x
    final_centers = centers.copy()
    sum_radii = -result.fun

    return (final_centers, final_radii, sum_radii)
