import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(r):
        return -np.sum(r)

    # Inequality constraints: circles must be inside the square
    def constraint_inside(r, centers):
        c = np.zeros(20)
        for i in range(10):
            c[2*i] = 2 * r[i] - centers[i, 0]
            c[2*i+1] = 2 * r[i] - (1 - centers[i, 1])
        return c

    # Inequality constraints: circles must not overlap
    def constraint_overlap(r, centers):
        c = []
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                c.append(r[i] + r[j] - dist)
        return np.array(c)

    # Bounds: radii must be non-negative
    bounds = [(0, None) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers)},
            {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers)}
        ],
        options={'ftol': 1e-10, 'maxiter': 500}
    )

    # Extract results
    centers_opt = centers
    radii_opt = result.x
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
