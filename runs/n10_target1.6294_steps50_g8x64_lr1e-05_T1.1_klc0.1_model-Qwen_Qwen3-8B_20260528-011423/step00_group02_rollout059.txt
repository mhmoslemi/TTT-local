import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced centers in grid
    centers = np.array([[i * 0.25, j * 0.25] for i in range(4) for j in range(4)])
    radii = np.zeros(10)
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[2:12])  # Radii are x[2:12]
    
    # Constraints
    def constraint_center_inside(x):
        # Centers are x[0:10], each with [x, y]
        # Radii are x[10:20]
        centers = x[:10].reshape(-1, 2)
        radii = x[10:]
        return np.array([x[i] >= radii[i] for i in range(10)] + [x[i] <= 1 - radii[i] for i in range(10)])
    
    def constraint_no_overlap(x):
        centers = x[:10].reshape(-1, 2)
        radii = x[10:]
        distances = np.sqrt(((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2).sum(axis=2))
        return distances - (radii[:, np.newaxis] + radii[np.newaxis, :])
    
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    
    # Bounds: x, y between 0 and 1; radii between 0 and 0.5
    bounds = [(0, 1) for _ in range(10)] + [(0, 0.5) for _ in range(10)]
    
    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, np.concatenate([centers.flatten(), np.zeros(10)]), method='SLSQP', bounds=bounds, constraints=constraints)
    
    optimized_x = result.x
    centers_opt = optimized_x[:10].reshape(-1, 2)
    radii_opt = optimized_x[10:]
    sum_radii = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii)
