import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: 10 circles at regular grid positions
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radii: assume all radii are equal for starting point
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # We minimize -sum to maximize sum

    def constraints(radii, centers):
        # Constraint: all radii must be non-negative
        bounds = [(0, 1) for _ in range(10)]
        # Constraint: circles must be fully inside the unit square
        inside = [(radii[i] <= centers[i, 0] - 1e-8) and (radii[i] <= 1 - centers[i, 0] - 1e-8) and
                  (radii[i] <= centers[i, 1] - 1e-8) and (radii[i] <= 1 - centers[i, 1] - 1e-8)
                  for i in range(10)]
        # Constraint: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    res = minimize(objective, radii, method='SLSQP', bounds=[(0, 1) for _ in range(10)],
                   constraints={'type': 'ineq', 'fun': lambda x: constraints(x, centers)})
    
    # Update the optimized centers if the solution is valid
    if res.success and constraints(res.x, centers):
        radii = res.x
        # Adjust the centers to maximize the radii while still satisfying the constraints
        # Simple iterative adjustment of centers
        for _ in range(10):
            for i in range(10):
                for j in range(i + 1, 10):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx * dx + dy * dy)
                    if dist < radii[i] + radii[j] - 1e-8:
                        # Move circles apart
                        direction = np.array([dx, dy]) / dist
                        step = (radii[i] + radii[j] - dist) / 2
                        centers[i, :] += direction * step
                        centers[j, :] -= direction * step

        # Ensure circles are inside the square
        for i in range(10):
            x, y = centers[i]
            radii[i] = min(radii[i], x - 1e-8)
            radii[i] = min(radii[i], 1 - x - 1e-8)
            radii[i] = min(radii[i], y - 1e-8)
            radii[i] = min(radii[i], 1 - y - 1e-8)

        sum_radii = np.sum(radii)
    else:
        # If optimization fails, fall back to initial guess
        sum_radii = np.sum(radii)

    return centers, radii, sum_radii
