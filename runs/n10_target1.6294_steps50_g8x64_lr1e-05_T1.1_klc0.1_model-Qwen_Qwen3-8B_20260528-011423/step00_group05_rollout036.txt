import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii (simple grid layout)
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    # Objective function to maximize the sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Negative for minimization

    # Constraints
    def constraint_inside(radii_vec):
        # All circles are within the unit square
        return np.array([r <= 0.5 for r in radii_vec])  # Assuming r <= 0.5 for safe placement

    def constraint_no_overlap(x):
        centers = x[:10, :2]
        radii = x[10:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -1.0  # Violation
        return 0.0  # No violation

    # Optimization bounds
    bounds = []
    for _ in range(10):
        bounds.append((0, 1))  # x-coordinates
        bounds.append((0, 1))  # y-coordinates
    for _ in range(10):
        bounds.append((0, 0.5))  # radii

    # Initial combined parameter vector
    x0 = np.vstack((centers, radii)).ravel()

    # Constraints in the form required by scipy.optimize.minimize
    cons = [
        {'type': 'ineq', 'fun': constraint_inside}
    ]

    # Optimization
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'disp': False}
    )

    # Extract the result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
