import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        radius_array = radii.reshape(-1, 1)
        inside = np.all((radius_array <= centers[:, 0]) & (centers[:, 0] <= 1 - radius_array) &
                        (radius_array <= centers[:, 1]) & (centers[:, 1] <= 1 - radius_array), axis=1)
        inside = np.all(inside)

        # Constraint 2: No overlap between circles
        dists = cdist(centers, centers)
        np.fill_diagonal(dists, np.inf)
        radii_sum = radii.reshape(-1, 1) + radii.reshape(1, -1)
        no_overlap = np.all(dists >= radii_sum)
        return inside and no_overlap

    # Optimization setup
    bounds = [(0, 1.0) for _ in range(10)]
    res = minimize(
        lambda x: -np.sum(x),
        x0=radii,
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': lambda x: 1.0 - constraints(x, centers)}],
        method='SLSQP',
        options={'disp': False, 'ftol': 1e-9}
    )

    # Update centers and radii based on optimization result
    best_radii = res.x
    # Ensure circles are packed without overlap and inside the square (fallback if needed)
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    return (centers, best_radii, -res.fun)
