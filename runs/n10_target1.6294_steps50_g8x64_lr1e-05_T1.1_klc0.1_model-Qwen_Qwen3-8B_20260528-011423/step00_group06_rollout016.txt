import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess: 10 circles placed in a grid
    initial_centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    initial_centers = np.vstack((initial_centers, [0.5, 0.5]))  # add one more circle near center
    initial_radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            if radii[i] > 1:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Define bounds for radii
    bounds = [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        jac=lambda x: np.zeros_like(x),
        options={'maxiter': 1000, 'ftol': 1e-8, 'disp': False}
    )

    # Extract results
    radii = result.x
    sum_radii = -result.fun

    # Ensure radii are valid
    radii = np.clip(radii, 0, 1)
    centers = np.zeros((10, 2))

    # Adjust initial guess to match the number of radii
    centers[:len(initial_centers)] = initial_centers

    return centers, radii, sum_radii
