import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: spread circles evenly and set a small radius
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.08)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        # radii must satisfy r <= x - 0 and 1 - x <= r for all x
        # r <= y - 0 and 1 - y <= r for all y
        constraints = []
        for i in range(10):
            for j in range(10):
                # Constraint 2: Circles do not overlap
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.hypot(dx, dy)
                    constraints.append(dist - (radii[i] + radii[j]) >= 0)
        return constraints

    def bounds(radii):
        return [(0, 1.0 - 1e-6) for _ in range(10)]  # All radii in (0, 1)

    # Optimization setup
    n = 10
    x0 = np.concatenate((centers.ravel(), radii))

    def objective_func(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)

    def constraint_func(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        constraints = []
        for i in range(n):
            for j in range(n):
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.hypot(dx, dy)
                    constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    def bounds_func(x):
        return [(0, 1.0 - 1e-6) for _ in range(2*n + n)]

    # Run optimization
    result = minimize(
        objective_func,
        x0,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraint_func},
        bounds=bounds_func,
        tol=1e-7,
        options={'maxiter': 1000, 'disp': False}
    )

    centers_opt = result.x[:2*n].reshape(n, 2)
    radii_opt = result.x[2*n:]

    # Ensure circles are inside the square
    for i in range(n):
        x, y = centers_opt[i]
        r = radii_opt[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            raise ValueError(f"Circle {i} is not fully inside the square")

    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
