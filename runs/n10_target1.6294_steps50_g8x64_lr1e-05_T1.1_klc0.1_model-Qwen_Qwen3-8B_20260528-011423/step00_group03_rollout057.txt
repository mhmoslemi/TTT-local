import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function to minimize (negative sum of radii to maximize)
    def objective(radii):
        return -np.sum(radii)

    # Constraint: each circle must fit in the unit square
    def square_bounds(radii):
        return np.sum(radii <= 1 - np.sqrt(2 * (radii ** 2)))  # Just a simplified check for each radius

    # Constraint: no two circles overlap
    def no_overlap(radii):
        n = len(radii)
        dx = centers[np.newaxis, :, 0] - centers[:, np.newaxis, 0]
        dy = centers[np.newaxis, :, 1] - centers[:, np.newaxis, 1]
        distances = np.sqrt(dx ** 2 + dy ** 2)
        min_dist = np.min(distances[distances > 1e-8])  # Skip self-distance
        return min_dist - np.sum(radii) >= 0  # Ensures minimum distance is >= sum of radii

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0=np.ones(10),
            bounds=[(0, 1 - 1e-3)] * 10,  # All radii must be less than 1 - small epsilon
            constraints=[
                {'type': 'ineq', 'fun': square_bounds},
                {'type': 'ineq', 'fun': no_overlap}
            ],
            method='SLSQP',
            options={'maxiter': 200}
        )

    # Extract results
    radii = result.x
    sum_radii = np.sum(radii)
    centers = np.array(centers).astype(float)  # Ensure numeric type

    return (centers, radii, sum_radii)
