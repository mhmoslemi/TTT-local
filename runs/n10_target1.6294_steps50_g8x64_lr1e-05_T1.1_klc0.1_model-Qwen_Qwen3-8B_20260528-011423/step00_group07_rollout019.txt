import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid with small radius
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    radii = np.ones(9) * 0.05
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more circle at center
    radii = np.append(radii, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        # Ensure circles are within the square
        def in_square(r):
            return np.all((radii <= 1 - r) & (radii >= r))
        if not in_square(radii):
            return np.inf
        # Ensure no overlaps
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimize
    bounds = [(0, 1 - 1e-6) for _ in range(10)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    if not result.success:
        print("Optimization failed", file=sys.stderr)
        return np.array([[0.5, 0.5]] * 10), np.zeros(10), 0.0

    centers = np.array(centers)
    sum_radii = np.sum(result.x)
    return centers, result.x, sum_radii
