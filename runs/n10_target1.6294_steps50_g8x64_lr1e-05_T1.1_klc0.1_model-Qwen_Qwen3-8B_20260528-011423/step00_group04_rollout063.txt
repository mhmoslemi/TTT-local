import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.array([[i * 0.25, j * 0.25] for i in range(4) for j in range(4)])
    radii = np.array([0.1] * 10)
    x_bounds = [(0, 1) for _ in range(10 * 2)]
    r_bounds = [(0, 0.5) for _ in range(10)]
    constraints = []

    # Constraint: each circle must be fully inside the unit square
    def circle_inside(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return 1
        return 0

    constraints.append({'type': 'ineq', 'fun': circle_inside})

    # Constraint: distance between any two circles >= sum of their radii
    def no_overlap(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return 1
        return 0

    constraints.append({'type': 'ineq', 'fun': no_overlap})

    # Objective: maximize sum of radii (minimize negative sum)
    def objective(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        return -np.sum(radii_flat)

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                      bounds=zip(x_bounds, r_bounds), constraints=constraints)

    if result.success:
        centers_opt = result.x[:20].reshape(10, 2)
        radii_opt = result.x[20:]
        sum_radii = -result.fun
        return (centers_opt, radii_opt, sum_radii)
    else:
        # Fallback in case of failure (e.g., due to optimization issues)
        centers = np.array([[0.25, 0.25], [0.75, 0.25], [0.25, 0.75], [0.75, 0.75],
                            [0.5, 0.1], [0.5, 0.9], [0.1, 0.5], [0.9, 0.5], [0.5, 0.5], [0.5, 0.5]])
        radii = np.array([0.1] * 10)
        sum_radii = 1.0
        return (centers, radii, sum_radii)
