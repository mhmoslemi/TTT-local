import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.08)

    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraints
    def constraint1(radii, centers):
        # Constraint: all circles are within the unit square
        return np.r_[radii, (centers[:, 0] - radii) - 0, 1 - (centers[:, 0] + radii), (centers[:, 1] - radii) - 0, 1 - (centers[:, 1] + radii)]

    def constraint2(radii, centers):
        # Constraint: no two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return dists - radii[:, np.newaxis] - radii[np.newaxis, :]

    # Optimization
    def optimize():
        bounds = [(0, 1) for _ in range(10)]
        cons = [
            {'type': 'ineq', 'fun': lambda r: constraint1(r, centers)},
            {'type': 'ineq', 'fun': lambda r: constraint2(r, centers)}
        ]
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)
        return result.x

    # Run optimization
    optimized_radii = optimize()
    optimized_centers = centers

    # Sum of radii
    sum_radii = np.sum(optimized_radii)

    # Return the result
    return (optimized_centers, optimized_radii, sum_radii)
