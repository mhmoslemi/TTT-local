import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        row = i // 3
        col = i % 3
        centers[i, 0] = 0.2 + col * 0.4
        centers[i, 1] = 0.2 + row * 0.4

    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Bounds: radii >= 0 and <= 0.5 (since radius can't exceed 0.5 to stay within square)
        bounds = [(0, 0.5) for _ in range(10)]
        # Ensure circles don't overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        # Ensure circles are within the unit square
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        return True

    # Optimization with constraints
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 0.5) for _ in range(10)],
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r)}]
    )

    # Extract results
    best_radii = result.x
    best_sum_radii = np.sum(best_radii)
    centers_updated = centers.copy()

    return (centers_updated, best_radii, best_sum_radii)
