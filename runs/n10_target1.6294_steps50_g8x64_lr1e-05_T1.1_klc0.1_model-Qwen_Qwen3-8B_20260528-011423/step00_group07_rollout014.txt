import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles placed in a grid-like pattern
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)  # We minimize the negative sum to maximize the sum

    def constraints(centers, radii):
        # Constraint 1: Circle is fully inside the unit square
        bounds = np.array([(r, 1 - r) for r in radii])
        return np.all((centers >= bounds) & (centers <= 1 - bounds))

    def non_overlap(centers, radii):
        # Constraint 2: No overlap between circles
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx ** 2 + dy ** 2)
                dists[j, i] = dists[i, j]
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    def optimize():
        # Optimization with bounds and constraints
        bounds = [(0, 1) for _ in range(20)]  # 10 centers * 2 dimensions

        def objective_func(x):
            centers = x[:20].reshape(10, 2)
            radii = x[20:]
            return -np.sum(radii)

        def constraint_func(x):
            centers = x[:20].reshape(10, 2)
            radii = x[20:]
            # Constraint 1: within bounds
            within_bounds = np.all((centers >= radii[:, np.newaxis]) & (centers <= 1 - radii[:, np.newaxis]))
            if not within_bounds:
                return -np.inf
            # Constraint 2: no overlap
            dists = np.zeros((10, 10))
            for i in range(10):
                for j in range(i + 1, 10):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dists[i, j] = np.sqrt(dx ** 2 + dy ** 2)
                    dists[j, i] = dists[i, j]
            no_overlap = np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
            if not no_overlap:
                return -np.inf
            return 0

        result = minimize(objective_func, np.concatenate([centers.ravel(), radii]), bounds=bounds,
                          constraints={'type': 'ineq', 'fun': constraint_func}, method='SLSQP')
        return result.x

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        x = optimize()

    centers = x[:20].reshape(10, 2)
    radii = x[20:]
    sum_radii = np.sum(radii)
    return centers, radii, sum_radii
