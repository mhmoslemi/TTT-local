import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles at a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize the sum by minimizing the negative
    
    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            r_i = radii[i]
            # Ensure circle is inside the unit square
            if not (r_i <= centers[i, 0] <= 1 - r_i and r_i <= centers[i, 1] <= 1 - r_i):
                return False
        # Ensure no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    bounds = [(0.0, 0.5) for _ in range(10)]
    
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "custom", "fun": constraints},
        method="SLSQP"
    )
    
    if not result.success:
        # Fallback to the initial guess if optimization fails
        return (np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)]),
                np.full(10, 0.1), np.sum(np.full(10, 0.1)))
    
    centers_opt = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii_opt = result.x
    sum_radii = -result.fun
    
    return (centers_opt, radii_opt, sum_radii)
