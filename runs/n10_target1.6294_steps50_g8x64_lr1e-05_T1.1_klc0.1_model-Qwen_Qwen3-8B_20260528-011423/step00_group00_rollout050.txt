import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced circles inside the square
    centers = np.linspace(0.1, 0.9, 10).reshape(-1, 1)
    centers = np.hstack([centers, centers])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = np.random.rand()
            centers[i, 1] = np.random.rand()
        # Reassign centers based on radii
        for i in range(10):
            centers[i, 0] = np.random.rand() * (1 - 2 * radii[i]) + radii[i]
            centers[i, 1] = np.random.rand() * (1 - 2 * radii[i]) + radii[i]
        # Check square constraints
        for i in range(10):
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return np.inf
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return np.inf
        # Check circle constraints
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)
    
    # Optimize
    result = minimize(objective, radii, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints})
    
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = np.random.rand() * (1 - 2 * result.x[i]) + result.x[i]
        centers[i, 1] = np.random.rand() * (1 - 2 * result.x[i]) + result.x[i]
    
    return (centers, result.x, -result.fun)
