import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles placed in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Constraint 1: All circles must be inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers[i][0], centers[i][1], radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Bounds: Each center (x, y) is between 0 and 1, and each radius is non-negative
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x_i
        bounds.append((0.0, 1.0))  # y_i
        bounds.append((0.0, 1.0))  # r_i

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate((initial_centers.flatten(), initial_radii)),
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'disp': False, 'ftol': 1e-9, 'maxiter': 1000}
    )

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun  # Objective was negative sum
    return (centers, radii, sum_radii)
