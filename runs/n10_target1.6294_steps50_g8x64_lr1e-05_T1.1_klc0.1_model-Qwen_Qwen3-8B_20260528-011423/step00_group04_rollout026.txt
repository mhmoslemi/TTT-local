import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles evenly in a grid
    initial_centers = np.kron(np.indices((2, 5)).reshape(10, 2), np.array([0.25, 0.25]))
    initial_radii = np.full(10, 0.1)

    # Define the objective function to minimize (negative sum of radii)
    def objective(radii, centers):
        return -np.sum(radii)

    # Define constraints
    def constraint_center_inside(radii, centers):
        # Each circle is inside the unit square
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0] - radii)

    def constraint_overlap(radii, centers):
        # Distance between centers >= sum of radii
        distances = cdist(centers, centers)
        return np.minimum(distances - radii[:, np.newaxis] - radii[np.newaxis, :], 0)

    # Optimization settings
    bounds = [(0, 1) for _ in range(10)]  # Radii must be between 0 and 1

    # Initial guess for optimization
    x0 = np.concatenate((initial_centers.reshape(10 * 2), initial_radii))

    # Optimization problem setup
    cons = [
        {'type': 'ineq', 'fun': lambda x: constraint_center_inside(x[10:], x[:10].reshape(10, 2))},
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x[10:], x[:10].reshape(10, 2))}
    ]

    # Perform the optimization
    result = minimize(
        fun=lambda x: objective(x[10:], x[:10].reshape(10, 2)),
        x0=x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP'
    )

    # Extract the result
    centers = result.x[:10].reshape(10, 2)
    radii = result.x[10:]
    sum_radii = -objective(radii, centers)

    return (centers, radii, sum_radii)
