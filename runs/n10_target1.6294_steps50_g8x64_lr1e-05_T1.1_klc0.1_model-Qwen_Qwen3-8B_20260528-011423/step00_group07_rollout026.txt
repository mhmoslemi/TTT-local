            x_i, y_i = x[i*2], x[i*2+1]
            if not (r <= x_i <= 1 - r and r <= y_i <= 1 - r):
                return False
        # No two circles overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = x[i*2] - x[j*2]
                dy = x[i*2+1] - x[j*2+1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Initial guess: place circles evenly spaced
    centers = np.array([(0.1 + i*0.2, 0.1 + j*0.2) for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)
    x0 = centers.flatten()

    # Optimization with constraints
    result = minimize(objective, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints})
    x_opt = result.x
    centers = np.reshape(x_opt, (10, 2))
    radii = np.array([min(1 - centers[i, 0], 1 - centers[i, 1], centers[i, 0], centers[i, 1]) for i in range(10)])
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
