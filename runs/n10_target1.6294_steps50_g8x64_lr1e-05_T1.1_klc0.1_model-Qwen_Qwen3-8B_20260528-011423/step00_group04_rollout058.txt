import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: spread out circles with equal radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii
    
    # Constraints: 1. Circles fully inside the unit square
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        inside = np.all((centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1) & np.all(
            (1 - centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1)
        return np.sum(inside) == 10  # All circles must be inside
    
    # Constraints: 2. No circles overlap
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                distance = math.hypot(dx, dy)
                if distance < radii_flat[i] + radii_flat[j]:
                    return -1.0  # Violated
        return 0.0  # Not violated
    
    # Flatten centers and radii into a single array for optimization
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Constraints in the form required by scipy.optimize.minimize
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Bounds: x, y in [0, 1], radii >= 0
    bounds = [(0.0, 1.0) for _ in range(60)] + [(0.0, 1.0) for _ in range(10)]
    
    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)
    
    # Extract result
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:]
    sum_radii_result = -result.fun
    
    return (centers_result, radii_result, sum_radii_result)
