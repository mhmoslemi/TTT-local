import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly placed)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    
    # Objective function: maximize the sum of radii -> minimize the negative sum
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints: 
    # 1. All circles must be inside the unit square
    # 2. No two circles overlap
    # 3. Radii are non-negative
    
    constraints = []
    
    # Constraint: radii must be non-negative
    constraints.append({'type': 'ineq', 'fun': lambda r: r})
    
    # Constraint: each circle must be inside the unit square
    for i in range(10):
        def constraint_i(r):
            r_i = r[i]
            return r_i - (1 - centers[i, 0] - r_i)  # x >= r_i and x <= 1 - r_i
        constraints.append({'type': 'ineq', 'fun': constraint_i})
        def constraint_j(r):
            r_j = r[i]
            return r_j - (1 - centers[i, 1] - r_j)  # y >= r_i and y <= 1 - r_i
        constraints.append({'type': 'ineq', 'fun': constraint_j})
    
    # Constraint: no two circles overlap
    for i in range(10):
        for j in range(i+1, 10):
            def constraint_ij(r):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                r_i = r[i]
                r_j = r[j]
                return dist - (r_i + r_j)
            constraints.append({'type': 'ineq', 'fun': constraint_ij})
    
    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=constraints)
    
    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii_final = -result.fun
    
    return (centers_final, radii_final, sum_radii_final)
