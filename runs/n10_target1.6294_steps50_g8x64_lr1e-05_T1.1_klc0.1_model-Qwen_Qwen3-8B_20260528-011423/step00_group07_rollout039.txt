import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: placing 10 circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds(radii):
        return [(0, 1 - 0.1) for _ in range(10)]

    def constraint_func(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            x, y = centers[i]
            if radii[i] > x or radii[i] > 1 - x or radii[i] > y or radii[i] > 1 - y:
                return False
        return True

    # Define optimization problem
    res = minimize(
        fun=objective,
        x0=radii,
        constraints=[
            {'type': 'ineq', 'fun': constraint_func},
            {'type': 'ineq', 'fun': constraints}
        ],
        bounds=bounds(radii),
        method='SLSQP'
    )

    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = res.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
