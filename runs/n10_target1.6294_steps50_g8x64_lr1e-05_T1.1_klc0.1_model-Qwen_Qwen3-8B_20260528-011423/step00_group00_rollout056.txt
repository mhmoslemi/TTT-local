import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.array([0.1] * 10)

    # Optimization function to minimize negative sum of radii
    def objective(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative to maximize sum

    # Constraints: each circle must be fully inside the unit square
    def square_constraint(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0]) - 1e-8
    square_cons = {'type': 'ineq', 'fun': square_constraint}

    # Constraints: distance between any two centers >= sum of their radii
    def circle_overlap_constraint(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        dist = np.sqrt(np.sum((centers[:, np.newaxis] - centers[np.newaxis, :])**2, axis=2))
        max_allowed = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dist - max_allowed + 1e-8
    circle_cons = {'type': 'ineq', 'fun': circle_overlap_constraint}

    # Bounds: radii >= 0
    bounds = [(0, None) for _ in range(20)]  # 10 centers with 2 coordinates each, 10 radii

    # Run optimization
    result = minimize(objective, x0=np.concatenate([initial_centers.ravel(), initial_radii]),
                      method='SLSQP',
                      bounds=bounds,
                      constraints=[square_cons, circle_cons])

    # Extract results
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun

    # Return result
    return (centers, radii, sum_radii)
