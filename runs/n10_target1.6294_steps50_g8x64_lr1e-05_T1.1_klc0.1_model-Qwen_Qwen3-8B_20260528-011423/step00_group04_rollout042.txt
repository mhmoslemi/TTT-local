import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    
    def objective(x):
        # Flatten the input for easier processing
        x_flat = x.reshape(10, 3)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        
        # Objective is to minimize negative sum of radii
        sum_r = -np.sum(radii)
        
        # Constraints
        constraints = []
        
        # 1. Each circle must be fully inside the unit square
        for i in range(10):
            r = radii[i]
            if r < 0:
                return np.inf
            if centers[i, 0] < r or centers[i, 0] > 1 - r:
                return np.inf
            if centers[i, 1] < r or centers[i, 1] > 1 - r:
                return np.inf
        
        # 2. No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        
        return sum_r
    
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    
    # Initial guess
    x0 = np.hstack((centers.reshape(-1), radii))
    
    # Optimization with bounds and constraints
    result = minimize(objective, x0, bounds=bounds, method='SLSQP', tol=1e-8)
    
    # Extract the result
    x_opt = result.x
    centers = x_opt[:20].reshape(10, 2)
    radii = x_opt[20:]
    
    # Check if the optimization was successful
    if result.success:
        sum_radii = np.sum(radii)
        return (centers, radii, sum_radii)
    else:
        # Fallback if optimization fails
        centers = np.random.rand(10, 2)
        radii = np.random.rand(10)
        sum_radii = np.sum(radii)
        return (centers, radii, sum_radii)
