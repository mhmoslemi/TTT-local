import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: evenly spaced centers with initial radius
    centers = np.array([[i / 9 + 0.05, j / 9 + 0.05] for j in range(10) for i in range(10)])
    centers = centers[:10]  # Take only 10 centers
    radii = np.full(10, 0.1)  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraint_overlap(i, j, radii):
        def constraint(x):
            # x is the vector of radii
            x = np.array(x)
            dist = np.linalg.norm(centers[i] - centers[j])
            return dist - (x[i] + x[j])
        return constraint

    def constraint_boundary(i, radii):
        def constraint(x):
            x = np.array(x)
            dist = np.linalg.norm(centers[i])
            return 1 - dist - x[i]
        return constraint

    # Define constraints
    constraints = []

    for i in range(10):
        # Boundary constraint for each circle
        cons = {'type': 'ineq', 'fun': constraint_boundary(i, radii)}
        constraints.append(cons)

    for i in range(10):
        for j in range(i + 1, 10):
            # Overlap constraint between circles i and j
            cons = {'type': 'ineq', 'fun': constraint_overlap(i, j, radii)}
            constraints.append(cons)

    # Bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(10)]

    # Initial guess for radii
    initial_radii = np.full(10, 0.1)

    # Optimization
    result = minimize(
        fun=objective,
        x0=initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Extract optimized values
    radii_opt = result.x
    sum_radii_opt = -result.fun

    # Ensure constraints are satisfied and radii are valid
    centers = np.array(centers)
    for i in range(10):
        if radii_opt[i] < 0:
            radii_opt[i] = 0
        if np.any(np.linalg.norm(centers[i]) + radii_opt[i] > 1):
            radii_opt[i] = max(0, 1 - np.linalg.norm(centers[i]))
        if np.any(np.linalg.norm(centers[i] - 1) + radii_opt[i] > 1):
            radii_opt[i] = max(0, 1 - np.linalg.norm(centers[i] - 1))

    # Re-verify all constraints after adjustment
    for i in range(10):
        for j in range(i + 1, 10):
            if np.linalg.norm(centers[i] - centers[j]) < radii_opt[i] + radii_opt[j]:
                radii_opt[i] = max(0, radii_opt[i] - (np.linalg.norm(centers[i] - centers[j]) - radii_opt[j]))
                radii_opt[j] = max(0, radii_opt[j] - (np.linalg.norm(centers[i] - centers[j]) - radii_opt[i]))

    sum_radii_opt = np.sum(radii_opt)

    return (centers, radii_opt, sum_radii_opt)
