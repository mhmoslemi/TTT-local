import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles placed in a grid-like pattern
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.array([0.1] * 10)
    
    def objective(x):
        # x is a vector of 20 values: 10 centers (each with x, y) and 10 radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Objective is to minimize the negative sum of radii (maximize sum of radii)
        return -np.sum(radii)
    
    def constraints(x):
        # Constraint 1: All circles are inside the unit square
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            for j in range(10):
                if i == j:
                    continue
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return 1.0  # Overlap detected
        for i in range(10):
            if radii[i] < 0:
                return 1.0  # Negative radius
            if not (radii[i] <= centers[i][0] <= 1 - radii[i] and radii[i] <= centers[i][1] <= 1 - radii[i]):
                return 1.0  # Circle out of bounds
        return 0.0
    
    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        bounds=[
            (0.0, 1.0) for _ in range(20)  # centers
        ] + [(0.0, 1.0) for _ in range(10)]  # radii
    )
    
    # Extract optimized result
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii_opt = -result.fun
    return (centers_opt, radii_opt, sum_radii_opt)
