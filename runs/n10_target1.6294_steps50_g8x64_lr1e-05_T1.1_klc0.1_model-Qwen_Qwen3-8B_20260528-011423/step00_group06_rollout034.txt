import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_bounds(centers, radii):
        # Ensure circles are fully inside the unit square
        return np.all((radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) &
                      (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1]))

    def constraint_overlap(centers, radii):
        # Ensure circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    def constraint_func(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        if not constraint_bounds(centers, radii):
            return np.inf
        if not constraint_overlap(centers, radii):
            return np.inf
        return 0

    # Flatten the centers and radii into a single vector for optimization
    x0 = np.concatenate([initial_centers.flatten(), initial_radii])

    # Set bounds for centers (between 0 and 1)
    bounds_centers = [(0, 1) for _ in range(60)]
    # Set bounds for radii (non-negative)
    bounds_radii = [(0, 1) for _ in range(10)]

    bounds = bounds_centers + bounds_radii

    # Setup optimization
    result = minimize(
        fun=constraint_func,
        x0=x0,
        bounds=bounds,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    x_opt = result.x
    centers_opt = x_opt[:60].reshape(10, 2)
    radii_opt = x_opt[60:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
