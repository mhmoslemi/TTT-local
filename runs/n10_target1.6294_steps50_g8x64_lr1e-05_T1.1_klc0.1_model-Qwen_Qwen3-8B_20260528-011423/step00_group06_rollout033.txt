import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii (random positions and equal radii)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        # x has shape (22,) -> 10 centers (2 per circle) + 10 radii
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints: 1. Circle inside unit square
    def constraint_inside(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        return np.r_[centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat,
                     centers_flat[:, 1] - radii_flat, 1 - centers_flat[:, 1] - radii_flat]

    # Constraints: 2. No overlap between circles
    def constraint_no_overlap(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        dists = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Optimization problem
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(objective, x0=np.r_[centers.ravel(), radii], bounds=bounds, constraints=constraints)

    # Extract results
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
