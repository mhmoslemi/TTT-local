import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraint_box(radii, centers):
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return -1  # Violates box constraint
        return 0

    def constraint_overlap(radii, centers):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -1  # Violates overlap constraint
        return 0

    def constraint_all(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = 0.1 + i * 0.2
            centers[i, 1] = 0.1 + (i // 5) * 0.2
        return constraint_box(radii, centers) + constraint_overlap(radii, centers)

    bounds = [(0.0, 0.5) for _ in range(10)]

    result = minimize(
        objective,
        initial_radii,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_box, 'args': (initial_centers,)},
            {'type': 'ineq', 'fun': constraint_overlap, 'args': (initial_centers,)}
        ]
    )

    radii = result.x
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = 0.1 + i * 0.2
        centers[i, 1] = 0.1 + (i // 5) * 0.2

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
