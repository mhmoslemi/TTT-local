import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for circle centers and radii
    initial_centers = np.random.rand(10, 2)
    initial_radii = np.random.rand(10) * 0.1

    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Objective: minimize the negative sum of radii (maximize sum of radii)
        sum_radii = np.sum(radii)
        # Constraint 1: all circles must be inside the unit square
        for i in range(10):
            xi, yi, ri = centers[i, 0], centers[i, 1], radii[i]
            if not (ri <= xi <= 1 - ri and ri <= yi <= 1 - ri):
                return np.inf
        # Constraint 2: no overlapping between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx*dx + dy*dy)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return -sum_radii  # minimize negative sum to maximize sum

    # Flatten initial guess
    x0 = np.concatenate([initial_centers.reshape(-1), initial_radii])

    # Bounds for each variable: x and y between 0 and 1, radii non-negative
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])  # radius up to 0.5 to be inside square

    # Optimization
    result = minimize(objective, x0, bounds=bounds, method='L-BFGS-B', tol=1e-8)

    # Extract results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun  # since we minimized negative sum

    return (centers, radii, sum_radii)
