import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced circles in the square
    n_circles = 10
    centers = np.zeros((n_circles, 2))
    radii = np.full(n_circles, 0.1)
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x[2:])  # Negative for minimization

    # Constraints
    def constraints(x):
        # Extract centers and radii from x
        centers = x[:2*n_circles].reshape(n_circles, 2)
        radii = x[2*n_circles:]
        
        # Constraint 1: All circles are inside the unit square
        for i in range(n_circles):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf  # Violation
        # Constraint 2: No two circles overlap
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf  # Violation
        return 0  # All constraints satisfied

    # Flatten centers and radii for optimization
    initial_guess = np.concatenate([centers.flatten(), radii])
    
    # Minimize negative sum of radii with constraints
    result = minimize(objective, initial_guess, constraints={"type": "ineq", "fun": constraints}, method='SLSQP')

    # Extract results
    x_result = result.x
    centers = x_result[:2*n_circles].reshape(n_circles, 2)
    radii = x_result[2*n_circles:]
    sum_radii = -objective(x_result)

    return (centers, radii, sum_radii)
