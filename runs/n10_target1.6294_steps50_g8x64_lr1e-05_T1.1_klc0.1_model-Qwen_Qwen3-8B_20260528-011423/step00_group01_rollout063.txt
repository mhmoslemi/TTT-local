import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced centers
    centers = np.array([[i / 9 + 0.05, j / 9 + 0.05] for j in range(10) for i in range(10)])
    centers = centers[:10]
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Enforce circles inside the square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return np.inf
            if radii[i] > centers[i, 0] or radii[i] > centers[i, 1]:
                return np.inf
        # Enforce non-overlapping circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Set up optimization
    res = minimize(objective, radii, method='SLSQP', constraints={
        'type': 'ineq',
        'fun': constraints
    }, bounds=[(0, 1 - 0.1) for _ in range(10)], tol=1e-6)

    # Extract results
    centers = np.array(centers)
    radii = res.x
    sum_radii = -res.fun

    return (centers, radii, sum_radii)
