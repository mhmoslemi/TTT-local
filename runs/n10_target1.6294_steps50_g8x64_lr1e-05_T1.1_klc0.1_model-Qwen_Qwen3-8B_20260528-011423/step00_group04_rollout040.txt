import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.array([r <= 1 - r for r in radii])

    def constraint_no_overlap(centers, radii):
        diff = centers[:, np.newaxis, :] - centers[np.newaxis, :, :]
        dist = np.sqrt(np.sum(diff ** 2, axis=2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraint_callback(c):
        if np.any(c < 0):
            raise ValueError("Constraint violation")

    def objective_with_constraints(radii_flat):
        nonlocal centers
        radii = radii_flat.reshape((10,))
        # Update centers to respect circle size and position
        for i in range(10):
            x, y = centers[i]
            if radii[i] > 0:
                centers[i, 0] = np.clip(x, radii[i], 1 - radii[i])
                centers[i, 1] = np.clip(y, radii[i], 1 - radii[i])
        # Compute constraints
        dist_matrix = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        constraints = dist_matrix - (radii[:, np.newaxis] + radii[np.newaxis, :])
        if np.any(constraints < -1e-8):
            return np.inf  # Penalize overlap
        return -np.sum(radii)

    # Optimization
    bounds = [(0, 1 - 1e-8) for _ in range(10)]
    result = minimize(
        objective_with_constraints,
        x0=radii,
        method='L-BFGS-B',
        bounds=bounds,
        options={'maxiter': 1000, 'ftol': 1e-10}
    )

    radii_opt = result.x
    centers_opt = np.copy(centers)
    for i in range(10):
        if radii_opt[i] > 0:
            centers_opt[i, 0] = np.clip(centers[i, 0], radii_opt[i], 1 - radii_opt[i])
            centers_opt[i, 1] = np.clip(centers[i, 1], radii_opt[i], 1 - radii_opt[i])

    sum_radii_opt = np.sum(radii_opt)
    return (centers_opt, radii_opt, sum_radii_opt)
