import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced centers with equal radii
    centers = np.array([[i / 9 + 0.05, j / 9 + 0.05] for i in range(10) for j in range(10)])
    radii = np.full(10, 0.05)

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(centers, radii):
        # Constraints for being inside the unit square
        for i in range(10):
            x, y = centers[i]
            if x < radii[i] or x > 1 - radii[i] or y < radii[i] or y > 1 - radii[i]:
                return False
        # Constraints for non-overlapping circles
        dists = pdist(centers)
        for d, (i, j) in zip(dists, [(a, b) for a in range(10) for b in range(a+1, 10)]):
            if d < radii[i] + radii[j]:
                return False
        return True

    # Optimize
    result = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        bounds=[(0, 1 - 1e-5) for _ in range(10)],
        constraints=[{'type': 'ineq', 'fun': lambda x: 0 if not constraints(centers, x) else 1}]
    )

    # Update centers based on optimal radii
    best_radii = result.x
    best_centers = np.array([[i / 9 + 0.05, j / 9 + 0.05] for i in range(10) for j in range(10)])
    
    # Ensure constraints are satisfied (due to numerical precision issues)
    for i in range(10):
        x, y = best_centers[i]
        if x < best_radii[i] or x > 1 - best_radii[i] or y < best_radii[i] or y > 1 - best_radii[i]:
            best_radii[i] = min(best_radii[i], x, 1 - x, y, 1 - y)
    
    dists = pdist(best_centers)
    for d, (i, j) in zip(dists, [(a, b) for a in range(10) for b in range(a+1, 10)]):
        if d < best_radii[i] + best_radii[j]:
            best_radii[i] = (d - best_radii[j]) / 2
            best_radii[j] = (d - best_radii[i]) / 2
    
    sum_radii = np.sum(best_radii)
    
    return (best_centers, best_radii, sum_radii)
