import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii_flat):
        return -np.sum(radii_flat)

    def constraints(radii_flat):
        # Reshape radii
        radii = radii_flat.reshape(-1)
        # Check if all circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Check if circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist_sq = dx * dx + dy * dy
                if dist_sq < (radii[i] + radii[j]) ** 2:
                    return False
        return True

    # Optimize
    result = minimize(
        objective,
        radii.flatten(),
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        bounds=[(0, 0.5) for _ in range(10)]
    )

    radii = result.x.reshape(-1)
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
