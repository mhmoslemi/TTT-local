import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -sum(radii)

    def constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Constraint 1: circles must be inside the square
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx ** 2 + dy ** 2)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    bounds = []
    for _ in range(10):
        bounds.append((0, 1))  # x coordinate
        bounds.append((0, 1))  # y coordinate
    for _ in range(10):
        bounds.append((0, 0.5))  # radius

    result = minimize(objective, np.concatenate((centers.flatten(), radii)),
                      method='SLSQP',
                      bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints},
                      tol=1e-6)

    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
