import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x[10:])  # Negative because we minimize

    # Bounds for centers (x and y) and radii (non-negative)
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        bounds.append((0.0, 1.0 - 0.001))  # radius (avoiding precision issues)

    # Constraints: circle must be inside the square
    def square_constraints(x):
        constraints = []
        for i in range(10):
            x_i = x[i*3]
            y_i = x[i*3 + 1]
            r_i = x[i*3 + 2]
            constraints.append((x_i - r_i >= 0.0, x_i + r_i <= 1.0,
                               y_i - r_i >= 0.0, y_i + r_i <= 1.0))
        return constraints

    # Constraints: no overlapping between circles
    def circle_constraints(x):
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                x_i = x[i*3]
                y_i = x[i*3 + 1]
                r_i = x[i*3 + 2]
                x_j = x[j*3]
                y_j = x[j*3 + 1]
                r_j = x[j*3 + 2]
                dist = np.hypot(x_i - x_j, y_i - y_j)
                constraints.append((dist >= r_i + r_j,))
        return constraints

    # Combine all constraints
    cons = []
    for c in square_constraints(x):
        cons.append({'type': 'ineq', 'fun': lambda x, *args: args[0], 'args': (c[0],)})
        cons.append({'type': 'ineq', 'fun': lambda x, *args: args[1], 'args': (c[1],)})
        cons.append({'type': 'ineq', 'fun': lambda x, *args: args[2], 'args': (c[2],)})
        cons.append({'type': 'ineq', 'fun': lambda x, *args: args[3], 'args': (c[3],)})
    for c in circle_constraints(x):
        cons.append({'type': 'ineq', 'fun': lambda x, *args: args[0], 'args': (c[0],)})

    # Optimization
    result = minimize(objective, x0=np.ravel([centers, radii]), bounds=bounds, constraints=cons)

    # Extract results
    x_flat = result.x
    centers = np.array([x_flat[i*3:i*3+2] for i in range(10)])
    radii = np.array([x_flat[i*3+2] for i in range(10)])
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
