import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = 0.1 + 0.2 * (i % 5)
        centers[i, 1] = 0.1 + 0.2 * (i // 5)
    radii = np.ones(10) * 0.1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint functions
    def constraint_inside(radii):
        # Constraint: all circles are inside the unit square
        return np.array([r <= 1 - r for r in radii])

    def constraint_overlap(radii):
        # Get pairwise distances
        dists = pdist(centers)
        # Get sum of radii for each pair
        sum_radii = np.array([radii[i] + radii[j] for i, j in zip(*np.triu_indices(10, 1))])
        # Constraint: dist >= sum_radii
        return dists - sum_radii

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
