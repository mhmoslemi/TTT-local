```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a square grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraint_circle_inside(radii, centers):
        return radii - (1 - np.max(centers, axis=1))  # Ensure r <= 1 - x and r <= 1 - y
    
    def constraint_circle_overlap(radii, centers):
        dists = np.linalg.norm(centers[:, np.newaxis, :] - centers[np.newaxis, :, :], axis=2)
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Setup optimization
    bounds = [(0, 1 - 1e-6) for _ in range(10)]
    cons = []
    
    # Constraint for circles inside the square
    cons.append({'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, initial_centers)})
    
    # Constraint for circles not overlapping
    cons.append({'type': 'ineq', 'fun': lambda r: constraint_circle_overlap(r, initial_centers)})
    
    # Optimization
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )
    
    # Extract results
    centers = initial_centers
    radii = result.x
    sum_radii = -result.fun
    
    # Check feasibility (optional for debugging)
    # Ensure all circles are inside the square
    assert np.all(radii <= 1 - np.min(centers, axis=1))
    # Ensure no overlapping
    dists = np.linalg.norm(centers[:, np.newaxis, :] - centers[np.newaxis, :, :], axis=2)
    assert np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    return (centers, radii, sum_radii)
```
