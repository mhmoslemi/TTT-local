import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.15  # Initial guess for radii

    # Objective function: maximize the sum of radii
    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints: circle must be inside the unit square
    def square_constraint(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.minimum(centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat)

    # Non-overlapping constraint: distance between centers >= sum of radii
    def no_overlap_constraint(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        dist = cdist(centers_flat, centers_flat)
        sum_radii = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        return dist - sum_radii

    # Bounds: x and y between 0 and 1, radii non-negative
    bounds = []
    for i in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': no_overlap_constraint}
    ]

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints=cons)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
