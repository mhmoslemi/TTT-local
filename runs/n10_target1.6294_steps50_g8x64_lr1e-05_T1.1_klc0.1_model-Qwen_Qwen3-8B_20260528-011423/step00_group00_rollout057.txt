import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced grid with radii
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)
    
    # Objective function to maximize sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)
    
    # Constraint: each circle is inside the unit square
    def square_inside(x):
        return np.array([x[0] - x[1], 1 - x[0] - x[1], x[2] - x[3], 1 - x[2] - x[3]], dtype=float)
    
    # Constraint: no two circles overlap
    def circle_overlap(x):
        dist = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x[2*i] - x[2*j]
                dy = x[2*i+1] - x[2*j+1]
                dist[i, j] = np.sqrt(dx*dx + dy*dy) - (x[10+i] + x[10+j])
        return np.concatenate(dist.flatten())
    
    # Flatten the parameters
    def objective_flat(x):
        return objective(x[10:])
    
    def square_inside_flat(x):
        return square_inside(x[10:])[0]
    
    def circle_overlap_flat(x):
        return circle_overlap(x[10:])
    
    # Constraints in the form of (type, fun, kwargs)
    constraints = [
        {'type': 'ineq', 'fun': square_inside_flat},
        {'type': 'ineq', 'fun': circle_overlap_flat}
    ]
    
    # Bounds: radii >= 0
    bounds = [(0, None) for _ in range(10)]
    
    # Initial guess for all parameters
    x0 = np.concatenate((centers.flatten(), radii))
    
    # Optimization
    result = minimize(objective_flat, x0, method='SLSQP', bounds=bounds, constraints=constraints)
    
    # Extract the result
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii_result = -objective(radii_result)
    
    return (centers_result, radii_result, sum_radii_result)
