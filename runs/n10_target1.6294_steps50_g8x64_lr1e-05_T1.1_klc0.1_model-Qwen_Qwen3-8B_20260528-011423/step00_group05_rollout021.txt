import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective(x):
        # Flatten x into centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Objective: maximize the sum of radii (minimize negative sum)
        return -np.sum(radii_flat)

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Constraint 1: Circles must be inside the unit square
        # r_i <= x_i <= 1 - r_i for all i
        constraints1 = []
        for i in range(10):
            x_i = centers_flat[i, 0]
            y_i = centers_flat[i, 1]
            r_i = radii_flat[i]
            constraints1.append(x_i - r_i)
            constraints1.append(1 - x_i - r_i)
            constraints1.append(y_i - r_i)
            constraints1.append(1 - y_i - r_i)
        # Constraint 2: Circles must not overlap
        constraints2 = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints2.append(dist - (radii_flat[i] + radii_flat[j]))
        return constraints1 + constraints2

    # Bounds for x and y coordinates (each between 0 and 1)
    x_bounds = [(0, 1) for _ in range(10 * 2)]
    # Bounds for radii (between 0 and 0.5)
    r_bounds = [(0, 0.5) for _ in range(10)]

    # Combine bounds
    all_bounds = x_bounds + r_bounds

    # Optimization
    result = minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=all_bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'ftol': 1e-10, 'maxiter': 1000}
    )

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
