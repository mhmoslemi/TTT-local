import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equal radii and evenly spaced centers
    num_circles = 10
    centers = np.random.rand(num_circles, 2) * 0.8 + 0.1  # Random initial positions inside the square
    radii = np.full(num_circles, 0.1)  # Initial radius
    sum_radii = np.sum(radii)
    
    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize the sum

    def constraints(radii, centers):
        # Constraints: circles must be within bounds
        # r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i
        bounds = np.zeros(num_circles * 2)
        for i in range(num_circles):
            bounds[2*i] = 0
            bounds[2*i + 1] = 0
        return bounds

    def constraint_overlap(radii, centers):
        # Constraint: no two circles can overlap
        # dist(center_i, center_j) >= r_i + r_j
        for i, j in itertools.combinations(range(num_circles), 2):
            dist = np.linalg.norm(centers[i] - centers[j])
            if dist < radii[i] + radii[j] - 1e-8:
                return dist - radii[i] - radii[j]
        return 0  # No overlap

    def optimize():
        # Optimization problem: maximize sum of radii
        bounds = [(0, 1) for _ in range(num_circles)]
        cons = {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers,)}
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)
        if result.success:
            return result.x, centers, result.fun
        else:
            return radii, centers, sum_radii

    # Run the optimization
    radii_opt, centers_opt, sum_radii_opt = optimize()
    
    # Ensure all circles are within the unit square
    for i in range(num_circles):
        x, y = centers_opt[i]
        r = radii_opt[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            return np.array([[0.5, 0.5]] * num_circles), np.zeros(num_circles), 0.0

    return centers_opt, radii_opt, sum_radii_opt
