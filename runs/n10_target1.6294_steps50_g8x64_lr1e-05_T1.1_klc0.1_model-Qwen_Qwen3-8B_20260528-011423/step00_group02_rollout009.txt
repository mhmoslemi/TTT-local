import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers (random points in the unit square)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)  # Initial guess for radii

    def objective(x):
        # x is flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # We minimize -sum_radii to maximize sum_radii

    def constraint_circle_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        # Check if circles are inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i][0], centers_flat[i][1], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1  # Violation
        return 0

    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.hypot(dx, dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1  # Overlap
        return 0

    # Prepare initial x for optimization
    x0 = np.empty(60)
    x0[:60] = np.concatenate([centers.flatten(), radii.flatten()])

    # Set up constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Bounds: x and y must be in [0, 1], radii must be >= 0
    bounds = []
    for i in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])  # x, y, r for each circle

    # Run optimization
    result = minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        tolerance=1e-6
    )

    # Extract results
    x_result = result.x
    centers_result = x_result[:60].reshape(10, 2)
    radii_result = x_result[60:].reshape(10)
    sum_radii_result = -result.fun  # Since we minimized -sum_radii

    return (centers_result, radii_result, sum_radii_result)
