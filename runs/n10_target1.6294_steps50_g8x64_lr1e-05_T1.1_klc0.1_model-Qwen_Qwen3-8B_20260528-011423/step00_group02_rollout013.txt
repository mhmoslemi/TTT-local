import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles evenly spaced with small radius
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)
    
    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum
    
    def constraints(radii, centers):
        cons = []
        # Constraint: radii must be non-negative
        for i in range(10):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        # Constraint: circles must be inside the unit square
        for i in range(10):
            def inside(i=i):
                x, y = centers[i]
                return r[i] <= x - r[i] and r[i] <= y - r[i] and x + r[i] <= 1 - r[i] and y + r[i] <= 1 - r[i]
            cons.append({'type': 'ineq', 'fun': inside})
        # Constraint: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                def no_overlap(i=i, j=j):
                    dx = centers[i][0] - centers[j][0]
                    dy = centers[i][1] - centers[j][1]
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - (radii[i] + radii[j])
                cons.append({'type': 'ineq', 'fun': no_overlap})
        return cons
    
    # Flatten centers and radii for optimization
    x0 = np.concatenate((centers.flatten(), radii))
    
    def objective_flat(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)
    
    def constraints_flat(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        cons = []
        # Non-negative radii
        for i in range(10):
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: x[60 + i]})
        # Circles inside the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            def inside(x, i=i):
                return x[60 + i] <= x_i - x[60 + i] and x[60 + i] <= y_i - x[60 + i] and x_i + x[60 + i] <= 1 - x[60 + i] and y_i + x[60 + i] <= 1 - x[60 + i]
            cons.append({'type': 'ineq', 'fun': inside})
        # Circles not overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                x_i, y_i = centers_flat[i]
                x_j, y_j = centers_flat[j]
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                def no_overlap(x, i=i, j=j):
                    dx = x_i - x_j
                    dy = y_i - y_j
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - (r_i + r_j)
                cons.append({'type': 'ineq', 'fun': no_overlap})
        return cons
    
    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective_flat, x0, method='SLSQP', constraints=constraints_flat, tol=1e-6)
    
    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
