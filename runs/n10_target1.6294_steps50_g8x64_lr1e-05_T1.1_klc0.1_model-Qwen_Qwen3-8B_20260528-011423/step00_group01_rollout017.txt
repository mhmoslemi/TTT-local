import numpy as np
from scipy.optimize import minimize
import os

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((10, 2))
    radii = np.ones(10) * 0.1
    for i in range(10):
        centers[i, 0] = (i % 5) * 0.3 + 0.1
        centers[i, 1] = (i // 5) * 0.3 + 0.1

    def objective(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize sum_radii

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)

        # Constraint 1: All circles are within the unit square
        for i in range(10):
            r = radii_flat[i]
            x_i, y_i = centers_flat[i]
            if not (r <= x_i <= 1 - r and r <= y_i <= 1 - r):
                return np.inf
        # Constraint 2: No overlap between any two circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0

    # Initial guess
    x0 = np.concatenate([centers.reshape(10 * 2), radii])

    # Optimization
    result = minimize(objective, x0, method='L-BFGS-B', bounds=[(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)], constraints={'type': 'ineq', 'fun': constraints})

    # Extract result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
