import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii (simple grid layout)
    centers = np.random.rand(10, 2) * 0.8 + 0.1  # Random positions within the square
    radii = np.random.rand(10) * 0.1  # Initial small radii

    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize -sum_radii to maximize sum_radii

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        constraints = []

        # Ensure circles are inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i][0], centers_flat[i][1], radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                constraints.append(1.0)
            else:
                constraints.append(0.0)

        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.hypot(dx, dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j:
                    constraints.append(1.0)
                else:
                    constraints.append(0.0)

        return constraints

    # Flatten the initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization bounds: each circle's center is in [0,1] and radius is >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])  # radius max is 0.5 to stay inside square

    # Create constraints as a list of inequality constraints
    cons = []
    for i in range(60):
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: constraints(x)[i]})

    # Run the optimization
    result = minimize(objective, x0=x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract the result
    optimized_centers = result.x[:60].reshape(10, 2)
    optimized_radii = result.x[60:]
    sum_radii = -result.fun  # Since we minimize -sum_radii

    return (optimized_centers, optimized_radii, sum_radii)
