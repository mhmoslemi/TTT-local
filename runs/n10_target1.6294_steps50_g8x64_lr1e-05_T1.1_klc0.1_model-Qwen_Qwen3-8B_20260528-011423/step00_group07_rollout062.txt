import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

    def constraints(radii_vec):
        # Ensure all circles are within the unit square
        for i in range(10):
            if radii_vec[i] < 0:
                return False
            if radii_vec[i] > 1 - 2 * np.min(centers[i])
                return False
        # Ensure circles do not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return False
        return True

    def objective_wrapper(x):
        if not constraints(x):
            return float('inf')  # Penalize invalid solutions
        return -np.sum(x)

    result = minimize(objective_wrapper, radii, method='L-BFGS-B', bounds=[(0, 1) for _ in range(10)])
    radii_result = result.x
    centers_result = centers
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
