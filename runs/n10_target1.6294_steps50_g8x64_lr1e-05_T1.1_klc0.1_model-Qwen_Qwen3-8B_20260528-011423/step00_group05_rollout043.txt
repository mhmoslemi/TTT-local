import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Boundary constraint: circles must fit inside unit square
        def boundary_constraint(x):
            return np.minimum(x - radii, 1 - x - radii)
        return np.concatenate([np.sum(boundary_constraint(centers[:, 0]), axis=0),
                               np.sum(boundary_constraint(centers[:, 1]), axis=0)])

    def non_overlap_constraint(radii, centers):
        # Non-overlap constraint: distance between any two centers >= sum of their radii
        n = len(radii)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    # Optimization
    initial_guess = np.concatenate([centers.flatten(), radii])
    bounds = [(0, 0.5) for _ in range(len(initial_guess))]

    # Objective function (negative sum of radii)
    def objective_func(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return -np.sum(radii)

    # Constraints
    def constraints_func(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Boundary constraints
        boundary_con = np.concatenate([np.minimum(centers[:, 0] - radii, 1 - centers[:, 0] - radii),
                                       np.minimum(centers[:, 1] - radii, 1 - centers[:, 1] - radii)])
        # Non-overlap constraints
        nonoverlap_con = []
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                nonoverlap_con.append(dist - (radii[i] + radii[j]))
        return np.concatenate([boundary_con, nonoverlap_con])

    # Optimization
    result = minimize(objective_func, initial_guess, bounds=bounds, constraints={'type': 'ineq', 'fun': constraints_func})
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
