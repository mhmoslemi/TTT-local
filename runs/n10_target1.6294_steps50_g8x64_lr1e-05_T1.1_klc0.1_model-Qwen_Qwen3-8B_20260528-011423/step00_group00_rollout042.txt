import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: distribute circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radii: same for simplicity
    radii = np.full(10, 0.15)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[10:])

    # Constraints: circle must be inside the square
    def square_constraint(x):
        r = x[10:]
        return np.ones_like(r) * (1 - 2 * r)  # (1 - 2r) >= 0 => r <= 0.5

    # Constraints: distance between centers should be >= sum of radii
    def circle_constraints(x):
        centers = x[:10].reshape(10, 2)
        radii = x[10:]
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        r_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dist - r_sum

    # Bounds: radii must be non-negative and at most 0.5
    bounds = [(0, 0.5)] * 10

    # Initial guess with centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': circle_constraints}
    ]

    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-8)

    # Extract results
    centers_opt = result.x[:10 * 2].reshape(10, 2)
    radii_opt = result.x[10 * 2:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
