import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles placed in grid pattern
    centers = np.array([[i * 0.25, j * 0.25] for j in range(2) for i in range(5)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are inside the unit square
        centers = np.array([[i * 0.25, j * 0.25] for j in range(2) for i in range(5)])
        for i in range(10):
            x, y = centers[i]
            if radii[i] > 0:
                if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                    return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                x1, y1 = centers[i]
                x2, y2 = centers[j]
                dist = np.hypot(x1 - x2, y1 - y2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    bounds = [(0.0, 0.5) for _ in range(10)]  # Max radius limited to 0.5 for initial guess

    result = minimize(objective, radii, bounds=bounds, constraints={"type": "ineq", "fun": constraints})

    # Final centers and radii
    centers = np.array([[i * 0.25, j * 0.25] for j in range(2) for i in range(5)])
    sum_radii = result.fun * -1

    return (centers, result.x, sum_radii)
