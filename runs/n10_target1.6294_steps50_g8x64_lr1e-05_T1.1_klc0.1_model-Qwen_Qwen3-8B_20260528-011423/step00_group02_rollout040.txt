import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced points with small radii
    centers = np.array([[0.1 + i*0.1, 0.1 + j*0.1] for i in range(10) for j in range(1)])
    radii = np.full(10, 0.05)
    
    def objective(radii):
        return -sum(radii)
    
    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        # Each circle's radius must be <= 1 - max(center_x, center_y)
        for i in range(10):
            x, y = centers[i]
            if radii[i] > 1 - max(x, y) or radii[i] > 1 - min(x, y):
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Define optimization bounds
    bounds = [(0.0, 1.0 - 1e-6) for _ in range(10)]
    
    # Use minimize to maximize the sum of radii
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        method='SLSQP'
    )
    
    # Extract result
    optimized_radii = result.x
    sum_radii = -result.fun
    
    # Return centers (unchanged), radii, and sum of radii
    return (centers, optimized_radii, sum_radii)
