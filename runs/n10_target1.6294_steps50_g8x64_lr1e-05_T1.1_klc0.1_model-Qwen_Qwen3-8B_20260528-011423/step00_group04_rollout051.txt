import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: each circle must fit inside the unit square
    def square_constraint(radii):
        return np.array([np.min(centers[i, 0] - radii[i]) for i in range(10)] +
                        [np.min(centers[i, 1] - radii[i]) for i in range(10)] +
                        [np.min(1 - centers[i, 0] - radii[i]) for i in range(10)] +
                        [np.min(1 - centers[i, 1] - radii[i]) for i in range(10)])

    # Constraint: no two circles overlap
    def no_overlap_constraint(radii):
        dists = np.zeros(10 * 10)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[10 * i + j] = np.sqrt(dx**2 + dy**2) - radii[i] - radii[j]
        return dists

    # Optimization problem
    bounds = [(0.0, 1.0) for _ in range(10)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: square_constraint(r)},
        {'type': 'ineq', 'fun': lambda r: no_overlap_constraint(r)}
    ]

    result = minimize(objective, centers.flatten(), bounds=bounds, constraints=cons, method='SLSQP')

    # Extract results
    radii = result.x
    centers_opt = np.reshape(centers, (10, 2))
    centers_opt = np.where(centers_opt <= 1 - radii[:, None], centers_opt, 1 - radii[:, None])
    centers_opt = np.where(centers_opt >= radii[:, None], centers_opt, radii[:, None])
    sum_radii = np.sum(radii)

    return (centers_opt, radii, sum_radii)
