import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: Circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - np.min(centers[i, :]):
                return False
        # Constraint 2: No overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define optimization problem
    bounds = [(0.0, 1.0 - 1e-5) for _ in range(10)]  # Small margin to avoid edge issues

    # Use a wrapper to handle constraints
    def optimize():
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            result = minimize(
                objective,
                radii,
                method='L-BFGS-B',
                bounds=bounds,
                options={'ftol': 1e-8, 'gtol': 1e-8, 'maxiter': 1000}
            )
        return result

    # Run optimization
    result = optimize()
    radii = result.x
    centers = np.array([[0.5, 0.5], [0.5, 0.25], [0.5, 0.75], [0.25, 0.5], [0.75, 0.5],
                       [0.25, 0.25], [0.25, 0.75], [0.75, 0.25], [0.75, 0.75], [0.5, 0.5]])

    # Check constraints and adjust
    while not constraints(radii):
        # If constraints are violated, reset and try another optimization
        # This is a simplified approach for the sake of example
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
        radii = np.ones(10) * 0.1
        result = optimize()

    sum_radii = np.sum(radii)
    return centers, radii, sum_radii
