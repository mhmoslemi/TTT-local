import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for the centers and radii
    # Start with equal radii and evenly spaced in a grid
    n = 10
    centers = np.zeros((n, 2))
    radii = np.full(n, 0.1)

    # Define the objective function to minimize negative sum of radii
    def objective(x):
        # x is a flat array: [x0, y0, r0, x1, y1, r1, ..., xn-1, yn-1, rn-1]
        centers_flat = x[:-n]
        radii_flat = x[-n:]
        centers = centers_flat.reshape(n, 2)
        radii = radii_flat

        # Objective: maximize sum of radii, so minimize negative of it
        return -np.sum(radii)

    # Define constraints: 1. Circles are inside the square
    # 2. Circles do not overlap
    cons = []

    # Constraint 1: Each circle must be inside the unit square
    for i in range(n):
        # x_i - r_i >= 0
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: x[-n + i] - x[-n + i + n]})
        # 1 - x_i - r_i >= 0
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - x[-n + i] - x[-n + i + n]})
        # y_i - r_i >= 0
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: x[-n + i + 1] - x[-n + i + n + 1]})
        # 1 - y_i - r_i >= 0
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - x[-n + i + 1] - x[-n + i + n + 1]])

    # Constraint 2: No two circles overlap
    for i in range(n):
        for j in range(i + 1, n):
            def constraint(x, i=i, j=j):
                x1, y1, r1 = x[-n + i], x[-n + i + 1], x[-n + i + n]
                x2, y2, r2 = x[-n + j], x[-n + j + 1], x[-n + j + n]
                return math.dist([x1, y1], [x2, y2]) - r1 - r2
            cons.append({'type': 'ineq', 'fun': constraint})

    # Initial guess flattened
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization options
    options = {'disp': False, 'ftol': 1e-10, 'maxiter': 1000}

    # Run the optimization
    result = minimize(objective, x0, constraints=cons, method='SLSQP', options=options)

    # Extract the optimized centers and radii
    optimized_x = result.x
    centers_opt = optimized_x[:-n].reshape(n, 2)
    radii_opt = optimized_x[-n:]

    # Calculate sum of radii
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
