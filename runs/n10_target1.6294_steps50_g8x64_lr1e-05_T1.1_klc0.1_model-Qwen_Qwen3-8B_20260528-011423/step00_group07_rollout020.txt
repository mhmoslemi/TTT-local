import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: place circles in a grid with equal spacing
    initial_centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    initial_radii = np.ones(9) * 0.1  # 9 circles in a 3x3 grid
    initial_centers = np.vstack([initial_centers, [0.5, 0.5]])  # 10th circle in the center
    initial_radii = np.append(initial_radii, 0.1)

    def objective_function(params):
        centers = params[:20].reshape(10, 2)
        radii = params[20:]

        # Check if all circles are inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf

        # Check if circles are non-overlapping
        dists = pdist(centers)
        for d, i, j in zip(dists, range(len(dists)), range(len(dists))):
            if d < radii[i] + radii[j]:
                return np.inf

        return -np.sum(radii)

    def constraint_function(params):
        centers = params[:20].reshape(10, 2)
        radii = params[20:]

        # Check if all circles are inside the unit square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf

        # Check if circles are non-overlapping
        dists = pdist(centers)
        for d, i, j in zip(dists, range(len(dists)), range(len(dists))):
            if d < radii[i] + radii[j]:
                return np.inf

        return 0

    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]

    result = minimize(objective_function, np.concatenate([initial_centers.flatten(), initial_radii]),
                      method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_function},
                      tol=1e-8)

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
