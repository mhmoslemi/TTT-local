import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: regular grid with some spacing
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)  # We minimize the negative sum to maximize the sum

    def constraints(centers, radii):
        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii), axis=1)
        if not np.all(inside):
            return np.inf
        # Constraint 2: No two circles overlap
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0]) ** 2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1]) ** 2))
        dists[np.diag_indices(10)] = np.inf
        min_dist = np.min(dists)
        if min_dist < np.min(radii) + np.min(radii):
            return np.inf
        return 0

    def objective_and_constraints(radii, centers):
        obj = -np.sum(radii)
        cons = constraints(centers, radii)
        return obj, cons

    def objective_wrapper(params):
        radii = params
        obj, cons = objective_and_constraints(centers, radii)
        return obj

    def constraints_wrapper(params):
        radii = params
        obj, cons = objective_and_constraints(centers, radii)
        return [cons]

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(10)]
    result = minimize(objective_wrapper, radii, method='SLSQP', bounds=bounds, constraints=constraints_wrapper, tol=1e-6)

    # Extract results
    radii_opt = result.x
    centers_opt = centers
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
