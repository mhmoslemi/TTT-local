import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # All radii start small

    def objective(radii):
        return -np.sum(radii)  # Want to maximize sum of radii

    def constraints(radii, centers):
        # Constraint to ensure all circles are inside the unit square
        # r <= x <= 1 - r and r <= y <= 1 - r
        return np.array([radii[i] <= 1 - centers[i, 0] for i in range(10)] +
                        [radii[i] <= centers[i, 0] for i in range(10)] +
                        [radii[i] <= 1 - centers[i, 1] for i in range(10)] +
                        [radii[i] <= centers[i, 1] for i in range(10)])

    def non_overlap_constr(radii, centers):
        # Constraint to ensure no two circles overlap
        # Distance between centers >= sum of radii
        dist_mat = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dist_mat - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization
    bounds = [(0, 1) for _ in range(10)]  # All radii must be non-negative

    res = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraints(r, centers)},
            {'type': 'ineq', 'fun': lambda r: non_overlap_constr(r, centers)}
        ],
        jac=lambda r: -np.ones_like(r)  # Negative gradient for maximization
    )

    # Update centers based on optimized radii
    # Use a random initial guess if the solution is invalid
    if res.success:
        # Refine centers to prevent overlap and stay within bounds
        centers = np.random.rand(10, 2)
        new_radii = res.x
        for i in range(10):
            x, y = centers[i]
            new_radii[i] = min(new_radii[i], x, 1 - x, y, 1 - y)
            for j in range(i + 1, 10):
                dx = abs(centers[i, 0] - centers[j, 0])
                dy = abs(centers[i, 1] - centers[j, 1])
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < new_radii[i] + new_radii[j]:
                    new_radii[i] = (dist - new_radii[j]) * 0.9
                    new_radii[j] = (dist - new_radii[i]) * 0.9
        sum_radii = np.sum(new_radii)
    else:
        # Fallback to random placement if optimization fails
        centers = np.random.rand(10, 2)
        new_radii = np.random.rand(10) * 0.1
        for i in range(10):
            new_radii[i] = min(new_radii[i], centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1])
        sum_radii = np.sum(new_radii)

    return (centers, new_radii, sum_radii)
