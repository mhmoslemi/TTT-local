import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: 10 circles evenly spaced in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all equal
    radii = np.ones(10) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii, centers):
        constraints = []
        # Constraint: each circle is inside the unit square
        for i in range(10):
            r = radii[i]
            if r > 0:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (centers[i, 0] - r)})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 0] - r})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (centers[i, 1] - r)})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 1] - r})
        # Constraint: no two circles overlap
        dists = pdist(centers)
        for d, i, j in zip(dists, range(len(dists)), range(len(dists))):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: d - (r[i] + r[j])})
        return constraints

    def objective_func(x):
        return -np.sum(x)

    def constraint_func(x):
        constraints = []
        # Circle inside the square
        for i in range(10):
            r = x[i]
            if r > 0:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (centers[i, 0] - r)})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 0] - r})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (centers[i, 1] - r)})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 1] - r})
        # No overlapping
        dists = pdist(centers)
        for d, i, j in zip(dists, range(len(dists)), range(len(dists))):
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: d - (x[i] + x[j])})
        return constraints

    # Initial guess
    x0 = radii.copy()

    # Optimization
    result = minimize(objective_func, x0, constraints=constraint_func(x0), method='SLSQP', tol=1e-8)

    # Extract result
    radii_opt = result.x
    sum_radii = -result.fun
    centers_opt = centers.copy()

    return (centers_opt, radii_opt, sum_radii)
