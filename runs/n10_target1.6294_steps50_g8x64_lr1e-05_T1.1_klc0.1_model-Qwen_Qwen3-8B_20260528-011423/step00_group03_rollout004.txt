import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimizer settings
    bounds = [(0.0, 0.5) for _ in range(10)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=[lambda x: constraints(x)])

    # Final centers and radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x

    for i in range(10):
        x, y = centers[i]
        if radii[i] > x or radii[i] > 1 - x or radii[i] > y or radii[i] > 1 - y:
            radii[i] = min(radii[i], x, 1 - x, y, 1 - y)

    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
