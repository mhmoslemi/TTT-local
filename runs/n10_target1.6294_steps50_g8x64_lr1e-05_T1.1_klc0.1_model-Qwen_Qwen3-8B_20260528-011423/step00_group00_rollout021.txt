import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles placed at the corners and centers of a grid
    centers = np.array([
        [0.1, 0.1], [0.1, 0.5], [0.1, 0.9],
        [0.5, 0.1], [0.5, 0.5], [0.5, 0.9],
        [0.9, 0.1], [0.9, 0.5], [0.9, 0.9],
        [0.5, 0.3]
    ])
    radii = np.array([0.1] * 10)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        # All circles are fully inside the unit square
        for r in radii:
            if r > 0.5:
                return False
        return True

    def bounds(radii):
        # Radii must be non-negative and <= 0.5
        return [(0, 0.5) for _ in range(10)]

    def inequality_constraints(center_vars):
        centers = center_vars[:20].reshape(10, 2)
        radii = center_vars[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Flatten centers and radii for optimization
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization setup
    result = minimize(
        fun=lambda x: -np.sum(x[20:]),  # Objective: maximize sum of radii
        x0=initial_guess,
        bounds=bounds(radii),
        constraints=[
            {'type': 'ineq', 'fun': inequality_constraints, 'jac': None}
        ]
    )

    # Extract optimal centers and radii
    optimal_centers = result.x[:20].reshape(10, 2)
    optimal_radii = result.x[20:]

    # Ensure radii satisfy the unit square constraint
    for i in range(10):
        r = optimal_radii[i]
        if r > 0.5:
            optimal_radii[i] = 0.5

    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
