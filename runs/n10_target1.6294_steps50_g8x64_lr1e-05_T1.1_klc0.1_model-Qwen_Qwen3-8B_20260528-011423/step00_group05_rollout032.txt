import numpy as np
from scipy.optimize import minimize
import numpy.random as npr

def run_packing():
    # Initial guess: random positions and equal radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square and not overlapping
    def constraint_inside(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([1 - r - x[i, 0] for i, r in enumerate(radii)] +
                        [1 - r - x[i, 1] for i, r in enumerate(radii)] +
                        [x[i, 0] - r for i, r in enumerate(radii)] +
                        [x[i, 1] - r for i, r in enumerate(radii)])

    def constraint_overlap(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        dists = np.zeros(45)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                idx += 1
        return dists

    # Bounds: each x and y is between 0 and 1, each radius is non-negative
    bounds = []
    for i in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])
    
    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii]),
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
