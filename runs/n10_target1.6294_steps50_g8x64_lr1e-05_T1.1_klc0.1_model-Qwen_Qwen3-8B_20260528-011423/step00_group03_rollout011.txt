import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    # Start with circles placed in a grid pattern
    centers = np.zeros((10, 2))
    radii = np.zeros(10)
    for i in range(10):
        row = i // 3
        col = i % 3
        centers[i, 0] = col + 0.5
        centers[i, 1] = row + 0.5
        radii[i] = 0.1

    # Objective function to minimize the negative of the sum of radii
    def objective(x):
        # Flatten input for easier handling
        x = x.reshape(-1, 2)
        r = x[:, 0]  # radii
        centers = x[:, 1:]  # centers
        sum_radii = np.sum(r)
        # Check if all circles fit inside the unit square
        if np.any(r < 0) or np.any((centers[:, 0] - r) < 0) or np.any((centers[:, 1] - r) < 0) or np.any((1 - centers[:, 0] - r) < 0) or np.any((1 - centers[:, 1] - r) < 0):
            return np.inf
        # Check if circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < r[i] + r[j]:
                    return np.inf
        return -sum_radii  # minimize the negative to maximize the sum

    # Bounds for each parameter: [r, x, y]
    bounds = []
    for _ in range(10):
        bounds.append((0.0, 0.5))  # r in [0, 0.5]
        bounds.append((0.0, 1.0))  # x in [0, 1]
        bounds.append((0.0, 1.0))  # y in [0, 1]

    # Bounds for the first iteration to avoid divergence
    initial_guess = []
    for i in range(10):
        initial_guess.extend([radii[i], centers[i, 0], centers[i, 1]])

    # Solve optimization
    result = minimize(objective, initial_guess, bounds=bounds, method='L-BFGS-B', options={'ftol': 1e-8})

    # Extract results
    x = result.x
    centers = np.zeros((10, 2))
    radii = np.zeros(10)
    for i in range(10):
        radii[i] = x[3 * i]
        centers[i, 0] = x[3 * i + 1]
        centers[i, 1] = x[3 * i + 2]

    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
