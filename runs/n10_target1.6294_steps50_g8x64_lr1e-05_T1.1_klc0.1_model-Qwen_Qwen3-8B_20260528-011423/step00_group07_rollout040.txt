import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid-like formation
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.array([0.1] * 10)

    def objective_function(x):
        # Unpack variables into centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Objective: maximize the sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraint_inside(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        return np.array([r - (1 - np.sqrt((centers[i, 0] - 0)**2 + (centers[i, 1] - 0)**2)) for i, r in enumerate(radii)]) \
               + np.array([r - (1 - np.sqrt((centers[i, 0] - 1)**2 + (centers[i, 1] - 0)**2)) for i, r in enumerate(radii)]) \
               + np.array([r - (1 - np.sqrt((centers[i, 0] - 0)**2 + (centers[i, 1] - 1)**2)) for i, r in enumerate(radii)]) \
               + np.array([r - (1 - np.sqrt((centers[i, 0] - 1)**2 + (centers[i, 1] - 1)**2)) for i, r in enumerate(radii)])

    def constraint_overlap(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(radii[i] + radii[j] - dist)
        return np.array(constraints)

    def constraint_non_negative(x):
        return x[20:]

    # Set up optimization problem
    bounds = []
    for _ in range(20):
        bounds.append((0.0, 1.0))  # bounds for x and y coordinates
    for _ in range(10):
        bounds.append((0.0, 1.0))  # bounds for radii

    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]

    # Run optimization
    result = minimize(objective_function, x0=np.concatenate((initial_centers.ravel(), initial_radii)), bounds=bounds, constraints=cons, method='SLSQP')

    # Extract results
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
