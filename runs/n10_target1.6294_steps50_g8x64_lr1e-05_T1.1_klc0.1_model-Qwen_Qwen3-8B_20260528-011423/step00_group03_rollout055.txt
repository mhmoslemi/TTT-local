import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers and radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial radii smaller than 0.1 to avoid overlap

    def objective(radii, centers):
        # Objective function: maximize the sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraints: all circles inside the unit square
        def inside(r, c):
            return (c[0] - r >= 0) and (c[1] - r >= 0) and (1 - c[0] - r >= 0) and (1 - c[1] - r >= 0)

        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            if not inside(radii[i], centers[i]):
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization parameters
    bounds = [(0, 0.5) for _ in range(10)]  # Radii can't exceed 0.5

    # Optimization using scipy.optimize.minimize
    result = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: np.array([constraints(x, centers)])}
        ],
        method='SLSQP'
    )

    # Extract the final result
    centers_opt = np.copy(centers)
    radii_opt = result.x
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
