import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (random positions in the unit square)
    centers = np.random.rand(10, 2)
    radii = np.zeros(10) + 0.1  # Initial guess for radii

    def objective_function(params):
        # Unpack parameters into centers and radii
        flatten = params.reshape(-1)
        centers_flat = flatten[:20]
        radii_flat = flatten[20:]
        centers = centers_flat.reshape(10, 2)
        radii = radii_flat

        # Compute sum of radii
        sum_radii = np.sum(radii)

        # Objective is to minimize the negative of sum of radii (maximize sum)
        return -sum_radii

    def constraints(params):
        flatten = params.reshape(-1)
        centers_flat = flatten[:20]
        radii_flat = flatten[20:]
        centers = centers_flat.reshape(10, 2)
        radii = radii_flat

        # Constraint 1: Each circle is fully inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -np.inf  # Invalid configuration

        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf  # Invalid configuration

        # All constraints satisfied
        return 0

    # Optimization bounds
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y for each center
        bounds.extend([(0.0, 0.5)])  # radius for each circle (max 0.5 to fit inside square)

    # Optimization
    result = minimize(objective_function, params=np.concatenate([centers.flatten(), radii]), 
                      method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Extract the result
    params = result.x
    centers = params[:20].reshape(10, 2)
    radii = params[20:]
    sum_radii = -result.fun

    # Visualization (optional)
    plt.figure(figsize=(6, 6))
    for i in range(10):
        circle = plt.Circle(centers[i], radii[i], edgecolor='r', facecolor='none')
        plt.gca().add_patch(circle)
    plt.xlim(0, 1)
    plt.ylim(0, 1)
    plt.axis('equal')
    plt.title(f"Sum of Radii: {sum_radii:.4f}")
    plt.show()

    return centers, radii, sum_radii
