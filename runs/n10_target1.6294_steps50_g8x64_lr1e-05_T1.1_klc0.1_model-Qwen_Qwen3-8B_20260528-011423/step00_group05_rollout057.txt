import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.array([r <= 1 - r for r in radii])

    def constraint_no_overlap(radii, centers):
        n = len(centers)
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraint_non_negative(radii):
        return radii

    # Optimization setup
    bounds = [(0, 1 - 1e-6) for _ in range(10)]

    # Initial constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x, centers)},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]

    # Minimize negative of objective to maximize radii sum
    result = minimize(
        objective,
        radii,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=constraints
    )

    # Update centers based on the optimal radii
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Final radii
    final_radii = result.x
    sum_radii = np.sum(final_radii)

    return (centers, final_radii, sum_radii)
