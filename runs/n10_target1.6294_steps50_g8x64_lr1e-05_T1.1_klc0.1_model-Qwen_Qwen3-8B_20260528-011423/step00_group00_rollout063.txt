import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place 10 circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum
    
    def constraints_func(x):
        # x is a flat array of centers and radii
        centers = x[:10*2].reshape(10, 2)
        radii = x[10*2:]
        
        # Constraint 1: All circles are within the unit square
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                r_i = radii[i]
                r_j = radii[j]
                if dist < r_i + r_j:
                    return np.inf
        
        return -np.sum(radii)  # Return negative sum to minimize
    
    # Objective function based on the flat x array
    def objective_func(x):
        return constraints_func(x)
    
    # Bounds for centers and radii
    bounds = []
    for _ in range(10):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
        bounds.append((0, 0.5))  # radius (max possible is 0.5)
    
    # Optimization settings
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective_func, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds)
    
    centers_opt = result.x[:10*2].reshape(10, 2)
    radii_opt = result.x[10*2:]
    sum_radii_opt = -result.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
