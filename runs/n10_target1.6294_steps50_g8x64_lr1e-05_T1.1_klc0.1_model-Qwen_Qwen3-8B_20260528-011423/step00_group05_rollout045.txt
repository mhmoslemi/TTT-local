import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: randomly place 10 circles in the unit square
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective_function(params):
        # params is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        # Constraint: circles must be inside the unit square
        for i in range(10):
            x, y, r = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
        # Constraint: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return -sum_radii  # Minimize negative of sum to maximize it

    # Flatten centers and radii for optimization
    initial_guess = np.r_[centers.reshape(10 * 2), radii]
    # Optimization with bounds: centers between 0 and 1, radii between 0 and 0.5
    bounds = [(0, 1) for _ in range(10 * 2)] + [(0, 0.5) for _ in range(10)]
    result = minimize(objective_function, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-8)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii_opt = -result.fun  # Convert back to sum of radii

    return (centers_opt, radii_opt, sum_radii_opt)
