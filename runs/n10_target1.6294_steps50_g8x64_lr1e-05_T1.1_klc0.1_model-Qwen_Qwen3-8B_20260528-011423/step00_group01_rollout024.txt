import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.spatial.distance import cdist

def run_packing():
    def objective(params):
        # Reshape parameters to centers and radii
        centers = params[:60].reshape(10, 2)
        radii = params[60:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize the negative sum to maximize the sum

    def constraints(params):
        centers = params[:60].reshape(10, 2)
        radii = params[60:]
        # Constraint 1: all circles are inside the unit square
        inside = np.all((centers - radii[:, np.newaxis]) >= 0, axis=1) & \
                 np.all((1 - centers - radii[:, np.newaxis]) >= 0, axis=1)
        if not np.all(inside):
            return np.inf
        
        # Constraint 2: no two circles overlap
        dist_matrix = cdist(centers, centers)
        for i in range(10):
            for j in range(i + 1, 10):
                if dist_matrix[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Initial guess: circle centers at grid points, radii all 0.1
    initial_centers = np.array([[i * 0.2, j * 0.2] for i in range(5) for j in range(5)])
    initial_radii = np.full(10, 0.1)
    params_initial = initial_centers.flatten() + initial_radii

    bounds = Bounds(
        lb=np.concatenate([np.zeros(20), np.zeros(10)]),
        ub=np.concatenate([np.ones(20), np.ones(10)])
    )

    result = minimize(
        objective,
        params_initial,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
