import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles at a grid
    np.random.seed(0)
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1

    # Objective function: minimize negative sum of radii
    def objective(x):
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints
    def constraint1(x):
        # All circles are inside the unit square
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        left = centers[:, 0] - radii
        right = 1 - centers[:, 0] - radii
        bottom = centers[:, 1] - radii
        top = 1 - centers[:, 1] - radii
        return np.concatenate([left, right, bottom, top])

    def constraint2(x):
        # No two circles overlap
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Bounds: radii >= 0, centers between 0 and 1
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # radius and x, y

    # Optimization
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]
    result = minimize(objective, np.concatenate([radii, centers.flatten()]), method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    # Extract results
    radii_result = result.x[::2]
    centers_result = result.x[1::2].reshape(10, 2)
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
