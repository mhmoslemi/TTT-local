import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(x, radii):
        # Constraints: circles are inside the unit square
        for i in range(10):
            if x[i*2] < radii[i] or x[i*2] > 1 - radii[i] or x[i*2 + 1] < radii[i] or x[i*2 + 1] > 1 - radii[i]:
                return True
        # Constraints: circles do not overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = x[i*2] - x[j*2]
                dy = x[i*2 + 1] - x[j*2 + 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return True
        return False

    initial_centers = np.random.rand(10, 2)
    initial_radii = np.random.rand(10)
    initial_radii = np.clip(initial_radii, 0, 0.5)

    bounds = []
    for i in range(10):
        bounds.append((0, 1))
        bounds.append((0, 1))
    bounds += [(0, 0.5)] * 10

    result = minimize(objective, x0=np.concatenate((initial_centers.reshape(20), initial_radii)), bounds=bounds, method='SLSQP', constraints=(lambda x: -constraints(x, x[10:20])))

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[10:20]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
