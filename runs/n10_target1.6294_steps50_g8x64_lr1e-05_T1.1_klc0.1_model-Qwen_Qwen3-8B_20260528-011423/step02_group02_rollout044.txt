import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint for square boundaries: r <= x <= 1 - r and r <= y <= 1 - r
    def constraint_square(radii, centers):
        constraints = []
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            constraints.append(r - x)
            constraints.append(x - (1 - r))
            constraints.append(r - y)
            constraints.append(y - (1 - r))
        return np.array(constraints)

    # Constraint for non-overlapping circles: distance >= sum of radii
    def constraint_overlap(radii, centers):
        n = len(radii)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Initial guess for radii
    initial_guess = np.ones(10) * 0.1

    # Create constraint for square boundaries
    square_constraint = {'type': 'ineq', 'fun': constraint_square, 'args': (centers,)}

    # Create constraint for non-overlapping circles
    overlap_constraint = {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers,)}

    # Optimization setup
    result = minimize(
        objective,
        initial_guess,
        bounds=bounds,
        constraints=[square_constraint, overlap_constraint],
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    radii = result.x
    centers_result = centers.copy()
    sum_radii = np.sum(radii)

    # Final check: ensure all circles are within the square and non-overlapping
    for i in range(10):
        x, y = centers_result[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            warnings.warn("One or more circles are not fully inside the square.", UserWarning)

    return (centers_result, radii, sum_radii)
