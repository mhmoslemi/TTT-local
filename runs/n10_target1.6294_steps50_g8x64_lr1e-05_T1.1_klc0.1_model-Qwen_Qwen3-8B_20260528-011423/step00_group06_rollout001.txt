```python
import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equal radii and evenly spaced centers
    initial_centers = np.array([
        [0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],
        [0.3, 0.1], [0.3, 0.3], [0.3, 0.5], [0.3, 0.7], [0.3, 0.9]
    ])
    initial_radii = np.full(10, 0.05)

    def objective_function(params):
        # Unpack parameters into centers and radii
        params = params.reshape(20)
        centers = params[:20].reshape(10, 2)
        radii = params[20:]

        # Objective is to maximize sum of radii, so minimize negative sum
        sum_radii = np.sum(radii)
        return -sum_radii

    def constraint_inside(params):
        params = params.reshape(20)
        centers = params[:20].reshape(10, 2)
        radii = params[20:]
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -1.0
        return 0.0

    def constraint_overlap(params):
        params = params.reshape(20)
        centers = params[:20].reshape(10, 2)
        radii = params[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -1.0
        return 0.0

    # Set up optimization problem
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y for each center
    bounds.extend([(0.0, 0.5) for _ in range(10)])  # radii cannot be larger than 0.5

    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(
        objective_function,
        x0=np.concatenate([initial_centers.ravel(), initial_radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract results
    params = result.x
    centers = params[:20].reshape(10, 2)
    radii = params[20:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
```
