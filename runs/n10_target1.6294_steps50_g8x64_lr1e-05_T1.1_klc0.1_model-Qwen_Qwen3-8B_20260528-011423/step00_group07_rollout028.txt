import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles at regular grid positions
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    centers = np.vstack([centers, [0.5, 0.5], [0.5, 0.25], [0.5, 0.75]])
    radii = np.ones(10) * 0.1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: 1. Each circle is inside the unit square
    #              2. No two circles overlap
    def constraint_inside(radii):
        return np.array([r <= 1 - np.max(np.abs(centers[i] - np.array([0.5, 0.5]))[0])
                         for i, r in enumerate(radii)])

    def constraint_overlap(radii):
        dists = np.zeros(10 * 9 // 2)
        idx = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx*dx + dy*dy) - (radii[i] + radii[j])
                idx += 1
        return dists

    # Bounds on radii: non-negative
    bounds = [(0, 1)] * 10

    # Constraints in minimize
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
    ]

    # Optimization
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    centers = np.vstack([centers, [0.5, 0.5], [0.5, 0.25], [0.5, 0.75]])
    radii = res.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
