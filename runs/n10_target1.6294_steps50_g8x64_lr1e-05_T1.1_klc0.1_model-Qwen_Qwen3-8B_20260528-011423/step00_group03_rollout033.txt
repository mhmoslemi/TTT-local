import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place 10 circles equally spaced in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            x, y = centers[i]
            x = 0.1 + i * 0.2
            y = 0.1 + (i // 5) * 0.2
            centers[i] = [x, y]
            if radii[i] < 0:
                return False
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False

        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Use numerical optimization with bounds
    bounds = [(0, 0.5) for _ in range(10)]
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints}
    )

    # Check if the solution is valid and extract the result
    if result.success and constraints(result.x):
        centers = np.array([[0.1 + i * 0.2, 0.1 + (i // 5) * 0.2] for i in range(10)])
        sum_radii = np.sum(result.x)
        return (centers, result.x, sum_radii)
    else:
        # Fallback to initial guess if optimization fails
        centers = np.array([[0.1 + i * 0.2, 0.1 + (i // 5) * 0.2] for i in range(10)])
        sum_radii = np.sum(radii)
        return (centers, radii, sum_radii)
