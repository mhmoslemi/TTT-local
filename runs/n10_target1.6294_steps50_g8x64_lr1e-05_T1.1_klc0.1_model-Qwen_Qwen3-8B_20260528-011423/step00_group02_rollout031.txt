import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles arranged in a square grid
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - 1e-6:
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    return False
        return True

    # Optimization settings
    bounds = [(0, 1) for _ in range(10)]
    constraints = [{'type': 'ineq', 'fun': lambda r: constraints(r)}]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract result
    optimal_radii = result.x
    # Ensure the circles are within the square and no overlaps
    for i in range(10):
        if optimal_radii[i] < 0:
            optimal_radii[i] = 0
        if optimal_radii[i] > 1 - 1e-6:
            optimal_radii[i] = 1 - 1e-6
    # Check overlaps again with optimal radii
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = math.sqrt(dx*dx + dy*dy)
            if dist < optimal_radii[i] + optimal_radii[j] - 1e-6:
                # Adjust centers to avoid overlap
                direction = np.array([dx, dy]) / dist
                overlap = optimal_radii[i] + optimal_radii[j] - dist
                centers[i] += direction * overlap / 2
                centers[j] -= direction * overlap / 2

    sum_radii = np.sum(optimal_radii)
    return (centers, optimal_radii, sum_radii)
