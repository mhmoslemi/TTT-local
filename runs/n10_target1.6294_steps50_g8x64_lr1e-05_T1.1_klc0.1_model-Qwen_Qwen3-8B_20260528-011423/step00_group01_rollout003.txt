import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.ones(10) * 0.1

    def objective_function(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing the negative

    def constraints_function(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        # Constraint 1: All circles are inside the square
        inside = np.all((centers >= radii[:, np.newaxis]) & (centers <= 1 - radii[:, np.newaxis]), axis=1)
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf  # Penalize overlapping
        return 0

    # Prepare initial guess
    initial_guess = np.concatenate((initial_centers.reshape(-1), initial_radii))

    # Set up optimization problem
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1)] for _ in range(2))  # x and y for each center
        bounds.extend([(0, 0.5)] for _ in range(10))  # radii (must be <= 0.5 to fit in square)

    constraints = [
        {'type': 'ineq', 'fun': constraints_function}
    ]

    result = minimize(
        objective_function,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
