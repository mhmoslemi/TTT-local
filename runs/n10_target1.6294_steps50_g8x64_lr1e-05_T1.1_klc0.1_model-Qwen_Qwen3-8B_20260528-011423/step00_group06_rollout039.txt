import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles placed in a grid pattern
    centers = np.array([[0.1 * i + 0.05, 0.1 * j + 0.05] for i in range(10) for j in range(1)])
    radii = np.ones(10) * 0.05

    def objective(radii):
        return -np.sum(radii)  # Minimize the negative sum to maximize the sum

    def constraints(radii):
        # Ensure all radii are non-negative
        if np.any(radii < 0):
            return -np.inf
        # Ensure all circles are inside the unit square
        for i in range(10):
            if radii[i] > 1 - 2 * centers[i, 0] or radii[i] > 1 - 2 * centers[i, 1]:
                return -np.inf
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return np.sum(radii)

    # Optimize using the constraints
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1) for _ in range(10)]
    )

    # Extract the optimized values
    optimal_radii = result.x
    optimal_centers = centers.copy()
    sum_radii = -result.fun

    return (optimal_centers, optimal_radii, sum_radii)
