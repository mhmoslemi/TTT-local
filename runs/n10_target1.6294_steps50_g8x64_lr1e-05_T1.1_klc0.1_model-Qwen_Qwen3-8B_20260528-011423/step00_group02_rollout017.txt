import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    # Duplicate for 10 circles, adjust positions slightly to avoid overlap
    centers = np.vstack([centers, np.array([[0.7, 0.7], [0.8, 0.6], [0.6, 0.8], [0.9, 0.5], [0.5, 0.9],
                                            [0.7, 0.3], [0.3, 0.7], [0.8, 0.2], [0.2, 0.8],
                                            [0.5, 0.5]])])
    radii = np.ones(10) * 0.15  # Initial radius estimate

    def objective(radii):
        return -sum(radii)  # Maximize sum of radii

    def constraints(radii):
        # Constraints for boundary (all circles inside the unit square)
        b = []
        for i in range(10):
            b.append(radii[i] <= np.min([centers[i, 0] - 0.0, 1 - centers[i, 0] - 0.0,
                                        centers[i, 1] - 0.0, 1 - centers[i, 1] - 0.0]))
        # Constraint for non-overlapping circles
        b.append(np.array([0.0 for _ in range(10)]))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                b.append(dist - radii[i] - radii[j] >= 0)
        return b

    def bounds(radii):
        return [(0.0, 1.0) for _ in range(10)]

    result = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': lambda x: constraints(x)}], bounds=bounds(radii), tol=1e-6)
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
