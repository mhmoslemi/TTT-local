import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles at evenly spaced positions with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.zeros(10)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Bounds for radii: each radius must be >= 0 and <= 0.5 (max possible given square size)
        # We'll use a list of constraints for optimization
        constraints = []
        
        # Each circle's center must be within [r, 1 - r] bounds
        for i in range(10):
            r = radii[i]
            # Center must be in [r, 1 - r] both in x and y
            # If r > 0.5, it's invalid (center cannot be within [r, 1 - r])
            if r > 0.5:
                return False
            # Apply bounds to center coordinates
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Convert centers to a flat array for optimization
    initial_guess = np.concatenate([centers.flatten(), radii])
    
    # Define bounds for radii: each radius is between 0 and 0.5
    bounds = [(0, 0.5) for _ in range(10)]
    
    # Define inequality constraints: distance between centers must be >= sum of radii
    def dist_constraint(radii_arr):
        # Extract radii from the flat array
        r = radii_arr[-10:]
        # Ensure all centers are within [r, 1 - r]
        for i in range(10):
            if not (r[i] <= centers[i, 0] <= 1 - r[i] and r[i] <= centers[i, 1] <= 1 - r[i]):
                return [np.inf]
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < r[i] + r[j]:
                    return [np.inf]
        return [0]
    
    # Optimization using Sequential Least Squares Programming
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': dist_constraint}
    )
    
    optimized_radii = result.x[-10:]
    optimized_centers = centers.copy()  # Centers are fixed; only radii are optimized
    
    sum_radii = np.sum(optimized_radii)
    
    return (optimized_centers, optimized_radii, sum_radii)
