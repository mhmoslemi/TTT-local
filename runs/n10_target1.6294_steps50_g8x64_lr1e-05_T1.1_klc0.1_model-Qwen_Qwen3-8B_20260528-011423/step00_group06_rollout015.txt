import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    
    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radius = np.sum(radii_flat)
        return -sum_radius  # Minimize negative sum to maximize sum

    # Constraints
    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        # Constraint 1: Every circle is inside the square
        inside = np.array([r <= x_i <= 1 - r and r <= y_i <= 1 - r for r, (x_i, y_i) in zip(radii_flat, centers_flat)])
        # Constraint 2: Circles do not overlap
        # Distance between centers >= sum of radii
        distances = np.array([[np.linalg.norm(centers_flat[i] - centers_flat[j]) for j in range(10)] for i in range(10)])
        overlapping = np.array([distances[i, j] < radii_flat[i] + radii_flat[j] for i in range(10) for j in range(i+1, 10)])
        return np.concatenate([inside, overlapping])

    # Bounds for centers (0 <= x, y <= 1) and radii (0 <= r <= 0.5)
    bounds = []
    for _ in range(10):
        bounds += [(0, 1), (0, 1)]  # x and y for each center
    for _ in range(10):
        bounds += [(0, 0.5)]  # radius for each circle

    # Optimization
    try:
        result = minimize(objective, np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    except:
        result = minimize(objective, np.concatenate([centers.flatten(), radii]), bounds=bounds)

    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:]
    sum_radii = -result.fun
    return (centers_result, radii_result, sum_radii)
