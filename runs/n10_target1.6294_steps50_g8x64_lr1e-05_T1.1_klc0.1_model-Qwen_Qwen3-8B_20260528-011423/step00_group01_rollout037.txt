import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([
            [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
            [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
        ])
        # Bounds for each radius
        bounds = [(0, 1 - 0.1) for _ in range(10)]  # Assume initial centers are 0.1 away from edges
        # Constraint: radii must be within bounds
        radius_bounds = [(0, 1 - centers[i, 0]) for i in range(10)]
        radius_bounds += [(0, 1 - centers[i, 1]) for i in range(10)]
        # Constraint: no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1 - 0.1) for _ in range(10)]  # Initial bounds based on initial centers
    )

    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = result.x
    sum_radii = np.sum(radii)

    # Final check for constraints
    for i in range(10):
        if radii[i] > 1 - 0.1:
            radii[i] = 1 - 0.1
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j]:
                radii[i] = (dist - radii[j]) / 2
                radii[j] = (dist - radii[i]) / 2

    return (centers, radii, sum_radii)
