import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (randomly placed)
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.random.uniform(0.05, 0.1, 10)

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        # Bounds constraint: each circle must be within the unit square
        bounds = []
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            bounds.append((r, 1 - r))
            bounds.append((r, 1 - r))
        # Non-overlapping constraint: distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                bounds.append((dist - (radii[i] + radii[j]), np.inf))
        return bounds

    def objective_with_bounds(radii, centers):
        return -np.sum(radii)

    def constraints_with_bounds(radii, centers):
        constraints = []
        # Bounds constraint: each circle must be within the unit square
        for i in range(10):
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - 0})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - r})
        # Non-overlapping constraint: distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: dist - (r[i] + r[j])})
        return constraints

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="A constraint violation was encountered")
        result = minimize(
            fun=objective_with_bounds,
            x0=radii,
            args=(centers,),
            constraints=constraints_with_bounds(radii, centers),
            bounds=[(0, 1)] * 20,
            method='SLSQP',
            options={'maxiter': 200, 'disp': False}
        )

    # Extract results
    centers_result = centers.copy()
    radii_result = result.x[:10]
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
