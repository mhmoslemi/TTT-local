import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1  # Initial radius

    def objective(radii_vec):
        return -np.sum(radii_vec)  # Maximize sum of radii by minimizing negative sum

    def constraint_distances(radii_vec):
        cons = []
        centers_array = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_array[i, 0] - centers_array[j, 0]
                dy = centers_array[i, 1] - centers_array[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                cons.append(dist - (radii_vec[i] + radii_vec[j]))
        return cons

    def constraint_boundaries(radii_vec):
        cons = []
        centers_array = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            r = radii_vec[i]
            x, y = centers_array[i]
            cons.append(r - (x - 0))  # x - r >= 0
            cons.append(r - (1 - x))  # 1 - x - r >= 0
            cons.append(r - (y - 0))  # y - r >= 0
            cons.append(r - (1 - y))  # 1 - y - r >= 0
        return cons

    # Bounds for radii
    bounds = [(0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_distances},
            {'type': 'ineq', 'fun': constraint_boundaries}
        ],
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract final values
    centers_final = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii_final = result.x
    sum_radii = -result.fun

    return (centers_final, radii_final, sum_radii)
