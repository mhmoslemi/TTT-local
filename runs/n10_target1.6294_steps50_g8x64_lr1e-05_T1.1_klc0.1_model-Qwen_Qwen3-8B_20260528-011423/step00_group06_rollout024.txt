import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints_func(centers, radii):
        constraints = []
        # Constraint 1: All circles are within the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r >= 0) * 1e6)
            constraints.append((x + r <= 1) * 1e6)
            constraints.append((y - r >= 0) * 1e6)
            constraints.append((y + r <= 1) * 1e6)
        # Constraint 2: No overlapping between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append((dist >= radii[i] + radii[j]) * 1e6)
        return np.sum(constraints)

    def objective_func(x):
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        return -np.sum(radii)

    def constraint_func(x):
        radii = x[::2]
        centers = x[1::2].reshape(10, 2)
        return constraints_func(centers, radii)

    # Optimization setup
    x0 = np.concatenate([radii, centers.flatten()])
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # Radii
        bounds.append((0, 1))  # X center
        bounds.append((0, 1))  # Y center

    # Setup optimization with constraints
    cons = [{'type': 'ineq', 'fun': constraint_func}]
    result = minimize(objective_func, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6, options={'maxiter': 1000})

    # Extract final solution
    final_radii = result.x[::2]
    final_centers = result.x[1::2].reshape(10, 2)
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
