import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    radii = np.full(10, 0.1)

    def objective_function(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        constraints = []

        # Boundary constraints: each circle is inside the unit square
        for i in range(10):
            x, y = centers_flat[i]
            r = radii_flat[i]
            constraints.append((x - r >= 0) * 1e6)
            constraints.append((x + r <= 1) * 1e6)
            constraints.append((y - r >= 0) * 1e6)
            constraints.append((y + r <= 1) * 1e6)

        # Non-overlap constraints: distance between centers >= sum of radii
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.hypot(dx, dy)
                r1 = radii_flat[i]
                r2 = radii_flat[j]
                constraints.append((dist - r1 - r2 >= 0) * 1e6)

        # Non-negative radii
        for i in range(10):
            constraints.append(radii_flat[i] >= 0)

        return np.array(constraints)

    # Bounds for parameters: centers in [0,1], radii in [0, 0.5]
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # centers x, y
        bounds.append((0.0, 0.5))  # radii

    # Optimization
    initial_params = np.concatenate([centers.reshape(10 * 2), radii])
    result = minimize(
        fun=objective_function,
        x0=initial_params,
        method='SLSQP',
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        jac=None,
        tol=1e-6,
        options={"disp": False, "maxiter": 1000}
    )

    if result.success:
        centers_opt = result.x[:60].reshape(10, 2)
        radii_opt = result.x[60:]
        sum_radii_opt = -result.fun
    else:
        centers_opt = centers
        radii_opt = radii
        sum_radii_opt = np.sum(radii)

    return (centers_opt, radii_opt, sum_radii_opt)
