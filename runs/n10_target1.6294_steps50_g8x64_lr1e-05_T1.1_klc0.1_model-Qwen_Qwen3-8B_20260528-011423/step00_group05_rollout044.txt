import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square
    def constraint_inside(radii):
        return radii

    # Constraint: circles do not overlap
    def constraint_overlap(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i+1, 10):
                if dists[i,j] < radii[i] + radii[j]:
                    return -1
        return 0

    # Optimization problem setup
    bounds = [(0, 0.5) for _ in range(10)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: r[i] for i in range(10)},  # Ensure radii are non-negative
        {'type': 'ineq', 'fun': constraint_inside},  # Circles inside the square
        {'type': 'ineq', 'fun': constraint_overlap}  # Circles do not overlap
    ]

    # Optimization
    result = minimize(objective, radii, bounds=bounds, constraints=cons)

    # Extract final result
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    sum_radii = result.fun * -1
    return (centers, result.x, sum_radii)
