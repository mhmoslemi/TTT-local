import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii
    
    def constraints_func(x):
        # x is flattened array of centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        
        # Constraint 1: All circles are within the unit square
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf  # Penalize violation
        
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf  # Penalize overlap
        
        return -np.sum(radii)  # Return negative sum as we are minimizing
    
    # Initial guess
    x0 = np.concatenate([centers.ravel(), radii])
    
    # Set up optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message=".*The trust-constr solver is not available.*")
        result = minimize(
            fun=constraints_func,
            x0=x0,
            method='SLSQP',
            options={'ftol': 1e-9, 'maxiter': 1000}
        )
    
    if not result.success:
        return (centers, radii, np.sum(radii))
    
    # Extract the optimized result
    best_centers = result.x[:60].reshape(10, 2)
    best_radii = result.x[60:]
    best_sum = np.sum(best_radii)
    
    return (best_centers, best_radii, best_sum)
