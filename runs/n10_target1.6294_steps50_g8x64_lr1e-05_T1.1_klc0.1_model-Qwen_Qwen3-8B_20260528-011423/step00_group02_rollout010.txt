import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: 10 circles placed in a square grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: Each circle is inside the square
        # r <= x <= 1 - r and r <= y <= 1 - r
        # => 2r <= 1 => r <= 0.5
        # This constraint is already satisfied as we start with radii <= 0.1 and increase
        # Only check if radii exceeds 0.5
        if np.any(radii > 0.5):
            return np.inf
        # Constraint 2: Distance between centers >= sum of radii
        dist = cdist(centers, centers)
        diag = np.diag(dist)
        dist[dist < 1e-8] = np.inf  # Avoid division by zero
        for i in range(10):
            for j in range(i+1, 10):
                if dist[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Prepare initial values
    x0 = np.concatenate((centers.flatten(), radii))

    def objective_with_params(x):
        centers = x[:50].reshape(10, 2)
        radii = x[50:]
        return -np.sum(radii)

    def constraints_with_params(x):
        centers = x[:50].reshape(10, 2)
        radii = x[50:]
        # Constraint 1: radii <= 0.5
        if np.any(radii > 0.5):
            return np.inf
        # Constraint 2: distance between centers >= sum of radii
        dist = cdist(centers, centers)
        diag = np.diag(dist)
        dist[dist < 1e-8] = np.inf
        for i in range(10):
            for j in range(i+1, 10):
                if dist[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    bounds = [(-1e-5, 0.5)] * 50 + [(0, 0.5)] * 10
    constraints_dict = {'type': 'ineq', 'fun': constraints_with_params}
    result = minimize(objective_with_params, x0, method='SLSQP', bounds=bounds, constraints=[constraints_dict], tol=1e-6, options={'maxiter': 1000})

    centers_final = result.x[:50].reshape(10, 2)
    radii_final = result.x[50:]
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
