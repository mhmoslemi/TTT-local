import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    radii = np.full(10, 0.05)

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x[10:])  # Negative for minimization

    # Constraints: 1. Each circle is inside the square
    #              2. No overlap between circles
    def constraint_inside(x):
        centers = x[:10*2].reshape((10, 2))
        radii = x[10:]
        return np.maximum(0, np.min(np.vstack([centers[:, 0] - radii, 1 - centers[:, 0] - radii, 
                                              centers[:, 1] - radii, 1 - centers[:, 1] - radii]), axis=0))

    def constraint_overlap(x):
        centers = x[:10*2].reshape((10, 2))
        radii = x[10:]
        distances = np.sqrt(((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2).sum(axis=2))
        return np.min(distances - (radii[:, np.newaxis] + radii[np.newaxis, :]))  # Negative for minimization

    # Combine constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds: All coordinates between 0 and 1, radii >= 0
    bounds = [(0, 1) for _ in range(10*2)] + [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(objective, np.concatenate([centers.flatten(), radii]), 
                      method='SLSQP', bounds=bounds, constraints=constraints)

    centers_opt = result.x[:10*2].reshape((10, 2))
    radii_opt = result.x[10:]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
