```python
import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_bounds(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        return 0

    def constraint_overlap(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Flatten initial guess
    x0 = np.concatenate([initial_centers.ravel(), initial_radii])

    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_center_bounds},
        {'type': 'ineq', 'fun': constraint_overlap},
    ]

    # Bounds for x and y coordinates (each must be between 0 and 1 - r)
    # Bounds for radii (each must be non-negative)
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # x_i
        bounds.append((0, 1))  # y_i
        bounds.append((0, 1))  # r_i

    # Optimization
    result = minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'maxiter': 1000, 'ftol': 1e-9, 'eps': 1e-8}
    )

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -objective(radii)

    return (centers, radii, sum_radii)
```
