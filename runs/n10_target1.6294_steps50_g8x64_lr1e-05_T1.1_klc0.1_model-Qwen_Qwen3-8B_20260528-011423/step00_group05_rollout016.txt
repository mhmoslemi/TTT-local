import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective_function(params):
        centers = params[:20].reshape(10, 2)
        radii = params[20:].reshape(10)
        sum_r = np.sum(radii)
        return -sum_r  # Minimize negative sum to maximize sum

    def constraints(params):
        centers = params[:20].reshape(10, 2)
        radii = params[20:].reshape(10)
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
        # Constraint 2: No circles overlap
        for i, j in itertools.combinations(range(10), 2):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < radii[i] + radii[j]:
                return np.inf
        return 0

    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x and y for each center
        bounds.extend([0, 1])  # radius for each circle

    result = minimize(
        objective_function,
        x0=np.concatenate((centers.flatten(), radii)),
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:].reshape(10)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
