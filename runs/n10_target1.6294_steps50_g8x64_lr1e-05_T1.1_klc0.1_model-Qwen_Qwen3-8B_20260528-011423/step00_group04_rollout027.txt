import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    np.random.seed(42)

    # Initial guess: randomly place 10 circles with small radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x[10:])  # Minimize negative sum of radii

    # Constraints: 1. Circles stay within the unit square
    #              2. Circles do not overlap
    #              3. Radii must be non-negative

    # Flatten centers and radii for optimization
    x0 = np.concatenate((centers.reshape(20), radii))

    def constraint1(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf  # Penalize overlapping circles
        return 0

    def constraint2(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        return 0

    def constraint3(x):
        radii = x[20:]
        for r in radii:
            if r < 0:
                return np.inf
        return 0

    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2},
        {'type': 'ineq', 'fun': constraint3}
    ]

    # Bounds: centers between [0,1], radii >= 0
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]

    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract final solution
    best_centers = result.x[:20].reshape(10, 2)
    best_radii = result.x[20:]
    sum_radii = np.sum(best_radii)

    return (best_centers, best_radii, sum_radii)
