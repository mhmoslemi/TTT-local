import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.array([0.1] * 10)

    def objective(x):
        # x is a flattened array of centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        sum_radii = np.sum(radii)
        return -sum_radii  # We minimize the negative to maximize the sum

    def constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        constraints = []
        # Constraint 1: Each circle is inside the unit square
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            constraints.append((x_i - r_i >= 0) * 1e6)
            constraints.append((x_i + r_i <= 1) * 1e6)
            constraints.append((y_i - r_i >= 0) * 1e6)
            constraints.append((y_i + r_i <= 1) * 1e6)
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii[i]
                r_j = radii[j]
                constraints.append((dist - (r_i + r_j) >= 0) * 1e6)
        return np.array(constraints)

    # Bounds: centers between 0 and 1, radii between 0 and 0.5
    bounds = []
    for _ in range(10):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
    for _ in range(10):
        bounds.append((0.0, 0.5))  # radii

    result = minimize(objective, x0=np.concatenate([initial_centers.flatten(), initial_radii]),
                      method='SLSQP',
                      bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints},
                      tol=1e-6,
                      options={'maxiter': 1000})

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
