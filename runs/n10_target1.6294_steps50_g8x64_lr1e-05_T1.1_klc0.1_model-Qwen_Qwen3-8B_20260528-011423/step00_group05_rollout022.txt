import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective(x):

    def constraints(x):
        # Ensure all circles are inside the unit square
        for i in range(10):
            r = radii[i]
            x_i, y_i = x[2*i], x[2*i+1]
            if not (r <= x_i <= 1 - r and r <= y_i <= 1 - r):
                return np.inf
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = x[2*i] - x[2*j]
                dy = x[2*i+1] - x[2*j+1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Initial guess: centers at (0.1, 0.1), (0.1, 0.3), ..., (0.1, 0.9) and (0.3, 0.1), etc.
    centers = np.array([
        [0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],
        [0.3, 0.1], [0.3, 0.3], [0.3, 0.5], [0.3, 0.7], [0.3, 0.9]
    ])
    x0 = np.concatenate([centers[:, 0], centers[:, 1]])
    radii = np.array([0.05] * 10)

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0=x0,
            method='SLSQP',
            constraints={'type': 'ineq', 'fun': constraints},
            bounds=[(0, 1) for _ in range(20)]
        )

    centers = np.reshape(result.x, (10, 2))
    radii = np.array([0.05] * 10)
    for i in range(10):
        r = radii[i]
        x_i, y_i = centers[i]
        if not (r <= x_i <= 1 - r and r <= y_i <= 1 - r):
            radii[i] = np.min([r, x_i - 0, 1 - x_i - r, y_i - 0, 1 - y_i - r])

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
