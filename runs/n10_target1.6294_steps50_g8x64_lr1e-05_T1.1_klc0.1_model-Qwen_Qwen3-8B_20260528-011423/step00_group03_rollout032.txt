import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        # Constraint 1: All circles are within the square
        # Each center must be at least radius away from the square's boundaries
        # radii must be non-negative
        for i in range(10):
            if radii[i] < 0 or radii[i] > 1 - np.min(np.abs(centers[i] - [0, 0])) or radii[i] > 1 - np.min(np.abs(centers[i] - [1, 1])):
                return np.inf
        # Constraint 2: Circles cannot overlap
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0
    
    # Optimization problem
    bounds = [(0, 1 - 0.01) for _ in range(10)]  # Small margin to avoid edge issues
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP"
    )
    
    # Extract results
    centers = np.array(result.x).reshape(10, 2)
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
