import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii (randomly placed circles)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)

    # Constraints
    def constraints(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # 1) All circles must be inside the square
        in_square = [r <= min(c[0], 1 - c[0], c[1], 1 - c[1]) for c, r in zip(centers, radii)]
        # 2) No circles can overlap
        overlap = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                overlap.append(dist < radii[i] + radii[j])
        return in_square + overlap

    # Bounds for radii (non-negative) and centers (within [0,1])
    bounds = [(0, 1) for _ in range(60)] + [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    # Extract result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
