import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place centers in a grid and assign equal radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective_function(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize sum

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        constraints = []

        # Constraint: circles are inside the unit square
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append(x_i - r_i)
            constraints.append(1 - x_i - r_i)
            constraints.append(y_i - r_i)
            constraints.append(1 - y_i - r_i)

        # Constraint: non-overlapping circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.hypot(dx, dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                constraints.append(dist - r_i - r_j)

        # Constraint: non-negative radii
        for i in range(10):
            constraints.append(radii_flat[i])

        return constraints

    # Initial guess as a flattened array
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Setup optimization
    bounds = []
    for _ in range(60):  # 10 centers * 2 + 10 radii
        bounds.append((0, 1))
    for _ in range(10):  # 10 radii
        bounds.append((0, 0.5))  # Max radius is 0.5 to fit in unit square

    # Optimization options
    options = {
        'disp': False,
        'ftol': 1e-9,
        'maxiter': 1000
    }

    result = minimize(objective_function, initial_guess, method='SLSQP',
                      bounds=bounds, constraints={'type': 'ineq', 'fun': constraints},
                      options=options)

    # Extract result
    best_centers = result.x[:60].reshape(10, 2)
    best_radii = result.x[60:]
    sum_radii = -result.fun  # Since we minimized negative sum

    return (best_centers, best_radii, sum_radii)
