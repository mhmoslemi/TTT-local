import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[i * 0.25 + 0.05, j * 0.25 + 0.05] for i in range(10) for j in range(1)])
    radii = np.full(10, 0.1)

    # Objective function to maximize the sum of radii (minimize negative sum)
    def objective(x):
        radii_flat = x[:10]
        centers_flat = x[10:]
        centers = centers_flat.reshape(10, 2)
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints
    def constraints(x):
        radii_flat = x[:10]
        centers_flat = x[10:]
        centers = centers_flat.reshape(10, 2)
        radii = radii_flat

        # Constraint 1: All circles are inside the unit square
        constraints = []
        for i in range(10):
            x_i, y_i = centers[i]
            constraints.append(x_i - radii[i] >= 0)
            constraints.append(1 - x_i - radii[i] >= 0)
            constraints.append(y_i - radii[i] >= 0)
            constraints.append(1 - y_i - radii[i] >= 0)

        # Constraint 2: No overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]) >= 0)

        return constraints

    # Bounds for radii and centers
    bounds = []
    for i in range(10):
        bounds.append((0, 1.0))  # radii
        bounds.append((0, 1.0))  # x-coordinate
        bounds.append((0, 1.0))  # y-coordinate

    # Initial guess
    x0 = np.concatenate([radii.flatten(), centers.flatten()])

    # Optimization: minimize -sum_radii
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    # Extract the result
    x_opt = result.x
    radii_opt = x_opt[:10]
    centers_opt = x_opt[10:].reshape(10, 2)
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
