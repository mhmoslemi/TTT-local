import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 10 circles
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    radii = np.ones(10) * 0.05

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]) & (radii >= 0), axis=1)
        # Constraint 2: No two circles overlap
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        overlap = np.triu(np.less_equal(dist, radii[:, np.newaxis] + radii[np.newaxis, :]), k=1)
        return np.all(inside) and np.all(overlap)

    def objective_with_constraints(radii):
        return -np.sum(radii)

    def constraints_with_constraints(radii):
        centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
        inside = np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]) & (radii >= 0), axis=1)
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        overlap = np.triu(np.less_equal(dist, radii[:, np.newaxis] + radii[np.newaxis, :]), k=1)
        if not np.all(inside) or not np.all(overlap):
            return np.inf
        return 0

    # Initial guess for radii
    initial_radii = np.ones(10) * 0.05
    result = minimize(objective_with_constraints, initial_radii, constraints=constraints_with_constraints)

    optimal_radii = result.x
    optimal_centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    sum_radii = np.sum(optimal_radii)

    return (optimal_centers, optimal_radii, sum_radii)
