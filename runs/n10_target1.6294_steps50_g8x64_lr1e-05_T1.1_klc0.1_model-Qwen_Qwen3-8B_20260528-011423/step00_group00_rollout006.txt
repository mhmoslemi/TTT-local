import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (randomly placed within the square)
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1

    # Objective function to minimize: negative sum of radii (maximize it)
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        # reshape to centers and radii
        x_flat = x.reshape(10, 3)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        sum_radii = np.sum(radii)
        return -sum_radii  # minimize negative sum to maximize sum

    # Constraints
    def constraint1(x):
        # Ensure circles are within the square
        x_flat = x.reshape(10, 3)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        # Check if each circle is within [r, 1 - r]
        valid = np.all((centers >= radii[:, np.newaxis]) & (centers <= 1 - radii[:, np.newaxis]), axis=1)
        return np.sum(valid)

    def constraint2(x):
        # Ensure circles do not overlap
        x_flat = x.reshape(10, 3)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        distances = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        # Ensure distance >= sum of radii
        valid = np.all(distances >= radii[:, np.newaxis] + radii[np.newaxis, :], axis=1)
        return np.sum(valid)

    def constraint3(x):
        # Ensure radii are non-negative
        x_flat = x.reshape(10, 3)
        radii = x_flat[:, 2]
        return np.sum(radii >= 0)

    # Setup the optimization
    bounds = [(0, 1) for _ in range(10 * 2)] + [(0, 0.5) for _ in range(10)]  # x, y in [0, 1], r in [0, 0.5]

    res = minimize(
        objective,
        x0=centers.reshape(-1).tolist() + radii.tolist(),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint1},
            {'type': 'ineq', 'fun': constraint2},
            {'type': 'ineq', 'fun': constraint3}
        ],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    # Extract results
    x_opt = res.x
    centers_opt = x_opt[:10*2].reshape(10, 2)
    radii_opt = x_opt[10*2:].reshape(10)
    sum_radii_opt = -res.fun

    # Return results
    return (centers_opt, radii_opt, sum_radii_opt)
