import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[10:])

    # Constraints: circle must be inside the unit square
    def square_constraints(x):
        return np.array([
            x[i*2] - x[i*2 + 1] + 1e-8,  # x - r >= 0
            1 - x[i*2] - x[i*2 + 1] + 1e-8,  # 1 - x - r >= 0
            x[i*2] - x[i*2 + 1] + 1e-8,  # y - r >= 0
            1 - x[i*2] - x[i*2 + 1] + 1e-8   # 1 - y - r >= 0
        for i in range(10)])

    # Constraints: no two circles may overlap
    def circle_constraints(x):
        r = x[10:]
        centers = x[:20].reshape(10, 2)
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return np.array([dists[i, j] - r[i] - r[j] for i in range(10) for j in range(i+1)])

    # Bounds: centers between 0 and 1, radii between 0 and 1
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0=np.concatenate((centers.flatten(), radii)),
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': square_constraints},
                {'type': 'ineq', 'fun': circle_constraints}
            ],
            method='SLSQP'
        )

    # Extract results
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
