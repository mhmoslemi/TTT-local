import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    def bounds(radii):
        return [(0, 1 - 0.1) for _ in range(10)]  # Assuming a basic initial layout
    
    # Set up optimization problem
    res = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}],
                   bounds=bounds(radii), tol=1e-5)
    
    # Update centers based on optimized radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    sum_radii = -res.fun
    return (centers, res.x, sum_radii)
