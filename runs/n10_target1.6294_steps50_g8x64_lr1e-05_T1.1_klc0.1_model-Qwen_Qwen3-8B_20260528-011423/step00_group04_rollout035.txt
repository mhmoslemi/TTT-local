import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective_function(params):
        # Extract centers and radii from parameters
        centers = params[:60].reshape(10, 2)
        radii = params[60:].reshape(10)
        
        # Objective: minimize negative sum of radii (equivalent to maximizing sum of radii)
        sum_radii = np.sum(radii)
        
        # Constraints: circles must be inside the unit square
        inside = np.all((centers >= radii[:, np.newaxis]) & (centers <= 1 - radii[:, np.newaxis]), axis=1)
        if not np.all(inside):
            return np.inf  # Penalize if any circle is out of bounds
        
        # Constraints: circles must not overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return np.inf  # Penalize if circles overlap
        
        return -sum_radii  # Minimize negative sum to maximize sum

    # Setup bounds for parameters: centers in [0,1], radii >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # centers
        bounds.extend([(0.0, 0.5)])              # radii (upper bound for radius to stay within square)
    
    # Optimization
    result = minimize(objective_function, np.concatenate([centers.flatten(), radii]), bounds=bounds)
    
    # Extract centers, radii, and sum_radii from result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii)
