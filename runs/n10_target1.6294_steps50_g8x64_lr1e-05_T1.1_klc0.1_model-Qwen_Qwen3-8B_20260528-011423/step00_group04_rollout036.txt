import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: spread circles evenly across the square
    # This is just a starting point and may be refined during optimization
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Initial random radii

    # Objective function: maximize sum of radii
    def objective(x):
        # x is flattened array of centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)  # minimize negative sum to maximize sum

    # Constraints: 1. Each circle fits inside the unit square
    #               2. Circles do not overlap
    #               3. Radii are non-negative
    def constraint_circle_inside(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([r - (x - 0.5)**2 - (y - 0.5)**2 for r, (x, y) in zip(radii, centers)])  # x^2 + y^2 <= (r + 0.5)^2

    def constraint_no_overlap(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        dists = np.array([[np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2) for (x_i, y_i), (x_j, y_j) in itertools.combinations(centers, 2)]])
        sum_radii = np.array([radii[i] + radii[j] for i, j in itertools.combinations(range(10), 2)])
        return dists - sum_radii

    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Bounds: radii must be non-negative
    bounds = [(0, 0.5) for _ in range(60)]  # Max radius is 0.5

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints=cons)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
