import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.08)
    
    def objective(radii, centers):
        return -np.sum(radii)
    
    def constraints(radii, centers):
        # Constraint: all circles are inside the square
        def in_square(r, c):
            return (c[0] - r >= 0 and c[0] + r <= 1 and c[1] - r >= 0 and c[1] + r <= 1)
        
        # Constraint: no overlap between circles
        def no_overlap(r1, r2, c1, c2):
            dx = c1[0] - c2[0]
            dy = c1[1] - c2[1]
            dist = np.sqrt(dx*dx + dy*dy)
            return dist >= r1 + r2
        
        # Constraints for all circles
        constraints = []
        for i in range(10):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] if in_square(r[i], centers[i]) else -np.inf})
        
        for i in range(10):
            for j in range(i+1, 10):
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: no_overlap(r[i], r[j], centers[i], centers[j])})
        
        return constraints
    
    # Optimization
    result = minimize(lambda x: objective(x, centers), radii, method='SLSQP', 
                      constraints=constraints(radii, centers), bounds=[(0, 1) for _ in range(10)])
    
    return (centers, result.x, -result.fun)
