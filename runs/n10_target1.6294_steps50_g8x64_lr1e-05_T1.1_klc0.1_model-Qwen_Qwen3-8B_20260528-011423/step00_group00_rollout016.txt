import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles arranged in a square grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)

    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x[2:])  # Negative because we minimize

    # Bounds for each circle's center and radius
    bounds = []
    for _ in range(10):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    # Constraint: each circle is inside the unit square
    def square_bounds(x):
        for i in range(10):
            r = x[2 + 3*i]
            if not (r <= x[0 + 3*i] <= 1 - r and r <= x[1 + 3*i] <= 1 - r):
                return -1
        return 0

    # Constraint: circles do not overlap
    def non_overlap(x):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x[0 + 3*i] - x[0 + 3*j]
                dy = x[1 + 3*i] - x[1 + 3*j]
                dist = np.sqrt(dx**2 + dy**2)
                r1 = x[2 + 3*i]
                r2 = x[2 + 3*j]
                if dist < r1 + r2 - 1e-8:
                    return -1
        return 0

    # Wrap constraints
    cons = [
        {'type': 'ineq', 'fun': square_bounds},
        {'type': 'ineq', 'fun': non_overlap}
    ]

    # Initial guess flattened
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization
    result = minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 500}
    )

    # Extract result
    centers_opt = np.reshape(result.x[:30], (10, 2))
    radii_opt = result.x[30:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
