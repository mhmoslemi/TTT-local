import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        n = len(radii)
        r = radii.reshape(n, 1)
        c = centers
        # Constraint 1: circles inside the square
        constraints1 = np.zeros(n)
        for i in range(n):
            constraints1[i] = r[i] - (c[i, 0] - r[i]) * (c[i, 0] + r[i]) * (c[i, 1] - r[i]) * (c[i, 1] + r[i])
        # Constraint 2: circles do not overlap
        constraints2 = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = c[i, 0] - c[j, 0]
                dy = c[i, 1] - c[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraints2.append(dist - (r[i] + r[j]))
        return np.array(constraints1), np.array(constraints2)

    # Objective function
    def objective_func(x):
        return -np.sum(x)

    # Bounds: radii >= 0
    bounds = [(0, 1) for _ in range(10)]

    # Constraints
    cons = []
    # Constraint for circles inside the square
    for i in range(10):
        def cons_i(radii):
            return radii[i] - (centers[i, 0] - radii[i]) * (centers[i, 0] + radii[i]) * (centers[i, 1] - radii[i]) * (centers[i, 1] + radii[i])
        cons.append({'type': 'ineq', 'fun': lambda r, i=i: cons_i(r)})
    # Constraint for circles not overlapping
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            def cons_ij(radii):
                return dist - (radii[i] + radii[j])
            cons.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: cons_ij(r)})

    # Optimization
    result = minimize(objective_func, radii, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-6)

    # Extract result
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    # Ensure constraints are satisfied
    assert np.all(radii >= 0), "All radii must be non-negative"
    assert np.all(radii <= 1 - centers[:, 0] - centers[:, 1]), "Radii must be inside the square"
    assert np.allclose(np.sqrt((centers[:, 0, np.newaxis] - centers[np.newaxis, :, 0])**2 + (centers[:, 1, np.newaxis] - centers[np.newaxis, :, 1])**2) - (radii[np.newaxis, :] + radii[:, np.newaxis]), 0, atol=1e-3), "Circles must not overlap"

    return (centers, radii, sum_radii)
