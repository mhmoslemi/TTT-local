import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced points with small radii
    centers = np.array([[i/9 + 0.05, j/9 + 0.05] for j in range(10) for i in range(10)])
    centers = centers[::2]  # Only take every other point to get 10 points
    radii = np.full(10, 0.05)
    
    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii
    
    def constraints(radii, centers):
        n = len(radii)
        cons = []
        for i in range(n):
            for j in range(i+1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                cons.append(dist - (radii[i] + radii[j]))
        for i in range(n):
            x, y = centers[i]
            cons.append(x - radii[i])
            cons.append(1 - x - radii[i])
            cons.append(y - radii[i])
            cons.append(1 - y - radii[i])
        return cons
    
    def bounds(radii):
        return [(0, 1) for _ in range(len(radii))]
    
    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="The objective function returned a value of 0.0 at the initial point")
        result = minimize(
            fun=lambda r: objective(r, centers),
            x0=radii,
            method='SLSQP',
            bounds=bounds(radii),
            constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r, centers)}],
            options={'disp': False}
        )
    
    optimized_radii = result.x
    centers = centers[:len(optimized_radii)]
    sum_radii = np.sum(optimized_radii)
    
    return (centers, optimized_radii, sum_radii)
