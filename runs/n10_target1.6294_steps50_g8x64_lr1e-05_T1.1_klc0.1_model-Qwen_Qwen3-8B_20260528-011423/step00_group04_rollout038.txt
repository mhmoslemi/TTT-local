import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(10, 2)
    # Initial guess for radii (all 0.1)
    radii = np.full(10, 0.1)
    
    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint: circles must be inside the unit square
        def in_bounds(r):
            return (radii <= 1 - radii) & (radii <= 1 - radii)  # Check x and y bounds
        # Constraint: no two circles overlap
        def no_overlap(i, j):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            return dist >= radii[i] + radii[j]
        # Combined constraints
        in_bounds_constr = {'type': 'ineq', 'fun': lambda r: r - 1e-6}
        no_overlap_constrs = []
        for i in range(10):
            for j in range(i+1, 10):
                no_overlap_constrs.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: no_overlap(i, j)})
        return [in_bounds_constr] + no_overlap_constrs

    # Optimize
    result = minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        constraints=constraints(radii, centers),
        method='SLSQP',
        bounds=[(0, 1 - 1e-6) for _ in range(10)]
    )

    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
