import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.array([0.1] * 10)
    
    def objective(radii):
        return -np.sum(radii)  # We want to maximize the sum, so minimize the negative
    
    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        # For each circle, r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i
        # We need to ensure that 0 <= r_i <= 0.5 (since 2r <= 1)
        r = radii
        inside = np.all((r <= 0.5) & (r >= 0))
        if not inside:
            return np.inf
        
        # Constraint 2: No two circles overlap
        # For each pair (i, j), distance between centers >= r_i + r_j
        centers = np.vstack([np.array([0.1 + i*0.2, 0.1 + j*0.2]) for i in range(5) for j in range(2)])
        dists = np.sqrt(((centers[:, 0][:, np.newaxis] - centers[:, 0])[np.triu_indices(10, 1)]**2 +
                         (centers[:, 1][:, np.newaxis] - centers[:, 1])[np.triu_indices(10, 1)]**2))
        sum_radii = r[np.triu_indices(10, 1)]
        overlapping = np.any(dists < sum_radii)
        if overlapping:
            return np.inf
        return 0
    
    # Define bounds for radii
    bounds = [(0, 0.5) for _ in range(10)]
    
    # Optimization
    result = minimize(objective, radii, method='L-BFGS-B', bounds=bounds, constraints={"type": "ineq", "fun": constraints})
    
    # Extract result
    final_radii = result.x
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    sum_radii = np.sum(final_radii)
    
    return (centers, final_radii, sum_radii)
