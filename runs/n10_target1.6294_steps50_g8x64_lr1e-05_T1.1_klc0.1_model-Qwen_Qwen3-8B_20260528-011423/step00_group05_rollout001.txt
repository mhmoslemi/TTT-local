import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.05

    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii

    def constraints(c, r):
        # Box constraints: circles must be inside the unit square
        box_cons = []
        for i in range(10):
            x, y = c[i]
            r_i = r[i]
            box_cons.append((x - r_i >= 0, x + r_i <= 1))
            box_cons.append((y - r_i >= 0, y + r_i <= 1))
        return box_cons

    def non_overlap_constraints(c, r):
        # Non-overlap constraints: distance between centers >= sum of radii
        n = len(c)
        cons = []
        for i in range(n):
            for j in range(i+1, n):
                dx = c[i, 0] - c[j, 0]
                dy = c[i, 1] - c[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                cons.append((dist >= r[i] + r[j]))
        return cons

    # Combine constraints into a single list
    def all_constraints(c, r):
        box = constraints(c, r)
        non_overlap = non_overlap_constraints(c, r)
        return box + non_overlap

    # Optimization
    def objective_fun(x):
        return -np.sum(x)

    def constraint_fun(x):
        c = x[:60].reshape(10, 2)
        r = x[60:]
        constraints = all_constraints(c, r)
        return [c[0] for c in constraints]

    # Set up bounds and constraints for the optimization
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])
    bounds.extend([(0, 0.5) for _ in range(10)])  # Radii are between 0 and 0.5

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Set up optimization problem
    res = minimize(
        objective_fun,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_fun},
        tol=1e-6,
        options={'maxiter': 500}
    )

    # Extract results
    centers_opt = res.x[:60].reshape(10, 2)
    radii_opt = res.x[60:]
    sum_radii = -res.fun

    return (centers_opt, radii_opt, sum_radii)
