import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for j in range(5) for i in range(2)])
    radii = np.full(10, 0.05)
    
    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[2:12])
    
    # Constraints: ensure circles are fully inside the unit square
    def constraint_center(x, i):
        r = x[2 + i]
        cx, cy = x[:2 + i], x[2 + i + 1]
        return np.array([cx - r, cy - r, 1 - cx - r, 1 - cy - r])
    
    # Constraint for each circle being within the square
    def constraint_square(x):
        return np.concatenate([constraint_center(x, i) for i in range(10)])
    
    # Constraints: distance between any two centers >= sum of their radii
    def constraint_overlap(x):
        dists = np.zeros(45)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x[2*i] - x[2*j]
                dy = x[2*i + 1] - x[2*j + 1]
                dist = np.sqrt(dx*dx + dy*dy)
                dists[10*i + j] = dist - (x[2 + i] + x[2 + j])
        return dists
    
    # Define bounds for all variables (x, y, r)
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # r
    
    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': lambda x: constraint_square(x)},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Initial guess (flattened)
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Perform optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons)
    
    # Extract centers, radii, and sum of radii
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
