import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced points inside the unit square
    num_circles = 10
    centers = np.random.rand(num_circles, 2)
    radii = np.ones(num_circles) * 0.1

    # Objective function: minimize negative sum of radii (maximize sum of radii)
    def objective(x):
        # Flatten input for easier reshaping
        x_flat = x.reshape(-1)
        centers = x_flat.reshape(num_circles, 2)
        radii = np.abs(x_flat[num_circles:])  # Radii are the latter part of the flattened array
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints: circles must be inside the unit square
    bounds = []
    for i in range(num_circles):
        bounds.extend([(0, 1), (0, 1), (0, 1)])  # x, y, radius for each circle

    # Constraints: circles must not overlap
    def constraint_overlap(x):
        x_flat = x.reshape(-1)
        centers = x_flat[:num_circles*2].reshape(num_circles, 2)
        radii = np.abs(x_flat[num_circles*2:])
        violations = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                min_distance = radii[i] + radii[j]
                violations.append(distance - min_distance)
        return np.min(violations)

    # Add constraints to ensure no overlap
    constraints = [{'type': 'ineq', 'fun': lambda x: constraint_overlap(x)}]

    # Constraints: circles must be inside the unit square
    def constraint_inside(x):
        x_flat = x.reshape(-1)
        centers = x_flat[:num_circles*2].reshape(num_circles, 2)
        radii = np.abs(x_flat[num_circles*2:])
        for i in range(num_circles):
            x, y, r = centers[i, 0], centers[i, 1], radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -1
        return 0

    constraints.append({'type': 'ineq', 'fun': lambda x: constraint_inside(x)})

    # Optimization
    result = minimize(objective, np.concatenate([centers.ravel(), radii]), bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    x_flat = result.x
    centers = x_flat[:num_circles*2].reshape(num_circles, 2)
    radii = np.abs(x_flat[num_circles*2:])
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
