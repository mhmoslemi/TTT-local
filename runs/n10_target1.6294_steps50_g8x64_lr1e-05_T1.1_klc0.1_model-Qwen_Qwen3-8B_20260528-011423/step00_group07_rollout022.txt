import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: maximize the sum of the radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_1(radii):
        # Each circle is inside the unit square
        return np.array([r for r in radii if not (0 <= r <= 1 - r)])
    
    def constraint_2(radii):
        # No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.array([dists[i, j] - (radii[i] + radii[j]) for i in range(10) for j in range(i+1)])
    
    constraints = [
        {'type': 'ineq', 'fun': lambda r: r},
        {'type': 'ineq', 'fun': constraint_1},
        {'type': 'ineq', 'fun': constraint_2}
    ]

    # Bounds for radii
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimize
    result = minimize(objective, np.full(10, 0.1), bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
