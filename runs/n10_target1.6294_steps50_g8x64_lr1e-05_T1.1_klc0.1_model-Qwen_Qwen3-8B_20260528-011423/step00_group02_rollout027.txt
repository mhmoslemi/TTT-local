import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_1(radii):
        # Ensure circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            for j in range(10):
                if i == j:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Use a custom constraint to check if all circles are valid
    def constraint_box(radii):
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        return True

    # Combine constraints into a single function
    def constraints(radii):
        return constraint_box(radii) and constraint_1(radii)

    # Set up bounds for radius and centers
    bounds = [(0, 0.5) for _ in range(10)]
    # We'll optimize radii and centers together with constraints

    # We'll reparameterize the problem to optimize the centers and radii
    # by flattening them into a single vector

    # Initial guess for all variables
    initial_guess = np.concatenate((centers.flatten(), radii))

    def objective_vectorized(x):
        # Reshape x to (10, 3) where the first two entries are x and y, third is radius
        # Split into circles
        n_circles = 10
        centers_flat = x[:n_circles * 2].reshape(n_circles, 2)
        radii_flat = x[n_circles * 2:].reshape(n_circles)
        return -np.sum(radii_flat)

    def constraint_box_vectorized(x):
        centers_flat = x[:n_circles * 2].reshape(n_circles, 2)
        radii_flat = x[n_circles * 2:].reshape(n_circles)
        for i in range(n_circles):
            cx, cy = centers_flat[i]
            r = radii_flat[i]
            if cx - r < 0 or cx + r > 1 or cy - r < 0 or cy + r > 1:
                return False
        return True

    def constraint_overlap_vectorized(x):
        centers_flat = x[:n_circles * 2].reshape(n_circles, 2)
        radii_flat = x[n_circles * 2:].reshape(n_circles)
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return False
        return True

    def constraint_bounds(x):
        return [x[i] >= 0 for i in range(n_circles * 3)]

    # Optimize
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=UserWarning)
        result = minimize(
            objective_vectorized,
            initial_guess,
            method='L-BFGS-B',
            bounds=[(0, 1) for _ in range(n_circles * 3)],
            constraints=[
                {'type': 'ineq', 'fun': constraint_box_vectorized},
                {'type': 'ineq', 'fun': constraint_overlap_vectorized}
            ]
        )

    # Extract results
    centers_flat = result.x[:n_circles * 2].reshape(n_circles, 2)
    radii_flat = result.x[n_circles * 2:].reshape(n_circles)
    sum_radii = np.sum(radii_flat)

    return (centers_flat, radii_flat, sum_radii)
