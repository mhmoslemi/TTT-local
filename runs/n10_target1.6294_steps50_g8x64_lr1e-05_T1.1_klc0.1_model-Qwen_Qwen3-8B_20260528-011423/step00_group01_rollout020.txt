import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1  # Small initial radii

    def objective_function(params):
        # Flatten parameters into centers and radii
        params_flat = params.reshape(10, 3)
        centers = params_flat[:, :2]
        radii = params_flat[:, 2]

        # Objective: maximize sum of radii, so minimize negative sum
        sum_radii = -np.sum(radii)
        return sum_radii

    def constraint_circle_inside(params):
        params_flat = params.reshape(10, 3)
        centers = params_flat[:, :2]
        radii = params_flat[:, 2]
        constraints = []
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            constraints.append(x - r >= 0)
            constraints.append(1 - x - r >= 0)
            constraints.append(y - r >= 0)
            constraints.append(1 - y - r >= 0)
        return np.array(constraints)

    def constraint_no_overlap(params):
        params_flat = params.reshape(10, 3)
        centers = params_flat[:, :2]
        radii = params_flat[:, 2]
        constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]) >= 0)
        return np.array(constraints)

    # Combine constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization bounds: centers in [0,1], radii >= 0
    bounds = [(0, 1) for _ in range(10 * 2)] + [(0, 1) for _ in range(10)]

    # Initial parameters: flatten centers and radii
    initial_params = np.concatenate([centers.reshape(10 * 2), radii])

    # Set up optimization
    result = minimize(
        objective_function,
        initial_params,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract result
    params_flat = result.x.reshape(10, 3)
    centers_result = params_flat[:, :2]
    radii_result = params_flat[:, 2]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
