import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3],
    ])
    radii = np.array([0.1] * 10)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint: circles must be fully inside the unit square
        radii = np.atleast_1d(radii)
        in_bounds = np.all(np.logical_and(radii <= 1 - centers[:, 0], radii <= 1 - centers[:, 1]), axis=1)
        in_bounds &= np.all(np.logical_and(radii >= centers[:, 0], radii >= centers[:, 1]), axis=1)
        if not np.all(in_bounds):
            return np.inf
        # Constraint: no overlapping between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0.0

    # Use scipy.optimize.minimize with constraints
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=UserWarning)
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            constraints=[{'type': 'ineq', 'fun': lambda x: constraints(x, centers)}],
            bounds=[(0, 1 - 0.1) for _ in range(10)]
        )

    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3],
    ])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
