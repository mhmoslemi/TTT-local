import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Check if all radii are non-negative and within the square
        for i in range(10):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Check if all pairs of circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    def constraint_func(x):
        radii = x
        return -1 if not constraints(radii) else 0

    def constraint_func_min(x):
        return constraint_func(x)

    # Optimization
    bounds = [(0, 0.5) for _ in range(10)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_func_min})

    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
