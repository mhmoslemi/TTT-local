import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles evenly in a grid
    def initial_guess():
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
        return centers

    # Objective function: minimize the negative sum of radii
    def objective(x):
        radii = x[::2]
        centers = x[1::2].reshape(-1, 2)
        return -np.sum(radii)

    # Constraints: each circle is inside the square
    def square_constraints(x):
        radii = x[::2]
        centers = x[1::2].reshape(-1, 2)
        return np.concatenate([radii - centers[:, 0], 1 - radii - centers[:, 0],
                               radii - centers[:, 1], 1 - radii - centers[:, 1]])

    # Constraints: circles do not overlap
    def no_overlap_constraints(x):
        radii = x[::2]
        centers = x[1::2].reshape(-1, 2)
        n = len(centers)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    # Initial parameters
    centers = initial_guess()
    radii = np.full(10, 0.1)
    x_initial = np.concatenate([radii, centers.flatten()])

    # Setup optimization
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1 - 1e-6)])  # Radius bounds
        bounds.extend([(0, 1), (0, 1)])  # Center coordinates

    cons = [
        {'type': 'ineq', 'fun': lambda x: square_constraints(x)},
        {'type': 'ineq', 'fun': lambda x: no_overlap_constraints(x)}
    ]

    result = minimize(
        objective,
        x_initial,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract results
    x_opt = result.x
    radii_opt = x_opt[::2]
    centers_opt = x_opt[1::2].reshape(-1, 2)
    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
