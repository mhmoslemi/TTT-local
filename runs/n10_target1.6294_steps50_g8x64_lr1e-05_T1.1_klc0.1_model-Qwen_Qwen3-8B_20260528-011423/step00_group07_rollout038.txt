import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2) * 0.8 + 0.1  # Random initial positions within [0.1, 0.9]
    radii = np.random.rand(10) * 0.1 + 0.05  # Random initial radii

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)

    # Constraints: circles are fully inside the unit square
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.array([1 - r - x[i, 0] for i, r in enumerate(radii_flat)] +
                        [r - x[i, 0] for i, r in enumerate(radii_flat)] +
                        [1 - r - x[i, 1] for i, r in enumerate(radii_flat)] +
                        [r - x[i, 1] for i, r in enumerate(radii_flat)])

    # Constraints: circles do not overlap
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        dists = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Bounds: centers in [0, 1], radii in [0, 0.5]
    bounds = [(0, 1) for _ in range(60)] + [(0, 0.5) for _ in range(10)]

    # Define constraints
    cons_inside = {'type': 'ineq', 'fun': constraint_inside}
    cons_overlap = {'type': 'ineq', 'fun': constraint_overlap}
    constraints = [cons_inside, cons_overlap]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0=np.concatenate([centers.ravel(), radii]), method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
