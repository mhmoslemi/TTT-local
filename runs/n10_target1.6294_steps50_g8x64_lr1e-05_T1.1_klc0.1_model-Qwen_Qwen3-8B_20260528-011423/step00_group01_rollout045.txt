import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: distribute circles evenly
    centers = np.array([(i / 9 + 0.5, j / 9 + 0.5) for i in range(10) for j in range(10)][:10])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: All circles must be inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return np.inf
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return np.inf

        # Constraint 2: No two circles overlap
        dists = cdist(centers, centers)
        np.fill_diagonal(dists, np.inf)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    bounds = [(0, 1) for _ in range(10)]
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    if result.success:
        final_radii = result.x
        final_centers = centers.copy()
        sum_radii = np.sum(final_radii)
        return (final_centers, final_radii, sum_radii)
    else:
        return (centers, radii, np.sum(radii))
