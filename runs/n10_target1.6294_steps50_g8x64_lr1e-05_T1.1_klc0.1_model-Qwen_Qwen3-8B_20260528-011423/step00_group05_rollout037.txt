import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles at a regular grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # We maximize sum of radii by minimizing -sum(radii)

    def constraint_circle_inside(radii):
        # Ensure each circle is inside the unit square
        return np.array([r <= 1 - x - r and r <= 1 - y - r for x, y, r in zip(centers[:, 0], centers[:, 1], radii)])
    
    def constraint_no_overlap(radii):
        # Ensure no two circles overlap
        dists = np.sqrt(np.sum((centers[np.newaxis, :, :] - centers[:, np.newaxis, :])**2, axis=2))
        return np.array([dists[i, j] >= radii[i] + radii[j] for i in range(10) for j in range(i+1)])

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            bounds=[(0, 1 - 0.1) for _ in range(10)],
            constraints=[
                {'type': 'ineq', 'fun': constraint_circle_inside},
                {'type': 'ineq', 'fun': constraint_no_overlap}
            ]
        )

    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
