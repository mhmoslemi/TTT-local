import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced circles in a grid
    centers = np.array([[i * 0.25 + 0.125, j * 0.25 + 0.125] for i in range(3) for j in range(3)])
    radii = np.full(9, 0.1)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add one more circle in the center
    radii = np.append(radii, 0.1)
    
    def objective(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x.reshape(10, 3)[:, :2]
        radii_flat = x.reshape(10, 3)[:, 2]
        return -np.sum(radii_flat)  # We want to maximize the sum of radii
    
    def constraints(x):
        centers_flat = x.reshape(10, 3)[:, :2]
        radii_flat = x.reshape(10, 3)[:, 2]
        
        # Constraint 1: All circles are within the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0
    
    # Initial guess flattened
    x0 = np.array([centers[i, 0] for i in range(10)] + [centers[i, 1] for i in range(10)] + [radii[i] for i in range(10)])
    
    # Set up optimization
    res = minimize(objective, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints})
    
    # Extract result
    x_result = res.x
    centers_result = x_result.reshape(10, 3)[:, :2]
    radii_result = x_result.reshape(10, 3)[:, 2]
    sum_radii = np.sum(radii_result)
    
    return (centers_result, radii_result, sum_radii)
