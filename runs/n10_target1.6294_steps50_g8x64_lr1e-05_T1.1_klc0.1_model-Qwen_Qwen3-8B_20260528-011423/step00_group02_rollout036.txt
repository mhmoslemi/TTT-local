import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective_function(params):
        # Unpack parameters
        flat_centers = params[:60].reshape(10, 2)
        flat_radii = params[60:]
        radii = flat_radii
        # Objective: maximize sum of radii (minimize negative sum)
        sum_radii = -np.sum(radii)
        # Check if all circles are within the unit square
        for i in range(10):
            x, y = flat_centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                # Penalize out-of-bounds positions
                sum_radii -= 10000
        # Check if any circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = flat_centers[i, 0] - flat_centers[j, 0]
                dy = flat_centers[i, 1] - flat_centers[j, 1]
                distance = np.sqrt(dx*dx + dy*dy)
                if distance < radii[i] + radii[j]:
                    # Penalize overlapping circles
                    sum_radii -= 10000
        return sum_radii

    # Bounds for parameters: [0, 1] for centers, [0, 0.5] for radii
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # x-coordinate for center i
        bounds.append((0, 1))  # y-coordinate for center i
        bounds.append((0, 0.5))  # radius i

    # Optimization
    result = minimize(
        objective_function,
        x0=np.concatenate([centers.ravel(), radii]),
        method='SLSQP',
        bounds=bounds,
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    # Extract results
    params = result.x
    centers = params[:60].reshape(10, 2)
    radii = params[60:]
    sum_radii = -objective_function(params)

    # Return the result
    return (centers, radii, sum_radii)
