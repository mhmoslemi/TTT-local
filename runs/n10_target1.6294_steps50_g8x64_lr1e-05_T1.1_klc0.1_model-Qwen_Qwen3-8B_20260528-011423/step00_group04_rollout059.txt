import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess for circle centers and radii
    # Place circles in a grid with equal spacing
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.full(10, 0.05)

    def objective_function(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers = x.reshape(10, 2)
        radii = x[2::3]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers = x.reshape(10, 2)
        radii = x[2::3]
        constraints = []

        # Constraint: all circles are inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers[i, 0], centers[i, 1], radii[i]
            constraints.append((x_i - r_i >= 0) * 1e6)
            constraints.append((x_i + r_i <= 1) * 1e6)
            constraints.append((y_i - r_i >= 0) * 1e6)
            constraints.append((y_i + r_i <= 1) * 1e6)

        # Constraint: no two circles overlap
        for i, j in itertools.combinations(range(10), 2):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            r_i = radii[i]
            r_j = radii[j]
            constraints.append((dist - (r_i + r_j) >= 0) * 1e6)

        return np.array(constraints)

    # Bounds for x, y, and radii
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1), (0, 0.5)])

    # Initial guess
    x0 = np.array([initial_centers[i, 0] for i in range(10)] +
                  [initial_centers[i, 1] for i in range(10)] +
                  [initial_radii[i] for i in range(10)])

    # Use SLSQP method with bounds and constraints
    result = minimize(
        objective_function,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    x_opt = result.x
    centers = x_opt.reshape(10, 2)
    radii = x_opt[2::3]
    sum_radii = np.sum(radii)

    # Return the result
    return (centers, radii, sum_radii)
