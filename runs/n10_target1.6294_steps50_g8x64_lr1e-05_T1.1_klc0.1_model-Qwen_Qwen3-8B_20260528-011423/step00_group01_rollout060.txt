import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced on a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.ones(10) * 0.15

    def objective(radii):
        return -np.sum(radii)

    def constraints_func(centers, radii):
        n = centers.shape[0]
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraints.append(dist - (radii[i] + radii[j]))
        for i in range(n):
            x, y = centers[i]
            constraints.append(x - radii[i])
            constraints.append(1 - x - radii[i])
            constraints.append(y - radii[i])
            constraints.append(1 - y - radii[i])
        return constraints

    def objective_func(x):
        radii = x
        return -np.sum(radii)

    def constraint_func(x):
        centers = np.reshape(x[:20], (10, 2))
        radii = x[20:]
        cons = constraints_func(centers, radii)
        return cons

    # Initial guess: 20 values for centers (10*2) + 10 for radii
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Set up constraints
    cons = {'type': 'ineq', 'fun': constraint_func}

    # Optimization settings
    bounds = []
    for _ in range(20):  # centers x and y
        bounds.append((0, 1))
    for _ in range(10):  # radii
        bounds.append((0, 1))

    # Optimize
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective_func, initial_guess, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    centers_opt = np.reshape(result.x[:20], (10, 2))
    radii_opt = result.x[20:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
