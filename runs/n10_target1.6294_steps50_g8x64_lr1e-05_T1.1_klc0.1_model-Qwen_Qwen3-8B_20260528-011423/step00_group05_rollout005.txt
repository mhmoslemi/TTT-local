import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: 10 circles placed in a grid
    initial_centers = np.column_stack([np.linspace(0.1, 0.9, 10), np.linspace(0.1, 0.9, 10)])
    initial_radii = np.full(10, 0.1)

    def objective_function(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        # Constraint: circles must be inside the unit square
        return np.concatenate([radii - (initial_centers[:, 0] - 0.001), 
                              radii - (1 - initial_centers[:, 1] - 0.001)])

    def constraint_no_overlap(radii):
        # Constraint: no two circles overlap
        centers = initial_centers
        dists = np.zeros((10, 10))
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization bounds
    bounds = Bounds(0, 1 - 0.001)

    # Objective function and constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(objective_function, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints)

    centers = initial_centers
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
