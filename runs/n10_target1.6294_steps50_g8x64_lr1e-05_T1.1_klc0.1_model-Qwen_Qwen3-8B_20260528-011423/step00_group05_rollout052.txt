import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.08)

    def objective(radii):
        return -np.sum(radii)  # We minimize the negative sum to maximize the sum

    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        radii = radii.reshape(10)
        dist = np.zeros(10)
        for i in range(10):
            dist[i] = np.sqrt((centers[i, 0] - centers[i, 0])**2 + (centers[i, 1] - centers[i, 1])**2)
        for i in range(10):
            for j in range(i + 1, 10):
                dist[i, j] = np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2)
        for i in range(10):
            dist[i, i] = np.inf
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        for i in range(10):
            for j in range(i + 1, 10):
                if dist[i, j] < radii[i] + radii[j]:
                    return False
        return True

    result = minimize(objective, radii, method='SLSQP', constraints=constraints, bounds=[(0, 1)]*10)

    centers_final = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii_final = result.x
    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
