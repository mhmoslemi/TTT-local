import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radii: all the same
    radii = np.full(10, 0.1)

    # Objective function: minimize negative sum of radii
    def objective(x):
        radii = x[:10]
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints: circles must be inside the unit square and non-overlapping
    def constraint_inside(radii):
        return np.concatenate([radii, np.zeros(10)]), \
               np.array([r for r in radii] + [0.0 for _ in range(10)])

    def constraint_overlap(x):
        radii = x[:10]
        centers = x[10:]
        centers = centers.reshape((10, 2))
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        overlaps = np.triu(dists - radii[:, np.newaxis] - radii[np.newaxis, :], k=1)
        return np.min(overlaps)

    # Initial parameters for optimization
    x0 = np.concatenate([radii, centers.flatten()])

    # Bounds: radii >= 0 and centers in [0, 1]
    bounds = [(0, 1) for _ in range(10)] + [(0, 1) for _ in range(20)]

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=[
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ])

    # Extract results
    radii = result.x[:10]
    centers = result.x[10:].reshape((10, 2))

    # Ensure all circles are within the unit square (due to constraint_inside)
    for i in range(10):
        if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i] or \
           centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
            radii[i] = min(radii[i], min(centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1]))

    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
