import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint function for square boundaries: r_i <= x_i - r_i and r_i <= y_i - r_i
    def constraint_square(radii, centers):
        return np.array([radii[i] <= centers[i, 0] - radii[i] and radii[i] <= centers[i, 1] - radii[i] for i in range(10)])

    # Constraint function for non-overlapping circles
    def constraint_overlap(radii, centers):
        n = len(radii)
        dists = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                dists.append(dist)
        return np.array(dists)

    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(10)]

    # Optimization setup
    initial_guess = np.ones(10) * 0.1
    result = minimize(
        objective,
        initial_guess,
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_square, 'args': (centers,)},
            {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers,)}
        ],
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    radii = result.x
    centers_result = centers.copy()
    sum_radii = np.sum(radii)

    # Final check: ensure all circles are within the square and non-overlapping
    for i in range(10):
        x, y = centers_result[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            warnings.warn("One or more circles are not fully inside the square.", UserWarning)

    return (centers_result, radii, sum_radii)
