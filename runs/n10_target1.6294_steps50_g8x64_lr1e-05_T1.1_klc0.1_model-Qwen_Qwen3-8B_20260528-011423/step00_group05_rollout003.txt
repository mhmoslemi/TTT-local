import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced grid of 10 points in the unit square
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(centers, radii):
        # Constraint 1: All circles are inside the square
        for i in range(10):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_func(x):
        radii = x
        if not constraints(centers, radii):
            return np.inf
        return -np.sum(radii)

    def bounds_func(x):
        return [(0, 1.0) for _ in range(10)]

    # Optimization loop with simulated annealing
    for _ in range(10):
        new_centers = np.random.rand(10, 2)
        new_radii = np.random.rand(10) * 0.5
        if constraints(new_centers, new_radii):
            centers = new_centers
            radii = new_radii

    # Optimize with scipy
    res = minimize(objective_func, radii, bounds=bounds_func, method='L-BFGS-B', options={'ftol': 1e-8, 'maxiter': 100})
    optimal_radii = res.x
    if not constraints(centers, optimal_radii):
        optimal_radii = radii
    sum_radii = np.sum(optimal_radii)
    return (centers, optimal_radii, sum_radii)
