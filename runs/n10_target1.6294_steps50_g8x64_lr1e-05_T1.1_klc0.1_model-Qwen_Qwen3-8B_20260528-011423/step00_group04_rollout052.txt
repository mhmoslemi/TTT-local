import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    radii = np.clip(radii, 0, 0.5)  # Set an upper bound to stay within the unit square

    # Objective function to maximize the sum of radii
    def objective(x):
        # x is a flattened array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints: circle centers must be within the unit square
    def constraint_center(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        return np.sum((centers_flat - radii_flat[:, np.newaxis])[:, :, 0] < 0) + \
               np.sum((centers_flat - radii_flat[:, np.newaxis])[:, :, 0] > 1) + \
               np.sum((centers_flat - radii_flat[:, np.newaxis])[:, :, 1] < 0) + \
               np.sum((centers_flat - radii_flat[:, np.newaxis])[:, :, 1] > 1)

    # Constraint: circles must not overlap
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        overlap = 0
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    overlap += 1
        return overlap

    # Bounds: centers are in [0, 1], radii are in [0, 0.5]
    bounds = [(0, 1) for _ in range(20)] + [(0, 0.5) for _ in range(10)]

    # Initial guess
    x0 = np.concatenate([centers.ravel(), radii.ravel()])

    # Optimization with constraints
    def constraint_center_func(x):
        return constraint_center(x)

    def constraint_overlap_func(x):
        return constraint_overlap(x)

    constraints = [
        {'type': 'ineq', 'fun': constraint_center_func},
        {'type': 'ineq', 'fun': constraint_overlap_func}
    ]

    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract result
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:].reshape(10)
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
