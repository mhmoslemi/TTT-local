import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
from scipy.spatial.distance import pdist

def run_packing():
    def objective_function(params):
        # Extract centers and radii from parameters
        centers = params[:20].reshape(10, 2)
        radii = params[20:]
        
        # Check if all circles are inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii))
        if not inside:
            return np.inf  # Penalize if any circle is out of bounds
        
        # Check if circles are non-overlapping
        distances = pdist(centers)
        radii_sum = np.add.outer(radii, radii).flatten()
        overlaps = np.any(distances < radii_sum)
        if overlaps:
            return np.inf  # Penalize if any circles overlap
        
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    # Initial guess: place circles in a grid
    initial_centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    initial_radii = np.array([0.05] * 10)
    initial_params = np.concatenate([initial_centers.flatten(), initial_radii])

    # Optimization settings
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 0.999999)] * 2)  # x and y coordinates
        bounds.extend([(0.0, 0.999999)] * 1)  # radius

    result = minimize(objective_function, initial_params, method='L-BFGS-B', bounds=bounds, tol=1e-8)

    # Extract final result
    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
