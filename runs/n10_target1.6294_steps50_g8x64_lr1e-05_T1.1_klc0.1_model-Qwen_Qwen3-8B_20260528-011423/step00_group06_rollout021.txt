import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    initial_centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    initial_radii = np.full(10, 0.05)

    # Objective function: minimize the negative sum of radii
    def objective(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)

    # Constraints
    def constraint_centers(x):
        centers = x[:60].reshape(10, 2)
        return np.array([x < 0 for x in centers[:, 0]] + [x > 1 for x in centers[:, 0]] + \
                        [y < 0 for y in centers[:, 1]] + [y > 1 for y in centers[:, 1]])

    def constraint_radii(x):
        radii = x[60:]
        return radii < 0

    def constraint_overlap(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=-1))
        sums = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dists - sums

    # Optimization
    bounds = []
    for _ in range(10):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
    bounds += [(0, 1)] * 10  # radii

    constraints = [
        {'type': 'ineq', 'fun': lambda x: np.min(np.array([1 - x[i] for i in range(10)])) - 2 * x[50 + i] for i in range(10)},
        {'type': 'ineq', 'fun': lambda x: np.min(np.array([x[i] for i in range(10)])) - 2 * x[50 + i] for i in range(10)},
        {'type': 'ineq', 'fun': lambda x: np.min(np.array([1 - x[i + 10] for i in range(10)])) - 2 * x[50 + i] for i in range(10)},
        {'type': 'ineq', 'fun': lambda x: np.min(np.array([x[i + 20] for i in range(10)])) - 2 * x[50 + i] for i in range(10)},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(objective, np.r_[initial_centers.ravel(), initial_radii], bounds=bounds, constraints=constraints)

    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
