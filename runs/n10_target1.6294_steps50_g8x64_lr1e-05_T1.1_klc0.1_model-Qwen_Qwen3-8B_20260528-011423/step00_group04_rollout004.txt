import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid pattern with equal spacing
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        # Reshape into centers and radii
        x_flat = np.reshape(x, (-1, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        # Objective: maximize sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraint_inside(x):
        x_flat = np.reshape(x, (-1, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        # Check if all circles are inside the unit square
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0]) + \
               np.minimum(centers[:, 1] - radii, 1 - centers[:, 1])

    def constraint_no_overlap(x):
        x_flat = np.reshape(x, (-1, 3))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        # Compute pairwise distances and check against sum of radii
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2).sum(axis=2))
        constraints = (dist >= radii[:, np.newaxis] + radii[np.newaxis, :])
        return constraints.min(axis=1)

    # Bounds for each parameter: x and y between 0 and 1, r >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])

    # Initial guess flattened
    x0 = np.reshape(centers, (-1, 3)).flatten()

    # Constraints
    constraints_inside = {'type': 'ineq', 'fun': constraint_inside}
    constraints_no_overlap = {'type': 'ineq', 'fun': constraint_no_overlap}

    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints=[constraints_inside, constraints_no_overlap])

    # Extract result
    x_opt = result.x
    centers_opt = np.reshape(x_opt, (-1, 3))[:, :2]
    radii_opt = np.reshape(x_opt, (-1, 3))[:, 2]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
