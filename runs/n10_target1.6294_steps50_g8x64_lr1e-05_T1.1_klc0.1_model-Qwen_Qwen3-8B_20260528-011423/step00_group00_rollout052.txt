import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as dist

def run_packing():
    def objective_function(positions):
        return -np.sum(positions[-10:])  # Maximize sum of radii

    def constraints(positions):
        # Extract centers and radii
        centers = positions[:20].reshape(10, 2)
        radii = positions[20:]
        # Constraint 1: Each circle fully inside the unit square
        for i in range(10):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                d = np.linalg.norm(centers[i] - centers[j])
                if d < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Initial guess: spread circles evenly in the square
    positions = np.array([
        [0.1, 0.1, 0.1], [0.1, 0.3, 0.1], [0.1, 0.5, 0.1], [0.1, 0.7, 0.1], [0.1, 0.9, 0.1],
        [0.3, 0.1, 0.1], [0.3, 0.3, 0.1], [0.3, 0.5, 0.1], [0.3, 0.7, 0.1], [0.3, 0.9, 0.1],
        [0.5, 0.1, 0.1], [0.5, 0.3, 0.1], [0.5, 0.5, 0.1], [0.5, 0.7, 0.1], [0.5, 0.9, 0.1],
        [0.7, 0.1, 0.1], [0.7, 0.3, 0.1], [0.7, 0.5, 0.1], [0.7, 0.7, 0.1], [0.7, 0.9, 0.1],
        [0.9, 0.1, 0.1], [0.9, 0.3, 0.1], [0.9, 0.5, 0.1], [0.9, 0.7, 0.1], [0.9, 0.9, 0.1]
    ])

    # Setup optimization problem
    bounds = []
    for _ in range(20):
        bounds.append((0.01, 0.5))  # Radii are at least 0.01, at most 0.5
    for _ in range(10):  # x and y coordinates of centers
        bounds.append((0.01, 0.99))  # Center in [0.01, 0.99] to ensure radius fits

    result = minimize(
        objective_function,
        positions,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract results
    positions_opt = result.x
    centers = positions_opt[:20].reshape(10, 2)
    radii = positions_opt[20:]
    sum_radii = np.sum(radii)

    # Check final constraints
    for i in range(10):
        x, y = centers[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            sum_radii = 0.0
            break
    for i in range(10):
        for j in range(i + 1, 10):
            d = np.linalg.norm(centers[i] - centers[j])
            if d < radii[i] + radii[j]:
                sum_radii = 0.0
                break
        if sum_radii == 0.0:
            break

    return (centers, radii, sum_radii)
