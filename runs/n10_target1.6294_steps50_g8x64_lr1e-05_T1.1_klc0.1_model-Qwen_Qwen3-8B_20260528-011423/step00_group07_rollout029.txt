import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraint: radii must be non-negative
        if np.any(radii < 0):
            return np.inf

        # Constraint: circles must be inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return np.inf

        # Constraint: circles must not overlap
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf

        return 0

    # Flatten radii for optimization
    initial_guess = radii.flatten()
    bounds = [(0, 0.5) for _ in range(10)]  # Upper bound on radius to stay within square

    # Add constraints to the optimization problem
    cons = [
        {'type': 'ineq', 'fun': lambda r: -constraints(r.reshape(10))},
    ]

    # Perform the optimization
    result = minimize(
        fun=objective,
        x0=initial_guess,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'maxiter': 1000, 'ftol': 1e-9}
    )

    # Extract the optimal radii and centers
    optimal_radii = result.x
    optimal_sum = -result.fun  # Objective is negative sum, so negate to get sum

    return (centers, optimal_radii, optimal_sum)
