import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: equally spaced circles in a grid-like pattern
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        # Constraint: all circles are inside the square
        # Each radius <= x - 0 and x <= 1 - r, same for y
        x_coords = centers[:, 0] - radii
        y_coords = centers[:, 1] - radii
        x_max = 1 - radii
        y_max = 1 - radii
        return np.concatenate((x_coords, y_coords, x_max, y_max))
    
    def circle_distances(radii):
        # Distance between centers must be >= sum of radii
        dists = np.zeros(45)
        idx = 0
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[idx] = dist - radii[i] - radii[j]
                idx += 1
        return dists
    
    def wrapper(radii):
        # Optimization objective
        return objective(radii)
    
    # Optimization bounds: all radii >= 0
    bounds = [(0.0, 1.0) for _ in range(10)]
    
    # Constraints: all inside square, and no overlap
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraints(r)},
        {'type': 'ineq', 'fun': lambda r: circle_distances(r)}
    ]
    
    result = minimize(wrapper, radii, bounds=bounds, constraints=cons)
    
    # Extract the optimized result
    optimized_radii = result.x
    optimized_sum = np.sum(optimized_radii)
    
    return (centers, optimized_radii, optimized_sum)
