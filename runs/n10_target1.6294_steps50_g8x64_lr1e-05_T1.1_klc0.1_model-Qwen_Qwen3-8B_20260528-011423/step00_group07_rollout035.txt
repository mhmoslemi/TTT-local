import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers: evenly spaced grid
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    # Initial guess for radii: all 0.1
    radii = np.full(9, 0.1)
    # Add one more circle at the center
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = np.append(radii, 0.1)

    def objective(radii):
        return -sum(radii)

    def constraints(radii):
        # Convert radii to a numpy array
        radii = np.array(radii)
        # Check if all circles are within the unit square
        centers = np.array([[0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],
                            [0.3, 0.1], [0.3, 0.3], [0.3, 0.5], [0.3, 0.7], [0.3, 0.9]])
        # Check if circles stay within the square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Check if circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_and_constraints(radii):
        # Objective function to maximize sum of radii (minimize -sum)
        result = -sum(radii)
        # Constraints function
        constraint_result = constraints(radii)
        return result, constraint_result

    # Optimization settings
    bounds = [(0.0, 0.5) for _ in range(10)]
    result = minimize(
        lambda r: -sum(r),
        x0=radii,
        bounds=bounds,
        constraints={
            'type': 'ineq',
            'fun': lambda r: 0,
            'jac': lambda r: np.zeros_like(r),
            'method': 'SLSQP'
        },
        options={'disp': False, 'ftol': 1e-9}
    )

    # Extract centers, radii, and sum
    centers = np.array([[0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],
                        [0.3, 0.1], [0.3, 0.3], [0.3, 0.5], [0.3, 0.7], [0.3, 0.9]])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
