import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: ten circles arranged in a grid
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    radii = np.clip(radii, 0, 0.5)  # Initial radii between 0 and 0.5

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(c):
        # Ensure circles are fully inside the unit square
        x, y = c[:, 0], c[:, 1]
        r = radii
        return np.minimum(x - r, 1 - x - r, y - r, 1 - y - r)

    def inequality_constraints(c):
        # Ensure no two circles overlap
        dist = np.sqrt(((c[:, np.newaxis, 0] - c[np.newaxis, :, 0])**2 +
                        (c[:, np.newaxis, 1] - c[np.newaxis, :, 1])**2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Pack optimization
    def pack():
        def objective_func(x):
            # x is [centers..., radii...]
            centers = x[:10*2].reshape(10, 2)
            radii = x[10*2:]
            return -np.sum(radii)

        def constraint_func(x):
            centers = x[:10*2].reshape(10, 2)
            radii = x[10*2:]
            # Constraint 1: circle inside square
            x_centers, y_centers = centers[:, 0], centers[:, 1]
            r = radii
            r = np.clip(r, 0, 0.5)
            # Check if circles are within square
            inside = np.all((r <= x_centers) & (x_centers <= 1 - r) &
                            (r <= y_centers) & (y_centers <= 1 - r))
            if not inside:
                return np.inf

            # Constraint 2: circles do not overlap
            dist = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                            (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
            lower_bounds = np.minimum(radii[:, np.newaxis], radii[np.newaxis, :])
            lower = np.sum(np.maximum(0, dist - lower_bounds), axis=(0, 1))
            if lower < 1e-6:
                return np.inf

            return 0

        # Initial guess
        x0 = np.concatenate([centers.reshape(-1), radii])

        # Use L-BFGS-B optimizer
        result = minimize(objective_func, x0, bounds=[(0, 1) for _ in range(10*2)] +
                          [(0, 0.5) for _ in range(10)],
                          constraints={"type": "ineq", "fun": constraint_func, "jac": None},
                          method="L-BFGS-B",
                          options={"ftol": 1e-08, "gtol": 1e-08, "maxiter": 1000})
        
        centers_opt = result.x[:10*2].reshape(10, 2)
        radii_opt = result.x[10*2:]
        radii_opt = np.clip(radii_opt, 0, 0.5)
        sum_r = np.sum(radii_opt)
        return centers_opt, radii_opt, sum_r

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        centers, radii, sum_radii = pack()
    
    return centers, radii, sum_radii
