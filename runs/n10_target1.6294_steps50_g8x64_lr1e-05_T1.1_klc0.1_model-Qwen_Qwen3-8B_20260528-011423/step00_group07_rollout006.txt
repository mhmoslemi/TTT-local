import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place 10 circles in a grid-like pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25

    # Objective: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints:
    # 1. Each circle is inside the unit square
    # 2. No overlap between circles
    # 3. All radii are non-negative

    def constraint_inside(radii):
        return np.array([r for r in radii])

    def constraint_overlap(radii):
        # For each pair of circles
        dists = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                dists.append(dist - (radii[i] + radii[j]))
        return np.array(dists)

    # Setup the optimization problem
    bounds = [(0.0, 1.0) for _ in range(10)]
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(
        objective,
        x0=np.ones(10),
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    radii = result.x
    sum_radii = np.sum(radii)
    centers = np.array(centers)

    return (centers, radii, sum_radii)
