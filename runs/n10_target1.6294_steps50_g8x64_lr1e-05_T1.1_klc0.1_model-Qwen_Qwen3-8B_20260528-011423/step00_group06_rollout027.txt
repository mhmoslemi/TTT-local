import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Center positions and radii are all non-negative, and circles are inside the square
        if np.any(radii < 0):
            return np.inf
        if np.any(radii > 1 - np.min(centers, axis=0)):
            return np.inf

        # Check circle-circle overlaps
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            constraints=[{'type': 'ineq', 'fun': constraints}],
            bounds=[(0, 1 - np.min(centers, axis=0)) for _ in range(10)]
        )

    # Best solution
    best_radii = result.x
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    sum_radii = np.sum(best_radii)
    return (centers, best_radii, sum_radii)
