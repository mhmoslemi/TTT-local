import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: 10 circles with equal radii and centers spaced evenly in a grid
    initial_radii = np.full(10, 0.1)
    initial_centers = np.zeros((10, 2))
    for i in range(10):
        initial_centers[i, 0] = (i % 5) * 0.4
        initial_centers[i, 1] = (i // 5) * 0.4

    def objective_function(params):
        radii = params[:10]
        centers = params[10:].reshape(10, 2)
        # Objective: maximize sum of radii, so minimize negative sum
        sum_radii = -np.sum(radii)
        # Constraint 1: All circles are within the unit square
        for i in range(10):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                sum_radii = -np.inf
                break
        # Constraint 2: No two circles overlap
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    sum_radii = -np.inf
                    break
            if sum_radii == -np.inf:
                break
        return sum_radii

    # Optimization with bounds: centers in [0, 1], radii >= 0
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
        bounds.append((0, 1))  # radius
    bounds = bounds * 10  # 10 centers * 3 parameters each = 30 total parameters

    # Optimization with constraints to avoid invalid configurations
    def constraint1(params):
        radii = params[:10]
        centers = params[10:].reshape(10, 2)
        for i in range(10):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return -np.inf
        return 0

    def constraint2(params):
        radii = params[:10]
        centers = params[10:].reshape(10, 2)
        dists = cdist(centers, centers)
        for i in range(10):
            for j in range(i + 1, 10):
                if dists[i, j] < radii[i] + radii[j]:
                    return -np.inf
        return 0

    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Optimize starting from initial guess
    result = minimize(
        objective_function,
        x0=np.concatenate([initial_radii, initial_centers.reshape(10 * 2)]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 500}
    )

    if result.success:
        params = result.x
        radii = params[:10]
        centers = params[10:].reshape(10, 2)
        sum_radii = -objective_function(params)
        return (centers, radii, sum_radii)
    else:
        # Fallback to initial guess if optimization fails
        return (initial_centers, initial_radii, np.sum(initial_radii))
