import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of centers
    centers = np.array([[i/3, j/3] for i in range(4) for j in range(4)])
    centers[4:] = np.array([[0.5, 0.25], [0.5, 0.75], [0.25, 0.5], [0.75, 0.5]])
    radii = np.full(10, 0.1)  # Initial guess for radii

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum (i.e., maximize sum)

    def constraint_circle_inside(radii, centers):
        # All circles must be inside the unit square
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0] - radii, centers[:, 1] - radii, 1 - centers[:, 1] - radii)

    def constraint_no_overlap(radii, centers):
        # No two circles can overlap
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - radii[:, np.newaxis] - radii[np.newaxis, :]

    # Set up optimization problem
    bounds = [(0, 1) for _ in range(10)]  # radii must be between 0 and 1
    cons = []

    # Add constraints for circles to be inside unit square
    cons.append({'type': 'ineq', 'fun': lambda x: constraint_circle_inside(x, centers)})

    # Add constraints for no overlap between circles
    cons.append({'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x, centers)})

    # Define optimization problem
    def objective_wrapper(x):
        return objective(x, centers)

    result = minimize(objective_wrapper, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8, maxiter=1000)

    if result.success:
        sum_radii = -result.fun
        return (centers, result.x, sum_radii)
    else:
        warnings.warn("Optimization did not converge.")
        return (centers, np.zeros(10), 0.0)
