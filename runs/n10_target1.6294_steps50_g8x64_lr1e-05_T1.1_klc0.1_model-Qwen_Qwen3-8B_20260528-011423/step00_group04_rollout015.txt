import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (randomly placed inside the unit square)
    centers = np.random.rand(10, 2)
    # Initial guess for radii (all 0.1)
    radii = np.full(10, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        # r <= x - r <= 1 - r and r <= y - r <= 1 - r
        # This becomes x - r >= r and 1 - x - r >= r => x >= 2r and x <= 1 - 2r
        # So for each center (x, y), 2r <= x <= 1 - 2r and 2r <= y <= 1 - 2r
        # For each circle, we check this constraint
        r = radii
        x = centers[:, 0]
        y = centers[:, 1]
        constraints1 = np.zeros(10)
        for i in range(10):
            constraints1[i] = (x[i] - r[i]) - r[i]  # x - 2r >= 0
            constraints1[i] = np.minimum(constraints1[i], (1 - x[i] - r[i]) - r[i])  # 1 - x - 2r >= 0
            constraints1[i] = np.minimum(constraints1[i], (y[i] - r[i]) - r[i])  # y - 2r >= 0
            constraints1[i] = np.minimum(constraints1[i], (1 - y[i] - r[i]) - r[i])  # 1 - y - 2r >= 0
        constraints1 = np.maximum(constraints1, 0)

        # Constraint 2: No two circles overlap
        # Distance between centers >= sum of their radii
        dist = np.zeros(10 * 10)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i * 10 + j] = np.sqrt(dx ** 2 + dy ** 2) - (radii[i] + radii[j])
        constraints2 = np.zeros(10 * 10)
        for i in range(10):
            for j in range(i + 1, 10):
                constraints2[i * 10 + j] = dist[i * 10 + j]  # distance >= sum of radii
        constraints2 = np.maximum(constraints2, 0)

        constraints = np.concatenate((constraints1, constraints2))
        return constraints

    # Bounds for radii (non-negative)
    bounds = [(0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(objective,
                      x0=radii,
                      args=(centers,),
                      method='SLSQP',
                      bounds=bounds,
                      constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r, centers)}])

    # Extract results
    radii = result.x
    centers = np.random.rand(10, 2)  # Replaced with better centers if needed
    # Note: The above centers are placeholder; to improve, one would need to update centers based on the optimization
    # This is a simplified version, and for real results, a more advanced optimization approach with center updates is needed.
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
