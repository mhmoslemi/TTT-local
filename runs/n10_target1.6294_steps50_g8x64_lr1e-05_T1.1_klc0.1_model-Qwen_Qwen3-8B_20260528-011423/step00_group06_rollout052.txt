import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - 0.1:
                return np.inf
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        return 0

    def bounds(radii):
        return [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        options={'maxiter': 1000, 'disp': False}
    )

    # Extract results
    radii_opt = result.x
    centers_opt = centers
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
