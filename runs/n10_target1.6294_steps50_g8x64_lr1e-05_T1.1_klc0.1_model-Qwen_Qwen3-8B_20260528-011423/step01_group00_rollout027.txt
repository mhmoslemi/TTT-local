import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: randomly placed circles inside the square
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10)
    
    # Objective function: minimize the negative sum of radii
    def objective(x):
        # x is flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        x_centers = x[:20].reshape(10, 2)
        x_radii = x[20:]
        sum_radii = np.sum(x_radii)
        return -sum_radii  # Minimize negative sum to maximize sum
    
    # Constraints: each circle must be inside the unit square
    def square_constraint(x):
        x_centers = x[:20].reshape(10, 2)
        x_radii = x[20:]
        left_bounds = np.array([x_centers[i, 0] - x_radii[i] for i in range(10)])
        right_bounds = np.array([1 - x_centers[i, 0] - x_radii[i] for i in range(10)])
        bottom_bounds = np.array([x_centers[i, 1] - x_radii[i] for i in range(10)])
        top_bounds = np.array([1 - x_centers[i, 1] - x_radii[i] for i in range(10)])
        return np.concatenate([left_bounds, right_bounds, bottom_bounds, top_bounds])
    
    # Constraints: distance between centers >= sum of radii
    def no_overlap_constraint(x):
        x_centers = x[:20].reshape(10, 2)
        x_radii = x[20:]
        distances = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x_centers[i, 0] - x_centers[j, 0]
                dy = x_centers[i, 1] - x_centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                distances.append(dist - x_radii[i] - x_radii[j])
        return np.array(distances)
    
    # Initial guess vector
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Bounds: each coordinate is between 0 and 1, radii >= 0
    bounds = [(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)]
    
    # Constraints
    cons = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': no_overlap_constraint}
    ]
    
    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)
    
    # Extract result
    x_opt = result.x
    centers_opt = x_opt[:20].reshape(10, 2)
    radii_opt = x_opt[20:]
    sum_radii_opt = -result.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
