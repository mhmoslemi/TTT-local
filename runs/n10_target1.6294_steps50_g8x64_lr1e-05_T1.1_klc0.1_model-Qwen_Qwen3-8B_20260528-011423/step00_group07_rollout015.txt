import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: 10 circles placed at random positions with small radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(radii, centers):
        # Ensure circles are inside the unit square
        r = radii.reshape(-1, 1)
        x = centers[:, 0]
        y = centers[:, 1]
        return np.min([x - r, 1 - x - r, y - r, 1 - y - r])

    def constraint_no_overlap(centers, radii):
        # Ensure no overlap between circles
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        r_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dists - r_sum

    # Optimization setup
    bounds = [(0, 1) for _ in range(10)]  # bounds for radii
    cons = []

    # Add constraint for circles inside the square
    cons.append({'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)})

    # Add constraint for no overlap between circles
    cons.append({'type': 'ineq', 'fun': lambda r: constraint_no_overlap(centers, r)})

    # Optimization
    result = minimize(objective, radii, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-8)

    # Extract results
    centers = np.random.rand(10, 2)  # Random centers (fallback in case of issues)
    radii = result.x
    sum_radii = -result.fun

    # Ensure valid solution
    if sum_radii < 0.5:
        centers = np.random.rand(10, 2)
        radii = np.full(10, 0.05)
        sum_radii = 0.5

    return (centers, radii, sum_radii)
