import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly distributed)
    np.random.seed(42)
    initial_centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    initial_radii = np.full(10, 0.1)

    def objective(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        # Reshape to centers and radii
        flat_centers = x[:60].reshape((10, 2))
        flat_radii = x[60:].reshape(10)
        # Objective is to minimize negative sum of radii
        return -np.sum(flat_radii)

    def constraints(x):
        flat_centers = x[:60].reshape((10, 2))
        flat_radii = x[60:].reshape(10)
        # Constraint: all circles must be within the unit square
        for i in range(10):
            x_i, y_i, r_i = flat_centers[i, 0], flat_centers[i, 1], flat_radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint: no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = flat_centers[i, 0] - flat_centers[j, 0]
                dy = flat_centers[i, 1] - flat_centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < flat_radii[i] + flat_radii[j]:
                    return np.inf
        return 0

    bounds = [(0.1, 0.9) for _ in range(60)] + [(0.0, 0.5) for _ in range(10)]

    # Optimization
    result = minimize(objective, x0=np.concatenate((initial_centers.ravel(), initial_radii)), 
                      method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Extract the solution
    centers = result.x[:60].reshape((10, 2))
    radii = result.x[60:].reshape(10)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
