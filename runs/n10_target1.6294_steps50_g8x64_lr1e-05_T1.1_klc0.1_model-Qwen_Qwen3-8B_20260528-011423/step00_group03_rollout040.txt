import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess for circle centers (random within the square, with radii 0.1)
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.1)
    
    def objective_function(params):
        # Flatten parameters into centers and radii
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        # Ensure all radii are non-negative
        radii_flat = np.maximum(radii_flat, 0)
        # Check if all circles are within the unit square
        if np.any((centers_flat - radii_flat[:, np.newaxis]) < 0) or np.any((1 - centers_flat - radii_flat[:, np.newaxis]) < 0):
            return np.inf
        # Calculate the sum of radii
        sum_radii = np.sum(radii_flat)
        # Check for overlaps between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distance = np.sqrt(dx*dx + dy*dy)
                if distance < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return -sum_radii  # Minimize -sum_radii to maximize sum_radii
    
    # Optimization bounds: centers in [0, 1], radii >= 0
    bounds = [(0.0, 1.0) for _ in range(60)] + [(0.0, 1.0) for _ in range(10)]
    
    # Run the optimization
    result = minimize(objective_function, np.concatenate([centers.flatten(), radii]), bounds=bounds, method='L-BFGS-B', tol=1e-6)
    
    # Extract final centers, radii, and sum of radii
    optimized_centers = result.x[:60].reshape(10, 2)
    optimized_radii = result.x[60:]
    sum_radii = -result.fun
    
    return (optimized_centers, optimized_radii, sum_radii)
