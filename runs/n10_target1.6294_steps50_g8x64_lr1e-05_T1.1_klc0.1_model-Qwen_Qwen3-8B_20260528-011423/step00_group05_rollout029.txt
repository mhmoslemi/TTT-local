import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective_function(params):
        # params: flattened array of centers and radii
        centers_flat = params[:20].reshape(10, 2)
        radii_flat = params[20:]
        # Objective: maximize sum of radii (minimize negative sum)
        sum_radii = -np.sum(radii_flat)
        return sum_radii

    def constraints_function(params):
        centers_flat = params[:20].reshape(10, 2)
        radii_flat = params[20:]
        constraints = []

        # Constraint: each circle is fully inside the unit square
        for i in range(10):
            x, y = centers_flat[i]
            r = radii_flat[i]
            constraints.append((x - r >= 0) * 1e6)
            constraints.append((x + r <= 1) * 1e6)
            constraints.append((y - r >= 0) * 1e6)
            constraints.append((y + r <= 1) * 1e6)

        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                r_sum = radii_flat[i] + radii_flat[j]
                constraints.append((dist - r_sum >= 0) * 1e6)

        return np.array(constraints)

    # Bounds: center coordinates between 0 and 1, radii >= 0
    bounds = []
    for i in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y
        bounds.append((0.0, 1.0))  # radius

    # Initial parameters
    initial_params = np.concatenate([centers.flatten(), radii])

    # Optimization
    result = minimize(
        objective_function,
        initial_params,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints_function}
    )

    # Extract result
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii_result = -np.sum(radii_result)

    return (centers_result, radii_result, sum_radii_result)
