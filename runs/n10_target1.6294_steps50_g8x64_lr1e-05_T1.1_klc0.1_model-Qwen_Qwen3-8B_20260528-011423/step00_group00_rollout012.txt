import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equally spaced grid for centers and same radii
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 10) for y in np.linspace(0.1, 0.9, 10)])
    radii = np.full(10, 0.05)

    def objective_function(params):
        # params is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize the negative to maximize the sum

    def constraints(params):
        centers_flat = params[:60].reshape(10, 2)
        radii_flat = params[60:]
        constraints = []

        # Constraint: circles must be inside the unit square
        for i in range(10):
            x, y = centers_flat[i]
            r = radii_flat[i]
            constraints.append(r <= x - 0.0001)
            constraints.append(r <= 1.0 - x - 0.0001)
            constraints.append(r <= y - 0.0001)
            constraints.append(r <= 1.0 - y - 0.0001)

        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.sqrt(dx*dx + dy*dy)
                constraints.append(dist >= radii_flat[i] + radii_flat[j] - 0.0001)

        return constraints

    # Prepare initial parameters
    params_initial = np.concatenate([centers.flatten(), radii])

    # Set up optimization problem
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    constraints_list = constraints(params_initial)

    # Set up optimization
    result = minimize(
        objective_function,
        params_initial,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints_list,
        tol=1e-5
    )

    # Extract results
    centers_result = result.x[:60].reshape(10, 2)
    radii_result = result.x[60:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
