import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: centers at a grid and equal radii
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # We want to maximize sum of radii

    def constraints(radii):
        # Ensure circles are fully inside the unit square
        for i in range(10):
            if radii[i] > 1 - 2 * np.sqrt((centers[i, 0] - 0.5)**2 + (centers[i, 1] - 0.5)**2):
                return False
        # Ensure circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization with constraints
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1 - 2 * np.sqrt((centers[i, 0] - 0.5)**2 + (centers[i, 1] - 0.5)**2)) for i in range(10)],
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r)}],
        options={'ftol': 1e-9, 'disp': False}
    )

    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
