import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 10 circles with small radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square
    def square_constraint(radii, centers):
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0] - radii)

    # Constraints: circles must not overlap
    def no_overlap_constraint(radii, centers):
        n = len(radii)
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization bounds
    bounds = [(0, 0.5) for _ in range(10)]

    # Set up optimization problem
    res = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: np.min(square_constraint(r, centers))},
            {'type': 'ineq', 'fun': lambda r: np.min(no_overlap_constraint(r, centers))}
        ]
    )

    # Extract results
    best_radii = res.x
    best_centers = centers.copy()
    best_sum = np.sum(best_radii)

    return (best_centers, best_radii, best_sum)
