import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.ones(10) * 0.15

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraint: each circle is inside the unit square
    def square_inside(radii, centers):
        return np.all((centers - radii >= 0) & (1 - centers - radii >= 0))

    # Constraint: circles do not overlap
    def no_overlap(radii, centers):
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx*dx + dy*dy)
                idx += 1
        return np.all(dists >= radii[:n-1] + radii[1:])

    # Combine all constraints
    def constraints(radii, centers):
        return [
            {'type': 'ineq', 'fun': lambda x: square_inside(x, centers)},
            {'type': 'ineq', 'fun': lambda x: no_overlap(x, centers)}
        ]

    # Flatten the variables for optimization
    def flatten(centers, radii):
        return np.concatenate([centers.flatten(), radii])

    def unflatten(x):
        centers = x[:10*2].reshape(10, 2)
        radii = x[10*2:]
        return centers, radii

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=lambda x: -np.sum(x[10*2:]),
            x0=flatten(initial_centers, initial_radii),
            constraints=[{'type': 'ineq', 'fun': lambda x: square_inside(x[10*2:], x[:10*2])},
                         {'type': 'ineq', 'fun': lambda x: no_overlap(x[10*2:], x[:10*2])}],
            method='SLSQP',
            bounds=[(0, 1) for _ in range(10*2)] + [(0, 1) for _ in range(10)]
        )

    centers, radii = unflatten(result.x)
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
