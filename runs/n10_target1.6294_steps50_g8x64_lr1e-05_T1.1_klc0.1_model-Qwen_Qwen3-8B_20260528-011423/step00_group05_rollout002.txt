import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    # Define objective function to minimize (negative of sum of radii)
    def objective(x):
        # Reshape input to centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return -np.sum(radii)

    # Define constraints
    def constraint_circle_inside(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([r - (x - 1 + r) for r, x in zip(radii, centers[:, 0])])  # x - r >= 0
        # The above expression simplifies to checking if all circles are within boundaries

    def constraint_circle_inside_2(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([r - (1 - x - r) for r, x in zip(radii, centers[:, 0])])  # 1 - x - r >= 0

    def constraint_circle_inside_3(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([r - (y - r) for r, y in zip(radii, centers[:, 1])])  # y - r >= 0

    def constraint_circle_inside_4(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        return np.array([r - (1 - y - r) for r, y in zip(radii, centers[:, 1])])  # 1 - y - r >= 0

    def constraint_no_overlap(x):
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        dist = np.zeros((10, 10))
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i, j] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
        return dist.flatten()

    # Optimization bounds: centers in [0,1], radii in [0, 0.5]
    bounds = [(0, 1) for _ in range(60)] + [(0, 0.5) for _ in range(10)]

    # Define constraints for the optimizer
    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_circle_inside_2},
        {'type': 'ineq', 'fun': constraint_circle_inside_3},
        {'type': 'ineq', 'fun': constraint_circle_inside_4},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimize
    result = minimize(objective, np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints=cons)

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
