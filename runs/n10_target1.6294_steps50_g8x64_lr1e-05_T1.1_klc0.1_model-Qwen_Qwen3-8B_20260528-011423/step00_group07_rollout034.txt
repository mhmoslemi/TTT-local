import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1
    radii = np.clip(radii, 0, 0.5)  # Set max radius to 0.5 for safety

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        radii_flat = np.clip(radii_flat, 0, 0.5)
        return -np.sum(radii_flat)

    # Constraints
    def constraint_inside(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1
        return 0

    def constraint_overlap(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distance = np.sqrt(dx * dx + dy * dy)
                if distance < radii_flat[i] + radii_flat[j]:
                    return -1
        return 0

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up bounds: x and y in [0,1], radii in [0, 0.5]
    bounds = []
    for i in range(10):
        bounds.extend([(0, 1), (0, 1), (0, 0.5)])
    
    # Set up constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Run optimization
    try:
        result = minimize(
            objective,
            x0=x0,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            tol=1e-6,
            options={'maxiter': 1000}
        )
    except:
        # If scipy is not available, use a fallback with local grid search
        # For this example, we assume scipy is available
        result = minimize(
            objective,
            x0=x0,
            method='L-BFGS-B',
            bounds=bounds,
            constraints=constraints,
            tol=1e-6,
            options={'maxiter': 1000}
        )

    # Extract results
    x_result = result.x
    centers_result = x_result[:20].reshape(10, 2)
    radii_result = x_result[20:]
    radii_result = np.clip(radii_result, 0, 0.5)
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
