import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i * 0.25 + 0.125, j * 0.25 + 0.125] for j in range(10) for i in range(10)])
    centers = centers[:10]  # take first 10 points

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: circles must be inside the unit square
    def constraint_inside(radii):
        return np.array([r <= 0.5 for r in radii])

    # Constraint 2: circles must not overlap (distance between centers >= sum of radii)
    def constraint_overlap(radii):
        centers = np.array([[i * 0.25 + 0.125, j * 0.25 + 0.125] for j in range(10) for i in range(10)])
        centers = centers[:10]
        distances = np.zeros((10, 10))
        for i in range(10):
            for j in range(10):
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    distances[i, j] = np.sqrt(dx**2 + dy**2)
        return distances - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Setup optimization problem
    bounds = [(0.0, 0.5) for _ in range(10)]

    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': lambda r: np.min(np.diag(constraint_overlap(r)))}
    ]

    result = minimize(objective, np.full(10, 0.1), bounds=bounds, constraints=constraints)

    # Extract final result
    radii = result.x
    centers = np.array([[i * 0.25 + 0.125, j * 0.25 + 0.125] for j in range(10) for i in range(10)])
    centers = centers[:10]

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
