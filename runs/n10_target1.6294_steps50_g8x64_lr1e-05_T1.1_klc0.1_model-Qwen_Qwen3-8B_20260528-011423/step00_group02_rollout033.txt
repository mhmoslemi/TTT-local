import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place 10 circles in a grid inside the unit square
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints:
    # 1. Radii must be non-negative
    cons = ({'type': 'ineq', 'fun': lambda r: r})

    # 2. Circles must be fully inside the square
    def inside_box(radii):
        for i in range(10):
            if (centers[i, 0] - radii[i] < 0) or (centers[i, 0] + radii[i] > 1) or \
               (centers[i, 1] - radii[i] < 0) or (centers[i, 1] + radii[i] > 1):
                return -1
        return 0

    cons.append({'type': 'ineq', 'fun': inside_box})

    # 3. Circles must not overlap
    def no_overlap(radii):
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0

    cons.append({'type': 'ineq', 'fun': no_overlap})

    # Bounds: radii >= 0
    bounds = [(0, 1) for _ in range(10)]

    # Optimization: minimize the negative of the sum of radii
    result = minimize(objective, np.ones(10), method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    centers_result = centers.copy()
    radii_result = result.x
    sum_radii_result = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii_result)
