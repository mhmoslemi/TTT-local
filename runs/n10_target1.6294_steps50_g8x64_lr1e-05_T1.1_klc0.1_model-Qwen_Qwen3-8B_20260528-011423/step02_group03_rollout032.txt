import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles at a grid with spacing of 0.2
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(c):
        # Objective function to minimize (negative sum of radii)
        return -np.sum(c[20:])

    def constraints(c):
        centers = c[:20].reshape(10, 2)
        radii = c[20:]
        # Constraint 1: Circles must be inside the unit square
        inside = np.all((centers[:, 0] >= radii) & (centers[:, 0] <= 1 - radii) &
                        (centers[:, 1] >= radii) & (centers[:, 1] <= 1 - radii))
        if not inside:
            return np.inf

        # Constraint 2: Circles must not overlap
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf

        return -np.sum(radii)

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(constraints, x0=np.concatenate((centers.flatten(), radii)),
                          method='SLSQP',
                          bounds=[(0, 1) for _ in range(20)] + [(0, 1) for _ in range(10)])
    
    centers_opt = result.x[:20].reshape(10, 2)
    radii_opt = result.x[20:]
    sum_radii = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii)
