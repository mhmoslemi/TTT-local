import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    def initial_guess():
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        radii = np.full(10, 0.05)
        return centers, radii

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii, centers):
        constraints = []
        # Constraint: every circle is within the square
        for i in range(10):
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (1 - np.max(centers[i, :]))})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (1 - np.min(centers[i, :]))})
        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                def dist_constr(radii, i=i, j=j):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.hypot(dx, dy)
                    return dist - (radii[i] + radii[j])
                constraints.append({'type': 'ineq', 'fun': dist_constr})
        return constraints

    def optimize():
        centers, radii = initial_guess()
        bounds = [(0.0, 1.0 - 1e-5) for _ in range(20)]  # Ensure circles are fully inside the square
        result = minimize(objective, radii, args=(centers,), bounds=bounds, constraints=constraints(radii, centers),
                          method='SLSQP', tol=1e-6)
        return result.x, centers, -result.fun

    radii, centers, sum_radii = optimize()
    return (centers, radii, sum_radii)
