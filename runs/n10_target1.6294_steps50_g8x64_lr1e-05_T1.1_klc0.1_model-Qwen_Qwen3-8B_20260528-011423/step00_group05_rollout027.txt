import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25

    # Initial radii: small values
    radii = np.full(10, 0.1)

    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraints: circles must be fully inside the square
    def constraint_inside(radii, centers):
        return np.sum((centers[:, 0] - radii) >= 0) + np.sum((1 - centers[:, 0] - radii) >= 0) + \
               np.sum((centers[:, 1] - radii) >= 0) + np.sum((1 - centers[:, 1] - radii) >= 0)

    # Constraint: circles do not overlap
    def constraint_no_overlap(radii, centers):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Set up the optimization problem
    bounds = [(0.0, 0.5) for _ in range(10)]
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
    ]

    # Optimization
    result = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
