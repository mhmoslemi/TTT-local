import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 10 circles arranged in a grid
    centers = np.random.rand(10, 2) * 0.8 + 0.1  # Random centers within [0.1, 0.9]
    radii = np.full(10, 0.1)  # Initial radius guess

    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraints: circles must be within the unit square and non-overlapping
    def constraints(radii, centers):
        # Constraint 1: Circle within the square
        def in_square(r, c):
            x, y = c
            return (r <= x) and (x <= 1 - r) and (r <= y) and (y <= 1 - r)
        square_constraints = [
            {'type': 'ineq', 'fun': lambda r: in_square(r, centers[i]) for i in range(10)}
        ]

        # Constraint 2: Circles do not overlap
        def no_overlap(i, j):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            return dist >= radii[i] + radii[j]
        overlap_constraints = []
        for i in range(10):
            for j in range(i + 1, 10):
                overlap_constraints.append(
                    {'type': 'ineq', 'fun': lambda r, i=i, j=j: no_overlap(i, j)}
                )

        return square_constraints + overlap_constraints

    # Optimization setup: minimize the negative objective
    result = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        bounds=[(0, 0.5) for _ in range(10)],
        constraints=constraints(radii, centers),
        method='SLSQP',
        tol=1e-6
    )

    # Extract results
    radii_result = result.x
    centers_result = centers
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
