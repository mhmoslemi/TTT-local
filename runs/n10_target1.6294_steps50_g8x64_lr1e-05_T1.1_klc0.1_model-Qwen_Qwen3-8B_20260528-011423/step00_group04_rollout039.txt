import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: centers at (0.1, 0.1), (0.1, 0.9), (0.9, 0.1), (0.9, 0.9),
    # and 6 more spread out in the center
    initial_centers = np.array([
        [0.1, 0.1], [0.1, 0.9], [0.9, 0.1], [0.9, 0.9],
        [0.5, 0.2], [0.5, 0.8], [0.2, 0.5], [0.8, 0.5],
        [0.5, 0.5]
    ])
    initial_radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        return 0

    # Optimization setup
    bounds = [(0, 1) for _ in range(10)]
    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract result
    radii = result.x
    centers = initial_centers.copy()
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
