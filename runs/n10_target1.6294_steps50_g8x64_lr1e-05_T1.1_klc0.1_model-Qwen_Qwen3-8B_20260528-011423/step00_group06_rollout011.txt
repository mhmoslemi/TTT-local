import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)
    
    # Define the objective function: minimize the negative sum of radii
    def objective(x):
        return -np.sum(x[10:])  # minimize negative sum => maximize sum
    
    # Define constraints
    def constraint_center_inside(x):
        # Constraint: each circle is inside the unit square
        # x[0:10] = centers, x[10:] = radii
        centers = x[0:10].reshape(10, 2)
        radii = x[10:]
        return np.array([centers[i, 0] - radii[i] >= 0 for i in range(10)] +
                       [1 - centers[i, 0] - radii[i] >= 0 for i in range(10)] +
                       [centers[i, 1] - radii[i] >= 0 for i in range(10)] +
                       [1 - centers[i, 1] - radii[i] >= 0 for i in range(10)])
    
    def constraint_no_overlap(x):
        # Constraint: no two circles overlap
        centers = x[0:10].reshape(10, 2)
        radii = x[10:]
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return np.array([dist[i, j] - radii[i] - radii[j] >= 0 for i in range(10) for j in range(i+1)])
    
    # Define bounds for centers and radii
    bounds = []
    for i in range(10):
        bounds.append((0, 1))  # x-coordinate of center
        bounds.append((0, 1))  # y-coordinate of center
    for i in range(10):
        bounds.append((0, 0.5))  # radius (to ensure circle stays within square)
    
    # Set up optimization problem
    x0 = np.concatenate([centers.flatten(), radii])
    cons = [{'type': 'ineq', 'fun': constraint_center_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}]
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tolerance=1e-6)
    
    # Extract solution
    centers_opt = result.x[0:10].reshape(10, 2)
    radii_opt = result.x[10:]
    sum_radii_opt = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii_opt)
