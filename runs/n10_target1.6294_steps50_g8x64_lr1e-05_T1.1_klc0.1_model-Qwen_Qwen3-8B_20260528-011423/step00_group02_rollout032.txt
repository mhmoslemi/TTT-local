import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.full(10, 0.1)

    def objective_function(x):
        # x is a flat array containing centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize sum

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        constraints = []
        # Constraint: all circles are within the unit square
        for i in range(10):
            if radii_flat[i] > 0:
                x_i = centers_flat[i][0]
                y_i = centers_flat[i][1]
                constraints.append(1 - x_i - radii_flat[i])
                constraints.append(1 - y_i - radii_flat[i])
                constraints.append(x_i - radii_flat[i])
                constraints.append(y_i - radii_flat[i])
        # Constraint: no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.sqrt(dx**2 + dy**2)
                if radii_flat[i] + radii_flat[j] > 0:
                    constraints.append(dist - (radii_flat[i] + radii_flat[j]))
        return constraints

    # Convert into a format suitable for scipy.optimize
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # centers
        bounds.extend([(0.0, 0.5)])  # radii (limited to 0.5 to ensure fit in unit square)
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimize
    constraints_func = [
        {'type': 'ineq', 'fun': lambda x: constraints(x)}
    ]
    result = minimize(
        objective_function,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints_func,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract and format the result
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun  # Because we maximized using negative sum
    return (centers_opt, radii_opt, sum_radii)
