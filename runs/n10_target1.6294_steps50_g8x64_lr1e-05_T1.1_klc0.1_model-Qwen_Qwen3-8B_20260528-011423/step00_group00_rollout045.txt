import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(x):
        # Reshape x to centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Compute sum of radii
        sum_r = np.sum(radii)
        return -sum_r  # Minimize negative to maximize

    def constraints(x):
        # Reshape x to centers and radii
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Constraint 1: Each circle is inside the square
        constraints1 = []
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            constraints1.append(x_i - r_i)
            constraints1.append(1 - x_i - r_i)
            constraints1.append(y_i - r_i)
            constraints1.append(1 - y_i - r_i)
        # Constraint 2: Distance between centers >= sum of radii
        constraints2 = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                constraints2.append(dist - (radii[i] + radii[j]))
        return constraints1 + constraints2

    # Initial guess: place circles at a grid
    initial_centers = np.array([
        [0.1, 0.1], [0.1, 0.3], [0.1, 0.5], [0.1, 0.7], [0.1, 0.9],
        [0.3, 0.1], [0.3, 0.3], [0.3, 0.5], [0.3, 0.7], [0.3, 0.9]
    ])
    initial_radii = np.full(10, 0.05)
    x0 = initial_centers.flatten()  # Flatten centers
    x0 = np.concatenate((x0, initial_radii))  # Add radii

    # Set bounds: x and y between 0 and 1, radii ≥ 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])
    for _ in range(10):
        bounds.append((0, 1))

    # Solve the optimization problem
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    # Extract result
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_r = np.sum(radii)

    return (centers, radii, sum_r)
