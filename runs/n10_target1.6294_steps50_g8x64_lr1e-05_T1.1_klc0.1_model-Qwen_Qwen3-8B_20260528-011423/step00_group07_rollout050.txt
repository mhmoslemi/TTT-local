```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid of 10 points
    centers = np.array([[i/3, j/3] for i in range(3) for j in range(3)])
    centers = centers[:10]
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii, centers):
        return np.sum((radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) &
                      (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1]))

    def constraint_no_overlap(radii, centers):
        distances = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return np.sum(distances >= radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraints(radii):
        return [constraint_center_inside(radii, centers),
                constraint_no_overlap(radii, centers)]

    # Optimize
    result = minimize(objective, radii, constraints=constraints, method='SLSQP', bounds=[(0, 1.0) for _ in range(10)])
    radii_opt = result.x
    centers_opt = centers
    sum_radii = -result.fun
    return (centers_opt, radii_opt, sum_radii)
```
