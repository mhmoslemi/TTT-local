import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.random.rand(10, 2)
    # All radii start as 0.1
    radii = np.full(10, 0.1)
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        # Constraints: circles fully inside the unit square
        for i in range(10):
            if (centers_flat[i, 0] - radii_flat[i] < 0 or 
                centers_flat[i, 0] + radii_flat[i] > 1 or
                centers_flat[i, 1] - radii_flat[i] < 0 or
                centers_flat[i, 1] + radii_flat[i] > 1):
                return float('inf')
        # Constraint: no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return float('inf')
        return -sum_radii  # We minimize the negative sum to maximize the sum

    # Bounds: each center (x, y) is in [0,1], each radius is in [0, 0.5]
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radii])
    # Optimization
    result = minimize(objective, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-6)
    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun
    return (centers_opt, radii_opt, sum_radii_opt)
