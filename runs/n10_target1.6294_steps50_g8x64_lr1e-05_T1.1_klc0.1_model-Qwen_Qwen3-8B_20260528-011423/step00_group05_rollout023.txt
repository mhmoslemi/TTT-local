import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    # Objective function: minimize the negative sum of radii
    def objective(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints: 1. Circles are inside the unit square
    def constraint_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        inside = np.all((centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1) & \
                 np.all((1 - centers_flat - radii_flat[:, np.newaxis]) >= 0, axis=1)
        return np.sum(inside)

    # Constraints: 2. Circles do not overlap
    def constraint_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distance = np.linalg.norm(centers_flat[:, np.newaxis] - centers_flat[np.newaxis, :], axis=2)
        overlap = np.any(distance < radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :], axis=None)
        return not overlap

    # Optimization bounds: centers in [0,1], radii >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x and y coordinates
        bounds.extend([(0, 0.5)])       # radius (limited to 0.5 to stay inside square)

    # Setup optimization
    x0 = np.hstack((centers.flatten(), radii))

    # Use SLSQP method
    result = minimize(objective, x0, method='SLSQP', bounds=bounds,
                      constraints=[{'type': 'ineq', 'fun': constraint_inside},
                                   {'type': 'ineq', 'fun': constraint_overlap}])

    # Extract results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
