import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: centers at grid points with small radii
    centers = np.random.rand(10, 2) * 0.8 + 0.1  # Random initial positions inside the square
    radii = np.full(10, 0.1)  # Initial radii

    # Objective function to minimize negative sum of radii
    def objective(x):
        # x is flattened array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative of sum

    # Constraints
    constraints = []

    # Constraint: each circle is within the unit square
    def square_boundary(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return np.concatenate([
            centers_flat[:, 0] - radii_flat,
            1 - centers_flat[:, 0] - radii_flat,
            centers_flat[:, 1] - radii_flat,
            1 - centers_flat[:, 1] - radii_flat
        ])

    constraints.append({'type': 'ineq', 'fun': square_boundary})

    # Constraint: no overlap between circles
    def circle_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        distances = cdist(centers_flat, centers_flat)
        radii_sum = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        return distances - radii_sum

    constraints.append({'type': 'ineq', 'fun': circle_overlap})

    # Bounds for each variable: x in [0,1], y in [0,1], r >= 0
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])

    # Optimization with L-BFGS-B
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii]),
        method='L-BFGS-B',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8
    )

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii_opt = -result.fun  # Negative because we minimized negative sum

    return (centers_opt, radii_opt, sum_radii_opt)
