import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced centers with equal radii
    n = 10
    centers = np.random.rand(n, 2)
    radii = np.full(n, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(x):
        # x is a flattened array of centers and radii
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        # Constraint 1: Circles are fully inside the unit square
        for i in range(n):
            x_i, y_i, r_i = centers[i][0], centers[i][1], radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: Circles do not overlap
        for i in range(n):
            for j in range(i+1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Flatten centers and radii for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up optimization
    res = minimize(objective, x0, method='SLSQP', bounds=[(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)])
    if not res.success:
        warnings.warn("Optimization failed to converge")
        return (np.random.rand(n, 2), np.full(n, 0.1), 0.1*n)

    # Extract result
    centers_opt = res.x[:2*n].reshape(n, 2)
    radii_opt = res.x[2*n:]
    sum_radii = -res.fun
    return (centers_opt, radii_opt, sum_radii)
