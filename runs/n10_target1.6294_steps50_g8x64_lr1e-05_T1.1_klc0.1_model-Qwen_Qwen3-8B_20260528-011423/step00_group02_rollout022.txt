import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.empty((10, 2))
        for i in range(10):
            x, y = np.random.rand(2)
            centers[i, 0] = x * (1 - radii[i]) + radii[i]
            centers[i, 1] = y * (1 - radii[i]) + radii[i]
        # Check if all circles are inside the square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return -np.inf
        # Check if circles are non-overlapping
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return np.sum(radii)

    # Optimize with bounds and constraints
    res = minimize(lambda r: -constraints(r), radii, method='SLSQP', bounds=[(0, 0.5)] * 10)
    optimal_radii = res.x
    optimal_centers = np.array([[np.random.rand() * (1 - r) + r for _ in range(2)] for r in optimal_radii])
    # Ensure circles are not overlapping and within bounds
    for i in range(10):
        while not (optimal_radii[i] <= optimal_centers[i, 0] <= 1 - optimal_radii[i] and
                  optimal_radii[i] <= optimal_centers[i, 1] <= 1 - optimal_radii[i]):
            optimal_centers[i, 0] = np.random.rand() * (1 - optimal_radii[i]) + optimal_radii[i]
            optimal_centers[i, 1] = np.random.rand() * (1 - optimal_radii[i]) + optimal_radii[i]
        for j in range(i + 1, 10):
            dx = optimal_centers[i, 0] - optimal_centers[j, 0]
            dy = optimal_centers[i, 1] - optimal_centers[j, 1]
            dist = math.hypot(dx, dy)
            if dist < optimal_radii[i] + optimal_radii[j]:
                optimal_centers[j, 0] = np.random.rand() * (1 - optimal_radii[j]) + optimal_radii[j]
                optimal_centers[j, 1] = np.random.rand() * (1 - optimal_radii[j]) + optimal_radii[j]
    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
