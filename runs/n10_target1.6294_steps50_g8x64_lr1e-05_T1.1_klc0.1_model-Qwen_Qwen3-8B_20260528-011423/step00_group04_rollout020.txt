import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: spread the centers evenly in the square
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        # Ensure no circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        constraints={
            'type': 'ineq',
            'fun': constraints
        },
        bounds=[(0.0, 0.5) for _ in range(10)]
    )

    # After optimization, we refine the solution slightly
    # by adjusting centers to avoid overlaps and stay within bounds
    def refine_centers(radii):
        centers = np.random.rand(10, 2)  # Start with new centers
        for i in range(10):
            # Keep radii fixed, adjust centers to avoid overlap and stay inside square
            for j in range(10):
                if i == j:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < 2 * radii[i]:
                    # Push centers apart
                    direction = np.array([dx, dy]) / dist
                    overlap = dist - (radii[i] + radii[j])
                    step = overlap / 2
                    centers[i] += direction * step
                    centers[j] -= direction * step
        # Ensure all circles are within the unit square
        for i in range(10):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                centers[i, 0] = np.clip(centers[i, 0], radii[i], 1 - radii[i])
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                centers[i, 1] = np.clip(centers[i, 1], radii[i], 1 - radii[i])
        return centers

    # Refine the centers
    centers = refine_centers(result.x)
    sum_radii = np.sum(result.x)
    return (centers, result.x, sum_radii)
