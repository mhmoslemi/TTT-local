import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    init_centers = np.random.rand(10, 2)
    init_radii = np.random.rand(10) * 0.1

    def objective(x):
        # Unpack variables
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Objective is to maximize the sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraint_bounds(x):
        # Unpack variables
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Check if all circles are inside the square
        for i in range(10):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        return 0

    def constraint_overlap(x):
        # Unpack variables
        centers = x[:60].reshape(10, 2)
        radii = x[60:]
        # Check if any circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Set up the optimization problem
    x0 = np.concatenate([init_centers.ravel(), init_radii])
    bounds = []
    for _ in range(10):
        bounds.append((0.0, 1.0))  # x coordinate bounds
        bounds.append((0.0, 1.0))  # y coordinate bounds
        bounds.append((0.0, 0.5))  # radius bounds (maximum radius is 0.5)
    cons = [
        {'type': 'ineq', 'fun': constraint_bounds},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:]
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
