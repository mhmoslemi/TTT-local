import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(center):
        # Constraint 1: All circles are within the unit square
        # center[i, 0] - radii[i] >= 0, center[i, 0] + radii[i] <= 1
        # center[i, 1] - radii[i] >= 0, center[i, 1] + radii[i] <= 1
        # Constraint 2: No two circles overlap (distance >= sum of radii)
        # distance between center[i] and center[j] >= radii[i] + radii[j]
        # We use a vectorized approach for all pairs
        n = len(center)
        square_constraints = (
            (center[:, 0] - radii) >= 0,
            (center[:, 1] - radii) >= 0,
            (1 - center[:, 0] - radii) >= 0,
            (1 - center[:, 1] - radii) >= 0
        )
        dists = np.sqrt(np.sum((center[:, np.newaxis] - center[np.newaxis, :])**2, axis=2))
        overlap_constraints = dists >= radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.concatenate((square_constraints, overlap_constraints.flatten()))

    def objective_and_constraints(x):
        # x is a flattened array: [x0, y0, r0, x1, y1, r1, ..., x9, y9, r9]
        centers = x[:2*10].reshape(10, 2)
        radii = x[2*10:]
        return objective(radii), constraints(centers)

    # Initial guess
    initial_guess = np.random.rand(2*10 + 10)  # 2*10 centers and 10 radii
    initial_guess[:2*10] = np.random.rand(10, 2)
    initial_guess[2*10:] = np.full(10, 0.05)

    # Optimize
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            lambda x: objective_and_constraints(x)[0],
            initial_guess,
            method='L-BFGS-B',
            bounds=[(0, 1) for _ in range(2*10)] + [(0, 1) for _ in range(10)],
            constraints=[{'type': 'ineq', 'fun': lambda x: objective_and_constraints(x)[1]}]
        )

    centers = result.x[:2*10].reshape(10, 2)
    radii = result.x[2*10:]
    sum_radii = -objective(radii)
    return (centers, radii, sum_radii)
