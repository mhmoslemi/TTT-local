import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    # Place 10 circles in a grid-like configuration
    centers = np.array([
        [0.1, 0.1], [0.3, 0.1], [0.5, 0.1], [0.7, 0.1], [0.9, 0.1],
        [0.1, 0.3], [0.3, 0.3], [0.5, 0.3], [0.7, 0.3], [0.9, 0.3]
    ])
    radii = np.full(10, 0.05)

    def objective(x):
        # Reshape x to centers and radii
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        return -np.sum(radii_flat)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return 1.0
        return 0.0

    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return 1.0
        return 0.0

    # Define bounds for x, y coordinates and radii
    bounds = []
    for i in range(10):
        bounds.append((0.0, 1.0))  # x_i
        bounds.append((0.0, 1.0))  # y_i
        bounds.append((0.0, 0.5))  # r_i

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_circle_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}
        ],
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
