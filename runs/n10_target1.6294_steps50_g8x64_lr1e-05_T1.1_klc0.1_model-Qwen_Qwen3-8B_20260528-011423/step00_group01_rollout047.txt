import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        # Objective is to minimize the negative sum of radii
        return -np.sum(radii_flat)

    def constraints(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        constraints = []
        # Constraint 1: Each circle is inside the unit square
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            constraints.append(x_i - r_i >= 0)
            constraints.append(1 - x_i - r_i >= 0)
            constraints.append(y_i - r_i >= 0)
            constraints.append(1 - y_i - r_i >= 0)
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist_sq = dx**2 + dy**2
                dist = np.sqrt(dist_sq)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                constraints.append(dist - r_i - r_j >= 0)
        return constraints

    # Flatten initial guess
    x0 = np.concatenate([centers.ravel(), radii.ravel()])

    # Define bounds for x, y, and r
    bounds = []
    for i in range(10):
        bounds.extend([(0, 1), (0, 1), (0, 0.5)])  # x, y, r for each circle

    # Define constraints in the form required by scipy.optimize.minimize
    cons = []
    for c in constraints(x0):
        cons.append({'type': 'ineq', 'fun': lambda x, c=c: c})

    # Use the SLSQP method for optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    # Extract centers, radii, and sum of radii
    centers = result.x[:60].reshape(10, 2)
    radii = result.x[60:].reshape(10)
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
