import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid-like formation
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    # Objective function: maximize sum of radii (minimize -sum_radii)
    def objective(x):
        # x is a flat array: [x0, y0, r0, x1, y1, r1, ...]
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraint 1: circles must be inside the unit square
    def inside_box(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        return np.all((centers_flat[:, 0] - radii_flat >= 0) & (centers_flat[:, 0] + radii_flat <= 1) &
                      (centers_flat[:, 1] - radii_flat >= 0) & (centers_flat[:, 1] + radii_flat <= 1))

    # Constraint 2: no overlapping between any two circles
    def no_overlap(x):
        centers_flat = x[:20].reshape(10, 2)
        radii_flat = x[20:]
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return False
        return True

    # Combine constraints
    def constraints(x):
        return [inside_box(x), no_overlap(x)]

    # Bounds for each variable
    bounds = []
    for _ in range(10):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])  # x, y, radius (max radius is 0.5 to fit in square)

    # Initial guess flattened
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract result
    centers_result = result.x[:20].reshape(10, 2)
    radii_result = result.x[20:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
