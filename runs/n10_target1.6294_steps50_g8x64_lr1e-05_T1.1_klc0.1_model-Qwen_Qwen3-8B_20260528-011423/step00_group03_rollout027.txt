import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    initial_radii = np.full(10, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing -sum
    
    def constraints(radii):
        # Constraint: All circles are inside the square
        for i in range(10):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - np.min(np.abs(initial_centers[i, :])) - np.max(np.abs(initial_centers[i, :] - 1)):
                return False
        # Constraint: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = initial_centers[i, 0] - initial_centers[j, 0]
                dy = initial_centers[i, 1] - initial_centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    bounds = [(0, 1) for _ in range(10)]
    
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': lambda x: constraints(x)}],
        bounds=bounds
    )
    
    centers = initial_centers
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
