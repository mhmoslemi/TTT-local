import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced points with small radii
    centers = np.random.rand(10, 2)
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_bounds(radii):
        return np.sum((radii > 0) * (radii > 1 - np.sqrt(2) * radii))

    def constraint_overlap(radii, centers):
        n = len(centers)
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - (radii[i] + radii[j])
                idx += 1
        return np.sum(dist < 0)

    # Optimization problem
    bounds = [(0, 1 - 1e-6) for _ in range(10)]  # Ensure circles are inside the square

    constraints = [
        {'type': 'ineq', 'fun': constraint_bounds},
        {'type': 'ineq', 'fun': lambda r: -constraint_overlap(r, centers)}
    ]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'disp': False, 'ftol': 1e-9}
    )

    # After optimization, update centers if needed (if the radii changed significantly)
    # For simplicity, we do not update centers in this basic implementation

    return (centers, result.x, -result.fun)
