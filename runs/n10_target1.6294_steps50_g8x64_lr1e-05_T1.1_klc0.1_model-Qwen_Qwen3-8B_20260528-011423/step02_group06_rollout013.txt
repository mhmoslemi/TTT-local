import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure every circle is inside the square
        for i, r in enumerate(radii):
            x, y = centers[i]
            if not (x - r >= 0 and x + r <= 1 and y - r >= 0 and y + r <= 1):
                return np.inf

        # Ensure no overlap between circles
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}])
    radii = result.x
    sum_radii = np.sum(radii)

    # Final centers based on initial guess
    centers = np.array([[0.1 + i * 0.2, 0.1 + j * 0.2] for i in range(5) for j in range(2)])

    return (centers, radii, sum_radii)
