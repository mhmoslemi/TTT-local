import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: distribute points evenly in the square
    centers = np.random.rand(10, 2)
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.random.rand(10, 2)
        # Ensure circles are inside the square
        for i in range(10):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf
        # Ensure no two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization settings
    bounds = [(0, 0.5) for _ in range(10)]

    # Run optimization
    result = minimize(objective, radii, method='L-BFGS-B', bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints})

    # Extract best result
    best_radii = result.x
    best_sum_radii = np.sum(best_radii)

    # Generate final centers using a better packing heuristic (random placement with constraints)
    centers = np.random.rand(10, 2)
    for i in range(10):
        x, y = centers[i]
        if x < best_radii[i] or x > 1 - best_radii[i] or y < best_radii[i] or y > 1 - best_radii[i]:
            centers[i] = [best_radii[i] + np.random.rand(), best_radii[i] + np.random.rand()]

    return centers, best_radii, best_sum_radii
