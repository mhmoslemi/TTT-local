import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for the centers (random positions inside the square)
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (10, 2))
    radii = np.random.uniform(0.05, 0.1, 10)

    # Objective function to minimize (negative of sum of radii)
    def objective(x):
        centers_flat = x.reshape(-1, 2)
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints: 1) circles within the square, 2) no overlaps
    def constraint_square(x):
        centers_flat = x.reshape(-1, 2)
        return np.concatenate([
            centers_flat[:, 0] - radii,
            1 - centers_flat[:, 0] - radii,
            centers_flat[:, 1] - radii,
            1 - centers_flat[:, 1] - radii
        ])

    def constraint_overlap(x):
        centers_flat = x.reshape(-1, 2)
        dists = np.sqrt(np.sum((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :])**2, axis=2))
        return np.concatenate([dists[i, j] - (radii[i] + radii[j]) for i in range(10) for j in range(i+1)])

    # Initial constraint values
    cons = [
        {'type': 'ineq', 'fun': constraint_square},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization bounds
    bounds = [(0.0, 1.0) for _ in range(20)]  # 10 centers * 2 coordinates

    # Run the optimization
    result = minimize(objective, centers.flatten(), method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    centers_opt = result.x.reshape(10, 2)
    sum_radii_opt = -result.fun

    return (centers_opt, radii, sum_radii_opt)
