import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles placed in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
        for i in range(10):
            for j in range(i+1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization with constraints and bounds
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 0.5) for _ in range(10)],
        constraints=[{'type': 'ineq', 'fun': lambda x: constraints(x)}]
    )

    # Final centers and radii
    centers_final = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii_final = result.x

    # Check if circles are within the unit square
    for i in range(10):
        if not (radii_final[i] <= centers_final[i, 0] <= 1 - radii_final[i] and
                radii_final[i] <= centers_final[i, 1] <= 1 - radii_final[i]):
            raise ValueError("Circle not fully inside the unit square")

    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
