import numpy as np
import scipy.optimize as opt
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Initial hexagonal grid layout
    num_circles = 26
    max_r = 0.2  # Initial guess for the radius
    initial_centers = np.zeros((num_circles, 2))
    initial_radii = np.full(num_circles, max_r)
    
    # Hexagonal grid: 5 rows with 5, 5, 5, 5, 6 circles
    # Row spacing: sqrt(3)/2 * diameter = sqrt(3)/2 * 2r = sqrt(3)*r
    row_spacing = math.sqrt(3) * max_r
    col_spacing = 2 * max_r
    
    for i in range(5):
        for j in range(5 if i % 2 == 0 else 6):
            x = j * col_spacing
            y = i * row_spacing
            # Ensure circles stay within [0,1]x[0,1]
            if x > 1 - max_r:
                x = 1 - max_r
            if y > 1 - max_r:
                y = 1 - max_r
            initial_centers[i * (5 if i % 2 == 0 else 6) + j] = [x, y]

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum (i.e., maximize sum)

    def constraint_center_bounds(x):
        # x is a flattened array of centers and radii
        centers = x[:2 * num_circles].reshape(num_circles, 2)

            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:

                dy = centers[i, 1] - centers[j, 1]

    result = opt.minimize(

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
