import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initialize a hexagonal grid configuration
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Start with a staggered hexagonal grid
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    spacing = 1.0 / (cols - 1)

    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j * spacing
            y = i * spacing * np.sqrt(3) / 2
            centers[idx, 0] = x
            centers[idx, 1] = y
            radii[idx] = 0.05

    # Define objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # negative because we use minimization

    # Define constraints
    def constraints(x):
        # Extract centers and radii
        centers = x[:2*num_circles].reshape(num_circles, 2)
        radii = x[2*num_circles:]
        constraints = []

        # Boundary constraints: x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1
        for i in range(num_circles):
            r = radii[i]
            x_i, y_i = centers[i]
            constraints.append((x_i - r >= -1e-8))  # x - r >= 0
            constraints.append((x_i + r <= 1 + 1e-8))  # x + r <= 1
            constraints.append((y_i - r >= -1e-8))  # y - r >= 0
            constraints.append((y_i + r <= 1 + 1e-8))  # y + r <= 1

        # Non-overlap constraints: ||center_i - center_j|| >= r_i + r_j
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i, r_j = radii[i], radii[j]
                constraints.append((dist - r_i - r_j >= -1e-8))

        return constraints

    # Optimization bounds
    # All centers must be in [0,1]×[0,1]
    bounds = []
    for _ in range(num_circles):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
    # Radii must be non-negative
    for _ in range(num_circles):
        bounds.append((0, 1))  # radius

    # Combine initial centers and radii into a single vector
    x0 = np.concatenate([centers.flatten(), radii])

    # Run the optimization
    result = opt.minimize(

        constraints={'type': 'ineq', 'fun': constraints},

    return centers_opt, radii_opt, np.sum(radii_opt)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
