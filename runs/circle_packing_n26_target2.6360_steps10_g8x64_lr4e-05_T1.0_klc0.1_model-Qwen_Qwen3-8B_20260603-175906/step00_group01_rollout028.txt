import numpy as np
import scipy.optimize as opt

def run_packing():
    # Define the number of circles
    n_circles = 26

    # Initialize grid-based (hexagonal) layout
    # 5 rows, 6 columns, but only use 26 positions
    # Start with a grid that allows easy packing

    # Generate initial grid layout (hexagonal packing)
    # 5 rows, 6 columns
    row_spacing = np.sqrt(3) / 2  # vertical spacing for hexagonal packing
    col_spacing = 1.0  # horizontal spacing
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        row = i // 6
        col = i % 6
        x = col * col_spacing
        y = row * row_spacing
        if row % 2 == 1:
            x += col_spacing / 2  # offset every other row
        centers[i, 0] = x
        centers[i, 1] = y

    # Define the initial radii (all equal for initial approximation)
    radii = np.full(n_circles, 0.1)

    # Objective function to maximize sum of radii
    def objective(x, centers, radii):
        return -np.sum(radii)  # minimize negative of sum

    # Constraint: non-overlapping circles
    def non_overlap_constraint(x, centers, radii):
        n = centers.shape[0]
        cons = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                cons.append(dist - (radii[i] + radii[j]) + 1e-12)
        return np.array(cons)

    # Constraint: circles within unit square
    def boundary_constraint(x, centers, radii):
        n = centers.shape[0]
        cons = []
        for i in range(n):
            dx = centers[i, 0] - radii[i]
            dy = centers[i, 1] - radii[i]
            cons.append(dx)
            cons.append(1.0 - centers[i, 0] - radii[i])
            cons.append(dy)
            cons.append(1.0 - centers[i, 1] - radii[i])
        return np.array(cons)

    # Optimization bounds (centers and radii must be positive)
    bounds = []
    for i in range(n_circles):
        # Center bounds (within the square)
        bounds.append((0.0, 1.0))
        bounds.append((0.0, 1.0))
        # Radius bounds (non-negative)
        bounds.append((0.0, 1.0))

    # Create x as an array of all variables (centers and radii)
    x0 = np.concatenate([centers.flatten(), radii])

    options = {
        'disp': False,
        'ftol': 1e-12,
        'maxiter': 500
    }

    result = opt.minimize(
        fun=lambda x: objective(x, centers, radii),
        x0=x0,
        bounds=bounds,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
