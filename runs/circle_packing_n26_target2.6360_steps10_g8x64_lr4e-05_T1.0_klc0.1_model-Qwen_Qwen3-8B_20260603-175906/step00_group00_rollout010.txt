import numpy as np
import scipy.optimize as optimize

def generate_hexagonal_packing(num_circles, grid_rows, grid_cols):
    """
    Generate initial positions for a hexagonal grid packing.
    """
    # Hexagonal grid with row-wise offset
    positions = []
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j + 0.5 * i
            y = i
            if i % 2 == 1:
                x += 0.5
            positions.append((x, y))
    # Adjust to fit within the unit square
    positions = np.array(positions)
    positions -= np.min(positions, axis=0)
    positions /= np.max(positions, axis=0)
    return positions[:num_circles]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():

    grid_rows = 7
    grid_cols = 4
    num_circles = 26

    initial_radii = np.ones(num_circles)

    def objective(radii):
        return -np.sum(radii)

    def constraint_overlap(i, j, centers, radii):
        dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
        return dist - radii[i] - radii[j]

    def constraint_boundary(i, centers, radii):

        return x - radii[i] - 1e-12

        return 1 - x - radii[i] - 1e-12

        return 1 - y - radii[i] - 1e-12

    cons = []
    for i in range(num_circles):
