import numpy as np
import scipy.optimize as optimize
import matplotlib.pyplot as plt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial placement
    n = 26
    num_rows = 5  # For better packing
    num_cols_per_row = 5  # Can adjust as needed

    # Place circles in a hexagonal grid
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    x = 0.25
    y = 0.25
    for i in range(n):
        row = i // num_cols_per_row
        col = i % num_cols_per_row
        angle = 60 * row
        dx = 0.5 * np.cos(np.radians(angle))
        dy = 0.5 * np.sin(np.radians(angle))
        centers[i, 0] = x + dx * col
        centers[i, 1] = y + dy * row
        radii[i] = 0.15  # Initial radius

    # Define the cost function for optimization
    def cost_function(x):
        # x is an array of length 2n: [x1, y1, r1, x2, y2, r2, ..., xn, yn, rn]
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:].reshape(n)

        # Check if everything is valid
        valid, msg = validate_packing(centers_flat, radii_flat)
        if not valid:
            return np.inf
        
        # Compute the total sum of radii
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize sum_radii

    # Prepare initial guess
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization using simulated annealing
    result = optimize.minimize(
        fun=cost_function,
        x0=initial_guess,
        method='anneal',
        options={
            'temperature': 1.0,
            'maxiter': 1000,
            'tol': 1e-6,
            'initial_temp': 100,
            'cooling_schedule': 'fast',
            'acceptance_probability': 'boltzmann'
        }
    )

    # Extract the optimal centers and radii

    optimal_radii = result.x[2*n:].reshape(n)

    # Validate the final packing
    valid, msg = validate_packing(optimal_centers, optimal_radii)
    if not valid:

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
