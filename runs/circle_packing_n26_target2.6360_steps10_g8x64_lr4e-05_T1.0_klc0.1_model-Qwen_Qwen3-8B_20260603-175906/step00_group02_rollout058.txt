import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing (approximate)
    n_circles = 26
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))

    # Hexagonal grid spacing
    dx = 1.0 / (cols + 1)
    dy = np.sqrt(3) / (2 * (rows + 1))

    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)
    count = 0

    for row in range(rows):
        for col in range(cols):
            if count >= n_circles:
                break
            x = col * dx + dx / 2
            y = row * dy + dy / 2
            if row % 2 == 1:
                x += dx / 2
            centers[count] = [x, y]
            radii[count] = 0.02  # Initial guess
            count += 1

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles within unit square
    def constraint1(radii):
        return np.sum(radii)
    cons = [{'type': 'ineq', 'fun': lambda r: np.sum(r)}]

    # Constraint: circle centers within [0,1]^2
    def constraint2(x):
        return np.all((x >= -1e-12) & (x <= 1 + 1e-12))
    cons.append({'type': 'ineq', 'fun': lambda x: constraint2(x)})

    # Constraint: circles do not overlap
    def constraint3(radii, centers):
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0
    cons.append({'type': 'ineq', 'fun': lambda x: constraint3(x, centers)})

    # Optimization using SLSQP
    bounds = [(0, 1) for _ in range(n_circles)]
    result = opt.minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
