import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds

def run_packing():

    radii = np.zeros(num_circles)

    for i in range(5):
        for j in range(5 + (i % 2)):
            row = i
            col = j
            if i == 0 or i == 4:
                # Place corner circles at top and bottom
                centers[5 * i + j, 0] = col * (1.0 / 4)
                centers[5 * i + j, 1] = row * (1.0 / 4)
            else:
                # Middle rows
                centers[5 * i + j, 0] = col * (1.0 / 4)
                centers[5 * i + j, 1] = row * (1.0 / 4)
    
    # Initialize radii: start with a base radius
    base_radius = 0.04
    radii[:] = base_radius

    # Initial objective: maximize the sum of radii

        return -x[1:]  # Maximize sum of radii by minimizing the negative sum

    # Flatten centers and radii for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Constraints
    def constraint_center(x):

        radii_flat = x[2*26:]

        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (x_i - r_i >= 0 - 1e-8 and x_i + r_i <= 1 + 1e-8 and
                    y_i - r_i >= 0 - 1e-8 and y_i + r_i <= 1 + 1e-8):
                return -1
        return 0

    def constraint_overlap(x):

            for j in range(i + 1, 26):
                x1, y1 = centers_flat[i]

                r1 = radii_flat[i]
                r2 = radii_flat[j]
                dx = x1 - x2
                dy = y1 - y2
                dist_sq = dx * dx + dy * dy
                if dist_sq < (r1 + r2) ** 2 - 1e-8:

    # Bounds: centers in [0,1]^2, radii >= 0

    res = optimize.minimize(

    sum_radii = np.sum(radii)
    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
