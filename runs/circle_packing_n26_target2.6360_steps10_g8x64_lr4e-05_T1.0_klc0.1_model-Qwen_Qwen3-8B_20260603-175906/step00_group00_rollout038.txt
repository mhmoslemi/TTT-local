import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():

    n = 26

    grid_size = 5
    grid = np.zeros((n, 2))
    idx = 0

                grid[idx, 0] = j / (grid_size - 1)
                grid[idx, 1] = i / (grid_size - 1)
                idx += 1

    # Initial radii - all set to 0.1 for uniformity
    radii = np.full(n, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Negative to use minimization

    # Bounds for the radii (non-negative)
    bounds = Bounds(0, np.inf)

    # Constraints: circles must lie within the unit square
    def square_bounds(x, radii):

            x_c, y_c = centers[i]
            r = radii[i]
            if not (0 - r <= x_c <= 1 + r and 0 - r <= y_c <= 1 + r):
                return np.inf  # Penalize out-of-bound circles
        return 0

    # Constraints: circles must not overlap
    def no_overlap(x, radii):
        centers = x.reshape(26, 2)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j] - 1e-6:
                    return np.inf  # Penalize overlapping circles
        return 0

    initial_guess = np.reshape(grid, (n * 2, ))
    result = opt.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,

            {'type': 'ineq', 'fun': no_overlap, 'args': (radii,)},

            'maxiter': 500,
            'disp': False
        }

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            if dist < radii[i] + radii[j] - 1e-12:
