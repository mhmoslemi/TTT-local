import numpy as np
import scipy.optimize as opt
from typing import Tuple

def validate_packing(centers: np.ndarray, radii: np.ndarray) -> Tuple[bool, str]:
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        r = radii[i]

                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_grid_placement(n_circles: int) -> Tuple[np.ndarray, np.ndarray]:
    # Create a grid with 5 rows and 5 columns (25 circles), leave space for the 26th
    num_rows = 5
    num_cols = 5
    grid_centers = np.zeros((n_circles, 2))
    grid_radii = np.zeros(n_circles)

    spacing_x = 1.0 / (num_cols + 1)
    spacing_y = 1.0 / (num_rows + 1)

    # Place 25 circles in a grid
    for i in range(num_rows):
        for j in range(num_cols):
            x = j * spacing_x + spacing_x / 2
            y = i * spacing_y + spacing_y / 2
            grid_centers[i * num_cols + j] = [x, y]

    y = 1 - spacing_y / 2 - spacing_y / 2
    grid_centers[-1] = [x, y]
    grid_radii[-1] = spacing_x / 2 - 1e-5

    return grid_centers, grid_radii

def objective_function(x: np.ndarray, n_circles: int) -> float:
    # x is a 2D array of shape (n_circles, 2) with circle centers and radii
    centers = x[:, :2]
    radii = x[:, 2]

    return -np.sum(radii)

def constraint_overlap(x: np.ndarray, n_circles: int) -> np.ndarray:
    centers = x[:, :2]
    radii = x[:, 2]

    constraints = np.zeros((n_circles * (n_circles - 1) // 2, 1))
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            constraints[j - i - 1] = dist - radii[i] - radii[j]
    return constraints

def constraint_boundary(x: np.ndarray, n_circles: int) -> np.ndarray:
    centers = x[:, :2]
    radii = x[:, 2]

    constraints = np.zeros((n_circles * 4, 1))
    for i in range(n_circles):
        x_i, y_i = centers[i]
        r_i = radii[i]
        constraints[i * 4] = x_i - r_i
        constraints[i * 4 + 1] = x_i + r_i - 1
        constraints[i * 4 + 2] = y_i - r_i
        constraints[i * 4 + 3] = y_i + r_i - 1
    return constraints

    centers_initial, radii_initial = initial_grid_placement(n_circles)

        bounds.append((0, 1))  # y
