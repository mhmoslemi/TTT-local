import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based packing (grid of 5x5 = 25 circles, plus one extra)
    n_circles = 26
    grid_size = int(np.ceil(np.sqrt(n_circles)))
    spacing = 1.0 / (grid_size + 1)

    # Create an initial grid layout
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    for i in range(n_circles):
        row = i // (grid_size)
        col = i % (grid_size)
        x = col * spacing + spacing * 0.5
        y = row * spacing + spacing * 0.5
        centers[i, 0] = x
        centers[i, 1] = y

    # Initial guess with equal radii
    radii_initial = np.ones(n_circles) * 0.05  # Small initial guess to avoid overlap

    # Define the objective function (maximize sum of radii -> minimize negative of it)
    def objective(radii):
        return -np.sum(radii)

    # Bounds for radii: [0, 1]
    bounds = [(0.0, 1.0) for _ in range(n_circles)]

    # Constraints:
    # 1. Circles must stay within the unit square
    def circle_bounds(x):
        x, y, r = x[0], x[1], x[2]
        return np.array([
            x - r - 1e-8,   # x - r >= -1e-8
            1.0 - x - r - 1e-8,  # 1 - x - r >= -1e-8
            y - r - 1e-8,   # y - r >= -1e-8
            1.0 - y - r - 1e-8   # 1 - y - r >= -1e-8
        ])

    # 2. Circles must not overlap
    def non_overlap(x):
        x_centers = x[:2 * n_circles].reshape(n_circles, 2)
        x_radii = x[2 * n_circles:]
        overlaps = []
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = x_centers[i, 0] - x_centers[j, 0]
                dy = x_centers[i, 1] - x_centers[j, 1]
                dist = np.hypot(dx, dy)
                min_dist = x_radii[i] + x_radii[j]
                if dist < min_dist - 1e-8:
                    overlaps.append(dist - min_dist + 1e-8)
        return np.array(overlaps)

    # Combine all constraints
    def constraints(x):
        # Boundary constraints
        bounds = circle_bounds(x)
        # Non-overlap constraints
        overlaps = non_overlap(x)
        return np.concatenate([bounds, overlaps])

    # Define the optimization problem
    x0 = np.concatenate([centers.flatten(), radii_initial])
    result = opt.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        method='SLSQP',
        options={'disp': False, 'ftol': 1e-8, 'eps': 1e-8}

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
