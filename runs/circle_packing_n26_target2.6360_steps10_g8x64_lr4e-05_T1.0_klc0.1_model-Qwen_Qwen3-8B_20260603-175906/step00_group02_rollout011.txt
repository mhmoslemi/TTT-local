import numpy as np

import cvxpy as cp

def run_packing():

    num_circles = 26

    centers = np.array([[x, y] for y in grid for x in grid])
    centers = centers[:num_circles]  # Only take 26 circles

    # Perturb centers slightly to break symmetry and improve packing
    np.random.seed(42)
    perturb = np.random.normal(0, 0.05, size=centers.shape)
    centers += perturb

    # Optimization: maximize total radius
    # Define variables
    radii = cp.Variable(num_circles, nonneg=True)
    centers_var = cp.Variable((num_circles, 2))

    # Define constraints
    constraints = []
    for i in range(num_circles):
        x, y = centers_var[i]
        r = radii[i]
        # Stay within unit square
        constraints.append(x - r >= -1e-6)
        constraints.append(x + r <= 1 + 1e-6)
        constraints.append(y - r >= -1e-6)
        constraints.append(y + r <= 1 + 1e-6)

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = centers_var[i, 0] - centers_var[j, 0]
            dy = centers_var[i, 1] - centers_var[j, 1]
            dist = cp.sqrt(dx**2 + dy**2)
            constraints.append(dist >= radii[i] + radii[j] - 1e-6)

    # Objective: maximize sum of radii
    objective = cp.Maximize(cp.sum(radii))

    # Solve optimization
    problem = cp.Problem(objective, constraints)
    problem.solve(solver=cp.SCS, eps=1e-6, max_iter=1000)

    if problem.status != "optimal":
        raise RuntimeError(f"Optimization failed with status: {problem.status}")

    # Extract results
    centers_opt = centers_var.value
    radii_opt = radii.value
    sum_radii = np.sum(radii_opt)

    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
