import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

        centers[i, 1] = row * row_step + row_step / 4

    # Initial guess for radii (small value)

        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints
    def constraints(x):
        centers_flat = x[:n*2]
        radii_flat = x[n*2:]
        centers = centers_flat.reshape(n, 2)

            return np.inf
