import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal grid setup
    def initialize_hex_grid(num_circles):
        # Compute the number of rows needed for hexagonal packing
        rows = int(np.sqrt(num_circles)) + 1
        cols = int(np.ceil(num_circles / rows))
        centers = np.zeros((num_circles, 2))
        idx = 0
        for r in range(rows):
            for c in range(cols):
                if idx >= num_circles:
                    break
                # Offset even rows
                x = c * (1 + 1/np.sqrt(3)) / (cols - 1) if r % 2 == 0 else c * (1 + 1/np.sqrt(3)) / (cols - 1) + 1/(2 * np.sqrt(3)) 
                y = r * np.sqrt(3)/2 / (rows - 1)
                centers[idx] = [x, y]
                idx += 1
        return centers

    # Initial guess for radii (all 0.1)
    initial_radii = np.full(26, 0.1)

    # Function to compute sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize radii

    # Constraints
    def constraint_centers(radii, centers):
        for i in range(26):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return 1  # Constraint violation
        return 0

    def constraint_distances(radii):
        for i in range(26):
            for j in range(i+1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return 1  # Constraint violation
        return 0

    # Initial centers from hexagonal grid
    centers = initialize_hex_grid(26)

    # Optimization
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraint_centers(r, centers)},
            {'type': 'ineq', 'fun': lambda r: constraint_distances(r)},
        ],
        bounds=[(0, 1.0) for _ in range(26)]
    )

        valid, message = validate_packing(centers, radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
