import numpy as np
import cvxpy as cp

def run_packing():
    # Parameters
    n = 26
    square_side = 1.0
    eps = 1e-12

    # Initial grid-based placement
    grid_rows = 6
    grid_cols = 5
    grid_spacing = square_side / (grid_cols + 1)
    grid_centers = np.zeros((n, 2))
    index = 0
    for i in range(grid_rows):
        for j in range(grid_cols):
            grid_centers[index, 0] = j * grid_spacing + eps
            grid_centers[index, 1] = i * grid_spacing + eps
            index += 1

    # Initial radius estimate based on grid spacing
    initial_radii = np.ones(n) * (grid_spacing / 2 - eps)

    # Define optimization variables
    centers = cp.Variable((n, 2))
    radii = cp.Variable(n, non_negative=True)

    # Constraints
    constraints = []

    # Centering and boundary constraints
    for i in range(n):
        constraints.append(centers[i, 0] >= eps)
        constraints.append(centers[i, 0] <= 1 - eps)
        constraints.append(centers[i, 1] >= eps)
        constraints.append(centers[i, 1] <= 1 - eps)

    # Non-overlap constraints
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.sqrt(cp.sum_squares(centers[i] - centers[j]))
            constraints.append(dist >= radii[i] + radii[j] - eps)

    # Objective: maximize sum of radii
    objective = cp.Maximize(cp.sum(radii))

    # Solve optimization problem
    prob = cp.Problem(objective, constraints)
    prob.solve(solver=cp.SCS, max_iter=10000, eps=1e-6, verbose=False)

    if prob.status != 'optimal':
        print("Optimization failed")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    radii_opt = radii.value
    sum_radii = cp.sum(radii).value

    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
