import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26

    base_centers = []

            base_centers.append([x, y])

    # Combine base and extra centers
    centers_guess = np.vstack([base_centers, extra_centers])

    radii_guess = np.full(n, 0.02)

    # Flatten centers for optimization
    x0 = np.hstack([centers_guess.flatten(), radii_guess])

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(x):

        radii = x[2 * n:]
        return -np.sum(radii)

    # Constraints: each circle is within the unit square
    def constraint_square(x):
        centers = x[:2 * n].reshape(n, 2)
        radii = x[2 * n:]
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (0 - 1e-12 <= x_i - r_i <= 1 + 1e-12 and 0 - 1e-12 <= y_i - r_i <= 1 + 1e-12):
