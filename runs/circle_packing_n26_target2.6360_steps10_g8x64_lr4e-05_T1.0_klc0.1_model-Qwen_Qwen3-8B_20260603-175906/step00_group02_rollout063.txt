import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_circle_grid(n_circles):
    # Initial hexagonal grid
    # First row
    rows = int(np.ceil(n_circles ** 0.5))
    cols = int(np.ceil(n_circles / rows))
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)
    idx = 0
    for row in range(rows):
        for col in range(cols):
            if idx >= n_circles:
                break
            x = col + 0.5 * (col % 2)
            y = row + 0.5 * (row % 2)
            centers[idx] = [x, y]
            radii[idx] = 0.05
            idx += 1
    return centers, radii

def calculate_objective(params, n_circles, centers, radii):
    # params is a vector of x, y, r for each circle
    # reshape params
    params = params.reshape((n_circles, 3))
    centers = params[:, :2]
    radii = params[:, 2]
    # objective is the negative of the sum of radii (to minimize)
    return -np.sum(radii)

def calculate_constraints(params, n_circles, centers, radii):
    # params is a vector of x, y, r for each circle
    params = params.reshape((n_circles, 3))
    centers = params[:, :2]
    radii = params[:, 2]
    # constraints
    constraints = []
    # 1. All circles are within the square [0,1]x[0,1]
    for i in range(n_circles):
        x, y = centers[i]
        r = radii[i]
        constraints.append({'type': 'ineq', 'fun': lambda p, i=i, x=x, y=y, r=r: x - r})
        constraints.append({'type': 'ineq', 'fun': lambda p, i=i, x=x, y=y, r=r: 1 + r - x})
        constraints.append({'type': 'ineq', 'fun': lambda p, i=i, x=x, y=y, r=r: y - r})
        constraints.append({'type': 'ineq', 'fun': lambda p, i=i, x=x, y=y, r=r: 1 + r - y})
    # 2. All circles are non-overlapping
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            def constraint(p):
                params = p.reshape((n_circles, 3))
                centers_i = params[i, :2]
                radii_i = params[i, 2]
                centers_j = params[j, :2]
                radii_j = params[j, 2]
                dist = np.sqrt((centers_i[0] - centers_j[0])**2 + (centers_i[1] - centers_j[1])**2)
                return dist - (radii_i + radii_j) + 1e-8
            constraints.append({'type': 'ineq', 'fun': constraint})
    return constraints

def run_packing():
    n_circles = 26
    # Initial guess: hexagonal grid
    centers, radii = initial_circle_grid(n_circles)
    # Flatten centers and radii into a vector
    initial_params = np.concatenate([centers.reshape(-1), radii])
    # Bounds for x, y in [0,1], radii >= 0
    bounds = []
    for _ in range(n_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius
    # Optimization
    result = optimize.minimize(
        fun=calculate_objective,
        x0=initial_params,
        method='SLSQP',
        bounds=bounds,
        constraints=calculate_constraints(initial_params, n_circles, centers, radii)
    )
    # Extract optimal solution
    optimal_params = result.x
    centers = optimal_params[:n_circles*2].reshape((n_circles, 2))
    radii = optimal_params[n_circles*2:]
    # Validate
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
        return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
    sum_radii = np.sum(radii)
    return centers, radii, sum_radii
