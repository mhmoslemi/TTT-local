import numpy as np
import cvxpy as cp

def run_packing():
    # Define the number of circles
    n = 26

    # Initial grid points using hexagonal packing
    # We'll arrange in a hexagonal grid with 6 rows
    # Rows will have 5, 6, 6, 6, 6, 5 circles
    # Coordinates start from the bottom left corner
    centers = np.zeros((n, 2))
    col_count = 0
    row_count = 0
    offset = 0.4  # Horizontal offset between columns
    vertical_offset = np.sqrt(3) / 2 * 0.4  # Vertical offset for hexagonal packing

    for i in range(n):
        row = i // 5
        if row % 2 == 1:
            col = (i % 5) * offset
        else:
            col = (i % 5) * offset

        centers[i, 0] = col
        centers[i, 1] = row * vertical_offset

    # Create variables for centers and radii
    centers_var = cp.Variable((n, 2))
    radii_var = cp.Variable(n, nonneg=True)

    # Objective: Maximize the sum of radii
    objective = cp.Maximize(cp.sum(radii_var))

    # Constraints: No overlaps and stay within the unit square
    constraints = []

    # Boundary constraints: Each circle must stay within [0,1]x[0,1]
    for i in range(n):
        x, y = centers_var[i]
        r = radii_var[i]
        constraints.append(x - r >= 0)
        constraints.append(x + r <= 1)
        constraints.append(y - r >= 0)
        constraints.append(y + r <= 1)

    # Circle separation constraints: distance >= sum of radii
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.sqrt(cp.sum_squares(centers_var[i] - centers_var[j]))
            constraints.append(dist >= radii_var[i] + radii_var[j])

    # Solve the optimization problem
    prob = cp.Problem(objective, constraints)
    prob.solve()

    # Extract the solution
    centers_opt = centers_var.value
    radii_opt = radii_var.value
    total_radii = np.sum(radii_opt)

    # Return the result
    return centers_opt, radii_opt, total_radii
