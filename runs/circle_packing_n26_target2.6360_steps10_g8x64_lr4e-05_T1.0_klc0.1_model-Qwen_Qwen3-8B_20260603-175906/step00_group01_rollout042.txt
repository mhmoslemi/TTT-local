import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial estimate for the number of circles per row in hexagonal packing
    num_rows = 6
    num_circles_per_row = 5  # Adjust based on trial
    
    # Calculate vertical distance between rows to respect circle spacing
    vertical_spacing = np.sqrt(3) / 2  # For hexagonal packing
    
    # Generate initial centers using a hexagonal grid configuration
    centers = []
    radii = []
    
    for row in range(num_rows):
        for col in range(num_circles_per_row):
            x = col + 0.5  # Start from middle of a column
            y = row * vertical_spacing + 0.3  # Offset vertically to fit in square
            centers.append([x, y])
            radii.append(0.1)  # Initial estimate for radius

    centers = np.array(centers)
    radii = np.array(radii)
    
    # Define the optimization problem: maximize the sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Negative for minimization problem
    
    def constraint_non_overlap(i, j, centers, radii_vec):
        dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
        return dist - radii_vec[i] - radii_vec[j] - 1e-12
    
    def constraint_boundary(i, centers, radii_vec):
        x, y = centers[i]
        r = radii_vec[i]
        return (x - r >= -1e-12) * (x + r <= 1 + 1e-12) * (y - r >= -1e-12) * (y + r <= 1 + 1e-12) - 1e-12

    # Set up optimization bounds and constraints
    bounds = [(0, 0.5) for _ in range(26)]
    
    # Define constraints for non-overlap and boundary
    constraints = []
    for i in range(26):
        for j in range(i + 1, 26):
            constraints.append({
                'type': 'ineq',
                'fun': lambda r, i=i, j=j: constraint_non_overlap(i, j, centers, r),
                'args': (centers, r)
            })
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: constraint_boundary(i, centers, r),
            'args': (centers, r)
        })

    # Run the optimization
    result = optimize.minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-6
    )
    
    # Extract the final result
    final_radii = result.x
    final_centers = centers
    final_sum_radii = np.sum(final_radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
