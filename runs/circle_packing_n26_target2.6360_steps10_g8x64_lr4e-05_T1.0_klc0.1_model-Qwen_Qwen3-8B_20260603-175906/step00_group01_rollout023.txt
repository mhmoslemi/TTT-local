import scipy.optimize as opt
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for centers using a hexagonal grid
    num_circles = 26
    max_radius = 0.1  # Initial estimated maximum radius
    max_rows = 5  # Number of rows in hexagonal grid
    max_cols = 5  # Number of columns in hexagonal grid

    centers = []
    radii = []

    # Generate initial hexagonal grid
    for i in range(max_rows):
        for j in range(max_cols):
            x = j + (i % 2) * 0.5
            y = i * np.sqrt(3) / 2
            if x < 1 and y < 1:
                centers.append([x, y])
                radii.append(max_radius)

    centers = np.array(centers)
    radii = np.array(radii)

    # Objective function: maximize the sum of radii squared
    def objective(x):
        x_flat = x.reshape((-1, 2))
        r_flat = x[-num_circles:]
        r_flat[r_flat < 0] = 0  # Ensure non-negative radii

        # Check boundary constraints
        for i in range(num_circles):
            x_i, y_i = x_flat[i]
            r_i = r_flat[i]
            if not (0 - 1e-12 <= x_i - r_i <= 1 + 1e-12 and
                    0 - 1e-12 <= y_i - r_i <= 1 + 1e-12):
                return -np.inf

        # Check pairwise circle constraints
        dist = cdist(x_flat, x_flat)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dist[i, j] < r_flat[i] + r_flat[j] - 1e-12:
                    return -np.inf

        # Return negative sum of radii squared to maximize
        return -np.sum(r_flat ** 2)

    # Bounds for positions (x, y) and radii (r)
    bounds = []
    for i in range(num_circles):
        # Positions: [0, 1] for x and y
        bounds.append((0.0, 1.0))
        bounds.append((0.0, 1.0))
        # Radii: [0, max_radius]
        bounds.append((0.0, max_radius))

    # Initial guess: flatten centers and radii
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Run optimization
    result = opt.minimize(

        initial_guess,
        bounds=bounds,
        method='SLSQP',

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
