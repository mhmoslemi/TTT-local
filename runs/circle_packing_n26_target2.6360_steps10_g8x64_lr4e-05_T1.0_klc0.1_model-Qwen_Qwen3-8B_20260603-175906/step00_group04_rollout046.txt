import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial arrangement: hexagonal grid
    num_circles = 26
    rows = 5
    cols = 5
    # For hexagonal packing, each row is offset by half the spacing
    spacing = 0.25
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                x = j * spacing
            else:
                x = (j + 0.5) * spacing
            y = i * spacing * np.sqrt(3) / 2
            if i < rows and j < cols:
                centers[i * cols + j] = np.array([x, y])
                radii[i * cols + j] = 0.04
    
    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    def non_overlap_constr(x):
        centers_x, radii_x = x[:26], x[26:]
        n = len(centers_x)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_x[i, 0] - centers_x[j, 0]
                dy = centers_x[i, 1] - centers_x[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_x[i] + radii_x[j] - 1e-12:
                    return dist - (radii_x[i] + radii_x[j] - 1e-12)
        return 0.0

    def boundary_constr(x):
        centers_x, radii_x = x[:26], x[26:]
        for i in range(len(centers_x)):
            x_center, y_center = centers_x[i]
            r = radii_x[i]
            if not (0 - r <= x_center <= 1 + r) or not (0 - r <= y_center <= 1 + r):
                return 1.0
        return 0.0

    # Initial guess
    initial_guess = np.concatenate([centers, radii])

    # Bounds: centers in [0,1]^2, radii non-negative
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))  # x center
        bounds.append((0, 1))  # y center
        bounds.append((0, 1))  # radius

    # Optimize
    result = opt.minimize(
        objective,
        initial_guess,
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': non_overlap_constr},
            {'type': 'ineq', 'fun': boundary_constr}
        ],
        method='SLSQP',
        tol=1e-8
    )

    # Extract final centers and radii
    final_centers = result.x[:26].reshape((26, 2))
    final_radii = result.x[26:]
    final_sum_radii = np.sum(final_radii)

    # Validate
    is_valid, message = validate_packing(final_centers, final_radii)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return final_centers, final_radii, final_sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
