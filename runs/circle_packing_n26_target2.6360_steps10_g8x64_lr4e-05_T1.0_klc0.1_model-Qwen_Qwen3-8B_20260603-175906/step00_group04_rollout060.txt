from scipy.optimize import minimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define grid parameters
    num_rows = 6
    num_cols = 5
    total_circles = num_rows * num_cols

    # Initial grid-based center coordinates for hexagonal packing
    spacing = 1.0 / math.sqrt(3)  # For hexagonal packing
    centers = np.zeros((total_circles, 2))
    for i in range(num_rows):
        for j in range(num_cols):
            x = 0.5 + (j + 0.5) * spacing
            y = 0.5 + (i * (spacing * math.sqrt(3))) if i % 2 == 0 else 0.5 + i * (spacing * math.sqrt(3))
            centers[i * num_cols + j] = [x, y]

    # Initial guess for radii (equal radii as a starting point)
    initial_radii = np.full(total_circles, 1 / (4 * math.sqrt(3)))

    # Objective function to maximize total sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints for non-overlapping and boundary conditions
    def constraint_overlap(x):
        # x = [centers, radii]
        centers = x[:total_circles * 2].reshape(total_circles, 2)
        radii = x[total_circles * 2:].reshape(total_circles)
        for i in range(total_circles):
            for j in range(i + 1, total_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -1  # Violation of non-overlap constraint
        return 0

    def constraint_boundary(x):
        centers = x[:total_circles * 2].reshape(total_circles, 2)
        radii = x[total_circles * 2:].reshape(total_circles)
        for i in range(total_circles):
            x_center, y_center = centers[i]
            r = radii[i]
            if not (0 - 1e-12 <= x_center - r <= 1 + 1e-12 and 0 - 1e-12 <= y_center - r <= 1 + 1e-12):
                return -1  # Violation of boundary constraint
        return 0

    # Bounds for centers and radii (centers are in [0,1]^2)
    bounds_centers = [(0.0, 1.0) for _ in range(total_circles * 2)]
    bounds_radii = [(0.0, 1.0) for _ in range(total_circles)]

    # Combine bounds for the optimization problem
    bounds = bounds_centers + bounds_radii

    # Initial guess: centers and radii
    x0 = np.concatenate([centers.flatten(), initial_radii])

    # Optimization settings (using SLSQP method for equality constraints)
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary}
    ]

    result = minimize(

        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
