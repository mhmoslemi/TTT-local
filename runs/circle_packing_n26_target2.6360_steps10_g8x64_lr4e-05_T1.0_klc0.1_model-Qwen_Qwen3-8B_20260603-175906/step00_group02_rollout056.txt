from scipy.optimize import minimize

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    def initial_hexagonal_layout(num_circles: int) -> np.ndarray:
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        centers = np.zeros((num_circles, 2))
        idx = 0
        for r in range(rows):
            for c in range(cols):
                x = c + 0.5 * (r % 2)
                y = r
                if idx < num_circles:
                    centers[idx] = [x, y]
                idx += 1
        return centers

        return -np.sum(radii)  # Negative because we minimize

    def constraints(centers: np.ndarray, radii: np.ndarray) -> list:

        n = centers.shape[0]
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - x})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - y})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: y - r})

                    r_j = radii[j]
                    dist = np.linalg.norm(centers[i] - centers[j])
                    return dist - r_i - r_j - 1e-12
                constraints.append({'type': 'ineq', 'fun': dist_constraint})

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
            or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
