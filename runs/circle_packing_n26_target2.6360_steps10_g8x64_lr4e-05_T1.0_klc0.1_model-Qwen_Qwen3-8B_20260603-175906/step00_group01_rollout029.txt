import scipy.optimize as opt

def run_packing():

    n_circles = 26

    rows = 4
    cols = 7
    dx = 1 / (cols - 1)
    dy = np.sqrt(3) / (2 * (rows - 1))
    
    # Generate an initial grid of centers
    centers = np.zeros((n_circles, 2))
    for i in range(rows):

            x = col
            y = row * dy
            if i % 2 == 1:
                x += dx / 2
            centers[i * cols + j] = [x, y]

    # Assign initial radii to all circles
    radii = np.zeros(n_circles)

    # Define objective function to maximize sum of radii
    def objective(x):

        x_flat = x.reshape((n_circles, 4))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]

        return -np.sum(radii)  # Negative for minimization

    # Define constraints
    def constraint_center(x):
        x_flat = x.reshape((n_circles, 4))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        bounds = np.zeros(n_circles)
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12) or (x + r > 1 + 1e-12) or (y - r < -1e-12) or (y + r > 1 + 1e-12):
                bounds[i] = 1
        return bounds

    def constraint_overlap(x):
        x_flat = x.reshape((n_circles, 4))

        radii = x_flat[:, 2]
        overlaps = np.zeros(n_circles)
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    overlaps[i] = 1
                    overlaps[j] = 1

    # Define bounds for each circle: [x, y, r] in [0,1] and r >= 0
