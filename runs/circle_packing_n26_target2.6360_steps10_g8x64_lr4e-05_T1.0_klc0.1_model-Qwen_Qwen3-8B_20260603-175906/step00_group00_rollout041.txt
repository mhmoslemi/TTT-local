import numpy as np
import scipy.optimize as optimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Initial placement using hexagonal packing

    # Define hexagonal packing with spacing = r * sqrt(3) for rows and r * 2 for columns
    # Assume initial radius r = 0.01 for the first pass
    num_circles = 26
    initial_r = 0.01
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, initial_r)

    # Place circles in hexagonal grid
    rows, cols = np.indices((6, 5))
    for i in range(rows.shape[0]):
        for j in range(cols.shape[1]):
            r = initial_r + 0.005 * i  # Add spacing between rows
            angle = 2 * np.pi / 6 * j  # Hexagonal angle
            x = r + r * np.cos(angle)
            y = r + r * np.sin(angle)
            centers[i * 5 + j] = [x, y]
            radii[i * 5 + j] = r

    # Step 2: Adjust circles to fill gaps and optimize radius
    # Define constants
    bounds = [(0.0, 1.0) for _ in range(2)]  # for centers
    bounds += [(0.0, 1.0) for _ in range(num_circles)]  # for radii

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize the sum of radii

    def constraint_overlap(i, j, centers, radii):
        dx = centers[i, 0] - centers[j, 0]
        dy = centers[i, 1] - centers[j, 1]
        dist_sq = dx**2 + dy**2
        dist = np.sqrt(dist_sq)
        return dist - (radii[i] + radii[j])

    def constraint_center(x, y, r):
        return x - r - 1e-6  # Ensure x - r >= 0
    def constraint_center2(x, y, r):
        return 1.0 - x - r - 1e-6  # Ensure x + r <= 1
    def constraint_center3(x, y, r):
        return y - r - 1e-6  # Ensure y - r >= 0
    def constraint_center4(x, y, r):
        return 1.0 - y - r - 1e-6  # Ensure y + r <= 1

    # Initial guess for centers and radii
    x_guess = np.random.rand(num_circles, 2)
    r_guess = np.ones(num_circles) * initial_r

    # Optimize using scipy.optimize.minimize
    result = optimize.minimize(
        fun=lambda x: objective(x[num_circles:], x[:num_circles]),
        x0=np.concatenate((x_guess.flatten(), r_guess)),
        bounds=bounds,
        constraints=[

        ],

# Helper function for validation
def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
