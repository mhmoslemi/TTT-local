    r = cp.Variable(n)
    x = cp.Variable((n, 2))

    # Objective: Maximize the sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # Non-negative radii
    constraints.append(r >= 0)

    # Boundary constraints: x ± r in [0, 1]
    for i in range(n):

            constraints.append(dist >= r[i] + r[j])

    # Solve the problem
    prob = cp.Problem(objective, constraints)
    result = prob.solve(solver=cp.OSQP, eps=1e-6)
