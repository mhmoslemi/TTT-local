import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial hexagonal grid of circles
    num_circles = 26
    # Set a seed for reproducibility
    np.random.seed(42)

    # Hexagonal grid generation
    # First, place circles in a hexagonal grid
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    grid_centers = []
    grid_radii = []

    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)  # Offset for hexagonal packing
            y = i * np.sqrt(3) / 2
            grid_centers.append([x, y])
            grid_radii.append(0.05)  # Initial guess for radii

    # Convert to numpy arrays
    centers = np.array(grid_centers)
    radii = np.array(grid_radii)
    sum_radii = np.sum(radii)

    # Define objective function
    def objective(x):
        # x is a flat array containing all radii
        new_radii = x.reshape(-1)
        return -np.sum(new_radii)  # Minimize negative sum to maximize sum

    # Define constraints
    def constraint_boundary(x):
        new_radii = x.reshape(-1)
        for i in range(num_circles):
            x_c, y_c = centers[i]
            r = new_radii[i]
            if not (0 <= x_c - r and x_c + r <= 1 and 0 <= y_c - r and y_c + r <= 1):
                return -1
        return 0

    def constraint_overlap(x):
        new_radii = x.reshape(-1)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < new_radii[i] + new_radii[j] - 1e-12:
                    return -1
        return 0

    # Set bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(num_circles)]

    # Initial guess: start with the initial grid radii
    initial_guess = radii.copy()

    # Optimization settings
    constraints = [
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Run optimization
    result = optimize.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    optimal_radii = result.x

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
