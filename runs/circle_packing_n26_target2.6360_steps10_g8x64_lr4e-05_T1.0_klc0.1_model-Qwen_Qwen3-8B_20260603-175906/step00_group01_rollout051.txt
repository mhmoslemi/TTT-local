    # Start with a hexagonal grid with rows stacked diagonally

    rows = int(np.ceil(np.sqrt(max_circles)))
    cols = int(np.ceil(max_circles / rows))

    # Each row offset vertically by sqrt(3)/2 * r, horizontally by 2*r
    # We set initial radii all to 0.05 and adjust them based on the distances

    radii = np.full(max_circles, 0.05)

    y = 0.5  # Centered vertically

                break

            centers[idx, 1] = y - i * offset
        # Reduce vertical spacing slightly to ensure all circles fit within [0,1]
        y -= offset * 0.95

    # Now perform an optimization to maximize the sum of radii with constraints
    # Objective function: maximize sum(radii)

        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints: non-overlapping and staying within [0,1]

                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -1  # Violation of constraint
        return 0

        centers = x[:len(centers)*2].reshape((-1, 2))

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
