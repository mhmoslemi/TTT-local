def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    rows = 5
    cols = 5
    padding = 0.1  # Add padding to avoid boundary issues
    initial_radii = np.full(num_circles, 0.1)  # Initial guess for radii

    # Optimization: maximize the sum of radii given constraints
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii, so minimize negative sum

        return 0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
