import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial guess: 5x5 grid with some space
    n = 26
    grid_size = int(np.sqrt(n))  # 5x5 grid with some space
    spacing = 1.0 / (grid_size + 1)
    centers = np.zeros((n, 2))
    idx = 0
    for i in range(grid_size):
        for j in range(grid_size):
            if idx < n:
                centers[idx, 0] = (i + 1) * spacing
                centers[idx, 1] = (j + 1) * spacing
                idx += 1
            else:
                break

    # Initial radius
    r0 = 0.05
    radii = np.full(n, r0)

    # Define objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Optimization bounds: x in [0, 1], y in [0, 1], r >= 0
    bounds = [(0, 1) for _ in range(n * 2)] + [(0, 1) for _ in range(n)]

    # Define constraints
    def constraint_center(x):
        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:]
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
                return np.inf
            if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf
        return 0

    def constraint_overlap(x):
        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:]
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers_flat[i] - centers_flat[j])
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf
        return 0

    # Use SLSQP for constrained optimization
    cons = [
        {'type': 'ineq', 'fun': constraint_center},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Solve the optimization problem
    result = opt.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        tol=1e-8
    )

    # Extract the result
    if not result.success:
        raise RuntimeError("Optimization failed with message: {}".format(result.message))

    centers_opt = result.x[:n * 2].reshape(n, 2)
    radii_opt = result.x[n * 2:]

    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
