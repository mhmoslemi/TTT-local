import scipy.optimize as opt

def run_packing():

    num_circles = 26

        col = i % 5
        grid[i, 0] = col / 4.0  # x-coordinate
        grid[i, 1] = row / 4.0  # y-coordinate

    initial_radii = np.full(num_circles, 0.05)

        return -np.sum(radii)

        min_dist = np.min(dists[np.triu_indices_from(dists, k=1)])
        return radii[0] + radii[1] - min_dist + 1e-8  # Add a small tolerance for numerical stability

    # This defines a single constraint, but in practice, the minimum distance between all pairs is the most restrictive
    # The actual constraint is: for all i < j, |c_i - c_j| >= r_i + r_j
    # To enforce that, we'll use a single constraint that represents the minimum distance between any two circles

            'fun': constraint_overlap,

            'type': 'ineq',

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):

            return False, f"Circle {i} has negative radius {radii[i]}"

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
