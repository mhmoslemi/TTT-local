import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    grid_size = np.ceil(np.sqrt(num_circles))
    grid = np.zeros((num_circles, 2))

        row = i // grid_size
        col = i % grid_size
        x = col / (grid_size - 1)
        y = row / (grid_size - 1)
        grid[i] = [x, y]

    # Initial radii (based on spacing in grid)
    spacing = 1.0 / (grid_size - 1)
    initial_radii = np.full(num_circles, spacing / 2.0)

        return -np.sum(radii)  # Minimize negative of sum (equivalent to maximizing)

    constraints = []

    def in_bounds(x):
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        return np.all((centers >= -1e-12) & (centers <= 1 + 1e-12))
    constraints.append({'type': 'ineq', 'fun': in_bounds})

    # Constraint: all radii are non-negative

        return np.all(x[num_circles * 2:] >= -1e-12)
    constraints.append({'type': 'ineq', 'fun': non_negative})

    def no_overlap(x):

    # Initial guess
