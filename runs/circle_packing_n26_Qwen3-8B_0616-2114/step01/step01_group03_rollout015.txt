import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial square grid layout with padding for packing
    num_circles = 26
    grid_size = int(np.ceil(np.sqrt(num_circles)))
    spacing = 1.0 / (grid_size + 1)
    initial_centers = []
    for i in range(grid_size):
        for j in range(grid_size):
            x = spacing * (j + 1)
            y = spacing * (i + 1)
            initial_centers.append([x, y])
    
    # Initial radii: uniform for simplicity
    initial_radii = np.full(num_circles, 0.05)

    # Optimization bounds: x and y between 0 and 1, radii ≥ 0
    bounds = []
    for i in range(num_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    # Define variables: flattened centers and radii
    variables = np.array([x for center in initial_centers for x in center] + initial_radii.tolist())

    def objective(variables):
        centers = variables[:2*num_circles].reshape(num_circles, 2)
        radii = variables[2*num_circles:]
        return -np.sum(radii)  # Maximize sum of radii, so minimize negative sum

    def constraint_non_overlap(variables):
        centers = variables[:2*num_circles].reshape(num_circles, 2)
        radii = variables[2*num_circles:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return dist - (radii[i] + radii[j]) + 1e-8
        return 0.0

    def constraint_boundaries(variables):
        centers = variables[:2*num_circles].reshape(num_circles, 2)
        radii = variables[2*num_circles:]
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-8 or x + r > 1 + 1e-8 or y - r < -1e-8 or y + r > 1 + 1e-8:
                return 1e-8 - (x - r) if x - r < -1e-8 else 0
        return 0.0

    # Define constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_non_overlap},
        {'type': 'ineq', 'fun': constraint_boundaries}
    ]

    # Optimization
    result = optimize.minimize(
        objective,
        variables,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'maxiter': 500, 'ftol': 1e-9, 'eps': 1e-8}
    )

    # Extract solution
    centers = result.x[:2*num_circles].reshape(num_circles, 2)
    radii = result.x[2*num_circles:]
    sum_radii = np.sum(radii)

    # Final validation
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
    else:
        print(f"Validation passed. Sum of radii: {sum_radii:.4f}")
    
    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
