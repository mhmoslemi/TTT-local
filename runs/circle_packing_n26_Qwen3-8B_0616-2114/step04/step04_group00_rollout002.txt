import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def objective_function(params, n_circles):
    centers = params[:n_circles * 2].reshape((n_circles, 2))
    radii = params[n_circles * 2:].reshape((n_circles,))
    
    # Primary objective: maximize sum of radii
    sum_radii = np.sum(radii)
    # Penalize circles being too close to the edges
    edge_penalty = 0
    for i in range(n_circles):
        r = radii[i]
        x, y = centers[i]
        # Top/bottom boundaries
        edge_penalty += np.maximum(0, r - (1 - y - 1e-12))
        edge_penalty += np.maximum(0, r - (1 - (1 - y) - 1e-12))
        # Left/right boundaries
        edge_penalty += np.maximum(0, r - (1 - x - 1e-12))
        edge_penalty += np.maximum(0, r - (1 - (1 - x) - 1e-12))
    # Penalize circle overlaps
    overlap_penalty = 0
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                overlap_penalty += (radii[i] + radii[j] - dist)
    # Weight the penalties appropriately
    return -sum_radii + 100 * edge_penalty + 1 * overlap_penalty

def generate_hexagonal_config(n_circles):
    centers = np.zeros((n_circles, 2))
    row = 0
    col = 0
    for i in range(n_circles):
        if i % 2 == 0:
            centers[i] = [col / (n_circles // 2), row / (n_circles // 2)]
        else:
            centers[i] = [(col + 0.5) / (n_circles // 2), row / (n_circles // 2)]
        col += 1
        if col >= n_circles // 2:
            row += 1
            col = 0
    return centers

def generate_corner_config(n_circles):
    centers = np.zeros((n_circles, 2))
    idx = 0
    for x in [0.05, 0.95]:
        for y in [0.05, 0.95]:
            centers[idx] = [x, y]
            idx += 1
    for i in range(idx, n_circles):
        centers[i] = [0.5, 0.5]
    return centers

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n_circles = 26
    best_centers = np.zeros((n_circles, 2))
    best_radii = np.zeros(n_circles)
    best_sum_radii = 0.0

    # Generate a variety of initial configurations
    def generate_grid_config(num_rows, num_cols):
        centers = np.zeros((n_circles, 2))
        for i in range(n_circles):
            row = i // num_cols
            col = i % num_cols
            x = col / (num_cols - 1) * 0.95 + 0.025
            y = row / (num_rows - 1) * 0.95 + 0.025
            centers[i] = [x, y]
        return centers
    
    def generate_hex_config(rows, cols):
        centers = np.zeros((n_circles, 2))
        for i in range(n_circles):
            row = i // cols
            col = i % cols
            x = col / (cols - 1) * 0.9 + (0.05 if row % 2 == 0 else 0.05)
            y = row / (rows - 1) * 0.9 + 0.05
            centers[i] = [x, y]
        return centers

    def generate_corner_filled_config():
        centers = np.zeros((n_circles, 2))
        for i in range(n_circles):
            if i == 0:
                centers[i] = [0.05, 0.05]
            elif i == 1:
                centers[i] = [0.95, 0.05]
            elif i == n_circles - 2:
                centers[i] = [0.05, 0.95]
            elif i == n_circles - 1:
                centers[i] = [0.95, 0.95]
            else:
                centers[i] = [0.5, 0.5]
        return centers
    
    def generate_random_config():
        centers = np.random.rand(n_circles, 2)
        centers *= 0.9
        centers += 0.05
        return centers

    # Prepare initial configurations
    initial_configs = []
    # Regular grid
    initial_configs.append(generate_grid_config(6, 5))
    # Hexagonal grid
    initial_configs.append(generate_hex_config(5, 6))
    # Corner-filled
    initial_configs.append(generate_corner_filled_config())
    # Random scattered
    initial_configs.append(generate_random_config())
    # Hexagonal layout
    initial_configs.append(generate_hexagonal_config(n_circles))

    for centers in initial_configs:
        # Initialize radii with small values
        initial_radii = np.ones(n_circles) * 0.06
        # Flatten centers and radii for optimization
        initial_params = np.concatenate([centers.flatten(), initial_radii])
        # Optimization settings
        bounds = []
        for _ in range(n_circles * 2):
            bounds.append((0, 1))
        for _ in range(n_circles):
            bounds.append((0, 1))
        # Use SLSQP for optimization
        result = optimize.minimize(
            fun=lambda x: objective_function(x, n_circles),
            x0=initial_params,
            bounds=bounds,
            method="SLSQP",
            options={"maxiter": 500, "ftol": 1e-9, "eps": 1e-8, "disp": False}
        )
        # Extract the optimal solution
        optimal_params = result.x
        centers_opt = optimal_params[:n_circles * 2].reshape((n_circles, 2))
        radii_opt = optimal_params[n_circles * 2:].reshape((n_circles,))
        sum_radii = np.sum(radii_opt)
        # Validate the packing
        is_valid, message = validate_packing(centers_opt, radii_opt)
        if is_valid and sum_radii > best_sum_radii:
            best_centers = centers_opt
            best_radii = radii_opt
            best_sum_radii = sum_radii

    return best_centers, best_radii, best_sum_radii
