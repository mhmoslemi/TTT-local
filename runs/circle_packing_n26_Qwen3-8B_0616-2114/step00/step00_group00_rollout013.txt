        rows = 5
        cols = 5
        for i in range(rows):

                x = 0.2 + j * (1 - 2 * radius_initial) / (cols - 1)
                y = 0.2 + i * (1 - 2 * radius_initial) / (rows - 1)
                if i % 2 == 1:
                    x += (1 - 2 * radius_initial) / (cols - 1) / 2  # Offset for odd rows
                centers.append([x, y])
        
        # Place two circles in the corners as additional circles
        centers.append([0.05, 0.05])
        centers.append([0.95, 0.95])

        return centers, radii

    # Objective function to maximize sum of radii (equivalent to maximizing total area)
    def objective(x, num_circles, n_corners=2):
        radii = x[:num_circles]
        radius_corners = x[num_circles:]
        return -np.sum(radii) - np.sum(radius_corners)  # Negative for minimization

    # Constraint functions
    def boundary_constraint(x, num_circles, n_corners):
        radii = x[:num_circles]
        radius_corners = x[num_circles:]

            if radii[i] < 0:
                return -1
            if x[i*2] - radii[i] < 0 or x[i*2] + radii[i] > 1:
                return -1
            if x[i*2 + 1] - radii[i] < 0 or x[i*2 + 1] + radii[i] > 1:

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
