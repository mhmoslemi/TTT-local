import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
from typing import Tuple, List, Optional

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"


def initialize_centers(num_circles):
    """Initialize positions of circles in a grid-like pattern."""
    grid_size = int(np.ceil(np.sqrt(num_circles)))
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // grid_size
        col = i % grid_size
        x = col / (grid_size - 1)
        y = row / (grid_size - 1)
        centers[i] = [x, y]
    return centers


def objective_function(x, num_circles):
    """Objective function to maximize the sum of radii."""
    # x is a flattened array of [x1, y1, r1, x2, y2, r2, ...]
    centers = x[:num_circles * 2].reshape(num_circles, 2)
    radii = x[num_circles * 2:].reshape(num_circles)
    return -np.sum(radii)  # We minimize the negative sum (i.e., maximize the sum)


def constraint_boundary(x, num_circles):
    centers = x[:num_circles * 2].reshape(num_circles, 2)
    radii = x[num_circles * 2:].reshape(num_circles)
    for i in range(num_circles):
        x_i, y_i = centers[i]
        r_i = radii[i]
        if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
            return -1e6
        if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
            return -1e6
    return 0


def constraint_overlap(x, num_circles):
    centers = x[:num_circles * 2].reshape(num_circles, 2)
    radii = x[num_circles * 2:].reshape(num_circles)
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return -1e6
    return 0


def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    num_circles = 26
    centers = initialize_centers(num_circles)
    radii = np.full(num_circles, 0.01)  # Initial small radii to ensure starting point is valid

    # Flatten the centers and radii for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Define bounds for centers and radii
    bounds = []
    for i in range(num_circles):
        bounds += [(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)]  # x, y, r for each circle

    # Define constraints
    constraints = [
        {"type": "ineq", "fun": constraint_boundary, "args": (num_circles,)},
        {"type": "ineq", "fun": constraint_overlap, "args": (num_circles,)},

        x0=x0,
        method="SLSQP",
        bounds=bounds,
        constraints=constraints,

    return centers_opt, radii_opt, sum_radii
