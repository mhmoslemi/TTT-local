import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Number of circles
    n = 26

    # Initial guess: uniform grid placement
    grid_size = 5  # Number of rows and columns
    centers = np.zeros((n, 2))
    idx = 0
    for i in range(grid_size):
        for j in range(grid_size):
            if idx < n:
                centers[idx, 0] = j / (grid_size - 1)
                centers[idx, 1] = i / (grid_size - 1)
                idx += 1
            else:
                break

    # Initial uniform radius
    initial_radius = 0.05
    radii = np.full(n, initial_radius)

    # Objective function to maximize radius (we'll minimize -sum(radii))
    def objective(x):
        # x contains [radii_0, radii_1, ..., radii_25]
        radii_flat = x
        # Ensure non-negative radii
        if np.any(radii_flat < 0):
            return np.inf
        return -np.sum(radii_flat)

    # Bounds for radii: all radii >= 0
    bounds = [(0, np.inf) for _ in range(n)]

    # Constraint: centers must lie within [0, 1]^2
    def center_bounds(x):
        # x contains [x0, y0, x1, y1, ..., x25, y25]
        x_flat = x.reshape(-1, 2)
        return np.array([x_flat[i, 0] - 1e-12 for i in range(n)] + [x_flat[i, 1] - 1e-12 for i in range(n)])

    # Constraint: distance between centers >= sum of radii for each pair
    def non_overlap_constraint(x):
        x_flat = x.reshape(-1, 2)
        radii_flat = x[::2]  # Extract radii from x (every other element)
        centers_flat = x_flat
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii_flat[i] + radii_flat[j]) + 1e-12)
        return np.array(constraints)

    # Combine constraints: center bounds and non-overlap
    constraints = (
        {'type': 'ineq', 'fun': center_bounds},
        {'type': 'ineq', 'fun': non_overlap_constraint}
    )

    x0 = np.concatenate([centers.reshape(-1), radii])

    result = optimize.minimize(

        x0,
        bounds=bounds,
        constraints=constraints,
        method='L-BFGS-B',
        tol=1e-8

    # Extract the best solution
    best_x = result.x
    centers_opt = best_x[:n*2].reshape(-1, 2)
    radii_opt = best_x[n*2:]

    # Compute the total sum of radii
    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
