import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial guess: Hexagonal grid-based placement
    n = 26
    # We'll spread the circles in two rows for a hexagonal-like arrangement
    # Adjust the grid to fit in the square while maximizing radii
    # Start with a hexagonal grid of 2 rows and 13 columns
    # This gives a rough initial arrangement

    # Define the positions of centers
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // 13
        col = i % 13
        x = col / 12.0  # spread horizontally
        y = row / 2.0   # spread vertically
        centers[i] = [x, y]

    # Initial radii - assign equal radii for simplicity
    radii = np.ones(n) * 0.1  # initial guess

    # Objective function: maximize the sum of radii
    # We'll minimize negative sum of radii

    def objective(radii):
        return -np.sum(radii)

    # Constraints:
    # 1. Circle must be within the unit square
    def constraint_circle_inside(radii, centers):
        return np.sum((centers - radii[:, np.newaxis]) ** 2 <= 1.0 - 2 * radii[:, np.newaxis] + radii[:, np.newaxis] ** 2)

    # 2. Distance between centers must be ≥ sum of radii
    def constraint_no_overlap(centers, radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Define bounds for radii (must be non-negative)
    bounds = [(0.0, 1.0) for _ in range(n)]

    # Use a constrained optimization method
    result = optimize.minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=[
            {"type": "ineq", "fun": lambda r: constraint_circle_inside(r, centers)},
            {"type": "ineq", "fun": lambda r: constraint_no_overlap(centers, r)}

        method="SLSQP"

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
