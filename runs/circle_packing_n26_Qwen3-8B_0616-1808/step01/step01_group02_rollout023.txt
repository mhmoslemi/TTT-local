import numpy as np
import scipy.optimize as optimize

def run_packing():

    # For 26 circles, use a hexagonal packing with some circles placed near borders
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Hexagonal grid in the center
    # We'll place 6 circles in a hexagonal arrangement, then add more in a grid
    # This is a rough initial placement; actual optimization will refine positions
    grid_radius = 0.25
    center = np.array([0.5, 0.5])

        angle = 2 * np.pi * i / num_circles
        radius = grid_radius * (i // 6 + 1)
        x = center[0] + radius * np.cos(angle)
        y = center[1] + radius * np.sin(angle)
        centers[i] = np.array([x, y])
        radii[i] = 0.05 + 0.1 * (i // 6)

    # Define objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraints
    def constraints(x):
        centers[:] = x[:2*num_circles].reshape(num_circles, 2)
        radii[:] = x[2*num_circles:]
        # Check if centers are in [0,1]^2
        if np.any(centers < -1e-6) or np.any(centers > 1 + 1e-6):
            return np.inf
        # Check if radii are non-negative
        if np.any(radii < -1e-6):
            return np.inf
        # Check for overlaps

                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)

                    return np.inf

            x, y = centers[i]
            r = radii[i]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
