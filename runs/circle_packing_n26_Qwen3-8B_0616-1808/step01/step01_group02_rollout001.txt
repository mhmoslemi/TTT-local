import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26
    centers = np.random.rand(n, 2)
    radii = np.random.rand(n) * 0.1  # Small initial radii

    # Define objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Define constraints
    def constraint_center_in_square(centers):
        return np.all((centers >= 0) & (centers <= 1))

    def constraint_radius_nonnegative(radii):
        return np.all(radii >= 0)

    def constraint_no_overlap(centers, radii):
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

    # Optimization
    bounds = [(0, 1) for _ in range(2 * n)]  # x and y for each center
    bounds += [(0, 0.5) for _ in range(n)]  # radii bounds

    result = opt.minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method="SLSQP",
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraint_center_in_square(x[:2*n].reshape(n, 2))},
            {'type': 'ineq', 'fun': lambda x: constraint_radius_nonnegative(x[2*n:])},
            {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x[:2*n].reshape(n, 2), x[2*n:])}
        ]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
