import numpy as np
import cvxpy as cp
import math

def generate_hexagonal_initial_layout(n, square_size=1.0):
    """
    Generate an initial hexagonal packing of n circles in a square.
    """
    # Determine number of rows in hexagon
    rows = math.ceil(math.sqrt(n))
    cols = math.ceil(n / rows)
    
    # Calculate circle spacing to fit into square
    spacing = square_size / (cols - 1)
    radius = spacing / (2 * math.sqrt(3))  # Approximate radius for hexagonal packing
    
    centers = []
    for row in range(rows):
        for col in range(cols):
            x = col * spacing
            y = row * (spacing * math.sqrt(3) / 2)
            if row % 2 == 1:
                x += spacing / 2
            centers.append([x, y])
    
    # Trim or pad to get exactly n circles
    centers = centers[:n]
    if len(centers) < n:
        # Add extra circles in the last row
        for i in range(len(centers), n):
            x = i * spacing
            y = (rows - 1) * (spacing * math.sqrt(3) / 2)
            if (rows - 1) % 2 == 1:
                x += spacing / 2
            centers.append([x, y])
    
    return np.array(centers), np.full(n, radius)

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers, initial_radii = generate_hexagonal_initial_layout(n)
    
    # Define CVXPY variables
    radii = cp.Variable(n)
    centers = cp.Variable((n, 2))
    
    # Objective: Maximize the sum of radii
    objective = cp.Maximize(cp.sum(radii))
    
    # Constraints
    constraints = []
    
    # Ensure circles are inside the unit square
    for i in range(n):
        constraints.append(centers[i, 0] >= radii[i])
        constraints.append(centers[i, 0] <= 1 - radii[i])
        constraints.append(centers[i, 1] >= radii[i])
        constraints.append(centers[i, 1] <= 1 - radii[i])
    
    # Ensure circles do not overlap
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2)
            constraints.append(dist >= radii[i] + radii[j])
    
    # Add initial guesses to the problem
    for i in range(n):
        constraints.append(centers[i, 0] == initial_centers[i, 0])
        constraints.append(centers[i, 1] == initial_centers[i, 1])
        constraints.append(radii[i] == initial_radii[i])
    
    # Solve the optimization problem
    prob = cp.Problem(objective, constraints)
    prob.solve(solver=cp.SCS, max_iters=10000, eps=1e-6)
    
    if prob.status != 'optimal':
        print("Optimization failed")
        return np.zeros((n, 2)), np.zeros(n), 0.0
    
    # Extract results
    final_centers = centers.value
    final_radii = radii.value
    sum_radii = np.sum(final_radii)
    
    # Validate the packing
    is_valid, msg = validate_packing(final_centers, final_radii)
    if not is_valid:
        print(f"Packaging invalid: {msg}")
        return np.zeros((n, 2)), np.zeros(n), 0.0
    
    # Ensure all values are non-negative
    final_radii = np.maximum(final_radii, 0)
    
    return final_centers, final_radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
