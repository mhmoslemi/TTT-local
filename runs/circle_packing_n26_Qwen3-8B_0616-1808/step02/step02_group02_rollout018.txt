import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Initial grid-based packing (starting configuration)
    # Let's use a hexagonal grid with 5 rows of 5, 5, 5, 5, and 6 circles (approximate)
    # This is a rough setup to give a good initial guess
    rows = 5
    cols = 5
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Starting with regular grid, then perturb later
    for i in range(n):
        r = i // cols
        c = i % cols
        x = c / (cols - 1)
        y = r / (rows - 1)
        centers[i] = [x, y]

    # Initial guess for radii (based on spacing between neighboring circles)
    def compute_initial_radii(centers):
        d = np.min(np.linalg.norm(centers[i] - centers[j], axis=1) for i in range(n) for j in range(i+1, n))
        return np.full(n, d / 2)

    radii = compute_initial_radii(centers)

    # Objective function (to maximize sum of radii)
    def objective(radii):
        return -np.sum(radii)  # Minimize the negative sum

    # Constraints: non-overlap and boundary
    def constraint_overlap(i, j, radii):
        def cons(radii):
            d = np.linalg.norm(centers[i] - centers[j])
            return d - (radii[i] + radii[j])
        return cons

    def constraint_boundary(i, radii):
        def cons(radii):
            x, y = centers[i]
            r = radii[i]
            return (x - r) - 0.0  # x - r >= 0
        return cons

    def constraint_boundary_right(i, radii):
        def cons(radii):
            x, y = centers[i]
            r = radii[i]
            return (1.0 - x - r) - 0.0  # 1 - x - r >= 0
        return cons

    def constraint_boundary_top(i, radii):
        def cons(radii):
            x, y = centers[i]
            r = radii[i]
            return (1.0 - y - r) - 0.0  # 1 - y - r >= 0

    def constraint_boundary_bottom(i, radii):

            r = radii[i]

    for i in range(n):

    # Optimization
    result = opt.minimize(

        method='SLSQP',
