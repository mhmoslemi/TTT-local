import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import minimize, Bounds
from scipy.spatial.distance import cdist
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: grid-like arrangement for 26 circles
    # This is a grid of 5x6 (with one column or row missing) to fit in a unit square
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.05  # initial small radii

    # Initial grid-like arrangement
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col / 4.5
        centers[i, 1] = row / 4.5

    # Define objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Define constraints: boundary and non-overlap
    def constraint_boundary(radii):
        return np.sum((centers[:, 0] - radii) >= 0) + np.sum((1 - centers[:, 0] - radii) >= 0) + \
               np.sum((centers[:, 1] - radii) >= 0) + np.sum((1 - centers[:, 1] - radii) >= 0)

    def constraint_overlap(radii):
        dist = cdist(centers, centers)
        min_dist = np.min(dist[np.triu_indices_from(dist, k=1)])
        min_required_dist = np.min(radii[np.triu_indices_from(radii, k=1)] + radii[np.tril_indices_from(radii, k=-1)])
        return min_dist - min_required_dist

    # Optimize using constrained optimization
    bounds = Bounds(0, 1.0)  # radii between 0 and 1

    # Initial guess for radii
    radii_guess = np.ones(num_circles) * 0.05

    # Run optimization
    result = minimize(
        objective,
        radii_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_boundary},
            {'type': 'ineq', 'fun': constraint_overlap}
        ],
        options={'disp': False, 'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract result
    radii = result.x
    centers = centers
    sum_radii = np.sum(radii)

    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
