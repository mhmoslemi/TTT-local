import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Number of circles
    n = 26

    # Generate initial hexagonal grid
    # We will approximate a hexagonal packing in a unit square
    # Let's place 5 rows, with decreasing number of circles per row
    rows = 5
    cols_per_row = [6, 5, 5, 5, 5]  # Total of 26 circles

    # Initial grid layout
    centers = np.zeros((n, 2))
    radius_initial = 0.1

    # Define hexagonal grid coordinates (equilateral triangle spacing)
    # Side length of equilateral triangle: s = 2 * radius_initial
    # Horizontal spacing: s * sqrt(3)/2
    # Vertical spacing: s
    h_spacing = 2 * radius_initial * np.sqrt(3)/2
    v_spacing = 2 * radius_initial

    idx = 0
    for row in range(rows):
        for col in range(cols_per_row[row]):
            x = col * h_spacing
            y = row * v_spacing
            centers[idx, :] = [x, y]
            idx += 1
    
    # Normalize centers to fit within [0,1]^2
    centers /= np.max(centers)

    # Initial guess for radii
    radii = np.full(n, radius_initial)

    # Optimization setup
    # Define objective function to minimize (-sum of radii)
    def objective(x):
        centers_flat = x[:2 * n].reshape((n, 2))
        radii_flat = x[2 * n:]
        return -np.sum(radii_flat)

    # Define constraints
    # 1. Boundary constraints: each circle must be within [0,1]^2
    def boundary_constraints(x):
        centers_flat = x[:2 * n].reshape((n, 2))
        radii_flat = x[2 * n:]
        return np.concatenate([
            centers_flat[:, 0] - radii_flat,
            1 - centers_flat[:, 0] - radii_flat,
            centers_flat[:, 1] - radii_flat,
            1 - centers_flat[:, 1] - radii_flat
        ])

    # 2. Non-overlap constraints: distance between centers >= sum of radii
    def non_overlap_constraints(x):
        centers_flat = x[:2 * n].reshape((n, 2))
        radii_flat = x[2 * n:]
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii_flat[i] + radii_flat[j]))
        return np.array(constraints)

    # Optimization bounds
    bounds = []
    for _ in range(2 * n):
        bounds.append((0.0, 1.0))  # x and y coordinates
    for _ in range(n):
        bounds.append((0.0, 1.0))  # radii

    # Initial guess
    x0 = np.concatenate([
        centers.flatten(),
        radii
    ])

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': boundary_constraints},
            {'type': 'ineq', 'fun': non_overlap_constraints}
        ],
        method='L-BFGS-B',
        tol=1e-6
    )

    if not result.success:
        raise ValueError("Optimization failed: " + result.message)

    # Extract results
    centers_opt = result.x[:2 * n].reshape((n, 2))
    radii_opt = result.x[2 * n:]
    sum_radii_opt = np.sum(radii_opt)

    # Validate the result
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        raise ValueError("Validation failed: " + msg)

    return centers_opt, radii_opt, sum_radii_opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
