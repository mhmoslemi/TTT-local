import numpy as np
import scipy.optimize as opt
from scipy.spatial.distance import cdist

def run_packing():
    # Initial base configuration: staggered rows of 5 circles each
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initial grid
    rows = 5
    cols = 5
    gaps = 0.2  # gap between rows and columns
    for i in range(rows):
        for j in range(cols):
            # Create staggered layout
            if i % 2 == 0:
                x = j * (1.0 / (cols - 1)) + gaps
            else:
                x = j * (1.0 / (cols - 1)) + gaps + 0.05
            y = i * (1.0 / (rows - 1)) + gaps
            idx = i * cols + j
            if idx < n:
                centers[idx] = [x, y]
                # Initialize radii to fit in grid
                radii[idx] = 0.1

    # Now we will optimize the radii and positions
    # Objective: maximize sum of radii
    def objective(x):
        return -x  # minimize negative of sum of radii

    # Constraints: centers must be within square [0,1]^2
    bounds = [(0, 1) for _ in range(2 * n)]  # 2*n variables: 2 per center

    # Constraints: circle radii must be non-negative
    def non_negative(x):
        return x[n:]  # radii are the last n variables

    # Constraints: distance between centers must be ≥ sum of radii
    def no_overlap(x):
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        dists = cdist(centers_flat, centers_flat)
        radii_sum = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        return dists - radii_sum + 1e-8  # allow small tolerance

    # Constraint: centers must be within [0,1]^2
    def in_bounds(x):
        return x[:2*n] - np.array([0, 0])  # just ensuring within [0,1]

    # Constraint: radii ≥ 0
    def non_neg_radii(x):
        return x[2*n:]

    # Constraint: circles must not exceed the square
    def in_square(x):
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        return np.array([
            centers_flat[i, 0] - radii_flat[i] >= -1e-8 for i in range(n)
        ]) + np.array([
            centers_flat[i, 0] + radii_flat[i] <= 1 + 1e-8 for i in range(n)
        ]) + np.array([
            centers_flat[i, 1] - radii_flat[i] >= -1e-8 for i in range(n)
        ]) + np.array([
            centers_flat[i, 1] + radii_flat[i] <= 1 + 1e-8 for i in range(n)
        ])

    # Combine all constraints
    constraints = [
        {'type': 'ineq', 'fun': in_bounds},
        {'type': 'ineq', 'fun': non_neg_radii},
        {'type': 'ineq', 'fun': in_square},
        {'type': 'ineq', 'fun': no_overlap}
    ]

    x0 = np.concatenate((centers.flatten(), radii))

    result = opt.minimize(

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
