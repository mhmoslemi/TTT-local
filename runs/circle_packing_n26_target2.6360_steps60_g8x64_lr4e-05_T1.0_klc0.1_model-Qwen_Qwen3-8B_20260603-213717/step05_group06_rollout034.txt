import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Generate a hexagonal grid of points that fits well in the unit square
    grid_points = []
    num_rows = 5
    num_cols = 5
    for row in range(num_rows):
        for col in range(num_cols):
            x = col / (num_cols - 1)
            y = row / (num_rows - 1)
            if row % 2 == 1:
                y += 0.5 / (num_rows - 1)  # offset for staggered grid
            grid_points.append((x, y))
    
    # Add boundary and corner circles
    boundary_points = []
    for i in range(2):
        for j in range(2):
            boundary_points.append((i * 0.25, j * 0.25))

    all_points = grid_points + boundary_points

    radii = np.ones(n) * 0.1

        return -np.sum(radii_vec)  # minimize negative sum to maximize sum

    centers_vec = centers.flatten()
    radii_vec = radii.flatten()

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):

                            centers[i][1] - centers[j][1])
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
