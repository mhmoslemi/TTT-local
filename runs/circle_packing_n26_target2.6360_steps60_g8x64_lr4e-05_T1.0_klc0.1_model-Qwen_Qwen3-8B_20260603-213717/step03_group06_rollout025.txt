import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26

    centers = np.random.rand(n, 2)

    # Define initial radii (start with a small value)
    radii = np.full(n, 0.05)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints function: ensure non-overlapping and stay within the square
    def constraints(radii_vec, centers):
        radii_reshaped = radii_vec

        # Non-overlapping constraints: dist(c_i, c_j) >= r_i + r_j
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraint = dist - radii_reshaped[i] - radii_reshaped[j]
                constraints.append(constraint)

        # Boundary constraints: x - r >= 0, x + r <= 1, same for y
        for i in range(n):
            x, y = centers[i]
            r = radii_reshaped[i]
            constraints.append(x - r)
            constraints.append(1 - x - r)
            constraints.append(y - r)
            constraints.append(1 - y - r)

        return np.array(constraints)

    # Initial guess
    initial_guess = np.full(n, 0.05)

        x0=initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': lambda r: constraints(r, centers)},

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
