import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26

    grid_points = []
    # Define grid size for hexagonal packing
    cols = 5
    rows = 5
    offset_y = 1 / (2 * (rows - 1))  # Adjust for vertical spacing in hex grid
    for row in range(rows):
        for col in range(cols):
            x = col / (cols - 1)
            y = row / (rows - 1)
            if row % 2 == 1:
                y += offset_y
            grid_points.append((x, y))
    
    # Add boundary and corner circles
    boundary_points = []
    for i in range(2):
        for j in range(2):
            x = i * 0.25
            y = j * 0.25

            boundary_points.append((x, y))

    unique_points = list(set(grid_points + boundary_points))
    unique_points = unique_points[:n]

    radii = np.ones(n) * 0.1

        return -np.sum(radii_vec)  # minimize negative sum to maximize sum

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.hypot(centers[i][0] - centers[j][0],
                            centers[i][1] - centers[j][1])
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
