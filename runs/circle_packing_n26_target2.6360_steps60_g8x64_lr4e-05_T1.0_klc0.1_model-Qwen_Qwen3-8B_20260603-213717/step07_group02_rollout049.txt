import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    grid_points = []

            y = j / 4.0 + (i % 2) * 0.05
            grid_points.append((x, y))

    boundary_points = [
        (0.1, 0.1), (0.1, 0.9), (0.9, 0.1), (0.9, 0.9),

    ]
    grid_points += boundary_points

    adjusted_centers = []
    for x, y in grid_points:

        adjusted_centers.append((x, y))
    
    # Remove duplicates and limit to 26 points

    centers = np.array(adjusted_centers)

    def objective(vars):

        radius_vars = vars[2 * n:]
        return -np.sum(radius_vars)

    # Constraints to enforce non-overlapping and boundary conditions
    def constraints(vars):
        center_vars = vars[:2 * n].reshape(n, 2)
        radius_vars = vars[2 * n:]
        constraints = []
        # Non-overlapping constraint
        for i in range(n):
            for j in range(i + 1, n):
                dx = center_vars[i, 0] - center_vars[j, 0]
                dy = center_vars[i, 1] - center_vars[j, 1]

    result = opt.minimize(

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.hypot(dx, dy)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
