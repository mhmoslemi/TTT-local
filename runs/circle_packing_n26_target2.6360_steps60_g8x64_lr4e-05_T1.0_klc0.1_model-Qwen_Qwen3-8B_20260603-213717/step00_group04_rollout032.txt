import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    rows = 5  # Number of rows
    cols = 6  # Number of columns
    spacing = 1.0 / (cols - 1)  # Space between centers in x-direction
    spacing_y = np.sqrt(3) / 2 * spacing  # Space between centers in y-direction for hex grid
    centers = np.zeros((n_circles, 2))
    idx = 0
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                x = j * spacing
            else:
                x = (j + 0.5) * spacing
            y = i * spacing_y
            centers[idx, 0], centers[idx, 1] = x, y
            idx += 1

    # Initial guess for radii (based on spacing)
    radii = np.full(n_circles, 0.05)

    # Objective function: Maximize the sum of radii
    def objective(x):
        return -np.sum(x[n_circles:])  # Return negative to minimize

    # Constraints function: Ensures non-overlapping and boundary condition
    def constraints(x):
        radii = x[n_circles:]
        positions = x[:n_circles * 2].reshape((n_circles, 2))
        constraints = []

        # Boundary constraints
        for i in range(n_circles):
            x_i, y_i = positions[i]
            r_i = radii[i]
            if x_i - r_i < -tol or x_i + r_i > 1 + tol:
