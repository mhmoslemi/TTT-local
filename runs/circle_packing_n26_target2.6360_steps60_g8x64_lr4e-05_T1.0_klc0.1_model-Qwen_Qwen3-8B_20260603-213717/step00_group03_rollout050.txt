import numpy as np
import matplotlib.pyplot as plt
import scipy.optimize as optimize

def run_packing():
    # Initial placement using a hexagonal packing pattern
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(np.sqrt(n)))
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    radius_initial = 0.05  # Initial guess for radius

    # Create initial hexagonal packing in a square
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * (1 + 1e-3) + 0.5 * (1 - 1e-3) * (row % 2)
        y = row * (np.sqrt(3) * (1 + 1e-3)) / 2
        centers[i] = [x, y]
        radii[i] = radius_initial

    # Define objective function to maximize sum of radii
    def objective(x):
        # x is a vector of shape (n*2 + n) = (52,)
        # First n*2 elements are centers
        # Next n elements are radii
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        # To maximize sum of radii, minimize negative of sum
        return -np.sum(radii_flat)

    # Define constraints
    def constraint_centers(x):
        centers_flat = x[:n*2].reshape(n, 2)
        return np.r_[centers_flat[:, 0], centers_flat[:, 1]]

    def constraint_radii(x):
        radii_flat = x[n*2:].reshape(n)
        return radii_flat

    def constraint_bounds(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        # Enforce boundaries
        boundary_violations = np.sum(
            (centers_flat[:, 0] - radii_flat < 0) + 
            (centers_flat[:, 0] + radii_flat > 1) + 
            (centers_flat[:, 1] - radii_flat < 0) + 
            (centers_flat[:, 1] + radii_flat > 1)
        )
        return boundary_violations

    def constraint_overlaps(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        overlaps = 0
        for i in range(n):
            for j in range(i+1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-6:
                    overlaps += 1
        return overlaps

    # Define bounds for all variables
    bounds = []
    for _ in range(n*2 + n):
        bounds.append((0, 1))

    # Optimization setup
    x0 = np.r_[centers.flatten(), radii]
    result = optimize.minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=[

        maxiter=1000

    centers_opt = result.x[:n*2].reshape(n, 2)

    sum_radii = np.sum(radii_opt)

    # Validate the solution
    valid, msg = validate_packing(centers_opt, radii_opt)

    # Return the result
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
