import numpy as np
import scipy.optimize as optimize

def run_packing():

    num_circles = 26
    rows = 5
    cols_per_row = [5, 6, 6, 6, 3]
    
    # Initial spacing based on hexagonal packing
    radius_initial = 0.15  # Initial estimate for radius

    # Initialize centers based on hexagonal packing
    centers = []
    for row_idx in range(rows):
        row_centers = []
        for col_idx in range(cols_per_row[row_idx]):
            col = col_idx
            row = row_idx + (col // 2) * 0.5  # Staggered rows
            x = col * (1.0 / (cols_per_row[row_idx] - 1)) + 0.5 * (1.0 / (cols_per_row[row_idx] - 1)) if cols_per_row[row_idx] > 1 else 0.5
            y = row * (1.0 / (rows - 1)) + 0.5 * (1.0 / (rows - 1)) if rows > 1 else 0.5
            row_centers.append((x, y))
        centers.extend(row_centers)
    centers = np.array(centers)
    
    # Initialize radii
    radii = np.full(num_circles, radius_initial)
    
    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Negative for minimization

    # 1. Each circle must be within the unit square
    # 2. Distance between centers must be at least sum of radii

    def constraint_boundary(x, i):
        r = x[i + num_circles]
        x_center = x[i]
        y_center = x[i + num_circles + 1]
        if x_center - r < 0 or x_center + r > 1 or y_center - r < 0 or y_center + r > 1:
            return x_center - r + 1e-12  # Enforces x_center - r >= 0
        return 0

    def constraint_overlap(x, i, j):
        r_i = x[i + num_circles]
        r_j = x[j + num_circles]

        dy = x[i + num_circles + 1] - x[j + num_circles + 1]

            return dist - (r_i + r_j) + 1e-12  # Enforces dist >= r_i + r_j
        return 0

    x0 = np.concatenate([centers.flatten(), radii])

    constraints = []

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
