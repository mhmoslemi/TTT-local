import scipy.optimize as optimize

def run_packing():

    n_circles = 26

    col_count = 5
    spacing = 0.5  # Initial spacing between centers

    centers = []
    radii = []

        for col in range(col_count):
            x = col * spacing + (spacing / 2) * (row % 2)
            y = row * spacing * np.sqrt(3) / 2
            centers.append([x, y])
            radii.append(0.02)

    centers = np.array(centers)
    radii = np.array(radii)

    # Optimization variables
    x = np.concatenate([centers.flatten(), radii])

    def objective(x_flat):

        return -np.sum(radii)  # Negative for minimization

    def constraints(x_flat):
        centers = x_flat[:2 * n_circles].reshape(n_circles, 2)
        radii = x_flat[2 * n_circles:]
        constraints = []

        # Boundary constraints
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r >= -1e-12) * 1.0)
            constraints.append((x + r <= 1 + 1e-12) * 1.0)
