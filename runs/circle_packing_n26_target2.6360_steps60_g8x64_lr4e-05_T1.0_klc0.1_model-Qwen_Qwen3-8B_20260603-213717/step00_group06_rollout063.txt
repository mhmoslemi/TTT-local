import numpy as np
import scipy.optimize as opt
from scipy.spatial import cdist

def run_packing():
    # Initial guess for centers using hexagonal packing pattern
    num_circles = 26
    rows = 5
    cols = 6
    # Offset for hexagonal packing
    hex_offset = 0.5 * np.sqrt(3)
    centers = np.zeros((num_circles, 2))
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 1:
                centers[i * cols + j, 0] = j * (1.0 / (cols - 1)) + 0.5 / (cols - 1)
            else:
                centers[i * cols + j, 0] = j * (1.0 / (cols - 1))
            centers[i * cols + j, 1] = i * hex_offset / (rows - 1)

    # Initial guess for radii: all same radius
    radii = np.full(num_circles, 1.0 / (rows * cols))

    # Objective function to maximize (sum of radii)
    def objective(radii):
        return -np.sum(radii)

    # Constraints for boundary limits
    def boundary_constraint(radii):
        return np.array([radii[i] <= 1.0 - centers[i, 0] for i in range(num_circles)] +
                        [radii[i] <= 1.0 - centers[i, 1] for i in range(num_circles)] +
                        [radii[i] <= centers[i, 0] for i in range(num_circles)] +
                        [radii[i] <= centers[i, 1] for i in range(num_circles)])

    # Constraints for circle non-overlapping
    def non_overlap_constraint(radii):
        dists = cdist(centers, centers)
        return np.array([dists[i, j] >= radii[i] + radii[j] - 1e-8 for i in range(num_circles) for j in range(i + 1, num_circles)])

    # Optimization
    cons = [
        {'type': 'ineq', 'fun': boundary_constraint},
        {'type': 'ineq', 'fun': non_overlap_constraint}
    ]
    bounds = [(0, 1.0) for _ in range(num_circles)]

    result = opt.minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Extract the best solution
    best_radii = result.x
    best_centers = centers.copy()
    best_sum_radii = np.sum(best_radii)

    if not is_valid:

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
