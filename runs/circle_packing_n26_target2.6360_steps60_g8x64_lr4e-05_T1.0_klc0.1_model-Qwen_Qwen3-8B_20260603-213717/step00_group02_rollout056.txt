import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial configuration: hexagonal packing
    num_circles = 26
    num_rows = int(np.ceil(np.sqrt(num_circles)))
    num_cols = int(np.ceil(num_circles / num_rows))
    
    # Hexagonal grid: spacing between centers
    horizontal_spacing = 1.0 / (num_cols - 1)
    vertical_spacing = np.sqrt(3) / (2 * (num_rows - 1))
    
    # Initial centers
    centers = np.zeros((num_circles, 2))
    idx = 0
    for row in range(num_rows):
        for col in range(num_cols):
            if row % 2 == 0:
                x = col * horizontal_spacing
            else:
                x = (col - 0.5) * horizontal_spacing
            y = row * vertical_spacing
            if y + vertical_spacing > 1.0 - 1e-8:
                y = 1.0 - 1e-8
            centers[idx] = [x, y]
            idx += 1
    
    # Initial radii
    radii = np.full(num_circles, 1.0 / (num_rows * num_cols))
    
    # Optimization parameters
    bounds = []
    for i in range(num_circles):
        bounds.append((0.0, 1.0 - 1e-8))  # Radii must be nonnegative and fit in square
    
    # Define objective function: maximize sum of radii
    def objective(x):
        # x contains all radii, reshape to get the radii array
        radii = x.reshape(num_circles)
        if np.any(radii < 0):
            return -np.inf
        # Compute total area
        total_area = np.sum(np.pi * radii**2)
        return total_area
    
    # Optimization constraints: ensure circles are within bounds and not overlapping
    def constraint(x):
        radii = x.reshape(num_circles)
        # Check if any circle is outside the square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < 0 - 1e-8 or x_i + r_i > 1.0 + 1e-8:
                return -np.inf
            if y_i - r_i < 0 - 1e-8 or y_i + r_i > 1.0 + 1e-8:
                return -np.inf
        # Check for overlaps between any two circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -np.inf
        return 0
    
    # Optimization: maximize total area
    res = optimize.minimize(
        lambda x: -objective(x),
        x0=radii,
        bounds=bounds,
        constraints=[{"type": "ineq", "fun": constraint}]
    )
    
    # Extract final result
    final_radii = res.x
    final_centers = centers.copy()
    final_sum_radii = np.sum(final_radii)
    
    return final_centers, final_radii, final_sum_radii
