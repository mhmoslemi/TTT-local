import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    def generate_initial_centers(num_circles):

        centers = np.zeros((num_circles, 2))

            row = i // num_cols

            x = col / (num_cols - 1) if num_cols > 1 else 0.5
            y = row / (num_rows - 1) if num_rows > 1 else 0.5
            centers[i] = [x, y]
        return centers

    radii = np.full(n, 0.15)

    # Objective function to maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    def constraints(x):
        # Extract radii
        radii = x
        constraints = []

        # Non-overlapping constraints: dist(c_i, c_j) >= r_i + r_j
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))

        # Boundary constraints: x - r >= 0, x + r <= 1, same for y
        for i in range(n):
            x_center, y_center = centers[i]
            r = radii[i]
            constraints.append(x_center - r)
            constraints.append(1 - x_center - r)
            constraints.append(y_center - r)
            constraints.append(1 - y_center - r)

        method='COBYLA',
        bounds=bounds,

        tol=1e-8,
        options={'disp': False}

    def validate_packing(centers, radii):

    if not valid:

    return best_centers, best_radii, sum_radii
