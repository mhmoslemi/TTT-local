def generate_hexagonal_grid(num_circles: int, radius: float) -> np.ndarray:

    num_rows = int(np.ceil(np.sqrt(2 * num_circles)))
    num_cols = int(np.ceil(num_circles / num_rows))
    
    # Adjust for even/odd rows to account for hexagonal structure
    if num_rows % 2 == 0:
        num_cols = num_cols + 1
    elif num_rows % 2 == 1:
        num_cols = num_cols

                y += radius

                centers[idx] = [x, y]
                idx += 1
    return centers

def boundary_constraints(radii: np.ndarray, centers: np.ndarray) -> np.ndarray:

        constraints.append((1 + 1e-12) - y - r)  # y + r <= 1 + 1e-12
    return np.array(constraints)

def non_overlap_constraints(radii: np.ndarray, centers: np.ndarray) -> np.ndarray:
