import scipy.optimize as opt
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    radii = np.zeros(num_circles)

    # Step 1: Generate an efficient and boundary-respecting initial configuration
    # Place some circles near the corners and edges for better utilization
    # Also use a hexagonal-like pattern in the interior
    spacing = 0.3
    grid_size = int(np.ceil(np.sqrt(num_circles)))
    for i in range(num_circles):
        row = i // grid_size
        col = i % grid_size
        x = col * spacing + spacing / 2
        y = row * spacing * np.sqrt(3) / 2 + spacing / 2
        # Adjust for boundaries
        if x < 0.1:
            x = 0.1
        if x > 0.9:
            x = 0.9
        if y < 0.1:
            y = 0.1
        if y > 0.9:
            y = 0.9
        centers[i] = [x, y]
        radii[i] = 0.05  # Initialize small radius

    # Step 2: Define objective function with constraints
    def objective(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:].reshape(num_circles)

        # Check circles are within the square
        for i in range(num_circles):
            if not (0 <= centers_flat[i][0] - radii_flat[i] and centers_flat[i][0] + radii_flat[i] <= 1):
                return -np.inf
            if not (0 <= centers_flat[i][1] - radii_flat[i] and centers_flat[i][1] + radii_flat[i] <= 1):
                return -np.inf
            if radii_flat[i] < 0:
                return -np.inf

        # Check circles are not overlapping
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i][0] - centers_flat[j][0]

                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -np.inf

        # Return negative of sum of radii to minimize
        return -np.sum(radii_flat)

    # Step 3: Define bounds and initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

        bounds.append((0, 1))

    result = opt.minimize(

    centers_opt = result.x[:num_circles * 2].reshape(num_circles, 2)
