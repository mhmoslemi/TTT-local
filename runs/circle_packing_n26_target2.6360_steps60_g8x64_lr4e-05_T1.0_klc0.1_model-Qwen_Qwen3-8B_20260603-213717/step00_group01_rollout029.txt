import scipy.optimize as optimize

def run_packing():

    n = 26

    # Create initial hexagonal packing configuration
    def hexagonal_centers(n):
        num_rows = int(np.ceil(np.sqrt(n)))
        num_cols = int(np.ceil(n / num_rows))
        centers = []
        for i in range(num_rows):
            for j in range(num_cols):
                x = j + 0.5 * (i % 2)
                y = i * np.sqrt(3) / 2
                if x + 0.5 < 1 and y + 0.5 < 1:
                    centers.append([x, y])
                if len(centers) == n:
                    break
            if len(centers) == n:
                break
        return np.array(centers)

    def constraint_boundary(x, centers, radii):
        for i in range(n):
            x_center, y_center = centers[i]
            r = x[i]
            if x_center - r < -1e-12 or x_center + r > 1 + 1e-12:
                return -1.0
            if y_center - r < -1e-12 or y_center + r > 1 + 1e-12:
                return -1.0
        return 0.0

    def constraint_overlap(x, centers, radii):

            for j in range(i + 1, n):

                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < x[i] + x[j] - 1e-12:

    ]

    result = optimize.minimize(

        x0,
        bounds=bounds,
        constraints=cons,
