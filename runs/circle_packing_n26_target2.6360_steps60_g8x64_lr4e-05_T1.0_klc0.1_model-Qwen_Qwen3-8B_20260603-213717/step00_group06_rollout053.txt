import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
import matplotlib.pyplot as plt

def run_packing():
    # Number of circles
    n_circles = 26
    # Initial grid: hexagonal packing
    rows = 6
    cols = 5
    spacing = 1.0 / (cols + 1)  # Horizontal spacing
    vert_spacing = np.sqrt(3) / 2 * spacing  # Vertical spacing for hexagonal packing
    centers = np.zeros((n_circles, 2))
    for i in range(rows):
        for j in range(cols):
            x = (j + 0.5) * spacing
            y = (i + (j % 2)) * vert_spacing
            if y > 1.0:
                y = 1.0 - (y - 1.0) % 1.0
            centers[i * cols + j] = [x, y]

    # Initial radii based on spacing
    initial_radii = np.full(n_circles, spacing / 2)

    def objective(x):
        radii = x[:n_circles]
        return -np.sum(radii)

    def constraint_circle_boundary(x):
        radii = x[:n_circles]
        centers = x[n_circles:]
        centers = centers.reshape(n_circles, 2)

            r_i = radii[i]

                return -1
        return 0

    def constraint_circle_overlap(x):

                dist = np.sqrt(dx**2 + dy**2)
                if dist < r_i + r_j - 1e-12:

        return 0

    bounds = Bounds(np.zeros(2 * n_circles), np.inf)

        {'type': 'ineq', 'fun': constraint_circle_overlap},

    result = minimize(

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
            or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
