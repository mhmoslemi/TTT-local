import scipy.optimize as optimize
from scipy.spatial.distance import cdist
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial parameters
    n = 26
    num_rows = 6  # Number of rows in hexagonal packing
    num_cols = 5  # Number of columns in hexagonal packing
    num_centers = num_rows * num_cols  # Total number of centers

    # Radius of each circle
    r_initial = 0.055  # Initial radius based on trial and error

    centers = []
    for row in range(num_rows):

            x = col + 0.5 * (row % 2)

            centers.append([x, y])

    min_x, max_x = np.min(centers[:, 0]), np.max(centers[:, 0])
    min_y, max_y = np.min(centers[:, 1]), np.max(centers[:, 1])
    scale = min(1.0 / max_x, 1.0 / max_y)
    centers = centers * scale

    # Adjust centers to ensure they fit in [0, 1]^2
    centers[:, 0] = centers[:, 0] + (1.0 - max_x) / 2
    centers[:, 1] = centers[:, 1] + (1.0 - max_y) / 2

    # Initial radii
    radii = np.full(num_centers, r_initial)

    # Optimization parameters
    bounds = [(0.0, 1.0) for _ in range(2 * n)]  # Bounds for x and y of centers

    def objective(x):

        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraint_overlap(x):

        distances = cdist(centers_flat, centers_flat)
