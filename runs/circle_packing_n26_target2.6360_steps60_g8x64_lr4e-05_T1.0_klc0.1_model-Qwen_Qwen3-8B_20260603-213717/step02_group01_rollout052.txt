import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Number of circles
    n = 26

    # Generate an initial nested packing with some offset from the boundaries
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    radius_initial = 0.1

    # Create a nested grid pattern with circles arranged around the center
    # We'll place them in a circular pattern and then allow optimization
    angle = 2 * np.pi / n
    for i in range(n):
        theta = angle * i
        x = 0.5 + 0.4 * np.cos(theta)
        y = 0.5 + 0.4 * np.sin(theta)
        # Ensure circles are not too close to boundaries
        x = max(0.1, min(0.9, x))
        y = max(0.1, min(0.9, y))
        centers[i] = np.array([x, y])

    # Initialize radii as a uniform value
    radii[:] = radius_initial

    # Define objective function: maximize the sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize

    # Define constraints: non-overlap and boundary
    def constraints(radii_vec):
        centers_reshaped = centers
        radii_reshaped = radii_vec

        # Non-overlap constraints: dist(c_i, c_j) >= r_i + r_j
        non_overlap_constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_reshaped[i][0] - centers_reshaped[j][0]
                dy = centers_reshaped[i][1] - centers_reshaped[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraint = dist - radii_reshaped[i] - radii_reshaped[j]
                non_overlap_constraints.append(constraint)

        # Boundary constraints: x - r >= 0, x + r <= 1, same for y
        boundary_constraints = []
        for i in range(n):
            x, y = centers_reshaped[i]
            r = radii_reshaped[i]
            boundary_constraints.append(x - r)
            boundary_constraints.append(1 - x - r)
            boundary_constraints.append(y - r)
            boundary_constraints.append(1 - y - r)

        return np.concatenate([non_overlap_constraints, boundary_constraints])

    # Initial guess
    initial_guess = np.full(n, radius_initial)

    # Bounds: each radius must be >= 0
    bounds = [(0, 1) for _ in range(n)]

    # Use SLSQP to handle the constrained optimization
    result = opt.minimize(

        x0=initial_guess,
        method='SLSQP',

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
