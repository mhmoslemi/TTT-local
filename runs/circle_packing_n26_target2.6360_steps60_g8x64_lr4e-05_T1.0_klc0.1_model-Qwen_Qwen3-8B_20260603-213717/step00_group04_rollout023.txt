import numpy as np
import cvxpy as cp

def run_packing():
    # Define the number of circles and the square size
    n = 26
    size = 1.0

    # Define a grid layout (5x5 grid with one circle removed)
    rows = 5
    cols = 5
    centers = np.zeros((n, 2))
    idx = 0

    for i in range(rows):
        for j in range(cols):
            if i == 0 and j == 0:
                continue  # Skip one circle to fit 26 circles in 5x5 grid
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]
            idx += 1

    # Define the optimization problem
    # Variables
    radii = cp.Variable(n, non_negative=True)

    # Constraints
    constraints = []
    for i in range(n):
        x, y = centers[i]
        constraints.append(radii[i] <= x)       # Left boundary
        constraints.append(radii[i] <= 1 - x)   # Right boundary
        constraints.append(radii[i] <= y)       # Bottom boundary
        constraints.append(radii[i] <= 1 - y)   # Top boundary

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = cp.sqrt(dx**2 + dy**2)
            constraints.append(dist >= radii[i] + radii[j] - 1e-6)

    # Objective: Maximize the sum of radii
    objective = cp.Maximize(cp.sum(radii))

    # Solve the problem
    problem = cp.Problem(objective, constraints)
    problem.solve()

    # Extract results
    centers = np.array(centers)
    radii = np.array(radii.value)

    if not valid:

    sum_radii = np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
