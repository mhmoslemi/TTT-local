import numpy as np
import scipy.optimize as opt

def run_packing():
    num_circles = 26
    initial_radius_guess = 0.10  # Slightly increased for better exploration

    # Initial hybrid packing: hexagonal grid with corner adjustments
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5  # 5 columns, leaving space for corners
        col = i % 5
        x = col * (2 * initial_radius_guess) + initial_radius_guess
        y = row * (np.sqrt(3) * initial_radius_guess) + initial_radius_guess

        # Offset for odd rows
        if row % 2 == 1:
            x += initial_radius_guess

        # Ensure the circle stays within the square
        x = np.clip(x, 0, 1)
        y = np.clip(y, 0, 1)

        centers[i] = [x, y]

    # Objective function to maximize the sum of radii
    def objective(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:].reshape(num_circles)
        return -np.sum(radii_flat)  # Negative for minimization

    # Non-overlap constraint: distance between centers >= sum of radii
    def constraint_non_overlap(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:].reshape(num_circles)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return -1.0
        return 0.0

    # Boundary constraint: x - r >= 0 and x + r <= 1, same for y
    def constraint_boundary(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:].reshape(num_circles)

            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
                return -1.0
            if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1.0
        return 0.0

    # Optimization bounds
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))  # x
