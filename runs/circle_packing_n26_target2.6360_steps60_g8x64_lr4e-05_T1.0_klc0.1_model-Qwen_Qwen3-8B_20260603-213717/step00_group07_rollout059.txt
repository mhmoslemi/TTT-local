import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def compute_constraints(centers, radii):
    n = centers.shape[0]
    constraints = []
    # Boundary constraints
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i, x_val=x, y_val=y, r_val=r: (x_val - r_val)})

        constraints.append({'type': 'ineq', 'fun': lambda x, i=i, x_val=x, y_val=y, r_val=r: (y_val - r_val)})
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i, x_val=x, y_val=y, r_val=r: (1.0 - y_val - r_val)})
    # Non-overlap constraints

            constraints.append({'type': 'ineq', 'fun': constraint})
    return constraints

def objective_function(params, n, initial_centers, initial_radii):
