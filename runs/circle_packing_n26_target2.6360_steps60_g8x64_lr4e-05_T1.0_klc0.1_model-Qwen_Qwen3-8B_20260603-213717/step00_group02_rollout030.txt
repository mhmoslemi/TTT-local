import numpy as np
import scipy.optimize as optimize

def run_packing():

    # Create 26 circles in a hexagonal grid, adjusting to fit within the square
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Define a hexagonal layout
    def hexagonal_grid(num_circles):
        # Determine number of rows
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))

        # Adjust based on hexagonal grid
        for i in range(rows):
            for j in range(cols):
                x = j + 0.5 * i
                y = i
                if x >= 1 or y >= 1:
                    continue
                idx = i * cols + j
                if idx < num_circles:
                    centers[idx, 0] = x
                    centers[idx, 1] = y
                    radii[idx] = 0.05  # Initial radius

    hexagonal_grid(num_circles)

    x = np.concatenate([centers.flatten(), radii])

    def constraints(x):

        radii = x[num_circles*2:]

        for i in range(num_circles):

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"
