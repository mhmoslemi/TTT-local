import scipy.optimize as opt

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    radii = np.zeros(n)

    # This is a simplified hexagonal packing with 26 circles
    # Adjust the number of rows and columns for hexagonal grid
    # We'll manually define the positions to ensure they fit in [0,1]^2

    hex_rows = 6
    hex_cols = 5
    spacing = 0.25  # initial spacing
    offset = spacing * np.sqrt(3) / 2  # vertical offset for hexagonal packing

    for i in range(hex_rows):

            if i % 2 == 0:
                x = j * spacing
            else:
                x = (j + 0.5) * spacing
            y = i * offset
            centers[idx] = [x, y]

    # Adjust positions to fit within [0,1]^2
    centers[:, 0] /= 1.2
    centers[:, 1] /= 1.2

    # Initial radii
    initial_radius = 0.05
    radii[:] = initial_radius

    # Optimization objective: maximize sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints
    def constraints(radii_vec):
        centers_reshaped = centers
        radii_reshaped = radii_vec

        # Create a list of constraints
        constraints = []

        # Boundary constraints
        for i in range(n):
            x, y = centers_reshaped[i]
            r = radii_reshaped[i]
            constraints.append(x - r)
            constraints.append(1 - x - r)
            constraints.append(y - r)

                dist = np.sqrt(np.sum((centers_reshaped[i] - centers_reshaped[j]) ** 2))
                constraint = dist - radii_reshaped[i] - radii_reshaped[j]
                constraints.append(constraint)

        return np.array(constraints)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
