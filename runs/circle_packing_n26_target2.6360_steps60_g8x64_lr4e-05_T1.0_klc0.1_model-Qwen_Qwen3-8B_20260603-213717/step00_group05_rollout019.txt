import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess for centers (hexagonal packing)
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial radius
    r_initial = 0.05
    
    # First few rows of hexagonal packing
    rows = 5
    cols = 6
    radius_step = r_initial * np.sqrt(3) * 0.8  # Adjust step to fit in square
    dx = 0.5
    dy = radius_step * np.sqrt(3) / 2  # Height of hexagon
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                x = j * dx
            else:
                x = (j + 0.5) * dx
            y = i * dy
            if i < rows and j < cols:
                centers[i * cols + j] = [x, y]
                radii[i * cols + j] = r_initial

    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Negative for minimization

    # Define constraints
    def constraint_center_boundaries(x):
        x_centers = x[:2*num_circles].reshape(num_circles, 2)
        r = x[2*num_circles:]

            x_i, y_i = x_centers[i]
            r_i = r[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or
                y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return np.inf
        return 0

    def constraint_circle_overlap(x):

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
