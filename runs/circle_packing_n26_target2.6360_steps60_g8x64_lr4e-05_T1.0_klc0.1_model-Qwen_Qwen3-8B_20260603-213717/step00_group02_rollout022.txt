import numpy as np
import cvxpy as cp

def run_packing():

    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    grid_size = 5
    spacing = 1 / (grid_size + 1)

            idx = i * grid_size + j

            centers[idx, 1] = i * spacing + spacing / 2
            radii[idx] = spacing / 2 - 1e-4  # Start with small radii

    centers[25, 0] = 1 - spacing / 2
    centers[25, 1] = spacing / 2
    radii[25] = spacing / 2 - 1e-4

    # Define variables
    x = cp.Variable((num_circles, 2))
    r = cp.Variable(num_circles)

    # Objective: maximize sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # Each circle must be within the unit square
    for i in range(num_circles):
        constraints.append(x[i, 0] >= 0)
        constraints.append(x[i, 0] <= 1)
        constraints.append(x[i, 1] >= 0)
        constraints.append(x[i, 1] <= 1)

        constraints.append(r[i] >= 0)

            dist = cp.norm(x[i] - x[j])

    # Solve the optimization problem

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
