import numpy as np
import scipy.optimize as optimize
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Initialize centers in a 5x5 grid with some spacing
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.25 + 0.125
        centers[i, 1] = row * 0.25 + 0.125

    # Initial guess for radii (all equal)
    radii = np.full(n, 0.05)

    # Objective function to maximize the sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Negative for minimization

    # Bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(n)]

    # Constraints: distance between centers >= sum of radii
    def constraint_distances(radii_flat):
        radii = radii_flat.reshape(n)
        distances = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                distances[idx] = dist - (radii[i] + radii[j])
                idx += 1
        return distances

    # Constraints: circles inside the unit square
    def constraint_inside(radii_flat):
        radii = radii_flat.reshape(n)
        inside = np.zeros(n)
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                inside[i] = 1.0  # Violation
        return inside

    # Combine all constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_distances},
        {'type': 'ineq', 'fun': constraint_inside}
    ]

    # Initial guess
    initial_guess = radii.copy()

    # Optimization settings
    options = {
        'maxiter': 1000,
        'tol': 1e-8,
        'disp': False
    }

    # Run optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = optimize.minimize(
            fun=objective,
            x0=initial_guess,
            bounds=bounds,
            constraints=constraints,
            method='SLSQP',
            options=options
        )

    if not result.success:
        # Fallback to initial guess if optimization fails
        optimal_radii = radii
    else:
        optimal_radii = result.x

    # Final check for valid packing
    valid, message = validate_packing(centers, optimal_radii)
    if not valid:
        raise ValueError(f"Packing validation failed: {message}")

    sum_radii = np.sum(optimal_radii)
    return centers, optimal_radii, sum_radii


def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
