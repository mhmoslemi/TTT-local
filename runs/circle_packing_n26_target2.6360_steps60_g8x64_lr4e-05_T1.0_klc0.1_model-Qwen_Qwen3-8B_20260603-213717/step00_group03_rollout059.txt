import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Define the number of circles and initial parameters
    n = 26
    max_radius = 0.1  # Initial estimate of the maximum radius
    initial_radius = max_radius * np.ones(n)
    
    # First, generate an initial hexagonal arrangement
    centers = np.zeros((n, 2))
    
    # Place 7 circles in a hexagonal arrangement to start with
    for i in range(7):
        if i == 0:
            centers[i] = [0.5, 0.5]
        else:
            angle = 2 * np.pi / 6 * i
            radius = 0.35
            x = 0.5 + radius * np.cos(angle)
            y = 0.5 + radius * np.sin(angle)
            centers[i] = [x, y]
    
    # Add more circles to reach 26
    for i in range(7, n):
        angle = 2 * np.pi / 6 * i
        radius = 0.3
        x = 0.5 + radius * np.cos(angle)
        y = 0.5 + radius * np.sin(angle)
        centers[i] = [x, y]
    
    # Function to calculate total sum of radii
    def objective(radii):
        return -np.sum(radii)  # minimize negative sum (maximize sum)
    
    # Function to calculate the constraint for non-overlapping circles
    def constraint_non_overlap(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii = x[n*2:]
        for i in range(n):
            for j in range(i+1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1  # violation of non-overlapping constraint
        return 0
    
    # Function to calculate boundary constraint
    def constraint_boundary(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii = x[n*2:]
        for i in range(n):
            xcen, ycen = centers_flat[i]
            if xcen - radii[i] < 0 or xcen + radii[i] > 1 or ycen - radii[i] < 0 or ycen + radii[i] > 1:
                return -1  # violation of boundary constraint
        return 0
    
    # Initial guess: combine centers and radii
    x0 = np.concatenate((centers.flatten(), initial_radius))
    
    # Set bounds: centers in [0, 1], radii >= 0
    bounds = [(0, 1) for _ in range(n*2)] + [(0, np.inf) for _ in range(n)]

    ]

        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,

    if not valid:

        return np.zeros((n, 2)), np.zeros(n), 0.0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
