import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    grid_points = []

            x = i / 3.5
            y = j / 4.0 + (i % 2) * 0.1
            grid_points.append((x, y))

    adjusted_centers = []
    for x, y in grid_points:

        y = max(0.05, min(0.95, y))
        adjusted_centers.append((x, y))
    
    # Assign centers to the first n positions
    centers[:len(adjusted_centers)] = np.array(adjusted_centers)
    
    # Compute initial radius based on shortest distance between neighbors
    distances = []
    for i in range(len(adjusted_centers)):
        for j in range(i + 1, len(adjusted_centers)):
            dist = np.hypot(adjusted_centers[i][0] - adjusted_centers[j][0],

            distances.append(dist)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.hypot(centers[i][0] - centers[j][0],
                            centers[i][1] - centers[j][1])
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
