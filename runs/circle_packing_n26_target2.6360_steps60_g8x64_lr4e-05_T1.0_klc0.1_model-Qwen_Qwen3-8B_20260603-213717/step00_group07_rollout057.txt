import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid arrangement - 5x5 grid with 25 circles
    grid_size = 5
    spacing = 1.0 / (grid_size - 1)
    initial_centers = np.zeros((grid_size * grid_size, 2))
    for i in range(grid_size):
        for j in range(grid_size):
            x = i * spacing
            y = j * spacing
            initial_centers[i * grid_size + j] = [x, y]
    
    # Add one more circle at the center for optimization
    extra_circle = np.array([0.5, 0.5])
    initial_centers = np.vstack([initial_centers, extra_circle])
    
    # Define the initial guess for radii (small, just to get the optimization started)
    initial_radii = np.full(26, 0.05)
    
    # Define the optimization problem
    def objective_function(radii):
        return -np.sum(radii)  # Minimize the negative sum (maximize the sum)

    def constraint_overlap(i, j, radii):
        def constraint(x):
            return np.sqrt(np.sum((x[i] - x[j])**2)) - radii[i] - radii[j]
        return constraint

    def constraint_boundary(i, radii):
        def constraint(x):
            x_i, y_i = x[i]
            r = radii[i]
            return (x_i - r) - (-1e-12)  # left boundary
        return constraint

    def constraint_boundary_right(i, radii):
        def constraint(x):
            x_i, y_i = x[i]
            r = radii[i]
            return (x_i + r) - (1 + 1e-12)  # right boundary
        return constraint

    def constraint_boundary_top(i, radii):
        def constraint(x):
            x_i, y_i = x[i]
            r = radii[i]
            return (y_i + r) - (1 + 1e-12)  # top boundary
        return constraint

    def constraint_boundary_bottom(i, radii):
        def constraint(x):
            x_i, y_i = x[i]
            r = radii[i]
            return (y_i - r) - (-1e-12)  # bottom boundary
        return constraint

    # Create constraints
    constraints = []
    
    # Add boundary constraints
    for i in range(26):
        # Left boundary
        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i, initial_radii)})
        # Right boundary
        constraints.append({'type': 'ineq', 'fun': constraint_boundary_right(i, initial_radii)})
        # Top boundary
        constraints.append({'type': 'ineq', 'fun': constraint_boundary_top(i, initial_radii)})
        # Bottom boundary
        constraints.append({'type': 'ineq', 'fun': constraint_boundary_bottom(i, initial_radii)})
    
    # Add non-overlap constraints
    for i in range(26):
        for j in range(i + 1, 26):
            constraints.append({'type': 'ineq', 'fun': constraint_overlap(i, j, initial_radii)})
    
    # Set up optimization
    bounds = [(0, 1) for _ in range(26 * 2)]  # bounds for centers
    bounds += [(0, 1) for _ in range(26)]     # bounds for radii
    
    # Initial guess - centers and radii
    x0 = np.hstack((initial_centers.flatten(), initial_radii))
    
    # Run optimization
    result = opt.minimize(
        objective_function,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8
    )
    
    # Extract results

    radii = result.x[26 * 2:]
    
    # Validate the packing

    if not valid:
        raise ValueError(f"Packing validation failed: {message}")
    
    # Compute sum of radii
    sum_radii = np.sum(radii)
    
    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
