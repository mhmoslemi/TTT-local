import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    n = 26

    # Hexagonal packing in the unit square
    def hexagonal_grid(num_circles):
        positions = []
        row = 0
        col = 0
        while len(positions) < num_circles:
            # Place the circle in a hexagonal grid and scale to fit in the unit square
            x = (col + 0.5 * (row % 2)) / (np.sqrt(3) / 2)

                positions.append((x, y))

                row += 1
            if row > 10:
                break
        return np.array(positions)

    centers = hexagonal_grid(n)
    radii = np.full(n, 0.1)  # Initial guess for radius

    def objective(radii_vec):
        return -np.sum(radii_vec)  # minimize negative sum to maximize sum

    def constraints(radii_vec):
        # Re-arrange centers and radii
        centers_reshaped = centers
        radii_reshaped = radii_vec

        # Non-overlapping constraints: dist(c_i, c_j) >= r_i + r_j
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_reshaped[i, 0] - centers_reshaped[j, 0]
                dy = centers_reshaped[i, 1] - centers_reshaped[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraint = dist - radii_reshaped[i] - radii_reshaped[j]
                constraints.append(constraint)

            r = radii_reshaped[i]
            constraints.append(x - r)
            constraints.append(1 - x - r)
            constraints.append(y - r)
            constraints.append(1 - y - r)

        return np.array(constraints)

        x0=radii,
        method='SLSQP',
        bounds=bounds,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
