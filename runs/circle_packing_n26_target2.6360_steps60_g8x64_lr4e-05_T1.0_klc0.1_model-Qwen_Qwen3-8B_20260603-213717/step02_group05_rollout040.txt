import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.05  # Initial small radius

    # Generate initial hexagonal grid in the center
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    spacing = 1.0 / (cols + 1)
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        # Shift initial positions slightly inward to allow for growth near boundaries
        x = col * spacing + spacing / 2
        y = row * spacing * np.sqrt(3) / 2 + spacing / 2
        centers[i] = [x, y]

    # Define objective function to maximize sum of radii
    def objective(x):
        # x is a flat array of (centers, radii)
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:].reshape(num_circles)

        # Check if all circles stay within the square
        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (0 - r_i <= x_i <= 1 + r_i and 0 - r_i <= y_i <= 1 + r_i):
                return -np.inf  # Invalid configuration

        # Check if all circles are non-overlapping
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx*dx + dy*dy)

        # Return negative of sum of radii to minimize
        return -np.sum(radii_flat)

    # Optimization bounds
    bounds = [(0, 1) for _ in range(num_circles * 2)] + [(0, 1) for _ in range(num_circles)]

    # Optimization with SLSQP for better convergence
    result = opt.minimize(
