import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.05  # Initial small radius

    # Generate initial hexagonal grid inside the square
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))

        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing * np.sqrt(3) / 2 + spacing / 2
        centers[i] = [x, y]

    def objective(x):
        # x is a flat array of (centers, radii)
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)

        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (0 - r_i <= x_i <= 1 + r_i and 0 - r_i <= y_i <= 1 + r_i):

        for i in range(num_circles):
