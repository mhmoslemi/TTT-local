import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial centers and radii for 26 circles using a hexagonal pattern
    # Start with a 5x6 grid, offset rows for hexagonal pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    
    # Initial guess for radii: all equal to 0.05 (adjustable based on expected density)
    initial_radius = 0.05
    
    # Set initial centers in a hexagonal pattern
    num_rows = 6
    num_cols = 5
    for i in range(num_rows):
        for j in range(num_cols):
            x = j * (1 + initial_radius * 2) / (num_cols - 1)
            y = i * (np.sqrt(3) * initial_radius * 2) / (num_rows - 1)
            centers[i * num_cols + j] = [x, y]
            radii[i * num_cols + j] = initial_radius
    
    # Define objective function to maximize total radius
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraints: non-overlapping and within the square
    def constraints_func(radii, centers):
        n = len(radii)
        cons = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                cons.append(dist - (radii[i] + radii[j]) + 1e-12)  # >= 0
        return cons

    # Optimization with bounds and constraints
    bounds = [(0, 0.5) for _ in range(26)]  # Radii between 0 and 0.5

    # Initial guess for radii
    initial_guess = np.full(26, initial_radius)

    # Run the optimization
    result = optimize.minimize(
        fun=objective,
        x0=initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={
            'type': 'ineq',
            'fun': lambda x: constraints_func(x, centers)
        },
        tol=1e-6
    )

    # Extract final centers and radii
    final_radii = result.x
    final_centers = centers.copy()

    if not is_valid:
        raise ValueError(f"Invalid packing: {message}")

    # Compute total sum of radii
    sum_radii = np.sum(final_radii)

    return final_centers, final_radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
