import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    n = 26

    # Calculate the side length of the hexagon to fit within the unit square
    # The side length is chosen to allow for 5 rows with some spacing
    initial_radius = 0.15
    spacing = 2 * initial_radius
    rows = 5
    cols = 5
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Initial hexagonal grid
    for i in range(rows):
        for j in range(cols):
            # Hexagonal grid: vertical spacing is sqrt(3) * horizontal spacing
            x = j * spacing
            y = i * spacing * np.sqrt(3)
            idx = i * cols + j
            centers[idx] = [x, y]
            radii[idx] = initial_radius

    max_x = np.max(centers[:, 0])

        return -np.sum(radii_vec)  # minimize negative sum to maximize sum

        radii_reshaped = radii_vec

                dist = np.sqrt(np.sum((centers_reshaped[i] - centers_reshaped[j]) ** 2))
                constraint = dist - radii_reshaped[i] - radii_reshaped[j]
                constraints.append(constraint)

            x, y = centers_reshaped[i]

            constraints.append(x - r)
            constraints.append(1 - x - r)
            constraints.append(y - r)
            constraints.append(1 - y - r)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
