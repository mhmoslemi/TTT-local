import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26

    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Generate 16 in square grid centered
    square_grid = []
    for i in range(4):
        for j in range(4):
            x = 0.25 + i * 0.25
            y = 0.25 + j * 0.25
            square_grid.append((x, y))

    # Generate 10 in hexagonal grid around the edges
    hex_grid = []
    for i in range(5):
        for j in range(5):
            if i % 2 == 0 and j % 2 == 0:
                continue  # skip center

            y = j / 5.0 + 0.05
            hex_grid.append((x, y))

    # Combine and adjust to fit within [0,1]^2
    grid_points = square_grid + hex_grid
    adjusted_centers = []
    for x, y in grid_points:
        x = max(0.0, min(1.0, x))
        y = max(0.0, min(1.0, y))
        adjusted_centers.append((x, y))

    # Assign centers and initial radii based on nearest neighbors
    centers[:len(adjusted_centers)] = np.array(adjusted_centers)

    # Compute initial radius based on shortest distance between neighbors
    distances = []
    for i in range(len(adjusted_centers)):
        for j in range(i + 1, len(adjusted_centers)):
            dist = np.hypot(adjusted_centers[i][0] - adjusted_centers[j][0],

            distances.append(dist)

        initial_radius = 0.1

    def constraints(x):

        non_overlap_constraints = []

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.hypot(centers[i][0] - centers[j][0],
                            centers[i][1] - centers[j][1])
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
