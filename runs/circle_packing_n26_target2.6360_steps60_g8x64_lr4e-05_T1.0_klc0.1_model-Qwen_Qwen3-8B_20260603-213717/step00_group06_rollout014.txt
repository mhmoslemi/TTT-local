import numpy as np
import scipy.optimize as optimize

def generate_initial_hex_grid(num_circles):
    """Generate initial hexagonal grid of circles within the unit square."""
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    radius = 0.05  # Initial radius
    
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)  # Offset for even rows
            y = i * np.sqrt(3) / 2
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            centers.append([x, y])
    
    # Convert to numpy array
    centers = np.array(centers)
    
    # Adjust positions for better spacing
    centers = centers / (np.max(centers, axis=0) + 1e-6)  # Scale to fit in [0,1]^2
    radii = np.full(num_circles, radius)
    
    return centers, radii

def objective_function(params, centers, num_circles):
    """Objective function to maximize the sum of radii."""
    radii = params[:num_circles]
    return -np.sum(radii)  # Minimize negative sum

def constraint_circle_inside(params, centers, num_circles):
    """Constraint to ensure each circle lies within the unit square."""
    radii = params[:num_circles]
    for i in range(num_circles):
        x, y = centers[i]
        r = radii[i]
        if not (0 <= x - r and x + r <= 1 and 0 <= y - r and y + r <= 1):
            return -1  # Violates constraint
    return 0

def constraint_no_overlap(params, centers, num_circles):
    """Constraint to ensure no two circles overlap."""
    radii = params[:num_circles]
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j] - 1e-8:
                return -1  # Violates constraint
    return 0

def run_packing():

    num_circles = 26
    centers, radii = generate_initial_hex_grid(num_circles)
    
    # Flatten the parameters for optimization
    params = np.concatenate([centers.reshape(-1), radii])
    
    # Define bounds: positions between 0 and 1, radii >= 0

        {'type': 'ineq', 'fun': lambda p: constraint_circle_inside(p, centers, num_circles)},
        {'type': 'ineq', 'fun': lambda p: constraint_no_overlap(p, centers, num_circles)},
    ]
    
    # Optimization: maximize sum of radii
    result = optimize.minimize(
        fun=lambda p: objective_function(p, centers, num_circles),
        x0=params,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
