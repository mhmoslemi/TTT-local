import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Initial guess: hexagonal grid pattern
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Generate a hexagonal grid configuration
    for i in range(n):
        row = i // 5
        col = i % 5
        # Hexagonal grid with spacing adjusted to fit within the unit square
        x = col * 0.4 + 0.1

        if row % 2 == 1:
            x += 0.2  # Offset for hexagonal pattern
        centers[i] = [x, y]

    # Initial radius as 0.15, but we'll let the solver optimize it
    radii[:] = 0.15

    # Objective: maximize the sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraint function: ensures non-overlapping and boundary conditions
    def constraints(radii_vec):
        # Re-arrange radii
        radii_reshaped = radii_vec

        # Non-overlapping constraints
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraint = dist - radii_reshaped[i] - radii_reshaped[j]
                constraints.append(constraint)

        # Boundary constraints: x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1
        for i in range(n):
            x, y = centers[i]
            r = radii_reshaped[i]
            constraints.append(x - r)
            constraints.append(1 - x - r)
            constraints.append(y - r)
            constraints.append(1 - y - r)

        return np.array(constraints)

    # Initial guess: radii
    initial_guess = np.full(n, 0.15)

    # Bounds: each radius must be >= 0
    bounds = [(0, 1) for _ in range(n)]

    # Optimization method: Sequential Least Squares Programming (SLSQP)
    result = opt.minimize(
        fun=objective,
        x0=initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},

    )

    # Extract the optimal solution

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
