def initial_hexagonal_packing(n: int, square_size: float = 1.0) -> Tuple[np.ndarray, np.ndarray]:

    Generate an initial hexagonal close packing of n circles in a square.

    # and the number of columns is ceil(sqrt(n))
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    d = 2 * np.sqrt(3) * (square_size / (cols + 0.5))

    radii = np.full(n, 1.0 / np.sqrt(3) if rows > 1 else 1.0)
