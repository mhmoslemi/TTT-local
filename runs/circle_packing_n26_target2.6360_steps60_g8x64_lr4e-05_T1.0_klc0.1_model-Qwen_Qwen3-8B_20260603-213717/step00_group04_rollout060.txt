import numpy as np
import scipy.optimize as opt
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess of circle positions in a hexagonal lattice
    # We'll use a grid of 5 rows and 6 columns (total 30), but we'll keep only 26
    # Create a hexagonal grid
    num_circles = 26
    grid_rows = 5
    grid_cols = 6
    grid = np.zeros((grid_rows, grid_cols, 2))
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j / (grid_cols - 1) * 1.0
            y = i / (grid_rows - 1) * 1.0
            if i % 2 == 1:
                x += 0.5 / (grid_cols - 1)
            grid[i, j, :] = [x, y]

    # Only take the first 26 circles
    centers_initial = grid[:num_circles, :, :].reshape(-1, 2)
    radii_initial = np.full(num_circles, 0.02)

    # Define the objective function
    def objective(radii):
        return -np.sum(radii)  # Negative because we use minimization

    # Define the constraints (no overlap and within bounds)
    def constraint_overlap(x):
        centers = x[:num_circles, :2]
        radii = x[num_circles:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1
        return 0

    def constraint_bounds(x):
        centers = x[:num_circles, :2]
        radii = x[num_circles:]
        for i in range(num_circles):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return -1
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return -1
        return 0

    # Define bounds: centers must be within [0, 1]^2, radii ≥ 0
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
        bounds.append((0, 1))  # radius

    # Initial guess
    x0 = np.concatenate([centers_initial, radii_initial])

    # Optimization settings
    result = opt.minimize(
        fun=objective,
        x0=x0,
        bounds=bounds,
        constraints=[{"type": "ineq", "fun": constraint_overlap},
                     {"type": "ineq", "fun": constraint_bounds}],
        method="SLSQP",
        options={"ftol": 1e-8, "maxiter": 1000}
    )

    # Extract result
    centers_result = result.x[:num_circles, :2]
    radii_result = result.x[num_circles:]
    sum_radii = np.sum(radii_result)

    # Validate the packing (optional, but recommended for reliability)
    is_valid, msg = validate_packing(centers_result, radii_result)
    if not is_valid:
        print(f"Validation failed: {msg}")
        return centers_initial, radii_initial, np.sum(radii_initial)

    return centers_result, radii_result, sum_radii


def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
