import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds, BoundsError
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial layout: hexagonal grid with 5 rows
    n_circles = 26
    rows = 5
    cols = 5
    hex_radius = 1 / (2 * np.sqrt(3))  # Adjust this to fit the unit square
    hex_spacing = 2 * hex_radius
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                x = j * hex_spacing
            else:
                x = j * hex_spacing - hex_spacing / 2
            y = i * (np.sqrt(3) * hex_radius)
            idx = i * cols + j
            if idx < n_circles:
                centers[idx] = [x, y]
                radii[idx] = hex_radius

    # Define bounds for centers and radii
    bounds = []
    for _ in range(n_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    bounds = np.array(bounds).reshape(-1, 3)

    # Optimization variables: x, y, r for each circle
    x0 = np.zeros(n_circles * 3)
    for i in range(n_circles):
        x0[i * 3] = centers[i, 0]
        x0[i * 3 + 1] = centers[i, 1]
        x0[i * 3 + 2] = radii[i]

    # Optimization problem
    def objective(x):
        return -np.sum(x[2::3])  # We aim to maximize the sum of radii

    def constraint_non_overlap(x):
        centers = x[::3].reshape(-1, 2)
        radii = x[2::3]
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1.0  # Violation of non-overlap constraint
        return 0.0

    def constraint_boundary(x):
        centers = x[::3].reshape(-1, 2)
        radii = x[2::3]
        for i in range(n_circles):
            x_center, y_center = centers[i]
            r = radii[i]
            if x_center - r < -1e-8 or x_center + r > 1.0 + 1e-8:
                return -1.0
            if y_center - r < -1e-8 or y_center + r > 1.0 + 1e-8:
                return -1.0
        return 0.0

    constraints = [
        {'type': 'ineq', 'fun': constraint_non_overlap},
        {'type': 'ineq', 'fun': constraint_boundary}
    ]

    # Optimization settings
    options = {
        'disp': False,
        'maxiter': 1000,
        'ftol': 1e-10,
        'xtol': 1e-10
    }

            x0=x0,
            bounds=bounds,
            constraints=constraints,
            method='SLSQP',
            options=options
        )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
