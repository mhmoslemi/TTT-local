import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Define centers for a hexagonal close-packed arrangement
    # We use a hexagonal grid with small spacing
    # Start from the bottom left and move in a staggered pattern
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Define grid size and spacing
    rows = 5
    cols = 5
    spacing = 0.2  # Adjust spacing for tighter packing
    offset = spacing * np.sqrt(3) / 2  # Hexagonal spacing in y-direction

    for i in range(rows):
        for j in range(cols):
            # Staggered pattern: even rows are offset
            if i % 2 == 0:
                x = j * spacing
            else:
                x = (j + 0.5) * spacing
            y = i * offset
            idx = i * cols + j
            centers[idx] = [x, y]
    
    # Set initial radii to 0.1 (smallest radius that allows for packing)
    initial_radius = 0.1
    radii[:] = initial_radius
    centers[25] = [0.5, 0.5]  # Add an extra circle at the center if needed
    
    # Define objective function for optimization
    def objective(radii_vec, centers_arr):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

    def constraints(radii_vec, centers_arr):
        # Constraints: no overlap between circles and within the square
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers_arr[i] - centers_arr[j]) ** 2))
                constraint = dist - radii_vec[i] - radii_vec[j]
                constraints.append(constraint)
        for i in range(n):
            x, y = centers_arr[i]
            r = radii_vec[i]
            constraints.append(x - r)
            constraints.append(1 - x - r)
            constraints.append(y - r)
            constraints.append(1 - y - r)
        return constraints

    # Initial guess for radii
    initial_guess = np.full(n, initial_radius)

    # Optimization setup
    bounds = [(0, 1) for _ in range(n)]
    result = opt.minimize(
        fun=lambda r: objective(r, centers),
        x0=initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': lambda r: constraints(r, centers)},
        tol=1e-10
    )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
