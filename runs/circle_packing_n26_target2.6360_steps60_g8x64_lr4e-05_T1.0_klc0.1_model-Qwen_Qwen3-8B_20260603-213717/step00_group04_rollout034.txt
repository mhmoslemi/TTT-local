import numpy as np
import scipy.optimize as optimize

def run_packing():

    n = 26
    centers = np.zeros((n, 2))
    radii = np.ones(n) * 0.05  # Small initial radii

    # Define the objective: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraints:
    # 1. Each circle must lie within [0,1]^2
    # 2. Distance between circles must be at least the sum of their radii
    constraints = []

    # Center placement constraint (bounds on x and y)
    for i in range(n):
        def center_bounds(i):
            def constraint(x):
                r = x[i]
                x_center, y_center = centers[i]
                return (x_center - r >= -1e-12) and (x_center + r <= 1 + 1e-12) and \
                       (y_center - r >= -1e-12) and (y_center + r <= 1 + 1e-12)
            return constraint
        constraints.append({'type': 'ineq', 'fun': center_bounds(i)})

    # Pairwise distance constraint
    for i in range(n):
        for j in range(i + 1, n):
            def dist_constraint(i, j):
                def constraint(x):
                    r_i = x[i]
                    r_j = x[j]
                    x_i, y_i = centers[i]
                    x_j, y_j = centers[j]
                    dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
                    return dist - r_i - r_j + 1e-12
                return constraint
            constraints.append({'type': 'ineq', 'fun': dist_constraint(i, j)})

    # Use scipy.optimize to maximize sum of radii
    x0 = radii  # Initial guess for radii
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        constraints=constraints,

    )

    final_radii = result.x

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
