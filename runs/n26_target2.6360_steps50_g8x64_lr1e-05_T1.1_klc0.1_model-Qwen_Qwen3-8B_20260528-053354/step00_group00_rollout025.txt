import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)
    for i in range(n_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
        radii[i] = 0.05

    def objective(radii_and_centers):
        # Unpack
        radii = radii_and_centers[:n_circles]
        centers = radii_and_centers[n_circles:]
        centers = centers.reshape((n_circles, 2))
        # Objective: maximize the sum of radii
        return -np.sum(radii)

    def constraints(radii_and_centers):
        radii = radii_and_centers[:n_circles]
        centers = radii_and_centers[n_circles:]
        centers = centers.reshape((n_circles, 2))
        # Constraint 1: All circles are inside the unit square
        in_square = np.all((radii <= 1 - centers) & (radii <= centers), axis=1)
        # Constraint 2: No overlap between circles
        dists = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        no_overlap = np.all(dists >= radii_sum, axis=1)
        # Combine constraints
        valid = in_square & no_overlap
        return valid

    # Initial guess
    initial_guess = np.concatenate((radii, centers.flatten()))

    # Setup optimization
    bounds = []
    for _ in range(n_circles):
        bounds.append((0.0, 0.5))  # radii are between 0 and 0.5
    bounds += [(0.0, 1.0) for _ in range(2 * n_circles)]  # x and y coordinates between 0 and 1

    # Run optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-8
    )

    # Extract results
    final_radii = result.x[:n_circles]
    final_centers = result.x[n_circles:].reshape((n_circles, 2))
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
