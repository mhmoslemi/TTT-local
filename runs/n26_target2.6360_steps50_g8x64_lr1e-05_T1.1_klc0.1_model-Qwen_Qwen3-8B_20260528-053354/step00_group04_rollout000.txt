import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = i / 25
        centers[i, 1] = 0.5

    # Initial radii: small positive values
    radii = np.full(26, 0.05)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Constraints: 
    # 1. Each circle must be inside the unit square
    # 2. Distance between any two centers >= sum of radii
    def constraints(x):
        # Extract centers and radii from x
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]

        # Constraint 1: Each circle is inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False

        # Constraint 2: No overlap between circles
        dists = pdist(centers_flat)
        for i in range(len(dists)):
            dist = dists[i]
            r1 = radii_flat[i // 26]
            r2 = radii_flat[i % 26]
            if dist < r1 + r2 - 1e-8:
                return False

        return True

    # Combine all variables into one vector
    x0 = np.concatenate([centers.ravel(), radii])

    # Optimization
    result = minimize(objective, x0, constraints=lambda x: constraints(x))

    # Extract result
    centers_result = result.x[:52].reshape(26, 2)
    radii_result = result.x[52:]
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
