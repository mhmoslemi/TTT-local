import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: equally spaced grid for 26 circles
    centers = np.stack([np.linspace(0.1, 0.9, 5), np.linspace(0.1, 0.9, 5)], axis=1)
    centers = np.vstack([centers, centers[:5], centers[:5], centers[:5], centers[:5]])
    radii = np.full(26, 0.05)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints_func(radii, centers):
        # Constraint 1: All circles must be fully inside the unit square
        r = radii
        in_square = np.all((r <= 1 - r) & (r <= 1 - r), axis=0)
        # Constraint 2: No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        no_overlap = np.all(dists >= r[np.newaxis, :] + r[:, np.newaxis], axis=1)
        return np.concatenate([in_square, no_overlap])
    
    def constraint_func(x):
        # Flatten and reshape to get centers and radii
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:]
        # Check constraints
        in_square = np.all((radii_flat <= 1 - radii_flat) & (radii_flat <= 1 - radii_flat), axis=0)
        dists = np.sqrt(np.sum((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :])**2, axis=2))
        no_overlap = np.all(dists >= radii_flat[np.newaxis, :] + radii_flat[:, np.newaxis], axis=1)
        return np.concatenate([in_square, no_overlap])
    
    # Bounds: centers between 0 and 1, radii between 0 and 0.5
    bounds = [(0, 1) for _ in range(26*2)] + [(0, 0.5) for _ in range(26)]
    
    # Initial guess for the optimization
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Optimization setup
    cons = {'type': 'ineq', 'fun': constraint_func}
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)
    
    # Extract solution
    centers_sol = result.x[:26*2].reshape(26, 2)
    radii_sol = result.x[26*2:]
    sum_radii = -result.fun
    
    # Round to avoid numerical instability
    radii_sol = np.round(radii_sol, 8)
    centers_sol = np.round(centers_sol, 8)
    
    return (centers_sol, radii_sol, sum_radii)
