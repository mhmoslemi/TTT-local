import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing():
    # Number of circles
    n = 26
    
    # Initial guess: equally spaced grid with some radius
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    for i in range(n):
        row = i // 5
        col = i % 5
        centers[i, 0] = col / 4 + 0.1
        centers[i, 1] = row / 4 + 0.1
        radii[i] = 0.05
    
    # Define objective function: minimize the negative sum of radii
    def objective(x):
        return -np.sum(x[26:])
    
    # Define bounds: [x, y, r] for each circle
    bounds = []
    for i in range(n):
        x = x[2*i]
        y = x[2*i + 1]
        r = x[2*i + 2]
        # x must be in [r, 1 - r]
        bounds.append((r, 1 - r))
        bounds.append((r, 1 - r))
        bounds.append((0, 1))
    
    # Define nonlinear constraints
    def constraints(x):
        # Extract centers and radii
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        # Constraint 1: circle is inside the unit square
        for i in range(n):
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i] or \
               centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return np.inf
        # Constraint 2: distance between centers >= sum of radii
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0
    
    # Set up optimization problem
    x0 = np.concatenate([centers.flatten(), radii])
    result = minimize(objective, x0, bounds=bounds, constraints=NonlinearConstraint(constraints, 0, 0))
    
    # Extract result
    optimal_centers = result.x[:2*n].reshape(n, 2)
    optimal_radii = result.x[2*n:]
    optimal_sum_radii = -result.fun
    
    return (optimal_centers, optimal_radii, optimal_sum_radii)
