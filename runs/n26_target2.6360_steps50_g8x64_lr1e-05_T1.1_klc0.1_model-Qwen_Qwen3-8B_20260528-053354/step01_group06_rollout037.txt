import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[i / 5 + 0.25, j / 5 + 0.25] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [0.5, 0.5]])  # Add a central circle
    radii = np.full(26, 0.05)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize the sum of radii
    
    def constraints(radii):
        # Ensure all circles are inside the square
        for i in range(26):
            r = radii[i]
            if r > 1 - np.min([centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1]]):
                return False
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Optimization settings
    bounds = [(0, 1 - 1e-6) for _ in range(26)]  # Radii cannot exceed 1 - small epsilon to avoid edge issues
    
    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            bounds=bounds,
            constraints=[{'type': 'ineq', 'fun': lambda x: 1 - np.sum(x) - 2.636},  # Ensure sum is at least 2.636
                         {'type': 'ineq', 'fun': lambda x: constraints(x)}]
        )
    
    radii = result.x
    centers = np.array([[i / 5 + 0.25, j / 5 + 0.25] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [0.5, 0.5]])  # Keep the central circle fixed
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
