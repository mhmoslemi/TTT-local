import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: place 26 circles evenly spaced in the square
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.0  # 5 columns
        y = i // 5 / 4.0  # 5 rows
        centers[i] = [x, y]

    # Objective: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be inside the square [0,1] x [0,1]
    def square_inside(radii):
        return np.all(radii <= 0.5)  # Max radius to fit in the square

    # Constraints: circles must not overlap
    def no_overlap(centers, radii):
        dists = np.sqrt(((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2).sum(axis=2))
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization bounds: radii >= 0 and <= 0.5
    bounds = Bounds(np.zeros(26), np.full(26, 0.5))

    # Initial guess for radii
    init_radii = np.full(26, 0.1)

    # Optimization options
    options = {'disp': False, 'ftol': 1e-6, 'maxiter': 1000}

    # Optimization using scipy.optimize.minimize
    result = opt.minimize(
        fun=objective,
        x0=init_radii,
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': square_inside},
                     {'type': 'ineq', 'fun': lambda x: no_overlap(centers, x)}],
        method='SLSQP',
        options=options
    )

    # Update centers based on optimized radii
    # For simplicity, we assume a grid layout, which might not be optimal
    # but this is a starting point to allow the optimization to work
    # (for more accurate placement, a more sophisticated method would be needed)

    # Final values
    radii = result.x
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.0  # 5 columns
        y = i // 5 / 4.0  # 5 rows
        centers[i] = [x, y]
    sum_radii = result.fun * -1

    # Check if the sum is close to the target value (2.636)
    if abs(sum_radii - 2.636) > 0.01:
        print("Warning: The sum of radii is not close to the target value (2.636).")

    return (centers, radii, sum_radii)
