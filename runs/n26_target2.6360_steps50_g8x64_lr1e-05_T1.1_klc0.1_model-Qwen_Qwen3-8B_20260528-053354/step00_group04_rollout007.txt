import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with some initial radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2) * 0.9 + 0.05  # Avoid edges initially
    radii = np.full(num_circles, 0.02)  # Small initial radii

    # Objective function: maximize the sum of radii
    def objective(x):
        # x contains flattened centers and radii
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize

    # Constraints
    def constraint_centers(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        return centers_flat

    def constraint_radius(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        return radii_flat

    def constraint_edge(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1
        return 0

    def constraint_overlap(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1
        return 0

    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y coordinates
    bounds.extend([(0.0, 0.5)] * num_circles)  # radii

    # Set up the optimization problem
    x0 = np.concatenate((centers.flatten(), radii))
    cons = []
    cons.append({'type': 'ineq', 'fun': constraint_edge})
    cons.append({'type': 'ineq', 'fun': constraint_overlap})
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    centers_result = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii_result = result.x[num_circles * 2:]
    sum_radii = -result.fun  # Since we are maximizing

    return (centers_result, radii_result, sum_radii)
