import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = i // 5 + 0.1
        centers[i, 1] = i % 5 + 0.1
        radii[i] = 0.05

    # Objective function to maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraint: circles must be inside the unit square
    def constraint_inside(radii, centers):
        return np.array([r <= 1 - x for x, r in zip(centers[:, 0], radii)]) * np.array([r <= 1 - y for y, r in zip(centers[:, 1], radii)])
    
    # Constraint: circles must not overlap
    def constraint_overlap(radii, centers):
        n = len(radii)
        dist = np.zeros(n * (n - 1) // 2)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[(i*n + j - i*(i+1)//2 - i) : (i*n + j - i*(i+1)//2 - i) + 1] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
        return dist >= 0

    # Optimization bounds
    bounds = [(0.0, 0.5) for _ in range(26)]

    # Set up the optimization problem
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers)}
    ]
    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract the result
    centers_result = centers.copy()
    radii_result = result.x
    sum_radii_result = np.sum(radii_result)

    # Ensure all circles are fully inside the square
    for i in range(26):
        x, y = centers_result[i]
        r = radii_result[i]
        if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
            # Adjust position if circle is too close to the border
            dx = 0.5 * (x - r)
            dy = 0.5 * (y - r)
            centers_result[i, 0] = x - dx
            centers_result[i, 1] = y - dy
            radii_result[i] = r - dx - dy

    return (centers_result, radii_result, sum_radii_result)
