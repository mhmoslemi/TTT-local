import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    n = 26
    centers = np.array([[i / np.sqrt(n), j / np.sqrt(n)] for i in range(n) for j in range(n)])
    centers = centers[np.random.permutation(n)]
    radii = np.full(n, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[i / np.sqrt(n), j / np.sqrt(n)] for i in range(n) for j in range(n)])
        centers = centers[np.random.permutation(n)]
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    res = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': lambda x: constraints(x)},
        bounds=[(0, 1 - 1e-6) for _ in range(n)]
    )

    # Final centers (randomized for better packing)
    final_centers = np.array([[i / np.sqrt(n), j / np.sqrt(n)] for i in range(n) for j in range(n)])
    final_centers = final_centers[np.random.permutation(n)]
    final_radii = res.x

    # Ensure all circles are within the unit square
    final_centers = np.clip(final_centers, np.array([0, 0]), np.array([1, 1]))
    for i in range(n):
        final_centers[i, 0] = np.clip(final_centers[i, 0], final_radii[i], 1 - final_radii[i])
        final_centers[i, 1] = np.clip(final_centers[i, 1], final_radii[i], 1 - final_radii[i])

    sum_radii = np.sum(final_radii)
    return (final_centers, final_radii, sum_radii)
