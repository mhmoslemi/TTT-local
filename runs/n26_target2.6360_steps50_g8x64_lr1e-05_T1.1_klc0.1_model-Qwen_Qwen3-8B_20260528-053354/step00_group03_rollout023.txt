import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial small radius

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(centers, radii):
        # Constraint 1: Each circle is inside the unit square
        def inside(x, y, r):
            return (x - r >= 0) and (x + r <= 1) and (y - r >= 0) and (y + r <= 1)
        inside_constraint = lambda i: inside(centers[i, 0], centers[i, 1], radii[i])
        constraints = [inside_constraint(i) for i in range(26)]
        # Constraint 2: No two circles overlap
        def distance(x1, y1, r1, x2, y2, r2):
            dx = x1 - x2
            dy = y1 - y2
            dist = np.sqrt(dx*dx + dy*dy)
            return dist - (r1 + r2)
        no_overlap = lambda i, j: distance(centers[i, 0], centers[i, 1], radii[i],
                                           centers[j, 0], centers[j, 1], radii[j]) >= 0
        overlap_constraints = [no_overlap(i, j) for i in range(26) for j in range(i+1, 26)]
        return constraints + overlap_constraints

    def objective_with_radii(x, centers):
        radii = x
        return -np.sum(radii)

    def constraints_with_radii(x, centers):
        radii = x
        constraints = []
        # Inside constraints
        for i in range(26):
            if radii[i] < 0:
                constraints.append(-radii[i])
        # No overlap constraints
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    # Define bounds for radii: [0, 0.5] (max radius possible for circles in unit square)
    bounds = [(0, 0.5) for _ in range(26)]

    # Initial guess for radii
    initial_guess = np.full(26, 0.05)

    # Optimization
    result = minimize(
        fun=objective_with_radii,
        x0=initial_guess,
        bounds=bounds,
        constraints={
            'type': 'ineq',
            'fun': constraints_with_radii,
            'args': (centers,)
        },
        method='SLSQP'
    )

    # Extract results
    radii = result.x
    sum_radii = -result.fun

    # Check sum_radii is close to target
    if abs(sum_radii - 2.636) > 0.05:
        warnings.warn("Solution not close to target sum_radii of 2.636")

    return (centers, radii, sum_radii)
