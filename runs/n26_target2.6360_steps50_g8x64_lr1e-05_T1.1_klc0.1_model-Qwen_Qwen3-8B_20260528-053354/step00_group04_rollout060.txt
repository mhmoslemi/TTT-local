import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 26 circles arranged in a grid pattern
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum
    
    def constraints(radii):
        # Constraint 1: All circles must be within the unit square
        # x - r >= 0, 1 - x - r >= 0, y - r >= 0, 1 - y - r >= 0
        # Combine into a single constraint
        diff = np.abs(centers - radii[:, np.newaxis])  # shape (26, 26, 2)
        dist = np.sqrt(np.sum(diff**2, axis=2))  # shape (26, 26)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dist[i, j] < radii[i] + radii[j] - 1e-8:
                    return np.inf
        return 0

    # Set up optimization problem
    bounds = [(0.0, 1.0) for _ in range(num_circles)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints})
    
    # Extract final result
    best_radii = result.x
    best_centers = centers
    sum_radii = np.sum(best_radii)
    
    return (best_centers, best_radii, sum_radii)
