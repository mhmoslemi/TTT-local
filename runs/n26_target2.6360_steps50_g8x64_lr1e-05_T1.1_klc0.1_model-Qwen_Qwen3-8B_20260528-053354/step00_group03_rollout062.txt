import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)  # Initial radius guess

    def objective_function(x):
        # x is a 52-dimensional vector: first 26*2 for centers, then 26 for radii
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]

        # Objective: maximize the sum of radii
        return -np.sum(radii_flat)

    def constraints(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]

        # Constraint 1: Each circle is fully inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -np.inf

        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -np.inf

        # Constraint 3: All radii are non-negative
        for r in radii_flat:
            if r < 0:
                return -np.inf

        return 0.0

    # Optimization bounds: centers in [0,1], radii in [0, 1]
    bounds = [(0, 1) for _ in range(2*26)] + [(0, 1) for _ in range(26)]

    # Initial guess
    x0 = np.concatenate((centers.flatten(), radii))

    # Use SLSQP method with constraints and bounds
    result = minimize(
        objective_function,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        options={'maxiter': 1000, 'ftol': 1e-9, 'eps': 1e-8}
    )

    if not result.success:
        raise RuntimeError("Optimization failed to converge")

    centers_opt = result.x[:2*26].reshape(26, 2)
    radii_opt = result.x[2*26:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
