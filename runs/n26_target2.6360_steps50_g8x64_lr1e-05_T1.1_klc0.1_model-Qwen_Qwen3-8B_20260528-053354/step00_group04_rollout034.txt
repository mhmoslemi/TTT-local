import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid with some spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        x = (i % 5) / 4 + 0.1
        y = (i // 5) / 4 + 0.1
        radii[i] = 0.1
        centers[i] = [x, y]

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = []
        for i in range(num_circles):
            x = (i % 5) / 4 + 0.1 + radii[i] * np.random.uniform(-0.1, 0.1)
            y = (i // 5) / 4 + 0.1 + radii[i] * np.random.uniform(-0.1, 0.1)
            centers.append([x, y])
        centers = np.array(centers)
        dists = np.linalg.norm(centers[:, np.newaxis] - centers[np.newaxis, :], axis=-1)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dists[i, j] < radii[i] + radii[j]:
                    return False
        return True

    # Use a random optimization approach for demonstration
    def optimize():
        bounds = [(0, 0.5) for _ in range(num_circles)]
        result = minimize(objective, radii, method='L-BFGS-B', bounds=bounds)
        return result

    # Run the optimization
    result = optimize()
    radii = result.x
    centers = np.array([(i % 5) / 4 + 0.1 + radii[i] * np.random.uniform(-0.1, 0.1),
                        (i // 5) / 4 + 0.1 + radii[i] * np.random.uniform(-0.1, 0.1)]
                    for i in range(num_circles))
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
