import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import itertools

def run_packing():
    # Initial placement: evenly spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    idx = 0
    for i in range(6):
        for j in range(5):
            centers[idx] = [i / 4.0, j / 4.0]
            idx += 1
    centers[idx] = [0.5, 0.25]
    centers[idx+1] = [0.5, 0.75]
    centers[idx+2] = [0.25, 0.5]
    centers[idx+3] = [0.75, 0.5]
    centers[idx+4] = [0.25, 0.25]
    centers[idx+5] = [0.25, 0.75]
    centers[idx+6] = [0.75, 0.25]
    centers[idx+7] = [0.75, 0.75]
    centers[idx+8] = [0.5, 0.5]
    centers[idx+9] = [0.125, 0.125]
    centers[idx+10] = [0.875, 0.125]
    centers[idx+11] = [0.125, 0.875]
    centers[idx+12] = [0.875, 0.875]
    centers[idx+13] = [0.375, 0.375]
    centers[idx+14] = [0.625, 0.375]
    centers[idx+15] = [0.375, 0.625]
    centers[idx+16] = [0.625, 0.625]
    centers[idx+17] = [0.25, 0.125]
    centers[idx+18] = [0.25, 0.875]
    centers[idx+19] = [0.75, 0.125]
    centers[idx+20] = [0.75, 0.875]
    centers[idx+21] = [0.125, 0.5]
    centers[idx+22] = [0.875, 0.5]
    centers[idx+23] = [0.375, 0.25]
    centers[idx+24] = [0.625, 0.25]
    centers[idx+25] = [0.375, 0.75]
    centers[idx+26] = [0.625, 0.75]

    # Initial guess for radii
    radii = np.ones(num_circles) * 0.05

    # Objective function to maximize sum of radii, minimize negative sum
    def objective(radii):
        return -np.sum(radii)

    # Constraints: all circles must fit inside the unit square
    def square_constraint(radii):
        return np.array([(radii[i] <= 1 - centers[i, 0]) and (radii[i] <= 1 - centers[i, 1]) for i in range(num_circles)])

    # Constraints: no two circles overlap
    def overlap_constraints(radii):
        dists = pdist(centers)
        overlap = np.array([dists[i] < radii[i] + radii[j] for i, j in itertools.combinations(range(num_circles), 2)])
        return overlap

    # Combined constraints
    def constraints(radii):
        square = square_constraint(radii)
        overlapped = overlap_constraints(radii)
        return np.concatenate([square, overlapped])

    # Bounds on radii
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
