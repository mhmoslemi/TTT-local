import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all equal
    radii = np.full(26, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_func(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
            centers[i, 0] += radii[i]
            centers[i, 1] += radii[i]
        radii = radii.reshape(-1, 1)
        centers = centers.reshape(-1, 2)
        # Check boundaries
        if np.any(centers[:, 0] < radii) or np.any(centers[:, 0] > 1 - radii) or np.any(centers[:, 1] < radii) or np.any(centers[:, 1] > 1 - radii):
            return np.inf
        # Check distances between centers
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Optimization
    res = minimize(objective, radii, method='L-BFGS-B', bounds=[(0, 0.5) for _ in range(26)], constraints=[{'type': 'ineq', 'fun': constraint_func}])

    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
        centers[i, 0] += res.x[i]
        centers[i, 1] += res.x[i]
    sum_radii = res.fun * -1
    return (centers, res.x, sum_radii)
