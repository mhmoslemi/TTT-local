import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    n = 26
    centers = np.array([[i / np.sqrt(n), j / np.sqrt(n)] for i in range(n) for j in range(n)])
    radii = np.full(n, 0.05)

    # Optimization objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: all circles must be fully inside the unit square
    def in_square(radii):
        return np.all(radii <= 0.5)

    # Constraint: no two circles overlap
    def no_overlap(radii):
        dist = pdist(centers)
        return np.all(dist >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Constraints for optimization
    constraints = [
        {'type': 'ineq', 'fun': in_square},
        {'type': 'ineq', 'fun': no_overlap}
    ]

    # Bounds: radii must be non-negative
    bounds = [(0, 0.5) for _ in range(n)]

    # Perform optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, bounds=bounds, constraints=constraints)

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
