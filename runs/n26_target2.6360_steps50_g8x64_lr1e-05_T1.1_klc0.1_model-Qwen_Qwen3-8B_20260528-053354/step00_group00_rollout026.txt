import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial guess: evenly spaced grid
    n = 26
    x = np.linspace(0.1, 0.9, n)
    y = np.linspace(0.1, 0.9, n)
    centers = np.array([x, y]).T
    radii = np.full(n, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_distance(radii, centers):
        # Convert to 1D array for easier manipulation
        r = radii.reshape(-1)
        dists = pdist(centers)
        min_dist = np.min(dists)
        return min_dist - np.sum(r[np.newaxis, :] + r[:, np.newaxis])

    def constraint_bound(radii):
        r = radii.reshape(-1)
        return np.sum(r) - 2.636

    # Setup constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_distance, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_bound}
    ]

    # Bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(n)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    centers_opt = np.copy(centers)
    radii_opt = result.x
    sum_radii_opt = -result.fun

    # Check constraints
    for i in range(n):
        x_i, y_i = centers_opt[i]
        if not (radii_opt[i] <= x_i <= 1 - radii_opt[i] and radii_opt[i] <= y_i <= 1 - radii_opt[i]):
            raise ValueError("Circle is not fully inside the square")

    # Ensure non-overlapping
    dists = pdist(centers_opt)
    for i in range(n):
        for j in range(i + 1, n):
            d = np.linalg.norm(centers_opt[i] - centers_opt[j])
            if d < radii_opt[i] + radii_opt[j]:
                raise ValueError("Circles overlap")

    return (centers_opt, radii_opt, sum_radii_opt)
