import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid with equal spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    # Start with a simple grid layout
    spacing = 1.0 / np.sqrt(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) * spacing
        centers[i, 1] = (i // 5 + 0.5) * spacing
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(num_circles):
            r = radii[i]
            if centers[i, 0] - r < 0 or centers[i, 0] + r > 1 or centers[i, 1] - r < 0 or centers[i, 1] + r > 1:
                return False
        # Ensure circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0.0, 1.0) for _ in range(num_circles)]
    )

    return (centers, result.x, -result.fun)
