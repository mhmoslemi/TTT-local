import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: random points inside the unit square
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    
    def objective(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ...]
        centers_flat = x.reshape(26, 4)[:, :2]
        radii_flat = x.reshape(26, 4)[:, 2]
        # Objective: minimize the negative sum of radii
        return -np.sum(radii_flat)
    
    def constraints(x):
        centers_flat = x.reshape(26, 4)[:, :2]
        radii_flat = x.reshape(26, 4)[:, 2]
        # Each circle is inside the square
        for i in range(26):
            x_i, y_i, r_i = centers_flat[i][0], centers_flat[i][1], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0
    
    # Optimize
    initial_guess = np.concatenate([centers.reshape(-1), radii.reshape(-1)])
    result = minimize(objective, initial_guess, constraints={"type": "ineq", "fun": constraints}, method="SLSQP")
    
    if result.success:
        x_opt = result.x
        centers_opt = x_opt.reshape(26, 4)[:, :2]
        radii_opt = x_opt.reshape(26, 4)[:, 2]
        sum_radii = np.sum(radii_opt)
        return (centers_opt, radii_opt, sum_radii)
    else:
        # Fallback: return initial guess if optimization fails
        return (centers, radii, np.sum(radii))
