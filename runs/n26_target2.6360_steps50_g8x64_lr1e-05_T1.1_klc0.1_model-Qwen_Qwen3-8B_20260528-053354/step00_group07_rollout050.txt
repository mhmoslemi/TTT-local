import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid with small radii
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.05)
    for i in range(26):
        centers[i, 0] = i // 4 + 0.25
        centers[i, 1] = i % 4 + 0.25

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: 1. Circles inside the square, 2. No overlaps
    def constraints(radii):
        # Constraint 1: Circles inside the square
        for i in range(26):
            if radii[i] <= 0:
                continue
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        # Constraint 2: Circles not overlapping
        for i in range(26):
            for j in range(i + 1, 26):
                if radii[i] <= 0 or radii[j] <= 0:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=objective,
            x0=radii,
            method='L-BFGS-B',
            bounds=[(0, 1)] * 26,
            constraints={"type": "ineq", "fun": constraints},
            options={"ftol": 1e-8, "maxiter": 1000}
        )

    # Extract result
    final_radii = result.x
    centers = np.array([[x, y] for (x, y), r in zip(centers, final_radii) if r > 0])
    final_radii = final_radii[final_radii > 0]
    sum_radii = np.sum(final_radii)

    # Ensure correct shapes
    centers = np.resize(centers, (26, 2))
    final_radii = np.resize(final_radii, 26)
    sum_radii = np.sum(final_radii)

    return (centers, final_radii, sum_radii)
