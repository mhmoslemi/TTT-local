import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: equally spaced grid with radii initialized to zero
    num_circles = 26
    initial_centers = np.array([[i / np.sqrt(num_circles), j / np.sqrt(num_circles)] 
                               for i in range(num_circles) for j in range(num_circles)])
    initial_centers = initial_centers[:num_circles]
    initial_radii = np.zeros(num_circles)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum
    
    def constraint_inside(radii):
        centers = np.vstack([initial_centers[i] for i in range(num_circles)])
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return False
        return True
    
    def constraint_no_overlap(radii):
        centers = np.vstack([initial_centers[i] for i in range(num_circles)])
        dists = squareform(pdist(centers))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dists[i, j] < radii[i] + radii[j]:
                    return False
        return True
    
    def constraint_bounds(radii):
        return np.all(radii >= 0)
    
    # Optimization problem
    bounds = [(0, 1) for _ in range(num_circles)]
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap},
        {'type': 'ineq', 'fun': constraint_bounds}
    ]
    
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'disp': False, 'ftol': 1e-10}
    )
    
    if not result.success:
        warnings.warn("Optimization did not converge")
        return (initial_centers, initial_radii, 0.0)
    
    radii = result.x
    centers = np.vstack([initial_centers[i] for i in range(num_circles)])
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
