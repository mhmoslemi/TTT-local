import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)  # we minimize -sum to maximize sum

    def constraint_circle_inside(radii, centers):
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return 1.0
        return 0.0

    def constraint_no_overlap(radii, centers):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return 1.0
        return 0.0

    # Optimization problem
    bounds = [(0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
    ]

    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
