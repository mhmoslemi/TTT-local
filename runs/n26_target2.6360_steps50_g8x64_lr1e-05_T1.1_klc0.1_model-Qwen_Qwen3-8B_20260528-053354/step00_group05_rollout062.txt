import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.05)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add one more center
    radii = np.append(radii, 0.1)

    def sum_radii(radii):
        return np.sum(radii)

    def objective(radii, centers):
        return -sum_radii(radii)

    def constraints(radii, centers):
        n = len(radii)
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    def bounds(radii):
        return [(0, 1) for _ in range(len(radii))]

    # Initial guess for radii
    init_guess = np.full(len(centers), 0.1)

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=lambda x: objective(x, centers),
            x0=init_guess,
            method='SLSQP',
            bounds=bounds(init_guess),
            constraints={'type': 'ineq', 'fun': lambda x: constraints(x, centers)}
        )

    # Extract results
    centers_result = centers
    radii_result = result.x
    sum_radii_result = sum_radii(radii_result)

    return (centers_result, radii_result, sum_radii_result)
