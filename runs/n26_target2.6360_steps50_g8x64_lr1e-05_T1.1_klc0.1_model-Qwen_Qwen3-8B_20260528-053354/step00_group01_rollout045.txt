import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[i*0.1 + 0.05, j*0.1 + 0.05] for i in range(26) for j in range(1)])
    radii = np.full(26, 0.05)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii, centers):
        # Constraint 1: Each circle is entirely within the unit square
        for i in range(len(radii)):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False

        # Constraint 2: No two circles overlap
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    def minimize_with_constraints():
        x0 = radii.copy()
        bounds = [(0, 1.0) for _ in range(26)]
        constraints = []
        def _constraint(x):
            return constraints(x, centers)
        cons = {'type': 'success', 'fun': _constraint}
        constraints.append(cons)

        result = minimize(objective, x0, args=(centers,), bounds=bounds, constraints=constraints, method='SLSQP')
        if not result.success:
            return (centers, radii, np.sum(radii))
        return (centers, result.x, np.sum(result.x))

    # Try to optimize
    result = minimize_with_constraints()
    return result
