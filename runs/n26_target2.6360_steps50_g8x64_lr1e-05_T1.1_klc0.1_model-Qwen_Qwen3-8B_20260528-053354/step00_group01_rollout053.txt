import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced centers
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: same for all
    radii = np.ones(26) * 0.1

    # Optimization parameters
    bounds = [(0.0, 1.0 - 1e-6) for _ in range(26 * 2)]

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[12:])

    # Constraints: circle within the unit square
    def square_constraint(x):
        r = x[12:]
        return np.array([x[i] - r[i] for i in range(12, 12 + 26)] + [1 - x[i] - r[i] for i in range(12, 12 + 26)])

    # Constraints: circles do not overlap
    def no_overlap_constraint(x):
        r = x[12:]
        centers = x[:12].reshape((26, 2))
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (r[:, np.newaxis] + r[np.newaxis, :])

    # Constraint handling
    constraints = []
    constraints.append({'type': 'ineq', 'fun': square_constraint})
    constraints.append({'type': 'ineq', 'fun': lambda x: no_overlap_constraint(x).min()})

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="Optimization failed")
        result = minimize(objective, x0=np.concatenate((centers.flatten(), radii)), bounds=bounds, constraints=constraints)

    final_centers = result.x[:12].reshape((26, 2))
    final_radii = result.x[12:]
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
