import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.array([[i / np.sqrt(num_circles), j / np.sqrt(num_circles)] for i in range(num_circles) for j in range(num_circles)])
    centers = centers[:num_circles]
    radii = np.array([0.1] * num_circles)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        # Constraints: circles must be inside the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            for j in range(num_circles):
                if i == j:
                    continue
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        return True
    
    def optimize():
        bounds = [(0, 1) for _ in range(num_circles)]
        result = minimize(objective, radii, bounds=bounds, constraints=({
            'type': 'ineq',
            'fun': lambda r: 1 if constraints(r) else -np.inf
        }))
        return result
    
    result = optimize()
    radii = result.x
    sum_radii = np.sum(radii)
    
    # Adjust positions slightly to avoid overlaps and ensure they are inside the square
    for i in range(num_circles):
        r = radii[i]
        while True:
            x = np.random.uniform(r, 1 - r)
            y = np.random.uniform(r, 1 - r)
            if all(np.linalg.norm([x, y] - centers[j]) >= radii[j] + r for j in range(num_circles)):
                centers[i] = [x, y]
                break
    
    return (centers, radii, sum_radii)
