import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles evenly in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i] = [0.1 + col * 0.2, 0.1 + row * 0.2]
        radii[i] = 0.05

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Constraint: circles must be inside the unit square
    def constraint_inside(x):
        return np.sum((x[::2] - 0.5)**2 + (x[1::2] - 0.5)**2 - (x[2::2])**2) <= 0

    # Constraint: circles must not overlap
    def constraint_no_overlap(x):
        n = 26
        r = x[2::2]
        centers = np.column_stack([x[::2], x[1::2]])
        min_dist = np.inf
        for i in range(n):
            for j in range(i+1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                min_dist = min(min_dist, dist - (r[i] + r[j]))
        return min_dist >= 0

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(26 * 2)]

    # Initial guess flattened
    x0 = np.concatenate((centers.flatten(), radii))

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimize
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-6)

    # Extract results
    final_centers = np.column_stack((result.x[::2], result.x[1::2]))
    final_radii = result.x[2::2]
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
