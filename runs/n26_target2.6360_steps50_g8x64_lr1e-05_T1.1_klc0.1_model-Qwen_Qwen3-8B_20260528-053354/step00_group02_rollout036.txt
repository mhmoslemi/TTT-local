import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.1)
    # Initial positions in a grid pattern
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.3 + 0.1
        centers[i, 1] = row * 0.3 + 0.1
        radii[i] = 0.1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: each circle is within the unit square
    def constraint_square(radii):
        centers = get_centers(radii)
        return np.concatenate([
            centers[:, 0] + radii >= 1e-6,
            centers[:, 1] + radii >= 1e-6,
            1 - centers[:, 0] + radii >= 1e-6,
            1 - centers[:, 1] + radii >= 1e-6
        ])

    # Constraints: no two circles overlap
    def constraint_overlap(radii):
        centers = get_centers(radii)
        dists = np.sqrt(((centers[:, np.newaxis] - centers[np.newaxis, :])**2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Constraints: radii >= 0
    def constraint_radii(radii):
        return radii

    # Helper function to get centers from radii
    def get_centers(radii):
        # Simple initial grid pattern (should be replaced by optimization result)
        centers = np.zeros((26, 2))
        for i in range(26):
            row = i // 5
            col = i % 5
            centers[i, 0] = col * 0.3 + 0.1
            centers[i, 1] = row * 0.3 + 0.1
        return centers

    # Set up optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            constraints=[
                {'type': 'ineq', 'fun': constraint_square},
                {'type': 'ineq', 'fun': constraint_overlap},
                {'type': 'ineq', 'fun': constraint_radii}
            ],
            method='SLSQP',
            options={'maxiter': 100}
        )

    centers = get_centers(result.x)
    sum_radii = np.sum(result.x)
    return (centers, result.x, sum_radii)
