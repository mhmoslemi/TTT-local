import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: evenly spaced grid in the unit square
    def initial_centers():
        centers = []
        for i in range(26):
            x = i // 5 + (i % 5) / 5
            y = (i % 5) / 5
            centers.append([x, y])
        return np.array(centers)

    centers = initial_centers()
    radii = np.full(26, 0.1)
    sum_radii = np.sum(radii)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    constraints = []

    # Constraint: all circles must be inside the unit square
    def inside_box(radii, centers):
        return np.sum((centers - radii) ** 2) <= 1 - 2 * np.sum(radii)
    # This is not directly usable as a constraint for minimize, so we use a wrapper

    # Constraint: no two circles overlap
    def no_overlap(radii, centers):
        for i, j in itertools.combinations(range(26), 2):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            distance = np.sqrt(dx**2 + dy**2)
            if distance < radii[i] + radii[j]:
                return False
        return True

    # Objective for the optimizer
    def objective_func(radii_flat):
        radii = radii_flat.reshape(26)
        # Check constraints before evaluating
        if not no_overlap(radii, centers):
            return float('inf')
        return -np.sum(radii)

    # Bounds: radii >= 0
    bounds = [(0.0, 1.0) for _ in range(26)]

    # Optimization
    result = minimize(objective_func, radii, method='SLSQP', bounds=bounds)

    # Extract the result
    radii = result.x
    centers = initial_centers()
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
