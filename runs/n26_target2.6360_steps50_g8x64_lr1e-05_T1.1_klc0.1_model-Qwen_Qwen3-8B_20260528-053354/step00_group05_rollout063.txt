import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced grid of circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: 1. Circles are inside the square
    def constraint_inside(radii):
        return np.minimum(radii, 1 - radii)

    # Constraints: 2. No overlap between circles
    def constraint_overlap(centers, radii):
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Bounds for radii: non-negative and not exceeding 0.5 (since the circle must fit in square)
    bounds = [(0, 0.5) for _ in range(num_circles)]

    # Initial radii guesses
    radii_guess = np.full(num_circles, 0.05)

    # Optimization
    result = minimize(
        objective,
        radii_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_inside},
            {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers, radii_guess)}
        ]
    )

    # Extract results
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii = -result.fun

    # Return as required
    return (np.array(centers_final), np.array(radii_final), sum_radii)
