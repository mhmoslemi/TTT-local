import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Number of circles
    n = 26
    
    # Initial guess for centers (randomly placed inside the unit square)
    centers = np.random.rand(n, 2)
    
    # Initial guess for radii (uniform)
    radii = np.full(n, 0.1)
    
    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        # x is a flattened array: [x1, y1, r1, x2, y2, r2, ...]
        x_centers = x[:2*n].reshape(n, 2)
        r = x[2*n:]
        objective_value = -np.sum(r)  # Negative because we use minimize
        return objective_value
    
    # Constraints:
    # 1. Distance between centers >= sum of radii
    # 2. Each circle fits inside the unit square
    # 3. All radii are non-negative
    
    # Constraint 1: No two circles overlap
    cons_overlap = []
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_overlap(x):
                x_centers = x[:2*n].reshape(n, 2)
                r = x[2*n:]
                distance = np.linalg.norm(x_centers[i] - x_centers[j])
                return distance - (r[i] + r[j])
            cons_overlap.append({'type': 'ineq', 'fun': constraint_overlap})
    
    # Constraint 2: Each circle fits inside the square
    cons_boundary = []
    for i in range(n):
        def constraint_boundary(x):
            x_centers = x[:2*n].reshape(n, 2)
            r = x[2*n:]
            return np.array([x_centers[i][0] - r[i], 1 - x_centers[i][0] - r[i],
                             x_centers[i][1] - r[i], 1 - x_centers[i][1] - r[i]])
        cons_boundary.append({'type': 'ineq', 'fun': constraint_boundary})
    
    # Constraint 3: Non-negative radii
    cons_radii = [{'type': 'ineq', 'fun': lambda x, i=i: x[2*n + i]} for i in range(n)]
    
    # Combine all constraints
    cons = cons_overlap + cons_boundary + cons_radii
    
    # Bounds: centers in [0, 1], radii in [0, 0.5]
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 0.5) for _ in range(n)]
    
    # Run optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                     method='SLSQP', bounds=bounds, constraints=cons)
    
    # Extract result
    centers_opt = result.x[:2*n].reshape(n, 2)
    radii_opt = result.x[2*n:]
    sum_radii = -result.fun
    
    return (centers_opt, radii_opt, sum_radii)
