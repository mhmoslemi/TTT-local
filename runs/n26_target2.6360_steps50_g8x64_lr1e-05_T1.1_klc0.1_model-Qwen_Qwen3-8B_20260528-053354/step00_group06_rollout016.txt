import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[i / 5 + 0.25, j / 5 + 0.25] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, np.array([[0.5, 0.25], [0.25, 0.5], [0.75, 0.5], [0.5, 0.75]])])
    centers = centers[:26]
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        n = len(radii)
        centers = np.array([[i / 5 + 0.25, j / 5 + 0.25] for i in range(5) for j in range(5)])
        centers = np.vstack([centers, np.array([[0.5, 0.25], [0.25, 0.5], [0.75, 0.5], [0.5, 0.75]])])
        centers = centers[:26]
        for i in range(n):
            x, y = centers[i]
            if radii[i] < 0:
                return False
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    def objective_func(x):
        return -np.sum(x)

    def constraint_func(x):
        return constraints(x)

    bounds = [(0.0, 0.5) for _ in range(26)]

    result = minimize(objective_func, radii, bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_func})

    radii = result.x
    if constraint_func(radii):
        centers = np.array([[i / 5 + 0.25, j / 5 + 0.25] for i in range(5) for j in range(5)])
        centers = np.vstack([centers, np.array([[0.5, 0.25], [0.25, 0.5], [0.75, 0.5], [0.5, 0.75]])])
        centers = centers[:26]
    else:
        centers = np.array([[random.uniform(0.1, 0.9), random.uniform(0.1, 0.9)] for _ in range(26)])

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
