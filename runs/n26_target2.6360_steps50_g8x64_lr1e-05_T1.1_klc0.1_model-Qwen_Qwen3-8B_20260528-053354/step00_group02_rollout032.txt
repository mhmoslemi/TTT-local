import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    num_circles = 26
    centers = np.array([[i / 7 + 0.5, j / 7 + 0.5] for i in range(7) for j in range(4)])
    radii = np.full(num_circles, 0.05)
    
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize -sum to maximize sum
    
    def constraints(radii_vec):
        radii = radii_vec.copy()
        # Constraint 1: All circles are inside the square
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        # Constraint 2: No two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                if np.sqrt(dx**2 + dy**2) < radii[i] + radii[j]:
                    return False
        return True
    
    # Optimize
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        jac=None,
        bounds=[(0, 0.5) for _ in range(num_circles)]
    )
    
    # Extract results
    final_radii = result.x
    final_centers = centers
    sum_radii = -result.fun
    
    return (final_centers, final_radii, sum_radii)
