import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    spacing = 1.0 / np.sqrt(num_circles)  # Initial spacing for grid placement

    for i in range(num_circles):
        row = i // np.sqrt(num_circles)
        col = i % np.sqrt(num_circles)
        x = col * spacing + 0.5 * spacing
        y = row * spacing + 0.5 * spacing
        centers[i] = [x, y]
        radii[i] = 0.01  # Initial small radius

    def objective_function(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraints for circle positions and radii
        r = np.atleast_1d(radii)
        # Constraint 1: Each circle must be inside the unit square
        for i in range(num_circles):
            if r[i] < 0:
                return np.inf
            if centers[i, 0] - r[i] < 0 or centers[i, 0] + r[i] > 1:
                return np.inf
            if centers[i, 1] - r[i] < 0 or centers[i, 1] + r[i] > 1:
                return np.inf

        # Constraint 2: Distance between any two centers >= sum of their radii
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < r[i] + r[j]:
                    return np.inf

        return 0

    # Optimize radii with constraints
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    res = minimize(objective_function, radii, method='SLSQP', bounds=bounds, constraints={"type": "eq", "fun": constraints})
    radii_opt = res.x

    # Apply soft constraints to ensure the circles stay within the square
    for i in range(num_circles):
        if centers[i, 0] - radii_opt[i] < 0:
            radii_opt[i] = centers[i, 0]
        elif centers[i, 0] + radii_opt[i] > 1:
            radii_opt[i] = 1 - centers[i, 0]
        if centers[i, 1] - radii_opt[i] < 0:
            radii_opt[i] = centers[i, 1]
        elif centers[i, 1] + radii_opt[i] > 1:
            radii_opt[i] = 1 - centers[i, 1]

    # Final check for non-overlapping
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < radii_opt[i] + radii_opt[j]:
                radii_opt[i] = max(radii_opt[i], dist - radii_opt[j])
                radii_opt[j] = max(radii_opt[j], dist - radii_opt[i])

    # Ensure all radii are non-negative
    radii_opt = np.maximum(radii_opt, 0)

    sum_radii = np.sum(radii_opt)

    return (centers, radii_opt, sum_radii)
