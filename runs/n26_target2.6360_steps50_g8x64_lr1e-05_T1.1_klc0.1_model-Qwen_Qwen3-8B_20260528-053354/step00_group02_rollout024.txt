import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: All circles must be within the unit square
    def constraint1(radii):
        return np.sum(radii) - np.sum(np.minimum(centers, 1 - centers)) + 1e-6

    # Constraint 2: No two circles can overlap (distance between centers >= sum of radii)
    def constraint2(radii):
        dists = np.zeros(num_circles)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i] += math.sqrt(dx*dx + dy*dy) - (radii[i] + radii[j])
        return np.min(dists)

    # Initial guess: all radii start at a small value
    initial_radii = np.ones(num_circles) * 0.05

    # Optimization problem
    bounds = [(0, 1) for _ in range(num_circles)]

    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract result
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
