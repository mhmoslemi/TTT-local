import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.full(26, 0.1)

    def objective(r):
        return -np.sum(r)  # Maximize sum of radii by minimizing negative sum

    # Constraints
    def circle_in_square(r, center):
        return r - (1 - center[0] - r)  # r <= 1 - x - r => x + r <= 1
    def circle_in_square_y(r, center):
        return r - (1 - center[1] - r)  # r <= 1 - y - r => y + r <= 1

    # Define nonlinear constraints for non-overlapping
    def non_overlap(r, center):
        return np.array([np.sqrt((centers[i, 0] - center[0])**2 + (centers[i, 1] - center[1])**2) - (r[i] + r[j]) for i in range(26) for j in range(i+1, 26)])

    def constraint_func(r):
        constraints = []
        for i in range(26):
            constraints.append(circle_in_square(r[i], centers[i]))
            constraints.append(circle_in_square_y(r[i], centers[i]))
        return np.array(constraints)

    # Bounds for radius: [0, 0.5]
    bounds = Bounds(np.zeros(26), np.full(26, 0.5))

    # Constraints for non-overlapping circles
    def non_overlap_constraint(r):
        constraints = []
        for i in range(26):
            for j in range(i+1, 26):
                dist = np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2)
                constraints.append(dist - (r[i] + r[j]))
        return np.array(constraints)

    # Optimization
    res = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_func},
            {'type': 'ineq', 'fun': non_overlap_constraint}
        ]
    )

    # Extract result
    centers_result = centers
    radii_result = res.x
    sum_radii = -res.fun  # convert back to sum

    return (centers_result, radii_result, sum_radii)
