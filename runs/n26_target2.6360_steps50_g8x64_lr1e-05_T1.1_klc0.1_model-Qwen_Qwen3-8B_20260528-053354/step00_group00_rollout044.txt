import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1  # Initial radius
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints(radii_vec):
        # Ensure all circles are inside the unit square
        r = radii_vec
        for i in range(num_circles):
            if centers[i, 0] - r[i] < 0 or centers[i, 0] + r[i] > 1 or \
               centers[i, 1] - r[i] < 0 or centers[i, 1] + r[i] > 1:
                return False
        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Optimization with constraints
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 0.5)] * num_circles  # Maximum possible radius is 0.5
    )

    best_radii = result.x
    best_centers = np.copy(centers)
    # Optional refinement with random restarts to avoid local minima
    for _ in range(5):
        # Random perturbation
        perturbation = np.random.rand(num_circles, 2) * 0.1 - 0.05
        best_centers += perturbation
        best_centers = np.clip(best_centers, 0, 1)
        # Check if perturbation improves the result
        if constraints(best_radii):
            result = minimize(
                objective,
                best_radii,
                method='SLSQP',
                constraints=[{'type': 'ineq', 'fun': constraints}],
                bounds=[(0, 0.5)] * num_circles
            )
            best_radii = result.x

    sum_radii = np.sum(best_radii)
    return (best_centers, best_radii, sum_radii)
