import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid for 26 circles
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        x = i / (n_circles - 1)
        y = 0.5  # horizontal spacing, vertical centered
        centers[i] = [x, y]
    
    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # minimize negative of sum
    
    # Constraints
    def constraint_inside(radii, centers):
        return np.array([r <= 1 - r for r in radii])  # x - r >= 0 and x + r <= 1
    
    def constraint_overlap(radii, centers):
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                idx += 1
        return dists  # distances >= 0
    
    # Define the optimization problem
    bounds = [(0.0, 1.0) for _ in range(n_circles)]
    
    # Start optimization with initial guess
    results = minimize(objective, centers.flatten(), method='SLSQP',
                      bounds=bounds, constraints=[
                          {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers)},
                          {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers)},
                      ], jac=None, options={'ftol': 1e-8, 'maxiter': 1000})
    
    # Extract results
    best_radii = results.x
    centers_final = centers
    sum_radii = -results.fun
    
    return (centers_final, best_radii, sum_radii)
