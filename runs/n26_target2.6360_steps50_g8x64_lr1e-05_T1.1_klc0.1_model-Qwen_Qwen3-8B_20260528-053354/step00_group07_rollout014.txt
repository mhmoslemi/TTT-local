import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i // 5 + 0.25
        y = i % 5 + 0.25
        centers[i] = [x, y]

    # Objective: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles within the square and no overlap
    def constraints(radii):
        # Ensure radii are non-negative
        if np.any(radii < 0):
            return np.inf

        # Check if circles are within the unit square
        for i in range(num_circles):
            if centers[i][0] - radii[i] < 0 or centers[i][0] + radii[i] > 1:
                return np.inf
            if centers[i][1] - radii[i] < 0 or centers[i][1] + radii[i] > 1:
                return np.inf

        # Check for overlap between circles
        dists = pdist(centers)
        for d, i, j in zip(dists, range(len(dists)), range(len(dists))):
            if d < radii[i] + radii[j]:
                return np.inf

        return 0

    # Optimization
    result = minimize(
        objective,
        x0=np.ones(num_circles),
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1)] * num_circles
    )

    radii = result.x
    sum_radii = np.sum(radii)
    centers = np.array(centers)

    return (centers, radii, sum_radii)
