import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for circle centers and radii
    # Using a simple grid layout for initial guess of 26 circles
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    for i in range(n):
        row = i // 5
        col = i % 5
        x = 0.2 + col * 0.4
        y = 0.2 + row * 0.4
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective_func(x):
        radii = x[:n]
        centers = x[n:].reshape((n, 2))
        # Apply constraints
        # Check if any circle is outside the unit square
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return float('inf')
        # Check for overlaps between circles
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j] - 1e-8:
                    return float('inf')
        # Objective: maximize the sum of radii
        return -np.sum(radii)

    # Optimization bounds
    bounds = [(0, 0.5)] * n + [(0.0, 1.0)] * (2 * n)  # Radii and centers

    # Initial guess
    x0 = np.concatenate((radii, centers.flatten()))

    # Run optimization
    result = minimize(
        objective_func,
        x0=x0,
        method='L-BFGS-B',
        bounds=bounds,
        tol=1e-6,
        options={'disp': False}
    )

    # Extract optimized values
    optimized_radii = result.x[:n]
    optimized_centers = result.x[n:].reshape((n, 2))

    # Final constraint check and adjustment
    for i in range(n):
        x_i, y_i = optimized_centers[i]
        r_i = optimized_radii[i]
        if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
            optimized_radii[i] = 0
    for i in range(n):
        for j in range(i + 1, n):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            distance = math.hypot(dx, dy)
            if distance < optimized_radii[i] + optimized_radii[j] - 1e-8:
                optimized_radii[i] = 0

    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
