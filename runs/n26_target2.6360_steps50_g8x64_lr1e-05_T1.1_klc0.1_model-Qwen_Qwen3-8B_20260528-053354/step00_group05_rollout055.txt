import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    np.random.seed(42)
    num_circles = 26
    initial_centers = np.random.rand(num_circles, 2)
    initial_radii = np.random.rand(num_circles) * 0.1

    def objective_function(params):
        centers = params[:num_circles*2].reshape(num_circles, 2)
        radii = params[num_circles*2:]
        sum_radii = np.sum(radii)
        # Check if all circles are inside the unit square
        if np.any((centers - radii) < 0) or np.any((1 - centers - radii) < 0):
            return np.inf
        # Check for overlaps between circles
        dist_matrix = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dist_matrix[i, j] < radii[i] + radii[j]:
                    return np.inf
        return -sum_radii  # we minimize -sum_radii to maximize sum_radii

    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0, 1)] * 2)  # x and y coordinates for each center
        bounds.extend([(0, 0.5)])   # radius for each circle

    result = minimize(objective_function, x0=np.concatenate([initial_centers.flatten(), initial_radii]), 
                      bounds=bounds, method='L-BFGS-B', tol=1e-8)

    centers = result.x[:num_circles*2].reshape(num_circles, 2)
    radii = result.x[num_circles*2:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
