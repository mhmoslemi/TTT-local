import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial radius
    initial_radius = 0.05
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        radius = initial_radius * (1 + 0.1 * i)
        centers[i, 0] = 0.1 + col * 0.25
        centers[i, 1] = 0.1 + row * 0.25
        radii[i] = radius

    # Objective function: minimize negative sum of radii to maximize sum
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: Circles are inside the unit square
    def constraint_inside(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i, 0] = 0.1 + (i // 5) * 0.25
            centers[i, 1] = 0.1 + (i % 5) * 0.25
        for i in range(num_circles):
            r = radii[i]
            if not (r <= centers[i, 0] and centers[i, 0] <= 1 - r and
                    r <= centers[i, 1] and centers[i, 1] <= 1 - r):
                return -1
        return 0

    # Constraint 2: No overlap between circles
    def constraint_overlap(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i, 0] = 0.1 + (i // 5) * 0.25
            centers[i, 1] = 0.1 + (i % 5) * 0.25
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0

    # Optimization problem
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'ftol': 1e-10, 'maxiter': 500}
    )

    # Final result
    final_centers = np.zeros((num_circles, 2))
    final_radii = result.x
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        final_centers[i, 0] = 0.1 + col * 0.25
        final_centers[i, 1] = 0.1 + row * 0.25

    sum_radii = np.sum(final_radii)
    return (final_centers, final_radii, sum_radii)
