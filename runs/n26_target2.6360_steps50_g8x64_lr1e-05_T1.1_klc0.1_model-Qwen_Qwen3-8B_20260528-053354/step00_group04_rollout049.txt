import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import math

def run_packing():
    def objective_function(positions, radii):
        return -np.sum(radii)

    def constraints(positions, radii):
        # Constraint 1: All circles are inside the unit square
        # x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1
        # For each circle, check if the bounds are satisfied
        bounds = np.zeros_like(positions)
        for i in range(len(positions)):
            x, y = positions[i]
            r = radii[i]
            bounds[i, 0] = x - r
            bounds[i, 1] = x + r
            bounds[i, 2] = y - r
            bounds[i, 3] = y + r
        return np.concatenate((bounds[::2] >= 0, bounds[1::2] <= 1))

    def non_overlap_constraints(positions, radii):
        dists = pdist(positions)
        return dists - (radii[:, None] + radii[None, :]).flatten()

    def pack_circles():
        # Initial positions: place circles in a grid-like pattern
        initial_positions = np.zeros((26, 2))
        for i in range(26):
            x = i % 5 / 4
            y = i // 5 / 4
            initial_positions[i] = [x, y]

        # Initial radii: start with some small values
        initial_radii = np.full(26, 0.06)

        # Optimization setup
        bounds = []
        for i in range(26):
            x_bounds = [(0.0, 1.0)]
            y_bounds = [(0.0, 1.0)]
            bounds.append(x_bounds + y_bounds)

        # Add constraints
        cons = [
            {'type': 'ineq', 'fun': lambda pos, rad: constraints(pos, rad)},
            {'type': 'ineq', 'fun': lambda pos, rad: non_overlap_constraints(pos, rad)},
        ]

        # Minimize the negative sum of radii to maximize it
        result = minimize(
            objective_function,
            initial_positions,
            args=(initial_radii,),
            method='SLSQP',
            bounds=bounds,
            constraints=cons,
            jac=None,
            tol=1e-6,
            options={'maxiter': 1000}
        )

        centers = result.x.reshape(-1, 2)
        radii = result.fun * -1  # Since objective is negative sum of radii
        sum_radii = np.sum(radii)
        return centers, radii, sum_radii

    centers, radii, sum_radii = pack_circles()
    return (centers, radii, sum_radii)
