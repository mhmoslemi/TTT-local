import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import math

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i / np.sqrt(num_circles)
        y = (i % np.sqrt(num_circles)) / np.sqrt(num_circles)
        centers[i] = [x, y]

    # Objective function: maximize sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraints
    def constraint_inside(radii, centers):
        # All circles fully inside the unit square
        return np.array([r <= 1 - r for r in radii])

    def constraint_no_overlap(radii, centers):
        # Distance between centers >= sum of radii
        dist = squareform(pdist(centers))
        sum_radii = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dist - sum_radii

    # Optimization problem
    bounds = [(0, 1.0) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (radii, centers)}
    ]

    # Initial radii guess (all same)
    initial_radii = np.ones(num_circles) * 0.1

    # Optimization
    result = minimize(objective, initial_radii, bounds=bounds, constraints=constraints, method='SLSQP')
    radii = result.x
    sum_radii = np.sum(radii)
    centers = centers  # Use initial guess for centers as they are adjusted during optimization

    return (centers, radii, sum_radii)
