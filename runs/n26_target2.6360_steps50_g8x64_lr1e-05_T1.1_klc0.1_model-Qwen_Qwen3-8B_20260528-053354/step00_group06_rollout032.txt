import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05
    centers -= 0.5
    centers *= 0.5
    radii = np.clip(radii, 0, 0.5)

    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize sum

    def constraints(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        constraints = []
        # Ensure circles are inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf
        # Ensure circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0.0

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: x, y in [0, 1], radius >= 0 and <= 0.5 (approximate max radius)
    bounds = []
    for i in range(26):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 0.5))  # r

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    x_opt = result.x
    centers_opt = x_opt[:2*26].reshape(26, 2)
    radii_opt = x_opt[2*26:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
