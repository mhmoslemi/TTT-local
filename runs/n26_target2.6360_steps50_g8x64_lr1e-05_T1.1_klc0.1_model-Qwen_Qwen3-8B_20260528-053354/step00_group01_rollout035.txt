import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: place circles in a grid with small radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2) * 0.8 + 0.1  # Random positions inside the square
    radii = np.full(num_circles, 0.02)  # Initial small radii

    def objective(radii):
        return -np.sum(radii)  # Maximize sum by minimizing negative sum

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        for i in range(num_circles):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        # Constraint 2: No two circles overlap
        for i, j in itertools.combinations(range(num_circles), 2):
            r1, r2 = radii[i], radii[j]
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < r1 + r2:
                return False
        return True

    def bounds_func(radii):
        return [(0, 1) for _ in range(num_circles)]

    # Optimization
    result = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        method='SLSQP',
        bounds=bounds_func,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: 1 - constraints(x)}
        ]
    )

    radii_opt = result.x
    sum_radii = -result.fun

    # Final check for constraints
    centers_opt = centers.copy()
    for i in range(num_circles):
        r = radii_opt[i]
        centers_opt[i, 0] = max(0.0 + r, min(1.0 - r, centers_opt[i, 0]))
        centers_opt[i, 1] = max(0.0 + r, min(1.0 - r, centers_opt[i, 1]))

    # Final constraint check
    for i, j in itertools.combinations(range(num_circles), 2):
        r1, r2 = radii_opt[i], radii_opt[j]
        dx = centers_opt[i, 0] - centers_opt[j, 0]
        dy = centers_opt[i, 1] - centers_opt[j, 1]
        dist = np.sqrt(dx ** 2 + dy ** 2)
        if dist < r1 + r2:
            radii_opt[i] = max(0, radii_opt[i] - (dist - r1 - r2 + 1e-8))
            radii_opt[j] = max(0, radii_opt[j] - (dist - r1 - r2 + 1e-8))

    return (centers_opt, radii_opt, sum_radii)
