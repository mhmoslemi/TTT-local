import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(num_circles, 0.05)

    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

    def constraint_func(radii_vec):
        centers_new = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers_new[i, 0] = (i % 5 + 0.5) / 5 + np.random.normal(0, 0.05)
            centers_new[i, 1] = (i // 5 + 0.5) / 5 + np.random.normal(0, 0.05)
        radii = radii_vec
        # Check if circles are inside the unit square
        if np.any(radii < 0):
            return -np.inf
        if np.any(radii > 1 - np.min(np.abs(centers_new), axis=1)):
            return -np.inf
        # Check if circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_new[i, 0] - centers_new[j, 0]
                dy = centers_new[i, 1] - centers_new[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return np.sum(radii)

    # Create bounds and constraints
    bounds = [(0, 1 - 0.01) for _ in range(num_circles)]
    constraints = {'type': 'ineq', 'fun': constraint_func}

    # Optimize
    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Final results
    centers_final = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers_final[i, 0] = (i % 5 + 0.5) / 5 + np.random.normal(0, 0.05)
        centers_final[i, 1] = (i // 5 + 0.5) / 5 + np.random.normal(0, 0.05)
    radii_final = result.x
    sum_radii = result.fun

    return (centers_final, radii_final, sum_radii)
