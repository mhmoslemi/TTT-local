import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced grid (approximate circle centers)
    num_circles = 26
    centers = np.array([[i / np.sqrt(num_circles), j / np.sqrt(num_circles)] for i in range(num_circles) for j in range(num_circles)])
    centers = centers[:26]
    radii = np.ones(num_circles) * 0.1  # Initial guess for radii

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints: 1. Circles fully inside square
    #               2. Circles not overlapping
    #               3. All radii non-negative

    # Constraint 1: r <= x <= 1 - r, r <= y <= 1 - r
    def constraint_inside(x):
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:]
        return np.concatenate([
            centers_flat[:, 0] - radii_flat,
            1 - centers_flat[:, 0] - radii_flat,
            centers_flat[:, 1] - radii_flat,
            1 - centers_flat[:, 1] - radii_flat
        ])

    # Constraint 2: distance between centers >= sum of radii
    def constraint_overlap(x):
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:]
        distances = pdist(centers_flat)
        radii_sum = np.array([radii_flat[i] + radii_flat[j] for i, j in enumerate(range(num_circles))])
        return distances - radii_sum

    # Constraint 3: radii >= 0
    def constraint_non_negative(x):
        return x[2 * num_circles:]

    # Setup optimization problem
    x0 = np.concatenate([centers.reshape(-1), radii])

    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]

    # Run optimization
    result = minimize(objective, x0, method='SLSQP', constraints=constraints, tol=1e-8)

    # Extract results
    centers = result.x[:2 * num_circles].reshape(num_circles, 2)
    radii = result.x[2 * num_circles:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
