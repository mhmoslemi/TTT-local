import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles in a grid pattern
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 5) for y in np.linspace(0.1, 0.9, 5)])
    radii = np.full(25, 0.05)
    
    # Objective function to maximize the sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)
    
    # Constraints: circles must be inside the unit square and not overlap
    def constraint_inside(radii_flat):
        return np.sum((radii_flat <= 1 - centers[:, 0]) * (radii_flat <= 1 - centers[:, 1]))
    
    def constraint_overlap(radii_flat):
        n = len(centers)
        dists = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[idx] = dist - (radii_flat[i] + radii_flat[j])
                idx += 1
        return np.sum(dists > 0)
    
    # Initial guess for radii
    radii_initial = np.full(25, 0.05)
    
    # Setup optimization problem
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=UserWarning)
        result = minimize(
            objective,
            radii_initial,
            method='L-BFGS-B',
            bounds=[(0, 1) for _ in range(25)],
            constraints=[
                {'type': 'ineq', 'fun': constraint_inside},
                {'type': 'ineq', 'fun': constraint_overlap}
            ]
        )
    
    # Extract final results
    final_radii = result.x
    final_centers = centers[:len(final_radii)]
    sum_radii = np.sum(final_radii)
    return (final_centers, final_radii, sum_radii)
