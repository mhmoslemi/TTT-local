import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial random placement of circles
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: 1) circles are inside the square, 2) circles do not overlap
    def constraint_inside(radii):
        return np.sum((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]))

    def constraint_overlap(radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum((dists >= radii[:, np.newaxis] + radii[np.newaxis, :]))

    # Optimization problem
    def constraints(radii):
        return [
            {'type': 'ineq', 'fun': lambda r: constraint_inside(r)},
            {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
        ]

    # Bounds on radii
    bounds = [(0.0, 0.5) for _ in range(26)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Final result
    centers_result = centers.copy()
    radii_result = result.x
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
