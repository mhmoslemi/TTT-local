import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid with equal spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.05)  # Small initial radii

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(centers, radii):
        constraints = []
        # Constraint 1: Each circle is inside the unit square
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, x, y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x, y: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x, y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x, y: 1 - y - r})
        # Constraint 2: Circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                def dist(radii, centers):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist_sq = dx**2 + dy**2
                    return dist_sq - (radii[i] + radii[j])**2
                constraints.append({'type': 'ineq', 'fun': dist})
        return constraints

    def bounds(radii):
        return [(0.0, 1.0) for _ in range(num_circles)]

    # Setup for optimization
    res = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        method='SLSQP',
        jac=None,
        constraints=constraints(centers, radii),
        bounds=bounds(radii),
        tol=1e-8,
        options={'maxiter': 1000}
    )

    # Extract results
    centers = np.array(res.x.reshape((num_circles, 2)))
    radii = res.x
    sum_radii = -res.fun  # Since we maximized with negative sum

    return (centers, radii, sum_radii)
