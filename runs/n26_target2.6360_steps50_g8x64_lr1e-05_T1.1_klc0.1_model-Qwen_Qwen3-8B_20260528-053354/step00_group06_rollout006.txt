import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.ones(26) * 0.1

    # Objective function: minimize the negative sum of radii
    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraint 1: circles are fully inside the unit square
    def constraint_inside(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        inside = np.all((radii_flat <= centers_flat[:, 0]) & (radii_flat <= 1 - centers_flat[:, 0]) &
                        (radii_flat <= centers_flat[:, 1]) & (radii_flat <= 1 - centers_flat[:, 1]))
        return inside

    # Constraint 2: circles do not overlap
    def constraint_overlap(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        overlaps = np.zeros(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return False
        return True

    # Convert constraints to scipy format
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds for x and y coordinates (0 <= x, y <= 1), radii must be non-negative
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])
    bounds.extend([(0.0, 1.0) for _ in range(26)])  # Radii are part of the bounds

    # Run the optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints=cons)

    # Extract the result
    centers_opt = result.x[:2*26].reshape(26, 2)
    radii_opt = result.x[2*26:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
