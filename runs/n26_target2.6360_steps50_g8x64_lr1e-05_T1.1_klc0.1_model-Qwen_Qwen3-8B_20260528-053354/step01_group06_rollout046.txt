import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place 26 circles with small radii in a grid
    num_circles = 26
    centers = np.array([[x / 7.0, y / 7.0] for y in range(7) for x in range(7)])
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Each circle must be inside the unit square
        radii = np.clip(radii, 0, 1)
        for i in range(num_circles):
            if radii[i] > 1:
                return np.inf
        # Constraint 2: Circles must not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Prepare initial guess and bounds
    bounds = [(0, 1) for _ in range(num_circles)]
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
        options={"disp": False, "ftol": 1e-8, "maxiter": 1000}
    )

    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
