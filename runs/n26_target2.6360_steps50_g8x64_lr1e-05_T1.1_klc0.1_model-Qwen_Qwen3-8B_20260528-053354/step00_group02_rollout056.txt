import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        r = radii.copy()
        for i in range(num_circles):
            # Each circle must be inside the unit square
            if r[i] < 0:
                return np.inf
            if centers[i, 0] - r[i] < 0 or centers[i, 0] + r[i] > 1:
                return np.inf
            if centers[i, 1] - r[i] < 0 or centers[i, 1] + r[i] > 1:
                return np.inf

        # No overlap between circles
        dists = pdist(centers)
        for j in range(len(dists)):
            i1 = j // num_circles
            i2 = j % num_circles
            if i1 == i2:
                continue
            if dists[j] < r[i1] + r[i2]:
                return np.inf

        return 0

    res = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 0.5) for _ in range(num_circles)]
    )

    centers = np.array(centers)
    radii = res.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
