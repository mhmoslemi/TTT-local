import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 + 0.5
        y = i // 5 + 0.5
        centers[i] = [x, y]

    # Initial radii: all equal for starting point
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure each circle is within the unit square
        for i in range(26):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Ensure no circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization loop with constraints
    for _ in range(100):
        # Optimization function
        def objective_func(x):
            return -np.sum(x)

        # Optimization constraints
        cons = []
        for i in range(26):
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: x[i]})  # Radii non-negative
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - x[i] - centers[i, 0]})
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - x[i] - centers[i, 1]})
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: centers[i, 0] - x[i]})
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: centers[i, 1] - x[i]})

        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                cons.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: dist - x[i] - x[j]})

        # Optimization bounds
        bounds = [(0, 1) for _ in range(26)]

        # Run optimization
        result = minimize(objective_func, radii, method='SLSQP', bounds=bounds, constraints=cons)
        # Check if constraint is satisfied
        if constraints(result.x):
            radii = result.x
        else:
            break

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
