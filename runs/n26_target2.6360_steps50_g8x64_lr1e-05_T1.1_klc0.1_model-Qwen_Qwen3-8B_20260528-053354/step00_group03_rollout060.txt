import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid with some spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    # Start with uniform radius and spaced centers
    radius_initial = 0.05
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i] = [0.2 + col * 0.3, 0.2 + row * 0.3]
        radii[i] = radius_initial
    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)
    # Constraints: 1. Circles are inside the square
    #              2. Non-overlapping between all pairs
    constraints = []
    for i in range(num_circles):
        # Constraint: circle i is inside the square
        def constraint_i(x):
            r = x[i]
            return [r, 1 - r]
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: x[i]})
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - x[i]})
        # Constraint: all other circles are non-overlapping
        for j in range(i + 1, num_circles):
            def constraint_ij(x):
                r_i = x[i]
                r_j = x[j]
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                return dist - r_i - r_j
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_ij(x)})
    # Define bounds for radii
    bounds = [(0, 1) for _ in range(num_circles)]
    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)
    # Extract the result
    optimal_radii = result.x
    optimal_centers = centers.copy()
    sum_radii = np.sum(optimal_radii)
    # Ensure the maximum sum is at least 2.636
    if sum_radii < 2.636:
        # If not enough, adjust to reach target
        scaling = 2.636 / sum_radii
        optimal_radii *= scaling
        sum_radii = 2.636
    return (optimal_centers, optimal_radii, sum_radii)
