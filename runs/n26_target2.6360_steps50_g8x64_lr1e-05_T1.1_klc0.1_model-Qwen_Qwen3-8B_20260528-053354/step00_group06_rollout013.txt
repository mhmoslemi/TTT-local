import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid pattern with some spacing
    centers = np.zeros((26, 2))
    radius = 0.05
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) * 0.4
        centers[i, 1] = (i // 5 + 0.5) * 0.4
    radii = np.full(26, radius)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum

    # Constraint: each circle must fit in the unit square
    def in_square(radii, centers):
        return np.sum((radii - (1 - centers[:, 0]) < 0) + (radii - centers[:, 0] < 0) + 
                      (radii - (1 - centers[:, 1]) < 0) + (radii - centers[:, 1] < 0))

    # Constraint: circles must not overlap
    def no_overlap(radii, centers):
        dist = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 + 
                        (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        return np.sum(dist < radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization bounds
    bounds = [(0.0, 0.5) for _ in range(26)]

    # Initial guess for radii
    initial_guess = np.full(26, 0.05)

    # Optimization problem
    def constraints(radii):
        if in_square(radii, centers) > 0 or no_overlap(radii, centers) > 0:
            return np.inf
        return 0

    res = minimize(objective, initial_guess, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Update centers based on radii
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = np.random.uniform(radii[i], 1 - radii[i])
        centers[i, 1] = np.random.uniform(radii[i], 1 - radii[i])

    return centers, res.x, np.sum(res.x)

if __name__ == "__main__":
    centers, radii, sum_radii = run_packing()
    print("Sum of radii:", sum_radii)
