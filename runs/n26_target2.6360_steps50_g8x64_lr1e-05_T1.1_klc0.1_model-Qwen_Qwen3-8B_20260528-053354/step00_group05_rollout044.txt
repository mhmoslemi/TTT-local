import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x_center = i % 5 / 4
            y_center = i // 5 / 4
            centers[i] = [x_center, y_center]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1] or \
               radii[i] < centers[i, 0] or radii[i] < centers[i, 1]:
                return False
        return True

    def objective_func(x):
        radii[:] = x
        return -np.sum(x)

    # Optimization
    result = minimize(objective_func, radii, method='SLSQP', bounds=[(0, 1 - 0.1) for _ in range(num_circles)],
                      constraints=[{'type': 'ineq', 'fun': lambda x: constraints(x)}])

    if not result.success:
        raise RuntimeError("Optimization failed")

    return (centers, result.x, np.sum(result.x))
