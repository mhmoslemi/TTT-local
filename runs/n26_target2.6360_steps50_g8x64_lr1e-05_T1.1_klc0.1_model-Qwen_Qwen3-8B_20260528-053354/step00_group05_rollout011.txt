import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid of 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i / np.sqrt(num_circles)
        y = np.sqrt(num_circles - i) / np.sqrt(num_circles)
        centers[i] = [x, y]

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_centers(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x = i / np.sqrt(num_circles)
            y = np.sqrt(num_circles - i) / np.sqrt(num_circles)
            centers[i] = [x, y]
        return centers

    def constraint_radii(radii):
        return np.sum(radii)

    def constraint_overlap(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x = i / np.sqrt(num_circles)
            y = np.sqrt(num_circles - i) / np.sqrt(num_circles)
            centers[i] = [x, y]
        dist = cdist(centers, centers)
        diag = np.diag(dist)
        np.fill_diagonal(dist, np.inf)
        return np.min(dist - radii[:, np.newaxis] - radii[np.newaxis, :])

    # Optimization setup
    bounds = [(0.0, 0.5) for _ in range(num_circles)]  # max radius is 0.5
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': lambda r: np.sum(r) - 2.636},  # ensure sum is at least target
        {'type': 'eq', 'fun': constraint_radii},
    ]

    # Run optimization
    result = minimize(
        fun=objective,
        x0=np.zeros(num_circles),
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        options={'ftol': 1e-8, 'maxiter': 500}
    )

    # Extract results
    radii = result.x
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i / np.sqrt(num_circles)
        y = np.sqrt(num_circles - i) / np.sqrt(num_circles)
        centers[i] = [x, y]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
