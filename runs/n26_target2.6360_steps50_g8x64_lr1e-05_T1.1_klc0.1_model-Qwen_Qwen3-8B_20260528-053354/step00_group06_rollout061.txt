import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers (centered in the square)
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial guess for radii (same for all)
    radii = np.ones(26) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraint1(x):
        # Constraint 1: Each circle is within the unit square
        return np.minimum(x[:, 0] - x[:, 2], 1 - x[:, 0] - x[:, 2], x[:, 1] - x[:, 2], 1 - x[:, 1] - x[:, 2])

    def constraint2(x):
        # Constraint 2: No two circles overlap
        dist = np.sqrt(np.sum((x[:, np.newaxis, :2] - x[np.newaxis, :, :2])**2, axis=2))
        return dist - (x[:, np.newaxis, 2] + x[np.newaxis, :, 2])

    # Combine all constraints
    constraints = []

    # Constraint 1: All circles are within the square
    def constraint1_func(x):
        x_centers = x[:, :2]
        x_radii = x[:, 2]
        return np.minimum(x_centers[:, 0] - x_radii, 1 - x_centers[:, 0] - x_radii, x_centers[:, 1] - x_radii, 1 - x_centers[:, 1] - x_radii)
    constraints.append({'type': 'ineq', 'fun': constraint1_func})

    # Constraint 2: No two circles overlap
    def constraint2_func(x):
        x_centers = x[:, :2]
        x_radii = x[:, 2]
        dist = np.sqrt(np.sum((x_centers[:, np.newaxis, :] - x_centers[np.newaxis, :, :])**2, axis=2))
        return dist - (x_radii[:, np.newaxis] + x_radii[np.newaxis, :])
    constraints.append({'type': 'ineq', 'fun': constraint2_func})

    # Optimization
    bounds = [(0, 1) for _ in range(26*2)] + [(0, 1) for _ in range(26)]  # x, y for each center, and radius for each circle

    result = minimize(objective, np.hstack((centers.ravel(), radii)), method='SLSQP', constraints=constraints, bounds=bounds)

    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26*2:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
