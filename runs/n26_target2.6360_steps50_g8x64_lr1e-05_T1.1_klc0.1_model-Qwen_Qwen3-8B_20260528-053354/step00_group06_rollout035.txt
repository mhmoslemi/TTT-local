import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.array([(i / 5 + 0.1, j / 5 + 0.1) for i in range(6) for j in range(6)])
    centers = centers[:26]  # Take first 26 circles
    radii = np.full(26, 0.1)  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Each circle is inside the square
        for i in range(26):
            x, y = centers[i]
            if radii[i] < 0:
                return -np.inf
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return -np.inf

        # Constraint 2: No circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}])
    optimal_radii = result.x
    sum_radii = -result.fun
    return (centers, optimal_radii, sum_radii)
