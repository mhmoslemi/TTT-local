import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid with some spacing
    num_circles = 26
    initial_centers = np.array([[i / 6 + 0.05, j / 6 + 0.05] for j in range(6) for i in range(6)])
    initial_centers = initial_centers[:num_circles]
    initial_radii = np.ones(num_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x, y = initial_centers[i]
            centers[i, 0] = x + np.random.uniform(-0.1, 0.1)
            centers[i, 1] = y + np.random.uniform(-0.1, 0.1)
        # Ensure circles are inside the unit square
        for i in range(num_circles):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Ensure circles don't overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    result = minimize(constraints, initial_radii, method='L-BFGS-B', bounds=[(0, 1) for _ in range(num_circles)])
    radii = result.x
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x, y = initial_centers[i]
        centers[i, 0] = x + np.random.uniform(-0.1, 0.1)
        centers[i, 1] = y + np.random.uniform(-0.1, 0.1)
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
