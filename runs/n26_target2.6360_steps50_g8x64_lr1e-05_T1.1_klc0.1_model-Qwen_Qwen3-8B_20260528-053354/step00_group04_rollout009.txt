import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid (simplified for demonstration)
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # We want to maximize sum of radii
    
    def constraints(radii):
        # Create centers based on some logic (simplified for this example)
        centers = np.zeros((26, 2))
        for i in range(26):
            x = i // 5 + 0.1
            y = i % 5 + 0.1
            centers[i] = [x, y]
        # Ensure circles are inside the unit square
        for i in range(26):
            if radii[i] < 0:
                return np.inf
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf
        # Ensure circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)  # Return negative sum for minimization
    
    # Optimization settings
    bounds = [(0, 0.5) for _ in range(26)]
    res = minimize(objective, radii, method='L-BFGS-B', bounds=bounds)
    
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i // 5 + 0.1
        y = i % 5 + 0.1
        centers[i] = [x, y]
    
    sum_radii = res.fun
    return (centers, -res.success * res.fun, sum_radii)
