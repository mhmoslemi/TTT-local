import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (26, 2))
    radii = np.random.uniform(0.01, 0.05, 26)

    def objective(x):
        # x is a flattened array: first 52 entries are centers, next 26 entries are radii
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        # Objective: minimize the negative sum of radii (maximize sum of radii)
        return -np.sum(radii_flat)

    def constraints(x):
        # Constraint 1: All circles are within the unit square
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: No two circles overlap
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0

    # Setup optimization problem
    x0 = np.column_stack((centers, radii)).ravel()
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y positions
    for _ in range(26):
        bounds.extend([(0.0, 0.5)])  # radii (limit to 0.5 to ensure circles fit in square)
    
    # Optimization settings
    res = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6,
        options={'maxiter': 1000}
    )

    if res.success:
        centers_opt = res.x[:52].reshape(26, 2)
        radii_opt = res.x[52:]
        sum_radii = -res.fun
        return (centers_opt, radii_opt, sum_radii)
    else:
        # Fallback to initial guess if optimization fails
        return (centers, radii, np.sum(radii))
