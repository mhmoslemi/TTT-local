import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    centers = np.array([[i / np.sqrt(num_circles), j / np.sqrt(num_circles)] for i in range(num_circles) for j in range(num_circles)])
    centers = centers[:num_circles]
    radii = np.full(num_circles, 0.05)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        # Ensure circles are fully inside the unit square
        for i in range(num_circles):
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return np.inf
        # Ensure no two circles overlap
        dists = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0
    
    # Optimization
    bounds = Bounds(np.zeros(num_circles), np.full(num_circles, 1.0))
    result = minimize(objective, radii, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints}, bounds=bounds, tol=1e-6)
    
    # Extract the result
    centers_final = centers
    radii_final = result.x
    sum_radii_final = -result.fun
    
    return (centers_final, radii_final, sum_radii_final)
