import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    def initial_guess(n_circles):
        centers = np.zeros((n_circles, 2))
        for i in range(n_circles):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
        return centers

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_center_inside(x, y, r):
        return x - r >= 0 and x + r <= 1 and y - r >= 0 and y + r <= 1

    def constraint_no_overlap(centers, radii):
        n = len(centers)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Combine all constraints
    def constraints(centers, radii):
        n = len(radii)
        constraints = []
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda x, y, r: x - r, 'args': (x, y, r)})
            constraints.append({'type': 'ineq', 'fun': lambda x, y, r: 1 - x - r, 'args': (x, y, r)})
            constraints.append({'type': 'ineq', 'fun': lambda x, y, r: y - r, 'args': (x, y, r)})
            constraints.append({'type': 'ineq', 'fun': lambda x, y, r: 1 - y - r, 'args': (x, y, r)})
        return constraints

    # Initial guess for positions
    centers = initial_guess(26)
    # Initial guess for radii (we try to spread them evenly)
    radii = np.ones(26) * 0.1

    # Bounds on radii: non-negative
    bounds = [(0, 1) for _ in range(26)]

    # Optimization problem
    def objective_wrapper(x):
        centers[:len(x)//2] = x[:len(x)//2].reshape(-1, 2)
        radii = x[len(x)//2:]
        return -np.sum(radii)

    def constraint_wrapper(x):
        centers = x[:len(x)//2].reshape(-1, 2)
        radii = x[len(x)//2:]
        if not constraint_no_overlap(centers, radii):
            return np.inf
        return 0

    x0 = np.concatenate([centers.reshape(-1), radii])
    result = minimize(objective_wrapper, x0, bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_wrapper}, method='SLSQP')

    centers_opt = result.x[:len(x0)//2].reshape(-1, 2)
    radii_opt = result.x[len(x0)//2:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
