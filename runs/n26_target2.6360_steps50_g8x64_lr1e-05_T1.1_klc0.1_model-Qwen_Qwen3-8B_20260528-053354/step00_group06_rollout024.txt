import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import json

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.4 + 0.2
        centers[i, 1] = row * 0.4 + 0.2

    # Initial radii: all 0.1
    radii = np.full(num_circles, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square
    def in_square(radii, centers):
        return [np.all((r <= centers[i, 0] <= 1 - r) & (r <= centers[i, 1] <= 1 - r)) for i, r in enumerate(radii)]

    # Constraints: circles must not overlap
    def no_overlap(radii, centers):
        n = len(radii)
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
        return [dists[i, j] >= radii[i] + radii[j] for i in range(n) for j in range(i + 1, n)]

    # Function to pack circles
    def pack_circles(radii, centers):
        constraints = []
        constraints.append({'type': 'ineq', 'fun': lambda r: np.array(in_square(r, centers))})
        constraints.append({'type': 'ineq', 'fun': lambda r: np.array(no_overlap(r, centers))})
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            constraints=constraints,
            bounds=[(0, 1 - 0.01) for _ in range(num_circles)]
        )
        return result.x

    # Run the optimization
    final_radii = pack_circles(radii, centers)
    final_centers = centers.copy()

    # Return the result
    return (final_centers, final_radii, -objective(final_radii, final_centers))
