import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.array([[(i / 6 + 0.5) / 7, (j / 6 + 0.5) / 7] for i in range(7) for j in range(4)])
    radii = np.full(26, 0.1)

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii_flat):
        return -np.sum(radii_flat)

    # Constraints
    def constraint_radius(centers, radii_flat):
        for i in range(len(radii_flat)):
            rx, ry = centers[i]
            if rx < radii_flat[i] or rx > 1 - radii_flat[i] or ry < radii_flat[i] or ry > 1 - radii_flat[i]:
                return -np.inf
        return 0

    def constraint_overlap(center1, r1, center2, r2):
        dx = center1[0] - center2[0]
        dy = center1[1] - center2[1]
        dist = np.sqrt(dx**2 + dy**2)
        if dist < r1 + r2:
            return -np.inf
        return 0

    centers_flat = centers.flatten()

    bounds = [(0, 0.5) for _ in range(52)]

        'ftol': 1e-6,
        'maxiter': 1000
    }
