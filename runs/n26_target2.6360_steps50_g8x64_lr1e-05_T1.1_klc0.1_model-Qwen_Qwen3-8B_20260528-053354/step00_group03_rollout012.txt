import numpy as np
from scipy.optimize import minimize
from itertools import combinations

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    initial_centers = np.array([[i / 5 + 0.1, j / 5 + 0.1] for j in range(5) for i in range(5)])
    initial_centers = np.vstack((initial_centers, [0.5, 0.75], [0.5, 0.25], [0.25, 0.5], [0.75, 0.5]))  # Add a few more centers
    initial_centers = initial_centers[:26]  # Ensure exactly 26 circles
    
    # Initial radii guess: all equal
    initial_radii = np.full(num_circles, 0.1)
    
    # Objective function: maximize the sum of the radii (minimize negative sum)
    def objective(x):
        radii = x[:num_circles]
        centers = x[num_circles:]
        centers = centers.reshape(num_circles, 2)
        return -np.sum(radii)
    
    # Constraints
    constraints = []
    
    # Constraint: each circle is inside the square
    def inside_box(x):
        radii = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)
        return np.array([r <= min(c[0], 1 - c[0], c[1], 1 - c[1]) for r, c in zip(radii, centers)])
    
    constraints.append({'type': 'ineq', 'fun': inside_box})
    
    # Constraint: circles do not overlap
    def no_overlap(x):
        radii = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])
    
    constraints.append({'type': 'ineq', 'fun': no_overlap})
    
    # Bounds: radii >= 0
    bounds = [(0, None) for _ in range(num_circles)] + [(0, 1) for _ in range(num_circles * 2)]
    
    # Optimization
    result = minimize(objective, np.concatenate((initial_radii, initial_centers.flatten())), 
                     method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)
    
    # Extract results
    radii = result.x[:num_circles]
    centers = result.x[num_circles:].reshape(num_circles, 2)
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
