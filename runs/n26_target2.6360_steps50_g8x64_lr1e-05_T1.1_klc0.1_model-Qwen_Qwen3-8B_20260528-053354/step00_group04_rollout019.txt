import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid with equal spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col + 0.5
        centers[i, 1] = row + 0.5

    # Initial guess for radii: uniform
    radii = np.ones(num_circles) * 0.1
    sum_radii = np.sum(radii)

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        nonlocal sum_radii
        radii = x
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints: circles must be inside the unit square
    def bounds(x):
        return [(0, 1 - x[i]) for i in range(num_circles)]

    # Constraints: no overlap between circles
    def constraints(x):
        r = x
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Optimization setup
    bounds = [(0, 1) for _ in range(num_circles)]
    cons = {'type': 'ineq', 'fun': constraints}
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    radii_opt = result.x
    centers_opt = centers
    sum_radii_opt = np.sum(radii_opt)

    # Check if optimization was successful
    if not result.success:
        warnings.warn("Optimization did not converge successfully.")
        radii_opt = radii
        sum_radii_opt = sum_radii

    return (centers_opt, radii_opt, sum_radii_opt)
