import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import warnings

def run_packing():
    # Initial guess: evenly spaced circles
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
        radii[i] = 0.05
    
    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii
    
    def constraints(radii, centers):
        # Constraint 1: All circles fit inside the square
        def square_constraint(radii):
            r = radii.reshape(-1, 1)
            center_x = centers[:, 0].reshape(-1, 1)
            center_y = centers[:, 1].reshape(-1, 1)
            return np.concatenate([
                r - center_x + r,
                1 - center_x - r,
                r - center_y + r,
                1 - center_y - r
            ]).flatten()
        
        # Constraint 2: No overlapping between circles
        def circle_constraint(radii):
            r = radii.reshape(-1, 1)
            center_x = centers[:, 0].reshape(-1, 1)
            center_y = centers[:, 1].reshape(-1, 1)
            dists = np.abs(center_x[:, np.newaxis] - center_x) + np.abs(center_y[:, np.newaxis] - center_y)
            return np.concatenate([dists - r[:, np.newaxis] - r[np.newaxis, :], np.zeros_like(r)])
        
        return np.concatenate([square_constraint(radii), circle_constraint(radii)])
    
    def bounds(radii):
        return [(0.0, 1.0) for _ in range(26)]
    
    result = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraints(r, centers)},
        ],
        bounds=bounds(radii),
        method='SLSQP',
        options={'maxiter': 100, 'ftol': 1e-6, 'disp': False}
    )
    
    if not result.success:
        warnings.warn("Optimization did not converge")
        return (centers, radii, np.sum(radii))
    
    best_radii = result.x
    best_centers = centers.copy()
    sum_radii = np.sum(best_radii)
    
    return (best_centers, best_radii, sum_radii)
