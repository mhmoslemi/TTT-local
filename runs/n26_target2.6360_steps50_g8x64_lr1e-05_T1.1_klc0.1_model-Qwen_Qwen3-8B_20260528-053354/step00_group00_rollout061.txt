import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = 0.1 + col * 0.2
        y = 0.1 + row * 0.2
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x, y = centers[i]
            radius = radii[i]
            # Ensure circles are inside the unit square
            if not (radius <= x <= 1 - radius and radius <= y <= 1 - radius):
                return np.inf
            # Check for overlap with other circles
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, bounds=bounds, constraints={"type": "ineq", "fun": constraints}, method="SLSQP")

    # Update centers using a better initialization strategy (simple grid)
    centers = np.zeros((26, 2))
    for i in range(26):
        row = i // 5
        col = i % 5
        x = 0.1 + col * 0.2
        y = 0.1 + row * 0.2
        centers[i] = [x, y]

    # Fill in the optimized radii
    radii = result.x

    # Final check and validation
    for i in range(26):
        x, y = centers[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            raise ValueError("Circle out of bounds")
        for j in range(i + 1, 26):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j]:
                raise ValueError("Circles overlap")

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
