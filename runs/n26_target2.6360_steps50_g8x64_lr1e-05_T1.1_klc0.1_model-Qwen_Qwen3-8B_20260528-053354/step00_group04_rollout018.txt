import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    initial_centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        initial_centers[i] = [col / 4 + 0.1, row / 4 + 0.1]

    # Objective function: maximize sum of radii (minimize -sum_radii)
    def objective(x):
        radii = x[:num_circles]
        centers = x[num_circles:].reshape((num_circles, 2))
        sum_r = np.sum(radii)
        # Check if all circles are within the unit square
        if not np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1])):
            return float('inf')
        # Check if all circles are non-overlapping
        dists = cdist(centers, centers)
        np.fill_diagonal(dists, np.inf)
        if not np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :]):
            return float('inf')
        return -sum_r  # Minimize -sum_radii to maximize sum_radii

    # Initial guess: radii are all 0.1, centers are initial_centers
    initial_guess = np.concatenate([np.full(num_circles, 0.1), initial_centers.flatten()])

    # Bounds: radii must be non-negative, centers must stay within [0, 1]
    bounds = []
    for _ in range(num_circles):
        bounds.append((0, 1))  # radius bounds
    for _ in range(num_circles):
        bounds.append((0, 1))  # x-coordinate bounds
    for _ in range(num_circles):
        bounds.append((0, 1))  # y-coordinate bounds

    # Constraints
    constraints = []
    # Ensure circles are inside the unit square
    for i in range(num_circles):
        def constraint_circle_inside(x, i=i):
            radii = x[:num_circles]
            centers = x[num_circles:].reshape((num_circles, 2))
            return radii[i] - (1 - centers[i, 0])  # r <= 1 - x
        constraints.append({'type': 'ineq', 'fun': constraint_circle_inside})
        def constraint_circle_inside_y(x, i=i):
            radii = x[:num_circles]
            centers = x[num_circles:].reshape((num_circles, 2))
            return radii[i] - (1 - centers[i, 1])  # r <= 1 - y
        constraints.append({'type': 'ineq', 'fun': constraint_circle_inside_y})

    # Optimize
    result = minimize(objective, initial_guess, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract the result
    radii = result.x[:num_circles]
    centers = result.x[num_circles:].reshape((num_circles, 2))
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
