import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: spread circles evenly in the square
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 + 0.5 / 5
        y = i // 5 + 0.5 / 5
        centers[i] = [x, y]

    # Initial radii: all 0.1
    radii = np.full(26, 0.1)

    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

    def constraints(radii_vec):
        # Constraints: radii must be non-negative
        constraints = []
        for i in range(26):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})  # r >= 0

        # Constraints: circles must be inside the unit square
        for i in range(26):
            r = radii_vec[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (centers[i, 0] - r)})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (1 - centers[i, 0] - r)})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (centers[i, 1] - r)})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (1 - centers[i, 1] - r)})

        # Constraints: circles must not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                def dist_con(r):
                    r_i = r[i]
                    r_j = r[j]
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    return dist - r_i - r_j
                constraints.append({'type': 'ineq', 'fun': dist_con})

        return constraints

    result = minimize(objective, radii, constraints=constraints(radii))
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
