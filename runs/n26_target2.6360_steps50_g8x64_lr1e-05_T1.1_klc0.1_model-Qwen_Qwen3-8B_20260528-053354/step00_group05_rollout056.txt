import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = col / 4 + 0.125
        y = row / 4 + 0.125
        centers[i] = [x, y]
        radii[i] = 0.05

    # Objective function to minimize
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints: 1. Each circle is inside the square
    #              2. No overlap between any two circles
    def constraints_func(x):
        # x is a flat array [r0, r1, ..., r25, x0, x1, ..., x25, y0, y1, ..., y25]
        radii = x[:26]
        centers = x[26:].reshape(26, 2)

        # Constraint 1: Circles inside the square
        for i in range(26):
            x_i, y_i = centers[i]
            if not (radii[i] <= x_i <= 1 - radii[i] and radii[i] <= y_i <= 1 - radii[i]):
                return np.inf
        # Constraint 2: Circles don't overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0.0

    # Flatten centers and radii for optimization
    def objective_flat(x):
        radii = x[:26]
        centers = x[26:].reshape(26, 2)
        return -np.sum(radii)

    def constraints_flat(x):
        return constraints_func(x)

    # Bounds: radii must be non-negative, centers must be in [0,1]
    bounds = []
    for i in range(26):
        bounds.append((0, 1))  # radii[i]
        bounds.append((0, 1))  # x[i]
        bounds.append((0, 1))  # y[i]

    # Initial guess
    x0 = np.concatenate((radii, centers.flatten()))

    # Optimization
    result = minimize(objective_flat, x0, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints_flat}, tol=1e-6)

    # Extract results
    radii_opt = result.x[:26]
    centers_opt = result.x[26:].reshape(26, 2)

    # Check for success
    if result.success:
        sum_radii = np.sum(radii_opt)
    else:
        sum_radii = 0.0

    return (centers_opt, radii_opt, sum_radii)
