import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i + 0.5) / np.sqrt(num_circles)
        centers[i, 1] = 0.5

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        # Constraint 1: All circles are inside the square
        r = radii.reshape(-1, 1)
        in_bounds = (r <= (1 - r)[:, np.newaxis]) & (r <= (1 - r)[:, np.newaxis])
        in_bounds = in_bounds.all()
        if not in_bounds:
            return np.inf  # Penalize if any circle goes out of bounds

        # Constraint 2: No overlapping between any two circles
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=-1))
        radii_mat = radii[:, np.newaxis] + radii[np.newaxis, :]
        overlap = dist < radii_mat
        if overlap.any():
            return np.inf  # Penalize if any circles overlap

        return 0

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0=np.ones(num_circles), constraints={"type": "ineq", "fun": constraints})

    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
