import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced circles in grid
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4  # 5 columns, spaced 0.25 apart
        y = i // 5     # 5 rows
        centers[i] = [x, y]
    
    # Optimization variables: flattening centers and radii
    x0 = np.concatenate([centers.reshape(-1), np.ones(26)])
    
    # Objective function: maximize the sum of radii
    def objective(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        return -np.sum(radii)  # minimize negative sum
    
    # Constraints: all circles are inside the unit square
    def square_constraint(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        return np.array([1 - np.max(centers[:, 0] + radii),
                         1 - np.max(centers[:, 1] + radii),
                         1 - np.min(centers[:, 0] - radii),
                         1 - np.min(centers[:, 1] - radii)])
    
    # Constraints: no overlapping circles
    def non_overlap_constraint(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        dists = squareform(pdist(centers))
        min_dist = np.min(dists)
        sum_radii = np.sum(radii)
        return sum_radii - min_dist
    
    # Optimization bounds
    bounds = []
    for i in range(26):
        bounds.append((0.0, 1.0))  # x-coordinates
        bounds.append((0.0, 1.0))  # y-coordinates
        bounds.append((0.0, 1.0))  # radii
    
    # Set up the optimization problem
    constraints = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': non_overlap_constraint}
    ]
    
    # Run the optimization with a reasonable tolerance
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-5)
    
    # Extract the optimized centers and radii
    centers_opt = result.x[:52].reshape(26, 2)
    radii_opt = result.x[52:]
    sum_radii_opt = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii_opt)
