import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for centers (random points in the unit square)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)  # Initial radii

    def objective(radii):
        return -np.sum(radii)  # We minimize the negative sum to maximize the sum

    def constraints(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ...]
        # Reshape into centers and radii
        x_flat = x.reshape(26, 4)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        # Constraint 1: All circles are inside the unit square
        dist_to_edge = np.minimum(centers[:, 0], 1 - centers[:, 0]) + np.minimum(centers[:, 1], 1 - centers[:, 1])
        dist_to_edge -= radii
        constraints1 = np.concatenate([dist_to_edge, -radii])
        # Constraint 2: No two circles overlap
        dist_between_centers = cdist(centers, centers)
        dist_between_centers -= np.outer(radii, np.ones(26)) - np.outer(np.ones(26), radii)
        constraints2 = np.concatenate([dist_between_centers.flatten()])
        # Combine all constraints
        return np.concatenate([constraints1, constraints2])

    def constraint_func(x):
        return constraints(x)

    # Bounds for each parameter (x, y, r)
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])
    
    # Optimization settings
    initial_guess = np.concatenate([centers.reshape(26, 2), radii.reshape(26, 1)], axis=1).reshape(-1)
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_func},
        tol=1e-6
    )

    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26*2:].reshape(26)
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
