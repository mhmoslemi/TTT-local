import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Define initial guess for circle centers and radii
    # Start with evenly spaced grid in the unit square
    centers = np.array([[(i / 5 + 0.25, j / 5 + 0.25) for j in range(5)] for i in range(5)])
    centers = centers.reshape(-1, 2)
    radii = np.full(25, 0.1)  # Initial guess for radii

    # Optimization function to minimize negative sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraint: circles are inside the unit square
    def inside_box(radii, centers):
        return np.all((centers - radii[:, np.newaxis]) >= 0) & np.all((1 - centers - radii[:, np.newaxis]) >= 0)

    # Constraint: no two circles overlap
    def no_overlap(radii, centers):
        distances = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.all(distances >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Define the optimization problem
    bounds = [(0, 0.5) for _ in range(26)]  # Radii cannot exceed 0.5

    # Initial guess with 26 circles: 25 from grid + 1 extra in center
    # Add an extra circle in the center with radius 0.1
    center_center = np.array([[0.5, 0.5]])
    centers = np.vstack([centers, center_center])
    radii = np.hstack([radii, [0.1]])

    def objective_with_bounds(x):
        # x is a 1D array of radii
        radii = x
        return -np.sum(radii)

    def constraint_inside_box(x):
        # Check if all circles are inside the unit square
        radii = x
        return inside_box(radii, centers)

    def constraint_no_overlap(x):
        # Check if no circles overlap
        radii = x
        return no_overlap(radii, centers)

    # Set up constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside_box},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization
    result = minimize(
        objective_with_bounds,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'maxiter': 1000, 'ftol': 1e-8}
    )

    # Extract results
    best_radii = result.x
    best_centers = centers
    best_sum = np.sum(best_radii)

    # Ensure all circles are valid
    if best_sum < 2.636:
        # If the result is below the target, return the initial guess
        best_radii = np.repeat(0.1, 26)
        best_centers = np.array([[(i / 5 + 0.25, j / 5 + 0.25) for j in range(5)] for i in range(5)])
        best_centers = best_centers.reshape(-1, 2)
        best_centers = np.vstack([best_centers, [0.5, 0.5]])
        best_sum = np.sum(best_radii)

    return (best_centers, best_radii, best_sum)
