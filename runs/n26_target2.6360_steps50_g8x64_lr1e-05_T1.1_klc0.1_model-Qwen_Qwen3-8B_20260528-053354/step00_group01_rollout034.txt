import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (random positions in the unit square)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Initial radii, small to avoid overlap

    # Objective function: minimize the negative sum of radii (maximize sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint: each circle must be fully inside the unit square
    def inside_box(radii, centers):
        return np.sum((centers - radii[:, None] > 0) * (1 - centers - radii[:, None] > 0))

    # Constraint: no two circles overlap
    def no_overlap(radii, centers):
        dists = np.sqrt(np.sum((centers[:, None, :] - centers[None, :, :])**2, axis=2))
        return np.sum(dists >= radii[:, None] + radii[None, :])

    # Combine constraints
    def constraints(radii):
        # Box constraint
        box_constraints = []
        for i in range(26):
            box_constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - 0.0})
            box_constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1.0 - r[i]})
        # Non-overlap constraint
        box_constraints.append({'type': 'ineq', 'fun': lambda r: no_overlap(r, centers)})
        return box_constraints

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=constraints(radii))
    optimal_radii = result.x
    optimal_centers = centers  # Centers are not optimized in this basic approach

    # Check constraints to avoid invalid solutions
    valid = True
    for i in range(26):
        if optimal_radii[i] < 0:
            valid = False
            break
        if optimal_radii[i] > 1.0:
            valid = False
            break
    if not valid:
        optimal_radii = np.zeros(26)
        optimal_centers = np.random.rand(26, 2)

    # Ensure no overlap
    dists = np.sqrt(np.sum((optimal_centers[:, None, :] - optimal_centers[None, :, :])**2, axis=2))
    for i in range(26):
        for j in range(i + 1, 26):
            if dists[i, j] < optimal_radii[i] + optimal_radii[j]:
                optimal_radii[i] = max(optimal_radii[i], (dists[i, j] - optimal_radii[j]) * 0.5)
                optimal_radii[j] = max(optimal_radii[j], (dists[i, j] - optimal_radii[i]) * 0.5)

    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
