import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial guess: evenly spaced grid with radius 0.05
    num_circles = 26
    grid = np.linspace(0.05, 0.95, 5)
    centers = np.array([[x, y] for y in grid for x in grid])
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        centers = np.array([[x, y] for y in grid for x in grid])
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization with constraints
    bounds = [(0, 0.5) for _ in range(num_circles)]
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=({
            'type': 'ineq',
            'fun': lambda r: np.array([constraints(r)])
        }),
        options={'disp': False}
    )

    # Final centers and radii
    centers = np.array([[x, y] for y in grid for x in grid])
    radii = result.x
    sum_radii = -result.fun

    # Ensure all circles are within the unit square
    for i in range(num_circles):
        if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                radii[i] <= centers[i, 1] <= 1 - radii[i]):
            radii[i] = 0.0

    return (centers, radii, sum_radii)
