import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid with small radii
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 5) for y in np.linspace(0.1, 0.9, 5)])
    radii = np.full(25, 0.05)  # 25 circles for 5x5 grid
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more circle in the center
    radii = np.append(radii, 0.05)  # Adjust to make 26 circles

    def objective(radii):
        return -np.sum(radii)  # We minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are within the unit square
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(5)])
        centers = np.vstack([centers, [0.5, 0.5]])
        r = radii.reshape(-1, 1)
        for i in range(len(centers)):
            if 2*r[i] > 1:
                return False
        # Ensure no two circles overlap
        for i in range(len(centers)):
            for j in range(i+1, len(centers)):
                dist = np.sqrt((centers[i,0] - centers[j,0])**2 + (centers[i,1] - centers[j,1])**2)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Optimization
    result = minimize(objective, radii, method='L-BFGS-B', bounds=[(0, 1) for _ in range(26)])
    radii_opt = result.x
    sum_radii = -result.fun

    # Final check and adjustment
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [0.5, 0.5]])
    for i in range(len(centers)):
        if 2*radii_opt[i] > 1:
            radii_opt[i] = 0.5
    for i in range(len(centers)):
        for j in range(i+1, len(centers)):
            dist = np.sqrt((centers[i,0] - centers[j,0])**2 + (centers[i,1] - centers[j,1])**2)
            if dist < radii_opt[i] + radii_opt[j]:
                radii_opt[i] = (dist - radii_opt[j]) / 2
                radii_opt[j] = (dist - radii_opt[i]) / 2

    sum_radii = np.sum(radii_opt)

    return (centers, radii_opt, sum_radii)
