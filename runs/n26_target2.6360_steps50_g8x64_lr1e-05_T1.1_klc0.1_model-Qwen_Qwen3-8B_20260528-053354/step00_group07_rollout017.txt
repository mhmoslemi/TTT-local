import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: Place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Generate initial circular positions
    for i in range(num_circles):
        x = (i % 5 + 0.5) / 5.0
        y = (i // 5 + 0.5) / 5.0
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial small radius
    
    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraint 1: Each circle is inside the unit square
    def constraint_inside(radii, centers):
        return np.array([r <= (1 - r) for r in radii])
    
    # Constraint 2: No two circles overlap
    def constraint_overlap(radii, centers):
        dists = np.zeros(num_circles)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i] += np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
        return dists
    
    # Define bounds for radii
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    
    # Optimization problem
    def objective_wrapper(x):
        radii[:] = x
        return -np.sum(x)
    
    def constraint_inside_wrapper(x):
        return np.array([r <= (1 - r) for r in x])
    
    def constraint_overlap_wrapper(x):
        dists = np.zeros(num_circles)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i] += np.sqrt(dx**2 + dy**2) - (x[i] + x[j])
        return dists
    
    # Setup optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside_wrapper},
        {'type': 'ineq', 'fun': constraint_overlap_wrapper}
    ]
    
    # Run optimization
    try:
        result = minimize(objective_wrapper, radii, method='SLSQP',
                          bounds=bounds, constraints=constraints, tol=1e-6)
    except:
        warnings.warn("Optimization failed, returning initial guess")
        result = {'x': radii}
    
    radii = result['x']
    sum_radii = np.sum(radii)
    
    # Final centers: we keep the initial grid positions
    # (you would typically optimize the centers as well for better results)
    
    return (centers, radii, sum_radii)
