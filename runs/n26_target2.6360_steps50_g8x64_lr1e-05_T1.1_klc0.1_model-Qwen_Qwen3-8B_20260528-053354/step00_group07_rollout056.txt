import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place 26 circles in a grid layout
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Place circles in a grid: 5 rows x 5 columns with 1 extra circle
    for i in range(num_circles):
        if i < 25:
            row = i // 5
            col = i % 5
            centers[i, 0] = col * 0.3
            centers[i, 1] = row * 0.3
        else:
            # Place the last circle at the center
            centers[i, 0] = 0.5
            centers[i, 1] = 0.5
        radii[i] = 0.05  # Initial guess for radius
    
    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraint 1: Each circle must be fully inside the unit square
    def constraint_inside(radii, centers):
        return np.array([r <= x - r and r <= y - r and x + r <= 1 and y + r <= 1 for r, x, y in zip(radii, centers[:, 0], centers[:, 1])])
    
    # Constraint 2: No two circles overlap
    def constraint_overlap(radii, centers):
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return np.array([dists[i, j] >= radii[i] + radii[j] for i in range(num_circles) for j in range(i + 1, num_circles)])
    
    # Constraints as a list of dictionaries
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers), 'args': (centers,)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers), 'args': (centers,)}
    ]
    
    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(num_circles)]
    
    # Optimization
    result = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )
    
    # Extract results
    radii_opt = result.x
    centers_opt = centers.copy()  # Keep initial centers as the final centers (could be updated in future)
    sum_radii_opt = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii_opt)
