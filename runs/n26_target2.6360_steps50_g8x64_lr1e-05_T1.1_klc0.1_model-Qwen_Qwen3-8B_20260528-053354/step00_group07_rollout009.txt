import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2)  # random initial centers
    radii = np.full(num_circles, 0.1)  # initial radius

    def objective(radii, centers):
        return -np.sum(radii)  # maximize sum of radii

    def constraints(radii, centers):
        # Constraint 1: all circles are inside the unit square
        # x - r >= 0, 1 - x - r >= 0, same for y
        inside = np.all((centers - radii[:, np.newaxis] >= 0) &
                        (1 - centers - radii[:, np.newaxis] >= 0), axis=1)
        if not np.all(inside):
            return -np.inf  # invalid solution

        # Constraint 2: no overlapping between any two circles
        # distance between centers >= sum of radii
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return -np.inf  # overlapping occurs
        return 0

    # Combine radii and centers into a single vector
    initial_guess = np.concatenate([radii, centers.flatten()])
    bounds = [(0, 0.5)] * num_circles + [(0, 1)] * (2 * num_circles)

    def objective_func(x):
        radii = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)
        return -np.sum(radii)

    def constraint_func(x):
        radii = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)
        # Check if all circles are inside the square
        inside = np.all((centers - radii[:, np.newaxis] >= 0) &
                        (1 - centers - radii[:, np.newaxis] >= 0), axis=1)
        if not np.all(inside):
            return -np.inf
        # Check for overlaps
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    result = minimize(
        objective_func,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_func}
    )

    # Extract final radii and centers
    final_radii = result.x[:num_circles]
    final_centers = result.x[num_circles:].reshape(num_circles, 2)
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
