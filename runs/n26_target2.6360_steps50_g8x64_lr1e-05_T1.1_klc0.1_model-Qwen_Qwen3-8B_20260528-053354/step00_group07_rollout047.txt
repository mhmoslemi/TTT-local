import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.05)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col / 4.0 + 0.05
        centers[i, 1] = row / 4.0 + 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i, 0] = col / 4.0 + 0.05
            centers[i, 1] = row / 4.0 + 0.05
        constraints = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    constraints.append(dist - radii[i] - radii[j])
        return np.array(constraints)

    def bounds(radii):
        return [(0.0, 1.0 - 1e-8) for _ in range(num_circles)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints, 'args': ()},
        bounds=bounds(radii),
        options={'ftol': 1e-9, 'maxiter': 200}
    )

    return (centers, result.x, np.sum(result.x))
