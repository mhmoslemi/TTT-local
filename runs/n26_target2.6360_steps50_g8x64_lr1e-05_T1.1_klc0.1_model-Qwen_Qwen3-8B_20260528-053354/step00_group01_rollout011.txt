import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers: equally spaced grid
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        x = i / (n_circles - 1)
        y = (i // 5) / 5
        centers[i] = [x, y]
    radii = np.ones(n_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are within the unit square
        r = radii
        for i in range(n_circles):
            if r[i] > 1 - 2 * r[i]:  # r <= 1 - 2r => r <= 1/3
                return False
        # Ensure no overlaps
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Define bounds for radii
    bounds = [(0.0, 1.0 / 3) for _ in range(n_circles)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=({'type': 'ineq', 'fun': constraints}),
        tol=1e-6
    )

    # Final radii and sum
    final_radii = result.x
    final_sum = np.sum(final_radii)
    final_centers = centers

    return (final_centers, final_radii, final_sum)
