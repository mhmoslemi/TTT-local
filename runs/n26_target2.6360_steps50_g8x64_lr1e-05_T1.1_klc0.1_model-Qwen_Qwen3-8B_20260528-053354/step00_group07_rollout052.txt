import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[(i + 0.5) / 5, (j + 0.5) / 5] for i in range(5) for j in range(5)])
    initial_radii = np.array([0.05] * 25)  # 25 circles in 5x5 grid + one more circle

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            if i < 25:
                centers[i] = initial_centers[i]
            else:
                # Place the 26th circle at the center of the square
                centers[i] = [0.5, 0.5]
        radii = np.clip(radii, 0, 1)
        # Check if circles are within the square
        if np.any(radii > 1 - np.min(np.vstack([centers[:, 0], 1 - centers[:, 0], centers[:, 1], 1 - centers[:, 1]]), axis=0)):
            return np.inf
        # Check if circles are non-overlapping
        dists = cdist(centers, centers)
        np.fill_diagonal(dists, np.inf)
        if np.any(dists < radii[:, np.newaxis] + radii[np.newaxis, :]):
            return np.inf
        return -np.sum(radii)  # Return negative sum for minimize

    # Optimization
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1) for _ in range(26)]
    )

    centers = np.zeros((26, 2))
    for i in range(26):
        if i < 25:
            centers[i] = initial_centers[i]
        else:
            centers[i] = [0.5, 0.5]
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
