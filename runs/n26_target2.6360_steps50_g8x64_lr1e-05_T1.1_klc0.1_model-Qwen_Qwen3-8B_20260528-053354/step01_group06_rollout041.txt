import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equally spaced grid of 26 points
    centers = np.empty((26, 2))
    radii = np.zeros(26)
    
    # Initial grid of points
    grid_x, grid_y = np.meshgrid(np.linspace(0.1, 0.9, 5), np.linspace(0.1, 0.9, 5))
    centers[:25] = np.stack((grid_x.flatten(), grid_y.flatten()), axis=-1)
    centers[25] = [0.5, 0.5]
    radii[25] = 0.05  # Start with small radius for the central circle
    
    # Objective function: minimize negative sum of radii
    def objective(x):
        # x is a flat array of centers and radii
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:]
        
        # Check if all circles are inside the square
        if any(r > 0 and (c[0] - r < 0 or c[0] + r > 1 or c[1] - r < 0 or c[1] + r > 1) for c, r in zip(centers_flat, radii_flat)):
            return np.inf  # Penalize by large value
        
        # Check for overlap between circles
        sum_overlap = 0
        for i in range(26):
            for j in range(i + 1, 26):
                distance = np.linalg.norm(centers_flat[i] - centers_flat[j])
                sum_overlap += max(0, distance - (radii_flat[i] + radii_flat[j]))
        
        return -np.sum(radii_flat) + 100 * sum_overlap  # Add penalty for overlaps

    # Bounds for centers (x and y in [0,1]) and radii (r >= 0)
    bounds = []
    for i in range(26):
        bounds.extend([(0, 1), (0, 1)])
        bounds.append((0, 0.5))  # Max radius is 0.5 to fit in square

    # Initial guess
    x0 = np.concatenate((centers.reshape(26 * 2), radii))

    # Perform optimization
    result = minimize(objective, x0, method='L-BFGS-B', bounds=bounds, tol=1e-6)

    # Extract final centers, radii, and sum of radii
    centers_final = result.x[:26 * 2].reshape(26, 2)
    radii_final = result.x[26 * 2:]
    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
