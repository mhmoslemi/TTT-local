import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii (non-overlapping random points)
    def initial_guess(n_circles):
        centers = np.random.rand(n_circles, 2)
        radii = np.random.rand(n_circles)
        for i in range(n_circles):
            r = radii[i]
            centers[i, 0] = np.clip(centers[i, 0] + r, 0 + r, 1 - r)
            centers[i, 1] = np.clip(centers[i, 1] + r, 0 + r, 1 - r)
        return centers, radii

    n_circles = 26
    centers, radii = initial_guess(n_circles)

    def objective(x):
        # x contains [x1, y1, r1, x2, y2, r2, ...]
        # Reshape x to centers and radii arrays
        flat_centers = x[:n_circles * 2].reshape(n_circles, 2)
        flat_radii = x[n_circles * 2:].reshape(n_circles)
        # Ensure all radii are non-negative
        flat_radii = np.clip(flat_radii, 0, None)
        # Check if all circles are within the unit square
        max_radius = np.max(flat_radii)
        if max_radius > 1:
            return np.inf
        for i in range(n_circles):
            r_i = flat_radii[i]
            if not (r_i <= flat_centers[i, 0] <= 1 - r_i and r_i <= flat_centers[i, 1] <= 1 - r_i):
                return np.inf
        # Check for overlapping circles
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = flat_centers[i, 0] - flat_centers[j, 0]
                dy = flat_centers[i, 1] - flat_centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < flat_radii[i] + flat_radii[j]:
                    return np.inf
        # Objective is to minimize the negative of the sum of radii (maximize sum)
        return -np.sum(flat_radii)

    # Flatten initial guess into a vector for optimization
    x0 = np.concatenate([centers.reshape(n_circles * 2), radii])

    # Set bounds for all variables: centers [0,1], radii [0, 1]
    bounds = []
    for i in range(n_circles * 2):
        bounds.append((0, 1))
    for i in range(n_circles):
        bounds.append((0, 1))

    result = minimize(objective, x0, method='L-BFGS-B', bounds=bounds, tol=1e-8, options={'maxiter': 1000})

    # Extract final centers and radii
    final_centers = result.x[:n_circles * 2].reshape(n_circles, 2)
    final_radii = result.x[n_circles * 2:].reshape(n_circles)
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
