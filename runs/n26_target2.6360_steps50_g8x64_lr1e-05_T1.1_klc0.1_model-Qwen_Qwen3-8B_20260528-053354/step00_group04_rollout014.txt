import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: distribute circles in a grid with equal spacing
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Optimization variables: radii
    initial_radii = np.full(n_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii <= 1 - centers) & (radii <= centers), axis=1)
        if not np.all(inside):
            return np.inf

        # Constraint 2: No two circles overlap
        dists = pdist(centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        overlap = np.any(dists < radii_sum)
        if overlap:
            return np.inf

        return 0

    # Set up optimization
    bounds = [(0, 1) for _ in range(n_circles)]
    result = minimize(
        objective,
        initial_radii,
        method='L-BFGS-B',
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        tol=1e-6
    )

    # Extract result
    radii = result.x
    centers = np.array(centers)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
