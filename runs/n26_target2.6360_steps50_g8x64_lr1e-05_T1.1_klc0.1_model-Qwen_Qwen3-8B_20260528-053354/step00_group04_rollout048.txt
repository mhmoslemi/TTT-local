import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid points
    num_circles = 26
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    # Remove the center point to make space for the last circle
    centers = np.vstack((centers, [0.5, 0.5]))  # Add the center point
    centers = centers[:26]  # Ensure exactly 26 circles
    
    # Initial radius guess: uniform
    radii = np.full(num_circles, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Constraint 1: Each circle is inside the unit square
        # r_i <= x_i - r_i and r_i <= 1 - x_i - r_i
        # Similarly for y_i
        dist = np.sum((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2, axis=2)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                r_i = radii[i]
                r_j = radii[j]
                if dist[i, j] < (r_i + r_j) ** 2:
                    return np.inf  # Overlapping
        return 0
    
    def constraint_func(radii):
        return constraints(radii)
    
    def objective_func(radii):
        return -np.sum(radii)
    
    res = minimize(objective_func, radii, method='SLSQP',
                   constraints={'type': 'ineq', 'fun': constraint_func},
                   bounds=[(0, 0.5) for _ in range(num_circles)])
    
    if res.success:
        sum_radii = -res.fun
        return (centers, res.x, sum_radii)
    else:
        raise ValueError("Optimization failed to find a valid packing.")
