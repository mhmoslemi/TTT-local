import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[x / 5 + 0.1, y / 5 + 0.1] for x in range(5) for y in range(5)])
    centers = np.vstack([centers, np.array([[0.2, 0.75]], dtype=float), np.array([[0.75, 0.2]], dtype=float),
                         np.array([[0.75, 0.75]], dtype=float), np.array([[0.25, 0.5]], dtype=float),
                         np.array([[0.5, 0.25]], dtype=float), np.array([[0.5, 0.75]], dtype=float),
                         np.array([[0.75, 0.5]], dtype=float), np.array([[0.25, 0.25]], dtype=float),
                         np.array([[0.25, 0.75]], dtype=float), np.array([[0.75, 0.25]], dtype=float),
                         np.array([[0.5, 0.5]], dtype=float), np.array([[0.4, 0.5]], dtype=float),
                         np.array([[0.6, 0.5]], dtype=float), np.array([[0.5, 0.4]], dtype=float),
                         np.array([[0.5, 0.6]], dtype=float), np.array([[0.3, 0.5]], dtype=float),
                         np.array([[0.7, 0.5]], dtype=float), np.array([[0.5, 0.3]], dtype=float),
                         np.array([[0.5, 0.7]], dtype=float), np.array([[0.2, 0.5]], dtype=float),
                         np.array([[0.8, 0.5]], dtype=float), np.array([[0.5, 0.2]], dtype=float),
                         np.array([[0.5, 0.8]], dtype=float), np.array([[0.2, 0.2]], dtype=float),
                         np.array([[0.8, 0.8]], dtype=float)])
    radii = np.full(26, 0.07)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers[:len(radii)] = np.array([[(x + np.random.rand() * 0.05) for x in range(26)] for _ in range(2)])
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return 1.0
        return 0.0

    bounds = [(0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={"type": "ineq", "fun": constraints})
    centers = np.array([[x, y] for x, y in result.x[:len(centers)]])
    radii = result.x[:len(centers)]
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
