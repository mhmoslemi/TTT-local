import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = (i % 5) / 4.0 + 0.25
        y = (i // 5) / 4.0 + 0.25
        centers[i] = [x, y]

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: Circles must be fully inside the unit square
    def constraint_inside(radii):
        centers_inside = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x, y = centers[i]
            centers_inside[i] = [x, y]
        return np.array([radii[i] <= 1 - centers_inside[i][0] for i in range(num_circles)]) + \
               np.array([radii[i] <= centers_inside[i][0] for i in range(num_circles)])

    # Constraint 2: Circles must not overlap
    def constraint_overlap(radii):
        dists = np.zeros(num_circles * num_circles)
        for i in range(num_circles):
            for j in range(num_circles):
                if i != j:
                    dx = centers[i][0] - centers[j][0]
                    dy = centers[i][1] - centers[j][1]
                    dist = math.hypot(dx, dy)
                    dists[i * num_circles + j] = dist - (radii[i] + radii[j])
        return dists

    # Initial radii guess: all 0.1
    initial_radii = np.full(num_circles, 0.1)

    # Bounds for radii
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Setup constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Perform optimization
    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    radii = result.x
    centers = np.array([[x, y] for x, y in centers])
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
