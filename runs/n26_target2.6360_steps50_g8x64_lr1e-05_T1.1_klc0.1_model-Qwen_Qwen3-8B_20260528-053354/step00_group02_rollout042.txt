import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.random.rand(26, 2)
        # Ensure each circle is inside the square
        for i in range(26):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Ensure circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Initial guess: random positions and small radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1
    # We use a local optimizer and restart with different initial guesses
    result = minimize(objective, radii, method='SLSQP', constraints=constraints, options={'maxiter': 100})
    centers = np.random.rand(26, 2)
    for i in range(26):
        if not (result.x[i] <= centers[i, 0] <= 1 - result.x[i] and
                result.x[i] <= centers[i, 1] <= 1 - result.x[i]):
            result.x[i] = 0
    return (centers, result.x, np.sum(result.x))
