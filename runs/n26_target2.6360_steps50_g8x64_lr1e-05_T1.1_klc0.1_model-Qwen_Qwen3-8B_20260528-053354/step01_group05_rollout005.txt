import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    np.random.seed(42)
    centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
    radii = np.full(26, 0.05)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints_func(radii_vec):
        # Convert radii_vec to a numpy array
        radii = radii_vec.copy()
        # Ensure all radii are non-negative
        for i in range(26):
            if radii[i] < 0:
                return np.inf
        # Check if circles are within the unit square
        for i in range(26):
            if radii[i] > 0:
                x, y = centers[i]
                if x - radii[i] < 0 or x + radii[i] > 1:
                    return np.inf
                if y - radii[i] < 0 or y + radii[i] > 1:
                    return np.inf
        # Check if circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    best_sum = 0.0
    best_centers = np.zeros((26, 2))
    best_radii = np.zeros(26)

    # Try multiple initial configurations
    for _ in range(10):
        centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
        radii = np.full(26, 0.05)
        result = minimize(objective, radii, constraints=[{'type': 'ineq', 'fun': constraints_func}])
        if result.success:
            current_sum = np.sum(result.x)
            if current_sum > best_sum:
                best_sum = current_sum
                best_centers = centers
                best_radii = result.x

    return (best_centers, best_radii, best_sum)
