import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.ones(26) * 0.1  # Initial guess for radii

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[26:])

    # Bounds for centers and radii
    bounds = []
    for i in range(26):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
        bounds.append((0, 1))  # radius

    # Constraints: circle within the square
    def constraint_square(x):
        for i in range(26):
            r = x[26 + i]
            x_c = x[2*i]
            y_c = x[2*i + 1]
            if not (r <= x_c <= 1 - r and r <= y_c <= 1 - r):
                return -1
        return 0

    # Constraints: circle-to-circle non-overlapping
    def constraint_overlap(x):
        for i in range(26):
            x1 = x[2*i]
            y1 = x[2*i + 1]
            r1 = x[26 + i]
            for j in range(i + 1, 26):
                x2 = x[2*j]
                y2 = x[2*j + 1]
                r2 = x[26 + j]
                dx = x1 - x2
                dy = y1 - y2
                if np.sqrt(dx*dx + dy*dy) < r1 + r2:
                    return -1
        return 0

    # Create constraints for optimization
    constraints = []
    constraints.append({'type': 'ineq', 'fun': constraint_square})
    constraints.append({'type': 'ineq', 'fun': constraint_overlap})

    # Initial guess flattened into a 1D array
    x0 = np.concatenate((centers.flatten(), radii))

    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    x_result = result.x
    centers = np.reshape(x_result[:52], (26, 2))
    radii = x_result[26:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
