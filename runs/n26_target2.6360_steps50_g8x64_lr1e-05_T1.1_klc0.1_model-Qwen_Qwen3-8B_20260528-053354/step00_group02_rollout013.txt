import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(x):
        centers = x[:26*2].reshape(26, 2)
        radii = x[26*2:]
        # Constraint 1: All circles are inside the unit square
        for i in range(26):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii[i]
                r_j = radii[j]
                if dist < r_i + r_j:
                    return np.inf
        return 0.0

    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26*2:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
