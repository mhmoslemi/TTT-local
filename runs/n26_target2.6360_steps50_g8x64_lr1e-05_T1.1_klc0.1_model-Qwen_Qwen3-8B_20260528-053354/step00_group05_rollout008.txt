import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced circles along a grid
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles)

    def objective(radii):
        # Objective: maximize the sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: circles are within the unit square
        inside = np.all((radii <= np.minimum(centers[:, 0], 1 - centers[:, 0])) &
                        (radii <= np.minimum(centers[:, 1], 1 - centers[:, 1])), axis=1)
        inside = np.all(inside)
        # Constraints: circles do not overlap
        dist = cdist(centers, centers)
        radii_diag = np.diag(radii)
        overlap = np.any(dist < radii_diag[:, np.newaxis] + radii_diag[np.newaxis, :])
        return inside and not overlap

    # Optimization options
    bounds = [(0, 1) for _ in range(num_circles)]
    result = minimize(objective, radii, bounds=bounds, constraints=lambda x: np.array([constraints(x)]))

    # Extract the optimal result
    optimal_radii = result.x
    optimal_centers = centers  # In a real scenario, centers would be optimized as well

    # Check if the constraints are satisfied
    inside = np.all((optimal_radii <= np.minimum(optimal_centers[:, 0], 1 - optimal_centers[:, 0])) &
                    (optimal_radii <= np.minimum(optimal_centers[:, 1], 1 - optimal_centers[:, 1])), axis=1)
    dist = cdist(optimal_centers, optimal_centers)
    radii_diag = np.diag(optimal_radii)
    overlap = np.any(dist < radii_diag[:, np.newaxis] + radii_diag[np.newaxis, :])

    if not inside or overlap:
        raise ValueError("Optimization result violates constraints")

    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
