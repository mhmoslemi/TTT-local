import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly scattered circles with small radii
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05

    # Define the objective function to minimize (negative sum of radii)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Define constraints
    def constraint_center_inside(radii_vec):
        # Each circle is fully inside the unit square
        return np.array([r <= 1 - r for r in radii_vec])

    def constraint_no_overlap(centers, radii_vec):
        # No two circles overlap
        distances = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return distances - (radii_vec[:, np.newaxis] + radii_vec[np.newaxis, :])

    def constraint_non_negative(radii_vec):
        # All radii are non-negative
        return radii_vec

    # Optimization parameters
    bounds = [(0, 1) for _ in range(26)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_center_inside},
            {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(centers, r)},
            {'type': 'ineq', 'fun': constraint_non_negative}
        ]
    )

    # Extract results
    centers = np.random.rand(26, 2)  # Reinitializing centers due to constraints not properly handled in opt
    # This is a simplified approach, actual optimization may require more advanced setup
    radii = result.x
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
