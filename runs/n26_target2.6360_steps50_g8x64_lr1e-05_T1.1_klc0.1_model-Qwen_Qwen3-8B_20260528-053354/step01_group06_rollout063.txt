import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1  # initial guess for radii

    def objective_function(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraint1(radii):
        # Ensure each circle is entirely inside the unit square
        r = radii
        return np.array([r[i] <= 1 - r[i] for i in range(num_circles)])  # x and y directions

    def constraint2(radii):
        # Ensure no two circles overlap
        r = radii
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            # Initial guess for centers (can be adjusted during optimization)
            centers[i, 0] = (i % 5) / 4.0
            centers[i, 1] = (i // 5) / 4.0
        dist = cdist(centers, centers)
        diag = np.diag(dist)
        dist[np.diag_indices(num_circles)] = np.inf
        return dist - (r[:, np.newaxis] + r[np.newaxis, :])

    # Setup optimization
    bounds = [(0.0, 0.5) for _ in range(num_circles)]  # radii cannot exceed 0.5
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(
        objective_function,
        radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    # Update centers based on the final radii
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5) / 4.0  # simple grid placement
        centers[i, 1] = (i // 5) / 4.0

    return (centers, result.x, -result.fun)
