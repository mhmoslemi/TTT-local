import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        for j in range(n):
            if i * n + j < n:
                centers[i * n + j] = [i / (n - 1), j / (n - 1)]
    
    # Objective function to minimize: negative sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Constraint: circles must be inside the square
    def inside_box(radii, centers):
        x, y = centers.T
        return np.all(radii <= x - 0.01) and np.all(radii <= 1.0 - x - 0.01) and np.all(radii <= y - 0.01) and np.all(radii <= 1.0 - y - 0.01)
    
    # Constraint: circles must not overlap
    def no_overlap(radii, centers):
        dists = cdist(centers, centers)
        diag = np.diag(dists)
        np.fill_diagonal(dists, np.inf)
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Optimization bounds
    bounds = [(0.0, 0.5) for _ in range(n)]
    
    # Initial radii as a guess
    radii = np.zeros(n)
    
    # Optimization settings
    options = {
        'tol': 1e-6,
        'maxiter': 1000,
        'disp': False
    }
    
    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: inside_box(x, centers)},
            {'type': 'ineq', 'fun': lambda x: no_overlap(x, centers)}
        ],
        options=options
    )
    
    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
