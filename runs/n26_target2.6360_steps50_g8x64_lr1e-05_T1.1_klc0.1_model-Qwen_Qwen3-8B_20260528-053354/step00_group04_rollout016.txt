    def constraint_func(x):
        radii = x[::2]

        # Constraint 1: Circles within square
        for i in range(26):
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i] or centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
