import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess for the centers and radii
    # Start with evenly spaced points in the square
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.05)

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        constraints = []
        # Constraint for each circle to be inside the unit square
        for i in range(num_circles):
            r_i = radii[i]
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y, r_i=r_i: x - r_i})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y, r_i=r_i: 1 - x - r_i})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y, r_i=r_i: y - r_i})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y, r_i=r_i: 1 - y - r_i})
        
        # Constraint for non-overlapping circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                r_i = radii[i]
                r_j = radii[j]
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, r_i=r_i, r_j=r_j, dist=dist: dist - r_i - r_j})
        
        return constraints

    # Flatten the radii and centers for optimization
    x0 = np.concatenate([centers.reshape(-1), radii])

    def objective_vector(x):
        centers = x[:num_circles*2].reshape(num_circles, 2)
        radii = x[num_circles*2:]
        return -np.sum(radii)

    def constraint_vector(x):
        centers = x[:num_circles*2].reshape(num_circles, 2)
        radii = x[num_circles*2:]
        constraints = []
        for i in range(num_circles):
            r_i = radii[i]
            x_i, y_i = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, r_i=r_i, x_i=x_i, y_i=y_i: x_i - r_i})
            constraints.append({'type': 'ineq', 'fun': lambda r, r_i=r_i, x_i=x_i, y_i=y_i: 1 - x_i - r_i})
            constraints.append({'type': 'ineq', 'fun': lambda r, r_i=r_i, x_i=x_i, y_i=y_i: y_i - r_i})
            constraints.append({'type': 'ineq', 'fun': lambda r, r_i=r_i, x_i=x_i, y_i=y_i: 1 - y_i - r_i})
        
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                r_i = radii[i]
                r_j = radii[j]
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append({'type': 'ineq', 'fun': lambda r, r_i=r_i, r_j=r_j, dist=dist: dist - r_i - r_j})
        
        return constraints

    bounds = []
    for _ in range(num_circles*2):
        bounds.append((0, 1))
    for _ in range(num_circles):
        bounds.append((0, 1))

    result = minimize(objective_vector, x0, bounds=bounds, constraints=constraint_vector, method='SLSQP', tol=1e-6)

    centers = result.x[:num_circles*2].reshape(num_circles, 2)
    radii = result.x[num_circles*2:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
