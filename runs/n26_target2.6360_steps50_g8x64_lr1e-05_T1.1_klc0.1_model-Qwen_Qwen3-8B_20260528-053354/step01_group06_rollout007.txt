import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii (assuming equal radii for initial guess)
    initial_radius = 0.1
    radii = np.full(num_circles, initial_radius)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints: circles must be fully inside the square
    def constraint_inside(x, centers, radii):
        return np.array([r - x[i, 0] for i, r in enumerate(radii)] +
                        [1 - x[i, 0] - radii[i] for i in range(num_circles)] +
                        [r - x[i, 1] for i, r in enumerate(radii)] +
                        [1 - x[i, 1] - radii[i] for i in range(num_circles)])

    # Constraints: circles must not overlap
    def constraint_overlap(x, centers, radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.array([dists[i, j] - (radii[i] + radii[j]) for i in range(num_circles) for j in range(i+1, num_circles)])

    # Optimization bounds: x and y coordinates must be between r and 1 - r
    x_bounds = [(r, 1 - r) for r in radii]
    y_bounds = [(r, 1 - r) for r in radii]
    bounds = [(x_low, x_high) for x_low, x_high in zip(np.ravel(x_bounds), np.ravel(y_bounds))]

    # Optimize using scipy
    result = minimize(objective, x0=np.ravel(centers), bounds=bounds,
                      constraints=[{'type': 'ineq', 'fun': constraint_inside, 'args': (centers, radii)},
                                   {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers, radii)}])

    # Extract centers and radii from result
    centers_result = result.x.reshape(num_circles, 2)
    radii_result = np.array([np.min([1 - centers_result[i, 0], centers_result[i, 0], 1 - centers_result[i, 1], centers_result[i, 1]]) for i in range(num_circles)])
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
