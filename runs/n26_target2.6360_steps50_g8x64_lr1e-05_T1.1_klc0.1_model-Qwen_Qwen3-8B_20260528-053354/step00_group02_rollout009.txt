import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: circles arranged in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1  # initial guess for radii

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[26:])

    # Constraints: 
    # 1. Each circle must be inside the unit square
    # 2. Distance between centers >= sum of radii
    constraints = []

    # Constraint 1: Each circle is fully inside the unit square
    def in_box(x):
        centers = x[:26*2].reshape((num_circles, 2))
        radii = x[26:]
        return np.all((centers[:, 0] - radii >= 0) & (centers[:, 1] - radii >= 0) & (1 - centers[:, 0] - radii >= 0) & (1 - centers[:, 1] - radii >= 0))

    constraints.append({'type': 'ineq', 'fun': in_box})

    # Constraint 2: No overlap between circles
    def no_overlap(x):
        centers = x[:26*2].reshape((num_circles, 2))
        radii = x[26:]
        distances = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 + 
                            (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        sum_radii = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.min(distances - sum_radii)

    constraints.append({'type': 'ineq', 'fun': no_overlap})

    # Bounds for centers (x and y in [0,1]) and radii (>= 0)
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
    for i in range(num_circles):
        bounds.append((0, 1))  # radius

    # Initial guess
    x0 = np.concatenate((centers.flatten(), radii))

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-8)

    # Extract results
    centers_opt = result.x[:26*2].reshape((num_circles, 2))
    radii_opt = result.x[26:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
