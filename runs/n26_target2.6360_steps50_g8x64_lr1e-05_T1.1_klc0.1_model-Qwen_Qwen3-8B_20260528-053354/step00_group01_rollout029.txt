import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05
    sum_radii = np.sum(radii)
    
    # Objective function: minimize negative sum of radii
    def objective(x):
        # x is a flattened array of centers and radii
        x_flat = x.reshape(26, 4)  # [x, y, r, ...]
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        sum_r = np.sum(radii)
        return -sum_r
    
    # Constraints
    def constraint_inside(x):
        x_flat = x.reshape(26, 4)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        # Check if each circle is inside the unit square
        inside = (centers[:, 0] + radii <= 1) & (centers[:, 0] - radii >= 0) & \
                 (centers[:, 1] + radii <= 1) & (centers[:, 1] - radii >= 0)
        return np.all(inside)
    
    def constraint_overlap(x):
        x_flat = x.reshape(26, 4)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        # Ensure no two circles overlap
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    def constraint_radii(x):
        x_flat = x.reshape(26, 4)
        radii = x_flat[:, 2]
        return np.all(radii >= 0)
    
    # Initial guess (flattened)
    x0 = np.concatenate([centers.reshape(-1), radii])
    
    # Prepare constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_radii}
    ]
    
    # Optimization
    result = minimize(objective, x0, constraints=cons, method='SLSQP', tol=1e-6)
    
    # Extract final solution
    x_sol = result.x
    centers_sol = x_sol[:26*2].reshape(26, 2)
    radii_sol = x_sol[26*2:].reshape(26)
    sum_radii_sol = np.sum(radii_sol)
    
    return (centers_sol, radii_sol, sum_radii_sol)
