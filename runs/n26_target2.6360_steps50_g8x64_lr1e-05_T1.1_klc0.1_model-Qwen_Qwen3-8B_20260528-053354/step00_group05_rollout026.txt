import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radius guess: same for all circles
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        # Constraints: circles are inside the square
        return np.array([r - (1 - x - r) for x, r in zip(centers, radii)] + [r - (1 - y - r) for x, y, r in zip(centers[:, 0], centers[:, 1], radii)])

    def constraint_no_overlap(radii):
        # Constraints: distance between any two centers >= sum of their radii
        constraints = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                constraints.append(distance - (radii[i] + radii[j]))
        return np.array(constraints)

    # Define optimization problem
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    cons = [{'type': 'ineq', 'fun': constraint_circle_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}]
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract result
    centers_final = centers
    radii_final = res.x
    sum_radii = -res.fun

    return (centers_final, radii_final, sum_radii)
