import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: equally spaced grid of 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square
    def inside_box(radii, centers):
        for i in range(num_circles):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        return True

    # Constraints: circles must not overlap
    def no_overlap(radii, centers):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Combine constraints into a single function
    def constraints(radii):
        if not inside_box(radii, centers):
            return np.inf
        if not no_overlap(radii, centers):
            return np.inf
        return 0

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimization
    result = minimize(objective, np.ones(num_circles), bounds=bounds, constraints={"type": "ineq", "fun": constraints})

    # Extract results
    radii = result.x
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
