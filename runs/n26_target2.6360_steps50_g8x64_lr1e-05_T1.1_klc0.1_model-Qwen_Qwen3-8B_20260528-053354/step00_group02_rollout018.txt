import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (randomly distributed)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    def objective_function(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Convert radii to a 1D array
        radii = radii.reshape(-1)
        # Check if each circle is inside the square
        for i in range(26):
            if radii[i] > 0:
                if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                    return np.inf
                if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                    return np.inf
        # Check for overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                if radii[i] + radii[j] > np.linalg.norm(centers[i] - centers[j]):
                    return np.inf
        return -np.sum(radii)

    # Optimization
    result = minimize(objective_function, radii, method='L-BFGS-B', bounds=[(0, 1)]*26)

    # Extract result
    final_radii = result.x
    final_centers = centers.copy()
    final_sum_radii = -result.fun

    return (final_centers, final_radii, final_sum_radii)
