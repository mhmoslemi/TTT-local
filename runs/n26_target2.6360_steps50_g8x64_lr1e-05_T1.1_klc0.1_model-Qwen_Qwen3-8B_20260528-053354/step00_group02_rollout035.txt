import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: distribute 26 circles in a grid
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.05)

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: all circles are inside the unit square
    def square_constraint(radii, centers):
        return np.sum((radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) & 
                      (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1]))

    # Constraint: no two circles overlap
    def no_overlap_constraint(radii, centers):
        dists = np.abs(centers[:, np.newaxis, :] - centers[np.newaxis, :, :])
        min_dist = np.min(dists)
        return np.sum((min_dist >= np.sqrt(np.sum((radii[:, np.newaxis] + radii[np.newaxis, :])**2, axis=2)))) - 1e-6

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimization settings
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        jac=None,
        bounds=bounds,
        constraints=[
            {"type": "ineq", "fun": lambda x: square_constraint(x, centers)},
            {"type": "ineq", "fun": lambda x: no_overlap_constraint(x, centers)}
        ],
        options={"maxiter": 1000, "ftol": 1e-8}
    )

    # Apply result
    radii = result.x
    centers = np.random.rand(num_circles, 2)
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
