import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.array([[x / np.sqrt(num_circles), y / np.sqrt(num_circles)] for y in range(num_circles) for x in range(num_circles)])
    radii = np.full(num_circles, 0.1)

    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraint: circles must be inside the unit square
    def inside_box(x, y, r):
        return x >= r and x <= 1 - r and y >= r and y <= 1 - r

    # Constraint: circles must not overlap
    def no_overlap(c1, r1, c2, r2):
        dx = c1[0] - c2[0]
        dy = c1[1] - c2[1]
        return np.sqrt(dx**2 + dy**2) >= r1 + r2

    # Combine constraints into a single function
    def constraints(radii, centers):
        constraints = []
        for i in range(num_circles):
            # Inside box constraint
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: inside_box(centers[i, 0], centers[i, 1], r)})
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                # No overlap constraint
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: no_overlap(centers[i], r[i], centers[j], r[j])})
        return constraints

    result = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        constraints=constraints(radii, centers),
        method='SLSQP',
        bounds=[(0, 1 - 1e-6) for _ in range(num_circles)]
