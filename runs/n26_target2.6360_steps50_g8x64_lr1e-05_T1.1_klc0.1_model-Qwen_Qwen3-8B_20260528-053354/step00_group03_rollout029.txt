import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced grid for 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i + 1) / (num_circles + 1)
        centers[i, 1] = 0.5

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint1(radii):
        # All circles fit inside the unit square
        return np.sum((radii <= 1 - centers[:, 0]) * (radii <= 1 - centers[:, 1]))

    def constraint2(radii):
        # No two circles overlap
        dx = centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0]
        dy = centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1]
        dist = np.sqrt(dx**2 + dy**2)
        return np.sum(dist >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Bounds on radii
    bounds = [(0, 1 - 0.1) for _ in range(num_circles)]

    # Constraints to be satisfied (all must be >= 0)
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Use optimization to maximize the sum of radii
    result = minimize(objective, np.ones(num_circles), method='SLSQP',
                      bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract results
    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
