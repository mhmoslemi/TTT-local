import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = 0.2 + col * 0.25
        centers[i, 1] = 0.2 + row * 0.25
        radii[i] = 0.05

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        centers_flat = x.reshape(-1, 2)
        radii_flat = x[2::3]
        # Initial objective: minimize the negative of sum of radii (maximize sum of radii)
        sum_r = np.sum(radii_flat)
        # Constraints: circles must be within the unit square and not overlapping
        # Check if any circle goes out of bounds
        for i in range(26):
            if centers_flat[i, 0] < 0 or centers_flat[i, 0] > 1 or \
               centers_flat[i, 1] < 0 or centers_flat[i, 1] > 1:
                return np.inf
            if centers_flat[i, 0] - radii_flat[i] < 0 or centers_flat[i, 0] + radii_flat[i] > 1 or \
               centers_flat[i, 1] - radii_flat[i] < 0 or centers_flat[i, 1] + radii_flat[i] > 1:
                return np.inf
        # Check if circles are non-overlapping
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return -sum_r  # We minimize the negative to maximize sum_r

    def bounds_and_constraints(x):
        # Define bounds for all variables (x_i, y_i, r_i)
        bounds = []
        for i in range(26):
            bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])
        return bounds

    def constraints(x):
        # Constraints for circle bounds and non-overlap
        centers_flat = x.reshape(-1, 2)
        radii_flat = x[2::3]
        # Check if any circle goes out of bounds
        for i in range(26):
            if centers_flat[i, 0] - radii_flat[i] < 0 or centers_flat[i, 0] + radii_flat[i] > 1 or \
               centers_flat[i, 1] - radii_flat[i] < 0 or centers_flat[i, 1] + radii_flat[i] > 1:
                return np.inf
        # Check if circles are non-overlapping
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0.0  # No violation

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="Optimization failed")
        result = minimize(
            fun=objective,
            x0=np.concatenate([centers.flatten(), radii]),
            bounds=bounds_and_constraints,
            constraints={"type": "ineq", "fun": constraints},
            method='SLSQP',
            options={'ftol': 1e-8, 'disp': False}
        )

    # Extract results
    x_opt = result.x
    centers_opt = x_opt.reshape(-1, 2)
    radii_opt = x_opt[2::3]
    sum_r_opt = -result.fun  # Negate because we minimized the negative

    return (centers_opt, radii_opt, sum_r_opt)
