import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05
    
    def objective_function(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Maximize sum_radii by minimizing -sum_radii

    def constraints(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        
        # Constraint 1: Each circle fits inside the unit square
        constraints1 = []
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints1.append(x_i - r_i)
            constraints1.append(1 - x_i - r_i)
            constraints1.append(y_i - r_i)
            constraints1.append(1 - y_i - r_i)
        
        # Constraint 2: No two circles overlap
        constraints2 = []
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                constraints2.append(dist - r_i - r_j)
        
        return np.concatenate((constraints1, constraints2))

    def bounds(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        bounds = []
        for i in range(26):
            for coord in [0, 1]:
                bounds.append((0, 1))
            bounds.append((0, 0.5))
        return bounds

    # Set up the optimization problem
    x0 = np.vstack((centers.flatten(), radii)).flatten()
    res = minimize(
        objective_function,
        x0,
        method='SLSQP',
        bounds=bounds(x0),
        constraints={'type': 'ineq', 'fun': constraints, 'jac': None},
        options={'ftol': 1e-9, 'maxiter': 1000, 'disp': False}
    )

    # Extract the result
    x_opt = res.x
    centers_opt = x_opt[:2*26].reshape(26, 2)
    radii_opt = x_opt[2*26:]
    sum_radii_opt = -res.fun

    return (centers_opt, radii_opt, sum_radii_opt)
