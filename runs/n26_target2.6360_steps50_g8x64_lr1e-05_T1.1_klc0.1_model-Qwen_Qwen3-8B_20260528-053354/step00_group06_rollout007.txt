import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.array([[i/5 + 0.1, j/5 + 0.1] for i in range(5) for j in range(5)])
    radii = np.zeros(25)
    # Add one more circle near the center
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = np.append(radii, 0.01)
    
    # Objective function to minimize (negative of sum of radii)
    def objective(x):
        return -np.sum(radii)
    
    # Constraints: each circle within the unit square and no overlaps
    def constraint_square(x):
        return x - 0.0  # r <= x <= 1 - r
    def constraint_overlap(x):
        # Check all pairs of circles for overlap
        diff = centers[None, :, :] - centers[:, None, :]
        dist = np.linalg.norm(diff, axis=2)
        return np.min(dist - radii[None, :] - radii[:, None])  # distance - sum of radii
    
    # Flatten centers and radii for optimization
    x0 = np.concatenate([centers.reshape(-1), radii])
    cons = [
        {'type': 'ineq', 'fun': constraint_square, 'args': (x0[len(centers*2):],)},
        {'type': 'ineq', 'fun': constraint_overlap, 'args': (x0,)}
    ]
    
    # Optimization
    result = minimize(objective, x0, constraints=cons, method='SLSQP', tol=1e-6)
    
    # Extract result
    opt_centers = result.x[:centers.size].reshape(centers.shape)
    opt_radii = result.x[centers.size:].reshape(radii.shape)
    sum_radii = np.sum(opt_radii)
    
    return (opt_centers, opt_radii, sum_radii)
