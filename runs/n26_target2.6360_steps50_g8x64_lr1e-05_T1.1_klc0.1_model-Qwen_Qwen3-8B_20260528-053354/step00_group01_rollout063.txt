```python
import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1

    # Randomly perturb initial positions
    for i in range(num_circles):
        centers[i] = [random.uniform(0.1, 0.9), random.uniform(0.1, 0.9)]

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.sum((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]))

    def constraint_no_overlap(radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(objective, radii, method='SLSQP', constraints=cons, bounds=[(0, 1) for _ in range(num_circles)])
    radii_opt = result.x
    centers_opt = centers
    sum_radii_opt = result.fun * -1

    return (centers_opt, radii_opt, sum_radii_opt)
```
