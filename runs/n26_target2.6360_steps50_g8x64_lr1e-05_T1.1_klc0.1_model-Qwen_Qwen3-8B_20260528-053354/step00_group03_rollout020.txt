import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: equally spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i + 1) / (num_circles + 1)
        centers[i, 1] = 0.5

    radii = np.ones(num_circles) * 0.1

    def objective(radii):
        return -sum(radii)

    def constraints(radii):
        # Objective function for optimization
        return -sum(radii)

    # Optimization constraints
    def constraint_func(radii):
        nonlocal centers
        # Constraint 1: Circles fully inside the unit square
        for i in range(num_circles):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return -np.inf
        # Constraint 2: No two circles overlap
        distances = pdist(centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        for d, r1, r2 in zip(distances, radii, radii):
            if d < r1 + r2:
                return -np.inf
        return 0.0

    def constraint_func_wrapper(radii):
        return constraint_func(radii)

    # Optimize
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    res = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        jac=None,
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_func_wrapper},
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    if res.success:
        optimized_radii = res.x
        # Update centers based on optimized radii
        # Simple heuristic to adjust centers iteratively
        centers_new = np.copy(centers)
        for _ in range(10):
            for i in range(num_circles):
                for j in range(num_circles):
                    if i != j:
                        dist = np.linalg.norm(centers_new[i] - centers_new[j])
                        r1 = optimized_radii[i]
                        r2 = optimized_radii[j]
                        if dist < r1 + r2:
                            direction = (centers_new[j] - centers_new[i]) / dist
                            move = 0.1 * (r1 + r2 - dist)
                            centers_new[i] += direction * move
                            centers_new[j] -= direction * move
        # Re-check constraints and adjust again
        for _ in range(10):
            for i in range(num_circles):
                for j in range(num_circles):
                    if i != j:
                        dist = np.linalg.norm(centers_new[i] - centers_new[j])
                        r1 = optimized_radii[i]
                        r2 = optimized_radii[j]
                        if dist < r1 + r2:
                            # Move both centers
                            direction = (centers_new[j] - centers_new[i]) / dist
                            move = 0.1 * (r1 + r2 - dist)
                            centers_new[i] += direction * move
                            centers_new[j] -= direction * move
        # Final check to ensure all circles are within the unit square
        for i in range(num_circles):
            r = optimized_radii[i]
            x, y = centers_new[i]
            if x < r or x > 1 - r or y < r or y > 1 - r:
                centers_new[i] = [x, y]
                optimized_radii[i] = min(r, 1 - x, x, 1 - y)

        # Final update to centers
        centers = centers_new
        radii = optimized_radii
        sum_radii = sum(radii)
    else:
        sum_radii = 0.0
        radii = np.zeros(num_circles)
        centers = np.zeros((num_circles, 2))

    return (centers, radii, sum_radii)
