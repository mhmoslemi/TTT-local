import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.ones(26) * 0.1  # Initial guess for radii

    # Define objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraints
    def constraint_inside(radii):
        # Constraint: all circles are inside the unit square
        def inside(x, y, r):
            return (r <= x) and (x <= 1 - r) and (r <= y) and (y <= 1 - r)

        # Combine all constraints
        return 1 - np.sum([inside(x, y, r) for x, y, r in zip(centers[:, 0], centers[:, 1], radii)])

    def constraint_overlap(radii):
        # Constraint: no two circles overlap
        def overlap(i, j, r_i, r_j, x_i, y_i, x_j, y_j):
            dx = x_i - x_j
            dy = y_i - y_j
            dist = math.hypot(dx, dy)
            return dist < r_i + r_j

        # Combine all pairwise constraints
        return 1 - np.sum([overlap(i, j, r_i, r_j, x_i, y_i, x_j, y_j) for i in range(26) for j in range(i+1, 26) \
                          for x_i, y_i, r_i in zip(centers[i, 0], centers[i, 1], radii) \
                          for x_j, y_j, r_j in zip(centers[j, 0], centers[j, 1], radii)])

    # Optimize
    result = minimize(objective, radii, method='SLSQP',
                      constraints=[
                          {'type': 'ineq', 'fun': constraint_inside},
                          {'type': 'ineq', 'fun': constraint_overlap}
                      ],
                      bounds=[(0, 1) for _ in range(26)])
    
    # Extract the result
    optimized_radii = result.x
    optimized_centers = centers.copy()
    sum_radii = np.sum(optimized_radii)
    
    return (optimized_centers, optimized_radii, sum_radii)
