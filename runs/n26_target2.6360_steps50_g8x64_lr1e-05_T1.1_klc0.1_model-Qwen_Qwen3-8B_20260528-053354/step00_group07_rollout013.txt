import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
import sys

def run_packing():
    # Initial guess: 26 circles placed in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5) / 4.0 + 0.125
        centers[i, 1] = (i // 5) / 4.0 + 0.125

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Each circle must be inside the unit square
        for i in range(26):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False

        # No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Bounds for radii: [0, 0.5] since circles must be inside unit square
    bounds = [(0, 0.5) for _ in range(26)]

    # Optimization problem
    result = minimize(
        fun=objective,
        x0=np.full(26, 0.1),
        bounds=bounds,
        constraints={"type": "custom", "fun": constraints},
        method="SLSQP"
    )

    # Extract results
    radii = result.x
    if not constraints(radii):
        raise ValueError("Optimization failed to satisfy constraints.")

    # Final centers and radii
    centers_final = np.zeros((26, 2))
    for i in range(26):
        centers_final[i, 0] = (i % 5) / 4.0 + 0.125
        centers_final[i, 1] = (i // 5) / 4.0 + 0.125

    sum_radii = np.sum(radii)
    return (centers_final, radii, sum_radii)
