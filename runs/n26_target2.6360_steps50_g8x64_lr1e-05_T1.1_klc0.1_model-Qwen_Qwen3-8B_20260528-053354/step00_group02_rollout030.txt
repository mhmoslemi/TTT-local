import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    num_circles = 26
    grid_size = int(np.ceil(np.sqrt(num_circles)))
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // grid_size
        col = i % grid_size
        centers[i, 0] = col / (grid_size - 1)
        centers[i, 1] = row / (grid_size - 1)
    radii = np.ones(num_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraint_1(radii):
        # Each circle must be inside the unit square
        radii = np.array(radii)
        r_min = np.min(radii)
        r_max = np.max(radii)
        # Check if all circles are within the square
        if np.any(radii > 0.5):
            return -np.inf
        return 0

    def constraint_2(radii):
        # No two circles overlap
        radii = np.array(radii)
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        sum_radii = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.min(dists - sum_radii)

    def constraint_3(radii):
        # All radii must be non-negative
        return np.min(radii)

    # Set up optimization problem
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_1},
        {'type': 'ineq', 'fun': constraint_2},
        {'type': 'ineq', 'fun': constraint_3}
    ]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-8)

    # Final result
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
