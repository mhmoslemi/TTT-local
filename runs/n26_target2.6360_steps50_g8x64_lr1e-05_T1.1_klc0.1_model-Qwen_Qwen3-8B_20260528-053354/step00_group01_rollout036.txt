from scipy.optimize import minimize

def run_packing():
    # Initial guess: equally spaced circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))

    for i in range(num_circles):
        x = (i % 5) / 4.0 + 0.1
        y = (i // 5) / 4.0 + 0.1
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i] = [0.1 + (i % 5) / 4.0, 0.1 + (i // 5) / 4.0]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if centers[i][0] - radii[i] < 0 or centers[i][0] + radii[i] > 1:
                return False
            if centers[i][1] - radii[i] < 0 or centers[i][1] + radii[i] > 1:
                return False
        return True

    def objective_func(x):
        radii = x
        # Initial centers as placeholders for the actual solution
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i] = [0.1 + (i % 5) / 4.0, 0.1 + (i // 5) / 4.0]

                    return 1e6  # Penalize constraint violation
