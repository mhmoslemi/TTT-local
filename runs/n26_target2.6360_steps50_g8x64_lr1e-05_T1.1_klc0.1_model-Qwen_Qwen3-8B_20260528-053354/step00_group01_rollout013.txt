import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import itertools

def run_packing():
    # Initial guess: circle centers spaced approximately 0.2 units apart
    # with radii initialized to 0.1
    num_circles = 26
    centers = np.random.rand(num_circles, 2) * 0.8 + 0.1  # Initial guess inside the square
    radii = np.full(num_circles, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # We want to maximize the sum, so minimize -sum
    
    def constraints(radii):
        # Constraint: radii must be non-negative
        return radii >= 0
    
    def box_constraints(radii):
        # Constraint: all circles fit inside the unit square
        return np.sum((radii <= 1 - centers[:, 0]) * (radii <= 1 - centers[:, 1])) == num_circles
    
    def distance_constraints(radii):
        # Constraint: no two circles overlap
        distances = squareform(pdist(centers))
        for i, j in itertools.combinations(range(num_circles), 2):
            if distances[i, j] < radii[i] + radii[j]:
                return False
        return True
    
    def objective_with_constraints(radii):
        # Combine all constraints
        if not all(r >= 0 for r in radii):
            return np.inf
        if not box_constraints(radii):
            return np.inf
        if not distance_constraints(radii):
            return np.inf
        return -np.sum(radii)
    
    # Optimization using scipy.optimize.minimize
    res = minimize(
        objective_with_constraints,
        radii,
        method='L-BFGS-B',
        bounds=[(0, 1)] * num_circles,
        options={'maxiter': 500, 'disp': False}
    )
    
    # Extract results
    optimized_radii = res.x
    optimized_centers = centers  # Centers are not optimized explicitly but are adjusted during optimization
    sum_radii = np.sum(optimized_radii)
    
    # Return the result
    return (optimized_centers, optimized_radii, sum_radii)
