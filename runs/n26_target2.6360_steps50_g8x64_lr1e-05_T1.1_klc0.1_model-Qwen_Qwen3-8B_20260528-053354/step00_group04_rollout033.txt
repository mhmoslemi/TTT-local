import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i / (num_circles - 1)
        y = 0.5
        centers[i, 0] = x
        centers[i, 1] = y
    
    # Initial radii: uniform
    radii = np.full(num_circles, 0.1)
    sum_radii = 0.0
    
    # Objective function: maximize sum of radii (minimize -sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints
    constraints = []
    
    # Constraint: circle must be inside the unit square
    def inside_box(radii):
        return np.array([r <= 1 - r for r in radii])
    
    def inside_box_constraint(radii):
        return np.sum(1 - (radii <= 1 - radii))
    
    constraints.append({'type': 'ineq', 'fun': inside_box})
    
    # Constraint: no overlap between circles
    def no_overlap(radii):
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[idx] = dist - radii[i] - radii[j]
                idx += 1
        return dists
    
    constraints.append({'type': 'ineq', 'fun': no_overlap})
    
    # Bounds: all radii must be non-negative
    bounds = [(0, 1) for _ in range(num_circles)]
    
    # Minimize the negative sum to maximize the sum
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)
    
    # Extract final results
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii_final = -result.fun
    
    return (centers_final, radii_final, sum_radii_final)
