import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles with small radii
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.05)
    centers = np.vstack([centers, [0.5, 0.5]])  # add one more circle in the center
    radii = np.append(radii, 0.05)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii_flat):
        return -np.sum(radii_flat)

    # Constraints
    def constraint_centers(radii_flat):
        # Ensure all circles are inside the unit square
        centers_flat = np.vstack([centers[:25, :], np.array([np.nan, np.nan])]).flatten()
        r = radii_flat[:25]
        r_center = radii_flat[25]
        # Check if the center is inside the square
        if not (r_center <= centers[25, 0] <= 1 - r_center and r_center <= centers[25, 1] <= 1 - r_center):
            return -np.inf
        # Check if all other centers are inside their respective squares
        for i in range(25):
            if not (r[i] <= centers[i, 0] <= 1 - r[i] and r[i] <= centers[i, 1] <= 1 - r[i]):
                return -np.inf
        return 0

    def constraint_overlap(radii_flat):
        # Ensure no two circles overlap
        centers_flat = np.vstack([centers[:25, :], np.array([np.nan, np.nan])]).flatten()
        r = radii_flat[:25]
        r_center = radii_flat[25]
        dists = []
        for i in range(25):
            for j in range(i + 1, 25):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                dists.append(dist - (r[i] + r[j]))
        if not all(d >= -1e-6 for d in dists):
            return -np.inf
        dx = centers[25, 0] - centers[:25, 0]
        dy = centers[25, 1] - centers[:25, 1]
        dists_center = [np.sqrt(dx[i] ** 2 + dy[i] ** 2) - (r_center + r[i]) for i in range(25)]
        if not all(d >= -1e-6 for d in dists_center):
            return -np.inf
        return 0

    # Bounds: radii must be non-negative
    bounds = [(0.0, None) for _ in range(26)]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_centers},
                {'type': 'ineq', 'fun': constraint_overlap}
            ],
            tol=1e-6
        )

    # Extract final result
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)
    centers_opt = centers.copy()
    return (centers_opt, radii_opt, sum_radii)
