import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for centers (equally spaced grid)
    num_circles = 26
    centers = np.array([[x / 7, y / 7] for y in range(7) for x in range(7)])
    centers = centers[:num_circles]
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - np.min(centers[i])
                return np.inf
            if radii[i] > 1 - np.max(centers[i])
                return np.inf

        # Ensure no two circles overlap
        distance_matrix = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if distance_matrix[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization settings
    bounds = [(0, 1) for _ in range(num_circles)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Extract final results
    centers_final = centers[:num_circles]
    radii_final = result.x
    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
