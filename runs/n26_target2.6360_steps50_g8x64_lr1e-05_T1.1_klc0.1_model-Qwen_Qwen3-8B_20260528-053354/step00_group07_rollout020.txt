import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.25 + 0.125
        centers[i, 1] = row * 0.25 + 0.125
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        num_circles = len(radii)
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i, 0] = i % 5 * 0.1 + 0.05
            centers[i, 1] = i // 5 * 0.1 + 0.05
            radii[i] = 0.05
        # Boundary constraints: circle must be inside the unit square
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return np.inf
        # Non-overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    result = minimize(objective, radii, method='SLSQP', bounds=[(0, 0.5)] * 26, constraints={"type": "ineq", "fun": constraints})
    if result.success:
        sum_radii = result.fun
        return (centers, radii, sum_radii)
    else:
        raise RuntimeError("Optimization failed")
