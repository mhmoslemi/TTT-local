import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii (random positions inside the square)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    def objective(radii_):
        return -np.sum(radii_)  # We're maximizing the sum, so minimize the negative

    def constraints(radii_):
        # Constraint 1: All circles must be inside the square
        inside = np.all((radii_ <= 1 - centers[:, 0]) & (radii_ <= 1 - centers[:, 1]), axis=1)
        inside = np.all(inside)

        # Constraint 2: No overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_[i] + radii_[j]:
                    return False
        return True

    # Combine radii and centers into a single array for optimization
    bounds = [(0, 0.5) for _ in range(26)]  # Max radius is 0.5
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Wrap the constraints into a function compatible with scipy.optimize
    def constraint_func(x):
        # Split x into centers and radii
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:]
        # Check if all constraints are satisfied
        inside = np.all((radii_flat <= 1 - centers_flat[:, 0]) & (radii_flat <= 1 - centers_flat[:, 1]), axis=1)
        if not np.all(inside):
            return np.inf
        # Check for overlaps
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return 0

    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints={"type": "ineq", "fun": constraint_func},
                      method="SLSQP", options={"maxiter": 1000, "ftol": 1e-8, "iprint": 0})

    # Extract the final solution
    final_centers = result.x[:26*2].reshape(26, 2)
    final_radii = result.x[26*2:]
    sum_radii = -result.fun  # Since we minimized the negative

    return (final_centers, final_radii, sum_radii)
