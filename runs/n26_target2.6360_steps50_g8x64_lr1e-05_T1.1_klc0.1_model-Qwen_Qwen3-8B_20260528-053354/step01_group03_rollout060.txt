import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    radii = np.full(36, 0.05)  # Initial radii

    # Remove the last 10 circles to get 26
    centers = centers[:26]
    radii = radii[:26]

    # Optimization function
    def objective(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ...]
        x_flat = x.reshape(-1, 3)
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]

        # Compute constraints
        sum_radii = np.sum(radii)

        # Check if all circles are within the unit square
        if not np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                      (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii)):
            return np.inf

        # Check if circles are non-overlapping
        dists = cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf

        return -sum_radii  # Minimize -sum_radii to maximize it

    # Bounds for each parameter: [x, y, r]
    bounds = []
    for i in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])

    # Initial guess for the optimization
    x0 = np.empty(26 * 3)
    for i in range(26):
        x0[i * 3] = centers[i, 0]
        x0[i * 3 + 1] = centers[i, 1]
        x0[i * 3 + 2] = radii[i]

    # Run optimization
    result = minimize(objective, x0, bounds=bounds, method='L-BFGS-B', tol=1e-8)

    # Extract the result
    optimized = result.x.reshape(-1, 3)
    optimized_centers = optimized[:, :2]
    optimized_radii = optimized[:, 2]
    sum_radii = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
