import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1

    # Random initial positions
    for i in range(num_circles):
        centers[i, 0] = random.uniform(0.1, 0.9)
        centers[i, 1] = random.uniform(0.1, 0.9)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circle inside unit square
    def constraint_inside(radii):
        return np.concatenate([radii * 2 - 1, radii * 2 - 1])

    # Constraints: no overlapping between circles
    def constraint_overlap(radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=-1))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r)},
        {'type': 'ineq', 'fun': lambda r: np.min(constraint_overlap(r))}
    ]

    # Bounds on radii: non-negative
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        jac=lambda r: np.zeros_like(r),
        bounds=bounds,
        constraints=constraints
    )

    # Extract results
    centers = np.array(result.x)
    radii = np.array(result.x)
    sum_radii = np.sum(radii)

    # Ensure all circles are inside the square
    valid = (radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]) & (radii <= centers[:, 0]) & (radii <= centers[:, 1])
    centers = centers[valid]
    radii = radii[valid]

    # If after optimization, the number of circles is less than 26, adjust
    if len(radii) < 26:
        new_centers = np.zeros((26, 2))
        new_radii = np.zeros(26)
        new_centers[:len(radii)] = centers
        new_radii[:len(radii)] = radii
        centers = new_centers
        radii = new_radii

    # Final check
    centers = centers[:26]
    radii = radii[:26]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
