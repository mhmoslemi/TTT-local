import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = (i % 7) / 6.0 + 0.05
        centers[i, 1] = (i // 7) / 6.0 + 0.05
        radii[i] = 0.04

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraints for each circle: r <= x <= 1 - r and r <= y <= 1 - r
        for i in range(26):
            if radii[i] < 0:
                return False
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        # Constraints for non-overlapping circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimize
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r)}],
        bounds=[(0.0, 0.5)] * 26
    )

    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
