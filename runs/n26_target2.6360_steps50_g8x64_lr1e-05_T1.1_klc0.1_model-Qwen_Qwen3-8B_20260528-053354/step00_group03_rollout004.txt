```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    spacing = 1.0 / np.sqrt(num_circles)
    for i in range(num_circles):
        row = i // np.sqrt(num_circles)
        col = i % np.sqrt(num_circles)
        x = col * spacing + 0.5 * spacing
        y = row * spacing + 0.5 * spacing
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = 0.05  # Initial small radius

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: Circles must be inside the unit square
    def constraint_inside(radii):
        return np.array([r <= 1 - np.sqrt((centers[i, 0] - 0.5)**2 + (centers[i, 1] - 0.5)**2) for i, r in enumerate(radii)])

    # Constraint 2: Circles cannot overlap
    def constraint_overlap(radii):
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0]) ** 2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1]) ** 2))
        radii_grid = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dists - radii_grid

    # Optimization setup
    bounds = [(0.0, 1.0 - np.sqrt((centers[i, 0] - 0.5)**2 + (centers[i, 1] - 0.5)**2)) for i in range(num_circles)]
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8, options={'maxiter': 1000})

    # Extract results
    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
```
