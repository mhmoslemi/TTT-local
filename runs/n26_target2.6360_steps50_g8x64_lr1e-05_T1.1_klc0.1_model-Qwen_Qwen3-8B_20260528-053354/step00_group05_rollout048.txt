import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: circles evenly spaced
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i] = [i % 7 / 6, i // 7 / 6]  # 7x4 grid with some padding
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i] = [i % 7 / 6, i // 7 / 6]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(26):
            if radii[i] > 1 - np.abs(centers[i][0]) or radii[i] > 1 - np.abs(centers[i][1]):
                return False
        return True

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="Cleaving a non-convex problem")
        result = minimize(
            fun=objective,
            x0=radii,
            method='SLSQP',
            bounds=[(0, 1 - 0.01) for _ in range(26)],
            constraints=[{'type': 'ineq', 'fun': constraints}]
        )

    # Re-optimizing to refine the positions based on final radii
    def objective_with_centers(radii):
        return -np.sum(radii)

    def constraints_with_centers(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            # Optimal circle positions
            centers[i] = [i % 7 / 6, i // 7 / 6]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(26):
            if radii[i] > 1 - np.abs(centers[i][0]) or radii[i] > 1 - np.abs(centers[i][1]):
                return False
        return True

    result = minimize(
        fun=objective_with_centers,
        x0=result.x,
        method='SLSQP',
        bounds=[(0, 1 - 0.01) for _ in range(26)],
        constraints=[{'type': 'ineq', 'fun': constraints_with_centers}]
    )

    # After optimization, determine the positions
    centers = np.zeros((26, 2))
    for i in range(26):
        # Optimal grid
        centers[i] = [i % 7 / 6, i // 7 / 6]
    # Final radii
    radii = result.x
    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
