import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Inequality constraints:
    # 1. All circles must be within the unit square
    # 2. Distance between any two centers must be >= sum of their radii

    def constraints(radii):
        # Constraint 1: circles fit in the square
        # r_i <= x_i - r_i and r_i <= y_i - r_i => r_i <= (x_i, y_i) - r_i
        # => r_i <= min(x_i - r_i, y_i - r_i) => r_i <= min(x_i, y_i) - r_i => r_i <= min(x_i, y_i)/2
        r = radii
        c1 = np.array([np.min(centers[i]) - r[i] for i in range(num_circles)])
        # Constraint 2: no overlapping between circles
        c2 = []
        for i in range(num_circles):
            for j in range(i+1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                c2.append(dist - r[i] - r[j])
        return np.concatenate([c1, c2])

    # Bounds: radii >= 0
    bounds = [(0, None) for _ in range(num_circles)]
    
    # Initial guess for radii: uniform
    initial_guess = np.full(num_circles, 0.1)

    # Optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-5
    )

    # Extract results
    radii = result.x
    sum_radii = np.sum(radii)
    centers = np.array(centers)
    
    return (centers, radii, sum_radii)
