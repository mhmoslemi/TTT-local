import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: equally spaced grid of 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial placement: equally spaced grid
    for i in range(num_circles):
        x = (i % 5 + 0.5) / 5
        y = (i // 5 + 0.5) / 5
        centers[i] = [x, y]
        radii[i] = 0.05
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraint 1: each circle must be inside the unit square
    def square_constraint(radii, centers):
        return np.array([r - (1 - r - x) for r, x in zip(radii, centers[:, 0])])
    
    # Constraint 2: no two circles overlap
    def overlap_constraint(radii, centers):
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return np.array([dists[i, j] - (radii[i] + radii[j]) for i in range(num_circles) for j in range(i + 1, num_circles)])
    
    # Optimization setup
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    
    # Define constraints
    cons = []
    cons.append({'type': 'ineq', 'fun': lambda x: square_constraint(x, centers)})
    cons.append({'type': 'ineq', 'fun': lambda x: overlap_constraint(x, centers)})
    
    # Optimize
    result = minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )
    
    # Update centers based on optimized radii
    final_radii = result.x
    final_centers = centers.copy()
    
    # Final check: ensure all circles are still inside square and no overlap
    for i in range(num_circles):
        x, y = final_centers[i]
        r = final_radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            raise ValueError("One or more circles are outside the unit square")
    
    # Calculate sum of radii
    sum_radii = np.sum(final_radii)
    
    return (final_centers, final_radii, sum_radii)
