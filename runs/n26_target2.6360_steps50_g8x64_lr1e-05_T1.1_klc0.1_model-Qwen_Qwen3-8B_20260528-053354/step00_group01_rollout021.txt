import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for centers (spread out in the unit square)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)  # Initial radius for all circles
    
    def objective_function(x):
        # x contains all parameters: first 52 values are centers (26 circles x 2 coordinates)
        # next 26 values are radii
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        sum_radii = np.sum(radii_flat)
        # Constraint 1: All circles must be inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf  # Penalize by returning infinity
        # Constraint 2: No two circles overlap
        dist_matrix = cdist(centers_flat, centers_flat)
        for i in range(26):
            for j in range(i + 1, 26):
                d = dist_matrix[i, j]
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if d < r_i + r_j - 1e-8:  # Check for overlap with tolerance
                    return np.inf
        # Objective is to maximize sum of radii, so minimize negative of sum
        return -sum_radii
    
    # Initial guess for parameters: centers and radii
    initial_guess = np.concatenate([centers.flatten(), radii])
    
    # Setup bounds: centers must be in [0,1], radii >= 0
    bounds = []
    for _ in range(26):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        bounds.append((0.0, 1.0))  # radius
    # Use L-BFGS-B as it handles bounds well
    result = minimize(objective_function, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-8)
    
    # Extract the optimized result
    centers_opt = result.x[:52].reshape(26, 2)
    radii_opt = result.x[52:]
    sum_radii_opt = -result.fun
    
    return (centers_opt, radii_opt, sum_radii_opt)
