import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = (i + 0.5) / n
        centers[i, 1] = 0.5

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint: circles must be inside the unit square
    def inside(radii, centers):
        return np.all(radii <= 1 - centers[:, 0] - radii) and np.all(radii <= 1 - centers[:, 1] - radii)

    # Constraint: no overlap between circles
    def no_overlap(radii, centers):
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Combined constraint function
    def constraint(radii):
        if not inside(radii, centers):
            return np.inf
        if not no_overlap(radii, centers):
            return np.inf
        return 0

    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(n)]

    # Optimization
    result = minimize(
        objective,
        np.full(n, 0.01),  # initial guess: small radii
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint}
    )

    radii = result.x
    # Ensure all radii are non-negative and inside the square
    radii = np.clip(radii, 0.0, 1.0 - centers[:, 0] - radii)
    radii = np.clip(radii, 0.0, 1.0 - centers[:, 1] - radii)
    # Final check for overlaps
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j]:
                radii = np.clip(radii, 0.0, 1.0 - centers[:, 0] - radii)
                radii = np.clip(radii, 0.0, 1.0 - centers[:, 1] - radii)

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
