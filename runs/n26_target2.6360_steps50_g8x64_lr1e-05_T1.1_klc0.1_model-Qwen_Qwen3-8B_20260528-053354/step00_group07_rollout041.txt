import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all the same
    radii = np.ones(26) * 0.1

    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: All circles must be inside the unit square
        # radius_i <= x_i - 0, 1 - x_i - radius_i, y_i - 0, 1 - y_i - radius_i
        inside = np.ones_like(radii)
        for i in range(26):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                inside[i] = 0
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                inside[i] = 0
        return inside

    # Constraint 2: No two circles overlap
    def non_overlap(radii):
        dists = np.zeros((26, 26))
        for i in range(26):
            for j in range(26):
                if i == j:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy) - (radii[i] + radii[j])
        return dists

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=objective,
            x0=radii,
            method='SLSQP',
            constraints=[
                {'type': 'ineq', 'fun': constraints},
                {'type': 'ineq', 'fun': non_overlap}
            ],
            bounds=[(0, 1)] * 26
        )

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun  # since we maximized by minimizing negative sum

    return (centers, radii, sum_radii)
