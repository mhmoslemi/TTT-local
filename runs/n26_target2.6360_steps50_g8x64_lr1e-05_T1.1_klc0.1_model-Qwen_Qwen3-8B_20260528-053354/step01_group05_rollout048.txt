import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    np.random.seed(42)
    centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
    radii = np.full(26, 0.05)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints_func(radii_vec):
        # Reinitializing centers for this configuration
        centers_config = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
        # Constraints: circle inside the unit square
        for i in range(26):
            if radii_vec[i] > 0:
                if centers_config[i, 0] - radii_vec[i] < 0 or centers_config[i, 0] + radii_vec[i] > 1:
                    return np.inf
                if centers_config[i, 1] - radii_vec[i] < 0 or centers_config[i, 1] + radii_vec[i] > 1:
                    return np.inf
        # Constraints: no overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_config[i, 0] - centers_config[j, 0]
                dy = centers_config[i, 1] - centers_config[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return np.inf
        return -np.sum(radii_vec)

    best_sum = 0.0
    best_centers = np.zeros((26, 2))
    best_radii = np.zeros(26)

    for _ in range(10):
        centers_config = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
        radii_initial = np.full(26, 0.05)
        result = minimize(objective, radii_initial, constraints=[{'type': 'ineq', 'fun': constraints_func}])
        if result.success:
            current_sum = np.sum(result.x)
            if current_sum > best_sum:
                best_sum = current_sum
                best_centers = centers_config
                best_radii = result.x

    return (best_centers, best_radii, best_sum)
