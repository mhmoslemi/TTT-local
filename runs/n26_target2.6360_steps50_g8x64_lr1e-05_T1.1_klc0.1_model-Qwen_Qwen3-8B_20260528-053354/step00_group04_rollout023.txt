import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Optimization parameters
    initial_radius = 0.05
    radii = np.full(num_circles, initial_radius)

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circle within the square and no overlap
    def constraint_circle(x):
        return np.array([x[0] - x[2], 1 - x[0] - x[2], x[1] - x[2], 1 - x[1] - x[2]], dtype=np.float64)

    def constraint_overlap(x, i, j):
        dx = x[2*i] - x[2*j]
        dy = x[2*i+1] - x[2*j+1]
        dist = np.sqrt(dx*dx + dy*dy)
        return dist - (x[2*i+2] + x[2*j+2])

    # Create a constraint for each circle being within the unit square
    cons = []
    for i in range(num_circles):
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_circle(x)[0]})
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_circle(x)[1]})
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_circle(x)[2]})
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_circle(x)[3]})

    # Create constraints for non-overlapping circles
    for i in range(num_circles):
        for j in range(i+1, num_circles):
            cons.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)})

    # Flatten the centers and radii into a single array for optimization
    x0 = np.concatenate((centers.flatten(), radii))
    bounds = [(0, 1) for _ in range(2*num_circles)] + [(0, 1) for _ in range(num_circles)]

    # Run the optimization
    res = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    # Extract the result
    x_res = res.x
    centers = x_res[:2*num_circles].reshape(num_circles, 2)
    radii = x_res[2*num_circles:]
    sum_radii = np.sum(radii)

    # Return the result
    return (centers, radii, sum_radii)
