import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    np.random.seed(42)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Initialize centers with random positions
        n = len(radii)
        centers = np.random.rand(n, 2)
        # Ensure circles are within the unit square
        for i in range(n):
            x, y = centers[i]
            if x < radii[i] or x > 1 - radii[i] or y < radii[i] or y > 1 - radii[i]:
                return np.inf
        # Ensure circles don't overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)
    
    # Initial guess: small random radii
    initial_radii = np.random.rand(26) * 0.05
    
    # Optimize with constraints
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}]
    )
    
    # Extract results
    radii = result.x
    centers = np.random.rand(26, 2)
    for i in range(26):
        x, y = centers[i]
        while x < radii[i] or x > 1 - radii[i] or y < radii[i] or y > 1 - radii[i]:
            centers[i] = np.random.rand(2)
    
    # Ensure no overlapping
    while True:
        overlap = False
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    overlap = True
                    break
            if overlap:
                break
        if not overlap:
            break
        # Adjust positions slightly to avoid overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    direction = np.array([dx, dy]) / dist
                    adjustment = (radii[i] + radii[j] - dist) / 2 * direction
                    centers[i] += adjustment
                    centers[j] -= adjustment
    
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
