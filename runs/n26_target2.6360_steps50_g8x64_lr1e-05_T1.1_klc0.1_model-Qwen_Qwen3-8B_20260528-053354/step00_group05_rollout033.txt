import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced grid of 5x5 points (25 circles)
    # Add one more circle in the center
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        x = (i % 5) / 4.0
        y = (i // 5) / 4.0
        centers[i] = [x, y]
    
    # Initial radii: assume equal radii for simplicity
    radii = np.full(n, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(n):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        # Ensure no two circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(n)]  # Radii can't exceed 0.5

    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP"
    )

    # Extract the optimized result
    optimized_radii = result.x
    optimized_centers = centers
    sum_radii = -result.fun

    return (optimized_centers, optimized_radii, sum_radii)
