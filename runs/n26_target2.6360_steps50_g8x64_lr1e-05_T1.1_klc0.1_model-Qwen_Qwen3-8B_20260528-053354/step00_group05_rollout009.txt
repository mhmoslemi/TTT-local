import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Initial guess for centers
        centers = np.zeros((26, 2))
        # Place circles with equal spacing
        for i in range(26):
            row = i // 5
            col = i % 5
            centers[i, 0] = col / 4 + 0.1
            centers[i, 1] = row / 4 + 0.1
        # Initial guess for radii
        # Use heuristic: radius = (1 - 0.2) / (sqrt(2) + 1) * 0.75 = approx 0.14
        initial_guess = np.full(26, 0.14)
        # Use a random perturbation to help optimization escape local minima
        perturbation = np.random.uniform(-0.01, 0.01, 26)
        radii = initial_guess + perturbation
        radii = np.clip(radii, 0, 1 - 0.1)  # Ensure radius is feasible
        # Optimization
        bounds = [(0, 1 - 0.1) for _ in range(26)]
        constraints = []
        # Constraint: all circles inside the square
        for i in range(26):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - 0.1})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - 0.1 - r[i]})
        # Constraint: no overlapping between circles
        for i in range(26):
            for j in range(i + 1, 26):
                def dist_constraint(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - (r[i] + r[j])
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        # Optimization
        result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')
        if result.success:
            final_radii = result.x
            centers = np.zeros((26, 2))
            for i in range(26):
                row = i // 5
                col = i % 5
                centers[i, 0] = col / 4 + 0.1
                centers[i, 1] = row / 4 + 0.1
            return (centers, final_radii, -result.fun)
        else:
            # Fallback to initial guess if optimization fails
            return (np.zeros((26, 2)), np.full(26, 0.14), 0.14 * 26)

    # Run the packing function
    centers, radii, sum_radii = run_packing()
    return (centers, radii, sum_radii)
