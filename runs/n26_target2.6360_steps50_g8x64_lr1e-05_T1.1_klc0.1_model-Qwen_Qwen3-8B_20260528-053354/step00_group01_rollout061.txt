import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = col + 0.1
        y = row + 0.1
        r = 0.1
        centers[i] = [x, y]
        radii[i] = r

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints
    # 1. All circles are fully inside the unit square
    def constraint_inside(x):
        r = x[::2]
        centers = x[1::2].reshape(-1, 2)
        for i in range(26):
            cx, cy = centers[i]
            if cx - r[i] < 0 or cx + r[i] > 1 or cy - r[i] < 0 or cy + r[i] > 1:
                return -1
        return 0

    # 2. Circles do not overlap
    def constraint_non_overlap(x):
        r = x[::2]
        centers = x[1::2].reshape(-1, 2)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return -1
        return 0

    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_non_overlap}
    ]

    # Bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(26)]

    # Flatten the centers and radii into a single vector
    initial_guess = np.concatenate([radii, centers.flatten()])

    # Set up optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6
    )

    # Extract results
    radii_opt = result.x[::2]
    centers_opt = result.x[1::2].reshape(26, 2)

    # Calculate sum of radii
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
