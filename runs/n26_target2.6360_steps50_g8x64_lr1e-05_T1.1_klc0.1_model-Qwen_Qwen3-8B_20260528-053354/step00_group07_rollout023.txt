import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial guess for radii: uniform
    radii = np.full(26, 0.1)

    # Objective function to minimize negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_square(radii):
        # Each circle is fully inside the unit square
        return np.sum((radii - (1 - radii)) ** 2)

    def constraint_overlap(radii):
        # No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return np.sum((dists - (radii[:, np.newaxis] + radii[np.newaxis, :])) ** 2)

    # Optimization
    bounds = [(0, 1) for _ in range(26)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_square(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
    ]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    centers_result = centers.copy()
    radii_result = result.x
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
