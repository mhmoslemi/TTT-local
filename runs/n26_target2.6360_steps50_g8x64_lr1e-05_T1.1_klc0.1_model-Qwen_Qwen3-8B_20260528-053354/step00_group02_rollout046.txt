import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.25
        centers[i, 1] = row * 0.25

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_center_inside(radii):
        return np.all(np.abs(centers[:, 0]) - radii >= 0) and np.all(np.abs(centers[:, 1]) - radii >= 0)

    def constraint_no_overlap(radii):
        dists = pdist(centers)
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization settings
    bounds = [(0.0, 1.0) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Suppress warnings from scipy.optimize
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, np.zeros(num_circles), bounds=bounds, constraints=constraints, method='SLSQP')

    radii = result.x
    centers = np.array(centers)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
