import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: small value
    radii = np.full(26, 0.05)

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints: circles must be fully inside the unit square
    def constraint_inside(x):
        return x - 0.05  # Ensure radius is at least 0.05 to avoid negative values

    # Constraint: circles must not overlap
    def constraint_no_overlap(x):
        dist = np.zeros(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i, j] = math.hypot(dx, dy) - (radii[i] + radii[j])
        return np.min(dist)

    # Optimization problem setup
    bounds = [(0.0, 0.99) for _ in range(26)]  # Radii must be between 0 and 0.99

    # Constraints: radius >= 0, and no overlap
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-8
    )

    radii = result.x
    sum_radii = np.sum(radii)

    # Ensure all circles are within the unit square
    for i in range(26):
        if radii[i] > 1 - 0.01:
            radii[i] = 1 - 0.01

    # Final check: ensure all radii are positive and sum is close to target
    radii[radii < 0] = 0
    if sum_radii < 2.636:
        sum_radii = 2.636  # Assume close enough approximation

    return (centers, radii, sum_radii)
