import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: distribute circles evenly
    centers = np.column_stack((np.linspace(0.1, 0.9, 26), np.linspace(0.1, 0.9, 26)))
    radii = np.full(26, 0.05)

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_1(radii):
        # Ensure all circles are within the unit square
        return np.array([r for r in radii if r > 0.0])

    def constraint_2(radii):
        # Ensure no circles overlap
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dists[idx] = np.linalg.norm(centers[i] - centers[j]) - (radii[i] + radii[j])
                idx += 1
        return dists

    def constraint_3(radii):
        # Ensure all radii are non-negative
        return radii

    # Optimization setup
    bounds = [(0.0, 0.5) for _ in range(26)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_1},
        {'type': 'ineq', 'fun': constraint_2},
        {'type': 'ineq', 'fun': constraint_3}
    ]

    # Optimization
    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    centers = np.column_stack((np.linspace(0.1, 0.9, 26), np.linspace(0.1, 0.9, 26)))
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
