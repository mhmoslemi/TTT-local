import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid points
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 7) / 6.0
        centers[i, 1] = (i // 7) / 6.0
    
    # Initial radii: uniform
    radii = np.full(26, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Constraints: circles are inside the square
        inside = np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]), axis=0)
        if not np.all(inside):
            return np.inf  # Penalize for circles outside the square
        
        # Constraints: circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf  # Penalize for overlapping circles
        
        return -np.sum(radii)  # Objective function
    
    # Optimization with constraints
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=lambda x: constraints(x),
            x0=radii,
            method='L-BFGS-B',
            bounds=[(0, 1 - np.min(centers[:, 0]), 0, 1 - np.min(centers[:, 1])) for _ in range(26)]
        )
    
    optimized_radii = result.x
    optimized_centers = centers.copy()
    
    # Ensure all circles are fully inside the square
    for i in range(26):
        if optimized_radii[i] > 1 - optimized_centers[i, 0]:
            optimized_radii[i] = 1 - optimized_centers[i, 0]
        if optimized_radii[i] > 1 - optimized_centers[i, 1]:
            optimized_radii[i] = 1 - optimized_centers[i, 1]
    
    # Ensure no two circles overlap
    for i in range(26):
        for j in range(i + 1, 26):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < optimized_radii[i] + optimized_radii[j]:
                # Adjust radii to prevent overlap
                adjusted_radius = (dist - optimized_radii[j]) / 2
                if adjusted_radius < optimized_radii[i]:
                    optimized_radii[i] = adjusted_radius
                else:
                    optimized_radii[j] = dist - optimized_radii[i]
    
    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
