import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (a simple grid)
    centers = np.array([[i / 5 + 0.1, j / 5 + 0.1] for i in range(5) for j in range(5)])
    # Initial guess for radii (same for all)
    radii = np.full(25, 0.1)
    # Add one more circle in the center
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = np.append(radii, 0.1)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        constraints = []
        centers = np.array([[i / 5 + 0.1, j / 5 + 0.1] for i in range(5) for j in range(5)])
        centers = np.vstack([centers, [0.5, 0.5]])
        for i in range(26):
            r_i = radii[i]
            # Constraint 1: Circle i is inside the unit square
            for coord in [0, 1]:
                constraints.append((centers[i, coord] - r_i) >= 0)
                constraints.append((1 - centers[i, coord] - r_i) >= 0)
            # Constraint 2: Circle i does not overlap with other circles
            for j in range(i + 1, 26):
                r_j = radii[j]
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append((dist - r_i - r_j) >= 0)
        return constraints
    
    # Optimization setup
    bounds = [(0.0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)
    
    # Extract results
    centers = np.array([[i / 5 + 0.1, j / 5 + 0.1] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
