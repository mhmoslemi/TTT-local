import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid of 26 points in the unit square
    # Each circle starts with radius 0.05 in the center
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles) + 0.05
    centers[:n_circles//2, 1] = np.linspace(0.05, 0.95, n_circles//2)
    centers[n_circles//2:, 0] = np.linspace(0.05, 0.95, n_circles//2)
    centers = centers.reshape(-1, 2)

    # Objective function: minimize the negative sum of radii (maximize sum)
    def objective(x):
        # x is a flat array of (x1, y1, r1, x2, y2, r2, ..., x26, y26, r26)
        x_centers = x[:2*n_circles].reshape(n_circles, 2)
        r = x[2*n_circles:]
        sum_r = np.sum(r)
        return -sum_r  # minimize negative sum to maximize sum

    # Inequality constraints: distance between centers >= sum of radii
    def constraints(x):
        x_centers = x[:2*n_circles].reshape(n_circles, 2)
        r = x[2*n_circles:]
        constraints = []
        for i in range(n_circles):
            for j in range(i+1, n_circles):
                dx = x_centers[i, 0] - x_centers[j, 0]
                dy = x_centers[i, 1] - x_centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (r[i] + r[j]))
        return np.array(constraints)

    # Bounds: x and y in [0.05, 0.95], r >= 0
    bounds = []
    for i in range(n_circles):
        x_bounds = (0.05, 0.95)
        y_bounds = (0.05, 0.95)
        r_bounds = (0.0, 1.0)
        bounds.append(x_bounds)
        bounds.append(y_bounds)
        bounds.append(r_bounds)

    # Initial x vector
    x0 = np.concatenate([centers.ravel(), radii])

    # Optimization settings
    options = {'disp': False, 'ftol': 1e-8, 'maxiter': 1000}
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints}, options=options)

    # Extract the solution
    x_opt = result.x
    centers_opt = x_opt[:2*n_circles].reshape(n_circles, 2)
    radii_opt = x_opt[2*n_circles:]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
