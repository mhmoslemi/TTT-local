import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial radius is 0.04
    radii[:] = 0.04
    # Spread the circles in a grid-like pattern
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.4 + 0.2
        centers[i, 1] = row * 0.4 + 0.2

    def objective(radii, centers):
        return -np.sum(radii)

    def constraint_bounds(centers, radii):
        # Ensure circles are within the square
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return False
        return True

    def constraint_overlap(centers, radii):
        # Ensure circles don't overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def constraint_func(x):
        # x is a flat array of [r0, r1, ..., r25, x0, y0, x1, y1, ..., x25, y25]
        # Reshape to get radii and centers
        radii = x[:num_circles]
        centers = x[num_circles:].reshape((num_circles, 2))

        # Check if bounds are satisfied
        if not constraint_bounds(centers, radii):
            return np.inf

        # Check if circles don't overlap
        if not constraint_overlap(centers, radii):
            return np.inf

        return -np.sum(radii)

    # Initial guess: radii and centers
    initial_guess = np.concatenate([radii, centers.flatten()])

    # Bounds: radii >= 0, centers in [0, 1]
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))  # radius for i-th circle
        bounds.append((0, 1))  # x for i-th circle
        bounds.append((0, 1))  # y for i-th circle

    result = minimize(
        constraint_func,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        tol=1e-6
    )

    radii_result = result.x[:num_circles]
    centers_result = result.x[num_circles:].reshape((num_circles, 2))
    sum_radii_result = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii_result)
