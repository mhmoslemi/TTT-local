import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    # Objective function to minimize (negative sum of radii to maximize)
    def objective(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraints
    constraints = []

    # Constraint: every circle is within the unit square
    def constraint_square(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return 1.0
        return 0.0

    constraints.append({'type': 'ineq', 'fun': constraint_square})

    # Constraint: no two circles overlap
    def constraint_overlap(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return 1.0
        return 0.0

    constraints.append({'type': 'ineq', 'fun': constraint_overlap})

    # Bounds for centers and radii (centers in [0,1], radii >= 0)
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])
    for _ in range(26):
        bounds.extend([(0.0, 1.0)])

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=UserWarning)
        result = minimize(objective, np.concatenate([centers.flatten(), radii]), method='SLSQP', bounds=bounds, constraints=constraints)

    centers_opt = result.x[:52].reshape(26, 2)
    radii_opt = result.x[52:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
