import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    np.random.seed(42)
    n_circles = 26
    initial_centers = np.random.uniform(0.1, 0.9, (n_circles, 2))
    initial_radii = np.random.uniform(0.01, 0.05, n_circles)

    def objective_function(x):
        centers = x[:n_circles*2].reshape(n_circles, 2)
        radii = x[n_circles*2:]
        sum_r = np.sum(radii)
        return -sum_r  # Minimize negative sum to maximize sum

    def constraints(x):
        centers = x[:n_circles*2].reshape(n_circles, 2)
        radii = x[n_circles*2:]
        constraints = []
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (radii[i] + radii[j]))
        for i in range(n_circles):
            x_center = centers[i, 0]
            y_center = centers[i, 1]
            r = radii[i]
            constraints.append(x_center - r)
            constraints.append(1 - x_center - r)
            constraints.append(y_center - r)
            constraints.append(1 - y_center - r)
        return constraints

    # Initial guess
    x0 = np.concatenate((initial_centers.flatten(), initial_radii))

    # Optimization
    result = minimize(objective_function, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints}, bounds=[(0, 1) for _ in range(n_circles*2)] + [(0, 1) for _ in range(n_circles)])

    # Extract results
    centers = result.x[:n_circles*2].reshape(n_circles, 2)
    radii = result.x[n_circles*2:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
