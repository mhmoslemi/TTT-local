import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers (all at the same position)
    initial_centers = np.random.rand(26, 2)
    initial_radii = np.random.rand(26) * 0.1

    def objective_function(params):
        centers = params[:52].reshape(26, 2)
        radii = params[52:]
        sum_radii = np.sum(radii)
        # Constraint: all circles inside the unit square
        inside = np.all((np.min(centers, axis=1) >= radii) & (np.max(centers, axis=1) <= 1 - radii))
        # Constraint: no overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -sum_radii  # minimizing negative sum is equivalent to maximizing sum

    # Bounds for parameters
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # centers
    for _ in range(26):
        bounds.extend([(0.0, 0.15)])  # radii (assuming maximum radius is 0.15)

    # Optimization
    result = minimize(objective_function, np.concatenate([initial_centers.ravel(), initial_radii]),
                      method='L-BFGS-B', bounds=bounds, tol=1e-6)

    centers = result.x[:52].reshape(26, 2)
    radii = result.x[52:]
    sum_radii = -result.fun  # since we minimized negative sum

    return (centers, radii, sum_radii)
