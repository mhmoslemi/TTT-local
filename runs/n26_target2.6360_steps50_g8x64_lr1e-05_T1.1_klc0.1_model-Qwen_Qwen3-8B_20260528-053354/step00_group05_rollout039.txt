import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: equally spaced points in grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_radius(centers, radii):
        for i in range(26):
            if radii[i] < 0:
                return False
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return False
        return True

    def constraint_overlap(centers, radii):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def constraints(radii):
        return [constraint_radius(centers, radii), constraint_overlap(centers, radii)]

    bounds = [(0.0, 0.5) for _ in range(26)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraint_radius, 'args': (centers,)},
                     {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers,)}]
    )

    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
