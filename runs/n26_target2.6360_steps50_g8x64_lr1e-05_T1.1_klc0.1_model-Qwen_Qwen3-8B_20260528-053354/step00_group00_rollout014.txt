```python
import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    centers = centers[:26]  # Only take first 26 points
    radii = np.full(26, 0.1)  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraint1(radii):
        # Ensure each circle is inside the unit square
        for i in range(26):
            if radii[i] < 0:
                return False
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        return True

    def constraint2(radii):
        # Ensure circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization bounds
    bounds = [(0.0, 0.5) for _ in range(26)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'maxiter': 1000}
    )

    # Extract solution
    radii_opt = result.x
    sum_radii = -result.fun

    # Ensure the solution is valid
    if not constraint1(radii_opt) or not constraint2(radii_opt):
        raise ValueError("Optimization result does not satisfy constraints")

    return (centers, radii_opt, sum_radii)
```
