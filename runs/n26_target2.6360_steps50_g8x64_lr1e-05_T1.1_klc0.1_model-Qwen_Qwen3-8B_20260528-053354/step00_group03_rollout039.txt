import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    np.random.seed(42)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        centers = np.random.rand(26, 2)
        centers = np.clip(centers, radii.reshape(-1, 1), 1 - radii.reshape(-1, 1))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0
    
    def bounds():
        return [(0, 1) for _ in range(26)]
    
    radii_initial = np.random.uniform(0, 0.1, 26)
    result = minimize(objective, radii_initial, method='SLSQP', bounds=bounds, constraints=constraints)
    
    centers = np.random.rand(26, 2)
    centers = np.clip(centers, result.x.reshape(-1, 1), 1 - result.x.reshape(-1, 1))
    
    for i in range(26):
        for j in range(i + 1, 26):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < result.x[i] + result.x[j]:
                centers = np.random.rand(26, 2)
                centers = np.clip(centers, result.x.reshape(-1, 1), 1 - result.x.reshape(-1, 1))
    
    return (centers, result.x, -result.fun)
