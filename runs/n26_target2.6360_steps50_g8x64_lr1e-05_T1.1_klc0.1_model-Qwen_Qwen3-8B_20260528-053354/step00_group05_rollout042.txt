import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    radii = np.full(25, 0.05)  # Initial radius

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be within the unit square and not overlap
    def constraint1(radii):
        # Ensure each circle is within the unit square
        return np.min(radii - np.minimum(centers[:, 0], 1 - centers[:, 0]) - np.minimum(centers[:, 1], 1 - centers[:, 1]))

    def constraint2(radii):
        # Ensure no two circles overlap
        distances = cdist(centers, centers)
        radii_expanded = np.repeat(radii, len(radii), axis=0).reshape(-1, len(radii))
        return np.min(distances - radii_expanded - radii_expanded.T)

    # Objective and constraints
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=cons, tol=1e-8)

    # Extract results
    centers = centers[:len(result.x)]
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
