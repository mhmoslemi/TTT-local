import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    n_circles = 26
    centers = np.array([(i / (n_circles - 1), j / (n_circles - 1)) for j in range(n_circles) for i in range(n_circles)])
    radii = np.full(n_circles, 0.05)  # Initial guess for radii

    # Objective function: maximize the sum of radii
    def objective(radii_vector):
        return -np.sum(radii_vector)

    # Constraints: (1) circles must be inside the square, (2) circles must not overlap
    def constraint_inside(radii_vector):
        return np.array([r <= 1 - r for r in radii_vector], dtype=np.float64)

    def constraint_overlap(radii_vector):
        centers = np.reshape(centers, (-1, 2))
        dists = np.zeros((n_circles, n_circles))
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return np.array([dists[i, j] >= radii_vector[i] + radii_vector[j] for i in range(n_circles) for j in range(n_circles)], dtype=np.float64)

    # Setup optimization
    bounds = [(0, 1) for _ in range(n_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract result
    sum_radii = result.fun
    centers = np.reshape(centers, (-1, 2))
    radii = result.x

    return (centers, radii, -sum_radii)
