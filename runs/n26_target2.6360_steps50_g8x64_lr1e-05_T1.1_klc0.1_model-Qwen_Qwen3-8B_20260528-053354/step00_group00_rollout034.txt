import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as distance

def run_packing():
    # Initial guess: evenly spaced grid
    grid = np.array([[x / 5, y / 5] for x in range(5) for y in range(5)])
    centers = np.vstack([grid, np.array([[0.5, 0.1], [0.5, 0.9], [0.1, 0.5], [0.9, 0.5], [0.5, 0.5]])])
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Box constraints: each radius must be <= distance to boundary
        bounds = []
        for i in range(26):
            r = radii[i]
            d_left = centers[i, 0] - r
            d_right = 1 - centers[i, 0] - r
            d_bottom = centers[i, 1] - r
            d_top = 1 - centers[i, 1] - r
            bounds.append((0, min(d_left, d_right, d_bottom, d_top)))
        return bounds

    def non_overlap(radii):
        dists = distance.cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < radii[i] + radii[j]:
                    return False
        return True

    # Wrap objective and constraints into a format suitable for scipy.optimize.minimize
    def objective_func(radii_flat):
        radii = radii_flat.reshape(26)
        return -np.sum(radii)

    def constraint_func(radii_flat):
        radii = radii_flat.reshape(26)
        for i in range(26):
            r = radii[i]
            d_left = centers[i, 0] - r
            d_right = 1 - centers[i, 0] - r
            d_bottom = centers[i, 1] - r
            d_top = 1 - centers[i, 1] - r
            if r > min(d_left, d_right, d_bottom, d_top):
                return 1
        return 0

    def non_overlap_func(radii_flat):
        radii = radii_flat.reshape(26)
        dists = distance.cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < radii[i] + radii[j]:
                    return 1
        return 0

    # Set up constraints for scipy.optimize.minimize
    cons = (
        {'type': 'ineq', 'fun': constraint_func},
        {'type': 'ineq', 'fun': non_overlap_func}
    )

    # Initial guess
    initial_guess = np.full(26, 0.1)

    # Perform minimization (minimize negative sum to maximize)
    result = minimize(objective_func, initial_guess, constraints=cons, method='SLSQP')

    # Extract the optimal solution
    optimal_radii = result.x
    sum_radii = -result.fun

    # Return the result
    return (centers, optimal_radii, sum_radii)
