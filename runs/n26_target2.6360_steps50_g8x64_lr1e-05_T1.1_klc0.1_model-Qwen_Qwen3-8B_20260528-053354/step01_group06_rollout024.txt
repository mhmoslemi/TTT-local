import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers: evenly spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial guess for radii: same for all circles
    initial_radii = np.ones(num_circles) * 0.1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: All circles must be fully inside the square
    def constraint_inside(radii):
        return np.ones(num_circles) * (1 - 2 * radii) - np.concatenate([radii, radii])
    cons_inside = {'type': 'ineq', 'fun': constraint_inside}

    # Constraint 2: No two circles overlap
    def constraint_no_overlap(radii):
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[i, j] = dist - (radii[i] + radii[j])
                dists[j, i] = dist - (radii[i] + radii[j])
        return np.min(dists)
    cons_overlap = {'type': 'ineq', 'fun': constraint_no_overlap}

    # Constraints
    constraints = [cons_inside, cons_overlap]

    # Bounds for radii
    bounds = [(0, 1 - 2 * np.min(centers, axis=0))] * num_circles

    # Optimization
    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
