import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    radii = np.clip(radii, 0, 0.5)  # initial radii between 0 and 0.5

    def objective_function(x):
        # Flatten input to get centers and radii
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:].reshape(26)
        radii_flat = np.clip(radii_flat, 0, 0.5)  # clamp radii to be within [0, 0.5]

        # Check if all circles are inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return float('inf')

        # Check for overlaps between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return float('inf')

        # Objective is to maximize the sum of radii, so minimize negative sum
        return -np.sum(radii_flat)

    # Define bounds for centers and radii
    bounds = []
    for _ in range(26):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
        bounds.append((0, 0.5))  # radius

    # Optimization settings
    options = {'disp': False, 'ftol': 1e-6, 'maxiter': 1000}

    # Run the optimization
    result = minimize(objective_function, x0=np.concatenate([centers.flatten(), radii.flatten()]), bounds=bounds, method='L-BFGS-B', options=options)

    # Extract the optimized result
    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26*2:].reshape(26)
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
