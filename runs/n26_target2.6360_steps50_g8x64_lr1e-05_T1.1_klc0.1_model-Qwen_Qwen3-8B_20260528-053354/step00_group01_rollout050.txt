import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly distribute 26 circles in a grid
    # 5 rows and 6 columns to fit within unit square
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 6 / 5.0
        y = i // 6 / 5.0
        centers[i] = [x, y]

    # Initial radii: set to a small value such that circles do not overlap initially
    radii = np.full(26, 0.05)

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Constraints
    def constraints(x):
        # x is a flattened array of centers and radii
        centers = x[:2*26].reshape(26, 2)
        radii = x[2*26:]
        # Constraint 1: circles inside the unit square
        inside = np.all((centers[:, 0] + radii >= 0.01) &
                        (centers[:, 0] - radii <= 0.99) &
                        (centers[:, 1] + radii >= 0.01) &
                        (centers[:, 1] - radii <= 0.99))
        if not inside:
            return np.inf

        # Constraint 2: circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    return np.inf

        # Constraint 3: radii non-negative
        non_negative = np.all(radii >= 0)
        if not non_negative:
            return np.inf

        return 0.0

    # Prepare initial parameters
    x0 = np.concatenate((centers.flatten(), radii))

    # Minimize negative sum of radii with constraints
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        bounds=[(0, 1) for _ in range(2*26)] + [(0, 1) for _ in range(26)]
    )

    # Extract results
    centers_out = result.x[:2*26].reshape(26, 2)
    radii_out = result.x[2*26:]
    sum_radii = -result.fun  # convert back to sum of radii

    return (centers_out, radii_out, sum_radii)
