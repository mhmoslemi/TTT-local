import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(26, 2)
    # Initial guess for radii (equal for all)
    radii = np.full(26, 0.1)
    # Objective function: minimize the negative sum of radii
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        sum_r = np.sum(radii_flat)
        # Constraint: each circle must be inside the unit square
        for i in range(26):
            x_i, y_i, r_i = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint: circles must not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return -sum_r  # Minimize the negative sum (i.e., maximize sum)

    # Flatten centers and radii for optimization
    x0 = np.hstack((centers.flatten(), radii))
    # Bounds for centers (0 to 1) and radii (0 to 0.5)
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    # Options for optimization
    opt = {
        'method': 'L-BFGS-B',
        'bounds': bounds,
        'tol': 1e-4,
        'maxiter': 1000
    }
    # Run optimization
    result = minimize(objective, x0, **opt)
    # Extract results
    optimal_centers = result.x[:52].reshape(26, 2)
    optimal_radii = result.x[52:]
    optimal_sum = -result.fun
    return optimal_centers, optimal_radii, optimal_sum
