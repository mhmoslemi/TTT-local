import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a regular grid
    num_circles = 26
    centers = np.array([[i / np.sqrt(num_circles), j / np.sqrt(num_circles)] for i in range(num_circles) for j in range(num_circles)])
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are within bounds
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Ensure no overlap between circles
        dists = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dists[i, j] < radii[i] + radii[j]:
                    return False
        return True

    def evaluate(radii):
        if not constraints(radii):
            return np.inf
        return -np.sum(radii)

    result = minimize(evaluate, radii, method='SLSQP', bounds=[(0, 1)] * num_circles, options={'maxiter': 1000})

    centers = np.array([[x, y] for x, y in centers])
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
