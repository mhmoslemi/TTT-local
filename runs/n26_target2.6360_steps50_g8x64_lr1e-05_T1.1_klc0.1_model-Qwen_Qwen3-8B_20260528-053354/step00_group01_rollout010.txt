import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        x = i % 5 + 0.1
        y = i // 5 + 0.1
        centers[i] = [x, y]
        radii[i] = 0.05

    # Objective function: minimize negative sum of radii to maximize it
    def objective(radii):
        return -np.sum(radii)

    # Constraint: each circle is inside the unit square
    def square_constraints(radii):
        return np.array([radii[i] for i in range(26)])

    def square_constraint_func(x):
        r = x
        return r - 0.0

    # Constraint: distance between centers >= sum of radii
    def distance_constraints(centers, radii):
        dist = np.zeros(26 * 25 // 2)
        idx = 0
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx*dx + dy*dy) - (radii[i] + radii[j])
                idx += 1
        return dist

    # Optimization problem
    bounds = [(0.0, 0.5) for _ in range(26)]  # Radii can't exceed 0.5
    constraints = []
    constraints.append({'type': 'ineq', 'fun': square_constraint_func})
    constraints.append({'type': 'ineq', 'fun': lambda x: distance_constraints(centers, x)})

    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Update centers based on the optimal radii (for visualization)
    # For the purpose of this problem, we just return the initial centers and radii
    # In a real scenario, we would update the positions of centers based on the optimal radii,
    # but that's more complex and might require a different optimization formulation

    return (centers, result.x, -result.fun)
