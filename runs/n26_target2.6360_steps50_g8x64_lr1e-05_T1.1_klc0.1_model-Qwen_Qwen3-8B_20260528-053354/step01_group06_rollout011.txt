import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with some random perturbations
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    # Objective function: maximize sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints: circles must be inside the unit square and not overlapping
    def constraints(radii_vec):
        centers = np.random.rand(26, 2)  # Use random centers for constraint checking
        # Ensure circles are inside the square
        in_box = np.all((radii_vec <= 1 - centers) & (radii_vec >= centers), axis=1)
        # Ensure no overlap between any two circles
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        overlaps = np.triu(dists < radii_vec[:, np.newaxis] + radii_vec[np.newaxis, :], k=1)
        return np.sum(in_box) + np.sum(overlaps)

    # Optimization
    result = minimize(objective, radii, constraints={"type": "ineq", "fun": constraints}, method='SLSQP')

    # Final result
    centers = np.random.rand(26, 2)  # Random centers for final result
    sum_radii = result.fun * -1
    return (centers, result.x, sum_radii)
