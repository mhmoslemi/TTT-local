import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.04)
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more center at center of square
    radii = np.append(radii, 0.0)
    centers = centers[:26]
    radii = radii[:26]
    
    # Optimization bounds for centers
    bounds = [(0.0, 1.0) for _ in range(26*2)]
    
    # Optimization constraints
    def constraint_center_inside(x):
        centers = x.reshape(26, 2)
        return np.min([centers[:, 0] - radii, 1.0 - centers[:, 0], centers[:, 1] - radii, 1.0 - centers[:, 1]])

    def constraint_no_overlap(x):
        centers = x.reshape(26, 2)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -dist + radii[i] + radii[j]
        return 0.0

    cons = [{'type': 'ineq', 'fun': constraint_center_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}]
    
    # Optimization objective: maximize the sum of radii
    def objective(x):
        centers = x.reshape(26, 2)
        return -np.sum(radii)  # We minimize the negative sum to maximize it
    
    # Initial guess flattened
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Perform optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)
