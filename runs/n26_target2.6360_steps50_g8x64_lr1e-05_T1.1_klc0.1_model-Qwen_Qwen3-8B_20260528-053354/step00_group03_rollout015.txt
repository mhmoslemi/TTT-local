import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly placed circles with small radii
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.05)
    
    # Objective function: minimize the negative sum of radii
    def objective(x):
        return -np.sum(x[26:])
    
    # Constraints: 1. Circles inside the square
    #               2. Circles do not overlap
    # Flatten centers and radii for optimization
    x0 = np.concatenate((centers.flatten(), radii))
    
    # Constraints for circles inside the square
    def constraint_inside(x):
        centers = x[:26*2].reshape(26, 2)
        radii = x[26*2:]
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0] - radii, 
                          centers[:, 1] - radii, 1 - centers[:, 1] - radii)
    
    # Constraints for circles not overlapping
    def constraint_overlap(x):
        centers = x[:26*2].reshape(26, 2)
        radii = x[26*2:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -1.0
        return 0.0
    
    # Combine all constraints
    cons_inside = [{'type': 'ineq', 'fun': lambda x: constraint_inside(x)}]
    cons_overlap = [{'type': 'ineq', 'fun': lambda x: constraint_overlap(x)}]
    cons = cons_inside + cons_overlap
    
    # Bounds for radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(26*2)] + [(0.0, 0.5) for _ in range(26)]
    
    # Optimization: minimize -sum(radii)
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons)
    
    # Extract optimized values
    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26*2:]
    sum_radii = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii)
