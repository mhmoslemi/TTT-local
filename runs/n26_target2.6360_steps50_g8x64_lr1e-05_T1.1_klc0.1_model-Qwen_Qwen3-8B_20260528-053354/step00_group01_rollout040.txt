import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: Place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i] = [col / 4 + 0.1, row / 4 + 0.1]
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i] = [i % 5 / 4 + 0.1, i // 5 / 4 + 0.1]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    return False
        return True

    # Use scipy.optimize.minimize to maximize the sum of radii with constraints
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': lambda x: x},
            {'type': 'ineq', 'fun': lambda x: np.array([0.1 for _ in range(26)]) - x}
        ],
        tol=1e-6,
        options={'disp': False, 'maxiter': 1000}
    )

    best_radii = result.x
    best_centers = np.zeros((26, 2))
    for i in range(26):
        # Generate new centers using a grid to avoid overlap
        row = i // 5
        col = i % 5
        best_centers[i] = [col / 4 + 0.1, row / 4 + 0.1]

    # Ensure circles are within the unit square
    for i in range(26):
        if best_centers[i, 0] - best_radii[i] < 0 or best_centers[i, 0] + best_radii[i] > 1:
            best_radii[i] = np.min([best_centers[i, 0], 1 - best_centers[i, 0]])
        if best_centers[i, 1] - best_radii[i] < 0 or best_centers[i, 1] + best_radii[i] > 1:
            best_radii[i] = np.min([best_centers[i, 1], 1 - best_centers[i, 1]])

    # Final check for non-overlapping
    for i in range(26):
        for j in range(i + 1, 26):
            dx = best_centers[i, 0] - best_centers[j, 0]
            dy = best_centers[i, 1] - best_centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < best_radii[i] + best_radii[j] - 1e-6:
                if best_radii[i] > best_radii[j]:
                    best_radii[i] -= 0.01
                else:
                    best_radii[j] -= 0.01

    sum_radii = np.sum(best_radii)
    return best_centers, best_radii, sum_radii
