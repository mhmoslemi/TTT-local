import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: randomly distribute circles in the unit square
    np.random.seed(0)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    def objective_function(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_inside(radii):
        x = centers[:, 0]
        y = centers[:, 1]
        return np.array([radii[i] <= x[i] for i in range(26)] + [radii[i] <= 1 - x[i] for i in range(26)] +
                        [radii[i] <= y[i] for i in range(26)] + [radii[i] <= 1 - y[i] for i in range(26)])
    
    def constraint_overlap(radii):
        dist = cdist(centers, centers)
        diag = np.diag(dist)
        dist[np.diag_indices_from(dist)] = np.inf
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :]) >= 0

    bounds = [(0, 1) for _ in range(26)]

    # Optimization problem
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    result = minimize(objective_function, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Update centers and radii based on the result
    radii = result.x
    sum_radii = -result.fun

    # Ensure all circles are within the unit square
    for i in range(26):
        if radii[i] > centers[i, 0] or radii[i] > 1 - centers[i, 0]:
            radii[i] = min(centers[i, 0], 1 - centers[i, 0])
        if radii[i] > centers[i, 1] or radii[i] > 1 - centers[i, 1]:
            radii[i] = min(centers[i, 1], 1 - centers[i, 1])

    return (centers, radii, sum_radii)
