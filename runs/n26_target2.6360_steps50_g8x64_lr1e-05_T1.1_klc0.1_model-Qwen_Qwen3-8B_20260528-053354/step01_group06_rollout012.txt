import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: equally spaced grid of 26 points, with small radii
    initial_centers = np.array([(x, y) for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    initial_centers = np.vstack((initial_centers, [(0.5, 0.1), (0.5, 0.9), (0.1, 0.5), (0.9, 0.5)]))
    initial_radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.vstack((np.array([(x, y) for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)]), 
                             [(0.5, 0.1), (0.5, 0.9), (0.1, 0.5), (0.9, 0.5)]))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(26):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        return True

    # Optimization without external dependencies
    result = minimize(objective, initial_radii, method='SLSQP', bounds=[(0, 1)]*26, constraints=({'type': 'ineq', 'fun': constraints} if constraints(initial_radii) else None))
    
    centers = np.vstack((np.array([(x, y) for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)]), 
                         [(0.5, 0.1), (0.5, 0.9), (0.1, 0.5), (0.9, 0.5)]))
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
