import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    n = 26
    centers = np.zeros((n, 2))
    radii = np.full(n, 0.1)

    # Generate a grid of initial positions
    for i in range(n):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.4 + 0.2
        centers[i, 1] = row * 0.4 + 0.2

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers_new = np.zeros((n, 2))
        for i in range(n):
            centers_new[i, 0] = centers[i, 0] - 0.1 * radii[i] / 0.5
            centers_new[i, 1] = centers[i, 1] - 0.1 * radii[i] / 0.5
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_new[i, 0] - centers_new[j, 0]
                dy = centers_new[i, 1] - centers_new[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        for i in range(n):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        return 0

    bounds = [(0, 0.5) for _ in range(n)]
    res = minimize(objective, radii, method='L-BFGS-B', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = centers[i, 0] - 0.1 * res.x[i] / 0.5
        centers[i, 1] = centers[i, 1] - 0.1 * res.x[i] / 0.5

    return (centers, res.x, -res.fun)
