import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.ones(n_circles) * 0.1  # Initial radius

    # Initial grid coordinates
    grid_x = np.linspace(0.1, 0.9, 6)
    grid_y = np.linspace(0.1, 0.9, 6)
    grid = np.array([[x, y] for y in grid_y for x in grid_x])
    centers = grid[:n_circles]

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure radii are non-negative
        if np.any(radii < 0):
            return np.inf
        # Ensure circles are within the unit square
        if np.any(radii > 1 - np.min(np.vstack([np.min(centers, axis=0), 1 - np.max(centers, axis=0)]))):
            return np.inf
        # Ensure circles do not overlap
        dists = cdist(centers, centers)
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1) for _ in range(n_circles)],
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
