import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.3 + 0.15
        centers[i, 1] = row * 0.3 + 0.15
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: Circles must be inside the unit square
        for i in range(26):
            r = radii[i]
            if centers[i, 0] - r < 0 or centers[i, 0] + r > 1 or centers[i, 1] - r < 0 or centers[i, 1] + r > 1:
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Bounds: radii must be non-negative
    bounds = [(0, 0.5) for _ in range(26)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
