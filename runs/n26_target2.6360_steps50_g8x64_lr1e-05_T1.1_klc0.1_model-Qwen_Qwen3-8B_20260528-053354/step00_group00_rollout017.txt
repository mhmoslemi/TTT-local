import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(x):
        centers = x.reshape((26, 2))
        radii = np.abs(x[::2])  # Take absolute values for radii

        # Ensure circles are within the unit square
        for i in range(26):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return np.inf

        # Ensure no circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf

        return -np.sum(radii)

    # Initial guess: place circles in grid
    def initial_guess():
        centers = np.zeros((26, 2))
        for i in range(26):
            row = i // 5
            col = i % 5
            centers[i, 0] = col * 0.2 + 0.1
            centers[i, 1] = row * 0.2 + 0.1
        radii = np.ones(26) * 0.1
        return centers.ravel()

    x0 = initial_guess()
    bounds = [(0, 1) for _ in range(52)]  # 26 centers with 2 coordinates each

    # Optimization
    result = minimize(
        objective,
        x0,
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        method='SLSQP',
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    centers = result.x.reshape((26, 2))
    radii = np.abs(centers[:, 0])  # Extract radii (first coordinate)
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
