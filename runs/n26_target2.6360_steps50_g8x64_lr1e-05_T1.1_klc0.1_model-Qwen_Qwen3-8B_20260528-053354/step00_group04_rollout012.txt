import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (26, 2))
    radii = np.random.uniform(0.01, 0.05, 26)

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint 1: Radii must be non-negative
        cons = [{'type': 'ineq', 'fun': lambda r: r}]
        # Constraint 2: Circles must be inside the unit square
        for i in range(26):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 0] + r[i]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - centers[i, 0] + r[i]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 1] + r[i]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - centers[i, 1] + r[i]})

        # Constraint 3: Circles must not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                def cons_overlap(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    return dist - r[i] - r[j]
                cons.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: cons_overlap(r)})
        return cons

    # Optimize
    bounds = [(0, 0.5) for _ in range(26)]
    res = minimize(objective, radii, args=(centers,), method='SLSQP', bounds=bounds, constraints=constraints(radii, centers))

    # Extract results
    radii_opt = res.x
    centers_opt = centers
    sum_radii = -res.fun

    return (centers_opt, radii_opt, sum_radii)
