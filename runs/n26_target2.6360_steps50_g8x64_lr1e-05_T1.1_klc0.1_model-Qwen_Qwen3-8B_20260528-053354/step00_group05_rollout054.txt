import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced grid with some offset
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = (i % 5) / 4.0 + 0.1
        y = (i // 5) / 4.0 + 0.1
        centers[i] = [x, y]

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: each circle must be within the unit square
    def constraint_inside(radii, centers):
        return np.array([radii[i] <= centers[i, 0] for i in range(num_circles)])
    
    # Constraint: no overlapping circles
    def constraint_no_overlap(radii, centers):
        dist = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i, j] = np.sqrt(dx**2 + dy**2)
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Initial radii guess: small values
    initial_radii = np.full(num_circles, 0.05)

    # Setup optimization problem
    bounds = [(0, 1.0) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers).min()}
    ]

    # Run optimization
    result = minimize(objective, initial_radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
