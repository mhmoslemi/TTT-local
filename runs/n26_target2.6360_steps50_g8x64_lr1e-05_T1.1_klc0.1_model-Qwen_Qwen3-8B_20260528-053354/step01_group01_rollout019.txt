import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial guess: grid-like pattern with some spacing
    grid = np.linspace(0.1, 0.9, 6)
    idx = 0
    for i, x in enumerate(grid):
        for j, y in enumerate(grid):
            if idx < num_circles:
                centers[idx] = [x, y]
                radii[idx] = 0.03  # Small initial radius
            idx += 1

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraint_square_func(radii_vec):
        constraints = []
        for i in range(num_circles):
            constraints.append(centers[i, 0] - radii_vec[i])
            constraints.append(1 - centers[i, 0] - radii_vec[i])
            constraints.append(centers[i, 1] - radii_vec[i])
            constraints.append(1 - centers[i, 1] - radii_vec[i])
        return np.array(constraints)

    def constraint_overlap_func(radii_vec):
        dists = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                dists.append(dist - (radii_vec[i] + radii_vec[j]))
        return np.array(dists)

    bounds = [(0, np.inf) for _ in range(num_circles)]

    try:
        result = minimize(
            fun=objective,
            x0=radii,
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_square_func},
                {'type': 'ineq', 'fun': constraint_overlap_func}
            ],
            method='SLSQP',
            options={'ftol': 1e-9, 'maxiter': 1000}
        )
    except:
        result = minimize(
            fun=objective,
            x0=radii,
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_square_func},
                {'type': 'ineq', 'fun': constraint_overlap_func}
            ],
            method='L-BFGS-B',
            options={'ftol': 1e-9, 'maxiter': 1000}
        )

    optimal_radii = result.x
    optimal_centers = centers.copy()
    sum_radii = np.sum(optimal_radii)
    
    return (optimal_centers, optimal_radii, sum_radii)
