import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers (evenly spaced)
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i + 0.5) / np.sqrt(num_circles)
        centers[i, 1] = 0.5

    # Initial radius guess
    radii = np.full(num_circles, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Negative because we minimize

    # Constraints
    def constraint_circle_inside(x):
        return x - 0.0001  # Ensure r > 0 to avoid division by zero

    def constraint_circle_overlap(x):
        # x contains [radii, centers[0][0], centers[0][1], ...]
        # Unpack variables
        radii = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)

        # Check if all circles are fully inside the square
        for r, (x, y) in zip(radii, centers):
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -np.inf  # Feasibility violated

        # Check pairwise distances between centers
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf  # Overlap detected

        return 0  # No constraints violated

    # Prepare variables for optimization
    # x contains [radii, centers[0][0], centers[0][1], ...]
    bounds = []
    for _ in range(num_circles):
        bounds.append((0.0, 0.5))  # Radii must be between 0 and 0.5
    for _ in range(num_circles * 2):
        bounds.append((0.0, 1.0))  # Centers must be inside the unit square

    # Initial guess with radii and centers
    x0 = np.concatenate((radii, centers.flatten()))

    # Set up the optimization problem
    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_circle_overlap}
    ]

    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    # Extract results
    optimized_radii = result.x[:num_circles]
    optimized_centers = result.x[num_circles:].reshape(num_circles, 2)
    sum_radii = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
