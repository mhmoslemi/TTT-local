import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for circle centers and radii
    # Start with equal radii and spread out centers
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii, centers):
        constraints = []

        # Constraint: all circles are inside the unit square
        for i in range(26):
            r = radii[i]
            x, y = centers[i]
            if r > 1 - r:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - r})
            if r > x:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: x - r})
            if r > 1 - x:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (1 - x) - r})
            if r > y:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: y - r})
            if r > 1 - y:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: (1 - y) - r})

        # Constraint: no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                def dist_constraint(radii):
                    r_i = radii[i]
                    r_j = radii[j]
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - r_i - r_j
                constraints.append({'type': 'ineq', 'fun': dist_constraint})

        return constraints

    # Optimization
    res = minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        constraints=constraints(radii, centers),
        method='SLSQP'
    )

    # Extract results
    sum_radii = -res.fun
    return (centers, res.x, sum_radii)
