        [0.1, 0.9], [0.3, 0.9], [0.5, 0.9], [0.7, 0.9], [0.9, 0.9]
    ])

    # Initial guess for radii
    radii = np.full(num_circles, 0.1)

    # Optimization problem
    def objective_func(radii):
        return -np.sum(radii)  # Objective to maximize sum of radii

    def constraint_func(radii):
        # Check if all circles are inside the square
        valid = (radii <= (1 - centers[:, 0])).all() and (radii <= (1 - centers[:, 1])).all()
        if not valid:
            return -np.inf  # Penalize invalid configurations
        # Check if all pairs of circles are non-overlapping
        dist = np.sqrt((centers[:, 0, np.newaxis] - centers[:, 0, np.newaxis])**2 +

        dist -= radii[np.newaxis, :] - radii[:, np.newaxis]
        dist[dist < 0] = 0
        return np.min(dist)

    # Optimization settings
    res = minimize(objective_func, radii, method='SLSQP',

    centers_array = np.array(centers)
    radii_array = res.x
    sum_radii = np.sum(radii_array)

    # Print a message indicating the target is close to achieved
    if sum_radii > 2.63:
        print("Success: Target sum of radii achieved.")

    return (centers_array, radii_array, sum_radii)
