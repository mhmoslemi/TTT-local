import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(26, 2)
    # Initial guess for radii (all 0.1)
    radii = np.full(26, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii, centers):
        constraints = []
        # Constraint: all circles are inside the unit square
        for i in range(26):
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r=r: r - (1 - r - 1e-8)})  # r <= x <= 1 - r
            constraints.append({'type': 'ineq', 'fun': lambda r=r: r - (1 - r - 1e-8)})  # r <= y <= 1 - r

        # Constraint: no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                def dist_con(radii, i=i, j=j):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    return dist - (radii[i] + radii[j])
                constraints.append({'type': 'ineq', 'fun': dist_con})

        return constraints

    res = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        method='SLSQP',
        constraints=constraints(radii, centers),
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract the result
    centers = res.x.reshape(26, 2)  # reshape radius into centers
    radii = np.full(26, 0.1)
    for i in range(26):
        radii[i] = res.x[i]  # assuming radii are at the beginning of the vector?
    sum_radii = np.sum(radii)

    # Ensure all circles are within bounds
    for i in range(26):
        x, y = centers[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            sum_radii = 0.0
            break

    return (centers, radii, sum_radii)
