import numpy as np
from scipy.optimize import minimize
from scipy.spatial import distance

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles) * 0.1  # Initial radii

    def objective(x):
        # Flatten x into centers and radii
        x_flat = x.reshape((num_circles, 2 + 1))
        centers = x_flat[:, :2]
        radii = x_flat[:, 2]
        
        # Objective is to minimize the negative sum of radii
        sum_radii = -np.sum(radii)
        
        # Constraints: circles must be fully inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii), axis=1)
        inside = np.where(inside, 0, 1e6)  # Penalize for being out of bounds

        # Constraints: circles must not overlap
        def pairwise_distance(c1, c2):
            return np.linalg.norm(c1 - c2)
        
        min_dist = np.inf
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                d = pairwise_distance(centers[i], centers[j])
                min_dist = min(min_dist, d - (radii[i] + radii[j]))
        
        # Penalize for overlaps
        overlap_penalty = np.maximum(0, -min_dist) * 1e6

        # Total cost
        total_cost = sum_radii + np.sum(inside) + overlap_penalty
        return total_cost

    # Bounds for x: [0, 1] for centers, [0, 0.5] for radii
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0, 1), (0, 1), (0, 0.5)])
    
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                      bounds=bounds, method='L-BFGS-B', tol=1e-6, options={'maxiter': 1000})

    # Extract solution
    x_opt = result.x
    centers_opt = x_opt.reshape((num_circles, 2))
    radii_opt = x_opt.reshape((num_circles, 3))[:, 2]
    
    # Compute sum of radii
    sum_radii_opt = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii_opt)
