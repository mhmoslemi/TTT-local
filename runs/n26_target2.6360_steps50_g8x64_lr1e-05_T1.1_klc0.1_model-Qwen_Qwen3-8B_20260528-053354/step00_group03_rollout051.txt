import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        x = col / 4 + 0.25
        y = row / 4 + 0.25
        centers[i] = [x, y]
        radii[i] = 0.1

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints: ensure circles are inside the square
    def inside_box(x):
        r = x[-num_circles:]
        return np.all(r <= 0.5) and np.all(0.5 <= x[:-num_circles])

    # Constraints: ensure circles do not overlap
    def no_overlap(x):
        centers = x[:-num_circles].reshape((num_circles, 2))
        radii = x[-num_circles:]
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.all(dists >= radii_sum - 1e-8)

    # Bounds for radii (non-negative, and to keep circles inside square)
    bounds = [(0, 0.5) for _ in range(num_circles)]

    # Initial guess with radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization
    constraints = [
        {'type': 'ineq', 'fun': inside_box},
        {'type': 'ineq', 'fun': no_overlap}
    ]
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-5)

    # Extract results
    centers_final = result.x[:-num_circles].reshape((num_circles, 2))
    radii_final = result.x[-num_circles:]
    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
