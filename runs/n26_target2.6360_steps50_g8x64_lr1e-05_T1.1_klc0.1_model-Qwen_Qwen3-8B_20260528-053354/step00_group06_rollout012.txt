from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.1)
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    def objective(radii):
        return -sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = (i % 5 + 0.5) / 5
            centers[i, 1] = (i // 5 + 0.5) / 5
            centers[i, 0] += np.random.uniform(-0.2, 0.2)
            centers[i, 1] += np.random.uniform(-0.2, 0.2)
        for i in range(26):
            x1, y1 = centers[i]
            r1 = radii[i]
            if x1 - r1 < 0 or x1 + r1 > 1 or y1 - r1 < 0 or y1 + r1 > 1:
                return np.inf
            for j in range(i + 1, 26):
                x2, y2 = centers[j]
                r2 = radii[j]
                dx = x1 - x2
                dy = y1 - y2
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r1 + r2:
                    return np.inf
        return 0

    def optimize():

        return result
