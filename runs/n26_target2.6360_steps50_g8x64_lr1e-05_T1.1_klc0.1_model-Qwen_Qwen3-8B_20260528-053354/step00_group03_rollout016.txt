import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x / 5 + 0.25, y / 5 + 0.25] for y in range(6) for x in range(6)])
    centers = centers[:26]
    radii = np.full(26, 0.05)
    
    def objective(radii_flat):
        return -np.sum(radii_flat)
    
    def constraints(radii_flat):
        radii = radii_flat.reshape(-1)
        centers = np.array([np.random.rand(2) for _ in range(26)])
        # Constraint 1: circles are inside the unit square
        for i in range(26):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf
        # Constraint 2: circles do not overlap
        dists = pdist(centers)
        for d, (i, j) in zip(dists, [(a, b) for a in range(26) for b in range(a+1, 26)]):
            if d < radii[i] + radii[j]:
                return np.inf
        return 0
    
    # Optimization
    result = minimize(objective, radii, constraints=[lambda r: constraints(r)], method='SLSQP')
    
    centers = np.array([np.random.rand(2) for _ in range(26)])
    radii = result.x
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
