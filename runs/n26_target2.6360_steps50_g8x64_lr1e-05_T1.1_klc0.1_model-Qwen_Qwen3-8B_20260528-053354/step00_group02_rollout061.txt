```python
import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: centers on a grid and equal radii
    n_circles = 26
    centers = np.array([(i / (n_circles - 1), j / (n_circles - 1)) for i in range(n_circles) for j in range(n_circles)])
    centers = centers[:n_circles]
    radii = np.full(n_circles, 0.1)

    # Objective function to minimize: negative of sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: Each circle must be inside the square (r_i <= x_i <= 1 - r_i, r_i <= y_i <= 1 - r_i)
    def constraint_inside(radii):
        return np.array([r - (1 - r) for r in radii])

    # Constraint 2: No overlap between any two circles
    def constraint_no_overlap(radii, centers):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=-1))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Combine all constraints
    cons = []
    cons.append({'type': 'ineq', 'fun': lambda r: constraint_inside(r)})
    cons.append({'type': 'ineq', 'fun': lambda r: np.min(constraint_no_overlap(r, centers))})

    # Bounds: radii must be non-negative
    bounds = Bounds(np.zeros(n_circles), np.inf)

    # Optimization
    result = opt.minimize(objective, radii, method='SLSQP', constraints=cons, bounds=bounds)

    # Extract the results
    centers_final = centers
    radii_final = result.x
    sum_radii = -result.fun

    return (centers_final, radii_final, sum_radii)
```
