import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    radii = np.full(25, 0.05)  # Initial radius for all circles
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more circle in the center
    radii = np.append(radii, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        n = len(radii)
        centers = np.array([[x + i * 0.1, y + j * 0.1] for j in range(n) for i in range(n)])
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        valid = (dists >= radii[:, np.newaxis] + radii[np.newaxis, :]).all()
        if not valid:
            return -np.inf
        return np.sum(radii)

    result = minimize(objective, radii, method='SLSQP', constraints=[{'type': 'ineq', 'fun': constraints}])

    centers = np.array([[x + i * 0.1, y + j * 0.1] for j in range(len(result.x)) for i in range(len(result.x))])
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = np.append(result.x[:-1], 0.05)

    return (centers, radii, result.fun)
