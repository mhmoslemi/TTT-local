import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.1)  # Initial radius guess
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more circle at center

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing -sum

    def constraints(radii):
        # Ensure circles are within the unit square
        r = radii.reshape(-1, 1)
        centers = np.vstack([np.array([0.5, 0.5]), np.array([i / 5, j / 5] for i in range(5) for j in range(5))])
        distances = cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if distances[i, j] < r[i] + r[j]:
                    return False
        return True

    # Optimization settings
    bounds = [(0, 1) for _ in range(26)]
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'maxiter': 1000, 'ftol': 1e-9}
    )

    centers = np.vstack([np.array([i / 5, j / 5] for i in range(5) for j in range(5)), [0.5, 0.5]])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
