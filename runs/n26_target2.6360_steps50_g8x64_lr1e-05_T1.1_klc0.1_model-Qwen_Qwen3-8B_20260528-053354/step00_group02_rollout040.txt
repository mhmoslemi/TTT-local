import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (randomly placed)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Initial radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Bounds for radii (non-negative, and circles must fit in unit square)
        bounds = [(0, 1 - np.min(centers, axis=0))]  # Adjust per circle
        # No overlap constraint
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraint = dist - (radii[i] + radii[j])
                if constraint < 0:
                    return True
        return False

    # Optimization
    result = minimize(
        objective,
        radii,
        method='L-BFGS-B',
        bounds=[(0, 1 - np.min(centers, axis=0))] * 26,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
