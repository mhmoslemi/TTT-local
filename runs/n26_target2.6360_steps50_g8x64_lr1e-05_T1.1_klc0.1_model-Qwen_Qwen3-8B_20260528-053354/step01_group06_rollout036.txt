import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    initial_centers = np.array([[i / 6 + 0.1, j / 6 + 0.1] for i in range(6) for j in range(5)])
    initial_radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        centers = np.array([np.array([x, y]) for x, y, r in zip(initial_centers[:, 0], initial_centers[:, 1], radii)])
        # Constraint 1: All circles are inside the unit square
        in_bounds = np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]), axis=0)
        # Constraint 2: No circles overlap
        distances = cdist(centers, centers)
        distances[np.arange(num_circles)[:, None] == np.arange(num_circles)] = np.inf
        overlap = np.any(distances < radii[:, None] + radii[None, :], axis=(0, 1))
        return in_bounds & ~overlap

    bounds = [(0, 0.5) for _ in range(num_circles)]

    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints={
            'type': 'ineq',
            'fun': lambda r: np.sum(constraints(r))
        }
    )

    centers = np.array([np.array([x, y]) for x, y, r in zip(initial_centers[:, 0], initial_centers[:, 1], result.x)])
    radii = result.x
    sum_radii = -result.fun  # Convert back to positive

    # Ensure all constraints are met (for safety)
    if not np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1])):
        raise ValueError("Some circles are outside the unit square.")
    distances = cdist(centers, centers)
    distances[np.arange(num_circles)[:, None] == np.arange(num_circles)] = np.inf
    if np.any(distances < radii[:, None] + radii[None, :]):
        raise ValueError("Some circles overlap.")

    return (centers, radii, sum_radii)
