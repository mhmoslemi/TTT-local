import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 26 equally spaced circles along a grid with some spacing
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.05)
    
    # Initial positions: grid-like arrangement
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i] = [0.1 + col * 0.22, 0.1 + row * 0.22]
    
    # Objective function: minimize the negative sum of radii (maximize sum of radii)
    def obj(radii):
        return -np.sum(radii)
    
    # Constraints: circles must be within the unit square
    def constraint_inside(radii):
        return np.sum((radii <= 0.5) | (radii <= 1 - centers[:, 0]) | (radii <= 1 - centers[:, 1]))
    
    # Constraints: circles must not overlap
    def constraint_overlap(radii):
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0]) ** 2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1]) ** 2))
        return np.min(dists - (radii[:, np.newaxis] + radii[np.newaxis, :])) + 0.001
    
    # Optimization settings
    bounds = [(0, 0.5) for _ in range(26)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Perform the optimization
    result = minimize(obj, radii, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)
    
    # Return the result
    return (centers, result.x, -result.fun)
