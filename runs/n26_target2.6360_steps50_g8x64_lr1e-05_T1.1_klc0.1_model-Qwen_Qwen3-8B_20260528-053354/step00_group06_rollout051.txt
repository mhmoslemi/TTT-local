import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid with some spacing
    # We use a simple grid with 5 rows and 5 columns of centers
    # Start with equal radii and adjust in optimization
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        x = (i % 5) / 4.0 + 0.25
        y = (i // 5) / 4.0 + 0.25
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize

    def constraints(radii, centers):
        # Constraint: each circle is inside the square
        circle_bounds = []
        for i in range(num_circles):
            r = radii[i]
            circle_bounds.append(r - centers[i, 0] + r <= 0)
            circle_bounds.append(r - centers[i, 1] + r <= 0)
            circle_bounds.append(centers[i, 0] + r - 1 + r <= 0)
            circle_bounds.append(centers[i, 1] + r - 1 + r <= 0)
        # Constraint: circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                circle_bounds.append(dist - radii[i] - radii[j] <= 0)
        return circle_bounds

    # Prepare for optimization
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    initial_guess = radii.copy()

    # Run optimization
    result = minimize(
        objective,
        initial_guess,
        args=(centers,),
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    # Extract results
    optimized_radii = result.x
    sum_radii = -result.fun

    # Ensure the centers are still valid after optimization
    for i in range(num_circles):
        x, y = centers[i]
        r = optimized_radii[i]
        if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
            optimized_radii[i] = min(optimized_radii[i], 0.5)
        if optimized_radii[i] < 0:
            optimized_radii[i] = 0.0

    return (centers, optimized_radii, sum_radii)
