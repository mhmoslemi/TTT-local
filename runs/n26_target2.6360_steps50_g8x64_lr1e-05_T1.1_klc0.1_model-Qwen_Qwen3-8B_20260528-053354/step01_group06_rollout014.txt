import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.ones(num_circles) * 0.1

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        constraints = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    def bounds(radii):
        return [(0, 1) for _ in range(num_circles)]

    # Optimization
    result = minimize(
        fun=lambda x: -np.sum(x),
        x0=radii,
        method='SLSQP',
        constraints=[
            {
                'type': 'ineq',
                'fun': lambda x: np.array([np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2) - (x[i] + x[j]) for j in range(i+1, num_circles) for i in range(num_circles)]) if i < j else []
            for i in range(num_circles)
            for j in range(i+1, num_circles)
        ],
        bounds=bounds(radii)
    )

    radii_result = result.x
    centers_result = centers
    sum_radii = np.sum(radii_result)

    # Check if all circles are within the unit square
    for i in range(num_circles):
        if radii_result[i] > centers_result[i, 0] or radii_result[i] > 1 - centers_result[i, 0]:
            radii_result[i] = min(radii_result[i], centers_result[i, 0])
        if radii_result[i] > centers_result[i, 1] or radii_result[i] > 1 - centers_result[i, 1]:
            radii_result[i] = min(radii_result[i], centers_result[i, 1])

    return (centers_result, radii_result, sum_radii)
