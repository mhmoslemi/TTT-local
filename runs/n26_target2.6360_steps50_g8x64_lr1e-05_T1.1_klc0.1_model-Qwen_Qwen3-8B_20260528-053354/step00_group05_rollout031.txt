import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import sys

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: uniform
    radii = np.ones(26) * 0.1

    # Objective function: minimize negative sum of radii (equivalent to maximizing sum)
    def objective(x):
        nonlocal centers, radii
        radii = x[:26]
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraint: all circles must lie within the unit square
    def in_square(x):
        nonlocal centers, radii
        r = x[:26]
        return (np.all(centers[:, 0] + r <= 1) and 
                np.all(centers[:, 0] - r >= 0) and 
                np.all(centers[:, 1] + r <= 1) and 
                np.all(centers[:, 1] - r >= 0))

    # Constraint: no two circles overlap
    def no_overlap(x):
        nonlocal centers, radii
        r = x[:26]
        dists = squareform(pdist(centers))
        return np.all(dists >= r[:, np.newaxis] + r[np.newaxis, :])

    # Optimization constraints
    cons = [
        {'type': 'ineq', 'fun': in_square},
        {'type': 'ineq', 'fun': no_overlap}
    ]

    # Bounds for radii
    bounds = [(0.0, 0.5) for _ in range(26)]

    # Optimization settings
    result = minimize(
        objective,
        x0=radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'ftol': 1e-10, 'maxiter': 500}
    )

    # Extract results
    radii_result = result.x
    sum_radii_result = -result.fun
    centers_result = centers.copy()

    return (centers_result, radii_result, sum_radii_result)
