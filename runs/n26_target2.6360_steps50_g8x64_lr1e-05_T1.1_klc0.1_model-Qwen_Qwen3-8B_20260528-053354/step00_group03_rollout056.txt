import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = (i + 0.5) / np.sqrt(n)
        centers[i, 1] = (i + 0.5) / np.sqrt(n)

    # Objective: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be inside the unit square and not overlap
    def constraint_inside(radii):
        return np.sum((radii <= (1 - centers[:, 0])) * (radii <= (1 - centers[:, 1])) * (radii >= 0))

    def constraint_no_overlap(radii):
        distances = cdist(centers, centers)
        return np.sum((distances >= radii[:, np.newaxis] + radii[np.newaxis, :]) * 1.0)

    # Optimization problem
    bounds = [(0.0, 1.0) for _ in range(n)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(objective, np.ones(n) * 0.1, method='SLSQP', bounds=bounds, constraints=constraints)

    centers_result = centers.copy()
    radii_result = result.x
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
