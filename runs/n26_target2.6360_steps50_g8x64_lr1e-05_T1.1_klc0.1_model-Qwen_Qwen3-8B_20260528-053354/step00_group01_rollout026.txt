```python
import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint

def run_packing():
    # Initial guess: evenly spaced grid points
    def initial_guess():
        centers = []
        radii = []
        for i in range(26):
            x = i // 5 + 0.5 * (i % 5) / 4
            y = (i % 5) / 4 + 0.5 * (i // 5) / 4
            centers.append([x, y])
            radii.append(0.05)
        return np.array(centers), np.array(radii)

    centers, radii = initial_guess()
    n = len(centers)

    def objective(x):
        return -np.sum(x)

    def constraints(x):
        x_flat = x.reshape(n, 2)
        r_flat = x[n:]
        cons = []
        # Box constraints for centers and radii
        for i in range(n):
            x_i, y_i = x_flat[i]
            r_i = r_flat[i]
            cons.append(x_i - r_i >= 0)
            cons.append(1 - x_i - r_i >= 0)
            cons.append(y_i - r_i >= 0)
            cons.append(1 - y_i - r_i >= 0)
        # Non-overlapping constraint
        for i in range(n):
            for j in range(i + 1, n):
                dx = x_flat[i, 0] - x_flat[j, 0]
                dy = x_flat[i, 1] - x_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                cons.append(dist >= r_flat[i] + r_flat[j])
        return np.array(cons)

    # Flatten centers and radii
    x0 = np.concatenate((centers.reshape(n, 2), radii.reshape(n, 1))).ravel()
    bounds = Bounds(
        lb=np.concatenate((np.zeros((n, 2)), np.zeros((n, 1)))),
        ub=np.concatenate((np.ones((n, 2)), np.ones((n, 1))))
    )
    cons = LinearConstraint([], -np.inf, np.inf, shape=x0.shape)
    res = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'ftol': 1e-9, 'maxiter': 1000}
    )
    x = res.x
    centers = x[:n*2].reshape(n, 2)
    radii = x[n*2:].reshape(n)
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
```
