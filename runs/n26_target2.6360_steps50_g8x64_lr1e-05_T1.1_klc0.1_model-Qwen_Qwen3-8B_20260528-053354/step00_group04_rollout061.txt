import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: centers of circles placed in a grid-like pattern with a small radius
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize the sum
    
    def constraints(radii):
        # Constraint 1: All circles are within the unit square
        # Constraint 2: No two circles overlap
        # Convert radii to a numpy array
        radii = np.array(radii)
        centers = np.random.rand(26, 2)
        # Check if all circles are within the square
        for i in range(26):
            if radii[i] > 1 - 2 * radii[i]:
                return np.inf  # Not valid configuration
        # Check if any two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return np.inf  # Overlap detected
        return 0
    
    def bounds(radii):
        return [(0, 1 - 2 * radii[i]) for i in range(26)]
    
    # Optimization: maximize sum of radii subject to constraints
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    
    # Extract result
    centers = np.random.rand(26, 2)
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
