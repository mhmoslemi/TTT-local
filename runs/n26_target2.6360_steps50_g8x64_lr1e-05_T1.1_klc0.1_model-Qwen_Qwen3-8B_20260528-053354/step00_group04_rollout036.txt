import numpy as np
from scipy.optimize import minimize
from scipy.spatial import distance

def run_packing():
    # Initial guess: place 26 circles at evenly spaced grid points
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: start with small values
    radii = np.ones(num_circles) * 0.05

    # Objective function to maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints:
    # 1. Circles are fully inside the unit square
    def constraint_inside(radii):
        return np.array([r <= 1 - r for r in radii])

    # 2. No two circles overlap: distance between centers >= sum of radii
    def constraint_no_overlap(radii):
        dist_matrix = distance.cdist(centers, centers)
        return np.array([dist_matrix[i, j] >= radii[i] + radii[j] for i in range(num_circles) for j in range(i + 1, num_circles)])

    # 3. Radii must be non-negative
    def constraint_non_negative(radii):
        return np.array(radii >= 0)

    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=cons)

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
