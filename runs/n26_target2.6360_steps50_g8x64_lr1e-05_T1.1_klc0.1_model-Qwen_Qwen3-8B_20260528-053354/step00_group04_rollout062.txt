import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    num_circles = 26
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    radii = np.full(num_circles, 0.05)

    def objective(reshape_params):
        params = reshape_params.reshape(-1, 2)
        radii = params[:, 0]
        centers = params[:, 1:]
        sum_radii = np.sum(radii)
        return -sum_radii

    def constraints(reshape_params):
        params = reshape_params.reshape(-1, 2)
        radii = params[:, 0]
        centers = params[:, 1:]
        constraints = []
        for i in range(num_circles):
            # Constraint: circle i is inside the unit square
            for j in [0, 1]:
                constraints.append(radii[i] <= 1 - centers[i, j])
                constraints.append(radii[i] <= centers[i, j])
            # Constraint: no overlap with other circles
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                constraints.append(dist >= radii[i] + radii[j])
        return np.array(constraints)

    # Optimize
    initial_params = np.concatenate([radii, centers.reshape(-1)])
    result = minimize(
        objective,
        initial_params,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        bounds=[(0, 1) for _ in range(num_circles * 3)]
    )

    # Extract result
    optimized_params = result.x
    radii = optimized_params[:num_circles]
    centers = optimized_params[num_circles:].reshape(num_circles, 2)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
