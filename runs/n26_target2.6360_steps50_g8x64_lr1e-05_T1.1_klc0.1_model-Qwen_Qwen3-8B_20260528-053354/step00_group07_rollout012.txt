import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guesses for centers and radii (simple grid-like)
    centers = np.random.rand(26, 2) * 0.8 + 0.1  # Avoid edges
    radii = np.ones(26) * 0.1

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        # x is flattened array [x1, y1, r1, x2, y2, r2, ...]
        x_flat = x.reshape(26, 4)
        centers_flat = x_flat[:, :2]
        radii_flat = x_flat[:, 2]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints: each circle must be inside the unit square
    def square_constraint(x):
        x_flat = x.reshape(26, 4)
        centers_flat = x_flat[:, :2]
        radii_flat = x_flat[:, 2]
        left = centers_flat[:, 0] - radii_flat
        right = 1.0 - centers_flat[:, 0] - radii_flat
        bottom = centers_flat[:, 1] - radii_flat
        top = 1.0 - centers_flat[:, 1] - radii_flat
        return np.concatenate([left, right, bottom, top])

    # Constraints: no two circles overlap
    def overlap_constraint(x):
        x_flat = x.reshape(26, 4)
        centers_flat = x_flat[:, :2]
        radii_flat = x_flat[:, 2]
        dist_sq = np.sum((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2, axis=2)
        radii_sum = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        return dist_sq - radii_sum

    # Initial guess flattened
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Bounds for x and y (each between 0 and 1) and radii (non-negative)
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])  # max radius is 0.5 to fit in square

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': overlap_constraint}
    ]

    # Optimization
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract result
    x_res = result.x
    centers_res = x_res[:26*2].reshape(26, 2)
    radii_res = x_res[26*2:].reshape(26)
    sum_radii_res = np.sum(radii_res)

    return (centers_res, radii_res, sum_radii_res)
