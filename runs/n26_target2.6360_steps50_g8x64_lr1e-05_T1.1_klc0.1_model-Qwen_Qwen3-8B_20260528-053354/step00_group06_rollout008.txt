import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import random

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[i/5 + 0.1, j/5 + 0.1] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.1)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add one more circle in the center
    radii = np.append(radii, 0.1)
    centers = centers[:26]
    radii = radii[:26]

    # Define objective function to minimize negative sum of radii
    def objective(x):
        radii_flat = x[:26]
        centers_flat = x[26:]
        centers = centers_flat.reshape(26, 2)
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Define constraints
    def constraint1(x):
        radii_flat = x[:26]
        centers_flat = x[26:]
        centers = centers_flat.reshape(26, 2)
        for i in range(26):
            r_i = radii_flat[i]
            x_i, y_i = centers[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1
        return 0

    def constraint2(x):
        radii_flat = x[:26]
        centers_flat = x[26:]
        centers = centers_flat.reshape(26, 2)
        dists = pdist(centers)
        radii = radii_flat
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[j - i - 1] < radii[i] + radii[j]:
                    return -1
        return 0

    # Initial guess for the optimization
    initial_guess = np.concatenate((radii, centers.flatten()))

    # Optimization
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        constraints=cons,
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    radii_final = result.x[:26]
    centers_final = result.x[26:].reshape(26, 2)
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
