import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers (evenly spaced grid)
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial guess for radii (all equal)
    radii = np.full(26, 0.1)

    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        return -np.sum(x[26:])

    # Bounds: radii >= 0, and each radius <= 0.5 (max possible radius for a circle in unit square)
    bounds = [(0, 0.5) for _ in range(26)]

    # Constraints: 1) circles are fully inside the unit square
    #               2) no two circles overlap
    constraints = []

    def inside_box(centers, radii):
        return np.all((radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) &
                      (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1]))

    def no_overlap(centers, radii):
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Constraints in terms of variables
    def constraint_inside_box(x):
        return inside_box(centers, x[26:])

    def constraint_no_overlap(x):
        return no_overlap(centers, x[26:])

    constraints.append({'type': 'ineq', 'fun': constraint_inside_box})
    constraints.append({'type': 'ineq', 'fun': constraint_no_overlap})

    # Optimization
    try:
        result = minimize(objective, np.concatenate((centers.flatten(), radii)), bounds=bounds, constraints=constraints)
    except:
        # Fallback: use initial guess if optimization fails
        result = {'x': np.concatenate((centers.flatten(), radii))}

    centers_opt = result['x'][:52].reshape(26, 2)
    radii_opt = result['x'][52:]
    sum_radii = -result['fun']

    return (centers_opt, radii_opt, sum_radii)
