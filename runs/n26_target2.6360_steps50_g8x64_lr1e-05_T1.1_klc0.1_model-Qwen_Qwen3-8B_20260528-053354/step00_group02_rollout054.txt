import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: regular grid with small radii
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.05)
    sum_radii = np.sum(radii)
    bounds = [(0, 1) for _ in range(26 * 2)]
    constraints = []
    radii_bounds = [(0, 1) for _ in range(26)]
    radii_bounds = [(0.0, 1.0) for _ in range(26)]
    radii_bounds = [(0.0, 1.0) for _ in range(26)]

    def objective(x):
        return -np.sum(x[26:])  # Maximize sum of radii

    def constraint1(x):
        # Ensure all circles are fully inside the unit square
        for i in range(26):
            r = x[26 + i]
            x_center = x[2*i]
            y_center = x[2*i + 1]
            if not (r <= x_center <= 1 - r and r <= y_center <= 1 - r):
                return 1.0
        return 0.0

    def constraint2(x):
        # Ensure circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                r_i = x[26 + i]
                r_j = x[26 + j]
                dx = x[2*i] - x[2*j]
                dy = x[2*i + 1] - x[2*j + 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r_i + r_j - 1e-8:
                    return 1.0
        return 0.0

    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization settings
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=constraints,
        constraints=constraints,
        options={'maxiter': 1000, 'ftol': 1e-6, 'eps': 1e-8}
    )

    if result.success:
        centers_opt = result.x[:52].reshape(26, 2)
        radii_opt = result.x[52:]
        sum_radii_opt = -result.fun
        return (centers_opt, radii_opt, sum_radii_opt)
    else:
        warnings.warn("Optimization did not converge")
        return (centers, radii, sum_radii)
