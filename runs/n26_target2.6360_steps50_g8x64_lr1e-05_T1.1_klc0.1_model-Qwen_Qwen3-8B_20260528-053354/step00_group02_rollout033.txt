import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial grid placement for 26 circles
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        x = col / 4.5
        y = row / 4.5
        centers[i] = [x, y]
        radii[i] = 0.1  # Initial small radius

    def objective(radii):
        return -sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure circles are within the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if centers[i][0] - radii[i] < 0 or centers[i][0] + radii[i] > 1:
                return False
            if centers[i][1] - radii[i] < 0 or centers[i][1] + radii[i] > 1:
                return False

        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds():
        return [(0, 0.5) for _ in range(num_circles)]

    # Optimization setup
    res = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds(),
        constraints=({'type': 'ineq', 'fun': constraints}),
        options={'disp': False}
    )

    # Extract results
    optimized_radii = res.x
    optimized_centers = centers.copy()
    sum_radii = sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
