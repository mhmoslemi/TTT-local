import numpy as np
import scipy.optimize as opt
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    centers = np.array([[i / 7, j / 7] for i in range(8) for j in range(8)][:26])
    radii = np.ones(26) * 0.1  # Initial radius estimate
    sum_radii = 2.6

    def objective(radii, centers):
        return -np.sum(radii)  # Objective is to maximize sum of radii

    def constraint_overlap(radii, centers):
        dists = cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < radii[i] + radii[j]:
                    return -1
        return 0

    def constraint_bounds(radii, centers):
        for i in range(26):
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return -1
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return -1
        return 0

    # Optimization with constraints
    bounds = [(0, 1) for _ in range(26)]
    cons = [
        {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_bounds, 'args': (centers,)}
    ]

    result = opt.minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    radii_opt = result.x
    centers = np.array([[i / 7, j / 7] for i in range(8) for j in range(8)][:26])
    sum_radii = np.sum(radii_opt)

    return (centers, radii_opt, sum_radii)
