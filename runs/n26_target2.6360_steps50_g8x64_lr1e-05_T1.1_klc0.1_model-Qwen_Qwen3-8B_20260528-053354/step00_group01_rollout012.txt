import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    x = np.linspace(0.1, 0.9, 5)
    y = np.linspace(0.1, 0.9, 5)
    centers = np.array([[xi, yi] for xi in x for yi in y])
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(5)])
        radii = radii.reshape(-1)
        # Ensure all circles are fully inside the square
        for i in range(num_circles):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        # Ensure no two circles overlap
        dist = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dist[i, j] < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization settings
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-6)

    radii_opt = result.x
    centers_opt = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(5)])
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
