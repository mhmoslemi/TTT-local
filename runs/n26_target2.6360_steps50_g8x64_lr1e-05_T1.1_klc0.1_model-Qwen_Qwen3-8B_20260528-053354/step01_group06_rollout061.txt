import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.25 + 0.125
        centers[i, 1] = row * 0.25 + 0.125
        radii[i] = 0.05

    def objective(radii):
        return -sum(radii)

    def constraints(radii):
        centers_new = np.zeros((26, 2))
        for i in range(26):
            centers_new[i, 0] = centers[i, 0] + random.uniform(-0.05, 0.05)
            centers_new[i, 1] = centers[i, 1] + random.uniform(-0.05, 0.05)
            centers_new[i, 0] = np.clip(centers_new[i, 0], radii[i], 1 - radii[i])
            centers_new[i, 1] = np.clip(centers_new[i, 1], radii[i], 1 - radii[i])
            radii[i] = np.clip(radii[i], 0, 0.5)
        return centers_new

    def objective_func(radii):
        centers_new = np.zeros((26, 2))
        for i in range(26):
            centers_new[i, 0] = centers[i, 0] + random.uniform(-0.05, 0.05)
            centers_new[i, 1] = centers[i, 1] + random.uniform(-0.05, 0.05)
            centers_new[i, 0] = np.clip(centers_new[i, 0], radii[i], 1 - radii[i])
            centers_new[i, 1] = np.clip(centers_new[i, 1], radii[i], 1 - radii[i])
            radii[i] = np.clip(radii[i], 0, 0.5)
        return -sum(radii)

    # Optimize the radii
    result = minimize(objective_func, radii, method='L-BFGS-B')
    radii = result.x
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = random.uniform(radii[i], 1 - radii[i])
        centers[i, 1] = random.uniform(radii[i], 1 - radii[i])

    # Check constraints
    for i in range(26):
        for j in range(i + 1, 26):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < radii[i] + radii[j]:
                radii[i] = min(radii[i], dist - radii[j])
                radii[j] = min(radii[j], dist - radii[i])

    # Final check to ensure all circles are inside the square and non-overlapping
    for i in range(26):
        if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
            radii[i] = min(radii[i], min(centers[i, 0], 1 - centers[i, 0]))
        if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
            radii[i] = min(radii[i], min(centers[i, 1], 1 - centers[i, 1]))

    sum_radii = sum(radii)
    return (centers, radii, sum_radii)
