import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    radii = np.full(36, 0.1)  # Start with small radii
    
    # Filter to 26 circles
    centers = centers[:26]
    radii = radii[:26]
    
    # Define objective function: minimize negative sum of radii (equivalent to maximizing sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Define constraints
    constraints = []
    
    # Constraint 1: All circles must be inside the unit square (r_i <= x_i <= 1 - r_i, same for y_i)
    def inside_box(x, r):
        return 1 - r - x <= 1e-8 and x - r >= -1e-8
    # This constraint is linear and can be handled with bounds, not with a constraint function
    # So we just set bounds on the radii and centers
    
    # Constraint 2: Distance between centers >= sum of radii
    def distance_constraint(c):
        dist = cdist(c[:, np.newaxis, :], c[np.newaxis, :, :], metric='euclidean')
        dist[np.arange(c.shape[0]), np.arange(c.shape[0])] = np.inf
        return dist - np.outer(radii, radii)
    
    # Define bounds for centers (each coordinate is in [0,1])
    bounds = [(0.0, 1.0) for _ in range(2 * 26)]
    
    # Define optimization problem
    def optimize():
        options = {'ftol': 1e-9, 'disp': False}
        result = minimize(objective, radii, method='SLSQP',
                          bounds=bounds,
                          constraints=[{'type': 'ineq', 'fun': lambda r: np.min(distance_constraint(centers))}])
        return result
    
    # Run the optimization
    try:
        result = optimize()
        centers = np.array(centers)
        radii = result.x
        sum_radii = -result.fun
        
        # Final check: ensure all circles are inside the square and no overlaps
        valid = True
        for i in range(26):
            if radii[i] < 0:
                valid = False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                valid = False
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    valid = False
        
        if not valid:
            warnings.warn("Optimization result may not satisfy all constraints.")
            return (np.zeros((26, 2)), np.zeros(26), 0.0)
        
        return (centers, radii, sum_radii)
    
    except Exception as e:
        warnings.warn(f"Optimization failed: {e}")
        return (np.zeros((26, 2)), np.zeros(26), 0.0)
