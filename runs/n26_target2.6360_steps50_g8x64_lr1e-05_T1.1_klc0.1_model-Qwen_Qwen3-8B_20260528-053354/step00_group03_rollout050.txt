import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii (randomly placed circles)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    centers[:, 0] = np.clip(centers[:, 0], 0.1, 0.9)
    centers[:, 1] = np.clip(centers[:, 1], 0.1, 0.9)
    radii = np.random.rand(26) * 0.05  # small initial radii

    def objective(x):
        # x is flattened: [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        x_vals = x[:2*26].reshape(26, 2)
        r_vals = x[2*26:]
        # Objective: minimize negative sum of radii (maximize sum)
        return -np.sum(r_vals)

    def constraints(x):
        x_vals = x[:2*26].reshape(26, 2)
        r_vals = x[2*26:]
        # Constraint: each circle must be inside the unit square
        for i in range(26):
            x_i, y_i = x_vals[i]
            r_i = r_vals[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False
        # Constraint: circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = x_vals[i, 0] - x_vals[j, 0]
                dy = x_vals[i, 1] - x_vals[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r_vals[i] + r_vals[j]:
                    return False
        return True

    # Prepare bounds for optimization
    bounds = []
    for i in range(26):
        bounds.extend([(0.1, 0.9), (0.1, 0.9), (0.0, 0.5)])
    # Initial guess
    x0 = np.concatenate([centers.ravel(), radii])
    # Optimize
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints={"type": "equality", "fun": constraints})
    # Extract results
    centers_opt = result.x[:2*26].reshape(26, 2)
    radii_opt = result.x[2*26:]
    sum_radii = -result.fun
    return (centers_opt, radii_opt, sum_radii)
