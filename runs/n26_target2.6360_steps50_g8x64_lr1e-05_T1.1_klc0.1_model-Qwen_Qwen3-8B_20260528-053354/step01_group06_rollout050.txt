import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.05)
    for i in range(num_circles):
        row, col = divmod(i, 5)
        centers[i] = [0.1 + col * 0.2, 0.1 + row * 0.2]
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraint_overlap(radii):
        centers_r = np.array([np.random.rand(2) for _ in range(num_circles)])
        # To simplify, we use the initial guess for centers instead of random
        centers_r = centers
        dists = pdist(centers_r)
        sum_radii = np.sum(radii)
        for i in range(len(dists)):
            dist = dists[i]
            r1 = radii[i // num_circles]
            r2 = radii[i % num_circles]
            if dist < r1 + r2:
                return dist - r1 - r2
        return 0
    
    def constraint_bound(radii):
        for i in range(num_circles):
            r = radii[i]
            if centers[i, 0] < r or centers[i, 0] > 1 - r or centers[i, 1] < r or centers[i, 1] > 1 - r:
                return r
        return 0
    
    # Optimization setup
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_bound}
    ]
    
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=UserWarning)
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)
    
    final_radii = result.x
    final_centers = centers
    sum_radii = np.sum(final_radii)
    
    return (final_centers, final_radii, sum_radii)
