import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced grid
    n = 26
    num_points = n
    centers = np.zeros((num_points, 2))
    for i in range(num_points):
        centers[i, 0] = (i + 0.5) / np.sqrt(num_points)
        centers[i, 1] = (i + 0.5) / np.sqrt(num_points)

    # Initial radii: equal for all
    radii = np.full(num_points, 0.1)

    # Objective function: minimize the negative sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraint: each circle must be inside the unit square
    def constraint_inside(x, idx):
        r = x[idx]
        x_center = centers[idx, 0]
        y_center = centers[idx, 1]
        return r - (x_center - r)  # Ensure x_center - r >= 0
    def constraint_inside2(x, idx):
        r = x[idx]
        x_center = centers[idx, 0]
        y_center = centers[idx, 1]
        return r - (1 - x_center - r)  # Ensure 1 - x_center - r >= 0
    def constraint_inside3(x, idx):
        r = x[idx]
        x_center = centers[idx, 0]
        y_center = centers[idx, 1]
        return r - (y_center - r)  # Ensure y_center - r >= 0
    def constraint_inside4(x, idx):
        r = x[idx]
        x_center = centers[idx, 0]
        y_center = centers[idx, 1]
        return r - (1 - y_center - r)  # Ensure 1 - y_center - r >= 0

    # Constraint: no two circles overlap
    def constraint_overlap(x, idx1, idx2):
        r1 = x[idx1]
        r2 = x[idx2]
        dx = centers[idx1, 0] - centers[idx2, 0]
        dy = centers[idx1, 1] - centers[idx2, 1]
        dist = math.hypot(dx, dy)
        return dist - (r1 + r2)

    # Build constraints
    constraints = []
    for i in range(num_points):
        constraints.append({'type': 'ineq', 'fun': lambda x, idx=i: constraint_inside(x, idx)})
        constraints.append({'type': 'ineq', 'fun': lambda x, idx=i: constraint_inside2(x, idx)})
        constraints.append({'type': 'ineq', 'fun': lambda x, idx=i: constraint_inside3(x, idx)})
        constraints.append({'type': 'ineq', 'fun': lambda x, idx=i: constraint_inside4(x, idx)})
    for i in range(num_points):
        for j in range(i + 1, num_points):
            constraints.append({'type': 'ineq', 'fun': lambda x, idx1=i, idx2=j: constraint_overlap(x, idx1, idx2)})

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(num_points)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'disp': False, 'maxiter': 1000}
    )

    # Extract results
    final_radii = result.x
    final_centers = centers.copy()
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
