import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    n_circles = 26
    centers = np.random.rand(n_circles, 2)
    radii = np.full(n_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraint_circle_inside(radii, centers):
        # Ensure each circle is fully inside the square
        return np.array([r <= 1 - r for r in radii])

    def constraint_no_overlap(radii, centers):
        # Ensure no two circles overlap
        distances = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return distances - (radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraint_bounds(radii):
        # Ensure all radii are non-negative
        return radii

    # Optimize
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': constraint_circle_inside, 'args': (centers,)},
            {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (radii, centers)},
            {'type': 'ineq', 'fun': constraint_bounds, 'args': (radii,)}
        ],
        bounds=[(0, 1) for _ in range(n_circles)]
    )

    # Extract results
    centers_opt = centers
    radii_opt = result.x
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
