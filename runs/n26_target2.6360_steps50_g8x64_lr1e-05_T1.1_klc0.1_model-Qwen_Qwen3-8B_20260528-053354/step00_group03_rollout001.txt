import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place 26 circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial placement: staggered grid
    for i in range(num_circles):
        row = i // 4
        col = i % 4
        x = col * 0.3
        y = row * 0.3
        centers[i] = [x, y]
        radii[i] = 0.05
    
    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints:
    # 1. Each circle must be inside the unit square
    def inside_square(radii, centers):
        return np.sum((1 - centers[:, 0] - radii) * (centers[:, 0] - radii) * (1 - centers[:, 1] - radii) * (centers[:, 1] - radii))
    
    # 2. Circles must not overlap
    def no_overlap(radii, centers):
        dist_sq = np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2)
        r_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.sum(np.maximum(0, r_sum**2 - dist_sq))
    
    # Define constraints for optimization
    def constraint_inside(radii):
        return inside_square(radii, centers)
    
    def constraint_overlap(radii):
        return no_overlap(radii, centers)
    
    # Create bounds for radii: [0, 0.5]
    bounds = [(0, 0.5) for _ in range(num_circles)]
    
    # Define constraint functions for scipy.optimize
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Perform optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6, options={'maxiter': 1000})
    
    # Update centers based on optimized radii (for display purpose)
    # (Note: This is a simplified approach for visualization. A better method would update centers as well.)
    # We instead use the initial grid centers but scale radii to maximize sum within constraints
    
    # Extract solution
    optimized_radii = result.x
    optimized_sum = np.sum(optimized_radii)
    
    return (centers, optimized_radii, optimized_sum)
