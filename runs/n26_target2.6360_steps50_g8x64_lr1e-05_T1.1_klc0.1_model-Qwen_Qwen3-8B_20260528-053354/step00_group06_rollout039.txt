import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
        radii[i] = 0.05
    
    # Objective function: maximize sum of radii
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraint 1: circles are fully inside the unit square
    def in_square(radii, centers):
        r = radii.reshape(-1, 1)
        x = centers[:, 0]
        y = centers[:, 1]
        return np.concatenate([x - r, 1 - x - r, y - r, 1 - y - r])
    
    # Constraint 2: circles do not overlap
    def no_overlap(radii, centers):
        r = radii.reshape(-1, 1)
        dist = pdist(centers)
        return dist - r.reshape(-1, 1) - r.reshape(1, -1)
    
    # Bounds and constraints
    bounds = [(0.0, 0.5) for _ in range(26)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: in_square(r, centers)},
        {'type': 'ineq', 'fun': lambda r: no_overlap(r, centers)}
    ]
    
    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6
    )
    
    # Extract results
    centers = result.x.reshape(-1, 1)  # Adjusting for center positions
    radii = result.fun * -1
    sum_radii = np.sum(radii)
    
    return (np.c_[centers, np.zeros(26)], radii, sum_radii)
