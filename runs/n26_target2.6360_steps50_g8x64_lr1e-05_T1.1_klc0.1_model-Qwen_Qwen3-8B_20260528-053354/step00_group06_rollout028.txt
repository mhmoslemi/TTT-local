import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = col / 4.5 + 0.05
        y = row / 4.5 + 0.05
        centers[i] = [x, y]
        radii[i] = 0.1

    def objective(radii):
        return -sum(radii)  # We minimize the negative sum to maximize the sum

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x = i % 5 / 4.5 + 0.05
            y = i // 5 / 4.5 + 0.05
            centers[i] = [x, y]
        # Constraint: each circle must fit in the unit square
        bounds = [(0, 0.5) for _ in range(26)]
        # Constraint: distance between centers >= sum of radii
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds_func(radii):
        return [(0, 0.5) for _ in range(26)]

    # Attempt to optimize
    result = minimize(objective, radii, method='SLSQP', bounds=bounds_func, constraints=constraints)
    radii_opt = result.x
    sum_radii = -result.fun

    # Create centers based on optimized radii
    centers_opt = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.5 + 0.05
        y = i // 5 / 4.5 + 0.05
        centers_opt[i] = [x, y]

    return (centers_opt, radii_opt, sum_radii)
