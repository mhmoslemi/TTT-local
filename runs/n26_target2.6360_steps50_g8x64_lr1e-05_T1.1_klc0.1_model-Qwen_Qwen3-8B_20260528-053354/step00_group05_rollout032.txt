import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess for centers and radii (random positions)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles are inside the unit square
    def constraint_inside(radii):
        return np.sum((radii <= 1 - centers[:, 0]) * (radii <= 1 - centers[:, 1]))

    # Constraints: circles do not overlap
    def constraint_overlap(radii):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -1.0
        return 0.0

    # Optimization problem
    bounds = [(0, 0.5) for _ in range(26)]  # radii cannot exceed 0.5
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
    ]

    # Initial guess for radii
    initial_guess = np.random.rand(26) * 0.1

    # Perform optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'maxiter': 1000, 'ftol': 1e-8}
    )

    # Extract results
    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
