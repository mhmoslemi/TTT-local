import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.05)
    x = np.linspace(0.1, 0.9, 5)
    y = np.linspace(0.1, 0.9, 5)
    idx = 0
    for i in range(5):
        for j in range(5):
            if idx < 26:
                centers[idx] = [x[i], y[j]]
                idx += 1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: each circle must be fully inside square
    def constraint_inside(radii):
        return np.sum((radii <= 0.5) * 1.0)

    # Constraints: no two circles overlap
    def constraint_noverlap(radii):
        n = len(radii)
        dist = np.zeros((n, n))
        for i in range(n):
            for j in range(n):
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist[i, j] = np.sqrt(dx**2 + dy**2)
        return np.sum((dist >= radii[i] + radii[j] for i in range(n) for j in range(n) if i != j)) - (n * (n - 1) / 2)

    # Box constraints: radii must be non-negative and <= 0.5
    bounds = [(0, 0.5) for _ in range(26)]

    # Initial radii
    initial_radii = np.full(26, 0.05)

    # Optimization
    result = minimize(objective, initial_radii, bounds=bounds, constraints=[
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_noverlap}
    ])

    # Extract results
    optimized_radii = result.x
    optimized_sum = np.sum(optimized_radii)

    return (centers, optimized_radii, optimized_sum)
