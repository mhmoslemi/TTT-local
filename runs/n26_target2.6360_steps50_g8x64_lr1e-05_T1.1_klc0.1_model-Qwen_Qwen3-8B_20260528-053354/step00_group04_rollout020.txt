import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly distribute 26 circles with small radii
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.01)
    sum_radii = 0.0

    # Objective function to minimize negative sum of radii (maximize sum)
    def objective(radii_flat):
        nonlocal centers
        radii = radii_flat.reshape(26)
        sum_r = np.sum(radii)
        return -sum_r

    # Constraints: 1. All circles are inside the square
    #              2. No overlap between any pair of circles
    def constraint_inside(radii_flat):
        nonlocal centers
        radii = radii_flat.reshape(26)
        for i in range(26):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return -1
        return 0

    def constraint_overlap(radii_flat):
        nonlocal centers
        radii = radii_flat.reshape(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return -1
        return 0

    # Initial radii
    initial_radii = np.array([0.01] * 26)

    # Setup optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0.0, 0.5) for _ in range(26)]
    )

    # Extract results
    centers = np.random.rand(26, 2)  # Overwrite with better initial guess if needed
    radii = result.x
    sum_radii = -result.fun

    # Ensure all circles are inside the square
    for i in range(26):
        x, y = centers[i]
        if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
            # Adjust center to be inside the square
            dx = max(radii[i] - x, 0)
            dy = max(radii[i] - y, 0)
            centers[i] = [x + (dx - dx) / 2, y + (dy - dy) / 2]

    return centers, radii, sum_radii
