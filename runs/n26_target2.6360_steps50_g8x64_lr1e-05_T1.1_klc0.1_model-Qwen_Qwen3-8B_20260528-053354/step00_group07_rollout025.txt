import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    radii = np.ones(26) * 0.1
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.4 + 0.1
        centers[i, 1] = row * 0.4 + 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((len(radii), 2))
        for i in range(len(radii)):
            centers[i, 0] = i % 5 * 0.4 + 0.1
            centers[i, 1] = i // 5 * 0.4 + 0.1
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    bounds = [(0.0, 0.5) for _ in range(26)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r).min()}],
        bounds=bounds,
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    radii_opt = result.x
    centers_opt = np.zeros((26, 2))
    for i in range(26):
        row = i // 5
        col = i % 5
        centers_opt[i, 0] = col * 0.4 + 0.1
        centers_opt[i, 1] = row * 0.4 + 0.1

    sum_radii = np.sum(radii_opt)

    # Optionally refine with more accurate centers
    for _ in range(5):
        dist = np.sqrt(((centers_opt[:, np.newaxis, :] - centers_opt[np.newaxis, :, :]) ** 2).sum(axis=2))
        constraints = dist - (radii_opt[:, np.newaxis] + radii_opt[np.newaxis, :])
        indices = np.argwhere(constraints < 0)
        for i, j in indices:
            if i != j:
                centers_opt[i] = (centers_opt[i] + centers_opt[j]) / 2
                radii_opt[i] = min(radii_opt[i], radii_opt[j] + 0.001)
                radii_opt[j] = min(radii_opt[j], radii_opt[i] + 0.001)

    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
