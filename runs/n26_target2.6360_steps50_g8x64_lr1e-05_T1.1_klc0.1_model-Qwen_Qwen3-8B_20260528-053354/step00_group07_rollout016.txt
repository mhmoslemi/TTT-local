import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: distribute circles evenly in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = (i % 5 + 0.5) / 5
        y = (i // 5 + 0.5) / 5
        centers[i] = [x, y]
    
    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints
    def constraint_distances(radii):
        # Constraint: distance between centers >= sum of radii
        dists = np.zeros(num_circles * num_circles)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                dists[ i*num_circles + j ] = dist - (radii[i] + radii[j])
                dists[ j*num_circles + i ] = dist - (radii[j] + radii[i])
        return dists
    
    def constraint_boundary(radii):
        # Constraint: circle is inside the unit square
        dists = np.zeros(num_circles)
        for i in range(num_circles):
            dists[i] = min(centers[i, 0] - radii[i], 1 - centers[i, 0] - radii[i],
                           centers[i, 1] - radii[i], 1 - centers[i, 1] - radii[i])
        return dists
    
    # Non-negativity constraint
    def constraint_non_negative(radii):
        return radii
    
    # Optimization
    bounds = [(0, 1) for _ in range(num_circles)]
    cons = [
        {'type': 'ineq', 'fun': constraint_distances},
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]
    
    result = minimize(objective, x0=np.ones(num_circles), bounds=bounds, constraints=cons)
    
    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
