import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: assume all the same
    radii = np.full(26, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Constraint: all circles are inside the square
    def bounds(x):
        return [(0, 1 - x[i]) for i in range(26)]  # x[i] is radius for circle i

    # Constraint: no overlap between any two circles
    def non_overlap(x):
        dists = pdist(centers)
        radii_sum = x[:, np.newaxis] + x[np.newaxis, :]
        return dists - np.diag(radii_sum)

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            bounds=[(0, 1) for _ in range(26)],
            constraints=[
                {'type': 'eq', 'fun': non_overlap, 'jac': lambda x: -np.diag(2 * x)},
                {'type': 'ineq', 'fun': lambda x: x - np.diag(2 * x)},  # Ensure radius <= 0.5
            ]
        )

    radii_opt = np.abs(result.x)
    centers_opt = centers.copy()
    sum_radii = np.sum(radii_opt)

    # Post-processing: check if sum_radii is close to target value
    if sum_radii < 2.6:
        # If sum is too low, adjust radii manually to match target
        scaling = 2.6 / sum_radii
        radii_opt *= scaling

    # Ensure all circles are fully inside the square
    radii_opt = np.clip(radii_opt, 0, 0.5)

    return (centers_opt, radii_opt, sum_radii)
