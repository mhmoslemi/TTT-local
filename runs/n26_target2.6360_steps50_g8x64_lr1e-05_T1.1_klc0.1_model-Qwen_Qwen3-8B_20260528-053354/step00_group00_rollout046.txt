import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.1)  # Initial radius

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: all circles must be within the unit square and no overlaps
    def constraints(radii, centers):
        # Ensure each circle is within the unit square
        in_bounds = np.all((radii <= 1 - centers) & (radii <= centers), axis=1)
        in_bounds = np.all(in_bounds)
        
        # Ensure no overlaps between circles
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        overlaps = np.any(dists < radii[:, np.newaxis] + radii[np.newaxis, :])
        
        return in_bounds and not overlaps

    # Optimization bounds: radii >= 0
    bounds = [(0, 1)] * num_circles

    # Define the optimization problem
    def optimization_func(params):
        nonlocal centers, radii
        radii = params
        return -np.sum(radii)

    result = minimize(optimization_func, radii, bounds=bounds, method='SLSQP')

    # Extract the result
    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii_opt = np.sum(radii_opt)

    # Ensure constraints are satisfied
    # (This is a simplified version; real optimization often requires more advanced constraints)

    return (centers_opt, radii_opt, sum_radii_opt)
