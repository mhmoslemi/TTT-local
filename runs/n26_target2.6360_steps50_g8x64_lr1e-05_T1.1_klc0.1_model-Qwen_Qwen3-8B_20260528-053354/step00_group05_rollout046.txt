import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 + 0.5
        y = i // 5 + 0.5
        centers[i] = [x, y]

    radii = np.full(26, 0.1)

    # Objective function to minimize: negative sum of radii to maximize it
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints
    def constraint_center_inside(radii_vec):
        return np.array([r - (1 - (x + r) > 1) * (1 - x - r) - (x - r < 0) * x for x, r in zip(centers, radii_vec)])

    def constraint_no_overlap(radii_vec):
        dists = pdist(centers)
        radii_pairwise = np.array([radii_vec[i] + radii_vec[j] for i, j in enumerate(range(26))])
        return np.array(dists - radii_pairwise)

    # Optimization
    res = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': constraint_center_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}
        ],
        bounds=[(0, 1 - 0.1) for _ in range(26)]  # Initial radius guess
    )

    # Extract results
    centers_final = centers
    radii_final = res.x
    sum_radii_final = np.sum(radii_final)
    return (centers_final, radii_final, sum_radii_final)
