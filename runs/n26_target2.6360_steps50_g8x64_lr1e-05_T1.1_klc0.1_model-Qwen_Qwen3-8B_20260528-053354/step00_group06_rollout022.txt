import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (uniformly spaced)
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]
    radii = np.full(26, 0.1)

    def objective(radii_flattened):
        return -np.sum(radii_flattened)  # Minimize negative sum to maximize sum

    def constraints(radii_flattened):
        radii = radii_flattened.reshape(26)
        # Ensure each circle is inside the unit square
        for i in range(26):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0.0

    # Optimization settings
    bounds = [(0.0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    radii = result.x
    centers = centers  # Keep initial centers; could be optimized further
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
