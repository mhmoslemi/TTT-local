import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.ones(num_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(num_circles):
            r_i = radii[i]
            if not (r_i <= centers[i, 0] <= 1 - r_i and r_i <= centers[i, 1] <= 1 - r_i):
                return False
        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx*dx + dy*dy)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    bounds = [(0, 0.5) for _ in range(num_circles)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=({
            'type': 'ineq',
            'fun': lambda x: 1 - (np.any([not (x[i] <= centers[i, 0] <= 1 - x[i] and x[i] <= centers[i, 1] <= 1 - x[i]) for i in range(num_circles)]) or np.any([np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2) < x[i] + x[j] for i in range(num_circles) for j in range(i + 1, num_circles)]))
        }),
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    if not result.success:
        return centers, radii, np.sum(radii)

    final_radii = result.x
    final_centers = centers.copy()
    sum_radii = np.sum(final_radii)
    return final_centers, final_radii, sum_radii
