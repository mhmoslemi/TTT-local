import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    radii = np.full(36, 0.05)  # Initial radius for each circle
    radii[:26] = 0.05  # We take only the first 26 circles

    # Function to compute the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Function to check constraints
    def constraints(centers, radii):
        # Constraint 1: Circles are fully inside the square
        inside = np.all((radii[:, np.newaxis] <= centers[:, 0]) &
                        (radii[:, np.newaxis] <= 1 - centers[:, 0]) &
                        (radii[:, np.newaxis] <= centers[:, 1]) &
                        (radii[:, np.newaxis] <= 1 - centers[:, 1]), axis=1)
        if not np.all(inside):
            return False
        
        # Constraint 2: No two circles overlap
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    def objective_func(x):
        nonlocal centers, radii
        radii = x
        return -np.sum(radii)

    def constraint_func(x):
        nonlocal centers
        return constraints(centers, x)

    bounds = [(0, 0.5) for _ in range(26)]  # Each radius is between 0 and 0.5
    result = minimize(objective_func, radii[:26], method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_func})
    
    # Extract the result
    optimal_radii = result.x
    sum_radii = np.sum(optimal_radii)
    return (centers[:26], optimal_radii, sum_radii)
