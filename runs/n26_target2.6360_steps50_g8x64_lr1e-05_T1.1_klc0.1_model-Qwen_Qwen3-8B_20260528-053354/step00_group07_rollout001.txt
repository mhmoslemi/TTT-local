import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    centers = np.array([[i/5, j/5] for i in range(5) for j in range(5)])
    centers = centers[:26]
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are fully inside the unit square
        for i in range(26):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        # Ensure no two circles overlap
        dists = cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < radii[i] + radii[j]:
                    return False
        return True

    # Optimization settings
    bounds = [(0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, bounds=bounds, constraints=lambda x: constraints(x))

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
