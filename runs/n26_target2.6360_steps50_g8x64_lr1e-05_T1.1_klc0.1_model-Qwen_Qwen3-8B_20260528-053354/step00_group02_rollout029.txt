import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints
    def constraint_square(radii):
        # Ensure each circle is inside the unit square
        return np.sum((radii - np.min(centers, axis=0)) ** 2 <= radii**2)

    def constraint_overlap(radii):
        # Ensure no two circles overlap
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return np.min(dists - radii[:, np.newaxis] - radii[np.newaxis, :])

    # Optimization problem setup
    bounds = [(0.0, 1.0) for _ in range(num_circles)]  # All radii must be between 0 and 1

    # Initial guess for radii (all equal for starting point)
    initial_radii = np.ones(num_circles) * 0.1

    # Define constraints in the form required by minimize
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_square(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
    ]

    # Run optimization
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract the result
    radii = result.x
    sum_radii = -result.fun  # Since we minimized negative sum
    centers = np.array(centers)

    return (centers, radii, sum_radii)
