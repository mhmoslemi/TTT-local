import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = 0.2 + col * 0.25
        centers[i, 1] = 0.2 + row * 0.25
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i, 0] = 0.2 + (i // 5) * 0.25
            centers[i, 1] = 0.2 + (i % 5) * 0.25
            centers[i, 0] += radii[i] * 2.0
            centers[i, 1] += radii[i] * 2.0

        # Ensure circles are within the unit square
        for i in range(num_circles):
            if centers[i, 0] + radii[i] > 1.0 or centers[i, 0] - radii[i] < 0.0:
                return False
            if centers[i, 1] + radii[i] > 1.0 or centers[i, 1] - radii[i] < 0.0:
                return False

        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    result = minimize(objective, radii, method='L-BFGS-B', bounds=[(0, 1.0) for _ in range(num_circles)])
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)

    # Final positioning
    centers_opt = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers_opt[i, 0] = 0.2 + col * 0.25
        centers_opt[i, 1] = 0.2 + row * 0.25
        centers_opt[i, 0] += radii_opt[i] * 2.0
        centers_opt[i, 1] += radii_opt[i] * 2.0

    # Final validation
    for i in range(num_circles):
        if centers_opt[i, 0] + radii_opt[i] > 1.0 or centers_opt[i, 0] - radii_opt[i] < 0.0:
            centers_opt[i, 0] = np.clip(centers_opt[i, 0], radii_opt[i], 1.0 - radii_opt[i])
        if centers_opt[i, 1] + radii_opt[i] > 1.0 or centers_opt[i, 1] - radii_opt[i] < 0.0:
            centers_opt[i, 1] = np.clip(centers_opt[i, 1], radii_opt[i], 1.0 - radii_opt[i])

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = centers_opt[i, 0] - centers_opt[j, 0]
            dy = centers_opt[i, 1] - centers_opt[j, 1]
            dist = math.hypot(dx, dy)
            if dist < radii_opt[i] + radii_opt[j]:
                centers_opt[i, 0] += 0.01
                centers_opt[i, 1] += 0.01
                centers_opt[j, 0] -= 0.01
                centers_opt[j, 1] -= 0.01

    return (centers_opt, radii_opt, sum_radii)
