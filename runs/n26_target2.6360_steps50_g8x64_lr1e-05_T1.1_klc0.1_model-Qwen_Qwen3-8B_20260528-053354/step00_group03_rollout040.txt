import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: centers arranged in a grid, radii small
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.05)
    centers[::2, 0] = np.linspace(0.1, 0.9, 13)
    centers[1::2, 0] = np.linspace(0.2, 0.8, 13)
    centers[::2, 1] = np.linspace(0.1, 0.9, 13)
    centers[1::2, 1] = np.linspace(0.2, 0.8, 13)

    def objective(radii_vec):
        return -np.sum(radii_vec)  # We maximize sum by minimizing negative sum

    def constraints(radii_vec):
        centers = update_centers(radii_vec)
        # Check boundaries
        for i in range(26):
            x, y = centers[i]
            r = radii_vec[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return False
        # Check pairwise distances
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_vec[i] + radii_vec[j]:
                    return False
        return True

    def update_centers(radii_vec):
        # Update centers based on current radii
        # This function simulates a simple heuristic to shift centers if they are too close
        centers = np.zeros((26, 2))
        centers[::2, 0] = np.linspace(0.1, 0.9, 13)
        centers[1::2, 0] = np.linspace(0.2, 0.8, 13)
        centers[::2, 1] = np.linspace(0.1, 0.9, 13)
        centers[1::2, 1] = np.linspace(0.2, 0.8, 13)
        for i in range(26):
            x, y = centers[i]
            r = radii_vec[i]
            centers[i, 0] = np.clip(x, r, 1 - r)
            centers[i, 1] = np.clip(y, r, 1 - r)
        return centers

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=objective,
            x0=radii,
            method='L-BFGS-B',
            bounds=[(0, 1) for _ in range(26)],
            constraints={'type': 'custom', 'fun': constraints},
            options={'ftol': 1e-5, 'maxiter': 1000}
        )

    radii_opt = result.x
    centers_opt = update_centers(radii_opt)
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
