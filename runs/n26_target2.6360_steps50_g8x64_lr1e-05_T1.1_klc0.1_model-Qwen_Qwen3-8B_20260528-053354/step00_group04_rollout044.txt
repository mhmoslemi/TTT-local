import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.random.rand(26, 2)  # Placeholder for centers
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(26):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        return True

    # Initial guess: small random radii
    initial_radii = np.random.rand(26) * 0.05

    # Optimizer settings
    result = minimize(objective, initial_radii, method='SLSQP', 
                      constraints={"type": "eq", "fun": lambda x: constraints(x)},
                      bounds=[(0, 1)] * 26,
                      options={'maxiter': 100, 'ftol': 1e-9, 'disp': False})

    # If optimization failed, return empty
    if not result.success:
        return np.array([]), np.array([]), 0.0

    # Generate centers with a simple packing strategy (randomly with spacing)
    centers = np.random.rand(26, 2)
    radii = result.x
    sum_radii = np.sum(radii)

    # Return tuple (centers, radii, sum_radii)
    return centers, radii, sum_radii
