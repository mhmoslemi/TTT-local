import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import warnings

def run_packing():
    # Initial guess: evenly spaced grid
    n_circles = 26
    centers = np.random.rand(n_circles, 2)
    radii = np.random.rand(n_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.random.rand(n_circles, 2)
        r = radii.reshape(-1, 1)
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return dist - (r + r[np.newaxis, :])

    def bounds(radii):
        return [(0, 1) for _ in radii]

    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)
    radii = res.x
    centers = np.random.rand(n_circles, 2)
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
