import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        x = i // 5 + 0.1
        y = i % 5 + 0.1
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial small radius

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x, y = centers[i]
            x += radii[i]
            y += radii[i]
            centers[i] = [x, y]

        # Constraint: circles must be inside the square
        for i in range(num_circles):
            if centers[i][0] + radii[i] > 1 or centers[i][0] - radii[i] < 0:
                return np.inf
            if centers[i][1] + radii[i] > 1 or centers[i][1] - radii[i] < 0:
                return np.inf

        # Constraint: no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j]:
                    return np.inf

        return 0

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, method='SLSQP', constraints={'type': 'eq', 'fun': constraints})

    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x, y = centers[i]
        x += result.x[i]
        y += result.x[i]
        centers[i] = [x, y]

    sum_radii = result.fun
    return (centers, result.x, sum_radii)
