import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid points with small radii
    centers = np.array([[x / 5, y / 5] for x in range(5) for y in range(5)])
    radii = np.full(25, 0.02)
    centers = np.vstack((centers, [0.5, 0.5]))
    radii = np.append(radii, 0.02)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii
    
    def constraints(radii):
        centers = np.array([[-1, -1]] * 26)  # Placeholder for centers
        # Update centers based on radii? This is a simplified approach
        # In practice, centers would be optimized along with radii
        # Here for demonstration, we just return a placeholder
        return []
    
    def bounds(radii):
        return [(0, 1) for _ in range(26)]
    
    # Optimization setup
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        options={'maxiter': 100, 'ftol': 1e-9}
    )
    
    # Extract results
    centers = np.array([[x, y] for x in range(5) for y in range(5)])
    centers = np.vstack((centers, [0.5, 0.5]))
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
