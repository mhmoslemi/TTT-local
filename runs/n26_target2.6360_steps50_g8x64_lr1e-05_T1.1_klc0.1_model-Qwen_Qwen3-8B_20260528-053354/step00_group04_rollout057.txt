import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radii: all set to 0.1
    radii = np.full(26, 0.1)

    def objective(x):
        # x is a flat array: [r0, r1, ..., r25]
        # Reshape into radii
        r = x.reshape((26,))
        # Objective function: maximize sum of radii, so minimize negative sum
        return -np.sum(r)

    def constraints(x):
        r = x.reshape((26,))
        # Constraint 1: distance between centers >= sum of radii
        distances = pdist(centers)
        sum_radii = np.add.outer(r, r)
        constraint1 = distances - sum_radii
        constraint1 = np.where(constraint1 < -1e-8, -1e8, constraint1)  # Enforce distance >= sum of radii
        # Constraint 2: circles inside the unit square
        constraint2 = np.minimum(r, centers[:, 0] - r) + np.minimum(r, 1 - centers[:, 0] - r)
        constraint2 = np.minimum(constraint2, np.minimum(r, centers[:, 1] - r) + np.minimum(r, 1 - centers[:, 1] - r))
        # Combine constraints
        return np.concatenate((constraint1, constraint2))

    # Optimization
    bounds = [(0, 1)] * 26
    result = minimize(objective, x0=radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Extract results
    r = result.x
    centers = np.array(centers)
    sum_radii = np.sum(r)

    return (centers, r, sum_radii)
