import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        x = col / 4.0 + 0.25
        y = row / 4.0 + 0.25
        centers[i] = [x, y]
        radii[i] = 0.1

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: Circles are fully inside the square
        for i in range(num_circles):
            r = radii[i]
            if not (r <= centers[i][0] <= 1 - r and r <= centers[i][1] <= 1 - r):
                return np.inf  # Violate constraint

        # Constraint 2: No two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                r1, r2 = radii[i], radii[j]
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < r1 + r2:
                    return np.inf  # Violate constraint

        return 0.0

    # Define variables
    bounds = [(0.0, 0.5) for _ in range(num_circles * 2)]
    x0 = np.concatenate((centers.flatten(), radii))

    # Optimization
    result = minimize(
        lambda x: objective(x[num_circles * 2:], x[:num_circles * 2].reshape((num_circles, 2))),
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=lambda x: constraints(x[num_circles * 2:], x[:num_circles * 2].reshape((num_circles, 2))),
        options={'disp': False}
    )

    centers_opt = result.x[:num_circles * 2].reshape((num_circles, 2))
    radii_opt = result.x[num_circles * 2:]
    sum_radii = -objective(radii_opt, centers_opt)

    return (centers_opt, radii_opt, sum_radii)
