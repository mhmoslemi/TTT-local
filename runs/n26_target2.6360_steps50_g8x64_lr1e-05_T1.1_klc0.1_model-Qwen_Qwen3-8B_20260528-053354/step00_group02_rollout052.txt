import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: spread circles evenly in the square with small radii
    centers = np.random.rand(26, 2) * 0.9 + 0.05
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize sum

    def constraints(radii):
        # Constraint: all circles are inside the unit square
        r = radii
        for i in range(26):
            if r[i] > 1 - 2 * r[i]:
                return False
        # Constraint: no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0.0, 1.0 - 2 * r) for r in radii]
    )

    # Return the result
    return (centers, result.x, -result.fun)
