import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess for centers and radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles) * 0.1  # Small initial radii

    def objective(x):
        # x is a flat array: centers followed by radii
        centers_flat = x[:num_circles*2].reshape(num_circles, 2)
        radii_flat = x[num_circles*2:].reshape(num_circles)
        return -np.sum(radii_flat)  # We want to maximize the sum of radii

    def constraints(x):
        centers_flat = x[:num_circles*2].reshape(num_circles, 2)
        radii_flat = x[num_circles*2:].reshape(num_circles)
        constraints = []

        # Constraint: circles are fully inside the unit square
        for i in range(num_circles):
            r_i = radii_flat[i]
            x_i, y_i = centers_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf

        # Constraint: no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j:
                    return np.inf

        return 0

    # Set up optimization problem
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])
        bounds.extend([(0.0, 0.5)])  # Max radius is 0.5 to fit in unit square

    # Initial guess flattened
    x0 = np.concatenate([centers.flatten(), radii.flatten()])

    # Minimize the negative sum (equivalent to maximizing sum)
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-3)

    # Extract results
    centers_result = result.x[:num_circles*2].reshape(num_circles, 2)
    radii_result = result.x[num_circles*2:].reshape(num_circles)
    sum_radii_result = -result.fun  # Since we minimized the negative sum

    return (centers_result, radii_result, sum_radii_result)
