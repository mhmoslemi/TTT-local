import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: 26 circles arranged in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.02)
    
    # Initial grid layout
    for i in range(num_circles):
        x = i % 5 / 4.0 + 0.1
        y = i // 5 / 4.0 + 0.1
        centers[i, 0] = x
        centers[i, 1] = y

    # Define objective function
    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii
    
    # Define constraints
    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x = i % 5 / 4.0 + 0.1
            y = i // 5 / 4.0 + 0.1
            centers[i, 0] = x
            centers[i, 1] = y
        
        # Constraint 1: circles are within the unit square
        for i in range(num_circles):
            r = radii[i]
            if r > 1 - 0.1 or r < 0:
                return False
        
        # Constraint 2: circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        
        return True
    
    # Optimization
    try:
        result = minimize(objective, radii, method='SLSQP', bounds=[(0, 1 - 0.1)]*num_circles,
                          constraints=[{'type': 'ineq', 'fun': lambda x: 1 - 0.1 - x[i]} for i in range(num_circles)])
        radii = result.x
        sum_radii = -result.fun
        
        # Final positions with optimized radii
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x = i % 5 / 4.0 + 0.1
            y = i // 5 / 4.0 + 0.1
            centers[i, 0] = x
            centers[i, 1] = y
        
        return (centers, radii, sum_radii)
    
    except:
        # Fallback: return initial layout if optimization fails
        return (centers, radii, np.sum(radii))
