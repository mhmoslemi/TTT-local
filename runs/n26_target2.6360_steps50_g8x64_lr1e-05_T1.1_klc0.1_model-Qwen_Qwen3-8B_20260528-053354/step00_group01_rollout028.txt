import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    radii = np.full(25, 0.1)  # Adjust if necessary
    
    # Define objective function to maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Define constraints
    def constraint_center_inside(radii):
        return np.all(radii <= 0.5)  # since max radius is 0.5 to fit in unit square
    
    def constraint_no_overlap(radii):
        # Function to check if any two circles overlap
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Define bounds for radii
    bounds = [(0.0, 0.5) for _ in range(26)]
    
    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraint_center_inside},
                     {'type': 'ineq', 'fun': constraint_no_overlap}]
    )
    
    # Update centers if needed (just for demonstration, but the optimization should already account for it)
    # In reality, the optimization process should adjust centers as well if needed
    # But for simplicity, we keep the initial centers and only optimize radii
    
    sum_radii = result.fun * -1
    return (centers, result.x, sum_radii)
