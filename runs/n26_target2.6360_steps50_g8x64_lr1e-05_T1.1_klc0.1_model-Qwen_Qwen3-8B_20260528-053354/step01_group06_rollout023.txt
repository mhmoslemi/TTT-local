import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = (i % 5 + 0.5) / 5.0
            centers[i, 1] = (i // 5 + 0.5) / 5.0
            centers[i, 0] += radii[i]
            centers[i, 1] += radii[i]
            centers[i, 0] = np.clip(centers[i, 0], 0, 1 - radii[i])
            centers[i, 1] = np.clip(centers[i, 1], 0, 1 - radii[i])
        # Ensure circles are within bounds
        for i in range(26):
            dist = np.sqrt((centers[i, 0] - centers[0, 0])**2 + (centers[i, 1] - centers[0, 1])**2)
            if dist < radii[i] + radii[0]:
                return 1.0
        for i in range(1, 26):
            for j in range(i):
                dist = np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2)
                if dist < radii[i] + radii[j]:
                    return 1.0
        return 0.0

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(26)]
    res = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    radii = res.x
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5.0 + radii[i]
        centers[i, 1] = (i // 5 + 0.5) / 5.0 + radii[i]
        centers[i, 0] = np.clip(centers[i, 0], 0, 1 - radii[i])
        centers[i, 1] = np.clip(centers[i, 1], 0, 1 - radii[i])

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
