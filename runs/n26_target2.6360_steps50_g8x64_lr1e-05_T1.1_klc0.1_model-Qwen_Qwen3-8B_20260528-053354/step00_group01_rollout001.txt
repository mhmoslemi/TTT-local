import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles uniformly in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    side = 1.0
    # Approximate spacing based on a grid of 5x5 (25 circles) with one extra circle
    spacing = 1.0 / 5.0
    for i in range(26):
        x = i % 5 * spacing
        y = i // 5 * spacing
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial small radius for optimization

    # Define objective function to maximize sum of radii (minimize negative sum)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Define constraint: circle must be inside the unit square
    def inside_constraint(radii_vec, centers):
        idx = np.argmin(radii_vec)
        x, y = centers[idx]
        r = radii_vec[idx]
        return (x - r) >= 0 and (y - r) >= 0 and (x + r) <= 1 and (y + r) <= 1

    # Define constraint: circles cannot overlap
    def no_overlap_constraint(radii_vec, centers):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_vec[i] + radii_vec[j] - 1e-8:
                    return False
        return True

    # Optimization bounds (radius can't be negative and needs to be small to fit)
    bounds = [(0, 0.2) for _ in range(26)]

    # Initial guess
    initial_guess = copy.deepcopy(radii)

    # Setup constraints
    constraints = []
    # Inside square constraint
    constraints.append({'type': 'ineq', 'fun': lambda x: inside_constraint(x, centers)})
    # No overlap constraint
    constraints.append({'type': 'ineq', 'fun': lambda x: no_overlap_constraint(x, centers)})

    # Perform optimization
    result = minimize(objective, initial_guess, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-8)

    # Extract final result
    final_radii = result.x
    final_centers = centers
    final_sum = np.sum(final_radii)

    return (final_centers, final_radii, final_sum)
