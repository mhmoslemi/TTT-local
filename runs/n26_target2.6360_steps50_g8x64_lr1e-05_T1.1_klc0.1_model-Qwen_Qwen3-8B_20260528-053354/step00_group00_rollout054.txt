import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.random.rand(26, 2)
    radii = np.array([0.025] * 26)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        # Bounds for radii
        bounds = [(0, 1 - np.min(centers[:, 0]) - np.min(centers[:, 1])) for _ in radii]
        # Non-overlapping constraint
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        return True
    
    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1 - np.min(centers[:, 0]) - np.min(centers[:, 1])) for _ in range(26)],
        constraints=[{'type': 'ineq', 'fun': lambda r: np.sqrt((centers[0, 0] - centers[1, 0])**2 + (centers[0, 1] - centers[1, 1])**2) - (r[0] + r[1])}]
    )
    
    # Final radii
    final_radii = result.x
    sum_radii = np.sum(final_radii)
    
    # Final centers (unchanged in this simple approach, could be optimized)
    final_centers = centers
    
    return (final_centers, final_radii, sum_radii)
