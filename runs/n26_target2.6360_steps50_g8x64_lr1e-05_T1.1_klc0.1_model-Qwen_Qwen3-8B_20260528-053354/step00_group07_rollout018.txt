import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i // 7 + 0.5
        y = i % 7 + 0.5
        centers[i] = [x, y]

    # Initial radii: all the same
    radii = np.ones(num_circles) * 0.1

    # Define constraints for optimization
    def objective(radii):
        return -np.sum(radii)

    # Constraint: all circles are within the unit square
    def constraint_position(radii):
        x, y = centers.T
        return np.concatenate([
            radii - (1 - x) + 1e-6,
            radii - (1 - y) + 1e-6
        ])

    # Constraint: no overlapping between circles
    def constraint_overlap(radii):
        dists = np.zeros(num_circles * (num_circles - 1) // 2)
        k = 0
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[k] = dist - (radii[i] + radii[j])
                k += 1
        return dists

    # Objective function to minimize (negative of sum of radii)
    def objective_func(x):
        return -np.sum(x)

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_position},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds for radii
    bounds = [(0, 1) for _ in range(num_circles)]

    # Use a local optimization method for simplicity
    result = minimize(objective_func, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
