import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: place 26 circles equally spaced in a grid
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles) * 0.05
    sum_radii = np.sum(radii)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii, centers):
        # All circles must be inside the unit square
        inside = np.all((radii <= 1 - centers) & (radii <= centers), axis=1)
        # No two circles overlap
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        overlap = np.triu(dist < radii[:, np.newaxis] + radii[np.newaxis, :], k=1)
        return np.concatenate([inside, overlap])

    def optimize():
        # Initial guess: perturb centers slightly to avoid overlap
        centers = np.random.rand(num_circles, 2)
        radii = np.random.rand(num_circles) * 0.1

        # Bounds for radii: non-negative, and must fit within the unit square
        bounds = Bounds(0, 1 - np.min(centers, axis=0) - np.max(centers, axis=0))

        # Optimization problem
        result = opt.minimize(
            lambda x: -np.sum(x[:num_circles]), 
            x0=np.concatenate([radii, centers.flatten()]),
            bounds=bounds,
            constraints=[
                opt.Constraints(
                    fun=lambda x: np.concatenate([
                        np.all((x[:num_circles] <= 1 - x[num_circles:]).reshape(num_circles, 2) & (x[:num_circles] <= x[num_circles:]), axis=1),
                        np.triu(np.sqrt(np.sum(
                            (x[num_circles:].reshape(num_circles, 2)[:, np.newaxis, :] - x[num_circles:].reshape(num_circles, 2)[np.newaxis, :, :]**2), axis=2)) 
                            < x[:num_circles][:, np.newaxis] + x[:num_circles][np.newaxis, :], k=1), axis=None)
                    ]),
                    type='ineq'
                )
            ],
            method='SLSQP'
        )

        # Extract results
        optimized_radii = result.x[:num_circles]
        optimized_centers = result.x[num_circles:].reshape(num_circles, 2)
        final_sum = np.sum(optimized_radii)

        # Check if all circles are inside the unit square
        inside = np.all((optimized_radii <= 1 - optimized_centers) & (optimized_radii <= optimized_centers), axis=1)

        # Check if no two circles overlap
        dist = np.sqrt(np.sum((optimized_centers[:, np.newaxis, :] - optimized_centers[np.newaxis, :, :])**2, axis=2))
        overlap = np.triu(dist < optimized_radii[:, np.newaxis] + optimized_radii[np.newaxis, :], k=1)

        if np.any(~inside) or np.any(overlap):
            # Fallback to initial guess
            optimized_radii = radii
            optimized_centers = centers
            final_sum = sum_radii
        return optimized_centers, optimized_radii, final_sum

    centers, radii, sum_radii = optimize()
    return (centers, radii, sum_radii)
