import numpy as np
from scipy.optimize import minimize
import time

def run_packing():
    num_circles = 26
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((num_circles, 2))
    indices = np.arange(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (indices[i] % 5 + 0.5) / 5
        centers[i, 1] = (indices[i] // 5 + 0.5) / 5

    # Initial radii: uniform
    radii = np.full(num_circles, 1 / 10)

    def objective_func(radii):
        return -np.sum(radii)

    def constraint_radius(radii, i):
        return radii[i] - 1e-8  # Ensure non-negative radii

    def constraint_position(radii, i):
        x_center, y_center = centers[i]
        return radii[i] - (1 - x_center - radii[i])  # radii[i] <= 1 - x_center - radii[i] => x_center + 2*radii[i] <= 1

    def constraint_position2(radii, i):
        x_center, y_center = centers[i]
        return radii[i] - (1 - y_center - radii[i])  # radii[i] <= 1 - y_center - radii[i] => y_center + 2*radii[i] <= 1

    def constraint_overlap(radii):
        total = 0.0
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                total += max(0, dist - (radii[i] + radii[j]))
        return total

    # Constraints
    constraints = []

    # Non-negative radii
    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_radius(x, i)})

    # Position constraint for x-coordinate
    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_position(x, i)})

    # Position constraint for y-coordinate
    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_position2(x, i)})

    # Overlap constraint
    constraints.append({'type': 'ineq', 'fun': lambda x: constraint_overlap(x)})

    # Initial guess
    initial_guess = radii.copy()

    # Optimization
    result = minimize(
        objective_func,
        initial_guess,
        method='SLSQP',
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000, 'disp': False}
    )

    if not result.success:
        print("Optimization failed.")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    radii = result.x
    centers = np.zeros((num_circles, 2))
    indices = np.arange(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (indices[i] % 5 + 0.5) / 5
        centers[i, 1] = (indices[i] // 5 + 0.5) / 5

    sum_radii = np.sum(radii)

    # Validate constraints
    for i in range(num_circles):
        x_center, y_center = centers[i]
        if radii[i] > (1 - x_center - radii[i]) or radii[i] > (1 - y_center - radii[i]):
            radii[i] = min(radii[i], 1 - x_center - radii[i], 1 - y_center - radii[i])

    for i in range(num_circles):
        x_center, y_center = centers[i]
        for j in range(i + 1, num_circles):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < (radii[i] + radii[j]):
                radii[i] = (dist - radii[j]) / 2
                radii[j] = (dist - radii[i]) / 2

    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
