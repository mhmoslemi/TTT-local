import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid of 26 circles
    centers = np.array([[i/5.0, j/5.0] for i in range(5) for j in range(6)])
    radii = np.full(26, 0.07)
    
    def objective(radii, centers):
        return -np.sum(radii)
    
    def constraints(radii, centers):
        # Constraint: circles must be inside the unit square
        # x - r >= 0, 1 - x - r >= 0, same for y
        r = np.atleast_1d(radii)
        constraints = []
        for i in range(len(r)):
            constraints.append((centers[i, 0] - r[i]) >= 0)
            constraints.append((1 - centers[i, 0] - r[i]) >= 0)
            constraints.append((centers[i, 1] - r[i]) >= 0)
            constraints.append((1 - centers[i, 1] - r[i]) >= 0)
        
        # Constraint: circles cannot overlap
        for i in range(len(r)):
            for j in range(i + 1, len(r)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append((dist - r[i] - r[j]) >= 0)
        
        return constraints
    
    def objective_func(x):
        return -np.sum(x)
    
    def constraint_func(x):
        r = np.atleast_1d(x)
        constraints = []
        for i in range(len(r)):
            constraints.append(centers[i, 0] - r[i] >= 0)
            constraints.append(1 - centers[i, 0] - r[i] >= 0)
            constraints.append(centers[i, 1] - r[i] >= 0)
            constraints.append(1 - centers[i, 1] - r[i] >= 0)
        
        for i in range(len(r)):
            for j in range(i + 1, len(r)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - r[i] - r[j] >= 0)
        
        return constraints
    
    # Set up optimization problem
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=objective_func,
            x0=radii,
            constraints={"type": "ineq", "fun": constraint_func},
            method="SLSQP",
            bounds=[(0, 1) for _ in range(26)]
        )
    
    if result.success:
        optimized_radii = result.x
        sum_radii = np.sum(optimized_radii)
        return (centers, optimized_radii, sum_radii)
    else:
        return (centers, radii, np.sum(radii))
