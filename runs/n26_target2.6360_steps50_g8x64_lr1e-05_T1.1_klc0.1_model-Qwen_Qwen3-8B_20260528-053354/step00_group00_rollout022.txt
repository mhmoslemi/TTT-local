import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: equally spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i + 1) / (num_circles + 1)
        centers[i, 1] = 0.5
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are inside the unit square
        for r in radii:
            if r < 0:
                return False
        # Ensure circles are fully inside the unit square
        for i in range(num_circles):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Ensure no circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    bounds = [(0, 0.5) for _ in range(num_circles)]
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    # Final result
    final_radii = result.x
    final_centers = copy.deepcopy(centers)
    final_sum = np.sum(final_radii)

    return (final_centers, final_radii, final_sum)
