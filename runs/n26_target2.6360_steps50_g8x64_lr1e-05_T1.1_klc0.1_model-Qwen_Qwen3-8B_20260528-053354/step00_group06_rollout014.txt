import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initialize centers with random positions
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    # Objective function: maximize the sum of radii
    def objective(x):
        return -x[26:]  # Minimize negative sum of radii

    # Constraints: 1. Circles must be inside the square
    #              2. Circles must not overlap
    def constraint_inside(x):
        r = x[26:]
        return np.all(r <= x[:26, 0] - r) & np.all(r <= 1 - x[:26, 0] - r) & np.all(r <= x[:26, 1] - r) & np.all(r <= 1 - x[:26, 1] - r)

    def constraint_overlap(x):
        r = x[26:]
        dists = np.sqrt(np.sum((x[:26, np.newaxis, :] - x[:26, :, np.newaxis]) ** 2, axis=2))
        return np.all(dists >= r[np.newaxis, :] + r[:, np.newaxis])

    # Initial guess: flatten centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Constraints in optimization format
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds: x must be in [0,1], radii must be non-negative
    bounds = [(0, 1) for _ in range(52)] + [(0, np.inf) for _ in range(26)]

    # Perform optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract result
    x_opt = result.x
    centers_opt = x_opt[:52].reshape(26, 2)
    radii_opt = x_opt[52:]
    sum_radii = -result.fun

    # Return the result
    return (centers_opt, radii_opt, sum_radii)
