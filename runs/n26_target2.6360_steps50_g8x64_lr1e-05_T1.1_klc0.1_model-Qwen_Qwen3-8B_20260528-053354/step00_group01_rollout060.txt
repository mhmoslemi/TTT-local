import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = 0.2 + col * 0.4
        centers[i, 1] = 0.2 + row * 0.4

    # Initial radii: all equal
    radii = np.ones(26) * 0.1

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints: each circle is inside the unit square
    def square_constraint(x, i):
        r_i = x[i]
        return (x[i * 2] - r_i >= 0) and (x[i * 2 + 1] - r_i >= 0) and (1 - x[i * 2] - r_i >= 0) and (1 - x[i * 2 + 1] - r_i >= 0)

    # Constraints: circles do not overlap
    def overlap_constraint(x, i, j):
        dx = x[i * 2] - x[j * 2]
        dy = x[i * 2 + 1] - x[j * 2 + 1]
        dist = np.sqrt(dx*dx + dy*dy)
        r_i = x[i]
        r_j = x[j]
        return dist - (r_i + r_j) >= 0

    # Flatten centers and radii into a single vector for optimization
    x0 = np.hstack((centers.reshape(-1), radii))

    # Define constraints
    constraints = []

    # Square constraint for all circles
    for i in range(26):
        constraint = {
            'type': 'ineq',
            'fun': lambda x, i=i: square_constraint(x, i)
        }
        constraints.append(constraint)

    # Overlap constraint for all pairs of circles
    for i in range(26):
        for j in range(i + 1, 26):
            constraint = {
                'type': 'ineq',
                'fun': lambda x, i=i, j=j: overlap_constraint(x, i, j)
            }
            constraints.append(constraint)

    # Optimization options
    options = {
        'disp': False,
        'ftol': 1e-6,
        'maxiter': 1000
    }

    # Run optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0=x0, constraints=constraints, method='SLSQP', options=options)

    # Extract results
    optimized_x = result.x
    centers = optimized_x[:26*2].reshape(26, 2)
    radii = optimized_x[26*2:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
