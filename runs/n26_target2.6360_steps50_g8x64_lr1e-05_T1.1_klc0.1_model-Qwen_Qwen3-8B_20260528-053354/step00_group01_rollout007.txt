import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: 26 circles arranged in a grid with some randomness
    num_circles = 26
    initial_radius = 0.05
    centers = np.random.uniform(0.1, 0.9, (num_circles, 2))
    radii = np.full(num_circles, initial_radius)

    # Objective function: minimize the negative sum of radii
    def objective(x):
        return -np.sum(x[1:])

    # Constraints: all circles must be inside the square
    def constraint_inside(x):
        return np.sum((x[1:] <= x[0:num_circles*2:2] - 1e-6) * (x[1:] >= x[0:num_circles*2:2] - 1e-6))

    # Constraint: distance between centers >= sum of radii
    def constraint_no_overlap(x):
        centers = x[0:num_circles*2:2].reshape(num_circles, 2)
        radii = x[1:num_circles*2+1:2]
        distances = np.sqrt(((centers[:, np.newaxis] - centers[np.newaxis, :]) ** 2).sum(axis=2))
        sum_radii = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.sum((distances >= sum_radii - 1e-6) * (distances - sum_radii + 1e-6))

    # Flatten centers and radii into a 1D array for optimization
    x0 = np.zeros(num_circles * 2 + 1)
    x0[0:num_circles * 2] = np.random.uniform(0.1, 0.9, num_circles * 2)
    x0[num_circles * 2] = initial_radius

    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': lambda x: constraint_inside(x)},
        {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x)}
    ]

    # Bounds for centers (must be within [0.1, 0.9] to leave space for radius)
    bounds = [(0.1, 0.9) for _ in range(num_circles * 2)] + [(0., 1.e5)]

    # Optimization
    result = minimize(objective, x0, constraints=cons, bounds=bounds, method='SLSQP', tol=1e-6)

    # Extract results
    centers_opt = result.x[0:num_circles*2].reshape(num_circles, 2)
    radii_opt = result.x[num_circles*2::2]
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
