import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # For simplicity, we will randomly initialize the centers
    # and radii, then perform a local optimization to maximize the sum of radii
    # This is a simplified heuristic; for a precise solution, more advanced algorithms like simulated annealing are typically used
    
    # Initialize centers randomly within the unit square
    for i in range(num_circles):
        centers[i] = np.random.uniform(low=0.1, high=0.9, size=2)
        radii[i] = 0.05  # Start with a small radius

    # Objective function to maximize the sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Minimize -sum for maximization

    # Constraint: each circle is inside the unit square
    def in_bounds(x):
        return np.all((x >= 0.01) & (x <= 0.99))

    # Constraint: distance between centers >= sum of radii
    def no_overlap(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            for j in range(i+1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1
        return 0

    # Flatten the vector for optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds for centers and radii
    bounds = []
    for i in range(num_circles):
        bounds.append((0.01, 0.99))  # x
        bounds.append((0.01, 0.99))  # y
        bounds.append((0.01, 0.5))   # radius (smaller to avoid overflow)

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': in_bounds, 'args': (centers.flatten(),)},
        {'type': 'ineq', 'fun': no_overlap, 'args': (x0,)}
    ]

    # Optimization
    result = minimize(objective, x0, method='L-BFGS-B', bounds=bounds, constraints=constraints)

    # Extract results
    centers_opt = result.x[:num_circles*2].reshape((num_circles, 2))
    radii_opt = result.x[num_circles*2:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
