import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_0(radii):
        return np.sum(radii) - 2.636

    def constraint_1(radii):
        dists = pdist(centers)
        r_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dists - np.min(r_sum, axis=1)

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(26)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_1(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_0(r)}
    ]

    # Initial guess for radii
    radii_initial = np.full(26, 0.1)

    # Optimization
    result = minimize(objective, radii_initial, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    # Ensure circles are within the unit square
    for i in range(26):
        r = radii[i]
        x, y = centers[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            raise ValueError(f"Circle {i} is not fully inside the unit square")

    return (centers, radii, sum_radii)
