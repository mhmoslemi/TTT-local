import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced grid with some small radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2) * 0.8 + 0.1  # Random initial positions inside the square
    radii = np.full(num_circles, 0.01)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraint 1: All circles must be inside the square
        # r <= x - r <= 1 - r => x >= 2r and x <= 1 - 2r => r <= (1 - x)/2 and r <= x/2
        # Same for y-coordinate
        circle_radii = np.reshape(radii, (num_circles,))
        x_min = np.maximum(circle_radii, np.abs(centers[:, 0] - 1))
        x_max = np.maximum(circle_radii, np.abs(centers[:, 0]))
        x_inside = (x_min <= 1 - x_max) * np.ones(num_circles)
        x_inside = np.sum(x_inside) == num_circles

        # Constraint 2: No overlap between circles
        distances = pdist(centers)
        radii_sum = np.tile(circle_radii, (num_circles, 1))
        radii_sum += np.tile(circle_radii.reshape(-1, 1), (1, num_circles))
        distances = distances - radii_sum
        dist_ge_zero = (distances >= 0) * np.ones(distances.shape)
        dist_ge_zero = np.sum(dist_ge_zero) == distances.size

        # Constraint 3: All radii are non-negative
        non_negative = (radii >= 0).all()

        return x_inside and dist_ge_zero and non_negative

    # Optimization: minimize negative sum of radii with constraints
    bounds = [(0, 0.5) for _ in range(num_circles)]
    result = minimize(objective, radii, bounds=bounds, constraints=constraints)

    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
