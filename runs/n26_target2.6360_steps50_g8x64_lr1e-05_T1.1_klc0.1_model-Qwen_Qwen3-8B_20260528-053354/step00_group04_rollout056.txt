import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (random positions in the square)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: all circles must be inside the square and no overlap
    def constraint_inside(radii):
        return np.sum((radii <= (1 - centers[:, 0])) & (radii <= (1 - centers[:, 1])) & (radii >= 0))

    def constraint_overlap(radii):
        n = len(centers)
        distances = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.hypot(dx, dy)
                distances[idx] = dist - (radii[i] + radii[j])
                idx += 1
        return np.min(distances)

    # Set up optimization problem
    bounds = [(0.0, 1.0) for _ in range(26)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
    ]

    # Optimize
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        tol=1e-6
    )

    # Extract final results
    final_radii = result.x
    final_centers = centers
    final_sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, final_sum_radii)
