import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i // 5
        y = i % 5
        centers[i] = [x / 4 + 0.5, y / 4 + 0.5]
    
    # Initial radii: all set to 0.1
    radii = np.full(num_circles, 0.1)
    sum_radii = sum(radii)
    
    # Objective function: maximize the sum of radii
    def objective(x):
        nonlocal radii
        radii = x[:(num_circles)]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize sum
    
    # Constraints: circles must be fully inside the unit square
    def constraint_inside(x):
        nonlocal radii
        r = x[:(num_circles)]
        return np.array([r[i] <= 1 - centers[i][0] for i in range(num_circles)] +
                        [r[i] <= 1 - centers[i][1] for i in range(num_circles)])
    
    # Constraints: circles must not overlap
    def constraint_no_overlap(x):
        nonlocal radii
        r = x[:(num_circles)]
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return dists - (r[:, None] + r[None, :])
    
    # Setup optimization
    bounds = [(0.0, 1.0 - 0.05) for _ in range(num_circles)]  # Ensure at least 0.05 margin
    constraints_inside = [{'type': 'ineq', 'fun': constraint_inside}]
    constraints_overlap = [{'type': 'ineq', 'fun': constraint_no_overlap}]
    constraints = constraints_inside + constraints_overlap
    
    # Optimization
    result = minimize(objective, x0=radii, bounds=bounds, constraints=constraints)
    
    if result.success:
        centers = np.array(centers)
        radii = np.array(result.x)
        sum_radii = np.sum(radii)
    else:
        # If optimization fails, return initial guess
        centers = np.array(centers)
        radii = np.array(radii)
        sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
