import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.05)  # Initial small radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum, i.e., maximize sum

    def constraint_circle_inside(radii, centers):
        # Ensure each circle is fully inside the unit square
        x, y = centers[:, 0], centers[:, 1]
        return np.minimum(x - radii, 1 - x - radii, y - radii, 1 - y - radii)

    def constraint_no_overlap(radii, centers):
        # Ensure no two circles overlap: distance between centers >= sum of radii
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization setup
    bounds = [(0, 0.5) for _ in range(num_circles)]  # Radii can't exceed 0.5
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
    ]
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'maxiter': 1000, 'ftol': 1e-9}
    )

    # Extract results
    centers = np.random.rand(num_circles, 2)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
