import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid of 26 points (5x5 grid with one point missing)
    centers = np.array([[i/4 + 0.1, j/4 + 0.1] for i in range(5) for j in range(5)])
    centers = np.vstack((centers, [0.5, 0.5]))  # add one extra point at center
    centers = centers[:26]  # ensure exactly 26 points

    # Initial radii: all same
    radii = np.array([0.05] * 26)

    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraint_circle_inside(radii):
        # Ensure each circle is fully inside the square
        r = radii
        return np.sum((np.min(centers, axis=0) - r) ** 2)  # not used directly in constraints, just for reference

    def constraint_no_overlap(radii):
        # Ensure no two circles overlap
        r = radii
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.min(dists - r[np.newaxis, :] - r[:, np.newaxis])  # ensure all distances >= sum of radii

    # Optimization bounds: radii >= 0, and circles fit in square
    bounds = [(0.0, 1.0 - x) for x in np.min(centers, axis=0)]

    # Constraints: no overlap and all circles inside the square
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r)},
        {'type': 'ineq', 'fun': lambda r: np.min(1.0 - centers - r)}
    ]

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=RuntimeWarning)
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract result
    centers_final = centers
    radii_final = result.x
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
