import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initialize centers with random positions inside the unit square
    centers = np.random.rand(26, 2)
    # Set initial radii to 0.1
    radii = np.full(26, 0.1)
    
    def objective_function(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        # Reshape to centers and radii
        flat_centers = x[:2*26].reshape(26, 2)
        flat_radii = x[2*26:].reshape(26)
        # Objective: maximize sum of radii, so minimize -sum(radii)
        return -np.sum(flat_radii)
    
    def constraint_inside(x):
        flat_centers = x[:2*26].reshape(26, 2)
        flat_radii = x[2*26:].reshape(26)
        # Check if all circles are inside the unit square
        for i in range(26):
            x_i, y_i = flat_centers[i]
            r_i = flat_radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1.0
        return 0.0
    
    def constraint_no_overlap(x):
        flat_centers = x[:2*26].reshape(26, 2)
        flat_radii = x[2*26:].reshape(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = flat_centers[i, 0] - flat_centers[j, 0]
                dy = flat_centers[i, 1] - flat_centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < flat_radii[i] + flat_radii[j]:
                    return -1.0
        return 0.0
    
    # Flatten initial centers and radii for optimization
    x0 = np.concatenate([centers.flatten(), radii])
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])
    
    # Set up constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    
    # Run the optimization
    result = minimize(
        fun=objective_function,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )
    
    # Extract result
    x_opt = result.x
    centers_opt = x_opt[:2*26].reshape(26, 2)
    radii_opt = x_opt[2*26:].reshape(26)
    sum_radii_opt = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii_opt)
