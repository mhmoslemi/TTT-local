import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess for centers (equally spaced grid)
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.1)

    # Objective function: minimize negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: Each circle must be entirely inside the unit square
    def constraint1(radii):
        return np.ones(num_circles) * (1 - radii) - np.abs(centers[:, 0] - radii) - np.abs(centers[:, 1] - radii)

    # Constraint 2: No two circles overlap
    def constraint2(radii):
        dists = pdist(centers)
        return dists - np.abs(radii[:, np.newaxis] + radii[np.newaxis, :])

    # Bounds for radii
    bounds = [(0, 1 - np.min(np.abs(centers[:, 0] - 0.5)))] * num_circles

    # Optimization problem
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Perform optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    centers = np.array(centers)
    sum_radii = -result.fun
    radii = result.x

    return (centers, radii, sum_radii)
