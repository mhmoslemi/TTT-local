import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: arrange circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial grid arrangement
    rows, cols = 5, 5  # 5x5 grid, 25 circles
    spacing = 1.0 / (cols + 1)
    for i in range(rows):
        for j in range(cols):
            centers[i * cols + j, 0] = j * spacing + spacing / 2
            centers[i * cols + j, 1] = i * spacing + spacing / 2
    # Add one more circle at the center (as a guess)
    centers[num_circles - 1, 0] = 0.5
    centers[num_circles - 1, 1] = 0.5
    
    # Initial radii
    radii[0:num_circles - 1] = 0.04
    radii[num_circles - 1] = 0.05
    
    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        # x contains all parameters: centers and radii
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        sum_r = np.sum(radii_flat)
        return -sum_r  # minimize negative sum to maximize
    
    # Constraints: 1. Circles are inside the square
    def constraint_inside(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            r = radii_flat[i]
            x_i, y_i = centers_flat[i]
            if not (r <= x_i <= 1 - r and r <= y_i <= 1 - r):
                return -1.0
        return 0.0
    
    # Constraints: 2. No overlap between circles
    def constraint_no_overlap(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 1e-8:
                    return -1.0
        return 0.0
    
    # Bounds for centers and radii
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0, 1), (0, 1)])
        bounds.extend([(0, 0.2)])  # Assume max radius is 0.2 for bounds

    # Initial x vector
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_inside},
                {'type': 'ineq', 'fun': constraint_no_overlap}
            ],
            tol=1e-8,
            options={'maxiter': 1000}
        )
    
    centers_opt = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii_opt = result.x[num_circles * 2:]
    sum_radii = -result.fun  # since we maximized the sum
    
    # Ensure all radii are non-negative and within bounds
    radii_opt = np.clip(radii_opt, 0, 0.2)
    
    # Final check for constraints
    for i in range(num_circles):
        r = radii_opt[i]
        x, y = centers_opt[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            raise ValueError("Circle is not fully inside the square.")
    
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = centers_opt[i, 0] - centers_opt[j, 0]
            dy = centers_opt[i, 1] - centers_opt[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii_opt[i] + radii_opt[j] - 1e-8:
                raise ValueError("Circles overlap.")
    
    return (centers_opt, radii_opt, sum_radii)
