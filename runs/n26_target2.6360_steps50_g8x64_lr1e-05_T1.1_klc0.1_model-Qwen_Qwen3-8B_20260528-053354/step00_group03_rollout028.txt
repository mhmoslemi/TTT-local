import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles evenly in a grid
    num_circles = 26
    centers = np.array([[i / 5 + 0.25, j / 5 + 0.25] for j in range(5) for i in range(5)])
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more circle in the center
    radii = np.ones(num_circles) * 0.1  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: All circles must be inside the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return False
        # Constraint 2: No two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_function(x):
        nonlocal radii
        radii = x
        return -np.sum(x)

    def constraint_function(x):
        nonlocal centers
        if not all(r >= 0 for r in x):
            return -np.inf
        for i in range(num_circles):
            if not (x[i] <= centers[i, 0] <= 1 - x[i] and x[i] <= centers[i, 1] <= 1 - x[i]):
                return -np.inf
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < x[i] + x[j]:
                    return -np.inf
        return np.sum(x)

    # Use scipy.optimize.minimize to maximize the sum of radii
    try:
        result = minimize(
            fun=objective_function,
            x0=radii,
            method='SLSQP',
            constraints=[{'type': 'ineq', 'fun': constraint_function}],
            bounds=[(0, 1) for _ in range(num_circles)]
        )
    except ImportError:
        # Fallback if scipy is not available
        return (centers, radii, np.sum(radii))
    
    # Extract the optimal solution
    optimal_radii = result.x
    optimal_centers = centers
    sum_radii = np.sum(optimal_radii)

    return (optimal_centers, optimal_radii, sum_radii)
