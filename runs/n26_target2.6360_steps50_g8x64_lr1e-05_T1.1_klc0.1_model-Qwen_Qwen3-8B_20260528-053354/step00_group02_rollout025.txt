import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = 0.2 + col * 0.25
        centers[i, 1] = 0.2 + row * 0.25
        radii[i] = 0.05
    sum_radii = np.sum(radii)

    # Define objective function
    def objective(x):
        nonlocal centers, radii
        # Update centers and radii with x
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        # Objective is to maximize sum of radii, so minimize negative sum
        return -np.sum(radii)

    # Define constraints
    def constraints(x):
        nonlocal centers, radii
        # Update centers and radii with x
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        # All circles must be inside the unit square
        in_square = []
        for i in range(26):
            x_i, y_i = centers[i]
            r_i = radii[i]
            in_square.append(1 - r_i - x_i >= 0)
            in_square.append(1 - r_i - y_i >= 0)
            in_square.append(r_i - x_i >= 0)
            in_square.append(r_i - y_i >= 0)
        # No overlapping circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Prepare initial guess
    x0 = np.concatenate([centers.reshape(52), radii])

    # Use SLSQP method with bounds and constraints
    bounds = []
    for i in range(26):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        bounds.append((0.0, 1.0))  # radius
    bounds = np.array(bounds)

    # Define constraint as a callable for scipy.optimize
    def constraint(x):
        return constraints(x)

    # Run optimization
    result = minimize(objective, x0=x0, bounds=bounds, constraints={"type": "ineq", "fun": constraint}, method='SLSQP')

    # Extract final result
    centers_final = result.x[:52].reshape(26, 2)
    radii_final = result.x[52:]
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
