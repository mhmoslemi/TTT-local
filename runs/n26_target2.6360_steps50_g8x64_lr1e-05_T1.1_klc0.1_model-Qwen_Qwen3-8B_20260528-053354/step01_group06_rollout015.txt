import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: arrange circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.01)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        radius = 0.01
        centers[i] = [col * (1 - 2 * radius) + radius, row * (1 - 2 * radius) + radius]
    
    def objective(radii, centers):
        return -np.sum(radii)  # maximize sum of radii
    
    def constraints(radii, centers):
        # Ensure circles are inside the unit square
        inside = (radii <= (1 - centers[:, 0] - radii)) & (radii <= (1 - centers[:, 1] - radii))
        inside = inside & (radii >= 0)
        # Ensure no overlap between circles
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0]) ** 2) + 
                        ((centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1]) ** 2))
        overlap = (dists < radii[np.newaxis, :] + radii[:, np.newaxis])
        inside = inside.all()
        overlap = overlap.any()
        if not inside or overlap:
            return np.inf
        return 0
    
    # Define bounds for radii
    bounds = [(0, 0.5) for _ in range(num_circles)]
    
    # Set up optimization problem
    def objective_func(x):
        radii[:] = x
        return -np.sum(x)
    
    def constraint_func(x):
        radii[:] = x
        return constraints(x, centers)
    
    result = minimize(objective_func, radii, bounds=bounds, constraints={"type": "ineq", "fun": constraint_func})
    
    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
