import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for circle centers and radii
    # Using a simple grid layout for initial guess of 26 circles
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    for i in range(n):
        row = i // 5
        col = i % 5
        x = 0.2 + col * 0.4
        y = 0.2 + row * 0.4
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_distances(centers, radii):
        n = len(radii)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j] - 1e-8:
                    return 1  # Violation of non-overlapping constraint
        return 0

    def constraint_bounds(radii, centers):
        for i in range(len(radii)):
            if radii[i] < 0:
                return 1  # Negative radius
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return 1  # Circle is outside the unit square
        return 0

    def objective_func(x):
        radii = x[:n]
        centers = x[n:]
        centers = centers.reshape((n, 2))
        # Apply constraints
        if constraint_bounds(radii, centers) != 0:
            return float('inf')
        if constraint_distances(centers, radii) != 0:
            return float('inf')
        return -np.sum(radii)

    # Optimize with bounds and constraints
    bounds = [(0, 0.5)] * n + [(0.0, 1.0)] * (2 * n)  # Radii and centers

    result = minimize(
        objective_func,
        x0=np.concatenate((radii, centers.flatten())),
        method='L-BFGS-B',
        bounds=bounds,
        tol=1e-6,
        options={'disp': False}
    )

    # Extract optimized values
    optimized_radii = result.x[:n]
    optimized_centers = result.x[n:].reshape((n, 2))

    # Check constraints after optimization
    for i in range(n):
        x, y = optimized_centers[i]
        if not (optimized_radii[i] <= x <= 1 - optimized_radii[i] and optimized_radii[i] <= y <= 1 - optimized_radii[i]):
            optimized_radii[i] = 0  # Clamp if out of bounds
    for i in range(n):
        for j in range(i + 1, n):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            distance = math.hypot(dx, dy)
            if distance < optimized_radii[i] + optimized_radii[j] - 1e-8:
                optimized_radii[i] = 0  # Adjust if overlap

    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
