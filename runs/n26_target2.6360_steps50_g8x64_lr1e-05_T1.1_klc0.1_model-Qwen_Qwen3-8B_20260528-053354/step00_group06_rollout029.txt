import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i / 5
        y = 0.5
        centers[i] = [x, y]
    
    # Initial radius guess: all radii start as 0.1
    radii = np.full(26, 0.1)
    sum_radii = np.sum(radii)
    
    # Define the objective function (minimize negative sum of radii)
    def objective(x):
        nonlocal sum_radii
        # Flatten x to get centers and radii
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:].reshape(26)
        sum_radii = np.sum(radii_flat)
        return -sum_radii
    
    # Define the constraints
    def constraint_center_inside(x):
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:].reshape(26)
        # Check if all circles are inside the square
        inside = np.all((radii_flat <= centers_flat[:, 0]) & (centers_flat[:, 0] <= 1 - radii_flat) &
                        (radii_flat <= centers_flat[:, 1]) & (centers_flat[:, 1] <= 1 - radii_flat))
        return inside
    
    def constraint_no_overlap(x):
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:].reshape(26)
        # Check pairwise distances
        dists = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        min_dist = np.min(dists)
        # If any pair is less than sum of their radii, this is a violation
        violating = np.any(dists < radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])
        return not violating
    
    # Optimization setup
    x0 = np.concatenate((centers.flatten(), radii))
    bounds = []
    for _ in range(26*2):
        bounds.append((0, 1))  # x and y centers must be in [0,1]
    for _ in range(26):
        bounds.append((0, 0.5))  # radii must be <= 0.5 to fit in the square
    
    # Set up constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    
    # Optimize
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            tol=1e-6
        )
    
    # Extract final centers, radii, and sum_radii
    centers_final = result.x[:26*2].reshape(26, 2)
    radii_final = result.x[26*2:].reshape(26)
    sum_radii_final = np.sum(radii_final)
    
    return (centers_final, radii_final, sum_radii_final)
