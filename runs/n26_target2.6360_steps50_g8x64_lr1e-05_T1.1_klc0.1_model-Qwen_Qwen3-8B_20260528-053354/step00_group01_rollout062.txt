import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii (uniform grid with equal radii)
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    radii = np.array([0.05] * 25)
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one more center in the middle
    radii = np.append(radii, 0.1)  # Initial guess for the middle circle

    def objective_function(params):
        centers = params[:52].reshape(26, 2)
        radii = params[52:]
        sum_radii = np.sum(radii)
        # Objective: minimize negative sum of radii (maximize sum of radii)
        return -sum_radii

    def constraint_function(params):
        centers = params[:52].reshape(26, 2)
        radii = params[52:]
        constraints = []
        # Constraint 1: All circles are inside the unit square
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -np.inf  # Penalize if outside
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf  # Penalize if overlapping
        return 0  # All constraints satisfied

    # Prepare initial parameters
    params = np.concatenate([centers.reshape(52), radii])

    # Set up optimization
    bounds = []
    for i in range(26):
        bounds.append((0.0, 1.0))  # x and y coordinates between 0 and 1
    for i in range(26):
        bounds.append((0.0, 1.0))  # radii between 0 and 1 (but realistically smaller)
    # Set up constraints and minimize
    constraints = [{'type': 'ineq', 'fun': constraint_function}]
    result = minimize(objective_function, params, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract results
    centers_result = result.x[:52].reshape(26, 2)
    radii_result = result.x[52:]
    sum_radii_result = -result.fun  # Since we minimized -sum_radii

    return (centers_result, radii_result, sum_radii_result)
