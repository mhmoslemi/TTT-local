import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid points with small radii
    centers = np.random.rand(26, 2) * 0.8 + 0.1  # Initial centers inside the square
    radii = np.full(26, 0.05)  # Initial small radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints_func(radii):
        # Constraint 1: All circles are inside the unit square
        r = np.clip(radii, 0, None)
        x = centers[:, 0]
        y = centers[:, 1]
        inside = (r <= x) & (x <= 1 - r) & (r <= y) & (y <= 1 - r)
        if not np.all(inside):
            return np.inf  # Penalize if any circle goes out of bounds

        # Constraint 2: No overlapping between circles
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf  # Penalize if circles overlap

        return 0.0  # No violation

    # Optimize using scipy.optimize.minimize
    bounds = [(0, 1) for _ in range(26)]  # Bounds for all radii

    try:
        result = minimize(
            objective,
            radii,
            method='L-BFGS-B',
            bounds=bounds,
            constraints={'type': 'ineq', 'fun': constraints_func},
            tol=1e-6
        )
    except:
        result = minimize(
            objective,
            radii,
            method='L-BFGS-B',
            bounds=bounds,
            tol=1e-6
        )

    optimized_radii = result.x
    optimized_centers = centers.copy()
    sum_radii = -result.fun

    return (optimized_centers, optimized_radii, sum_radii)
