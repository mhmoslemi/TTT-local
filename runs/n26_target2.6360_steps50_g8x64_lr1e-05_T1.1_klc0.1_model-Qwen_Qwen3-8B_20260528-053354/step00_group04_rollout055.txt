import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly placed circles
    centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
    radii = np.random.uniform(low=0.01, high=0.05, size=26)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(26):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return np.inf
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    def objective_wrapper(x):
        radii = x
        return -np.sum(radii)

    def constraint_wrapper(x):
        # Ensure all circles are within the unit square
        for i in range(26):
            r = x[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return np.inf
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < x[i] + x[j]:
                    return np.inf
        return 0

    # Optimization
    result = minimize(
        fun=objective_wrapper,
        x0=radii,
        bounds=[(0, 0.5)] * 26,
        constraints={"type": "ineq", "fun": constraint_wrapper},
        method="SLSQP"
    )

    radii_opt = result.x
    sum_radii = -result.fun

    # Convert to numpy arrays
    centers_array = np.array(centers)
    radii_array = np.array(radii_opt)

    return (centers_array, radii_array, sum_radii)
