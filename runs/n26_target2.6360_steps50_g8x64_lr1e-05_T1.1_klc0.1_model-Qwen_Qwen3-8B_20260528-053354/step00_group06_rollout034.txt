import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for circle centers (spread out in a grid)
    centers = np.array([[x / 5 + 0.1, y / 5 + 0.1] for x in range(5) for y in range(5)])
    centers = np.vstack([centers, np.array([[0.5, 0.1]], dtype=float)])
    centers = np.vstack([centers, np.array([[0.1, 0.5]], dtype=float)])
    centers = np.vstack([centers, np.array([[0.5, 0.5]], dtype=float)])
    
    # Initial guess for radii
    radii = np.ones(26) * 0.1

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        # Reshape into centers and radii
        x_flat = x
        centers_flat = x_flat[:2*26].reshape((26, 2))
        radii_flat = x_flat[2*26:]
        
        # Objective is to minimize the negative sum of radii
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    def constraints(x):
        x_flat = x
        centers_flat = x_flat[:2*26].reshape((26, 2))
        radii_flat = x_flat[2*26:]
        
        # Constraint 1: All circles are inside the unit square
        in_square = np.all((radii_flat <= centers_flat[:, 0]) & (radii_flat <= 1 - centers_flat[:, 0]) &
                          (radii_flat <= centers_flat[:, 1]) & (radii_flat <= 1 - centers_flat[:, 1]), axis=1)
        if not np.all(in_square):
            return np.inf
        
        # Constraint 2: No two circles overlap
        dists = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        radii_sum = radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :]
        overlap = np.any(dists < radii_sum - 1e-6)
        if overlap:
            return np.inf
        
        return 0

    # Optimization bounds (each center x and y between 0 and 1, each radius non-negative)
    bounds = []
    for _ in range(26):
        bounds.extend([(0, 1), (0, 1), (0, 0.5)])
    
    # Optimization
    result = minimize(objective, np.concatenate([centers.flatten(), radii]), 
                      bounds=bounds, constraints={"type": "ineq", "fun": constraints}, 
                      method="SLSQP", tol=1e-8, options={"maxiter": 1000})
    
    # Extract results
    optimal_x = result.x
    centers_opt = optimal_x[:2*26].reshape((26, 2))
    radii_opt = optimal_x[2*26:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
