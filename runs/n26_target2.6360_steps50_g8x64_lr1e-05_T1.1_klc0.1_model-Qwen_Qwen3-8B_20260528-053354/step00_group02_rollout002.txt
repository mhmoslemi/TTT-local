import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():

    centers = np.zeros((n_circles, 2))

        x = i / (n_circles - 1)
        y = (i % 5) / 4
        centers[i] = [x, y]

    radii = np.ones(n_circles) * 0.1

    def objective(radii):

        centers = np.zeros((n_circles, 2))
        for i in range(n_circles):
            x = i / (n_circles - 1)
            y = (i % 5) / 4
            centers[i] = [x, y]
        # Ensure all circles are inside the unit square
        def in_square(r):
            return r <= 1 - r
        in_square_cons = [in_square(r) for r in radii]
        # Ensure no circles overlap
        def no_overlap(i, j):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            return dist - (radii[i] + radii[j]) >= -1e-6
        no_overlap_cons = []
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                no_overlap_cons.append(no_overlap(i, j))
        return in_square_cons + no_overlap_cons

    def callback(x):
        pass  # No callback needed

    result = minimize(objective, radii, method='SLSQP',
                      constraints={'type': 'ineq', 'fun': constraints},
                      bounds=[(0, 1) for _ in range(n_circles)],
                      callback=callback)

    if not result.success:
        return (centers, radii, np.sum(radii))
    
    best_radii = result.x
    sum_radii = np.sum(best_radii)
    return (centers, best_radii, sum_radii)
