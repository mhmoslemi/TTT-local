import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        x = i / (n_circles - 1)
        y = (i // (n_circles // 2)) / (n_circles // 2)
        centers[i] = [x, y]
    
    # Objective function to minimize: negative sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints: each circle must be inside the square and no overlaps
    def constraint_inside(radii, centers):
        # Each circle must be inside the square
        return np.concatenate([radii - centers[:, 0], radii - (1 - centers[:, 0]),
                               radii - centers[:, 1], radii - (1 - centers[:, 1])])
    
    def constraint_overlap(radii, centers):
        # Distance between centers >= radii[i] + radii[j] for all i < j
        D = cdist(centers, centers)
        return np.concatenate([D[i, j] - (radii[i] + radii[j]) for i in range(n_circles) for j in range(i+1, n_circles)])
    
    # Initial radii
    radii = np.ones(n_circles) * 0.1
    
    # Set up optimization problem
    bounds = [(0, 0.5) for _ in range(n_circles)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers)}
    ]
    
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)
    
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii_final = np.sum(radii_final)
    
    return (centers_final, radii_final, sum_radii_final)
