```python
import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a hexagonal grid layout
    def initial_centers():
        centers = []
        for i in range(26):
            row = i // 7
            col = i % 7
            x = col + 0.5 * (1 - (row % 2))
            y = row * np.sqrt(3) / 2 + 0.5
            centers.append([x, y])
        return np.array(centers)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii, centers):
        return np.sum((centers[:, 0] - radii) >= 0) + np.sum((1 - centers[:, 0] - radii) >= 0) + \
               np.sum((centers[:, 1] - radii) >= 0) + np.sum((1 - centers[:, 1] - radii) >= 0)

    def constraint_overlap(radii, centers):
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        min_dist = np.min(dists[np.triu_indices(26, 1)])
        return min_dist - (radii[np.newaxis, :] + radii[:, np.newaxis])

    def constraints(radii, centers):
        return [constraint_center_inside(radii, centers), constraint_overlap(radii, centers)]

    def run():
        centers = initial_centers()
        radii = np.ones(26) * 0.1
        bounds = [(0, 1) for _ in range(26)]

        result = minimize(
            objective,
            radii,
            args=(centers,),
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            tol=1e-6,
            options={'ftol': 1e-9, 'maxiter': 1000}
        )

        if not result.success:
            warnings.warn("Optimization did not converge")
            return (centers, np.zeros(26), 0)

        return (centers, result.x, np.sum(result.x))

    return run()
```
