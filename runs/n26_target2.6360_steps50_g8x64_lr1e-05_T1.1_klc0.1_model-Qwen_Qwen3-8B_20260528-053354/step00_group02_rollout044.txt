import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.1)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add center circle
    radii = np.append(radii, 0.1)

    def objective(radii_flat):
        return -np.sum(radii_flat)  # Minimize negative sum to maximize sum

    def constraints(radii_flat):
        centers_flat = np.reshape(centers, (26, 2))
        radii = np.reshape(radii_flat, (26,))

        # Constraint: All circles are inside the square
        for i in range(26):
            if radii[i] > 1 - 2 * centers_flat[i, 0] or radii[i] > 1 - 2 * centers_flat[i, 1]:
                return False

        # Constraint: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distance = np.sqrt(dx*dx + dy*dy)
                if distance < radii[i] + radii[j]:
                    return False

        return True

    # Define bounds for radii
    bounds = [(0.0, 1.0) for _ in range(26)]

    # Define a function for optimization
    def objective_func(radii_flat):
        if not constraints(radii_flat):
            return np.inf  # Penalize invalid configurations
        return -np.sum(radii_flat)

    # Run optimization
    result = minimize(objective_func, radii, bounds=bounds, method='SLSQP', options={'maxiter': 1000})
    radii_opt = result.x
    centers_opt = centers
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
