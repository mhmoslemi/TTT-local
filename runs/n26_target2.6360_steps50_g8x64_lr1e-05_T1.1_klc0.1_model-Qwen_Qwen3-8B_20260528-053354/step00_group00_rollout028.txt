import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Start by placing circles in a grid with spacing equal to the radius
    # First, estimate a reasonable initial radius
    # For simplicity, assume circles are placed in a 5x5 grid with one circle removed
    grid_size = 5
    spacing = 1.0 / grid_size
    initial_radius = spacing * 0.6  # Adjust based on expected optimal value
    
    for i in range(num_circles):
        x = i % grid_size
        y = i // grid_size
        centers[i, 0] = x * spacing + spacing * 0.5
        centers[i, 1] = y * spacing + spacing * 0.5
        radii[i] = initial_radius
    
    # Objective function to maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negate for minimization

    # Constraint: each circle is fully inside the unit square
    def square_inside(radii):
        return np.all(radii <= 1 - centers[:, 0]) and np.all(radii <= 1 - centers[:, 1])

    # Constraint: no overlap between circles
    def no_overlap(radii):
        dist = cdist(centers, centers)
        return np.all(dist >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Combine constraints
    def constraints(radii):
        return [square_inside(radii), no_overlap(radii)]

    # Optimization setup
    bounds = [(0, 1) for _ in range(num_circles)]  # Each radius is between 0 and 1

    # Initial guess with uniform radii
    initial_guess = np.full(num_circles, initial_radius)

    # Optimization
    result = minimize(objective, initial_guess, bounds=bounds, constraints=constraints)

    # Extract the optimized values
    optimized_radii = result.x
    optimized_centers = centers.copy()

    # Ensure all constraints are satisfied
    # (This is redundant if the constraints are correctly implemented, but for safety)
    dist = cdist(optimized_centers, optimized_centers)
    is_overlap = np.any(dist < optimized_radii[:, np.newaxis] + optimized_radii[np.newaxis, :])
    if is_overlap:
        raise ValueError("Optimization failed to satisfy non-overlap constraint")
    is_outside = np.any(optimized_radii > 1 - optimized_centers[:, 0]) or np.any(
        optimized_radii > 1 - optimized_centers[:, 1]
    )
    if is_outside:
        raise ValueError("Optimization failed to satisfy square inside constraint")

    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
