import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: evenly spaced circles in the square
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Initial radii

    def objective(x):
        # x is a vector containing all centers and radii
        # Reshape into centers and radii
        flat_centers = x[:2*26].reshape(26, 2)
        flat_radii = x[2*26:]
        # Objective is to maximize the sum of radii, so minimize negative sum
        sum_radii = np.sum(flat_radii)
        return -sum_radii

    def constraints(x):
        # x is a vector containing all centers and radii
        # Reshape into centers and radii
        flat_centers = x[:2*26].reshape(26, 2)
        flat_radii = x[2*26:]
        constraints = []

        # 1. All circles must be fully inside the unit square
        for i in range(26):
            x_i, y_i = flat_centers[i]
            r_i = flat_radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                constraints.append(1.0)
            else:
                constraints.append(0.0)

        # 2. No two circles overlap
        for i, j in itertools.combinations(range(26), 2):
            dx = flat_centers[i][0] - flat_centers[j][0]
            dy = flat_centers[i][1] - flat_centers[j][1]
            distance = np.sqrt(dx**2 + dy**2)
            r_i = flat_radii[i]
            r_j = flat_radii[j]
            if distance < r_i + r_j:
                constraints.append(1.0)
            else:
                constraints.append(0.0)

        return np.array(constraints)

    # Define bounds: centers between 0 and 1, radii between 0 and 0.5
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])

    bounds.extend([(0.0, 0.5) for _ in range(26)])

    # Setup for optimization
    x0 = np.concatenate([centers.flatten(), radii])
    res = minimize(objective, x0, bounds=bounds, constraints={"type": "ineq", "fun": constraints}, method="SLSQP")

    # Extract the result
    final_centers = res.x[:2*26].reshape(26, 2)
    final_radii = res.x[2*26:]
    sum_radii = -res.fun

    return (final_centers, final_radii, sum_radii)
