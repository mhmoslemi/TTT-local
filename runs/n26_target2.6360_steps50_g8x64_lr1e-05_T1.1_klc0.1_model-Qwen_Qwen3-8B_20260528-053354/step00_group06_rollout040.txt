import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import math

def run_packing():
    # Initial guess: equally spaced grid
    n = 26
    centers = np.linspace(0.1, 0.9, n).reshape(-1, 1)
    centers = np.hstack((centers, centers))
    centers = centers[np.random.permutation(n), :]

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(radii):
        return np.sum((radii > 0) * (radii > 1 - np.min(np.abs(centers), axis=0)))  # Ensure circles are fully inside

    def constraint_no_overlap(radii):
        dist = pdist(centers)
        return np.sum((dist < np.abs(radii[:, np.newaxis] + radii[np.newaxis, :])) * (radii > 0))  # Ensure no overlaps

    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    bounds = [(0, 1 - 0.01) for _ in range(n)]  # Avoid edge cases by setting a small buffer

    result = minimize(objective, np.ones(n) * 0.1, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-5)

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
