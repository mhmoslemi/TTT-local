import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: evenly spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all the same
    radii = np.ones(num_circles) * 0.1

    # Optimization bounds
    bounds = [(0, 1 - r) for r in radii]

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Constraint 1: circles must be inside the unit square
    def constraint_inside(x):
        return np.array([x[i] <= 1 - x[i] for i in range(len(x))], dtype=float)

    # Constraint 2: no overlap between circles
    def constraint_overlap(x):
        dists = pdist(centers)
        radii_sum = np.sum(radii)
        return radii_sum - np.min(dists)

    # Combine constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)
    radii = result.x
    sum_radii = -result.fun

    # Final centers (fixed)
    centers = np.array(centers)
    return (centers, radii, sum_radii)
