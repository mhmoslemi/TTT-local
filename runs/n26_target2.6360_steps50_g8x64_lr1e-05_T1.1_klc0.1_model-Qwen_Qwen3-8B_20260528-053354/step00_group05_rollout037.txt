import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place 26 circles in a grid pattern
    n_circles = 26
    centers = np.array([[i / np.sqrt(n_circles), j / np.sqrt(n_circles)] 
                        for i in range(n_circles) for j in range(n_circles)])
    centers = centers[:n_circles]
    radii = np.full(n_circles, 0.1)  # Initial radius

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Constraints: circles must be within the square and not overlap
    def constraint_square(x, i):
        r = x[i]
        x_i, y_i = centers[i]
        return r - (1 - r - x_i)  # Ensure x_i + r <= 1
    def constraint_overlap(x, i, j):
        r_i = x[i]
        r_j = x[j]
        dx = centers[i, 0] - centers[j, 0]
        dy = centers[i, 1] - centers[j, 1]
        return np.sqrt(dx**2 + dy**2) - (r_i + r_j)  # Ensure distance >= sum of radii

    # Combine all constraints
    constraints = []
    for i in range(n_circles):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_square(x, i)})
        for j in range(i + 1, n_circles):
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)})

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=constraints, bounds=[(0, 1)] * n_circles)

    # Extract results
    final_centers = centers.copy()
    final_radii = result.x
    final_sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, final_sum_radii)
