import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.array([[0.1 + i*0.1, 0.1 + j*0.1] for i in range(26) for j in range(1)])
    radii = np.full(26, 0.05)

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_circle_inside(radii, centers):
        # Ensure each circle is inside the square
        return np.sum((centers[:, 0] - radii) >= 0) + np.sum((1 - centers[:, 0] - radii) >= 0) + \
               np.sum((centers[:, 1] - radii) >= 0) + np.sum((1 - centers[:, 1] - radii) >= 0)

    def constraint_no_overlap(radii, centers):
        # Ensure circles do not overlap
        distances = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum(distances >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization problem
    bounds = Bounds(0, 1)  # Radii must be between 0 and 1

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
            {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
        ],
        options={'ftol': 1e-5, 'disp': False}
    )

    # Extract solution
    optimal_radii = result.x
    optimal_sum_radii = np.sum(optimal_radii)
    centers = np.array([[0.1 + i*0.1, 0.1 + j*0.1] for i in range(26) for j in range(1)])

    return (centers, optimal_radii, optimal_sum_radii)
