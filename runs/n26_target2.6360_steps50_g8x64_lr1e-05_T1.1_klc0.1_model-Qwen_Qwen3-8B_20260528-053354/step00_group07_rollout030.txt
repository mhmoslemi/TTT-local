import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced points in the square
    centers = np.array([[x / 5 + 0.25, y / 5 + 0.25] for y in range(6) for x in range(6)])
    centers = centers[:26]  # Take first 26 points

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        n = len(radii)
        cons = []
        for i in range(n):
            for j in range(i + 1, n):
                dist_sq = np.sum((centers[i] - centers[j]) ** 2)
                # Constraint: distance^2 >= (r_i + r_j)^2 => dist_sq - (r_i + r_j)^2 >= 0
                cons.append(dist_sq - (radii[i] + radii[j]) ** 2)
        return np.array(cons)

    def bounds(radii):
        return [(0, 1) for _ in range(len(radii))]

    # Initial guess for radii
    radii = np.ones(26) * 0.1

    # Optimization
    res = minimize(objective, radii, method='SLSQP',
                   constraints={'type': 'ineq', 'fun': constraints},
                   bounds=bounds,
                   jac=lambda r: -np.ones(len(r)))

    # Check if optimization was successful
    if not res.success:
        warnings.warn("Optimization did not converge")
        return (centers, radii, np.sum(radii))

    # Update radii with optimal values
    radii = res.x

    # Check if all circles fit in the unit square
    for i in range(len(radii)):
        x, y = centers[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            warnings.warn("Circle does not fit in the unit square")
            return (centers, radii, np.sum(radii))

    return (centers, radii, np.sum(radii))
