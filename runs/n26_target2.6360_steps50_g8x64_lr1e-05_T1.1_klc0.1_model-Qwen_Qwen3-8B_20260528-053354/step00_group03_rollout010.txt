import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with some radius
    num_circles = 26
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 6) for y in np.linspace(0.1, 0.9, 4)])
    radii = np.full(num_circles, 0.1)  # Initial guess for radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure all circles are within the unit square
        for i in range(num_circles):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        # Ensure no overlap between circles
        dists = pdist(centers)
        for d, i, j in zip(dists, range(len(dists)), range(len(dists))):
            r_i = radii[i]
            r_j = radii[j]
            if d < r_i + r_j:
                return False
        return True

    # Optimization
    result = minimize(
        objective,
        radii,
        method='L-BFGS-B',
        bounds=[(0, 0.5)] * num_circles,
        constraints={"type": "ineq", "fun": constraints},
        options={'gtol': 1e-6, 'maxiter': 1000}
    )

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    # Validate
    if not constraints(radii):
        raise RuntimeError("Optimization failed to satisfy constraints")

    return (centers, radii, sum_radii)
