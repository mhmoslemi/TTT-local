import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 26 circles placed in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    # Initial radius guess
    init_radius = 0.05
    # Initial grid positions
    for i in range(num_circles):
        x = i % 5 / 4.0 + 0.1
        y = i // 5 / 4.0 + 0.1
        centers[i] = [x, y]
        radii[i] = init_radius

    def objective(radii_guess):
        return -np.sum(radii_guess)  # Minimize negative sum to maximize sum

    def constraints(radii_guess):
        # Constraint 1: All circles are within the unit square
        r = radii_guess
        for i in range(num_circles):
            if r[i] < 0:
                return False
            if r[i] > 1 - centers[i][0] or r[i] > 1 - centers[i][1]:
                return False
        # Constraint 2: No two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.hypot(dx, dy)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Optimization setup
    bounds = [(0.0, 1.0) for _ in range(num_circles)]
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'maxiter': 200}
    )

    # Extract the final solution
    final_radii = result.x
    final_centers = centers.copy()
    final_sum_radii = -result.fun

    # Ensure all radii are non-negative and within bounds
    final_radii = np.clip(final_radii, 0, 1)
    for i in range(num_circles):
        if final_radii[i] > 1 - final_centers[i][0] or final_radii[i] > 1 - final_centers[i][1]:
            final_radii[i] = np.min([1 - final_centers[i][0], 1 - final_centers[i][1]])

    # Final check for overlap
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = final_centers[i][0] - final_centers[j][0]
            dy = final_centers[i][1] - final_centers[j][1]
            dist = math.hypot(dx, dy)
            if dist < final_radii[i] + final_radii[j]:
                final_radii[i] = (dist - final_radii[j]) / 2
                final_radii[j] = (dist - final_radii[i]) / 2

    return (final_centers, final_radii, final_sum_radii)
