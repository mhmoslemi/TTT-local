import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: spread circles evenly in the square
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    def objective_function(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: All circles must be inside the unit square
        centers = np.random.rand(26, 2)  # Re-initialize centers for each evaluation
        for i in range(26):
            x, y = centers[i]
            if radii[i] > 0:
                if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                    return False
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define bounds for radii (non-negative)
    bounds = [(0, 0.5) for _ in range(26)]

    # Optimization
    result = minimize(objective_function, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Post-processing
    centers = np.random.rand(26, 2)
    sum_radii = result.fun * -1
    return (centers, result.x, sum_radii)
