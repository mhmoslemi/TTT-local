import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    radii = np.zeros(25)
    sum_radii = 0.0

    # Objective function: minimize the negative sum of radii (maximize sum)
    def objective(radii):
        nonlocal centers
        return -np.sum(radii)

    # Constraint 1: circles must remain inside the unit square
    def constraint_inside(radii):
        nonlocal centers
        return np.array([r <= 1 - r for r in radii], dtype=np.float64)

    # Constraint 2: circles must not overlap
    def constraint_overlap(radii):
        nonlocal centers
        n = len(radii)
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return np.array([dists[i, j] - (radii[i] + radii[j]) for i in range(n) for j in range(n)], dtype=np.float64)

    # Setup optimization
    bounds = [(0.0, 1.0 - 1e-6) for _ in range(25)]
    cons_inside = [{'type': 'ineq', 'fun': constraint_inside}]
    cons_overlap = [{'type': 'ineq', 'fun': constraint_overlap}]
    cons = cons_inside + cons_overlap

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract solution
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
