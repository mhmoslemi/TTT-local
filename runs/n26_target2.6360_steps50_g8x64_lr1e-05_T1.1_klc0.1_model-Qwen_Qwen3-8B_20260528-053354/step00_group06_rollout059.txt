import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Define initial guess: equally spaced grid of 26 circles
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        centers[i, 0] = (i + 1) / (n_circles + 1)
        centers[i, 1] = 0.5  # initial y-coordinate

    # Initial radii: small value to avoid overlap
    radii = np.full(n_circles, 0.01)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints:
    # 1. All circles are inside the unit square
    # 2. No two circles overlap
    # 3. All radii are non-negative

    def constraint_inside(x):
        r = x[n_circles:]
        centers_new = x[:n_circles*2].reshape(n_circles, 2)
        for i in range(n_circles):
            if not (r[i] <= centers_new[i, 0] <= 1 - r[i] and r[i] <= centers_new[i, 1] <= 1 - r[i]):
                return -1
        return 0

    def constraint_overlap(x):
        r = x[n_circles:]
        centers_new = x[:n_circles*2].reshape(n_circles, 2)
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers_new[i, 0] - centers_new[j, 0]
                dy = centers_new[i, 1] - centers_new[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return -1
        return 0

    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds: centers must be in [0, 1], radii must be >= 0
    bounds = [(0.0, 1.0) for _ in range(n_circles*2)] + [(0.0, 1.0) for _ in range(n_circles)]

    # Initial guess: flatten centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Minimize the negative of the objective function
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract results
    centers_result = result.x[:n_circles*2].reshape(n_circles, 2)
    radii_result = result.x[n_circles:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
