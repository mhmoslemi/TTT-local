import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as dist

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    radii = np.ones(centers.shape[0]) * 0.05

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii[:, np.newaxis] <= (1 - centers[:, 0]) - radii[:, np.newaxis]) & 
                        (radii[:, np.newaxis] <= (1 - centers[:, 1]) - radii[:, np.newaxis]), axis=1)
        inside = np.all(inside)

        # Constraint 2: No two circles overlap
        distances = dist.cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        no_overlap = np.all(distances >= radii_sum)
        return inside and no_overlap

    def optimize():
        def objective_func(radii_flat):
            radii = radii_flat.reshape(centers.shape[0])
            return -np.sum(radii)

        def constraint_func(radii_flat):
            radii = radii_flat.reshape(centers.shape[0])
            inside = np.all((radii <= (1 - centers[:, 0]) - radii) & (radii <= (1 - centers[:, 1]) - radii))
            if not inside:
                return -np.inf
            distances = dist.cdist(centers, centers)
            radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
            no_overlap = np.all(distances >= radii_sum)
            if no_overlap:
                return 0
            else:
                return -np.inf

        bounds = [(0, 0.5) for _ in range(centers.shape[0])]
        result = minimize(objective_func, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_func})
        return result.x, -result.fun

    radii, sum_radii = optimize()
    return (centers, radii, sum_radii)
