import numpy as np
from scipy.optimize import minimize

def run_packing():
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    indices = np.arange(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (indices[i] % 5 + 0.5) / 5
        centers[i, 1] = (indices[i] // 5 + 0.5) / 5

    radii = np.full(num_circles, 1 / 10)

    def objective_func(radii):
        return -np.sum(radii)

    def constraint_radius(radii, i):
        return radii[i] - 1e-8

    def constraint_position_x(radii, i):
        x_center, _ = centers[i]
        return radii[i] - (1 - x_center - radii[i])

    def constraint_position_y(radii, i):
        _, y_center = centers[i]
        return radii[i] - (1 - y_center - radii[i])

    def constraint_overlap(radii):
        total = 0.0
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                total += max(0, dist - (radii[i] + radii[j]))
        return total

    constraints = []

    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_radius(r, i)})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_position_x(r, i)})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_position_y(r, i)})

    constraints.append({'type': 'ineq', 'fun': lambda r: constraint_overlap(r)})

    initial_guess = radii.copy()

    result = minimize(
        objective_func,
        initial_guess,
        method='SLSQP',
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000, 'disp': False}
    )

    if not result.success:
        print("Optimization failed.")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    radii = result.x

    # Validate and adjust radii to ensure they fit within the square
    for i in range(num_circles):
        x_center, y_center = centers[i]
        if radii[i] > 1 - x_center - radii[i]:
            radii[i] = 1 - x_center - radii[i]
        if radii[i] > 1 - y_center - radii[i]:
            radii[i] = 1 - y_center - radii[i]

    # Validate and adjust radii to ensure no overlap
    for i in range(num_circles):
        x_center, y_center = centers[i]
        for j in range(i + 1, num_circles):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < (radii[i] + radii[j]):
                mid_dist = dist / 2
                radii[i] = mid_dist - radii[j]
                radii[j] = mid_dist - radii[i]

    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
