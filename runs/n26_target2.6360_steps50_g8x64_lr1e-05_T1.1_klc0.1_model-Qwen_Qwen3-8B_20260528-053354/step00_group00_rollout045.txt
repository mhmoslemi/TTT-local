import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    np.random.seed(42)
    for i in range(26):
        centers[i, 0] = np.random.uniform(0.1, 0.9)
        centers[i, 1] = np.random.uniform(0.1, 0.9)
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: circle must be inside the unit square
        # r_i <= x_i - r_i => x_i >= 2r_i, similarly for y_i
        in_box = np.all((centers[:, 0] - radii >= 0) & (centers[:, 1] - radii >= 0) &
                        (centers[:, 0] + radii <= 1) & (centers[:, 1] + radii <= 1))
        if not in_box:
            return np.inf
        # Constraints: circles must not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Initial radius guess
    initial_guess = np.full(26, 0.05)

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=objective,
            x0=initial_guess,
            method='SLSQP',
            constraints=[{'type': 'ineq', 'fun': constraints}],
            options={'ftol': 1e-8, 'maxiter': 500}
        )

    optimized_radii = result.x
    sum_radii = -result.fun

    # Return centers, radii, sum_radii
    return (centers, optimized_radii, sum_radii)
