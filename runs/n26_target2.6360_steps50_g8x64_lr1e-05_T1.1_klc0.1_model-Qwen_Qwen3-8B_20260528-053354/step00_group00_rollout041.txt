import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    
    # Place circles in a grid (rows and columns)
    for i in range(26):
        row = i // 5
        col = i % 5
        # Spread out the circles slightly to avoid initial overlap
        centers[i, 0] = col * 0.22 + 0.05 + np.random.uniform(-0.02, 0.02)
        centers[i, 1] = row * 0.22 + 0.05 + np.random.uniform(-0.02, 0.02)
        # Initial radius estimate based on spacing
        radii[i] = 0.04 + np.random.uniform(-0.01, 0.01)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(26):
            if (radii[i] > centers[i, 0]) or (radii[i] > 1 - centers[i, 0]) or \
               (radii[i] > centers[i, 1]) or (radii[i] > 1 - centers[i, 1]):
                return 1
        # Ensure no circles overlap
        for i in range(26):
            for j in range(i+1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return 1
        return 0
    
    # Optimization setup
    bounds = [(0.0, 0.2) for _ in range(26)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={"type": "ineq", "fun": constraints})
    
    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
