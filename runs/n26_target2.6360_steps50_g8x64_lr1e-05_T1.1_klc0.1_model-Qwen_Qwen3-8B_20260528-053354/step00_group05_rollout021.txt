import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: randomly placed circles with small radii
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.05)

    def objective_function(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:].reshape(26)
        return -np.sum(radii_flat)

    def constraint_function(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:].reshape(26)
        constraints = []

        # Constraint 1: All circles are inside the unit square
        for i in range(26):
            x_i, y_i, r_i = centers_flat[i][0], centers_flat[i][1], radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf  # Penalize if any circle is outside the square

        # Constraint 2: No overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                distance = np.sqrt(dx**2 + dy**2)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if distance < r_i + r_j - 1e-6:
                    return np.inf  # Penalize if circles overlap

        return 0

    # Initial guess
    x0 = np.concatenate([centers.reshape(1, -1)[0], radii])

    # Optimization settings
    bounds = []
    for _ in range(26):
        bounds += [(0, 1), (0, 1), (0, 1)]  # Bounds for x, y, r for each circle

    # Run optimization
    result = minimize(
        objective_function,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_function},
        tol=1e-6,
        options={'maxiter': 1000}
    )

    best_centers = np.reshape(result.x[:2*26], (26, 2))
    best_radii = np.reshape(result.x[2*26:], (26,))
    sum_radii = np.sum(best_radii)

    return (best_centers, best_radii, sum_radii)
