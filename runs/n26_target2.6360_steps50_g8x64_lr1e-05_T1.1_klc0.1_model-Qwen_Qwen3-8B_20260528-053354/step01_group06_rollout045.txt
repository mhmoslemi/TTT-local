import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    radii[0] = 0.1  # Start with a small radius for the first circle
    for i in range(1, 26):
        centers[i, 0] = (i % 5) / 4.0  # 5 columns
        centers[i, 1] = (i // 5) / 4.0  # 5 rows
        radii[i] = 0.1

    def objective(radii):
        return -sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Each circle is inside the unit square
        # r <= x - r <= 1 - r and r <= y - r <= 1 - r
        # So x - r >= r => x >= 2r and 1 - r >= x + r => x <= 1 - 2r
        # Similar for y
        # To ensure this, x - r >= r and 1 - r - x >= r, etc.
        for i in range(26):
            if centers[i, 0] - radii[i] < radii[i]:
                return False
            if centers[i, 0] + radii[i] > 1 - radii[i]:
                return False
            if centers[i, 1] - radii[i] < radii[i]:
                return False
            if centers[i, 1] + radii[i] > 1 - radii[i]:
                return False
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return False
        return True

    # Bounds for radii
    bounds = [(0.0, 0.5) for _ in range(26)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    centers_result = copy.deepcopy(centers)
    for i in range(26):
        if centers_result[i, 0] - result.x[i] < result.x[i]:
            centers_result[i, 0] = result.x[i] + result.x[i]
        if centers_result[i, 0] + result.x[i] > 1 - result.x[i]:
            centers_result[i, 0] = 1 - result.x[i] - result.x[i]
        if centers_result[i, 1] - result.x[i] < result.x[i]:
            centers_result[i, 1] = result.x[i] + result.x[i]
        if centers_result[i, 1] + result.x[i] > 1 - result.x[i]:
            centers_result[i, 1] = 1 - result.x[i] - result.x[i]

    return (np.array(centers_result), np.array(result.x), -result.fun)
