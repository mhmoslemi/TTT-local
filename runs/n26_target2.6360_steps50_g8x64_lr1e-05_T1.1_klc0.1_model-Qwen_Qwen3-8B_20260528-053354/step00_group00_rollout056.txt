import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: place circles in a grid with some initial radius
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.1)  # Initial radius
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    def objective_function(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.sum((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]) & (radii >= 0))

    def constraint_no_overlap(radii):
        dists = np.zeros(num_circles * (num_circles - 1) // 2)
        idx = 0
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                dists[idx] = dist - (radii[i] + radii[j])
                idx += 1
        return np.min(dists)

    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    bounds = [(0.0, 1.0 - 1e-6) for _ in range(num_circles)]

    result = minimize(objective_function, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    return (centers, result.x, -result.fun)
