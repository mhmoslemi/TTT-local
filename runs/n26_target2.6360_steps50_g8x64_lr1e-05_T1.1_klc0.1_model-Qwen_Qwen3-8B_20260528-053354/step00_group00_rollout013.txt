import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: uniformly distribute centers in the unit square
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Small initial radii

    # Objective function: minimize negative sum of radii to maximize it
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_center_inside(radii, centers):
        # Ensure each circle is within the square
        return np.sum((centers[:, 0] - radii) > 0) + np.sum((1 - centers[:, 0] - radii) > 0) + \
               np.sum((centers[:, 1] - radii) > 0) + np.sum((1 - centers[:, 1] - radii) > 0)

    def constraint_no_overlap(radii, centers):
        # Ensure circles do not overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization setup
    bounds = [(0, 1.0) for _ in range(26)]

    # Use L-BFGS-B optimizer
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=[{
            'type': 'ineq',
            'fun': lambda r: constraint_center_inside(r, centers),
        }, {
            'type': 'ineq',
            'fun': lambda r: constraint_no_overlap(r, centers),
        }],
        method='L-BFGS-B',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract result
    optimal_radii = result.x
    sum_radii = -result.fun

    # Ensure all circles are within the square
    for i in range(26):
        if optimal_radii[i] < 0:
            optimal_radii[i] = 0
        centers[i, 0] = np.clip(centers[i, 0], optimal_radii[i], 1 - optimal_radii[i])
        centers[i, 1] = np.clip(centers[i, 1], optimal_radii[i], 1 - optimal_radii[i])

    return (centers, optimal_radii, sum_radii)
