import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col + 0.1
        centers[i, 1] = row + 0.1
        radii[i] = 0.1

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_center_inside(radii):
        # Ensure each circle is fully inside the unit square
        distances_to_edges = np.min(np.array([
            centers[:, 0] - radii,
            1 - centers[:, 0] - radii,
            centers[:, 1] - radii,
            1 - centers[:, 1] - radii
        ]), axis=0)
        return distances_to_edges

    def constraint_no_overlap(radii):
        # Ensure no two circles overlap
        dists = np.zeros(26 * 26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[26 * i + j] = np.sqrt(dx**2 + dy**2)
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Build the optimization problem
    bounds = [(0.0, 1.0) for _ in range(26)]

    # Initial guess for radii
    initial_radius_guess = np.full(26, 0.1)

    # Optimization
    try:
        result = minimize(
            objective,
            initial_radius_guess,
            method='SLSQP',
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': lambda r: np.min(constraint_center_inside(r))},
                {'type': 'ineq', 'fun': lambda r: np.min(constraint_no_overlap(r))}
            ]
        )
    except:
        # Handle missing scipy.optimize
        print("Scipy.optimize not available.")
        sys.exit(1)

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
