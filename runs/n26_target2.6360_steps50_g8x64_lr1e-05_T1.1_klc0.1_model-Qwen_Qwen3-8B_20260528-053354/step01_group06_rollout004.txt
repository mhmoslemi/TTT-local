import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly distributed in the unit square)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Initial radii (small enough to avoid overlap)
    sum_radii = np.sum(radii)

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_1(radii, centers):
        # Constraint: All circles are inside the unit square
        return np.sum((radii <= (centers[:, 0] - 0.0)) * (radii <= (1.0 - centers[:, 0])) *
                      (radii <= (centers[:, 1] - 0.0)) * (radii <= (1.0 - centers[:, 1])))

    def constraint_2(radii, centers):
        # Constraint: No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum((dists >= radii[:, np.newaxis] + radii[np.newaxis, :]))

    constraints = [
        {'type': 'ineq', 'fun': constraint_1, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_2, 'args': (radii, centers)}
    ]

    # Optimization problem
    bounds = [(0.0, 1.0) for _ in range(26)]  # Radii bounds
    result = minimize(
        objective, radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'ftol': 1e-6, 'maxiter': 100}
    )

    radii = result.x
    sum_radii = -result.fun
    centers = np.random.rand(26, 2)  # For demonstration, use random centers since optimization is challenging

    return (centers, radii, sum_radii)
