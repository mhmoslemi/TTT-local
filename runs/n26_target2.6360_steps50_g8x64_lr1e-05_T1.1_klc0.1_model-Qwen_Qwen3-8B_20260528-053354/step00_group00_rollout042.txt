import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    initial_centers = np.random.rand(26, 2)
    initial_radii = np.random.rand(26)

    def objective_function(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative to maximize sum

    def constraints(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        constraints = []

        # Constraint 1: All circles are within the unit square
        for i in range(26):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i] or \
               centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return np.inf

        # Constraint 2: No overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j]:
                    return np.inf

        return 0

    bounds = []
    for _ in range(26):
        bounds.append((0.0, 1.0))  # x and y coordinates
    for _ in range(26):
        bounds.append((0.0, 1.0))  # radii

    result = minimize(objective_function, x0=np.concatenate([initial_centers.ravel(), initial_radii]),
                      method='SLSQP',
                      bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints},
                      tol=1e-8,
                      options={'maxiter': 1000})

    centers = result.x[:52].reshape(26, 2)
    radii = result.x[52:]
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
