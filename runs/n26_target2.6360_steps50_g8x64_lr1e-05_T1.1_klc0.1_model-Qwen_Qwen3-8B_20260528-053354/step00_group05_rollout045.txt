import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid points
    def initial_guess():
        centers = []
        for i in range(26):
            x = (i % 5) / 4.0 + 0.125
            y = (i // 5) / 4.0 + 0.125
            centers.append([x, y])
        return np.array(centers)

    def objective_function(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([(x + 0.125, y + 0.125) for x, y in zip(
            np.linspace(0.125, 0.875, 5), np.linspace(0.125, 0.875, 5)
        ) for _ in range(26)])
        centers[:26] = centers[:26] + (np.random.rand(26, 2) - 0.5) * 0.1
        radii = np.clip(radii, 0, 1.0)
        for i in range(26):
            x, y = centers[i]
            if x < radii[i] or x > 1 - radii[i] or y < radii[i] or y > 1 - radii[i]:
                return np.inf
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Initial guess for radii
    radii_initial = np.random.rand(26) * 0.1

    # Setup optimization problem
    result = minimize(
        fun=lambda x: constraints(x),
        x0=radii_initial,
        method='SLSQP',
        bounds=[(0.0, 1.0) for _ in range(26)]
    )

    # Generate centers based on optimized radii
    def generate_centers(radii):
        centers = []
        for i in range(26):
            x = np.random.rand() * 0.75 + 0.125
            y = np.random.rand() * 0.75 + 0.125
            while x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                x = np.random.rand() * 0.75 + 0.125
                y = np.random.rand() * 0.75 + 0.125
            centers.append([x, y])
        return np.array(centers)

    centers = generate_centers(result.x)
    sum_radii = np.sum(result.x)
    return (centers, result.x, sum_radii)
