import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)
    for i in range(n_circles):
        row = i // 5
        col = i % 5
        # Start with a radius of 0.1
        radii[i] = 0.1
        centers[i, 0] = col * 0.5 + 0.25
        centers[i, 1] = row * 0.5 + 0.25

    def objective(x):
        # x contains radii and centers in the order: r0, r1, ..., r25, x0, y0, x1, y1, ..., x25, y25
        r = x[:n_circles]
        centers_flat = x[n_circles:]
        centers = centers_flat.reshape((n_circles, 2))
        # Constraint: all circles are inside the unit square
        for i in range(n_circles):
            if not (r[i] <= centers[i, 0] <= 1 - r[i] and r[i] <= centers[i, 1] <= 1 - r[i]):
                return np.inf
        # Constraint: no two circles overlap
        dists = squareform(pdist(centers))
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                if dists[i, j] < r[i] + r[j]:
                    return np.inf
        # Objective: maximize sum of radii
        return -np.sum(r)

    def constraint_func(x):
        # Constraint: all radii are non-negative
        r = x[:n_circles]
        return r

    # Initial guess
    initial_guess = np.concatenate((radii, centers.flatten()))

    # Define constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_func}
    ]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, initial_guess, method='L-BFGS-B', constraints=constraints, tol=1e-8)

    # Extract result
    radii_opt = result.x[:n_circles]
    centers_opt = result.x[n_circles:].reshape((n_circles, 2))

    # Check if optimization was successful
    if result.success:
        sum_radii = np.sum(radii_opt)
    else:
        sum_radii = np.sum(radii)

    return (centers_opt, radii_opt, sum_radii)
