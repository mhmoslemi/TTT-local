import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for circle centers and radii
    centers = np.random.rand(26, 2)
    radii = np.zeros(26) + 0.05  # Initial radius for all circles

    # Objective function to maximize the sum of radii (minimize -sum_radii)
    def objective(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        return -np.sum(radii_flat)

    # Constraints
    constraints = []

    # Constraint: All circles are inside the unit square
    def inside_box(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        for i in range(26):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -1
        return 0
    constraints.append({'type': 'ineq', 'fun': inside_box})

    # Constraint: No two circles overlap
    def no_overlap(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        for i in range(26):
            for j in range(i+1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1
        return 0
    constraints.append({'type': 'ineq', 'fun': no_overlap})

    # Bounds for centers (x, y) in [0, 1], radii in [0, 0.5]
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])
    bounds.extend([(0.0, 0.5) for _ in range(26)])

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), 
                      method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract results
    centers_final = result.x[:2*26].reshape(26, 2)
    radii_final = result.x[2*26:]
    sum_radii = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii)
