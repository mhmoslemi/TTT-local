import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[x / 5 + 0.25, y / 5 + 0.25] for x in range(5) for y in range(5)])
    radii = np.full(25, 0.02)
    # Add one more circle in the center
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = np.append(radii, 0.02)

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_in_square(radii):
        # Check if centers are within [r, 1 - r] for each circle
        for i in range(len(radii)):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return -np.inf
        return 0

    def constraint_no_overlap(radii):
        # Check if all pairs of circles are non-overlapping
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    # Optimization problem setup
    bounds = [(0.0, 0.5) for _ in range(26)]  # Max radius is 0.5
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_in_square},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-6,
        options={'maxiter': 1000}
    )

    return (centers, result.x, -result.fun)
