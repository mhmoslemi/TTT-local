import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced points in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: small values
    radii = np.full(26, 0.05)

    def objective(r):
        return -np.sum(r)

    def constraints(r):
        # Constraints: each circle is inside the unit square
        for i in range(26):
            if r[i] > 1 - centers[i, 0] or r[i] > 1 - centers[i, 1]:
                return np.inf
        # Constraints: no overlap between circles
        dists = cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < r[i] + r[j]:
                    return np.inf
        return 0

    bounds = [(0, 1 - 0.01) for _ in range(26)]  # small buffer to avoid edge issues

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=({'type': 'ineq', 'fun': constraints}))

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
