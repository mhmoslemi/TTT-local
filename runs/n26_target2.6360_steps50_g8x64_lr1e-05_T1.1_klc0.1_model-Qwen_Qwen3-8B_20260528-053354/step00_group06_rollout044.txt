import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers: evenly spaced grid
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.01)  # Start with small radii
    centers = np.vstack([centers, [0.5, 0.5]])

    # Objective function: minimize negative sum of radii
    def objective(x):
        return -np.sum(radii)

    # Constraint: each circle is inside the unit square
    def constraint_inside(x):
        r = x[-26:]
        x_centers = x[:-26].reshape(26, 2)
        return np.minimum(r, 1 - r) - np.minimum(x_centers[:, 0], 1 - x_centers[:, 0]) - np.minimum(x_centers[:, 1], 1 - x_centers[:, 1])

    # Constraint: no overlap between circles
    def constraint_overlap(x):
        r = x[-26:]
        x_centers = x[:-26].reshape(26, 2)
        distances = np.sqrt(((x_centers[:, np.newaxis, :] - x_centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return distances - (r[np.newaxis, :] + r[:, np.newaxis])

    # Combine constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds for x and y coordinates (must be at least radius and at most 1-radius)
    bounds = []
    for i in range(26):
        bounds.extend([(r, 1 - r) for r in radii[i]] + [(0, 1) for _ in range(2)])

    # Combine all variables into a single array
    x0 = np.concatenate([centers.ravel(), radii])

    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    centers_opt = result.x[:-26].reshape(26, 2)
    radii_opt = result.x[-26:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
