import numpy as np
from scipy.optimize import minimize
from numpy.linalg import norm

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    radii = np.ones(25) * 0.1
    centers = np.vstack([centers, [0.5, 0.5]])  # Add one center at the middle
    radii = np.append(radii, 0.1)  # Initial radius for the center circle

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Enforce that all circles are within the unit square
        for i in range(26):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - np.min([centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1]]):
                return np.inf

        # Enforce no overlapping between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dist = norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Prepare bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(26)]

    # Run optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={"type": "ineq", "fun": constraints})

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
