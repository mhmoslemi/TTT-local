import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles at a grid with small radii
    num_circles = 26
    centers = np.array([(i / 6 + 0.5, j / 6 + 0.5) for i in range(6) for j in range(5)])
    radii = np.full(num_circles, 0.05)

    def objective_function(x):
        # x contains all radii in a flat array
        x = x.reshape((num_circles, 1))
        return -np.sum(x)  # Maximize sum of radii

    def constraints(x):
        x = x.reshape((num_circles, 1))
        centers_flat = np.array([centers[i] for i in range(num_circles)])
        # Constraint 1: All circles are inside the unit square
        constraints = []
        for i in range(num_circles):
            x_i = x[i]
            if x_i <= 0:
                continue
            constraints.append((centers_flat[i, 0] - x_i) >= 0)
            constraints.append((1 - centers_flat[i, 0] - x_i) >= 0)
            constraints.append((centers_flat[i, 1] - x_i) >= 0)
            constraints.append((1 - centers_flat[i, 1] - x_i) >= 0)
        # Constraint 2: No two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                r_i = x[i]
                r_j = x[j]
                constraints.append(dist >= r_i + r_j)
        return constraints

    # Prepare bounds for optimization
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Optimization
    result = minimize(
        fun=objective_function,
        x0=radii.flatten(),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-5,
        options={'maxiter': 1000}
    )

    if not result.success:
        return (centers, np.zeros_like(radii), 0.0)

    optimized_radii = result.x
    optimized_radii = optimized_radii.reshape((num_circles,))
    total_sum = np.sum(optimized_radii)

    # Check if circles are inside the square and no overlap (optional, for safety)
    centers_flat = centers
    for i in range(num_circles):
        x_i = optimized_radii[i]
        if x_i <= 0:
            continue
        if not (x_i <= centers_flat[i, 0] <= 1 - x_i and x_i <= centers_flat[i, 1] <= 1 - x_i):
            return (centers, np.zeros_like(radii), 0.0)
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = centers_flat[i, 0] - centers_flat[j, 0]
            dy = centers_flat[i, 1] - centers_flat[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < optimized_radii[i] + optimized_radii[j]:
                return (centers, np.zeros_like(radii), 0.0)

    return (centers, optimized_radii, total_sum)
