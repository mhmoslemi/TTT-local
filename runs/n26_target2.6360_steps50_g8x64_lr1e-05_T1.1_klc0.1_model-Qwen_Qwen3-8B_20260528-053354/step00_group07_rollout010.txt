import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.random.rand(num_circles, 2) * 0.8 + 0.1  # avoid edges
    radii = np.ones(num_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii

    def constraints(radii):
        # Ensure all circles are inside the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1)] * num_circles,
        constraints=[{'type': 'ineq', 'fun': constraints}],
        tol=1e-8
    )

    # Extract results
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
