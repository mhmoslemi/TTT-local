        for i in range(26):

                dx = centers[i, 0] - centers[j, 0]

                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:

    # Optimize
    result = minimize(objective, x0, constraints=constraints, method='SLSQP', bounds=[(0, 1) for _ in range(52)] + [(0, 1) for _ in range(26)])

    # Extract final solution
    centers = result.x[:52].reshape(26, 2)
