import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: spread circles evenly with small radii
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.05)

    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(x):
        return -np.sum(x[26:])

    # Bounds: x[0:26] are centers (0 to 1), x[26:] are radii (0 to 0.5)
    bounds = [(0, 1) for _ in range(26)] + [(0, 0.5) for _ in range(26)]

    # Constraints
    def constraint1(x):
        # Check if all circles are within the unit square
        centers = x[:26].reshape(26, 2)
        radii = x[26:]
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0]) + \
               np.minimum(centers[:, 1] - radii, 1 - centers[:, 1])

    def constraint2(x):
        # Check if no two circles overlap
        centers = x[:26].reshape(26, 2)
        radii = x[26:]
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Optimization
    result = minimize(objective, x0=np.concatenate((centers.flatten(), radii)), bounds=bounds, constraints=constraints)

    # Extract result
    centers_result = result.x[:26].reshape(26, 2)
    radii_result = result.x[26:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
