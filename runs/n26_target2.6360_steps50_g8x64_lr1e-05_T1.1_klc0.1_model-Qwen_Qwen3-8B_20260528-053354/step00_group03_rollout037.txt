import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 26 circles equally spaced
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]

    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x = i % 5 / 4
            y = i // 5 / 4
            centers[i] = [x, y]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        for i in range(26):
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return np.inf
        return 0

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=[('ineq', constraints, 0)])

    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]
    return (centers, result.x, -result.fun)
