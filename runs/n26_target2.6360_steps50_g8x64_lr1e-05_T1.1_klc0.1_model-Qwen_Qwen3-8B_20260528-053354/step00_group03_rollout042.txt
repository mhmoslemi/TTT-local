import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced grid with some spacing
    centers = np.zeros((26, 2))
    for i in range(26):
        x = (i % 5) / 4.0 + 0.25
        y = (i // 5) / 4.0 + 0.25
        centers[i] = [x, y]

    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: All circles are within the unit square
        for r in radii:
            if r < 0:
                return False
            if r > 1:
                return False
        # Constraint 2: No overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Use a penalty method to handle constraints
    def penalty_function(x):
        r = x.copy()
        # Constraint 1: All circles are within the unit square
        for i in range(26):
            r[i] = max(0, min(r[i], 1))
        # Constraint 2: No overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    penalty = 100 * (r[i] + r[j] - dist)
                    r[i] -= penalty / 2
                    r[j] -= penalty / 2
        return -np.sum(r)

    # Optimization with constraints
    result = minimize(penalty_function, radii, method='L-BFGS-B', bounds=[(0, 1)]*26)
    radii_opt = result.x
    sum_radii = -result.fun

    return (centers, radii_opt, sum_radii)
