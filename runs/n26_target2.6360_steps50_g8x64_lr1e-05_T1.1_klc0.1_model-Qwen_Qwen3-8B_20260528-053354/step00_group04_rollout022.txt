import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid points with small radii
    num_circles = 26
    centers = np.array([(i / (num_circles - 1), j / (num_circles - 1)) 
                       for i in range(num_circles) for j in range(num_circles)])
    centers = centers[:num_circles]  # Ensure exactly 26 circles
    radii = np.full(num_circles, 0.01)  # Initial radius

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum

    # Constraint: each circle is inside the unit square
    def in_square(radii):
        return np.sum((radii <= (1 - centers[:, 0])) * (radii <= (1 - centers[:, 1])) * (radii >= centers[:, 0]) * (radii >= centers[:, 1])) == num_circles

    # Constraint: no overlap between circles
    def no_overlap(radii):
        dists = np.abs(np.expand_dims(centers, axis=1) - np.expand_dims(centers, axis=0))
        dists = np.sqrt(np.sum(dists**2, axis=2))
        return np.all(dists >= radii[np.newaxis, :] + radii[:, np.newaxis])

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': lambda r: r},  # radii >= 0
        {'type': 'ineq', 'fun': in_square},
        {'type': 'ineq', 'fun': no_overlap},
    ]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=constraints, bounds=[(0, 1)] * num_circles,
                      options={'maxiter': 1000, 'disp': False, 'ftol': 1e-6})

    # Extract solution
    centers = centers[:num_circles]
    radii = result.x
    sum_radii = np.sum(radii)

    # Ensure all constraints are satisfied
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        if not no_overlap(radii):
            raise RuntimeError("Constraint violation: circles overlap")
        if not np.all(radii >= 0):
            raise RuntimeError("Constraint violation: negative radii")
        if not np.all((radii <= (1 - centers[:, 0])) & (radii <= (1 - centers[:, 1])) & (radii >= centers[:, 0]) & (radii >= centers[:, 1])):
            raise RuntimeError("Constraint violation: circle out of bounds")

    return (centers, radii, sum_radii)
