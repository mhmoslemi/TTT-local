import numpy as np
from scipy.optimize import minimize
import numpy.random as npr

def run_packing():
    # Initial guess: place circles in a grid with random offsets
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.1)
    
    # Objective function: maximize the sum of radii
    def objective(x):
        r = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)
        return -np.sum(r)
    
    # Constraints: 
    # 1. Each circle must be inside the unit square
    # 2. No two circles can overlap
    # 3. Radii must be non-negative
    
    def constraint_inside(x):
        r = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)
        return np.hstack([
            (centers[:, 0] - r) - 0,  # x - r >= 0
            (centers[:, 1] - r) - 0,  # y - r >= 0
            (1 - centers[:, 0] - r) - 0,  # 1 - x - r >= 0
            (1 - centers[:, 1] - r) - 0   # 1 - y - r >= 0
        ])
    
    def constraint_overlap(x):
        r = x[:num_circles]
        centers = x[num_circles:].reshape(num_circles, 2)
        dists = np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2)
        r_sum = r[:, np.newaxis] + r[np.newaxis, :]
        return np.sqrt(dists) - r_sum
    
    # Bounds: radii >= 0 and centers in [0, 1]
    bounds = [(0, np.inf)] * num_circles + [(0, 1)] * (2 * num_circles)
    
    # Initial guess
    x0 = np.concatenate([radii, centers.flatten()])
    
    # Set up constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Optimization
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons
    )
    
    # Extract results
    radii_opt = result.x[:num_circles]
    centers_opt = result.x[num_circles:].reshape(num_circles, 2)
    sum_radii = np.sum(radii_opt)
    
    return (centers_opt, radii_opt, sum_radii)
