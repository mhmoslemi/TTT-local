import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import math

def run_packing():
    # Initial guess: 26 circles placed in a grid with small radii
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.01)
    sum_radii = 0.26

    # Objective function to maximize the sum of radii (minimize negative sum)
    def objective(x):
        nonlocal sum_radii
        centers[:26] = x.reshape(26, 2)
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints: 1. Each circle is inside the unit square
    #              2. No two circles overlap (distance >= sum of radii)
    #              3. Radii are non-negative
    def constraint_inside(x):
        centers[:26] = x.reshape(26, 2)
        return np.sum((centers[:, 0] - radii) >= 0) + np.sum((1 - centers[:, 0] - radii) >= 0) + \
               np.sum((centers[:, 1] - radii) >= 0) + np.sum((1 - centers[:, 1] - radii) >= 0)

    def constraint_no_overlap(x):
        centers[:26] = x.reshape(26, 2)
        distances = np.zeros((26, 26))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distances[i, j] = np.sqrt(dx**2 + dy**2)
        return np.sum(distances >= (radii[i] + radii[j] for i, j in zip(range(26), range(26))))

    # Initial guess in flat format
    x0 = np.random.rand(26 * 2) * 0.8 + 0.1  # Random initial positions within the square

    # Bounds for centers (x, y) must be in [r, 1 - r], radii must be non-negative
    bounds = [(r, 1 - r) for _ in range(26 * 2)]
    for i in range(26):
        bounds[i*2] = (radii[i], 1 - radii[i])
        bounds[i*2 + 1] = (radii[i], 1 - radii[i])

    # Optimization
    result = minimize(
        fun=objective,
        x0=x0,
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}
        ],
        method='SLSQP'
    )

    centers_opt = result.x.reshape(26, 2)
    sum_radii = np.sum(radii)
    return (centers_opt, radii, sum_radii)
