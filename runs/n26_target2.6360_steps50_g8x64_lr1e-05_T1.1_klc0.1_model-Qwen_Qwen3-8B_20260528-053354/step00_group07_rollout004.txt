import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: evenly spaced grid in the unit square
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all set to 0.1 as a starting point
    radii = np.full(n_circles, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[26:])

    # Bounds: each center (x, y) must be in [r, 1-r], and r >= 0
    def bounds_func(x):
        # First 26 values are centers (x, y)
        # Next 26 values are radii
        centers = x[:26].reshape(n_circles, 2)
        radii = x[26:]
        bounds = []
        for i in range(n_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            bounds.append((r_i, 1 - r_i))
            bounds.append((r_i, 1 - r_i))
        bounds.extend([(0, 1 - r) for r in radii])
        return bounds

    # Constraints: distance between centers >= sum of radii
    def constraints_func(x):
        centers = x[:26].reshape(n_circles, 2)
        radii = x[26:]
        constraints = []
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        return constraints

    # Initial guess for centers and radii
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization options
    options = {
        'ftol': 1e-8,
        'maxiter': 1000,
        'disp': False
    }

    # Optimization
    result = minimize(objective, initial_guess, method='SLSQP',
                      bounds=Bounds(0, 1), constraints={'type': 'ineq', 'fun': constraints_func},
                      options=options)

    # Extract result
    centers_result = result.x[:26].reshape(n_circles, 2)
    radii_result = result.x[26:]
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
