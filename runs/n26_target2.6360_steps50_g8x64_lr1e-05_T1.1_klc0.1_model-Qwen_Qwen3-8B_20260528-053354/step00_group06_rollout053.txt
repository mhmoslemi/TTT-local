import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = generate_centers(radii)
        # Ensure circles are inside the unit square
        for i in range(26):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return np.inf
        # Ensure no circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    def generate_centers(radii):
        centers = []
        for _ in range(26):
            while True:
                x = np.random.uniform(radii[-1], 1 - radii[-1])
                y = np.random.uniform(radii[-1], 1 - radii[-1])
                overlap = False
                for i in range(len(centers)):
                    dx = x - centers[i][0]
                    dy = y - centers[i][1]
                    dist = np.sqrt(dx**2 + dy**2)
                    if dist < radii[-1] + radii[i]:
                        overlap = True
                        break
                if not overlap:
                    centers.append((x, y))
                    break
        return np.array(centers)

    initial_guess = np.random.random(26)
    initial_centers = generate_centers(initial_guess)

    bounds = [(0, 1) for _ in range(26)]

    # Optimization
    result = minimize(
        objective,
        initial_guess,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP"
    )

    centers = generate_centers(result.x)
    radii = result.x
    sum_radii = result.fun

    return (centers, radii, sum_radii)
