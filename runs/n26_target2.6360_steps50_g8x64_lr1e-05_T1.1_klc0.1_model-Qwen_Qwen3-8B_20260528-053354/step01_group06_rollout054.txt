import numpy as np
from scipy.optimize import minimize
import os

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial positions in a grid with spacing
    grid_size = 2
    for i in range(num_circles):
        row = i // grid_size
        col = i % grid_size
        centers[i, 0] = col * 0.4 + 0.1
        centers[i, 1] = row * 0.4 + 0.1
        radii[i] = 0.1

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure circles are within bounds
        r = radii.copy()
        for i in range(num_circles):
            if r[i] < 0:
                return np.inf
            if centers[i, 0] - r[i] < 0 or centers[i, 0] + r[i] > 1:
                return np.inf
            if centers[i, 1] - r[i] < 0 or centers[i, 1] + r[i] > 1:
                return np.inf

        # Ensure circles are non-overlapping
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r[i] + r[j]:
                    return np.inf
        return 0

    def objective_func(x):
        radii[:] = x
        return -np.sum(x)

    def constraint_func(x):
        radii[:] = x
        return constraints(x)

    # Set up the optimization problem
    bounds = [(0, 0.5) for _ in range(num_circles)]
    cons = {'type': 'ineq', 'fun': constraint_func}
    res = minimize(objective_func, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-5)

    # Extract results
    radii_opt = res.x
    centers_opt = centers.copy()

    # Ensure circles are within bounds and correct radii
    for i in range(num_circles):
        r = radii_opt[i]
        centers_opt[i, 0] = np.clip(centers_opt[i, 0], r, 1 - r)
        centers_opt[i, 1] = np.clip(centers_opt[i, 1], r, 1 - r)

    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
