import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.array([[x*0.1 + 0.05, y*0.1 + 0.05] for y in range(26) for x in range(13)])
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: Each circle must be fully inside the square
        # r_i <= x_i - r_i and r_i <= y_i - r_i => x_i + r_i <= 1 and y_i + r_i <= 1
        for i in range(26):
            if (centers[i, 0] + radii[i] > 1) or (centers[i, 1] + radii[i] > 1):
                return False
        # Constraint 2: No overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization problem
    def wrapper(radii):
        if not constraints(radii):
            return float('inf')
        return -np.sum(radii)

    # Optimization bounds
    bounds = [(0.0, 0.5) for _ in range(26)]

    result = minimize(wrapper, radii, bounds=bounds, method='L-BFGS-B', tol=1e-8)

    # Extract result
    optimal_radii = result.x
    optimal_centers = copy.deepcopy(centers)
    sum_radii = np.sum(optimal_radii)

    return (optimal_centers, optimal_radii, sum_radii)
