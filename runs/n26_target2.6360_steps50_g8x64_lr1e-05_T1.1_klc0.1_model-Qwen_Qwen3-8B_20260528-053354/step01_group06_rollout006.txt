import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: equally spaced grid of 26 points in the unit square
    num_circles = 26
    centers = np.indices((5, 5)).reshape(25, 2) / 4
    centers = np.vstack((centers, [0.5, 0.5]))
    radii = np.ones(num_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure each circle is inside the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if radii[i] > 1:
                return False
        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization with constraints
    def constraint_func(radii_vec):
        radii = radii_vec.reshape(num_circles)
        if not constraints(radii):
            return np.inf
        return 0

    # Initial guess
    x0 = radii.copy()

    # Bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimization options
    options = {'ftol': 1e-6, 'maxiter': 1000}

    result = minimize(objective, x0, bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_func}, options=options)

    # Extract result
    radii = result.x
    sum_radii = np.sum(radii)
    centers = centers.reshape(num_circles, 2)

    return (centers, radii, sum_radii)
