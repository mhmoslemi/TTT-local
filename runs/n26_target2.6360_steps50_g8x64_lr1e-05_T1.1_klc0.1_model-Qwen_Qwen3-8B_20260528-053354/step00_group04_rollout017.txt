import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i % 5 + 0.1  # 5 columns with spacing
        y = i // 5 + 0.1  # 5 rows with spacing
        centers[i] = [x, y]
    
    # Initial radii: small values
    radii = np.ones(num_circles) * 0.05

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # minimize the negative sum

    # Bounds: radii >= 0, and each circle must be within the square
    bounds = [(0, 1 - 0.05) for _ in range(num_circles)]  # 0.05 is initial radii

    # Constraints: no two circles can overlap
    def constraint_overlap(x):
        r = x
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return dist - r[i] - r[j]
        return 0.0

    # Constraint: circles must be fully inside the square
    def constraint_inside(x):
        r = x
        for i in range(num_circles):
            if not (r[i] <= centers[i, 0] <= 1 - r[i] and r[i] <= centers[i, 1] <= 1 - r[i]):
                return -1.0
        return 0.0

    # Set up constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_inside}
    ]

    # Optimize
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
