import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    for i in range(n):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0
        radii[i] = 0.1

    # Objective function to maximize sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints: circle inside the square and non-overlapping
    def constraint_inside(radii_vec):
        radii = radii_vec
        for i in range(n):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return -1
        return 0

    def constraint_overlap(radii_vec):
        radii = radii_vec
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0

    # Optimization
    def constraint_func(x):
        return constraint_inside(x), constraint_overlap(x)

    bounds = [(0.0, 0.5) for _ in range(n)]
    constraints = [{'type': 'ineq', 'fun': lambda x: constraint_func(x)[0]},
                   {'type': 'ineq', 'fun': lambda x: constraint_func(x)[1]}]

    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-6)

    centers_final = np.copy(centers)
    radii_final = result.x
    sum_radii_final = np.sum(radii_final)

    # Ensure all circles are within the square
    for i in range(n):
        x, y = centers_final[i]
        if x - radii_final[i] < 0 or x + radii_final[i] > 1 or y - radii_final[i] < 0 or y + radii_final[i] > 1:
            radii_final[i] = min(radii_final[i], 1 - np.min([x, y]))

    return (centers_final, radii_final, sum_radii_final)
