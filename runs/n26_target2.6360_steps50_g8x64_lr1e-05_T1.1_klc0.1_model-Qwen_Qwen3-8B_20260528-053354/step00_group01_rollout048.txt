import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for circle centers and radii
    # Place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = col + 0.2  # Add some spacing
        y = row + 0.2
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint: Each circle must be inside the unit square
        # x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1
        # We use the centers from the initial guess as a starting point
        centers = np.zeros((26, 2))
        for i in range(26):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return False
        return True

    def overlap(radii):
        # Check for overlap between circles
        centers = np.zeros((26, 2))
        for i in range(26):
            x, y = centers[i]
            for j in range(i + 1, 26):
                dx = centers[j, 0] - centers[i, 0]
                dy = centers[j, 1] - centers[i, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define the optimization problem
    bounds = [(0.0, 0.5) for _ in range(26)]  # Radii cannot exceed 0.5

    def objective_function(radii):
        return -np.sum(radii)

    def constraint_func(radii):
        # All circles must fit in the unit square
        for i in range(26):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return -1
        # No two circles may overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[j, 0] - centers[i, 0]
                dy = centers[j, 1] - centers[i, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -1
        return 0

    # Set up the optimization
    result = minimize(
        fun=objective_function,
        x0=radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraint_func},
        method="SLSQP"
    )

    if result.success:
        final_radii = result.x
        final_centers = centers
        sum_radii = np.sum(final_radii)
        return (final_centers, final_radii, sum_radii)
    else:
        warnings.warn("Optimization did not converge.")
        return (centers, radii, np.sum(radii))
