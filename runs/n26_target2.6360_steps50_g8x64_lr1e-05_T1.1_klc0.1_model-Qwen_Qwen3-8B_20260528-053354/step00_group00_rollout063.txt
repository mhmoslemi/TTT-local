import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: random positions with small radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: circles must be inside the unit square
    def constraint_inside(radii):
        centers_inside = np.all((centers[:, 0] >= radii) & (centers[:, 0] <= 1 - radii) &
                               (centers[:, 1] >= radii) & (centers[:, 1] <= 1 - radii), axis=1)
        return np.sum(centers_inside)

    # Constraint 2: circles must not overlap
    def constraint_overlap(radii):
        distances = np.zeros(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distances[i] += np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                distances[j] += np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
        return np.min(distances)

    # Combine constraints into a constraint function
    def constraint(x):
        # x is a vector of radii
        # Check inside constraint
        inside = np.all((centers[:, 0] >= x) & (centers[:, 0] <= 1 - x) &
                       (centers[:, 1] >= x) & (centers[:, 1] <= 1 - x), axis=1)
        if not np.all(inside):
            return -np.inf  # Penalize invalid positions
        # Check overlap constraint
        distances = np.zeros(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dist -= (x[i] + x[j])
                distances[i] += dist
                distances[j] += dist
        if np.min(distances) < 0:
            return -np.inf  # Penalize overlaps
        return 0  # No constraint violation

    # Optimization bounds: radii must be non-negative
    bounds = [(0, 0.5) for _ in range(26)]

    # Optimization options
    options = {
        'disp': False,
        'ftol': 1e-8,
        'maxiter': 1000,
        'method': 'SLSQP'
    }

    # Optimization with constraints
    result = minimize(objective, radii, bounds=bounds, constraints=[{'type': 'ineq', 'fun': constraint}], options=options)

    # Extract the optimized radii and centers
    optimized_radii = result.x
    sum_radii = np.sum(optimized_radii)

    # Return the result
    return (centers, optimized_radii, sum_radii)
