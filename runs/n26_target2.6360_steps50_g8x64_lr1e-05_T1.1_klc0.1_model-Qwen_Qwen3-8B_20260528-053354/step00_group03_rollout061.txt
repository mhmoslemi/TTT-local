import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.array([[x / 7, y / 7] for y in range(7) for x in range(7)])
    centers = centers[:num_circles]
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_overlap(i, j, radii):
        def constraint(x):
            x = x.reshape(-1)
            r_i, r_j = radii[i], radii[j]
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            return dist - (r_i + r_j)
        return constraint

    def constraint_boundary(i, radii):
        def constraint(x):
            r = radii[i]
            x_idx = np.where(radii == r)[0][0]
            x_val = x[x_idx]
            return np.min([x_val - r, 1 - x_val - r])
        return constraint

    # Add constraints
    cons = []

    for i in range(num_circles):
        cons.append({'type': 'ineq', 'fun': constraint_boundary(i, radii)})

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            cons.append({'type': 'ineq', 'fun': constraint_overlap(i, j, radii)})

    # Setup optimization
    bounds = [(0, 1) for _ in range(num_circles)]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    final_radii = result.x
    final_centers = centers
    sum_radii = -result.fun

    return (final_centers, final_radii, sum_radii)
