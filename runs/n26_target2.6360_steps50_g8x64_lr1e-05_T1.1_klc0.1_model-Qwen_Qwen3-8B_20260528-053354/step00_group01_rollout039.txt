import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place 26 circles in a grid
    centers = np.array([[x / 7 + 0.1, y / 7 + 0.1] for y in range(7) for x in range(7)])
    radii = np.full(26, 0.05)
    
    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraint: circles must be inside the unit square
    def in_square(radii, centers):
        return np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]) &
                      (radii >= centers[:, 0]) & (radii >= centers[:, 1]))
    
    # Constraint: circles must not overlap
    def no_overlap(radii, centers):
        n = len(radii)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Combined constraints
    def constraints(radii):
        if not in_square(radii, centers):
            return np.inf
        if not no_overlap(radii, centers):
            return np.inf
        return 0
    
    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints={'type': 'eq', 'fun': lambda r: constraints(r)},
        bounds=[(0, 1 - 0.1) for _ in range(26)]
    )
    
    # Extract results
    centers = np.array([[x / 7 + 0.1, y / 7 + 0.1] for y in range(7) for x in range(7)])
    radii = result.x
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
