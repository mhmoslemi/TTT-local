import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: 26 equally spaced circles in a grid-like pattern
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    centers = centers[:26]
    radii = np.full(26, 0.05)

    def objective(radii_flat):
        return -np.sum(radii_flat)

    def constraint_center_inside(radii_flat):
        r = radii_flat.reshape(-1, 1)
        centers_r = np.vstack([centers[i] for i in range(26)])
        return (centers_r - r)[:, 0] >= 0
        # return (centers_r - r)[:, 0] >= 0 and (1 - centers_r - r)[:, 0] >= 0

    def constraint_overlap(radii_flat):
        r = radii_flat.reshape(-1, 1)
        d = np.sum((np.vstack([centers[i] for i in range(26)]) - np.vstack([centers[j] for j in range(26)]))**2, axis=2)
        r_sum = r[:, np.newaxis] + r[np.newaxis, :]
        return d >= r_sum

    def pack_circles():
        # Initial guess for radii
        initial_guess = np.full(26, 0.05)

        # Define constraints
        constraints = []

        # Constraint: centers must be within the square [r, 1 - r]
        for i in range(26):
            def center_inside_i(radii_flat):
                r = radii_flat[i]
                return [centers[i][0] - r >= 0, 1 - centers[i][0] - r >= 0,
                        centers[i][1] - r >= 0, 1 - centers[i][1] - r >= 0]
            constraints.append({'type': 'ineq', 'fun': center_inside_i})

        # Constraint: no circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                def overlap_ij(radii_flat):
                    r_i = radii_flat[i]
                    r_j = radii_flat[j]
                    dx = centers[i][0] - centers[j][0]
                    dy = centers[i][1] - centers[j][1]
                    distance = np.sqrt(dx*dx + dy*dy)
                    return distance - (r_i + r_j)
                constraints.append({'type': 'ineq', 'fun': overlap_ij})

        # Optimization
        result = minimize(
            fun=objective,
            x0=initial_guess,
            method='SLSQP',
            constraints=constraints,
            bounds=[(0, 1.0) for _ in range(26)]
        )

        # Extract result
        radii = result.x
        sum_radii = -result.fun
        centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])[:26]
        return centers, radii, sum_radii

    try:
        from scipy.optimize import minimize
        return pack_circles()
    except ImportError:
        print("scipy.optimize not available, cannot run optimization.")
        return np.array([]), np.array([]), 0.0
