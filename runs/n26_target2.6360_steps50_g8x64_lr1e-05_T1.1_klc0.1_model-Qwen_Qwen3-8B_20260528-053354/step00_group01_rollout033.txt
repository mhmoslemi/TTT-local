import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 26 circles arranged in a grid pattern
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]
    radii = np.ones(26) * 0.1  # Initial radius

    def objective(radii):
        return -np.sum(radii)

    def constraint1(radii):
        # Each circle must be fully inside the unit square
        r = radii
        return np.minimum(r, 1 - r)  # Ensure r <= 1 - r for all

    def constraint2(radii):
        # No two circles overlap
        centers = np.array([np.random.rand(2) for _ in range(26)])
        dists = np.linalg.norm(centers[:, np.newaxis] - centers[np.newaxis, :], axis=-1)
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Define optimization problem
    bounds = [(0, 1) for _ in range(26)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint1(r)},
        {'type': 'ineq', 'fun': lambda r: constraint2(r)}
    ]

    res = minimize(objective, radii, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-5)

    centers = np.array([np.random.rand(2) for _ in range(26)])
    sum_radii = res.fun
    return (centers, -res.success * np.ones(26), sum_radii)
