import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with some random perturbation
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5.0 + np.random.uniform(-0.1, 0.1)
        centers[i, 1] = (i // 5 + 0.5) / 5.0 + np.random.uniform(-0.1, 0.1)

    # Initial radii: same for all, as a starting point
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = (i % 5 + 0.5) / 5.0
            centers[i, 1] = (i // 5 + 0.5) / 5.0

        # Constraint 1: Each circle is inside the unit square
        for i in range(26):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return np.inf
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return np.inf

        # Constraint 2: No overlaps between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf

        return 0

    # Optimization
    result = minimize(
        objective,
        radii,
        method='L-BFGS-B',
        bounds=[(0, 1) for _ in range(26)],
        constraints={'type': 'ineq', 'fun': constraints},
        options={'maxiter': 2000, 'ftol': 1e-6}
    )

    # Final centers and radii
    final_centers = np.zeros((26, 2))
    for i in range(26):
        final_centers[i, 0] = (i % 5 + 0.5) / 5.0
        final_centers[i, 1] = (i // 5 + 0.5) / 5.0

    return (final_centers, result.x, np.sum(result.x))
