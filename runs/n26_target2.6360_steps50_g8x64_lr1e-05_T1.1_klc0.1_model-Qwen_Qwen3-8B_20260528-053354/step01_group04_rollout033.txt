import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    for i in range(n):
        row = i // 5
        col = i % 5
        x = 0.2 + col * 0.4
        y = 0.2 + row * 0.4
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective_func(x):
        radii_vals = x[:n]
        centers_vals = x[n:].reshape((n, 2))
        # Check bounds
        for i in range(n):
            r = radii_vals[i]
            x_c, y_c = centers_vals[i]
            if not (r <= x_c <= 1 - r and r <= y_c <= 1 - r):
                return float('inf')
        # Check distances
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_vals[i, 0] - centers_vals[j, 0]
                dy = centers_vals[i, 1] - centers_vals[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii_vals[i] + radii_vals[j] - 1e-8:
                    return float('inf')
        return -np.sum(radii_vals)

    bounds = [(0, 0.5)] * n + [(0.0, 1.0)] * (2 * n)  # Radii and centers

    result = minimize(
        objective_func,
        x0=np.concatenate((radii, centers.flatten())),
        method='L-BFGS-B',
        bounds=bounds,
        tol=1e-6,
        options={'disp': False}
    )

    optimized_radii = result.x[:n]
    optimized_centers = result.x[n:].reshape((n, 2))

    # Check constraints after optimization
    for i in range(n):
        x_c, y_c = optimized_centers[i]
        r = optimized_radii[i]
        if not (r <= x_c <= 1 - r and r <= y_c <= 1 - r):
            optimized_radii[i] = 0.0
    for i in range(n):
        for j in range(i + 1, n):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            distance = math.hypot(dx, dy)
            if distance < optimized_radii[i] + optimized_radii[j] - 1e-8:
                optimized_radii[i] = 0.0

    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
