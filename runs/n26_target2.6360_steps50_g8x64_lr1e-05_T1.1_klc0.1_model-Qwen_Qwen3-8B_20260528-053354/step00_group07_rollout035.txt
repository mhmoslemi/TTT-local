import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    def initial_guess():
        centers = []
        radii = []
        for i in range(26):
            x = i / 5
            y = (i % 5) / 5
            centers.append([x, y])
            radii.append(0.05)
        return np.array(centers), np.array(radii)

    centers, radii = initial_guess()
    n = len(radii)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: circles are inside the square
        for i in range(n):
            if radii[i] > 0:
                if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                    return False
                if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                    return False

        # Constraints: circles do not overlap
        dist = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        if np.any(dist < radii_sum - 1e-6):
            return False

        return True

    def bounds(radii):
        return [(0, 1) for _ in range(n)]

    def objective_with_constraints(radii):
        if not constraints(radii):
            return np.inf
        return -np.sum(radii)

    result = minimize(objective_with_constraints, radii, method='SLSQP', bounds=bounds)
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii = -result.fun
    return (centers_final, radii_final, sum_radii)
