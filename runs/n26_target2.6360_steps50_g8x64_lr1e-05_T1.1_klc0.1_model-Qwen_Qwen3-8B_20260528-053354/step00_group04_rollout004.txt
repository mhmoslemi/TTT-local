import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers: equally spaced grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    
    # Initial guess for radii
    radii = np.zeros(26) + 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: Each circle is inside the unit square
        # Ensure that 0 <= x - r <= 1 - r and 0 <= y - r <= 1 - r
        inside = np.all((radii <= centers[:, 0] - 0.01) * (radii <= 1 - centers[:, 0] - 0.01), axis=1)
        if not np.all(inside):
            return np.inf  # Penalize for circles outside the square

        # Constraint 2: No two circles overlap
        dists = np.zeros((26, 26))
        for i in range(26):
            for j in range(i+1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)
                dists[j, i] = dists[i, j]
        overlaps = np.any(dists < radii[:, np.newaxis] + radii[np.newaxis, :])
        if overlaps:
            return np.inf  # Penalize for overlapping circles

        return 0  # No constraints violated

    # Define bounds on radii: non-negative
    bounds = [(0.0, 0.2) for _ in range(26)]  # Upper bound of 0.2 as a safe estimate

    # Use a constrained optimization solver
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Extract final solution
    final_radii = result.x
    final_centers = centers.copy()
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
