import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1  # Initial small radii

    # Define the objective function: maximize the sum of radii (we minimize the negative)
    def objective(radii):
        return -np.sum(radii)

    # Define constraint 1: all circles must be inside the unit square
    def inside_box(x):
        return np.array([x[i, 0] - radii[i] >= 0 and x[i, 1] - radii[i] >= 0 and
                         1 - x[i, 0] - radii[i] >= 0 and 1 - x[i, 1] - radii[i] >= 0
                         for i in range(num_circles)])

    # Define constraint 2: circles do not overlap
    def no_overlap(x):
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = x[i, 0] - x[j, 0]
                dy = x[i, 1] - x[j, 1]
                dists[i, j] = np.sqrt(dx * dx + dy * dy)
                dists[j, i] = dists[i, j]
        return np.array([dists[i, j] >= radii[i] + radii[j] for i in range(num_circles) for j in range(num_circles)])

    # Define the bounds for centers and radii
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))  # for x-coordinate
        bounds.append((0, 1))  # for y-coordinate
        bounds.append((0, 1))  # for radius

    # Set up the constraints
    cons = []
    cons.append({'type': 'ineq', 'fun': inside_box})
    cons.append({'type': 'ineq', 'fun': no_overlap})

    # Initial guess (centers and radii)
    initial_guess = []
    for i in range(num_circles):
        initial_guess.extend([np.random.uniform(0.1, 0.9), np.random.uniform(0.1, 0.9), radii[i]])

    # Optimize
    result = minimize(objective, initial_guess, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-6)

    # Extract the solution
    centers_opt = np.zeros((num_circles, 2))
    radii_opt = np.zeros(num_circles)
    for i in range(num_circles):
        centers_opt[i, 0] = result.x[2 * i]
        centers_opt[i, 1] = result.x[2 * i + 1]
        radii_opt[i] = result.x[2 * i + 2]

    return (centers_opt, radii_opt, -result.fun)
