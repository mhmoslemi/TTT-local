import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    radii = np.full(36, 0.04)  # Initial radius for each circle

    # Remove the extra circles beyond 26
    centers = centers[:26]
    radii = radii[:26]

    # Define the objective function to maximize the sum of radii (minimize negative sum)
    def objective(radii_):
        return -np.sum(radii_)

    # Define constraints
    def constraint_bounds(radii_):
        # Each circle must be inside the unit square
        return np.sum((radii_ <= np.min(np.abs(centers - np.array([0, 0])))) & 
                      (radii_ <= np.min(np.abs(centers - np.array([1, 1])))) &
                      (radii_ <= np.min(np.abs(centers - np.array([0, 1])))) &
                      (radii_ <= np.min(np.abs(centers - np.array([1, 0])))))

    def constraint_overlap(radii_):
        # No two circles may overlap
        dists = np.zeros((26, 26))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx * dx + dy * dy) - (radii_[i] + radii_[j])
        return np.min(dists)

    # Define constraints for optimization
    bounds = [(0, 0.1) for _ in range(26)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_bounds},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract results
    centers_result = centers
    radii_result = result.x
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
