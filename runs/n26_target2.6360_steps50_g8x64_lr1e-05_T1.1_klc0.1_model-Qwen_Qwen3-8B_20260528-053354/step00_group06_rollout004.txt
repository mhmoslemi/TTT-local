import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place 26 circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii as a small value
    radii = np.full(num_circles, 0.05)

    # Objective function to minimize: negative sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints
    def constraints(x):
        # x is a flat array, reshape to (num_circles, 2)
        x_flat = x.reshape((num_circles, 2))
        # Constraint 1: all circles within the unit square
        r = x[num_circles:]
        c = x_flat[:num_circles]
        in_square = np.all((r <= c[:, 0]) & (c[:, 0] <= 1 - r) & (r <= c[:, 1]) & (c[:, 1] <= 1 - r))
        if not in_square:
            return np.inf

        # Constraint 2: no overlapping between circles
        distances = np.sqrt(np.sum((c[:, np.newaxis, :] - c[np.newaxis, :, :]) ** 2, axis=2))
        radii_sum = r[:, np.newaxis] + r[np.newaxis, :]
        non_overlap = np.all(distances >= radii_sum)
        if not non_overlap:
            return np.inf

        return 0

    # Bounds: radii >= 0, positions in [0, 1]
    bounds = []
    for _ in range(num_circles):
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y
    for _ in range(num_circles):
        bounds.append((0, 0.5))  # radius

    # Initial guess as flat array
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization
    result = minimize(
        objective,
        initial_guess,
        method="SLSQP",
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        tol=1e-6,
        options={"disp": False}
    )

    # Extract results
    final_guess = result.x
    centers = final_guess[:num_circles * 2].reshape((num_circles, 2))
    radii = final_guess[num_circles * 2:].copy()
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
