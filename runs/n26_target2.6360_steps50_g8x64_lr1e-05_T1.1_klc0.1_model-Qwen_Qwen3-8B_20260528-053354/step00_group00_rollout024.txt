import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (random within the unit square)
    centers = np.random.rand(26, 2)
    radii = np.ones(26) * 0.1  # Initial radius

    # Objective function: maximize the sum of radii
    def objective(x):
        return -sum(x[26:])  # Negative for minimization

    # Bounds for centers and radii
    bounds = []
    for i in range(26):
        bounds.append((0, 1))  # x coordinate
        bounds.append((0, 1))  # y coordinate
        bounds.append((0, 1))  # radius

    # Constraints: circles must be inside the square
    def constraint_inside(x):
        for i in range(26):
            r = x[26 + i]
            x_center = x[2 * i]
            y_center = x[2 * i + 1]
            if not (r <= x_center <= 1 - r and r <= y_center <= 1 - r):
                return -1
        return 0

    # Constraints: circles must not overlap
    def constraint_overlap(x):
        for i in range(26):
            for j in range(i + 1, 26):
                r1 = x[26 + i]
                r2 = x[26 + j]
                x1, y1 = x[2 * i], x[2 * i + 1]
                x2, y2 = x[2 * j], x[2 * j + 1]
                dx = x1 - x2
                dy = y1 - y2
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r1 + r2:
                    return -1
        return 0

    # Constraints to be passed to minimize
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    result = minimize(
        objective,
        np.concatenate((centers.ravel(), radii)),
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract the solution
    optimal_centers = result.x[:52].reshape(26, 2)
    optimal_radii = result.x[52:]
    sum_radii = -result.fun  # Objective was negated

    return (optimal_centers, optimal_radii, sum_radii)
