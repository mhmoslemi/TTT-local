import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import sys

def run_packing():
    # Initial guess: evenly spaced grid
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        x = 1.0 / (n_circles**0.5) * (i % n_circles + 1)
        y = 1.0 / (n_circles**0.5) * (i // n_circles + 1)
        centers[i] = [x, y]
    
    # Objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints: all circles must be inside the square
    def constraint_inside(radii):
        centers_inside = np.zeros_like(centers)
        for i in range(n_circles):
            x, y = centers[i]
            if x - radii[i] < 0 or x + radii[i] > 1 or y - radii[i] < 0 or y + radii[i] > 1:
                return np.inf
        return 0
    
    # Constraint: no two circles can overlap
    def constraint_no_overlap(radii):
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0
    
    # All radii must be non-negative
    bounds = [(0.0, 1.0) for _ in range(n_circles)]
    
    # Optimization setup
    result = minimize(objective, np.ones(n_circles), 
                      method='SLSQP',
                      bounds=bounds,
                      constraints=[{'type': 'ineq', 'fun': constraint_inside},
                                   {'type': 'ineq', 'fun': constraint_no_overlap}],
                      options={'maxiter': 500, 'ftol': 1e-8})
    
    if result.success:
        radii = result.x
        centers = np.array(centers)
        sum_radii = np.sum(radii)
        return (centers, radii, sum_radii)
    else:
        # Fallback if optimization fails
        radii = np.full(n_circles, 0.1)
        centers = np.random.rand(n_circles, 2)
        for i in range(n_circles):
            centers[i] = [np.random.uniform(0.1, 0.9), np.random.uniform(0.1, 0.9)]
        sum_radii = np.sum(radii)
        return (centers, radii, sum_radii)
