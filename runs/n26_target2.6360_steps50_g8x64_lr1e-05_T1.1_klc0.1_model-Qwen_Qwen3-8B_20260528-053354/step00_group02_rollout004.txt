import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced in a grid (for 26 circles)
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i // 5 + 0.1
        y = i % 5 + 0.1
        centers[i] = [x, y]

    # Define constraints
    def constraints_func(centers_flat):
        centers = centers_flat.reshape(-1, 2)
        constraints = []
        for i in range(num_circles):
            for j in range(i+1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    def bounds_func(centers_flat):
        centers = centers_flat.reshape(-1, 2)
        bounds = []
        for i in range(num_circles):
            bounds.append((0.0, 1.0))
            bounds.append((0.0, 1.0))
        return bounds

    # Optimization variables
    variables = centers.flatten()

    # Initial radii based on spacing
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        min_dist = np.inf
        for j in range(num_circles):
            if i != j:
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                min_dist = min(min_dist, dist)
        radii[i] = min_dist / 2

    # Optimization function
    def objective_func(centers_flat):
        centers = centers_flat.reshape(-1, 2)
        sum_r = sum(radii)
        return -sum_r  # Minimize negative sum to maximize

    # Set up optimization
    res = minimize(
        objective_func,
        variables,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints_func},
        bounds=bounds_func,
        options={'maxiter': 500, 'ftol': 1e-8}
    )

    # Extract results
    centers_opt = res.x.reshape(-1, 2)
    sum_radii = -res.fun

    # Final radii calculation based on new centers
    radii_opt = np.zeros(num_circles)
    for i in range(num_circles):
        min_dist = np.inf
        for j in range(num_circles):
            if i != j:
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                min_dist = min(min_dist, dist)
        radii_opt[i] = min_dist / 2

    # Final check: enforce radii <= min(x, 1-x, y, 1-y)
    for i in range(num_circles):
        x, y = centers_opt[i]
        radii_opt[i] = min(radii_opt[i], x, 1 - x, y, 1 - y)

    return (centers_opt, radii_opt, sum_radii)
