import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: spread circles evenly in the square
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[26:])

    # Constraints
    def constraint1(x):
        # x[0:26] are centers, x[26:] are radii
        centers = x[0:26].reshape(26, 2)
        radii = x[26:]
        # Ensure all circles are within the unit square
        for i in range(26):
            if not (radii[i] <= centers[i][0] <= 1 - radii[i] and radii[i] <= centers[i][1] <= 1 - radii[i]):
                return -np.inf
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return -np.sum(radii)

    cons = [{'type': 'ineq', 'fun': constraint1}]

    # Bounds: centers between 0 and 1, radii non-negative
    bounds = [(0, 1) for _ in range(52)] + [(0, 1) for _ in range(26)]

    # Optimization
    result = minimize(objective, x0=np.concatenate((centers.flatten(), radii)), bounds=bounds, constraints=cons)

    # Extract results
    centers_opt = result.x[0:52].reshape(26, 2)
    radii_opt = result.x[52:]
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
