import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: spread out the centers in a grid
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)  # initial radius guess

    # Objective function: maximize sum of radii
    def obj(radii):
        return -np.sum(radii)

    # Constraints: circles must be within the unit square
    def square_bounds(radii, centers):
        return np.sum((radii - np.min(centers, axis=0)) <= 0) + np.sum((1 - radii - np.max(centers, axis=0)) <= 0)

    # Constraints: circles do not overlap
    def no_overlap(radii, centers):
        dist = np.linalg.norm(centers[:, np.newaxis] - centers[np.newaxis, :], axis=2)
        min_dist = np.min(dist[np.triu_indices(26, 1)])
        return min_dist - np.sum(radii) >= 0

    # Combine all constraints
    def constraints(radii, centers):
        return [square_bounds(radii, centers), no_overlap(radii, centers)]

    # Optimization
    result = minimize(
        fun=obj,
        x0=radii,
        method='SLSQP',
        jac=lambda x: -np.ones_like(x),
        bounds=[(0, 1) for _ in range(26)],
        constraints=[
            {'type': 'ineq', 'fun': lambda x: square_bounds(x, centers)},
            {'type': 'ineq', 'fun': lambda x: no_overlap(x, centers)}
        ],
        options={'maxiter': 500, 'ftol': 1e-9}
    )

    # Extract result
    optim_centers = centers
    optim_radii = result.x
    optim_sum_radii = np.sum(optim_radii)

    return (optim_centers, optim_radii, optim_sum_radii)
