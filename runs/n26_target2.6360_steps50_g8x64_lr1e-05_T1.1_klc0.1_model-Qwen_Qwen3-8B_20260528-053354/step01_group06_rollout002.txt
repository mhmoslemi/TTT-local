import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    initial_centers = np.array([
        [i / 5 + 0.1, j / 5 + 0.1] for i in range(5) for j in range(5)
    ])[:26]
    initial_radii = np.full(num_circles, 0.1)

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must be within the unit square and not overlapping
    def constraint_within_bounds(radii):
        return np.sum((radii <= 1 - initial_centers[:, 0]) & (radii <= initial_centers[:, 0]) &
                      (radii <= 1 - initial_centers[:, 1]) & (radii <= initial_centers[:, 1])) == num_circles

    def constraint_no_overlap(radii):
        dists = np.sqrt(np.sum((initial_centers[:, np.newaxis, :] - initial_centers[np.newaxis, :, :]) ** 2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :]) == np.sum(np.triu(dists >= radii[:, np.newaxis] + radii[np.newaxis, :], k=1))

    # Create bounds for radii: non-negative
    bounds = [(0, None) for _ in range(num_circles)]

    # Initial constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_within_bounds},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization
    result = minimize(objective, initial_radii, bounds=bounds, constraints=cons)

    centers = initial_centers
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
