import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced points in a grid
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    radii = np.zeros(25)
    # Add one more circle at the center
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = np.append(radii, 0.0)
    # Initial sum of radii
    sum_radii = 0.0

    # Objective function to maximize the sum of radii
    def objective(radii_vec):
        nonlocal sum_radii
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraint: all circles are inside the unit square
    def constraint_inside(radii_vec):
        nonlocal centers
        return np.array([r <= 1 - x - r for x, r in zip(centers[:, 0], radii_vec)])
    
    # Constraint: no two circles overlap
    def constraint_overlap(radii_vec):
        nonlocal centers
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.array([dists[i, j] >= radii_vec[i] + radii_vec[j] for i in range(len(centers)) for j in range(i+1)]).flatten()
    
    # Equality constraints for optimization
    cons = (
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    )

    # Optimization bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(len(centers))]

    # Initial guess for optimization
    initial_guess = np.zeros(len(centers))

    # Run optimization
    result = minimize(objective, initial_guess, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract final solution
    final_radii = result.x
    final_centers = centers
    final_sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, final_sum_radii)
