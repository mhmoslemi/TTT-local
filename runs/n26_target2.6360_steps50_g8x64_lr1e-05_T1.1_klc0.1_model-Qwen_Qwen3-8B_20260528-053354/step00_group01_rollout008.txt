import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        for j in range(n):
            if i * n + j < n:
                centers[i * n + j] = [j / (n - 1), i / (n - 1)]
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints: circles must be inside the unit square
    def inside_circle(radii):
        return [radii[i] <= 1 - np.sqrt((centers[i, 0])**2 + (centers[i, 1])**2) for i in range(n)]
    
    # Constraints: no two circles overlap
    def no_overlap(radii):
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(n):
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dists[i, j] = math.sqrt(dx**2 + dy**2)
        return [dists[i, j] >= radii[i] + radii[j] for i in range(n) for j in range(n) if i != j]
    
    # Initial guess for radii
    radii_initial = np.full(n, 0.1)
    
    # Setup optimization problem
    constraints = []
    constraints.append({'type': 'ineq', 'fun': lambda r: inside_circle(r)})
    constraints.append({'type': 'ineq', 'fun': lambda r: no_overlap(r)})
    
    result = minimize(objective, radii_initial, constraints=constraints, method='SLSQP', bounds=[(0, 1) for _ in range(n)])
    
    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii = result.fun * -1
    
    return (centers_final, radii_final, sum_radii)
