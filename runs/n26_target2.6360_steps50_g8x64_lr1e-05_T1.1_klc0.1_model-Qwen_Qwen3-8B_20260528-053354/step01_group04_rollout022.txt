import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    n = 26
    # Initial guess for circle centers and radii
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    for i in range(n):
        row = i // 5
        col = i % 5
        x = 0.2 + col * 0.4
        y = 0.2 + row * 0.4
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective_func(x):
        radii = x[:n]
        centers = x[n:].reshape((n, 2))
        # Check bounds constraint
        for i in range(n):
            r = radii[i]
            x_center, y_center = centers[i]
            if not (r <= x_center <= 1 - r and r <= y_center <= 1 - r):
                return float('inf')
        # Check distance constraint
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j] - 1e-8:
                    return float('inf')
        return -np.sum(radii)

    # Bounds: radii between 0 and 0.5, centers between 0 and 1
    bounds = [(0, 0.5)] * n + [(0, 1)] * (2 * n)

    # Initial guess
    initial_guess = np.concatenate((radii, centers.flatten()))

    # Optimize
    result = minimize(
        objective_func,
        initial_guess,
        method='L-BFGS-B',
        bounds=bounds,
        tol=1e-6,
        options={'disp': False}
    )

    optimized_radii = result.x[:n]
    optimized_centers = result.x[n:].reshape((n, 2))

    # Clip radii to ensure circles are inside the unit square
    for i in range(n):
        r = optimized_radii[i]
        x, y = optimized_centers[i]
        if x < r:
            optimized_radii[i] = x
        elif x > 1 - r:
            optimized_radii[i] = 1 - x
        if y < r:
            optimized_radii[i] = y
        elif y > 1 - r:
            optimized_radii[i] = 1 - y

    # Final check for overlaps
    for i in range(n):
        for j in range(i + 1, n):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            distance = math.hypot(dx, dy)
            if distance < optimized_radii[i] + optimized_radii[j] - 1e-8:
                # Reduce radii to prevent overlap
                overlap = (optimized_radii[i] + optimized_radii[j]) - distance
                optimized_radii[i] -= overlap / 2
                optimized_radii[j] -= overlap / 2

    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
