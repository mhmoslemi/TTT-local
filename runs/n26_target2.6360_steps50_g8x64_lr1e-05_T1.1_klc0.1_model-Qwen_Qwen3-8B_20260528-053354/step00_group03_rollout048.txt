import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 5x5 points inside the unit square
    # Adjusted to allow for spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial guess for centers: spread out points
    idx = 0
    for i in range(5):
        for j in range(5):
            if i == 0 or j == 0 or i == 4 or j == 4:
                centers[idx] = [i / 4, j / 4]
                idx += 1
    # Add more points to reach 26
    for i in range(4):
        centers[idx] = [i / 4, 0.5]
        idx += 1
    for i in range(4):
        centers[idx] = [0.5, i / 4]
        idx += 1
    for i in range(4):
        centers[idx] = [i / 4, 0.75]
        idx += 1
    for i in range(4):
        centers[idx] = [0.75, i / 4]
        idx += 1
    
    # Optimization bounds for radii
    bounds = [(0, 0.5) for _ in range(num_circles)]
    
    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)
    
    # Constraints: circles are inside the square and no two circles overlap
    def constraint_inside(x):
        r = np.reshape(x, (num_circles,))
        return np.array([r[i] + r[j] <= 1 - np.abs(centers[i, 0] - centers[j, 0]) - np.abs(centers[i, 1] - centers[j, 1]) for i in range(num_circles) for j in range(num_circles)])
    
    def constraint_overlap(x):
        r = np.reshape(x, (num_circles,))
        dist = np.sqrt((centers[:, 0][:, np.newaxis] - centers[:, 0]) ** 2 + (centers[:, 1][:, np.newaxis] - centers[:, 1]) ** 2)
        return np.array([dist[i, j] >= r[i] + r[j] for i in range(num_circles) for j in range(num_circles)])
    
    # Minimize negative of sum radii
    res = minimize(objective, x0=radii, bounds=bounds, constraints=[{'type': 'ineq', 'fun': constraint_inside}, {'type': 'ineq', 'fun': constraint_overlap}])
    
    # Extract results
    radii_opt = res.x
    sum_radii_opt = np.sum(radii_opt)
    
    return (centers, radii_opt, sum_radii_opt)
