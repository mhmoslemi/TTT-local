import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial radius for each circle (as a guess)
    initial_radius = 0.05
    radii[:] = initial_radius
    
    # Initial positions in a grid
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.4 + 0.1
        centers[i, 1] = row * 0.4 + 0.1
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize it

    def constraints(radii):
        constraints = []
        # Each circle must be within the unit square
        for i in range(num_circles):
            r = radii[i]
            for j in range(num_circles):
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.sqrt(dx*dx + dy*dy)
                    r1 = radii[i]
                    r2 = radii[j]
                    constraints.append(dist - (r1 + r2))
        # All radii must be non-negative
        for r in radii:
            constraints.append(r)  # r >= 0
        return constraints

    # Set up the optimization problem
    bounds = [(0, 1) for _ in range(num_circles)]
    cons = {'type': 'ineq', 'fun': lambda x: constraints(x)}
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
