import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all equal to 0.1
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Bounds constraint: radii >= 0
        for i in range(26):
            if radii[i] < 0:
                return False
        # Constraint: circles must stay within the unit square
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return False
        # Constraint: no overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Setup optimization
    bounds = [(0, 1) for _ in range(26)]
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "success", "fun": constraints},
        method="SLSQP",
        tol=1e-6
    )

    # Extract the result
    radii_opt = result.x
    centers_opt = np.copy(centers)
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
