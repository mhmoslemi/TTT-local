import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: random positions inside the square
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)  # Initial small radius

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraints: 1) Circles inside square, 2) No overlap
        # Flatten radii for optimization (optimization variables are radii)
        radii = radii.reshape(-1)
        constraints = []
        
        # Constraint: radii >= 0
        for i in range(len(radii)):
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i: x[i]})

        # Constraint: circle inside the square
        for i in range(len(radii)):
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - x[i] - centers[i, 0]})
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - x[i] - centers[i, 1]})
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i: x[i] - centers[i, 0]})
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i: x[i] - centers[i, 1]})
        
        # Constraint: circles do not overlap
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                def constraint(x):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = math.sqrt(dx*dx + dy*dy)
                    return dist - (radii[i] + radii[j])
                constraints.append({'type': 'ineq', 'fun': constraint})
        
        return constraints

    # Optimize
    result = minimize(objective, radii, constraints=constraints(radii), method='SLSQP', tol=1e-6)

    # Extract results
    centers = np.array(result.x.reshape(26, 2))
    radii = np.array(result.x)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
