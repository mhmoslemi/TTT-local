import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i] = [col / 4 + 0.125, row / 4 + 0.125]
    
    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints
    def constraints_func(x):
        # x is a vector of shape (2*num_circles + num_circles,) = (52 + 26,)
        centers = x[:2*num_circles].reshape(num_circles, 2)
        radii = x[2*num_circles:]
        
        # Constraint: all circles are within the unit square
        in_bounds = np.all((radii <= 1 - centers) & (centers <= 1 - radii), axis=1)
        if not np.all(in_bounds):
            return np.inf
        
        # Constraint: no overlaps between circles
        dists = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        no_overlap = np.all(dists >= radii_sum)
        if not no_overlap:
            return np.inf
        
        return 0
    
    # Optimization bounds: centers in [0, 1], radii >= 0
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y for each center
        bounds.extend([(0.0, 1.0)])              # radius for each circle
    
    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), np.ones(num_circles)])
    
    # Optimization
    result = minimize(objective, initial_guess, bounds=bounds, constraints={'type': 'eq', 'fun': constraints_func})
    
    # Extract results
    optimal_centers = result.x[:2*num_circles].reshape(num_circles, 2)
    optimal_radii = result.x[2*num_circles:]
    optimal_sum = np.sum(optimal_radii)
    
    return (optimal_centers, optimal_radii, optimal_sum)
