import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = np.random.uniform(radii[i], 1 - radii[i])
            centers[i, 1] = np.random.uniform(radii[i], 1 - radii[i])
        # Initial placement with random positions within bounds
        # Now apply constraints
        constraints = []
        for i in range(26):
            for j in range(i + 1, 26):
                dist = np.linalg.norm(centers[i] - centers[j])
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    # Initial guess: all radii set to 0.1
    initial_guess = np.array([0.1] * 26)

    # Optimization setup
    bounds = [(0, 0.5) for _ in range(26)]  # Radii can't exceed 0.5

    # Initial placement with random centers
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = np.random.uniform(0.1, 0.9)
        centers[i, 1] = np.random.uniform(0.1, 0.9)

    # Objective function
    def objective_func(radii):
        return -np.sum(radii)

    # Constraint functions
    def constraint_func(radii):
        constraints = []
        for i in range(26):
            for j in range(i + 1, 26):
                dist = np.linalg.norm(centers[i] - centers[j])
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)

    # Optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=RuntimeWarning)
        result = minimize(objective_func, initial_guess, bounds=bounds, constraints={"type": "ineq", "fun": constraint_func}, method="SLSQP")

    # Post-processing
    radii = result.x
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = np.random.uniform(radii[i], 1 - radii[i])
        centers[i, 1] = np.random.uniform(radii[i], 1 - radii[i])
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
