import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as dist

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial guess for radii
    radii = np.full(26, 0.1)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii

    def constraint_circle_inside(radii, centers):
        # Check if all circles are inside the unit square
        x = centers[:, 0]
        y = centers[:, 1]
        return (x - radii) >= 0.0001 & (1 - x - radii) >= 0.0001 & (y - radii) >= 0.0001 & (1 - y - radii) >= 0.0001

    def constraint_no_overlap(radii, centers):
        # Check if no two circles overlap
        d = dist.cdist(centers, centers)
        r_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return d >= r_sum - 1e-6

    def constraint_all_non_negative(radii):
        return radii >= 0.0

    bounds = [(0.0, 1.0) for _ in range(26)]

    # Optimization setup
    result = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_circle_inside, 'args': (centers,)},
            {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (centers,)},
            {'type': 'ineq', 'fun': constraint_all_non_negative, 'args': (radii,)}
        ],
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    # Extract results
    centers = np.array(result.x.reshape(26, 2))
    radii = np.array(result.x)
    sum_radii = -result.fun

    # Ensure the circles are fully inside the square
    for i in range(26):
        x, y = centers[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            print(f"Circle {i} is outside the square.")
            # Adjust to stay within the square
            dx = max(r - (x - r), 1 - x - r)
            dy = max(r - (y - r), 1 - y - r)
            centers[i] += np.array([dx, dy])

    # Final checks
    for i in range(26):
        for j in range(i + 1, 26):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist_ = np.sqrt(dx**2 + dy**2)
            if dist_ < radii[i] + radii[j] - 1e-8:
                print(f"Circles {i} and {j} overlap.")

    return (centers, radii, sum_radii)
