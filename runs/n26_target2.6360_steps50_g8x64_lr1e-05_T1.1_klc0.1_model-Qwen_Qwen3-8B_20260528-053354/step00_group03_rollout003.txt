import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Convert radii to a list of circles with centers
        centers = np.random.uniform(size=(26, 2))
        radii = np.clip(radii, 0, 1)
        for i in range(26):
            for j in range(i + 1, 26):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j]:
                    return True
        return False

    # Initial guess for radii: all equal
    initial_guess = np.ones(26) * 0.1

    # Optimization settings
    bounds = [(0, 1) for _ in range(26)]
    result = minimize(objective, initial_guess, bounds=bounds, method='SLSQP', constraints=constraints)

    # Final radii and centers
    radii = result.x
    centers = np.random.uniform(size=(26, 2))

    # Ensure circles are within bounds
    radii = np.clip(radii, 0, 1)
    for i in range(26):
        centers[i] = np.clip(centers[i], radii[i], 1 - radii[i])

    # Force non-overlapping constraints
    for i in range(26):
        for j in range(i + 1, 26):
            dist = np.linalg.norm(centers[i] - centers[j])
            if dist < radii[i] + radii[j]:
                adjustment = (radii[i] + radii[j] - dist) / 2
                centers[i] += adjustment * (centers[j] - centers[i]) / dist
                centers[j] -= adjustment * (centers[j] - centers[i]) / dist
                radii[i] = np.clip(radii[i], 0, 1 - np.min(np.abs(centers[i] - np.array([0, 0])))
                radii[j] = np.clip(radii[j], 0, 1 - np.min(np.abs(centers[j] - np.array([0, 0]))

    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
