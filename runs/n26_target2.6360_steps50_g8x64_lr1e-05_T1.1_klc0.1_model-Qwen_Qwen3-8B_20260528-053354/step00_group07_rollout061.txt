import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced points in the square
    centers = np.linspace(0.1, 0.9, 26).reshape(-1, 1)
    centers = np.column_stack((centers, centers))
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.column_stack((np.linspace(0.1, 0.9, 26), np.linspace(0.1, 0.9, 26)))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds(radii):
        return [(0.0, 1.0 - 0.01) for _ in range(26)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=radii,
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    centers = np.column_stack((np.linspace(0.1, 0.9, 26), np.linspace(0.1, 0.9, 26)))
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
