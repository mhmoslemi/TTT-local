import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid-like arrangement
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = 0.2 + col * 0.4
        y = 0.2 + row * 0.4
        centers[i] = [x, y]
        radii[i] = 0.1

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: circles must be fully inside the unit square
    def inside_box(radii, centers):
        return np.all(radii <= 1 - centers[:, 0] - radii) and np.all(radii <= 1 - centers[:, 1] - radii)

    # Constraint: circles must not overlap
    def no_overlap(radii, centers):
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[(i * (n - 1) // 2) + j - i - 1] = dist
        return np.all(dists >= radii[0:n] + radii[1:n])

    # Optimization with constraints
    cons = [
        {'type': 'ineq', 'fun': lambda r: inside_box(r, centers)},
        {'type': 'ineq', 'fun': lambda r: no_overlap(r, centers)},
    ]
    bounds = [(0, 1) for _ in range(26)]

    result = minimize(objective, radii, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-8)

    centers = np.array([[0.2 + col * 0.4, 0.2 + row * 0.4] for row in range(5) for col in range(5)])
