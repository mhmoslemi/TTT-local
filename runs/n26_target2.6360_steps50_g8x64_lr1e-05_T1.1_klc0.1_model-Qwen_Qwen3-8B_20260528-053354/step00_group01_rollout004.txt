import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii: place circles in a grid
    # This is a heuristic and may not yield the optimal solution
    # But it's a starting point for optimization

    # Initialize centers
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = (col + 0.5) / 5.0 + 0.1
        y = (row + 0.5) / 5.0 + 0.1
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial radius

    # Objective function to maximize sum of radii
    def objective(radii_param):
        return -np.sum(radii_param)

    # Constraints: circles must be inside the unit square
    def constraint_inside(radii_param):
        centers_inside = []
        for r in radii_param:
            centers_inside.append(1.0 - r)
        return np.array(centers_inside)

    # Constraints: distance between any two centers >= sum of their radii
    def constraint_overlap(radii_param):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i] = [0.5 + 0.1 * i, 0.5 + 0.1 * (i // 5)]  # Placeholder for initial centers, actual centers would come from optimization
        dists = np.zeros(26 * 25 // 2)
        idx = 0
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = math.hypot(dx, dy)
                dists[idx] = dist - (radii_param[i] + radii_param[j])
                idx += 1
        return dists

    # Optimization problem setup
    bounds = [(0.0, 1.0) for _ in range(26)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: 1.0 - r},  # Ensure each circle is inside the square
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    centers_opt = centers
    radii_opt = result.x
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
