import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.02)

    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii, centers):
        # Constraint: All circles must be inside the unit square
        r = radii.reshape(-1, 1)
        lb = np.zeros_like(r)
        ub = np.ones_like(r)
        constraints = []
        for i in range(26):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - (1 - r[i] - np.min(centers[i])))
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - (1 - r[i] - np.max(centers[i])))
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - (1 - r[i] - np.min(centers[i], axis=0)[0]))
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - (1 - r[i] - np.max(centers[i], axis=0)[0])))

        # Constraint: No two circles overlap
        dists = pdist(centers)
        for j, dist in enumerate(dists):
            if dist < 1e-8:
                continue
            idx1 = j // 26
            idx2 = j % 26
            if idx1 != idx2:
                constraints.append({'type': 'ineq', 'fun': lambda r, idx1=idx1, idx2=idx2: dist - (r[idx1] + r[idx2])})

        return constraints

    # Optimization
    result = minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        method='SLSQP',
        jac=lambda x: np.zeros_like(x),
        constraints=constraints(radii, centers),
        bounds=[(0, 1) for _ in range(26)]
    )

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -objective(radii, centers)

    return (centers, radii, sum_radii)
