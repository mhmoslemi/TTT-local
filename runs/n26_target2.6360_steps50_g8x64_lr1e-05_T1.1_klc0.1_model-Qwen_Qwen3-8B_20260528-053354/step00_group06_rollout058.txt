import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly placed circles with small radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.01)
    
    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x[1:])  # Negative because we minimize
    
    # Define constraints
    def constraint_center_inside(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        return np.all((radii_flat <= 1 - centers_flat) & (centers_flat <= radii_flat))
    
    def constraint_no_overlap(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        distances = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return np.all(distances >= radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])
    
    # Setup optimization problem
    bounds = []
    for i in range(num_circles):
        # Each circle has 2 center coordinates (x, y) and 1 radius
        bounds.extend([(0, 1) for _ in range(2)])  # x, y
        bounds.extend([(0, 1) for _ in range(1)])  # radius
    
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    
    # Initial guess
    x0 = np.zeros(num_circles * 3)
    x0[:num_circles * 2] = centers.flatten()
    x0[num_circles * 2:] = radii
    
    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints)
    
    # Extract results
    centers_result = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii_result = result.x[num_circles * 2:]
    sum_radii_result = -result.fun
    
    return (centers_result, radii_result, sum_radii_result)
