import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    r = 1 / np.sqrt(26)  # A rough estimate for the radius
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Objective function to minimize: negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: circles must be inside the square
    def inside_square(radii):
        return np.all(radii <= 0.5)  # Maximum possible radius is 0.5

    # Constraint: circles must not overlap
    def no_overlap(radii):
        # For each pair of circles, check distance >= sum of radii
        dists = np.zeros((26, 26))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)
                dists[j, i] = dists[i, j]
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Initial radii guess
    radii = np.full(26, r)

    # Optimization options
    bounds = [(0.0, 0.5) for _ in range(26)]

    # Combine objective and constraints
    def constraint_inside(radii):
        return inside_square(radii)

    def constraint_overlap(radii):
        return no_overlap(radii)

    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Run optimization
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP'
    )

    # Extract results
    centers_result = centers
    radii_result = result.x
    sum_radii = -result.fun

    return (centers_result, radii_result, sum_radii)
