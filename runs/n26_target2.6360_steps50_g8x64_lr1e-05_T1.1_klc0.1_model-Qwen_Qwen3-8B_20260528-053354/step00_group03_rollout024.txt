import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial random centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    
    # Objective function: minimize negative sum of radii (to maximize it)
    def objective(x):
        return -np.sum(x[26:])
    
    # Constraint 1: Each circle is fully inside the unit square
    def constraint_inside(x):
        centers = x[:26*2].reshape((26, 2))
        radii = x[26:]
        return np.array([r - (x - 1 + r) for x, r in zip(centers, radii)]).flatten()
    
    # Constraint 2: No two circles overlap
    def constraint_overlap(x):
        centers = x[:26*2].reshape((26, 2))
        radii = x[26:]
        distances = np.zeros((26, 26))
        for i in range(26):
            for j in range(i+1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distances[i, j] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
        return distances.flatten()
    
    # Bounds: radii >= 0, centers in [0, 1]
    bounds = [(0, 1) for _ in range(26*2)] + [(0, 1) for _ in range(26)]
    
    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints=cons)
    
    # Extract result
    centers_result = result.x[:26*2].reshape((26, 2))
    radii_result = result.x[26:]
    sum_radii_result = np.sum(radii_result)
    
    return (centers_result, radii_result, sum_radii_result)
