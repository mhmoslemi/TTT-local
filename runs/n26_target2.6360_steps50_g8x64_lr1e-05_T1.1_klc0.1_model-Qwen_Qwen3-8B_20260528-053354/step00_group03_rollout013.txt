import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced grid points in the unit square
    def initial_guess():
        n = 26
        x = np.linspace(0.1, 0.9, n)
        y = np.linspace(0.1, 0.9, n)
        centers = np.column_stack((np.repeat(x, n), np.tile(y, n)))
        return centers

    centers = initial_guess()
    radii = np.ones(26) * 0.1

    # Objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: all circles are inside the unit square
    def in_bounds(radii, centers):
        for i, (x, y) in enumerate(centers):
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        return True

    # Constraint: no two circles overlap
    def no_overlap(radii, centers):
        dists = pdist(centers)
        for i, d in enumerate(dists):
            if d < radii[i] + radii[np.argmax(dists[:i+1])]:
                return False
        return True

    # Combine all constraints into a single function
    def constraints(radii):
        if not in_bounds(radii, centers):
            return np.inf
        if not no_overlap(radii, centers):
            return np.inf
        return 0

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(26)]

    # Optimization setup
    result = minimize(
        objective,
        radii,
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    # Extract results
    centers = np.array(result.x).reshape(-1, 2)
    radii = result.x
    sum_radii = np.sum(radii)

    # Ensure all circles are inside the unit square
    for i, (x, y) in enumerate(centers):
        if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
            raise ValueError("One or more circles are not fully inside the unit square.")

    # Ensure no two circles overlap
    dists = pdist(centers)
    for i, d in enumerate(dists):
        if d < radii[i] + radii[np.argmax(dists[:i+1])]:
            raise ValueError("Two circles overlap.")

    return (centers, radii, sum_radii)
