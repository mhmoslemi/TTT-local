import numpy as np
from scipy.optimize import minimize

def run_packing():
    num_circles = 26
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((num_circles, 2))
    indices = np.arange(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (indices[i] % 5 + 0.5) / 5
        centers[i, 1] = (indices[i] // 5 + 0.5) / 5

    # Initial radii: uniform
    radii = np.full(num_circles, 1 / 10)

    def objective_func(radii_vec):
        return -np.sum(radii_vec)

    def constraint_radius(radii_vec, i):
        return radii_vec[i] - 1e-8  # Ensure non-negative radii

    def constraint_position(radii_vec, i, x_center, y_center):
        return radii_vec[i] - (1 - x_center - radii_vec[i])  # x[i] <= 1 - x_center - x[i]

    def constraint_position2(radii_vec, i, x_center, y_center):
        return radii_vec[i] - (1 - y_center - radii_vec[i])  # x[i] <= 1 - y_center - x[i]

    def constraint_overlap(radii_vec):
        total = 0.0
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii_vec[i]
            for j in range(i + 1, num_circles):
                x_j, y_j = centers[j]
                r_j = radii_vec[j]
                dx = x_i - x_j
                dy = y_i - y_j
                dist = np.sqrt(dx**2 + dy**2)
                total += max(0, dist - (r_i + r_j))
        return total

    # Constraints
    constraints = []

    # Non-negative radii
    for i in range(num_circles):
        x_center, y_center = centers[i]
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x_center=x_center, y_center=y_center: constraint_radius(r, i),
                            'args': (i, x_center, y_center)})

    # Position constraint for x-coordinate
    for i in range(num_circles):
        x_center, y_center = centers[i]
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x_center=x_center, y_center=y_center: constraint_position(r, i, x_center, y_center),
                            'args': (i, x_center, y_center)})

    # Position constraint for y-coordinate
    for i in range(num_circles):
        x_center, y_center = centers[i]
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x_center=x_center, y_center=y_center: constraint_position2(r, i, x_center, y_center),
                            'args': (i, x_center, y_center)})

    # Overlap constraint
    constraints.append({'type': 'ineq', 'fun': constraint_overlap})

    # Initial guess
    initial_guess = radii.copy()

    # Optimization
    result = minimize(
        objective_func,
        initial_guess,
        method='SLSQP',
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000, 'disp': False}
    )

    if not result.success:
        print("Optimization failed.")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    # Final radii
    radii = result.x

    # Validate constraints
    for i in range(num_circles):
        x_center, y_center = centers[i]
        r_i = radii[i]
        if r_i > (1 - x_center - r_i):
            radii[i] = 1 - x_center - r_i
        if r_i > (1 - y_center - r_i):
            radii[i] = 1 - y_center - r_i

    for i in range(num_circles):
        x_i, y_i = centers[i]
        r_i = radii[i]
        for j in range(i + 1, num_circles):
            x_j, y_j = centers[j]
            r_j = radii[j]
            dx = x_i - x_j
            dy = y_i - y_j
            dist = np.sqrt(dx**2 + dy**2)
            if dist < (r_i + r_j):
                # Adjust radii to avoid overlap
                radii[i] = (dist - r_j) / 2
                radii[j] = (dist - r_i) / 2

    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
