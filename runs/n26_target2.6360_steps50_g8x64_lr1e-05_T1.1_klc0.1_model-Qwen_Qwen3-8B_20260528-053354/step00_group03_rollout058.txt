import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    for i in range(26):
        row = i // 5
        col = i % 5
        x = col / 4 + 0.1
        y = row / 4 + 0.1
        centers[i] = [x, y]

    radii = np.ones(26) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = (i % 5) / 4 + 0.1
            centers[i, 1] = (i // 5) / 4 + 0.1
        for i in range(26):
            center_i = centers[i]
            r_i = radii[i]
            if center_i[0] - r_i < 0 or center_i[0] + r_i > 1 or center_i[1] - r_i < 0 or center_i[1] + r_i > 1:
                return 1  # Circle is not fully inside the square
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist_sq = dx * dx + dy * dy
                if dist_sq < (radii[i] + radii[j]) ** 2:
                    return 1  # Circles overlap
        return 0

    # Optimization
    res = minimize(objective, radii, method='SLSQP', bounds=[(0, 0.5)]*26, constraints={"type": "ineq", "fun": constraints})

    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5) / 4 + 0.1
        centers[i, 1] = (i // 5) / 4 + 0.1

    return (centers, res.x, np.sum(res.x))
