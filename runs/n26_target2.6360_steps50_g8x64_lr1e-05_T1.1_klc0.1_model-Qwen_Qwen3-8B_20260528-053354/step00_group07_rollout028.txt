import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    centers = np.array([(i/7, j/7) for i in range(8) for j in range(8) if i + j < 14])
    radii = np.full(len(centers), 0.04)  # small initial radius

    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        for i in range(len(radii)):
            r = radii[i]
            if r > 1 - np.min(centers[i]) or r > 1 - np.max(centers[i]):
                return False
        # Constraint 2: No two circles overlap
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimize with constraints
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1 - 1e-6) for _ in range(len(radii))]
    )

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
