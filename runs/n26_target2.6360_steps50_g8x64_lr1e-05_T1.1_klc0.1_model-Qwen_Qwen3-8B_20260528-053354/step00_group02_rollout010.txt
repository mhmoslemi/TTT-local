import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 non-overlapping circles
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1
    radii[radii > 0.5] = 0.5  # Ensure radii are small initially

    # Objective function to minimize: negative of sum of radii
    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ...]
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints: 1) circle is inside square
    #               2) circles do not overlap
    #               3) radii >= 0
    def constraint_inside(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        return np.minimum(centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat)

    def constraint_overlap(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        dist = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dist - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    def constraint_non_negative(x):
        return x[2*26:]

    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_non_negative}
    ]

    # Bounds: x and y in [0,1], radii in [0, 0.5]
    bounds = [(0, 1) for _ in range(2*26)] + [(0, 0.5) for _ in range(26)]

    # Initial guess as flat array
    x0 = np.concatenate([centers.flatten(), radii])

    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-6)

    if not result.success:
        raise ValueError("Optimization failed to converge")

    centers_opt = result.x[:2*26].reshape(26, 2)
    radii_opt = result.x[2*26:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
