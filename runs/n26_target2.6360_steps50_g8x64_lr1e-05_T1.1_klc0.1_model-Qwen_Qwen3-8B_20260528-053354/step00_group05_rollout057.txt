import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5) / 4.0 + 0.05
        centers[i, 1] = (i // 5) / 4.0 + 0.05

    radii = np.full(26, 0.05)

    def objective(radii_vec):
        return -np.sum(radii_vec)  # Maximize sum by minimizing negative sum

    def constraints(radii_vec):
        # Constraint 1: All circles are inside the unit square
        r = radii_vec
        for i in range(26):
            if r[i] < 0 or r[i] > 1 - 2 * r[i]:
                return False
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Set up constraints for optimization
    cons = {'type': 'ineq', 'fun': constraints}

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=cons,
        bounds=[(0, 1 - 2 * r) for r in radii],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    # Extract results
    optimized_radii = result.x
    optimized_centers = centers
    sum_radii = -result.fun

    return (optimized_centers, optimized_radii, sum_radii)
