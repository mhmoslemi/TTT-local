import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for circle centers and radii (simple grid layout)
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i] = [col * 0.3 + 0.1, row * 0.3 + 0.1]
        radii[i] = 0.04

    def objective(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize the negative to maximize sum

    def constraints(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        cons = []
        # Boundary constraints
        for i in range(26):
            x_min, x_max = centers_flat[i][0] - radii_flat[i], centers_flat[i][0] + radii_flat[i]
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: x_min - 0.0})
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: 1.0 - x_max})
            y_min, y_max = centers_flat[i][1] - radii_flat[i], centers_flat[i][1] + radii_flat[i]
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: y_min - 0.0})
            cons.append({'type': 'ineq', 'fun': lambda x, i=i: 1.0 - y_max})
        # Non-overlap constraints
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = math.sqrt(dx*dx + dy*dy)
                cons.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: dist - (radii_flat[i] + radii_flat[j])})
        return cons

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])
    # Bounds for radii (non-negative)
    bounds = [(0, 1.0) for _ in range(26)] + [(0, 1.0) for _ in range(52, 52 + 26)]
    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints)
    centers_opt = result.x[:52].reshape(26, 2)
    radii_opt = result.x[52:]
    sum_radii_opt = -result.fun
    return (centers_opt, radii_opt, sum_radii_opt)
