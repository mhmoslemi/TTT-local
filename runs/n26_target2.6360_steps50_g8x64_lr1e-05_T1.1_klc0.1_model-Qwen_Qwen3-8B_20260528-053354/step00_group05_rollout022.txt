```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.array([[i / 5 + 0.1, j / 5 + 0.1] for j in range(5) for i in range(5)])
    centers = np.vstack([centers, [0.5, 0.75], [0.5, 0.25], [0.75, 0.5], [0.25, 0.5]])
    centers = centers[:num_circles]
    radii = np.full(num_circles, 0.1)

    def objective(x):
        return -np.sum(radii)

    def constraint_bounds(x):
        for i in range(num_circles):
            x_i, y_i = x[2*i], x[2*i+1]
            r_i = radii[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False
        return True

    def constraint_overlap(x):
        dists = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i+1, num_circles):
                if dists[i,j] < radii[i] + radii[j]:
                    return False
        return True

    def constraint_all(x):
        return constraint_bounds(x) and constraint_overlap(x)

    # Optimization with bounds and constraints
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 1))
        bounds.append((0, 1))

    def constraint_func(x):
        return np.array([constraint_all(x)])

    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_func})

    # Extract solution
    x_sol = result.x
    centers = x_sol[:2*num_circles].reshape(num_circles, 2)
    radii = x_sol[2*num_circles:]
    sum_radii = np.sum(radii)

    # Check if constraints are satisfied
    for i in range(num_circles):
        x_i, y_i = centers[i]
        r_i = radii[i]
        if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
            raise ValueError("Constraint violation: circle out of bounds")
    dists = cdist(centers, centers)
    for i in range(num_circles):
        for j in range(i+1, num_circles):
            if dists[i,j] < radii[i] + radii[j]:
                raise ValueError("Constraint violation: circles overlapping")

    return (centers, radii, sum_radii)
```
