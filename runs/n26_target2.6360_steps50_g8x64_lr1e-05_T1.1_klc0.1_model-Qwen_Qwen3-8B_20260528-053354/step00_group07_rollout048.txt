import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.05)
    # Start with a simple grid packing
    for i in range(26):
        r = 0.05
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
        if i % 5 == 0:
            r = 0.05
        radii[i] = r

    # Objective function: minimize negative sum of radii (so maximize sum)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraint 1: Each circle must be fully inside the unit square
    def constraint_inside(radii_vec, centers):
        centers = np.array(centers)
        r_vec = radii_vec
        x = centers[:, 0]
        y = centers[:, 1]
        dist_to_edge = np.minimum(x - r_vec, 1 - x - r_vec, np.minimum(y - r_vec, 1 - y - r_vec))
        return dist_to_edge

    # Constraint 2: No two circles overlap
    def constraint_overlap(radii_vec, centers):
        r_vec = radii_vec
        dist = np.linalg.norm(centers[:, np.newaxis] - centers[np.newaxis, :], axis=2)
        r_sum = r_vec[:, np.newaxis] + r_vec[np.newaxis, :]
        return dist - r_sum

    # Constraint 3: All radii non-negative
    def constraint_non_negative(radii_vec):
        return radii_vec

    # Initial guess for radii
    initial_radii = np.full(26, 0.05)

    # Optimize
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=objective,
            x0=initial_radii,
            constraints=[
                {'type': 'ineq', 'fun': lambda x: constraint_inside(x, centers)},
                {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers)},
                {'type': 'ineq', 'fun': lambda x: constraint_non_negative(x)}
            ],
            method='SLSQP',
            bounds=[(0, 1) for _ in range(26)]
        )

    centers = np.array(centers)
    radii = result.x
    sum_radii = result.fun * -1

    return (centers, radii, sum_radii)
