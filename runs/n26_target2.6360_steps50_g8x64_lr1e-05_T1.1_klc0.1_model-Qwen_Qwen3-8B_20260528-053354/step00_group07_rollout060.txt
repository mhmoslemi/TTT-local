import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place 26 circles in a grid pattern with equal spacing
    num_circles = 26
    centers = np.random.rand(num_circles, 2)  # Random initial positions

    # Objective function: minimize negative sum of radii
    def objective(x):
        return -np.sum(x[1:])  # x[0] is the sum, x[1:] are radii

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(num_circles)]

    # Constraints: circles must be inside the unit square
    def constraint_inside(x):
        for i in range(num_circles):
            x_i = x[i * 2]
            y_i = x[i * 2 + 1]
            r_i = x[num_circles + i]
            if x_i < r_i or x_i > 1 - r_i or y_i < r_i or y_i > 1 - r_i:
                return -1
        return 0

    def constraint_no_overlap(x):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i, r_i = x[i * 2], x[i * 2 + 1], x[num_circles + i]
                x_j, y_j, r_j = x[j * 2], x[j * 2 + 1], x[num_circles + j]
                dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
                if dist < r_i + r_j - 1e-8:
                    return -1
        return 0

    # Prepare optimization variables
    x0 = np.concatenate([np.zeros(1), np.random.rand(num_circles)])  # x[0] is the sum of radii, x[1:] are radii
    res = minimize(objective, x0, bounds=bounds, constraints=[{'type': 'ineq', 'fun': constraint_inside}, {'type': 'ineq', 'fun': constraint_no_overlap}], method='SLSQP', options={'ftol': 1e-9, 'maxiter': 1000})

    # Extract results
    sum_radii = res.x[0]
