import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        try:
            from scipy.optimize import Bounds
        except ImportError:
            pass

    # Initial guess: centers of 26 equally spaced circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 + 0.5
        y = i // 5 + 0.5
        centers[i] = [x, y]
    radii = np.full(26, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints: (1) Each circle is within the unit square
    def circle_in_square(x):
        return np.sum((x[:, 0] - 0.5)**2 + (x[:, 1] - 0.5)**2 - (x**2)) <= 0

    # Constraint: (2) No two circles overlap
    def no_overlap(x, radii):
        n = len(radii)
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = x[i, 0] - x[j, 0]
                dy = x[i, 1] - x[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)
                dists[j, i] = dists[i, j]
        return np.sum(dists - radii[:, None] - radii[None, :]) <= 0

    # Constraint: (3) All radii are non-negative
    def non_negative_radii(x):
        return np.all(x >= 0)

    # Optimization setup
    n = 26
    x0 = np.concatenate((centers.flatten(), radii))

    def constraints(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        # Check circles in square
        in_square = np.sum((centers[:, 0] - 0.5)**2 + (centers[:, 1] - 0.5)**2 - (radii**2)) <= 0
        # Check no overlap
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)
                dists[j, i] = dists[i, j]
        no_overlap = np.sum(dists - radii[:, None] - radii[None, :]) <= 0
        # Check non-negative radii
        non_negative = np.all(radii >= 0)
        return [in_square, no_overlap, non_negative]

    cons = [{'type': 'ineq', 'fun': lambda x: constraints(x)[0]},
            {'type': 'ineq', 'fun': lambda x: constraints(x)[1]},
            {'type': 'ineq', 'fun': lambda x: constraints(x)[2]}]

    # Bounds for centers (within [0, 1])
    bounds_centers = [(0.0, 1.0) for _ in range(2*n)]
    # Bounds for radii (non-negative)
    bounds_radii = [(0.0, 1.0) for _ in range(n)]
    bounds = bounds_centers + bounds_radii

    # Solve the optimization problem
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    x = result.x
    centers = x[:2*n].reshape(n, 2)
    radii = x[2*n:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
