import numpy as np
from scipy.optimize import minimize
from itertools import combinations

def run_packing():
    # Initial guess: spread circles evenly across the square
    centers = np.array([[x / 26, y / 26] for x in range(26) for y in range(26)])
    radii = np.full(26, 0.05)
    
    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii
    
    def constraints(radii):
        # Constraint 1: All circles are within the unit square
        # Constraint 2: No two circles overlap
        N = len(radii)
        A = []
        b = []
        # Square boundaries
        for i in range(N):
            A.append([1, 0, -radii[i]])
            A.append([-1, 0, -radii[i]])
            A.append([0, 1, -radii[i]])
            A.append([0, -1, -radii[i]])
        # Circle overlaps
        for i, j in combinations(range(N), 2):
            dist = np.linalg.norm(centers[i] - centers[j])
            A.append([0, 0, -radii[i] - radii[j]])
            A.append([0, 0, -radii[i] - radii[j]])
            A.append([0, 0, -radii[i] - radii[j]])
            A.append([0, 0, -radii[i] - radii[j]])
            A.append([0, 0, -radii[i] - radii[j]])
            A.append([0, 0, -radii[i] - radii[j]])
            b.append(dist - (radii[i] + radii[j]))
        
        # Convert to inequality constraints: A @ x <= b
        # Objective: Maximize sum(radii) => minimize -sum(radii)
        res = minimize(lambda x: -np.sum(x), radii, method='SLSQP',
                       constraints=[{'type': 'ineq', 'fun': lambda x: np.dot(A, x) - b}],
                       bounds=[(0, 1) for _ in range(N)],
                       tol=1e-6)
        return res.x
    
    # Run optimization
    optimal_radii = constraints(radii)
    sum_radii = np.sum(optimal_radii)
    return (centers, optimal_radii, sum_radii)
