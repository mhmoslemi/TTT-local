import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for circle centers (evenly spaced)
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5  # Distribute along x-axis
        centers[i, 1] = (i // 5 + 0.5) / 5  # Distribute along y-axis

    # Initial guess for radii
    radii = np.ones(26) * 0.1

    # Objective function to maximize the sum of radii (minimize negative sum)
    def objective(x):
        radii = x[:26]
        return -np.sum(radii)

    # Constraint 1: Each circle is fully inside the square
    def constraint1(x):
        radii = x[:26]
        return np.array([r <= 1 - r for r in radii])

    # Constraint 2: No two circles overlap
    def constraint2(x):
        radii = x[:26]
        centers = np.reshape(x[26:], (26, 2))
        distances = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return distances - radii_sum

    # Optimization bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(26)]

    # Combine all constraints
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    # Initial guess: concatenate centers and radii
    x0 = np.concatenate((centers.ravel(), radii))

    # Perform optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    optimized_radii = result.x[:26]
    optimized_centers = np.reshape(result.x[26:], (26, 2))
    sum_radii = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
