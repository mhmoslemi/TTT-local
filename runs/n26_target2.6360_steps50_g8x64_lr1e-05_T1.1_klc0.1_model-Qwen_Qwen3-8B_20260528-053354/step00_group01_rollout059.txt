import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    num_circles = 26
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 7) for x in np.linspace(0.1, 0.9, 7)])
    radii = np.full(num_circles, 0.05)  # Initial small radius

    # Objective function to minimize: negative of sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: circles must be inside unit square
    def inside_box(radii):
        return np.array([r <= 1 - r for r in radii])

    # Constraint: no two circles overlap
    def no_overlap(radii, centers):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=-1))
        return np.array([dists[i, j] >= radii[i] + radii[j] for i in range(num_circles) for j in range(i + 1)]).flatten()

    # Optimization bounds: all radii must be non-negative
    bounds = [(0, 1) for _ in range(num_circles)]

    # Objective function with radii as input
    def obj_with_radii(x, centers):
        return -np.sum(x)

    # Constraint function with radii as input
    def constraints_with_radii(x, centers):
        radii = x
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=-1))
        overlap = np.array([dists[i, j] >= radii[i] + radii[j] for i in range(num_circles) for j in range(i + 1)]).flatten()
        inside = np.array([r <= 1 - r for r in radii])
        return np.concatenate([overlap, inside])

    # Initial radii
    initial_radii = np.full(num_circles, 0.05)

    # Optimization
    result = minimize(obj_with_radii, initial_radii, method='SLSQP', bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints_with_radii, 'args': (centers,)})

    # Extract result
    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
