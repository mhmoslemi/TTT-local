import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place 26 circles on a grid
    n = 26
    grid = np.zeros((n, 2))
    for i in range(n):
        grid[i, 0] = (i % 5 + 0.5) / 5
        grid[i, 1] = (i // 5 + 0.5) / 5
    
    # Initial radii: all the same for the grid
    radii = np.ones(n) * 0.1
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints:
    # 1. All circles are within the square
    # 2. No two circles overlap
    # 3. Radii must be non-negative
    
    def constraint_1(radii):
        centers = grid
        return np.array([np.min(centers[:, 0] - radii[i]) for i in range(n)]) + np.array([np.min(1 - centers[:, 0] - radii[i]) for i in range(n)]) + np.array([np.min(centers[:, 1] - radii[i]) for i in range(n)]) + np.array([np.min(1 - centers[:, 1] - radii[i]) for i in range(n)])
    
    def constraint_2(radii):
        centers = grid
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = math.hypot(dx, dy)
                dists[j, i] = dists[i, j]
        return np.array([dists[i, j] - radii[i] - radii[j] for i in range(n) for j in range(i + 1, n)])
    
    constraints = [
        {'type': 'ineq', 'fun': lambda x: constraint_1(x)},
        {'type': 'ineq', 'fun': lambda x: constraint_2(x)},
        {'type': 'ineq', 'fun': lambda x: x}
    ]
    
    result = minimize(objective, radii, method='SLSQP', constraints=constraints, tol=1e-6)
    
    centers = grid
    sum_radii = result.fun * -1
    return (centers, result.x, sum_radii)
