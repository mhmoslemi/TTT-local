import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: 26 circles placed in a grid-like pattern
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radii: 0.01
    radii = np.full(26, 0.01)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective_function(x):
        nonlocal radii
        radii = x[:26]
        return -np.sum(radii)

    # Constraints: 1. All circles within the unit square
    def square_constraint(x):
        return np.array([x[i] <= 1 - x[i + 26] for i in range(26)])  # x[i] is radius, x[i + 26] is center x-coordinate
    square_con = {'type': 'ineq', 'fun': square_constraint}

    def square_y_constraint(x):
        return np.array([x[i] <= 1 - x[i + 26 + 26] for i in range(26)])  # x[i] is radius, x[i + 26 + 26] is center y-coordinate
    square_y_con = {'type': 'ineq', 'fun': square_y_constraint}

    # 2. No overlapping circles
    def circle_overlap_constraint(x):
        nonlocal radii, centers
        radii = x[:26]
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = x[i + 26]
            centers[i, 1] = x[i + 26 + 26]
        constraints = []
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist - (radii[i] + radii[j]))
        return np.array(constraints)
    overlap_con = {'type': 'ineq', 'fun': circle_overlap_constraint}

    # 3. Non-negative radii
    def non_negative_con(x):
        return x[:26]
    non_neg_con = {'type': 'ineq', 'fun': non_negative_con}

    # Combine constraints
    constraints = [square_con, square_y_con, overlap_con, non_neg_con]

    # Bounds: radius and center coordinates between 0 and 1
    bounds = [(0, 1) for _ in range(26)] + [(0, 1) for _ in range(26)] + [(0, 1) for _ in range(26)]

    # Initial guess: combine centers and radii
    x0 = np.concatenate((radii, centers.flatten()))

    # Run optimization
    result = minimize(objective_function, x0, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract results
    optimal_radii = result.x[:26]
    optimal_centers = np.reshape(result.x[26:], (26, 2))
    sum_radii = np.sum(optimal_radii)

    return (optimal_centers, optimal_radii, sum_radii)
