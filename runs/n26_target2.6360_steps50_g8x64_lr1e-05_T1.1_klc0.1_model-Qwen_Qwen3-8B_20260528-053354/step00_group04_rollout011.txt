import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1
    
    # Define objective function to maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Define constraint: each circle must be inside the unit square
    def square_constraint(radii):
        return np.sum((radii <= 0.5) * 1)  # Ensure radii <= 0.5 for circles to fit
    
    # Define constraint: circles do not overlap
    def non_overlap_constraint(centers, radii):
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 + 
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Define bounds for radii: non-negative and less than or equal to 0.5
    bounds = [(0, 0.5) for _ in range(num_circles)]
    
    # Optimization setup
    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': square_constraint},
            {'type': 'ineq', 'fun': lambda x: np.min(non_overlap_constraint(centers, x))}
        ],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )
    
    # Update centers based on optimization
    # Simple gradient-based improvement is not feasible, so this is a placeholder
    # In practice, this would need to be part of the optimization process with updated centers
    # For the purpose of this example, we use a basic grid layout
    centers = np.zeros((26, 2))
    i, j = 0, 0
    for r in result.x:
        centers[i, 0] = (j + 0.5) / 5
        centers[i, 1] = (i + 0.5) / 5
        i += 1
        if i == 5:
            i = 0
            j += 1
    
    # Return the calculated results
    return (centers, result.x, -result.fun)
