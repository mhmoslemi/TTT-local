import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        radii = np.clip(radii, 0, 1e6)  # Ensure non-negative radii
        dist = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dist[i, j] < radii[i] + radii[j]:
                    return False
        # Check if any circle is outside the unit square
        for i in range(num_circles):
            if centers[i, 0] < radii[i] or centers[i, 0] > 1 - radii[i]:
                return False
            if centers[i, 1] < radii[i] or centers[i, 1] > 1 - radii[i]:
                return False
        return True

    # Set bounds for radii
    bounds = [(0, 1e6) for _ in range(num_circles)]

    # Optimization settings
    options = {
        'ftol': 1e-8,
        'maxiter': 1000,
        'disp': False
    }

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        options=options
    )

    # Extract the optimal radii and centers
    radii_opt = result.x
    sum_radii_opt = -result.fun

    # Ensure all radii are non-negative and clip to maximum possible
    radii_opt = np.clip(radii_opt, 0, 1e6)

    # Recheck constraints once more to ensure validity
    if not constraints(radii_opt):
        warnings.warn("Optimization may have failed to satisfy all constraints.")

    return (centers, radii_opt, sum_radii_opt)
