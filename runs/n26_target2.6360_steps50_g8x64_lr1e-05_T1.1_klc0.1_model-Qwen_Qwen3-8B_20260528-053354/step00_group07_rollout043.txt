import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: spread out points in a grid
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1 + 0.05

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        constraints = []
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        for i in range(26):
            # Distance to edges
            constraints.append(centers[i, 0] - radii[i])
            constraints.append(1 - centers[i, 0] - radii[i])
            constraints.append(centers[i, 1] - radii[i])
            constraints.append(1 - centers[i, 1] - radii[i])
        return constraints

    def objective_wrapper(x):
        radii = x[:26]
        centers = np.array(x[26:])
        centers = centers.reshape(26, 2)
        return -np.sum(radii)

    def constraint_wrapper(x):
        radii = x[:26]
        centers = np.array(x[26:])
        centers = centers.reshape(26, 2)
        constraints = []
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]))
        for i in range(26):
            # Distance to edges
            constraints.append(centers[i, 0] - radii[i])
            constraints.append(1 - centers[i, 0] - radii[i])
            constraints.append(centers[i, 1] - radii[i])
            constraints.append(1 - centers[i, 1] - radii[i])
        return constraints

    # Initial guess
    x0 = np.concatenate([radii, centers.flatten()])

    # Bounds for radii and centers
    bounds = []
    for i in range(26):
        bounds.append((0, 0.5))  # radius
        bounds.append((0, 1))  # x
        bounds.append((0, 1))  # y

    # Optimization
    result = minimize(objective_wrapper, x0, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint_wrapper})

    if result.success:
        radii_opt = result.x[:26]
        centers_opt = result.x[26:].reshape(26, 2)
        sum_radii = np.sum(radii_opt)
        return (centers_opt, radii_opt, sum_radii)
    else:
        # Fallback to initial random guess if optimization fails
        return (centers, radii, np.sum(radii))
