import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    centers = np.array([[x / 7, y / 7] for y in range(7) for x in range(7)])
    centers = centers[:26]
    radii = np.ones(26) * 0.1  # Initial radius

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: Each circle is fully inside the square
    def inside_box(radii, centers):
        return np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]))

    # Constraint 2: No two circles overlap
    def no_overlap(radii, centers):
        dists = np.zeros((26, 26))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx ** 2 + dy ** 2)
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Combine constraints
    def constraints(radii, centers):
        return np.concatenate([inside_box(radii, centers), no_overlap(radii, centers)])

    # Optimization
    result = minimize(
        objective,
        radii,
        constraints={"type": "ineq", "fun": lambda r: constraints(r, centers)},
        bounds=[(0, 1) for _ in range(26)],
        method="SLSQP",
        options={"maxiter": 1000, "ftol": 1e-8}
    )

    # Extract result
    radii_result = result.x
    sum_radii = np.sum(radii_result)
    centers_final = centers[:26]

    return (centers_final, radii_result, sum_radii)
