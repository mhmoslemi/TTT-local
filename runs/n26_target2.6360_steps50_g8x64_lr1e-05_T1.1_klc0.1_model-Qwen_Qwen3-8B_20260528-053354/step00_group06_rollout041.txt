import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 26 circles evenly spaced
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    
    def objective_function(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraint_inside(radii, centers):
        for i in range(26):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        return True
    
    def constraint_no_overlap(radii, centers):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    def bounds(radii):
        return [(0.0, 1.0) for _ in range(26)]
    
    def constraints(radii):
        return [
            {'type': 'ineq', 'fun': lambda r: constraint_inside(r, centers)},
            {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
        ]
    
    result = minimize(
        objective_function,
        radii,
        method='SLSQP',
        bounds=bounds(radii),
        constraints=constraints(radii),
        options={'maxiter': 100, 'ftol': 1e-8}
    )
    
    centers = np.random.rand(26, 2)
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
