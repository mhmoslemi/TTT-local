import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid with some random perturbation
    num_circles = 26
    centers = np.array([[i / np.sqrt(num_circles), j / np.sqrt(num_circles)] for i in range(num_circles) for j in range(num_circles)])
    centers = np.random.rand(num_circles, 2) * 0.5 + 0.25  # initial random placement inside the square

    # Radius initialization: small positive values
    radii = np.random.rand(num_circles) * 0.05

    def objective_function(radii):
        # Objective is to maximize sum of radii, so minimize negative sum
        return -np.sum(radii)

    def constraint_overlap(i, j, radii):
        def constraint(x):
            # x is a vector of radii, but we're using a scalar here to minimize dependency
            # For each pair of circles (i, j), check if their distance >= sum of radii
            # To simplify, we treat the radii as fixed and check for a fixed pair (i, j)
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            distance = np.sqrt(dx**2 + dy**2)
            return distance - (radii[i] + radii[j])
        return constraint

    # Add constraints for each pair of circles to ensure no overlap
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            # Use a lambda to capture i and j inside the constraint
            def add_constraint_pair(i, j):
                return {'type': 'ineq', 'fun': constraint_overlap(i, j, radii)}
            cons = add_constraint_pair(i, j)

    # Add constraints for each circle to be fully inside the square
    def constraint_inside(x, i):
        # x is the radius of circle i
        r = x[i]
        # Check if circle i is inside the square
        return 1 - r - centers[i, 0] >= 0  # right side
    for i in range(num_circles):
        # Create a constraint for the left side
        cons_left = {'type': 'ineq', 'fun': lambda x, i=i: 1 - r - centers[i, 0] >= 0}
        cons_right = {'type': 'ineq', 'fun': lambda x, i=i: centers[i, 0] - r >= 0}
        cons_top = {'type': 'ineq', 'fun': lambda x, i=i: 1 - r - centers[i, 1] >= 0}
        cons_bottom = {'type': 'ineq', 'fun': lambda x, i=i: centers[i, 1] - r >= 0}
        cons.append(cons_left)
        cons.append(cons_right)
        cons.append(cons_top)
        cons.append(cons_bottom)

    # Initial guess for the radii
    x0 = radii.copy()

    # Set up optimization options
    options = {'maxiter': 1000, 'disp': False}
    result = minimize(objective_function, x0, constraints=cons, method='SLSQP', options=options)

    # Extract results
    optimal_radii = result.x
    optimal_centers = centers.copy()

    # Ensure all radii are non-negative
    optimal_radii = np.maximum(optimal_radii, 0)

    # Final check for overlap and boundaries (simplified for the purpose of the function)
    for i in range(num_circles):
        r = optimal_radii[i]
        if not (r <= 1 - optimal_centers[i, 0] and r <= 1 - optimal_centers[i, 1] and
                r <= optimal_centers[i, 0] and r <= optimal_centers[i, 1]):
            optimal_radii[i] = 0
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = optimal_centers[i, 0] - optimal_centers[j, 0]
            dy = optimal_centers[i, 1] - optimal_centers[j, 1]
            distance = np.sqrt(dx**2 + dy**2)
            if distance < optimal_radii[i] + optimal_radii[j]:
                optimal_radii[i] = (distance - optimal_radii[j]) / 2
                optimal_radii[j] = (distance - optimal_radii[i]) / 2

    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
