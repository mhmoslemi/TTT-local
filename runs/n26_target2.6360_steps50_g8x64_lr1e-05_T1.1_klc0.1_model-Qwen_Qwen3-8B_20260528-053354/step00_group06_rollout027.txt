import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (randomly spread out)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)
    
    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize radii
    
    def constraints(radii, centers):
        # Ensure all circles are inside the unit square
        def in_square(r_i, x, y):
            return r_i <= x - r_i and x + r_i <= 1.0 and r_i <= y - r_i and y + r_i <= 1.0
        
        # Ensure no two circles overlap
        def no_overlap(r_i, r_j, x_i, y_i, x_j, y_j):
            dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
            return dist >= r_i + r_j
        
        constraints = []
        for i in range(26):
            for j in range(i + 1, 26):
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: no_overlap(r[i], r[j], centers[i, 0], centers[i, 1], centers[j, 0], centers[j, 1])})
        
        for i in range(26):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: in_square(r[i], centers[i, 0], centers[i, 1])})
        
        return constraints
    
    # Optimize using scipy
    bounds = [(0.0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, args=(centers,), method='SLSQP', bounds=bounds, constraints=constraints(radii, centers), tol=1e-6)
    
    # Extract results
    optimal_radii = result.x
    optimal_centers = centers.copy()
    sum_radii = np.sum(optimal_radii)
    
    return (optimal_centers, optimal_radii, sum_radii)
