import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: 26 circles arranged in a grid with spacing
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = col * 0.25 + 0.125
        y = row * 0.25 + 0.125
        centers[i] = [x, y]
        radii[i] = 0.04

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Circle inside the unit square
        for i in range(26):
            if radii[i] > 0:
                if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                    return 1
                if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                    return 1
        # Circle not overlapping
        dists = pdist(centers)
        for d, (i, j) in zip(dists, zip(range(26), range(26))):
            if i >= j:
                continue
            if d < radii[i] + radii[j]:
                return 1
        return 0

    def inequality_constraints(radii):
        return np.array([radii[i] for i in range(26)]).reshape(26, 1) >= 0

    # Optimization
    bounds = [(0, 0.5) for _ in range(26)]
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraints},
            {'type': 'ineq', 'fun': inequality_constraints}
        ],
        options={'ftol': 1e-9, 'maxiter': 500}
    )

    if not result.success:
        print("Optimization did not converge.")
        return (centers, radii, np.sum(radii))

    radii_opt = result.x
    # Update centers based on optimization (optional for more accurate results)
    # For simplicity, centers are kept static
    centers_opt = centers
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
