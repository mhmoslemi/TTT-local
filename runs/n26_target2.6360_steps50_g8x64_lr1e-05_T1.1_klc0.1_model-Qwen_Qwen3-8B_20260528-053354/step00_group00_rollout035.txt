import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid
    initial_centers = np.array([[i / 6 + 0.05, j / 6 + 0.05] for i in range(6) for j in range(6)])
    initial_centers = initial_centers[:26]
    initial_radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint: all circles are inside the square
        for i in range(26):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - 2 * centers[i, 0] or radii[i] > 1 - 2 * centers[i, 1]:
                return False
        # Constraint: no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Objective function and constraints
    def objective_func(radii):
        return -np.sum(radii)

    def constraint_func(radii):
        return constraints(radii, initial_centers)

    try:
        result = minimize(
            objective_func,
            initial_radii,
            method='SLSQP',
            constraints={'type': 'ineq', 'fun': constraint_func}
        )
    except:
        # Fallback if optimization fails
        result = {
            'x': initial_radii,
            'success': True
        }

    radii = result['x']
    sum_radii = -result['fun']
    centers = initial_centers

    return (centers, radii, sum_radii)
