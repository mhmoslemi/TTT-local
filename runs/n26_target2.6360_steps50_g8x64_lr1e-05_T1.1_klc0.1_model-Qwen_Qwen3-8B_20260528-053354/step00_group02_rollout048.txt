import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.array([[i / 5 + 0.1, j / 5 + 0.1] for j in range(5) for i in range(5)])
    radii = np.full(num_circles, 0.05)
    initial_sum = sum(radii)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -sum(radii)

    # Constraints: circles must be inside the unit square
    def square_boundary(radii):
        centers = np.array([[(i + 0.5) * np.sqrt(2 * 0.05 * radii[i]) for i in range(num_circles)] for _ in range(2)])
        return np.concatenate([radii, np.zeros(num_circles)])

    # Constraint functions for non-overlapping circles
    def non_overlap(radii):
        r = radii
        centers = np.array([[(i + 0.5) * np.sqrt(2 * 0.05 * r[i]) for i in range(num_circles)] for _ in range(2)])
        dists = np.sqrt((centers[0] - centers[0][:, np.newaxis])**2 + (centers[1] - centers[1][:, np.newaxis])**2)
        return dists - (r[np.newaxis, :] + r[:, np.newaxis])

    # Optimization problem
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': lambda r: r},
        {'type': 'ineq', 'fun': lambda r: np.min(r - square_boundary(r))},
        {'type': 'ineq', 'fun': lambda r: np.min(non_overlap(r))}
    ]

    # Run optimization
    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    centers = np.array([[(i + 0.5) * np.sqrt(2 * 0.05 * result.x[i]) for i in range(num_circles)] for _ in range(2)])
    sum_radii = result.fun * -1

    return (centers, result.x, sum_radii)
