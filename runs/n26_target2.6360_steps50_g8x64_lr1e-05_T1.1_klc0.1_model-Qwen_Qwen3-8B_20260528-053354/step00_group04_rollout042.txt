import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: evenly spaced centers with small radii
    centers = np.array([[x / 5 + 0.25, y / 5 + 0.25] for x in range(5) for y in range(5)])
    radii = np.full(25, 0.04)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add one more circle in the center
    radii = np.append(radii, 0.04)
    centers = centers[:26]
    radii = radii[:26]
    sum_radii = np.sum(radii)

    # Objective function: minimize negative sum of radii
    def objective(x):
        nonlocal sum_radii
        # x contains all variables as a 1D array: [x1, y1, r1, x2, y2, r2, ...]
        x_flat = x.reshape((-1, 3))
        centers_new = x_flat[:, :2]
        radii_new = x_flat[:, 2]
        sum_radii_new = np.sum(radii_new)
        # Constraints: circle inside the unit square
        inside = np.all((radii_new <= centers_new) & (centers_new <= 1 - radii_new), axis=1)
        # Constraints: circles not overlapping
        dists = np.sqrt(np.sum((centers_new[:, np.newaxis, :] - centers_new[np.newaxis, :, :]) ** 2, axis=2))
        overlap = np.triu(dists - radii_new[:, np.newaxis] - radii_new[np.newaxis, :], k=1)
        overlap = np.all(overlap >= 0)
        # Objective to minimize (we minimize negative sum of radii)
        return -sum_radii_new

    # Equality constraints: x, y, r all within bounds (but we handle this with bounds)
    # Inequality constraints: all circles inside the unit square and no overlaps
    def constraint_inside(x):
        x_flat = x.reshape((-1, 3))
        centers_new = x_flat[:, :2]
        radii_new = x_flat[:, 2]
        return np.all((radii_new <= centers_new) & (centers_new <= 1 - radii_new))

    def constraint_overlap(x):
        x_flat = x.reshape((-1, 3))
        centers_new = x_flat[:, :2]
        radii_new = x_flat[:, 2]
        dists = np.sqrt(np.sum((centers_new[:, np.newaxis, :] - centers_new[np.newaxis, :, :]) ** 2, axis=2))
        return np.all(dists - radii_new[:, np.newaxis] - radii_new[np.newaxis, :] >= 0)

    # Initial guess flattened
    x0 = np.array([centers[i][0] for i in range(26)] + [centers[i][1] for i in range(26)] + [radii[i] for i in range(26)])
    x0 = x0.reshape((-1, 3))

    # Bounds: x and y between 0 and 1; radii between 0 and 0.5
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])

    # Constraints: all circles inside the unit square and no overlaps
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization
    result = minimize(objective, x0.flatten(), method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6, options={'maxiter': 1000})

    # Extract solution
    x_sol = result.x.reshape((-1, 3))
    centers_sol = x_sol[:, :2]
    radii_sol = x_sol[:, 2]
    sum_radii_sol = -result.fun

    return (centers_sol, radii_sol, sum_radii_sol)
