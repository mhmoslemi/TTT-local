import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: distribute centers evenly in the square
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)  # Initial radius

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Ensure all circles are fully inside the unit square
        for i in range(26):
            r = radii[i]
            if centers[i, 0] - r < 0 or centers[i, 0] + r > 1 or centers[i, 1] - r < 0 or centers[i, 1] + r > 1:
                return False
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def objective_with_constraints(radii):
        if not constraints(radii):
            return float('inf')  # Penalize invalid configurations
        return -np.sum(radii)

    result = minimize(objective_with_constraints, radii, method='SLSQP', bounds=[(0, 1)]*26)
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
