```python
import numpy as np
from scipy.optimize import minimize
import warnings

# Helper function to calculate the squared distance between two points
def squared_distance(p1, p2):
    return (p1[0] - p2[0])**2 + (p1[1] - p2[1])**2

# Objective function: minimize negative sum of radii (maximize sum)
def objective(radii, centers):
    return -np.sum(radii)

# Inequality constraint: each circle must be inside the unit square
def square_constraint(radii, centers):
    return np.hstack([
        centers[:, 0] - radii,
        1 - centers[:, 0] - radii,
        centers[:, 1] - radii,
        1 - centers[:, 1] - radii
    ])

# Equality constraint: no two circles overlap
def circle_overlap_constraint(radii, centers):
    n = len(radii)
    constraints = []
    for i in range(n):
        for j in range(i + 1, n):
            d_sq = squared_distance(centers[i], centers[j])
            r_sum = radii[i] + radii[j]
            constraints.append(d_sq - r_sum**2)
    return np.array(constraints)

# Constraint handling
def constraints(radii, centers):
    constraints = []
    constraints.append({'type': 'ineq', 'fun': lambda x: square_constraint(x, centers)})
    constraints.append({'type': 'ineq', 'fun': lambda x: circle_overlap_constraint(x, centers)})
    return constraints

# Initial guess: place circles in a grid pattern
def initial_guess(num_circles):
    grid = np.zeros((num_circles, 2))
    idx = 0
    for i in range(5):
        for j in range(5):
            if idx >= num_circles:
                break
            grid[idx, 0] = (i + 0.5) / 4
            grid[idx, 1] = (j + 0.5) / 4
            idx += 1
    return grid

# Main function to run the packing
def run_packing():
    num_circles = 26
    initial_centers = initial_guess(num_circles)
    
    # Initial guess for radii: equal for all
    initial_radii = np.full(num_circles, 0.1)
    
    # Define optimization variables: radii
    n_vars = num_circles
    
    # Define bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(n_vars)]
    
    # Optimization problem
    result = minimize(
        fun=objective,
        x0=initial_radii,
        bounds=bounds,
        constraints=constraints,
        args=(initial_centers,),
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )
    
    # Extract results
    centers = initial_centers
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
```
