import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import itertools

def run_packing():
    # Initial guess: place 26 circles in a grid pattern
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles) * 0.05
    sum_radii = np.sum(radii)

    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        # Convert to centers and radii
        centers_flat = x[:num_circles*2].reshape(num_circles, 2)
        radii_flat = x[num_circles*2:].reshape(num_circles)
        # Check if circles are inside the unit square
        inside = np.all((radii_flat <= centers_flat[:, 0]) & (centers_flat[:, 0] <= 1 - radii_flat) &
                        (radii_flat <= centers_flat[:, 1]) & (centers_flat[:, 1] <= 1 - radii_flat))
        if not inside:
            return np.inf
        # Check if circles overlap
        distances = squareform(pdist(centers_flat))
        for i, j in itertools.combinations(range(num_circles), 2):
            if distances[i, j] < radii_flat[i] + radii_flat[j]:
                return np.inf
        # Objective: maximize sum of radii
        return -np.sum(radii_flat)

    # Initial guess in flat format
    flat_initial = np.concatenate([centers.flatten(), radii.flatten()])
    result = minimize(objective, flat_initial, method='L-BFGS-B', bounds=[(0, 1) for _ in range(num_circles*2)] +
                                   [(0, 0.5) for _ in range(num_circles)])
    # Extract the result
    flat_result = result.x
    centers = flat_result[:num_circles*2].reshape(num_circles, 2)
    radii = flat_result[num_circles*2:].reshape(num_circles)
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
