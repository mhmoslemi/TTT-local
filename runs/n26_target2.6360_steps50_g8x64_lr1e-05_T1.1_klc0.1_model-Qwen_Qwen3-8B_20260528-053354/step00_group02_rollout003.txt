import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: evenly spaced grid of 26 points with small radii
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = (i + 0.5) / np.sqrt(n)
        centers[i, 1] = (i + 0.5) / np.sqrt(n)
    radii = np.full(n, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Check if all circles are inside the square
        for i in range(n):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Check if circles do not overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def constraint_func(radii):
        if not constraints(radii):
            return np.inf
        return 0

    # Initial guess
    initial_guess = radii.copy()

    # Optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=[(0, 1.0) for _ in range(n)],
        constraints={'type': 'ineq', 'fun': constraint_func}
    )

    # Extract results
    radii_opt = result.x
    sum_radii = -result.fun

    # Ensure the final solution is feasible
    if not constraints(radii_opt):
        sum_radii = np.sum(radii_opt)

    centers_opt = centers.copy()
    return (centers_opt, radii_opt, sum_radii)
