import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 26 circles placed in a grid pattern inside the unit square
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Start with a square grid layout, spacing is 0.5 to keep them apart initially
    spacing = 0.5
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * spacing + spacing / 2
        centers[i, 1] = row * spacing + spacing / 2
        radii[i] = 0.05
    
    # Objective function: minimize negative sum of radii (maximize sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints: 1. Circles within the unit square, 2. No overlap between circles
    def constraint_circle_inside(radii):
        return np.array([r <= 1 - x - r and r <= 1 - y - r and r <= x + r and r <= y + r for r, x, y in zip(radii, centers[:, 0], centers[:, 1])])
    
    def constraint_no_overlap(radii):
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists[i, j] = dist - (radii[i] + radii[j])
                dists[j, i] = dist - (radii[i] + radii[j])
        return np.ndarray.flatten(dists)
    
    # Optimization
    bounds = [(0, 1 - 0.01) for _ in range(num_circles)]  # Radii must be small enough to allow some space
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r)}
    ]
    
    result = minimize(objective, radii, bounds=bounds, constraints=cons)
    
    # Extract results
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii_final = np.sum(radii_final)
    
    return (centers_final, radii_final, sum_radii_final)
