import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern with equal spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0
        radii[i] = 0.05  # Initial small radius

    def objective(r):
        return -np.sum(r)  # Minimize negative sum to maximize sum

    def constraints(r):
        # Ensure all circles are inside the unit square
        for i in range(num_circles):
            if r[i] < 0:
                return np.inf
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return np.inf
            if centers[i, 0] - r[i] < 0 or centers[i, 0] + r[i] > 1 or centers[i, 1] - r[i] < 0 or centers[i, 1] + r[i] > 1:
                return np.inf
        return 0

    bounds = [(0, 0.5) for _ in range(num_circles)]

    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-6)
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
