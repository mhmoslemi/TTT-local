import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: equally spaced in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
        radii[i] = 0.05

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints
    def constraint_center_inside(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        return np.array([center[0] - radius >= 0 for center, radius in zip(centers_flat, radii_flat)] + 
                       [center[0] + radius <= 1 for center, radius in zip(centers_flat, radii_flat)] + 
                       [center[1] - radius >= 0 for center, radius in zip(centers_flat, radii_flat)] + 
                       [center[1] + radius <= 1 for center, radius in zip(centers_flat, radii_flat)])
    
    def constraint_no_overlap(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        dists = np.array([[np.linalg.norm(centers_flat[i] - centers_flat[j]) for j in range(26)] for i in range(26)])
        return np.array([dists[i, j] - radii_flat[i] - radii_flat[j] for i in range(26) for j in range(i+1, 26)])

    # Initial guess for x: centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up the optimization problem
    cons = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Bounds: centers are in [0, 1], radii are non-negative
    bounds = [(0, 1) for _ in range(52)] + [(0, 1) for _ in range(26)]

    # Optimize
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract final centers and radii
    optimized_centers = result.x[:52].reshape(26, 2)
    optimized_radii = result.x[52:]
    total_sum = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, total_sum)
