import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 26 circles arranged in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all the same
    radii = np.ones(26) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        # Ensure all circles are inside the unit square
        x = centers[:, 0]
        y = centers[:, 1]
        return np.array([r - (x[i] - r) for i, r in enumerate(radii)] + 
                        [r - (1 - x[i] - r) for i, r in enumerate(radii)] +
                        [r - (y[i] - r) for i, r in enumerate(radii)] +
                        [r - (1 - y[i] - r) for i, r in enumerate(radii)])

    def constraint_no_overlap(radii):
        # Ensure no two circles overlap
        dists = np.zeros(26 * 25 // 2)
        idx = 0
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[idx] = np.sqrt(dx*dx + dy*dy) - (radii[i] + radii[j])
                idx += 1
        return dists

    # Optimization problem setup
    bounds = [(0.0, 0.5) for _ in range(26)]  # Radii can't exceed 0.5
    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract result
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
