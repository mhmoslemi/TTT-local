import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid points
    def initial_guess():
        centers = []
        for i in range(26):
            x = i // 5 + 0.5 * (i % 5) / 5
            y = 0.5
            centers.append([x, y])
        return np.array(centers)

    def objective(radii):
        # Objective function: minimize negative sum of radii (maximize sum)
        return -np.sum(radii)

    def constraints(centers, radii):
        # Constraints: circles must be within the unit square
        bounds = np.array([(r, 1 - r) for r in radii])
        lb = np.vstack([centers[:, 0] - radii, centers[:, 1] - radii]).T
        ub = np.vstack([centers[:, 0] + radii, centers[:, 1] + radii]).T
        return np.array([lb, ub])

    # Optimization setup
    centers = initial_guess()
    radii = np.ones(26) * 0.1  # Initial guess for radii

    # Define constraints for optimization
    cons = []

    # Constraint: circle must be within unit square
    def square_constraint(center, radii_index):
        r = radii[radii_index]
        return np.array([center[0] - r, center[1] - r, 1 - center[0] - r, 1 - center[1] - r])

    for i in range(26):
        cons.append({'type': 'ineq', 'fun': lambda x, i=i: x[2*i] - x[2*i + 1] - x[2*i + 2] - x[2*i + 3]})

    # Constraint: no overlap between circles
    def no_overlap_constraint(centers, radii):
        dist_matrix = cdist(centers, centers)
        for i in range(26):
            for j in range(i + 1, 26):
                dist = dist_matrix[i, j]
                if dist < radii[i] + radii[j]:
                    return dist - radii[i] - radii[j]
        return 0

    cons.append({'type': 'ineq', 'fun': no_overlap_constraint})

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=cons)

    # Extract the result
    centers = np.array([(x, 0.5) for x in np.linspace(0, 1, 5)] * 5)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
