import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        x = i / (n - 1)
        y = 0.5
        centers[i] = [x, y]
    
    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints
    def constraint_circle_inside(radii):
        # Constraint that all circles are inside the square
        return np.sum((radii > 0) * (np.all(centers - radii[:, np.newaxis] >= 0, axis=1) &
                                     np.all(1 - centers - radii[:, np.newaxis] >= 0, axis=1)))

    def constraint_no_overlap(radii):
        # Constraint that circles are non-overlapping
        dist = cdist(centers, centers)
        min_dist = np.min(dist[np.triu_indices(n, 1)])
        return min_dist - np.sum(radii[np.triu_indices(n, 1)] + radii[np.tril_indices(n, -1)]).mean()

    # Optimization
    bounds = [(0, 1.0) for _ in range(n)]
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r)}
    ]
    result = minimize(objective, centers.flatten(), bounds=bounds, constraints=cons, method='SLSQP')

    # Extract results
    radii = result.x
    centers = np.reshape(radii, (n, 2))
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
