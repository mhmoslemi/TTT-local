import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid of 26 points in [0,1]x[0,1]
    centers = np.array([[i/np.sqrt(26), j/np.sqrt(26)] for i in range(26) for j in range(26)])
    centers = centers[:26]  # Truncate to 26 points
    radii = np.full(26, 0.1)  # Initial guess for radii

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        # For each circle, check that its center is within [r, 1-r] in both x and y
        r = radii.reshape(-1, 1)
        inside = (centers[:, 0] >= r) & (centers[:, 0] <= 1 - r) & \
                 (centers[:, 1] >= r) & (centers[:, 1] <= 1 - r)
        inside = np.all(inside, axis=0)
        # Constraint 2: No two circles overlap
        distances = cdist(centers, centers)
        radii_sum = r + r.T
        overlap = distances < radii_sum
        overlap[range(26), range(26)] = False  # No self-overlap
        return np.all(~overlap)

    def bounds(radii):
        return [(0, 1 - np.min(centers, axis=0))] * 26

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=bounds(radii)
    )

    centers_final = centers
    radii_final = result.x
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
