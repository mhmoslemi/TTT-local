import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    n = 26
    # Initial guess for circle centers and radii
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    for i in range(n):
        row = i // 5
        col = i % 5
        x = 0.2 + col * 0.4
        y = 0.2 + row * 0.4
        centers[i] = [x, y]
        radii[i] = 0.05

    def objective_func(x):
        radii = x[:n]
        centers = x[n:].reshape((n, 2))
        # Apply constraints
        for i in range(n):
            if radii[i] < 0:
                return float('inf')
            x_i, y_i = centers[i]
            if not (radii[i] <= x_i <= 1 - radii[i] and radii[i] <= y_i <= 1 - radii[i]):
                return float('inf')
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                if distance < radii[i] + radii[j] - 1e-8:
                    return float('inf')
        return -np.sum(radii)

    # Initial guess
    x0 = np.concatenate((radii, centers.flatten()))

    # Bounds: radii between 0 and 0.5, centers between 0 and 1
    bounds = [(0, 0.5)] * n + [(0, 1)] * (2 * n)

    # Optimization
    result = minimize(
        objective_func,
        x0=x0,
        method='L-BFGS-B',
        bounds=bounds,
        tol=1e-6,
        options={'disp': False}
    )

    optimized_radii = result.x[:n]
    optimized_centers = result.x[n:].reshape((n, 2))

    # Clamp radii if out of bounds
    for i in range(n):
        x, y = optimized_centers[i]
        if not (optimized_radii[i] <= x <= 1 - optimized_radii[i] and optimized_radii[i] <= y <= 1 - optimized_radii[i]):
            optimized_radii[i] = 0

    # Adjust for any overlap
    for i in range(n):
        for j in range(i + 1, n):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            distance = math.hypot(dx, dy)
            if distance < optimized_radii[i] + optimized_radii[j] - 1e-8:
                optimized_radii[i] = 0

    sum_radii = np.sum(optimized_radii)
    return (optimized_centers, optimized_radii, sum_radii)
