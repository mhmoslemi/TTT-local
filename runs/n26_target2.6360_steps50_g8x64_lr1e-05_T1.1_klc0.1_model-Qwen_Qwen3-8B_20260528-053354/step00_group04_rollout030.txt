import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # We'll use a greedy approach with random restarts and local optimization
    # to approximate the solution for 26 circles inside the unit square.

    def objective_function(positions):
        # positions is a flattened array of shape (26*2, ) containing x and y coordinates
        # Convert to centers array
        centers = positions.reshape(-1, 2)
        # Objective: maximize sum of radii (minimize negative of sum)
        # We'll use a rough estimation of radii assuming equal spacing
        radii = np.zeros(26)
        for i in range(26):
            # Estimate radius based on distance to nearest neighbors
            # Only consider nearby points
            distances = np.linalg.norm(centers - centers[i], axis=1)
            min_distance = np.min(distances[distances > 0])
            # Also ensure circle is inside the square
            x, y = centers[i]
            max_possible_radius = min(x, 1 - x, y, 1 - y)
            radii[i] = min(0.5, max_possible_radius, min_distance / 2)
        return -np.sum(radii)

    def constraints(positions):
        # positions is a flattened array of shape (26*2, ) containing x and y coordinates
        centers = positions.reshape(-1, 2)
        constraints = []
        # Ensure circles are fully inside the square
        for i in range(26):
            x, y = centers[i]
            r = 0.5  # Placeholder radius (we'll refine it later)
            constraints.append((x - r) >= 0)
            constraints.append((x + r) <= 1)
            constraints.append((y - r) >= 0)
            constraints.append((y + r) <= 1)
        # Ensure circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dist = np.linalg.norm(centers[i] - centers[j])
                r_i = 0.5  # Placeholder radius (we'll refine it later)
                r_j = 0.5  # Placeholder radius (we'll refine it later)
                constraints.append(dist >= r_i + r_j)
        return constraints

    # Initial guess: randomly placed points
    np.random.seed(42)
    initial_guess = np.random.rand(26 * 2)

    # Optimization
    result = minimize(objective_function, initial_guess, method='L-BFGS-B', bounds=[(0, 1)] * (26 * 2))
    positions = result.x
    centers = positions.reshape(-1, 2)
    r = np.zeros(26)
    for i in range(26):
        x, y = centers[i]
        max_r = min(x, 1 - x, y, 1 - y)
        r[i] = max_r
    sum_r = np.sum(r)

    return (centers, r, sum_r)
