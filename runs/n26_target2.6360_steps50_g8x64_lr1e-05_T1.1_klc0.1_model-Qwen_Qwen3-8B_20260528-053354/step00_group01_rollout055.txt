import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers: spread them out
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 26) for y in np.linspace(0.1, 0.9, 26)])
    radii = np.full(26, 0.1)  # Start with small radii

    def objective(radii):
        return -np.sum(radii)  # Maximize the sum of radii

    def constraints(radii):
        # Ensure circles are within the unit square and do not overlap
        # 1. Boundary constraints
        for i in range(26):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # 2. Pairwise distance constraints
        for i in range(26):
            for j in range(i + 1, 26):
                dist_sq = (centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2
                if dist_sq < (radii[i] + radii[j])**2:
                    return False
        return True

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints={'type': 'ineq', 'fun': constraints},
        bounds=[(0, 1) for _ in range(26)]
    )

    # Extract results
    final_radii = result.x
    final_centers = centers
    sum_r = np.sum(final_radii)

    return (final_centers, final_radii, sum_r)
