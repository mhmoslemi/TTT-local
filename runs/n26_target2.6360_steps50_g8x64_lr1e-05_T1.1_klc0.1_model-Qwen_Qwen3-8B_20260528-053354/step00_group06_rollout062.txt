import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid-like fashion
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        x = 0.1 + col * 0.2
        y = 0.1 + row * 0.2
        centers[i] = [x, y]
        radii[i] = 0.05

    # Objective function to minimize (negative sum of radii)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint1(radii):
        # Ensure circles fit inside the unit square
        centers = np.zeros((26, 2))
        for i in range(26):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf
        return 0

    def constraint2(radii):
        # Ensure circles do not overlap
        centers = np.zeros((26, 2))
        for i in range(26):
            x, y = centers[i]
            for j in range(i + 1, 26):
                dx = centers[j, 0] - centers[i, 0]
                dy = centers[j, 1] - centers[i, 1]
                if np.sqrt(dx**2 + dy**2) < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Bounds for radii
    bounds = [(0, 0.5) for _ in range(26)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint1},
            {'type': 'ineq', 'fun': constraint2}
        ],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    sum_radii = -result.fun
    return (centers, np.array(result.x), sum_radii)
