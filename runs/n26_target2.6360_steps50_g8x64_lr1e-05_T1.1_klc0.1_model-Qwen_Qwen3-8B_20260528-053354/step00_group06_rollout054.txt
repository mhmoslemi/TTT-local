import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced grid with some small radii
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.01)
    
    # Initial positions: grid layout
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col / 4 + 0.125
        centers[i, 1] = row / 4 + 0.125

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(x):
        return -sum(x[26:])

    # Constraints: each circle must be inside the unit square
    def in_square(x):
        return np.all((x[26:] <= x[:26] - np.ones(num_circles)) &
                      (x[26:] <= 1 - x[:26]) &
                      (x[26:] >= x[:26] + np.ones(num_circles)) &
                      (x[26:] >= 0) &
                      (x[26:] >= 1 - x[:26]))

    # Constraints: no two circles overlap
    def no_overlap(x):
        distances = pdist(centers)
        sums = x[26:] + x[26:].reshape(-1, 1)
        return np.all(distances >= sums[np.triu_indices(num_circles, 1)])

    # Combine constraints
    def constraints(x):
        return np.array([in_square(x), no_overlap(x)])

    # Bounds: all circles have non-negative radii
    bounds = [(0, 1)] * (26 + num_circles)

    # Optimization
    result = minimize(
        objective,
        np.concatenate((centers.flatten(), radii)),
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6
    )

    # Extract results
    optimized_centers = result.x[:26*num_circles].reshape(num_circles, 2)
    optimized_radii = result.x[26*num_circles:]
    sum_radii = sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
