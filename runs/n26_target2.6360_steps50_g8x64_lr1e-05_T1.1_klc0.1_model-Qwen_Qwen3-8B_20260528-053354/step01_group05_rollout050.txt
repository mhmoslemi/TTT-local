import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    np.random.seed(42)
    centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraint_inside(x):
        # x is a flattened array of radii
        r = x.reshape(-1)
        for i in range(26):
            if r[i] > 0:
                if centers[i, 0] - r[i] < 0 or centers[i, 0] + r[i] > 1:
                    return -np.inf
                if centers[i, 1] - r[i] < 0 or centers[i, 1] + r[i] > 1:
                    return -np.inf
        return np.sum(r)

    def constraint_overlap(x):
        # x is a flattened array of radii
        r = x.reshape(-1)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return -np.inf
        return np.sum(r)

    # Try a few random initial configurations
    best_sum = 0
    best_centers = np.zeros((26, 2))
    best_radii = np.zeros(26)

    for _ in range(10):
        centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
        radii = np.full(26, 0.05)
        # Optimize radii with constraints
        result = minimize(
            objective,
            radii,
            constraints=[
                {'type': 'ineq', 'fun': constraint_inside},
                {'type': 'ineq', 'fun': constraint_overlap}
            ],
            method='SLSQP'
        )
        if result.success:
            current_centers = centers
            current_radii = result.x
            current_sum = np.sum(current_radii)
            if current_sum > best_sum:
                best_sum = current_sum
                best_centers = current_centers
                best_radii = current_radii

    return (best_centers, best_radii, best_sum)
