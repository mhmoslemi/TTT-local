import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid with some spacing
    def initial_guess():
        centers = np.zeros((26, 2))
        for i in range(26):
            row = i // 5
            col = i % 5
            centers[i, 0] = 0.2 + col * 0.25
            centers[i, 1] = 0.2 + row * 0.25
        return centers

    # Objective function: minimize negative sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraints: circles must be inside the square and non-overlapping
    def constraints(radii, centers):
        n = len(radii)
        cons = []

        # Circle inside the square
        for i in range(n):
            r = radii[i]
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r - 0.0})  # r >= 0
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1.0 - r - centers[i, 0]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1.0 - r - centers[i, 1]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: centers[i, 0] - r})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: centers[i, 1] - r})

        # Non-overlapping constraint
        for i in range(n):
            for j in range(i + 1, n):
                def dist_func(radii, i=i, j=j):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    return np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
                cons.append({'type': 'ineq', 'fun': dist_func})

        return cons

    # Run optimization
    centers = initial_guess()
    radii = np.full(26, 0.1)
    bounds = [(0.0, 0.5) for _ in range(26)]

    result = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        bounds=bounds,
        constraints=constraints(radii, centers),
        method='SLSQP'
    )

    return (centers, result.x, -result.fun)
