import numpy as np
from scipy.optimize import minimize

def run_packing():
    num_circles = 26
    # Initial guess: evenly spaced circles in a grid
    centers = np.zeros((num_circles, 2))
    indices = np.arange(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (indices[i] % 5 + 0.5) / 5
        centers[i, 1] = (indices[i] // 5 + 0.5) / 5

    # Initial radii: uniform
    radii = np.full(num_circles, 1 / 10)

    def objective_func(x):
        return -np.sum(x)

    def constraint_radius(x, i):
        return x[i] - 1e-8  # Ensure radii are non-negative

    def constraint_position(x, i, center_x, center_y):
        return x[i] - (1 - center_x - x[i])  # x[i] <= 1 - center_x - x[i]

    def constraint_position2(x, i, center_x, center_y):
        return x[i] - (1 - center_y - x[i])  # x[i] <= 1 - center_y - x[i]

    def constraint_overlap(x, centers, radii):
        total = 0.0
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                total += max(0, dist - (radii[i] + radii[j]))
        return total

    # Constraints
    constraints = []

    # Non-negative radii
    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_radius(x, i)})

    # Position constraint for x-coordinate
    for i in range(num_circles):
        center_x = centers[i, 0]
        center_y = centers[i, 1]
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i, cx=center_x, cy=center_y: constraint_position(x, i, cx, cy)})

    # Position constraint for y-coordinate
    for i in range(num_circles):
        center_x = centers[i, 0]
        center_y = centers[i, 1]
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i, cx=center_x, cy=center_y: constraint_position2(x, i, cx, cy)})

    # Overlap constraint
    constraints.append({'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers, radii)})

    # Initial guess
    initial_guess = radii.copy()

    # Optimization
    result = minimize(
        objective_func,
        initial_guess,
        method='SLSQP',
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000, 'disp': False}
    )

    if not result.success:
        print("Optimization failed.")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    radii = result.x
    centers = np.zeros((num_circles, 2))
    indices = np.arange(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (indices[i] % 5 + 0.5) / 5
        centers[i, 1] = (indices[i] // 5 + 0.5) / 5

    # Validate constraints
    for i in range(num_circles):
        x_center, y_center = centers[i]
        if radii[i] > (1 - x_center - radii[i]) or radii[i] > (1 - y_center - radii[i]):
            radii[i] = min(radii[i], 1 - x_center - radii[i], 1 - y_center - radii[i])

    for i in range(num_circles):
        x_center, y_center = centers[i]
        for j in range(i + 1, num_circles):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < (radii[i] + radii[j]):
                radii[i] = (dist - radii[j]) / 2
                radii[j] = (dist - radii[i]) / 2

    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
