import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced circles in grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all the same, such that circles just touch the square
    initial_radius = 0.2
    radii = np.full(num_circles, initial_radius)

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)

    # Constraints: radius >= 0, and circles do not overlap
    constraints = []

    # Radius >= 0
    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: x[i]})

    # Circle is inside the unit square
    for i in range(num_circles):
        def constraint_inside(x, i=i):
            r = x[i]
            return np.array([1 - r - centers[i, 0], 1 - r - centers[i, 1], r - centers[i, 0], r - centers[i, 1]])
        constraints.append({'type': 'ineq', 'fun': constraint_inside})

    # No two circles overlap
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            def constraint_overlap(x, i=i, j=j):
                r_i = x[i]
                r_j = x[j]
                dist = np.linalg.norm(centers[i] - centers[j])
                return dist - r_i - r_j
            constraints.append({'type': 'ineq', 'fun': constraint_overlap})

    # Bounds for radii
    bounds = [(0, 1) for _ in range(num_circles)]

    # Solve optimization problem
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii = -result.fun

    return (centers_final, radii_final, sum_radii)
