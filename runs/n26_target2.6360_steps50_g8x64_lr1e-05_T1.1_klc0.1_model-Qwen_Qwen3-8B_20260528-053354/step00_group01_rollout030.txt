import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    # We start with a grid-like configuration
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.4 + 0.1
        centers[i, 1] = row * 0.4 + 0.1
        radii[i] = 0.05

    # Define optimization function
    def objective(x):
        # x is a flat array: [x0, y0, r0, x1, y1, r1, ..., x25, y25, r25]
        centers_flat = x[:-26]
        radii_flat = x[-26:]
        centers = centers_flat.reshape(26, 2)
        radii = radii_flat
        sum_radii = np.sum(radii)
        return -sum_radii  # minimize negative sum to maximize sum

    def constraint_overlap(x):
        # Check for overlapping circles
        centers_flat = x[:-26]
        radii_flat = x[-26:]
        centers = centers_flat.reshape(26, 2)
        radii = radii_flat
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return -1.0  # violation
        return 0.0

    def constraint_bounds(x):
        # Check if all circles are within the unit square
        centers_flat = x[:-26]
        radii_flat = x[-26:]
        centers = centers_flat.reshape(26, 2)
        radii = radii_flat
        for i in range(26):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return -1.0  # violation
        return 0.0

    # Set up optimization problem
    x0 = np.concatenate([centers.flatten(), radii])
    bounds = []
    for i in range(26):
        bounds.extend([(0.0, 1.0 - 1e-6) for _ in range(2)])  # x and y bounds
        bounds.extend([(0.0, 1.0 - 1e-6) for _ in range(1)])  # radius bounds
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_bounds}
    ]

    # Run optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="Optimization failed")
        result = minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            options={'maxiter': 1000, 'ftol': 1e-10, 'eps': 1e-6}
        )

    if not result.success:
        raise ValueError("Optimization failed to converge")
    
    centers = result.x[:-26].reshape(26, 2)
    radii = result.x[-26:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
