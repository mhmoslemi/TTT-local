import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers (uniformly spread in the square)
    centers = np.array([[i / 5 + 0.25, j / 5 + 0.25] for i in range(26) for j in range(26)])
    centers = centers[np.random.permutation(26)]  # randomize initial positions

    # Initial radii, all set to 0.1
    radii = np.full(26, 0.1)

    # Objective function: minimize the negative sum of radii (maximize the sum)
    def objective(x):
        return -np.sum(x[26:])

    # Constraints
    def constraint_center_inside(x):
        return x[26:] - (x[:26] - np.repeat([0.1, 0.1], 26)).min(axis=1)  # r <= x, y - r >= 0

    def constraint_no_overlap(x):
        centers = x[:26].reshape(-1, 2)
        radii = x[26:]
        distances = np.sqrt(((centers[:, None, :] - centers[None, :, :]) ** 2).sum(axis=2))
        return distances - (radii[:, None] + radii[None, :])

    # Bounds: centers in [0,1], radii in [0, 0.2]
    bounds = [(0.0, 1.0) for _ in range(26 * 2)] + [(0.0, 0.2) for _ in range(26)]

    # Initial guess: flat vector of centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-5)

    # Extract results
    centers = result.x[:26 * 2].reshape(26, 2)
    radii = result.x[26:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
