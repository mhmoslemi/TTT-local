```python
import numpy as np
from scipy.optimize import minimize
import sys

def objective_function(positions, radii):
    """
    Objective function to maximize the sum of radii.
    """
    return -np.sum(radii)  # Negative for minimization

def constraint_inside(positions, radii):
    """
    Constraint to ensure all circles are inside the unit square.
    """
    # positions is a (n, 2) array of centers
    # radii is a (n,) array of radii
    n = positions.shape[0]
    for i in range(n):
        x, y = positions[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            return -1.0
    return 0.0

def constraint_no_overlap(positions, radii):
    """
    Constraint to ensure circles do not overlap.
    """
    n = positions.shape[0]
    for i in range(n):
        for j in range(i + 1, n):
            dx = positions[i, 0] - positions[j, 0]
            dy = positions[i, 1] - positions[j, 1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j]:
                return -1.0
    return 0.0

def run_packing():
    try:
        from scipy.optimize import Bounds
    except ImportError:
        print("scipy not available")
        return np.array([]), np.array([]), 0.0

    # Initial guess: spread out the centers
    positions = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05  # Initial small radii
    # Flatten positions and radii for optimization
    x0 = np.concatenate((positions.reshape(-1), radii))
    n = 26

    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside, 'args': (radii,)},
        {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (positions, radii,)}
    ]

    # Bounds for positions: x and y between 0 and 1
    bounds = Bounds([0, 0] * n, [1, 1] * n)

    # Define objective function wrapped for scipy
    def wrapper(x):
        positions = x[:2 * n].reshape(n, 2)
        radii = x[2 * n:]
        return objective_function(positions, radii)

    result = minimize(wrapper, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    positions_opt = result.x[:2 * n].reshape(n, 2)
    radii_opt = result.x[2 * n:]
    sum_radii = -result.fun  # Negative because we minimized -sum(radii)

    # Return the result
    return positions_opt, radii_opt, sum_radii
```
