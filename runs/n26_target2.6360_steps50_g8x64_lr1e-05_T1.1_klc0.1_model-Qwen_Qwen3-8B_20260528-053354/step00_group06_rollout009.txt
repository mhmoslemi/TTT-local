import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial grid-like placement
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = 0.2 + col * 0.3
        centers[i, 1] = 0.2 + row * 0.3
        radii[i] = 0.1
    
    def objective(radii, centers):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum
    
    def constraints_func(radii, centers):
        # Constraint 1: Each circle is within the unit square
        r = radii.reshape(-1, 1)
        x = centers[:, 0]
        y = centers[:, 1]
        in_box = np.all((x - r >= 0) & (x + r <= 1) & (y - r >= 0) & (y + r <= 1))
        if not in_box:
            return np.inf
        
        # Constraint 2: No overlapping circles
        dists = cdist(centers, centers)
        radii_sum = radii.reshape(-1, 1) + radii.reshape(1, -1)
        no_overlap = np.all(dists >= radii_sum)
        if not no_overlap:
            return np.inf
        
        return 0  # Satisfied
    
    # Optimize
    bounds = [(0.0, 1.0) for _ in range(num_circles)]
    result = minimize(
        objective,
        radii,
        args=(centers,),
        method='L-BFGS-B',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': lambda r: constraints_func(r, centers)}
    )
    
    optimized_radii = result.x
    optimized_centers = centers.copy()
    
    # Check constraints and adjust if needed
    r = optimized_radii.reshape(-1, 1)
    x = optimized_centers[:, 0]
    y = optimized_centers[:, 1]
    in_box = (x - r >= 0) & (x + r <= 1) & (y - r >= 0) & (y + r <= 1)
    if not np.all(in_box):
        # Enforce box constraints
        for i in range(num_circles):
            if not in_box[i, 0]:
                optimized_radii[i] = max(0.0, min(optimized_radii[i], np.min([x[i], 1 - x[i], y[i], 1 - y[i]])))
    
    dists = cdist(optimized_centers, optimized_centers)
    radii_sum = optimized_radii.reshape(-1, 1) + optimized_radii.reshape(1, -1)
    no_overlap = dists >= radii_sum
    if not np.all(no_overlap):
        # Enforce no overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dists[i, j] < radii_sum[i, j]:
                    max_radius = (dists[i, j] - radii_sum[i, j]) / 2
                    optimized_radii[i] = max(0.0, optimized_radii[i] - max_radius)
                    optimized_radii[j] = max(0.0, optimized_radii[j] - max_radius)
    
    sum_radii = np.sum(optimized_radii)
    print(f"Sum of radii: {sum_radii}")
    
    return (optimized_centers, optimized_radii, sum_radii)
