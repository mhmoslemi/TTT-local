import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)  # We maximize the sum of radii

    def constraints(radii, centers):
        # Ensure all circles are inside the unit square
        if np.any(radii > 1 - np.min(centers, axis=0)):
            return np.inf
        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    def objective_with_constraints(radii):
        if np.any(radii < 0):
            return np.inf
        return -np.sum(radii)

    # Use a local optimization method with bounds
    bounds = [(0.0, 1.0 - 1e-5) for _ in range(num_circles)]

    # Initial guess
    initial_guess = np.random.rand(num_circles) * 0.05

    result = minimize(
        objective_with_constraints,
        initial_guess,
        method='L-BFGS-B',
        bounds=bounds,
        options={'disp': False}
    )

    # Extract the optimized radii and centers
    radii = result.x
    # Ensure circles are not overlapping and are within the unit square
    centers = np.random.rand(num_circles, 2)
    for i in range(num_circles):
        for j in range(num_circles):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < radii[i] + radii[j]:
                centers[i] = np.random.rand(2)
    # Ensure circles are within the unit square
    radii = np.minimum(radii, 1 - np.min(centers, axis=0))

    # Compute sum of radii
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
