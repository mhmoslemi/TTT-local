import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid with small radii
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    radii = np.full(25, 0.05)
    centers = np.vstack([centers, [0.5, 0.5]])  # Add center circle
    radii = np.append(radii, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
        centers = np.vstack([centers, [0.5, 0.5]])
        radii = np.append(radii, 0.05)
        # Check if all circles are inside the unit square
        for i in range(len(centers)):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Check if circles are non-overlapping
        dists = cdist(centers, centers)
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Optimize
    result = minimize(
        objective,
        radii,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [0.5, 0.5]])
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
