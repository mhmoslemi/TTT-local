import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced circles
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = (i + 0.5) / n
        centers[i, 1] = 0.5

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_bounds(radii):
        # Each circle must be inside the unit square
        return np.array([radii[i] <= centers[i, 0] for i in range(n)] + 
                        [radii[i] <= 1 - centers[i, 0] for i in range(n)] +
                        [radii[i] <= centers[i, 1] for i in range(n)] +
                        [radii[i] <= 1 - centers[i, 1] for i in range(n)])

    def constraint_overlap(radii):
        # No two circles overlap
        D = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return D - radii_sum

    # Optimization setup
    bounds = [(0, 1) for _ in range(n)]
    cons = [{'type': 'ineq', 'fun': lambda r: constraint_bounds(r)},
            {'type': 'ineq', 'fun': lambda r: constraint_overlap(r).min()}]

    # Run optimization
    result = minimize(objective, np.ones(n), bounds=bounds, constraints=cons, method='SLSQP')

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
