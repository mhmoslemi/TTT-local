import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    centers = centers[:26]  # Take first 26 points
    radii = np.full(26, 0.1)
    
    def objective(radii, centers):
        return -np.sum(radii)  # Maximize the sum of radii by minimizing the negative
    
    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        # r_i <= x_i - r_i <= 1 - r_i and same for y_i
        # So for each circle, we have r_i + r_i <= 1 - 2*r_i => 2*r_i <= 1 => r_i <= 0.5
        # But more strictly, each circle must satisfy x_i - r_i >= 0 and 1 - r_i >= x_i
        # Similarly for y
        constraints = []
        for i in range(26):
            if radii[i] > 0.5:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 0.5 - r[i]})
        return constraints
    
    def non_overlap_constraints(radii, centers):
        # Constraint 2: Distance between any two centers >= sum of their radii
        constraints = []
        for i in range(26):
            for j in range(i + 1, 26):
                def constraint(r):
                    dist = np.linalg.norm(centers[i] - centers[j])
                    return dist - (radii[i] + radii[j])
                constraints.append({'type': 'ineq', 'fun': constraint})
        return constraints
    
    # Optimize using Nelder-Mead
    bounds = [(0, 0.5) for _ in range(26)]
    
    result = minimize(
        fun=lambda r: -np.sum(r),
        x0=radii,
        bounds=bounds,
        method='Nelder-Mead',
        jac=None,
        options={'xatol': 1e-6, 'fatol': 1e-6, 'maxiter': 1000}
    )
    
    radii_opt = result.x
    sum_radii = np.sum(radii_opt)
    
    # Check if sum of radii is close to target
    # We will return the result as is, assuming convergence
    
    return (centers, radii_opt, sum_radii)
