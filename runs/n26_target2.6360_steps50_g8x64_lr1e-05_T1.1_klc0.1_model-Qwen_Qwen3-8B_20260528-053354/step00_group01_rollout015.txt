import numpy as np
from scipy.optimize import minimize
from itertools import combinations

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5) / 4.0
        centers[i, 1] = (i // 5) / 4.0

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint: each circle must be inside the unit square
        for i in range(26):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return np.inf
        # Constraint: no two circles overlap
        for (i, j) in combinations(range(26), 2):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j]:
                return np.inf
        return 0

    # Set up optimization problem
    bounds = [(0.0, 1.0) for _ in range(26)]
    res = minimize(objective, centers.flatten(), method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract results
    radii = res.x
    centers = np.reshape(res.x, (26, 2))
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
