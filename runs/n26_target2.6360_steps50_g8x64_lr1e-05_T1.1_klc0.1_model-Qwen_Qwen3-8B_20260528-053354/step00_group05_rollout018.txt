import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.random.rand(26, 2)  # Initial random centers
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds():
        return [(0.0, 1.0) for _ in range(26)]

    # Initial guess: small radii
    initial_guess = np.random.rand(26) * 0.05

    # Set up optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints, 'jac': None},
        options={'ftol': 1e-10, 'maxiter': 1000}
    )

    # Extract results
    radii = result.x
    centers = np.random.rand(26, 2)  # Placeholder; in real scenario, this would be updated
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
