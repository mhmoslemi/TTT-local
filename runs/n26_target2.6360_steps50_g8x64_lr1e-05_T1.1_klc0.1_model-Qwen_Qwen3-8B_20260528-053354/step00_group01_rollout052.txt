import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: small positive values
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: Radii must be non-negative
        if np.any(radii < 0):
            return np.inf
        # Constraint 2: Circles must be inside the unit square
        for i in range(num_circles):
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return np.inf
        # Constraint 3: No overlap between circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        options={'disp': False}
    )

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
