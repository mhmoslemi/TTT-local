import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = (col + 0.5) / 5.0
        centers[i, 1] = (row + 0.5) / 5.0
        radii[i] = 0.05

    def objective(radii):
        # Objective: maximize sum of radii (minimize negative sum)
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: circles must be inside the unit square
        r = radii.reshape(-1, 1)
        inside = np.all((r <= 1 - r) & (r <= 1 - r), axis=1)
        return -np.sum(inside)

    def overlap_constraints(radii):
        # Constraints: no two circles can overlap
        r = radii.reshape(-1, 1)
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        sums = r + r[np.newaxis, :]
        return -np.sum(dists < sums)

    # Define bounds for radii
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Optimization: minimize negative sum of radii with constraints
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraints},
            {'type': 'ineq', 'fun': overlap_constraints}
        ]
    )

    # Extract result
    radii_result = result.x
    centers_result = centers.copy()
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
