import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid-like pattern with some spacing
    num_circles = 26
    centers = np.random.rand(num_circles, 2) * 0.8 + 0.1  # Initial random positions within [0.1, 0.9]
    radii = np.full(num_circles, 0.05)  # Initial guess for radii

    def objective_function(radii):
        # Objective function to minimize (negative of sum of radii to maximize)
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: circles fully inside the square and non-overlapping
        centers = np.random.rand(num_circles, 2) * 0.8 + 0.1  # Re-initialize centers for each constraint check
        # Constraint 1: circles inside square
        inside_square = np.all((radii <= 1 - centers) & (radii >= 0), axis=0)
        # Constraint 2: circles do not overlap
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        non_overlapping = np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :], axis=(0, 1))
        return np.concatenate((inside_square, non_overlapping))

    # Set up optimization problem
    bounds = [(0, 1 - 1e-5) for _ in range(num_circles)]  # Radii can't exceed 1 - small epsilon

    # Use a local optimizer as a heuristic to find good initial positions
    result = minimize(objective_function, radii, method='L-BFGS-B', bounds=bounds)

    # Final positions (random re-initialization might be better to avoid local optima)
    best_centers = np.random.rand(num_circles, 2) * 0.8 + 0.1
    best_radii = result.x

    # Enforce non-overlapping constraints again to ensure valid solution
    while True:
        # Compute distances between centers
        dists = np.sqrt(((best_centers[:, np.newaxis, :] - best_centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        non_overlapping = np.all(dists >= best_radii[:, np.newaxis] + best_radii[np.newaxis, :], axis=(0, 1))
        if non_overlapping:
            break
        else:
            # Adjust centers and radii slightly
            best_centers += np.random.normal(0, 0.05, best_centers.shape)
            best_centers = np.clip(best_centers, 0.1, 0.9)
            best_radii = np.clip(best_radii, 0, 0.5)

    sum_radii = np.sum(best_radii)
    return (best_centers, best_radii, sum_radii)
