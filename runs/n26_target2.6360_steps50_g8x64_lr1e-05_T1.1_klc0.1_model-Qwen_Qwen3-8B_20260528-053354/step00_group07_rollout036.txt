import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid with some padding
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii_flat):
        return -np.sum(radii_flat)

    # Constraints: 1. All circles must be inside the square
    # 2. No two circles may overlap
    def constraint_inside(radii_flat, centers_flat):
        # Extract radii and centers
        radii = radii_flat.reshape(-1, 1)
        centers = centers_flat.reshape(-1, 2)
        # Each circle must be inside the square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii))
        return inside

    def constraint_overlap(radii_flat, centers_flat):
        # Extract radii and centers
        radii = radii_flat.reshape(-1, 1)
        centers = centers_flat.reshape(-1, 2)
        # Compute pairwise distances
        dists = cdist(centers, centers)
        # Ensure distance >= sum of radii for each pair
        return np.all(dists >= radii + radii.T)

    # Bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Initial guess
    initial_centers = np.random.rand(num_circles, 2)
    initial_radii = np.random.rand(num_circles) * 0.1
    initial_flat = np.concatenate((initial_centers.flatten(), initial_radii))

    # Optimization
    cons = [
        {'type': 'ineq', 'fun': lambda x: constraint_inside(x[:num_circles], x[num_circles:]),
         'args': (initial_centers.flatten(),)},
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x[:num_circles], x[num_circles:]),
         'args': (initial_centers.flatten(),)}]
    result = minimize(objective, initial_flat, method='SLSQP', bounds=bounds, constraints=cons)

    # Extract solution
    optimal_flat = result.x
    centers = optimal_flat[num_circles:].reshape(num_circles, 2)
    radii = optimal_flat[:num_circles]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
