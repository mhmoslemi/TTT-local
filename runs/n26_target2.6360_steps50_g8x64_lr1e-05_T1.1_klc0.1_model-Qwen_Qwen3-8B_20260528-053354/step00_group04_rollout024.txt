import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    # We use 5 rows and 5 columns with some offset to allow better packing
    # Each circle has a small radius initially
    num_circles = 26
    centers = np.random.rand(num_circles, 2)  # Random initial positions
    radii = np.full(num_circles, 0.05)  # Small initial radius

    def objective(radii):
        return -np.sum(radii)  # We want to maximize sum of radii

    def constraints(radii):
        # Ensure circles are within the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf

        # Ensure circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimize using SciPy
    bounds = [(0, 0.5) for _ in range(num_circles)]  # Upper bound for radii
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract best result
    best_radii = result.x
    best_centers = centers
    sum_radii = np.sum(best_radii)

    # Return result
    return (best_centers, best_radii, sum_radii)
