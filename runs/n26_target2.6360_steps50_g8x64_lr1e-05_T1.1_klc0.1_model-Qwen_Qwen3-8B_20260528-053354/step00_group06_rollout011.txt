import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.random.rand(26, 2)
        # Ensure circles are within the unit square
        for i in range(26):
            x, y = centers[i]
            if x < radii[i] or x > 1 - radii[i] or y < radii[i] or y > 1 - radii[i]:
                return np.inf
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Initial guess: small radii and centered points
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05

    # Optimize
    result = minimize(objective, radii, method='COBYLA', bounds=[(0, 0.5)]*26)
    optimized_radii = result.x
    optimized_centers = centers.copy()
    # Ensure circles are within the unit square and no overlaps
    for i in range(26):
        x, y = optimized_centers[i]
        if x < optimized_radii[i] or x > 1 - optimized_radii[i] or y < optimized_radii[i] or y > 1 - optimized_radii[i]:
            optimized_radii[i] = min(optimized_radii[i], x, 1 - x, y, 1 - y)
    # Ensure no overlaps
    for i in range(26):
        for j in range(i + 1, 26):
            dx = optimized_centers[i, 0] - optimized_centers[j, 0]
            dy = optimized_centers[i, 1] - optimized_centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < optimized_radii[i] + optimized_radii[j]:
                overlap = dist - (optimized_radii[i] + optimized_radii[j])
                optimized_radii[i] -= overlap / 2
                optimized_radii[j] -= overlap / 2
                optimized_radii[i] = max(optimized_radii[i], 0)
                optimized_radii[j] = max(optimized_radii[j], 0)
    sum_radii = np.sum(optimized_radii)

    return (optimized_centers, optimized_radii, sum_radii)
