import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial radius estimate (for center circles)
    initial_radius = 1 / (2 * np.sqrt(2))  # Diagonal distance between centers
    # Place circles in a 5x6 grid with spacing
    for i in range(num_circles):
        row = i // 6
        col = i % 6
        x = col * (1 + initial_radius * 2) / 6
        y = row * (1 + initial_radius * 2) / 5
        centers[i] = [x, y]
        radii[i] = initial_radius
    
    # Objective function: maximize the sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)
    
    # Constraints
    def constraint1(radii_vec):
        # Ensure each circle is within the unit square
        centers_r = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers_r[i] = [col * (1 + radii_vec[i] * 2) / 6, row * (1 + radii_vec[i] * 2) / 5]
        for i in range(num_circles):
            x, y = centers_r[i]
            if not (radii_vec[i] <= x <= 1 - radii_vec[i] and radii_vec[i] <= y <= 1 - radii_vec[i]):
                return -1
        return 0
    
    def constraint2(radii_vec):
        # Ensure circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_vec[i] + radii_vec[j]:
                    return -1
        return 0
    
    # Set up optimization problem
    bounds = [(0, 1) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]
    
    # Run optimization
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", category=RuntimeWarning)
        result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-6)
    
    # Extract final values
    optimal_radii = result.x
    optimal_centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 6
        col = i % 6
        x = col * (1 + optimal_radii[i] * 2) / 6
        y = row * (1 + optimal_radii[i] * 2) / 5
        optimal_centers[i] = [x, y]
    
    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
