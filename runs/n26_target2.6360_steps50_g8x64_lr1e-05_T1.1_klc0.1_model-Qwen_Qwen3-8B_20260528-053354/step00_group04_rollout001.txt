import numpy as np
from scipy.optimize import minimize
from itertools import combinations

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.zeros((26, 2))
    radii = np.ones(26) * 0.1  # Initial radius estimate

    # Objective function: maximize the sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x[26:])

    # Constraints: each circle must be inside the unit square
    def constraint_inside(x):
        centers = x[:26*2].reshape(26, 2)
        radii = x[26:]
        return np.sum((centers[:, 0] - radii - 0.0) * (centers[:, 0] + radii - 1.0)) <= 0
    cons_inside = {'type': 'ineq', 'fun': constraint_inside}

    # Constraints: no two circles overlap
    def constraint_no_overlap(x):
        centers = x[:26*2].reshape(26, 2)
        radii = x[26:]
        for (i, j) in combinations(range(26), 2):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            distance = np.sqrt(dx**2 + dy**2)
            if distance < radii[i] + radii[j] - 1e-8:
                return distance - radii[i] - radii[j] + 1e-8
        return np.inf

    cons_overlap = {'type': 'ineq', 'fun': constraint_no_overlap}

    # Initial parameters
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: all radii must be non-negative
    bounds = [(0, 1) for _ in range(26)] + [(0, 0.5) for _ in range(26)]  # Radii can't exceed 0.5

    # Optimize
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=[cons_inside, cons_overlap], tol=1e-6)

    # Extract result
    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
