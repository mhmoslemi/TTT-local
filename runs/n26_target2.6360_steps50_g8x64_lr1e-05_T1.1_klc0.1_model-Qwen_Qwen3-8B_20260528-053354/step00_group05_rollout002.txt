import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initialize random centers within the square and radii
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    def objective_function(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        x_centers = x[:2*26].reshape(26, 2)
        x_radii = x[2*26:]
        sum_radii = np.sum(x_radii)
        # Check if all circles are fully inside the square
        if np.any(x_radii < 0) or np.any(x_centers[:, 0] - x_radii < 0) or np.any(1 - x_centers[:, 0] - x_radii < 0) or np.any(x_centers[:, 1] - x_radii < 0) or np.any(1 - x_centers[:, 1] - x_radii < 0):
            return np.inf  # Penalize for going out of bounds
        # Check if circles overlap
        dists = cdist(x_centers, x_centers)
        for i in range(26):
            for j in range(i + 1, 26):
                if dists[i, j] < x_radii[i] + x_radii[j]:
                    return np.inf  # Penalize for overlapping circles
        return -sum_radii  # Negative for minimization

    def bounds():
        # Bounds for each center (x, y) and radius
        bounds = []
        for _ in range(26):
            bounds.extend([(0, 1), (0, 1), (0, 0.5)])
        return bounds

    # Initial guess
    initial_guess = np.random.rand(2*26 + 26)
    # Optimize with constraints
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective_function, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-7)

    # Extract result
    optimal_x = result.x
    centers = optimal_x[:2*26].reshape(26, 2)
    radii = optimal_x[2*26:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
