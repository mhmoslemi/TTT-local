import math

            radii[i] = 0.15
        else:
            radii[i] = 0.2

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints_func(centers, radii):
        constraints = []
        # Constraint 1: All circles inside the unit square
        for i in range(len(radii)):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        # Constraint 2: No circles overlap
        for i in range(len(radii)):

                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def constraint_handler(x):
        # Unpack x into centers and radii
        centers = x[:26*2].reshape(26, 2)
        radii = x[26*2:]
        # Constraint 1: All circles inside the unit square
        for i in range(len(radii)):
            x_i, y_i = centers[i]
            if not (radii[i] <= x_i <= 1 - radii[i] and radii[i] <= y_i <= 1 - radii[i]):

        # Constraint 2: No circles overlap
        for i in range(len(radii)):

                    return np.inf
        return 0

    # Flatten centers and radii into a single array
