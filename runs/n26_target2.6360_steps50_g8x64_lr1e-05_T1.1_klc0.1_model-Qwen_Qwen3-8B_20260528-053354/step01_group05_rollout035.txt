import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    np.random.seed(42)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints_func(radii, centers):
        # Check if all circles are inside the unit square
        for i in range(26):
            if radii[i] > 0:
                if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                    return np.inf
        # Check for circle overlaps
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)
    
    best_sum = 0
    best_centers = np.zeros((26, 2))
    best_radii = np.zeros(26)
    
    for _ in range(10):
        centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
        radii = np.full(26, 0.05)
        
        # Define constraints for optimization
        cons = [{'type': 'ineq', 'fun': lambda r: constraints_func(r, centers)}]
        
        result = minimize(objective, radii, constraints=cons)
        
        if result.success:
            current_sum = np.sum(result.x)
            if current_sum > best_sum:
                best_sum = current_sum
                best_centers = centers
                best_radii = result.x
    
    return (best_centers, best_radii, best_sum)
