import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Define the unit square [0,1] x [0,1]
    square_bounds = np.array([[0.0, 1.0], [0.0, 1.0]])
    
    # Initial random positions for 26 circles
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    
    # Objective function: minimize the negative sum of radii (equivalent to maximizing)
    def objective(radii_vec):
        return -np.sum(radii_vec)
    
    # Constraint 1: Each circle must be within the unit square
    def square_constraint(radii_vec):
        centers = np.random.rand(26, 2)
        for i in range(26):
            x, y = centers[i]
            if not (radii_vec[i] <= x <= 1 - radii_vec[i] and radii_vec[i] <= y <= 1 - radii_vec[i]):
                return -1
        return 0
    
    # Constraint 2: Circles must not overlap
    def circle_overlap_constraint(radii_vec):
        centers = np.random.rand(26, 2)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return -1
        return 0
    
    # Initial guess for radii
    initial_guess = np.random.rand(26)
    
    # Set up constraints
    cons = [
        {'type': 'ineq', 'fun': square_constraint},
        {'type': 'ineq', 'fun': circle_overlap_constraint}
    ]
    
    # Perform optimization
    result = minimize(objective, initial_guess, method='SLSQP', constraints=cons)
    
    # Extract the optimized result
    optimized_radii = result.x
    # Generate final centers (random within square bounds considering radii)
    centers_final = np.zeros((26, 2))
    for i in range(26):
        r = optimized_radii[i]
        # Place in a random position ensuring radius fits within the square
        x = np.random.uniform(r, 1 - r)
        y = np.random.uniform(r, 1 - r)
        centers_final[i] = [x, y]
    
    sum_radii = np.sum(optimized_radii)
    
    return (centers_final, optimized_radii, sum_radii)
