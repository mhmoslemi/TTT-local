import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as dist

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i / np.sqrt(num_circles)
        y = (i // np.sqrt(num_circles)) / np.sqrt(num_circles)
        centers[i] = [x, y]
    
    # Optimization variables: radii and centers
    initial_guess = np.concatenate((centers.reshape(-1), np.zeros(num_circles)))
    
    # Objective function: maximize sum of radii
    def objective(x):
        radii = x[num_circles:]
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        return -np.sum(radii)  # Negative for minimization
    
    # Constraints
    def constraint_center_inside(x):
        radii = x[num_circles:]
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        return np.sum((centers[:, 0] - radii) >= 0) + np.sum((1 - centers[:, 0] - radii) >= 0) + \
               np.sum((centers[:, 1] - radii) >= 0) + np.sum((1 - centers[:, 1] - radii) >= 0)
    
    def constraint_no_overlap(x):
        radii = x[num_circles:]
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        distances = dist.cdist(centers, centers)
        return np.sum((distances - radii[:, np.newaxis] - radii[np.newaxis, :]) >= 0)
    
    cons = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    
    # Bounds: centers in [0,1], radii >= 0
    bounds = []
    for i in range(num_circles * 2):
        bounds.append((0.0, 1.0))
    for i in range(num_circles):
        bounds.append((0.0, 1.0))
    
    result = minimize(objective, initial_guess, method='SLSQP', bounds=bounds, constraints=cons)
    x_opt = result.x
    
    centers_opt = x_opt[:num_circles * 2].reshape(num_circles, 2)
    radii_opt = x_opt[num_circles:]
    sum_radii_opt = -result.fun
    
    return centers_opt, radii_opt, sum_radii_opt
