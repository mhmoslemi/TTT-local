import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    centers = np.array([[(i / 5, j / 5) for j in range(5)] for i in range(5)])
    centers = centers.reshape(-1, 2)
    radii = np.full(26, 0.04)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints(radii_vec):
        # Ensure all circles are inside the unit square
        radii = radii_vec.reshape(-1)
        for i in range(26):
            r = radii[i]
            if not (r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r):
                return False
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds_func(radii_vec):
        return [(0.0, 1.0) for _ in range(26)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        jac=None,
        bounds=bounds_func,
        constraints=({'type': 'ineq', 'fun': constraints}),
        options={'disp': False, 'ftol': 1e-8}
    )

    # Extract the optimized result
    optimized_radii = result.x
    optimized_sum_radii = np.sum(optimized_radii)

    return (centers, optimized_radii, optimized_sum_radii)
