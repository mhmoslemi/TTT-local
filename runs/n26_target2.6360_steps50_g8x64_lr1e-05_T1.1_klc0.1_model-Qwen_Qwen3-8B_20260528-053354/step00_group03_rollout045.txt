import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        x = (i % 5 + 0.5) / 5
        y = (i // 5 + 0.5) / 5
        centers[i] = [x, y]
    radii = np.full(n_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint: each circle must be inside the unit square
        for i in range(n_circles):
            r = radii[i]
            if r > 1 - 1e-8:
                return False
        # Constraint: no two circles overlap
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return False
        return True

    def bounds(radii):
        return [(0, 1 - 1e-8) for _ in range(n_circles)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=bounds,
        options={'ftol': 1e-8, 'maxiter': 500}
    )

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
