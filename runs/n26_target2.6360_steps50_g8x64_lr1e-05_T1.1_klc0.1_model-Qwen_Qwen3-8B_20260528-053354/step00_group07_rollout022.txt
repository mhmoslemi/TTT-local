import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: place circles evenly in rows and columns
    num_circles = 26
    initial_centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        initial_centers[i, 0] = 0.2 + col * 0.25
        initial_centers[i, 1] = 0.2 + row * 0.25
    initial_radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        constraints = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - (radii[i] + radii[j]) >= 0)
        return constraints

    def bounds(radii):
        return [(0, 1) for _ in range(num_circles)]

    result = minimize(
        objective,
        initial_radii,
        args=(initial_centers,),
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r, initial_centers)[0]}],
        bounds=bounds(initial_radii),
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    if not result.success:
        return (initial_centers, initial_radii, np.sum(initial_radii))

    radii = result.x
    centers = copy.deepcopy(initial_centers)
    sum_radii = np.sum(radii)

    # Check if radii are valid and adjust if needed
    for i in range(num_circles):
        if radii[i] < 0:
            radii[i] = 0
        if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
            radii[i] = 0
        if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
            radii[i] = 0
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
