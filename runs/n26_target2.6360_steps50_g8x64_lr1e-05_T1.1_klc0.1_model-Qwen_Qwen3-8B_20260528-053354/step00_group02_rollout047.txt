import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    # Initial radius
    initial_radius = 0.05
    # Distribute circles in a grid-like pattern
    for i in range(num_circles):
        row = i // 4
        col = i % 4
        x = col / 3.0
        y = row / 3.0
        centers[i] = [x, y]
        radii[i] = initial_radius

    # Objective function
    def objective(radii):
        return -np.sum(radii)

    # Constraints: circles must fit within the unit square
    def square_constraint(radii):
        return np.array([r for r in radii])

    # Constraints: circles must not overlap
    def non_overlap_constraint(radii):
        centers_current = np.zeros((len(radii), 2))
        for i in range(len(radii)):
            x = 0.0
            y = 0.0
            # Placeholder for actual update logic (this part is illustrative)
            centers_current[i] = [0.2 + i / 10, 0.2 + i / 10]
        distances = np.zeros((len(radii), len(radii)))
        for i in range(len(radii)):
            for j in range(len(radii)):
                if i != j:
                    dx = centers_current[i, 0] - centers_current[j, 0]
                    dy = centers_current[i, 1] - centers_current[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    distances[i, j] = dist - (radii[i] + radii[j])
        min_dist = np.min(distances[distances < 0])
        return np.array([min_dist])

    # Bounds for radii
    bounds = [(0, 0.5) for _ in range(num_circles)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': lambda r: r},  # Ensure radii are non-negative
        {'type': 'ineq', 'fun': square_constraint},  # Ensure circles fit in square
        {'type': 'ineq', 'fun': non_overlap_constraint}  # No overlap between circles
    ]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-5)

    # Extract results
    centers_result = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = 0.0
        y = 0.0
        # Placeholder for actual update logic (this part is illustrative)
        centers_result[i] = [0.2 + i / 10, 0.2 + i / 10]
    sum_radii = result.fun * -1
    return (centers_result, result.x, sum_radii)
