import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    def objective_function(x):
        # x is flattened array of centers (26*2) and radii (26)
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints_function(x):
        # x is flattened array of centers (26*2) and radii (26)
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:]
        constraints = []
        # Constraint 1: All circles are inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append(x_i - r_i)
            constraints.append(1 - x_i - r_i)
            constraints.append(y_i - r_i)
            constraints.append(1 - y_i - r_i)
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraints.append(dist - (radii_flat[i] + radii_flat[j]))
        return constraints

    # Prepare bounds for optimization
    bounds = []
    for i in range(26):
        bounds.append((0.0, 1.0))  # x_i
        bounds.append((0.0, 1.0))  # y_i
        bounds.append((0.0, 0.5))  # r_i (radius cannot exceed 0.5 to fit in unit square)

    # Initial guess flattened
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization options
    options = {
        'disp': False,
        'ftol': 1e-6,
        'maxiter': 2000
    }

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective_function,
            initial_guess,
            method='SLSQP',
            bounds=bounds,
            constraints={'type': 'ineq', 'fun': constraints_function},
            options=options
        )

    # Extract final centers, radii, and sum of radii
    final_centers = result.x[:26*2].reshape(26, 2)
    final_radii = result.x[26*2:]
    final_sum_radii = -result.fun  # Since we minimized -sum

    return (final_centers, final_radii, final_sum_radii)
