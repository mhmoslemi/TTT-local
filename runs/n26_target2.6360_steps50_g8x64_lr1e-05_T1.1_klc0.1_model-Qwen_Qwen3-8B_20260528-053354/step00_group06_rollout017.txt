import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    spacing = 1.0 / math.sqrt(num_circles)
    for i in range(num_circles):
        row = i // math.sqrt(num_circles)
        col = i % math.sqrt(num_circles)
        centers[i, 0] = col * spacing + 0.5 * spacing
        centers[i, 1] = row * spacing + 0.5 * spacing
        radii[i] = 0.01  # Initial small radius

    # Objective function to maximize sum of radii (minimize -sum)
    def objective(x):
        return -sum(radii)

    # Constraints: circles must be inside the unit square
    def constraint_inside(x):
        return np.sum((x[::2] - 0.5 * x[1::2])**2 + (x[1::2] - 0.5 * x[::2])**2 - (x[1::2])**2)

    # Constraints: circles must not overlap
    def constraint_no_overlap(x):
        dist_sq = np.sum((x[::2, np.newaxis] - x[1::2, np.newaxis].T)**2, axis=2)
        return np.sum(np.sqrt(dist_sq) - x[1::2] - x[::2])  # Ensure >= 0

    # Optimization bounds: x and y coordinates between 0 and 1, radius >= 0
    bounds = []
    for i in range(num_circles):
        bounds.extend([(0, 1), (0, 1), (0, 1)])

    # Initial guess concatenated
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization setup
    def constraint_inside_vec(x):
        # Split x into centers and radii
        centers_flat = x[:-num_circles]
        radii_flat = x[-num_circles:]
        centers = centers_flat.reshape(num_circles, 2)
        radii = radii_flat
        # Check if all circles are inside the square
        in_bounds = np.all((centers[:, 0] - radii >= 0) & (centers[:, 0] + radii <= 1) &
                           (centers[:, 1] - radii >= 0) & (centers[:, 1] + radii <= 1))
        return in_bounds

    def constraint_no_overlap_vec(x):
        # Split x into centers and radii
        centers_flat = x[:-num_circles]
        radii_flat = x[-num_circles:]
        centers = centers_flat.reshape(num_circles, 2)
        radii = radii_flat
        # Check if all circles are non-overlapping
        dist = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i, j] = dx * dx + dy * dy
                dist[j, i] = dist[i, j]
        min_dist = np.min(np.sqrt(dist) - radii[:, np.newaxis] - radii[np.newaxis, :])
        return min_dist >= 0

    # Use constraints as functions
    cons = [
        {'type': 'ineq', 'fun': constraint_inside_vec},
        {'type': 'ineq', 'fun': constraint_no_overlap_vec}
    ]

    # Optimization settings
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6, options={'maxiter': 1000})

    # Extract result
    x_result = result.x
    centers = x_result[:-num_circles].reshape(num_circles, 2)
    radii = x_result[-num_circles:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
