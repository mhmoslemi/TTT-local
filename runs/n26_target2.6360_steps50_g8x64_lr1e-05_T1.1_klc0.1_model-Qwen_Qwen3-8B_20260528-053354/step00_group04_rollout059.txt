import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(radii):
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return -np.inf
        return 0

    def constraint_no_overlap(radii):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return -np.inf
        return 0

    # Set up optimization
    bounds = [(0.0, 0.5) for _ in range(26)]  # Radii can't exceed 0.5
    constraints = [
        {
            'type': 'ineq',
            'fun': constraint_circle_inside
        },
        {
            'type': 'ineq',
            'fun': constraint_no_overlap
        }
    ]

    # Optimize
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract and format result
    radii = result.x
    centers = np.array(centers)
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
