import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii: place 26 circles on a grid
    centers = np.array([[i / 5.0, j / 5.0] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [[0.5, 0.25], [0.5, 0.75], [0.25, 0.5], [0.75, 0.5]]])
    radii = np.full(26, 0.05)

    def objective(x):
        # x is a flat array: [x0, y0, r0, x1, y1, r1, ...]
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        constraints = []

        # Constraint: each circle is within the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append((x_i - r_i >= 0) * 1e6)
            constraints.append((x_i + r_i <= 1) * 1e6)
            constraints.append((y_i - r_i >= 0) * 1e6)
            constraints.append((y_i + r_i <= 1) * 1e6)

        # Constraint: no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                constraints.append((dist - r_i - r_j >= 0) * 1e6)

        return np.array(constraints)

    # Bounds for all variables: x, y in [0, 1], radius >= 0
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 0.5)])  # radius <= 0.5

    result = minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        tol=1e-6,
        options={'maxiter': 500}
    )

    if not result.success:
        raise RuntimeError("Optimization failed")

    centers_opt = result.x[:2*26].reshape(26, 2)
    radii_opt = result.x[2*26:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
