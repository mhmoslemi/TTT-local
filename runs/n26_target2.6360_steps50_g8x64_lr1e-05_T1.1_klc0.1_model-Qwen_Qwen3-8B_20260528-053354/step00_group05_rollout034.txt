import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii (random points inside the square with small radii)
    np.random.seed(42)
    n_circles = 26
    centers = np.random.rand(n_circles, 2)
    radii = np.random.rand(n_circles) * 0.05
    initial_guess = np.hstack([centers.reshape(-1), radii])

    # Objective function: maximize the sum of radii (minimize -sum_radii)
    def objective(x):
        centers = x[:2*n_circles].reshape(n_circles, 2)
        radii = x[2*n_circles:]
        sum_radii = np.sum(radii)
        return -sum_radii

    # Constraints: all circles are inside the unit square
    def constraint_inside(x):
        centers = x[:2*n_circles].reshape(n_circles, 2)
        radii = x[2*n_circles:]
        return np.array([r - x[i, 0] for i, r in enumerate(radii)]) + np.array([r - (1 - x[i, 0]) for i, r in enumerate(radii)]) + \
               np.array([r - x[i, 1] for i, r in enumerate(radii)]) + np.array([r - (1 - x[i, 1]) for i, r in enumerate(radii)])

    # Constraint that all radii are non-negative
    def constraint_non_negative(x):
        return x[2*n_circles:]

    # Constraints: circles do not overlap
    def constraint_overlap(x):
        centers = x[:2*n_circles].reshape(n_circles, 2)
        radii = x[2*n_circles:]
        distances = np.zeros((n_circles, n_circles))
        for i in range(n_circles):
            for j in range(n_circles):
                if i == j:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distances[i, j] = np.sqrt(dx**2 + dy**2) - (radii[i] + radii[j])
        return distances

    # Setup constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_non_negative},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Bounds for centers (between 0 and 1) and radii (non-negative)
    bounds = [(0, 1) for _ in range(2 * n_circles)] + [(0, 1) for _ in range(n_circles)]

    # Run optimization
    result = minimize(objective, initial_guess, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract solution
    centers = result.x[:2*n_circles].reshape(n_circles, 2)
    radii = result.x[2*n_circles:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
