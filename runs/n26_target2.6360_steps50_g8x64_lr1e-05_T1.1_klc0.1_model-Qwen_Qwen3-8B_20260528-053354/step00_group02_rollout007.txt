```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    radii = np.ones(26) * 0.1

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints: all circles must be inside the square
    def inside_box(radii_vec):
        return np.all(radii_vec <= 0.5)

    # Constraint: no two circles overlap
    def no_overlap(radii_vec):
        dists = cdist(centers, centers)
        return np.all(dists >= radii_vec[:, np.newaxis] + radii_vec[np.newaxis, :])

    # Combine constraints
    def constraints(radii_vec):
        return np.concatenate([
            np.array([inside_box(radii_vec)]),
            np.array([no_overlap(radii_vec)])
        ])

    # Optimization settings
    bounds = [(0, 0.5) for _ in range(26)]
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    # Extract results
    radii_result = result.x
    sum_radii = -result.fun
    centers_result = centers.copy()

    # Ensure all circles are valid and non-overlapping
    valid = inside_box(radii_result) and no_overlap(radii_result)
    if not valid:
        warnings.warn("Optimization did not find a valid configuration.")

    return (centers_result, radii_result, sum_radii)
```
