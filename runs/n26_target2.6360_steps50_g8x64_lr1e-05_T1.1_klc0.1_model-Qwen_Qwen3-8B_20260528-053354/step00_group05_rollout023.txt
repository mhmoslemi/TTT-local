import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i] = [col * 0.3 + 0.1, row * 0.3 + 0.1]
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = []
        for i in range(26):

            centers.append([x, y])

                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    def bounds():
        return [(0, 0.5) for _ in range(26)]

    # Run optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-5
    )
