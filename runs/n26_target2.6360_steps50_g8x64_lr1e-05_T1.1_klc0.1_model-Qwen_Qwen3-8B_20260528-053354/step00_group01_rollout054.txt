import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess: distribute circles in a grid
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    # Initial guess for centers
    for i in range(n_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.3 + 0.15
        centers[i, 1] = row * 0.3 + 0.15

    # Initial guess for radii
    radii = np.random.uniform(low=0.01, high=0.05, size=n_circles)

    # Objective function: maximize sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Inequality constraints: circles must be fully inside the unit square
    def constraint_inside(radii_vec):
        return np.array([r for r in radii_vec])

    def constraint_inside_bounds(radii_vec):
        return np.array([r <= 1 - np.sqrt((centers[i, 0] ** 2 + centers[i, 1] ** 2)) for i, r in enumerate(radii_vec)])

    # Constraint function: no two circles overlap
    def constraint_overlap(x):
        # x is a flattened array: [r0, r1, ..., r25]
        radii_vec = x.reshape((n_circles,))
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.hypot(dx, dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return -dist + radii_vec[i] + radii_vec[j]
        return 0.0

    # Constraints
    cons_inside = {'type': 'ineq', 'fun': constraint_inside_bounds}
    cons_overlap = {'type': 'ineq', 'fun': constraint_overlap}

    # Bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(n_circles)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[cons_inside, cons_overlap],
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract results
    final_radii = result.x
    final_centers = centers
    final_sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, final_sum_radii)
