import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 6) for y in np.linspace(0.1, 0.9, 6)])
    centers = centers[:26]  # Take the first 26 points
    radii = np.ones(26) * 0.1  # Initial radius guess

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Ensure circles are within the unit square
        x = centers[:, 0]
        y = centers[:, 1]
        r = radii
        r_max = np.minimum(np.minimum(x - r, 1 - x - r), np.minimum(y - r, 1 - y - r))
        r_max[r_max < 0] = -np.inf
        return r_max

    # Constraint 2: Ensure no overlap between circles
    def non_overlap(radii):
        dist = cdist(centers, centers)
        r_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dist - r_sum

    # Set up optimization
    bounds = [(0., 0.5) for _ in range(26)]  # Radii can't exceed 0.5
    constraints_func = {'type': 'ineq', 'fun': constraints}
    non_overlap_func = {'type': 'ineq', 'fun': non_overlap}

    result = minimize(objective, radii, method='SLSQP',
                      bounds=bounds,
                      constraints=[constraints_func, non_overlap_func],
                      options={'maxiter': 500, 'ftol': 1e-9, 'eps': 1e-9})

    if not result.success:
        warnings.warn("Optimization did not converge.")
        radii = np.ones(26) * 0.1  # Fallback to initial guess
    else:
        radii = result.x

    # Final check to ensure circles are within the square
    r = radii
    x = centers[:, 0]
    y = centers[:, 1]
    r_max = np.minimum(np.minimum(x - r, 1 - x - r), np.minimum(y - r, 1 - y - r))
    r = np.minimum(r, r_max)

    sum_radii = np.sum(r)
    return (centers, r, sum_radii)
