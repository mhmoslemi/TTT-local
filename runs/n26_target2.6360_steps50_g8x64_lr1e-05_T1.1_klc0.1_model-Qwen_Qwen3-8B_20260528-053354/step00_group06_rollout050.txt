import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    
    # Objective function: maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraints
    def constraint_inside(radii, centers):
        # Each circle is fully inside the unit square
        return np.sum((centers[:, 0] - radii) > 0) + np.sum((1 - centers[:, 0] - radii) > 0) + \
               np.sum((centers[:, 1] - radii) > 0) + np.sum((1 - centers[:, 1] - radii) > 0)
    
    def constraint_no_overlap(radii, centers):
        # Distance between centers >= sum of radii
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2).sum(axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Bounds: radii >= 0
    bounds = [(0, 1) for _ in range(26)]
    
    # Initial parameters
    x0 = np.concatenate([centers.reshape(-1), radii])
    
    def objective_wrapper(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        return -np.sum(radii)
    
    def constraint_inside_wrapper(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        return np.sum((centers[:, 0] - radii) > 0) + np.sum((1 - centers[:, 0] - radii) > 0) + \
               np.sum((centers[:, 1] - radii) > 0) + np.sum((1 - centers[:, 1] - radii) > 0)
    
    def constraint_no_overlap_wrapper(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2).sum(axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Optimization
    try:
        result = minimize(
            objective_wrapper,
            x0,
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_inside_wrapper},
                {'type': 'ineq', 'fun': constraint_no_overlap_wrapper}
            ],
            options={'disp': False, 'ftol': 1e-9, 'maxiter': 1000}
        )
    except:
        # Fallback if scipy.optimize is not available
        result = None
    
    if result is None or not result.success:
        # Default values if optimization fails
        centers = np.random.rand(26, 2)
        radii = np.random.rand(26)
        sum_radii = np.sum(radii)
    else:
        centers = result.x[:52].reshape(26, 2)
        radii = result.x[52:]
        sum_radii = -objective(radii, centers)
    
    return (centers, radii, sum_radii)
