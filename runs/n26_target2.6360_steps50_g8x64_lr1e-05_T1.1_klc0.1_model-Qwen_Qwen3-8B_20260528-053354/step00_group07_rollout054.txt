import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles)

    # Objective function: minimize the negative sum of radii
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ...]
        x_centers = x[:num_circles*2].reshape(num_circles, 2)
        x_radii = x[num_circles*2:].reshape(num_circles)
        return -np.sum(x_radii)

    # Constraints
    def constraint_inside(x):
        x_centers = x[:num_circles*2].reshape(num_circles, 2)
        x_radii = x[num_circles*2:].reshape(num_circles)
        inside = np.all((x_centers[:, 0] >= x_radii) & (x_centers[:, 0] <= 1 - x_radii) &
                        (x_centers[:, 1] >= x_radii) & (x_centers[:, 1] <= 1 - x_radii))
        return inside

    def constraint_overlap(x):
        x_centers = x[:num_circles*2].reshape(num_circles, 2)
        x_radii = x[num_circles*2:].reshape(num_circles)
        # Check pairwise distances
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = x_centers[i, 0] - x_centers[j, 0]
                dy = x_centers[i, 1] - x_centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < x_radii[i] + x_radii[j]:
                    return -1.0
        return 0.0

    # Bounds for each variable
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0 - np.sqrt(2)/2)])

    # Initial guess for the optimization
    initial_guess = np.random.rand(num_circles*3)

    # Optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_inside},
            {'type': 'ineq', 'fun': constraint_overlap}
        ],
        options={'disp': False, 'ftol': 1e-10, 'maxiter': 1000}
    )

    # Extract the best solution
    best_centers = result.x[:num_circles*2].reshape(num_circles, 2)
    best_radii = result.x[num_circles*2:].reshape(num_circles)
    sum_radii = np.sum(best_radii)

    return (best_centers, best_radii, sum_radii)
