import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    num_circles = 26
    initial_centers = np.random.uniform(0.1, 0.9, (num_circles, 2))
    initial_radii = np.random.uniform(0.01, 0.05, num_circles)
    
    def objective(x):
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        radii = x[num_circles * 2:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative to maximize sum_radii

    def constraints(x):
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        radii = x[num_circles * 2:]
        
        # Constraint 1: All circles must be fully inside the unit square
        inside = np.all((radii[:, np.newaxis] <= centers) & (centers <= 1 - radii[:, np.newaxis]), axis=1)
        if not np.all(inside):
            return np.inf
        
        # Constraint 2: No two circles overlap
        dist = np.linalg.norm(centers[:, np.newaxis] - centers[np.newaxis, :], axis=2)
        distances = dist - radii[:, np.newaxis] - radii[np.newaxis, :]
        min_distance = np.min(distances[distances > 0])
        if min_distance < 0:
            return np.inf
        
        return 0
    
    bounds = []
    for _ in range(num_circles * 2):
        bounds.append((0.0, 1.0))
    for _ in range(num_circles):
        bounds.append((0.0, 0.5))  # Radii are at most 0.5 to fit inside the square
    
    result = minimize(objective, x0=np.concatenate([initial_centers.ravel(), initial_radii]), 
                      method='SLSQP', bounds=bounds, constraints=[{'type': 'ineq', 'fun': constraints}])
    
    if not isinstance(result, OptimizeResult) or not result.success:
        raise ValueError("Optimization failed")
    
    centers = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii = result.x[num_circles * 2:]
    sum_radii = -result.fun  # Since we maximized sum_radii by minimizing -sum_radii
    
    return (centers, radii, sum_radii)
