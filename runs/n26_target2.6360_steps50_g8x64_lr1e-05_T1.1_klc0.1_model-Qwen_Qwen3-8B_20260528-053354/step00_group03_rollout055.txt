import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers: evenly spaced grid with some perturbation
    centers = np.array([[(i + np.random.uniform(-0.1, 0.1)) / 5, (j + np.random.uniform(-0.1, 0.1)) / 5]
                        for i in range(5) for j in range(5)])
    radii = np.full(25, 0.1)  # Initial radii

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraint 1: All circles are inside the unit square
    def constraint_inside(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1.0
        return 0.0

    # Constraint 2: No two circles overlap
    def constraint_no_overlap(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1.0
        return 0.0

    # Constraints for optimization
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Initial guess for all variables
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: x and y coordinates must be between 0 and 1, radii must be non-negative
    bounds = [(0, 1) for _ in range(52)] + [(0, 1) for _ in range(26)]

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8)

    centers_result = result.x[:52].reshape(26, 2)
    radii_result = result.x[52:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
