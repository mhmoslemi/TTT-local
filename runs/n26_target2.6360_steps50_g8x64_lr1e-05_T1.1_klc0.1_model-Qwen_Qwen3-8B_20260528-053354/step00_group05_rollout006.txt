import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess: place circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # First, place circles in a grid to get an initial configuration
    # 5 rows and 6 columns
    for i in range(num_circles):
        row = i // 6
        col = i % 6
        centers[i, 0] = 0.2 + col * 0.25
        centers[i, 1] = 0.2 + row * 0.25
        radii[i] = 0.05  # Initial radius
    
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

    def constraints(radii_vec):
        # Constraint: circles must be fully inside the square
        for i in range(num_circles):
            r_i = radii_vec[i]
            if centers[i, 0] - r_i < 0 or centers[i, 0] + r_i > 1 or \
               centers[i, 1] - r_i < 0 or centers[i, 1] + r_i > 1:
                return False
        # Constraint: circles must not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return False
        return True
    
    result = minimize(objective, radii, method='SLSQP', constraints=[lambda x: constraints(x)])
    
    sum_radii = -result.fun
    
    # Ensure centers are valid (based on optimized radii)
    for i in range(num_circles):
        r_i = result.x[i]
        centers[i, 0] = 0.2 + (i // 6) * 0.25
        centers[i, 1] = 0.2 + (i % 6) * 0.25
        # Adjust positions if necessary (simple heuristic)
        if centers[i, 0] - r_i < 0:
            centers[i, 0] = r_i
        if centers[i, 0] + r_i > 1:
            centers[i, 0] = 1 - r_i
        if centers[i, 1] - r_i < 0:
            centers[i, 1] = r_i
        if centers[i, 1] + r_i > 1:
            centers[i, 1] = 1 - r_i
    
    return (centers, result.x, sum_radii)
