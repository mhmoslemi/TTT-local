import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
    radii = np.random.uniform(low=0.01, high=0.05, size=26)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Objective function to minimize
        return objective(radii)

    # Constraints: radius <= 0.5 for all
    def radius_bounds(radii):
        return np.all(radii <= 0.5)

    # Non-overlapping constraint between any two circles
    def non_overlapping(centers, radii):
        n = centers.shape[0]
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.all(dists >= radii_sum)

    # Objective function to maximize sum of radii
    def objective_func(x):
        radii = x[26:]
        total = np.sum(radii)
        # Check constraints
        if not np.all(radii <= 0.5):
            return np.inf
        if not non_overlapping(centers, radii):
            return np.inf
        return -total  # Return negative for minimization

    # Initial guess for all variables
    initial_guess = np.concatenate([centers.ravel(), radii])

    # Optimization
    result = minimize(objective_func, initial_guess, method='L-BFGS-B', bounds=[(0, 1)]*52)

    # Extract centers and radii from result
    centers_opt = result.x[:52].reshape(26, 2)
    radii_opt = result.x[52:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
