import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    # Start with a grid-like arrangement with small radii
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)
    
    # Place circles in a grid-like pattern
    for i in range(n_circles):
        x = i % 5
        y = i // 5
        centers[i, 0] = x / 4 + 0.25
        centers[i, 1] = y / 4 + 0.25
        radii[i] = 0.05
    
    def objective_function(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2*n_circles].reshape(n_circles, 2)
        radii_flat = x[2*n_circles:]
        
        # Objective: maximize sum of radii → minimize -sum(radii)
        sum_radii = np.sum(radii_flat)
        return -sum_radii
    
    def constraints_func(x):
        # Extract centers and radii
        centers_flat = x[:2*n_circles].reshape(n_circles, 2)
        radii_flat = x[2*n_circles:]
        
        # 1. Each circle must be fully inside the unit square
        for i in range(n_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf  # Violation
        
        # 2. No two circles overlap: distance between centers >= sum of radii
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                distance = math.hypot(dx, dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if distance < r_i + r_j:
                    return np.inf  # Violation
        
        # No constraints violated
        return 0
    
    # Prepare initial guess
    x_initial = np.r_[centers.flatten(), radii]
    
    # Set up optimization problem
    bounds = []
    for i in range(n_circles):
        bounds.extend([(0.0, 1.0 - 0.01), (0.0, 1.0 - 0.01), (0.0, 0.5)])  # x, y, radius
    
    # Options for optimization
    options = {
        'ftol': 1e-6,
        'maxiter': 500
    }
    
    # Run the optimization
    result = minimize(
        objective_function,
        x_initial,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints_func},
        options=options
    )
    
    # Extract final solution
    centers = result.x[:2*n_circles].reshape(n_circles, 2)
    radii = result.x[2*n_circles:]
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
