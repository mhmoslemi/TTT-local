import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    # Start with equally spaced circles along rows and columns
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i] = [col / 4 + 0.25, row / 4 + 0.25]
        radii[i] = 0.1

    # Objective function: minimize negative sum of radii
    def objective(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    # Constraint 1: circles must be entirely inside the unit square
    def constraint_inside(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        return np.minimum(centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat)

    # Constraint 2: circles must not overlap
    def constraint_no_overlap(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        dists = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])

    # Optimization setup
    x0 = np.concatenate([centers.ravel(), radii])
    bounds = [(0, 1) for _ in range(num_circles * 2)] + [(0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Suppress warnings for optimization convergence
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-8, options={'maxiter': 1000})

    # Extract results
    centers_result = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii_result = result.x[num_circles * 2:]
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
