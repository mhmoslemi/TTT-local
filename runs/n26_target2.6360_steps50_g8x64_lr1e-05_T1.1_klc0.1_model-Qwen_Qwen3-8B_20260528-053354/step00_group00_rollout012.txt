import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial guess: evenly spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: all equal and small
    radii = np.full(num_circles, 0.05)

    # Optimization function
    def objective(x):
        # x contains radii
        return -np.sum(x)

    # Constraint: all circles must fit inside the unit square
    def constraint_fit(x):
        return np.array([r for r in x if r > 0]) <= 0.5

    # Constraint: no two circles overlap
    def constraint_overlap(x):
        dists = pdist(centers)
        radii = x
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :]) >= 0

    # Optimization options
    bounds = [(0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_fit},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Run the optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract results
    centers = centers
    radii = result.x
    sum_radii = -result.fun

    # Ensure all circles are within the unit square
    for i in range(num_circles):
        r = radii[i]
        if r > 0:
            x, y = centers[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                radii[i] = 0

    return (centers, radii, sum_radii)
