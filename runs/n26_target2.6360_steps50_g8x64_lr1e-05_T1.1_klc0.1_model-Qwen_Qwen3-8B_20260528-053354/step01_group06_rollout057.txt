import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 + 0.5 * (i // 5) / 5
        y = i // 5 + 0.5 * (i % 5) / 5
        centers[i] = [x, y]
    radii = np.full(26, 0.1)

    def objective(r):
        return -np.sum(r)

    def constraints(r, centers):
        r = r.reshape(-1)
        n = len(r)
        cons = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                cons.append(dist - (r[i] + r[j]))
        return np.array(cons)

    def bounds(r):
        return [(0, 1.0) for _ in range(26)]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints, 'args': (centers,)},
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    # Update centers to be fully within the unit square
    centers = np.array(centers)
    radii = result.x
    for i in range(26):
        x, y = centers[i]
        r = radii[i]
        x = max(r, min(x, 1 - r))
        y = max(r, min(y, 1 - r))
        centers[i] = [x, y]

    # Verify and update constraints
    for i in range(26):
        x, y = centers[i]
        r = radii[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            radii[i] = min(r, x - r, 1 - x - r, y - r, 1 - y - r)
            centers[i] = [max(r, min(x, 1 - r)), max(r, min(y, 1 - r))]

    # Final constraints check
    for i in range(26):
        for j in range(i + 1, 26):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = math.hypot(dx, dy)
            if dist < radii[i] + radii[j]:
                radii[i] = (dist - radii[j]) / 2
                radii[j] = (dist - radii[i]) / 2
                centers[i] = [max(radii[i], min(centers[i, 0], 1 - radii[i])),
                              max(radii[i], min(centers[i, 1], 1 - radii[i]))]
                centers[j] = [max(radii[j], min(centers[j, 0], 1 - radii[j])),
                              max(radii[j], min(centers[j, 1], 1 - radii[j]))]

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
