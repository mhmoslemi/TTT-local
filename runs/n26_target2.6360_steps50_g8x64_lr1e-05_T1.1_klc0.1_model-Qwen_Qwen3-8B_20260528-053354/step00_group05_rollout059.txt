import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Define objective function to minimize negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    def constraint_centers(radii):
        # Ensure circles are within the square
        return np.sum((centers[:, 0] - radii) > 0) + np.sum((1 - centers[:, 0] - radii) > 0) + \
               np.sum((centers[:, 1] - radii) > 0) + np.sum((1 - centers[:, 1] - radii) > 0)

    def constraint_overlap(radii):
        # Ensure circles don't overlap
        dist = cdist(centers, centers)
        radii_sum = np.outer(radii, radii)
        return np.sum(dist < radii_sum)

    # Prepare bounds for radii
    bounds = [(0.0, 1.0) for _ in range(26)]

    # Setup constraints for optimization
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_centers(r)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r)}
    ]

    # Run optimization
    result = minimize(objective, np.ones(26), bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
