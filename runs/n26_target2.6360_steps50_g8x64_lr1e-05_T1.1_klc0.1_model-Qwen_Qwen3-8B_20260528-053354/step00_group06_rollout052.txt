import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    num_circles = 26
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, num=5) for y in np.linspace(0.1, 0.9, num=5)])
    centers = np.vstack((centers, [[0.5, 0.5]]))  # Add center for better packing
    centers = np.vstack((centers, [[0.1, 0.5], [0.9, 0.5], [0.5, 0.1], [0.5, 0.9]]))  # Add boundary points
    centers = np.vstack((centers, [[0.25, 0.25], [0.75, 0.25], [0.25, 0.75], [0.75, 0.75]]))  # Add more points
    centers = np.vstack((centers, [[0.4, 0.4], [0.6, 0.4], [0.4, 0.6], [0.6, 0.6]]))  # Add center points
    centers = centers[:26]  # Ensure exactly 26 points
    
    # Initial radii: same for all circles
    radii = np.full(num_circles, 0.1)
    
    # Objective function: minimize negative sum of radii to maximize it
    def objective(x):
        nonlocal radii
        radii = x[:num_circles]
        return -np.sum(radii)
    
    # Constraints: circles must be inside the unit square and not overlap
    constraints = []
    
    # Constraint: each circle is inside the unit square
    def inside_box(r, i):
        x, y = centers[i]
        return (x - r >= 0) * (x + r <= 1) * (y - r >= 0) * (y + r <= 1)
    
    for i in range(num_circles):
        def box_constr(r):
            return inside_box(r, i)
        constraints.append({'type': 'ineq', 'fun': box_constr})
    
    # Constraint: no overlap between circles
    def no_overlap(r, i):
        for j in range(num_circles):
            if i == j:
                continue
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < r[i] + r[j]:
                return -1e-6  # slightly negative to enforce constraint
        return 0
    
    for i in range(num_circles):
        def overlap_constr(r):
            return no_overlap(r, i)
        constraints.append({'type': 'ineq', 'fun': overlap_constr})
    
    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(num_circles)]
    
    # Run optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)
    
    # Extract result
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
