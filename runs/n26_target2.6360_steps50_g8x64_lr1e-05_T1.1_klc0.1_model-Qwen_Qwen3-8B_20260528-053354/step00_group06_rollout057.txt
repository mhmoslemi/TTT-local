import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint

def run_packing():
    # Initial guess: spread circles evenly in the unit square
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Ensure circles are fully inside the unit square
        r = radii.reshape(-1, 1)
        x_centers = centers[:, 0]
        y_centers = centers[:, 1]
        return np.hstack([
            x_centers - r,
            1 - x_centers - r,
            y_centers - r,
            1 - y_centers - r
        ])

    def non_overlap_constraints(radii):
        r = radii.reshape(-1, 1)
        dist = np.sqrt((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 + 
                       (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2)
        return dist - r - r[np.newaxis, :]

    # Optimization bounds
    bounds = Bounds(np.zeros(26), np.inf)

    # Constraints: circles must be inside the square
    # r <= x - r => r <= (1 - r)/2 ??
    # Actually, ensure that x - r >= 0 and 1 - x - r >= 0 => r <= x and r <= 1 - x
    # Same for y
    # So we can just constrain radii to be <= min(x_center, 1 - x_center, y_center, 1 - y_center)
    # We can't directly do it in constraints, so we use bounds
    # Initial guess for radii
    radii = np.minimum(
        np.minimum(centers[:, 0], 1 - centers[:, 0]),
        np.minimum(centers[:, 1], 1 - centers[:, 1])
    )

    # Define constraints
    constraints = [
        LinearConstraint(np.eye(26), -np.inf, np.inf, keep_feasible=True),
        LinearConstraint(
            np.hstack([
                np.zeros((26, 26)),
                np.eye(26)
            ]),
            np.zeros(26),
            np.inf,
            keep_feasible=True
        ),  # To ensure non-overlap
    ]

    # Run optimization
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    centers = np.random.rand(26, 2)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
