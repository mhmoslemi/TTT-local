import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(26, 2)
    radii = np.ones(26) * 0.1  # Initial small radii

    # Objective function: maximize sum of radii
    def objective(x):
        # x is a flattened array: [x1, y1, r1, x2, y2, r2, ...]
        x_vals = x[:26*2].reshape(26, 2)
        r_vals = x[26*2:]
        sum_radii = np.sum(r_vals)
        return -sum_radii  # Minimize negative sum to maximize sum

    # Constraints
    def constraint_square(x):
        x_vals = x[:26*2].reshape(26, 2)
        r_vals = x[26*2:]
        # Check if all circles are inside the square
        x_inside = (x_vals[:, 0] + r_vals <= 1) & (x_vals[:, 0] - r_vals >= 0)
        y_inside = (x_vals[:, 1] + r_vals <= 1) & (x_vals[:, 1] - r_vals >= 0)
        return np.all(x_inside & y_inside)

    def constraint_overlap(x):
        x_vals = x[:26*2].reshape(26, 2)
        r_vals = x[26*2:]
        # Check if all pairs of circles are non-overlapping
        for i in range(26):
            for j in range(i + 1, 26):
                dx = x_vals[i, 0] - x_vals[j, 0]
                dy = x_vals[i, 1] - x_vals[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r_vals[i] + r_vals[j]:
                    return -1
        return 0

    # Initial bounds for x, y, and radii
    bounds = []
    for _ in range(26):
        bounds.extend([(0, 1), (0, 1), (0, 1)])  # [x, y, r]

    # Optimization
    result = minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_square},
            {'type': 'ineq', 'fun': constraint_overlap}
        ],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    # Extract result
    x_vals = result.x[:26*2].reshape(26, 2)
    r_vals = result.x[26*2:]
    sum_radii = -result.fun  # Because we maximized the sum by minimizing its negative

    return (x_vals, r_vals, sum_radii)
