import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    warnings.filterwarnings("ignore", message=".*The detected data type.*")

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.random.rand(26, 2)
        centers[:, 0] = np.random.rand(26)
        centers[:, 1] = np.random.rand(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return 1
        return 0

    # Initial guess with random centers and uniform radii
    initial_centers = np.random.rand(26, 2)
    initial_radii = np.full(26, 0.1)

    # Bounds for radii
    bounds = [(0, 1) for _ in range(26)]

    # Optimization
    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints)

    centers = initial_centers
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
