import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (26, 2))
    radii = np.full(26, 0.02)

    def objective(radii, centers):
        return -np.sum(radii)

    def constraint_bounds(radii, centers):
        for i in range(26):
            r_i = radii[i]
            if r_i < 0:
                return False
            if r_i > 1 - np.min(centers[i])
            or r_i > 1 - np.max(centers[i]):
                return False
        return True

    def constraint_overlap(radii, centers):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_bounds(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers)}
    ]

    result = minimize(
        fun=objective,
        x0=radii,
        args=(centers,),
        method='SLSQP',
        constraints=constraints,
        options={'ftol': 1e-6, 'maxiter': 1000}
    )

    centers_result = centers
    radii_result = result.x
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
