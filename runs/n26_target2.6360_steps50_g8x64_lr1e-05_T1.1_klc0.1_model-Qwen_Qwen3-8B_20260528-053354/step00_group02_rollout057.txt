import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    centers = np.array([[i / np.sqrt(num_circles), j / np.sqrt(num_circles)] for i in range(num_circles) for j in range(num_circles)])
    radii = np.full(num_circles, 0.01)  # Initial small radii
    centers = centers[:num_circles]  # Trim if necessary

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Ensure circles are inside the unit square
        for i in range(num_circles):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - centers[i, 0]:
                return np.inf
            if radii[i] > 1 - centers[i, 1]:
                return np.inf

        # Ensure circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Define bounds for radii (non-negative)
    bounds = Bounds(np.zeros(num_circles), np.ones(num_circles))

    # Optimization problem
    result = minimize(objective, radii, bounds=bounds, constraints=constraints, method='SLSQP', options={'ftol': 1e-9})

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
