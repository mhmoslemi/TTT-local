import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)
    
    def objective(radii):
        return -sum(radii)  # We minimize negative sum to maximize sum
    
    def constraints(centers, radii):
        # Constraint 1: Circles are inside the unit square
        inside = np.all((radii[:, np.newaxis] <= centers) & (centers <= 1 - radii[:, np.newaxis]), axis=1)
        if not np.all(inside):
            return False
        
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        
        return True
    
    # Objective function
    def objective_func(radii):
        return -np.sum(radii)
    
    # Bounds for radii
    bounds = [(0.0, 0.5) for _ in range(26)]
    
    # Initial guess for radii
    initial_radii = np.full(26, 0.1)
    
    # Optimization
    result = minimize(objective_func, initial_radii, method='L-BFGS-B', bounds=bounds)
    
    # Extract results
    optimized_radii = result.x
    optimized_centers = np.random.rand(26, 2)  # Placeholder if centers are not optimized, but in reality centers should be optimized too
    
    # Check constraints and handle errors
    try:
        if not constraints(optimized_centers, optimized_radii):
            raise ValueError("Constraints not satisfied")
    except Exception as e:
        warnings.warn(f"Optimization failed: {e}")
        
    return (optimized_centers, optimized_radii, np.sum(optimized_radii))
