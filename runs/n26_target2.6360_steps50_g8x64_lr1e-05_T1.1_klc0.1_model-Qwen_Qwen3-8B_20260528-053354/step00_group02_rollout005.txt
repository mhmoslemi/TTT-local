import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x[26:])  # Minimize negative sum to maximize sum

    # Constraint 1: Every circle is fully inside the unit square
    def constraint_inside(x):
        centers = x[:26*2].reshape(26, 2)
        radii = x[26:]
        return np.minimum(centers[:, 0] - radii, 1 - centers[:, 0]) - 1e-8

    # Constraint 2: No two circles overlap
    def constraint_no_overlap(x):
        centers = x[:26*2].reshape(26, 2)
        radii = x[26:]
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.minimum(dists - (radii[:, np.newaxis] + radii[np.newaxis, :]), 1e-8)

    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Bounds for centers (0 <= x, y <= 1)
    bounds = [(0, 1) for _ in range(26 * 2)] + [(0, 1) for _ in range(26)]

    # Initial guess flattened
    x0 = np.concatenate([centers.ravel(), radii])

    # Optimization
    result = minimize(objective, x0, bounds=bounds, constraints=cons, method='SLSQP')

    # Extract result
    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
