import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    centers = np.array([[(i / (num_circles - 1)), (j / (num_circles - 1))] for j in range(num_circles) for i in range(num_circles)])
    centers = centers[np.random.permutation(num_circles)]
    radii = np.ones(num_circles) * 0.1
    radii[0] = 0.15  # Adjust first circle to start with a slightly larger radius

    # Objective function: minimize negative sum of radii (i.e., maximize sum)
    def objective(x):
        nonlocal radii
        radii = x[:num_circles]
        return -np.sum(radii)

    # Constraints: each circle must be inside the unit square
    def square_inside(x):
        radii = x[:num_circles]
        return np.array([r <= 1 - r for r in radii])

    # Constraint: distance between centers >= sum of radii
    def no_overlap(x):
        radii = x[:num_circles]
        centers = x[num_circles:]
        centers = centers.reshape((num_circles, 2))
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Setup optimization
    x0 = np.concatenate((radii, centers.flatten()))
    cons = [
        {'type': 'ineq', 'fun': square_inside},
        {'type': 'ineq', 'fun': no_overlap}
    ]
    bounds = [(0, 1) for _ in range(num_circles)] + [(0, 1) for _ in range(2 * num_circles)]

    # Run optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-8, options={'maxiter': 500})
    radii_opt = result.x[:num_circles]
    centers_opt = result.x[num_circles:].reshape((num_circles, 2))
    sum_radii_opt = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii_opt)
