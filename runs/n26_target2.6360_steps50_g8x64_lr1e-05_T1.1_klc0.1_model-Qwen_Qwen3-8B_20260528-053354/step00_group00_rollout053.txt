import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1

    def objective(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers = x[:52].reshape(26, 2)
        radii = x[52:]
        constraints = []

        # Constraint 1: Each circle is inside the unit square
        for i in range(26):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf
            constraints.append((x_i - r_i) >= 0)
            constraints.append((x_i + r_i) <= 1)
            constraints.append((y_i - r_i) >= 0)
            constraints.append((y_i + r_i) <= 1)

        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii[i]
                r_j = radii[j]
                if dist < r_i + r_j:
                    return np.inf
                constraints.append(dist >= r_i + r_j)

        return 0

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Setup optimization
    res = minimize(objective, x0, method='SLSQP', constraints=[('ineq', constraints)], tol=1e-8)

    # Extract results
    centers = res.x[:52].reshape(26, 2)
    radii = res.x[52:]
    sum_radii = -res.fun

    return (centers, radii, sum_radii)
