import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: arrange circles in a grid pattern
    num_circles = 26
    initial_centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        initial_centers[i, 0] = col * 0.2 + 0.1
        initial_centers[i, 1] = row * 0.2 + 0.1
    initial_radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -sum(radii)

    def constraints_func(x):
        # x = [centers..., radii...]
        centers = x[:2*num_circles].reshape(num_circles, 2)
        radii = x[2*num_circles:]
        # Constraint 1: all circles are inside the unit square
        for i in range(num_circles):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Constraint 2: no circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -sum(radii)

    # Prepare initial guess
    x0 = np.concatenate([initial_centers.flatten(), initial_radii])

    # Define bounds and constraints for optimization
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0, 1)] * 2)  # bounds for centers
        bounds.extend([(0, 1)])     # bounds for radii

    result = minimize(constraints_func, x0, method='L-BFGS-B', bounds=bounds, tol=1e-6)

    # Extract results
    centers = result.x[:2*num_circles].reshape(num_circles, 2)
    radii = result.x[2*num_circles:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
