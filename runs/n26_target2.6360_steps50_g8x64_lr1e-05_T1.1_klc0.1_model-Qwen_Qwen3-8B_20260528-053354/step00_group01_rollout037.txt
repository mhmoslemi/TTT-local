```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import warnings

def run_packing():
    # Initial guess: place circles in a grid with some spacing
    centers = np.zeros((26, 2))
    radii = np.ones(26) * 0.1  # initial radius
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.5 + 0.25
        centers[i, 1] = row * 0.5 + 0.25

    def objective(radii):
        return -np.sum(radii)  # we minimize -sum to maximize sum

    def constraints(radii):
        # Ensure all circles are within the unit square
        r = radii.reshape(-1, 1)
        in_square = (centers - r >= 0).all(axis=1) & (1 - centers - r >= 0).all(axis=1)
        if not in_square.all():
            return np.inf

        # Ensure circles do not overlap
        dist = cdist(centers, centers)
        r_sum = r + r.reshape(-1, 1)
        overlap = (dist < r_sum).any()
        if overlap:
            return np.inf

        return 0

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            bounds=[(0, 1) for _ in range(26)],
            constraints=[
                {'type': 'ineq', 'fun': constraints}
            ],
            options={'ftol': 1e-9, 'maxiter': 1000, 'disp': False}
        )

    # Check for success and return results
    if result.success:
        return (centers, result.x, -result.fun)
    else:
        return (centers, np.zeros(26), 0.0)
```
