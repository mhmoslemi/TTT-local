import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    
    # Initialize centers with a grid-like pattern, scaled down
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = (col + 0.5) / 5.0
        centers[i, 1] = (row + 0.5) / 5.0
        radii[i] = 0.05

    # Objective function to minimize: negative of sum of radii
    def objective(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ... x26, y26, r26]
        # Reshape into centers and radii
        flat_centers = x[:2*26].reshape(26, 2)
        flat_radii = x[2*26:].reshape(26)
        
        # Ensure radii are non-negative
        flat_radii = np.maximum(flat_radii, 0)
        
        # Constraint 1: all circles are inside the unit square
        inside = np.all((flat_radii <= flat_centers[:, 0]) & (flat_centers[:, 0] <= 1 - flat_radii) &
                        (flat_radii <= flat_centers[:, 1]) & (flat_centers[:, 1] <= 1 - flat_radii), axis=1)
        if not np.all(inside):
            return float('inf')  # Penalize if any circle is outside the square

        # Constraint 2: no two circles overlap
        for i, j in itertools.combinations(range(26), 2):
            dx = flat_centers[i, 0] - flat_centers[j, 0]
            dy = flat_centers[i, 1] - flat_centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < flat_radii[i] + flat_radii[j]:
                return float('inf')  # Penalize if circles overlap

        # Return negative sum of radii to maximize
        return -np.sum(flat_radii)

    # Bounds for x, y, and r for each circle
    bounds = []
    for _ in range(26):
        bounds.extend([(0, 1), (0, 1), (0, 0.5)])

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds)

    # Extract results
    x_opt = result.x
    flat_centers = x_opt[:2*26].reshape(26, 2)
    flat_radii = x_opt[2*26:].reshape(26)
    sum_radii = np.sum(flat_radii)

    return (flat_centers, flat_radii, sum_radii)
