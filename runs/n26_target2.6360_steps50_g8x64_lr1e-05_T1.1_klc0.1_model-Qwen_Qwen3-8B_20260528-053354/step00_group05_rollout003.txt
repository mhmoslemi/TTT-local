import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid, with equal spacing
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.05)
    grid = np.linspace(0.1, 0.9, 5)
    count = 0
    for i in range(5):
        for j in range(5):
            if count < num_circles:
                centers[count] = [grid[i], grid[j]]
                count += 1

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(centers, radii):
        return np.array([radii[i] <= (1 - centers[i, 0]) and radii[i] <= (1 - centers[i, 1]) for i in range(num_circles)])

    def constraint_no_overlap(centers, radii):
        n = num_circles
        distances = np.zeros(int(n * (n - 1) / 2))
        k = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                distances[k] = dist - radii[i] - radii[j]
                k += 1
        return distances

    def objective_wrapper(x):
        # x is a flat array of radii
        radii = x
        return -np.sum(radii)

    def constraint_circle_inside_wrapper(x):
        radii = x
        return np.array([radii[i] <= (1 - centers[i, 0]) and radii[i] <= (1 - centers[i, 1]) for i in range(num_circles)])

    def constraint_no_overlap_wrapper(x):
        radii = x
        n = num_circles
        distances = np.zeros(int(n * (n - 1) / 2))
        k = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                distances[k] = dist - radii[i] - radii[j]
                k += 1
        return distances

    bounds = [(0, 0.5) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside_wrapper},
        {'type': 'ineq', 'fun': constraint_no_overlap_wrapper}
    ]

    result = minimize(
        objective_wrapper,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8
    )

    radii_opt = result.x
    sum_radii = np.sum(radii_opt)
    centers_final = centers.copy()
    return centers_final, radii_opt, sum_radii
