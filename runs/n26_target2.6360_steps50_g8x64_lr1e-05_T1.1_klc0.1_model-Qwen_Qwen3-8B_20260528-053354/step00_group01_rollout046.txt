import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: distribute circles evenly
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radius guess: small value
    radii = np.full(26, 0.05)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # We minimize the negative sum

    # Constraints
    def constraint_bounds(x):
        # Ensure circles are within bounds
        r = x
        lb = np.zeros(26)
        ub = np.full(26, 1.0 - 1e-6)
        return np.all((r >= lb) & (r <= ub))

    def constraint_overlap(x):
        # Ensure circles do not overlap
        r = x
        n = 26
        dist = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i, j] = np.sqrt(dx**2 + dy**2)
        return np.all(dist >= r[i] + r[j] for i in range(n) for j in range(i + 1, n))

    # Wrap constraints for optimization
    def constraint_func(x):
        if not constraint_bounds(x):
            return np.inf
        if not constraint_overlap(x):
            return np.inf
        return 0

    # Setup bounds and constraints
    bounds = [(0, 1 - 1e-6) for _ in range(26)]
    cons = {'type': 'ineq', 'fun': constraint_func}

    # Run optimization
    result = minimize(objective, radii, bounds=bounds, constraints=cons, method='SLSQP', tol=1e-6)

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
