import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    centers = np.zeros((26, 2))
    radii = np.full(26, 0.1)
    for i in range(26):
        x = i // 5 / 5.0
        y = i % 5 / 5.0
        centers[i] = [x, y]
    
    # Define objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Define constraints: 1. Each circle must fit in the unit square
    #                          2. No two circles overlap
    def constraint_fit(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i] = [i // 5 / 5.0, i % 5 / 5.0]
        for i in range(26):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        return True
    
    def constraint_overlap(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x, y = centers[i]
            centers[i] = [x, y]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Wrap the constraints in a way suitable for scipy.optimize
    def constraint(x):
        if not constraint_fit(x):
            return np.inf
        if not constraint_overlap(x):
            return np.inf
        return 0
    
    # Bounds: radii must be non-negative
    bounds = [(0, 0.5) for _ in range(26)]
    
    # Optimization settings
    options = {'disp': False, 'ftol': 1e-8, 'maxfev': 10000}
    
    # Run optimization
    try:
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraint}, options=options)
    except:
        # If scipy.optimize is not available, return initial guess
        return (centers, radii, np.sum(radii))
    
    radii_result = result.x
    centers_result = np.zeros((26, 2))
    for i in range(26):
        x = i // 5 / 5.0
        y = i % 5 / 5.0
        centers_result[i] = [x, y]
    
    sum_radii = np.sum(radii_result)
    return (centers_result, radii_result, sum_radii)
