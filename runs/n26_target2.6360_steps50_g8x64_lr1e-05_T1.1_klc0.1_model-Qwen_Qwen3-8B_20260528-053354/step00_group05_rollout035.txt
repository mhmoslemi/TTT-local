import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    
    # Objective function: maximize sum of radii (equivalent to minimizing negative sum)
    def objective(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        return -np.sum(radii_flat)

    # Constraints: 1) All circles inside the unit square
    def constraint_inside(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        return np.concatenate([radii_flat, 1 - radii_flat - centers_flat[:, 0], centers_flat[:, 0], 1 - radii_flat - centers_flat[:, 1], centers_flat[:, 1]])

    # Constraint: 2) No overlap between circles
    def constraint_no_overlap(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        dists = np.zeros(26 * 25 // 2)
        idx = 0
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dists[idx] = np.sqrt(dx**2 + dy**2) - (radii_flat[i] + radii_flat[j])
                idx += 1
        return dists

    # Bounds: All centers must be in [0, 1], all radii must be >= 0 and <= 1
    bounds = [(0, 1) for _ in range(52)] + [(0, 1) for _ in range(26)]

    # Optimization problem
    res = minimize(
        objective,
        np.concatenate([centers.flatten(), radii]),
        constraints=[
            {'type': 'ineq', 'fun': constraint_inside},
            {'type': 'ineq', 'fun': constraint_no_overlap}
        ],
        bounds=bounds,
        method='SLSQP',
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract result
    centers_opt = res.x[:52].reshape(26, 2)
    radii_opt = res.x[52:]
    sum_radii = -res.fun

    return (centers_opt, radii_opt, sum_radii)
