import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.full(n_circles, 0.1)
    
    # Initial grid placement
    for i in range(n_circles):
        row = i // 5
        col = i % 5
        centers[i] = [col * 0.25 + 0.125, row * 0.25 + 0.125]
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraint_circle_inside(radii, centers):
        # Ensure each circle is inside the unit square
        x, y = centers.T
        return (x - radii) >= 0.0, (1.0 - x - radii) >= 0.0, (y - radii) >= 0.0, (1.0 - y - radii) >= 0.0
    
    def constraint_no_overlap(radii, centers):
        # Ensure no two circles overlap
        dist = cdist(centers, centers)
        radii_sum = radii.reshape(-1, 1) + radii.reshape(1, -1)
        return dist - radii_sum >= 0.0
    
    # Optimization problem
    bounds = [(0.0, 0.5) for _ in range(n_circles)]
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)[0]},
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)[1]},
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)[2]},
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)[3]},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)[0]}
    ]
    
    result = minimize(objective, radii, bounds=bounds, constraints=constraints)
    
    centers = np.array(centers)
    radii = np.array(result.x)
    sum_radii = -result.fun
    
    return (centers, radii, sum_radii)
