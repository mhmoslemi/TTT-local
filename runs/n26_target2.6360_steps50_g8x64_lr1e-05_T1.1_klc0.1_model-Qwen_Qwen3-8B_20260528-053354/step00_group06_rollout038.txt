import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers (randomly placed)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Constraint 1: Each circle must be within the unit square
        inbounds = []
        for i in range(26):
            r = radii[i]
            inbounds.append(r <= centers[i, 0] - r)
            inbounds.append(r <= centers[i, 1] - r)
            inbounds.append(1 - r - centers[i, 0] >= r)
            inbounds.append(1 - r - centers[i, 1] >= r)
        # Constraint 2: Circles must not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = math.hypot(dx, dy)
                inbounds.append(distance >= radii[i] + radii[j])
        return inbounds

    # Bounds for all radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(26)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'disp': False, 'ftol': 1e-9}
    )

    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
