import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: circles arranged in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = i % 5 + 0.5
        centers[i, 1] = i // 5 + 0.5

    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            centers[i, 0] = i % 5 + 0.5
            centers[i, 1] = i // 5 + 0.5
        # Ensure circles are inside the unit square
        for i in range(26):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        # Ensure circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Define the optimization problem
    bounds = [(0, 1) for _ in range(26)]
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, bounds=bounds, constraints={"type": "ineq", "fun": constraints})

    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = i % 5 + 0.5
        centers[i, 1] = i // 5 + 0.5

    return (centers, result.x, -result.fun)
