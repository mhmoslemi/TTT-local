import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult

def run_packing():
    # Initial guess: evenly spaced circles in grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial guess for radii: all equal
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.array([r <= 1 - np.sqrt(centers[i, 0]**2 + centers[i, 1]**2) for i, r in enumerate(radii)])

    def constraint_no_overlap(radii):
        total = []
        for i in range(26):
            for j in range(i + 1, 26):
                dist = np.linalg.norm(centers[i] - centers[j])
                total.append(dist - (radii[i] + radii[j]))
        return np.array(total)

    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(objective, radii, method='SLSQP', constraints=constraints, tol=1e-8)

    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
