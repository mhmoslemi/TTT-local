import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers (randomly placed in the unit square)
    np.random.seed(42)  # For reproducibility
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Initial small radii

    def objective_function(radii):
        return -np.sum(radii)  # We minimize negative sum to maximize the sum

    def constraints(radii):
        # Constraint: all circles must be inside the unit square
        # r_i <= x_i - r_i => x_i >= 2*r_i, same for y_i
        # r_i <= 1 - x_i - r_i => x_i <= 1 - 2*r_i
        # So we ensure 2*r_i <= x_i <= 1 - 2*r_i for each circle
        # This is handled via bounds, not constraints

        # Constraint: no two circles overlap
        # Distance between any two centers >= r_i + r_j
        N = len(radii)
        for i in range(N):
            for j in range(i + 1, N):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define bounds for radii (non-negative and within square constraints)
    bounds = [(0, 0.5) for _ in range(26)]  # max radius is 0.5 for non-overlapping

    # Optimization
    result = minimize(
        fun=objective_function,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        options={"disp": False}
    )

    radii_opt = result.x
    sum_radii = -result.fun

    # Check if constraints are satisfied (optional)
    # We assume the optimization ensures they are

    # Return the result
    return (centers, radii_opt, sum_radii)
