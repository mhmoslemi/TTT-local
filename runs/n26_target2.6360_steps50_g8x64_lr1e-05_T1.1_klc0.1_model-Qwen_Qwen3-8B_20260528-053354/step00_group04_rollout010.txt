import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid in unit square
    centers = np.array([[x / 5, y / 5] for x in range(5) for y in range(5)])
    centers = np.vstack((centers, np.array([[0.5, 0.25], [0.5, 0.75], [0.25, 0.5], [0.75, 0.5]])))
    centers = np.vstack((centers, np.array([[0.25, 0.25], [0.25, 0.75], [0.75, 0.25], [0.75, 0.75]])))
    centers = np.vstack((centers, np.array([[0.5, 0.5]])))

    # Initial radii
    radii = np.full(26, 0.01)

    def objective(radii, centers):
        return -np.sum(radii)  # We minimize the negative sum to maximize it

    def constraint_box(radii, centers):
        # Ensure all circles are within the unit square
        return np.concatenate([radii, 1 - radii - centers[:, 0], 1 - radii - centers[:, 1]])

    def constraint_overlap(radii, centers):
        # Ensure no two circles overlap
        dists = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dists - radii_sum

    # Define bounds for radii
    bounds = [(0, 1) for _ in range(26)]

    # Set up optimization problem
    result = minimize(
        fun=lambda r: objective(r, centers),
        x0=radii,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraint_box(r, centers)},
            {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers).flatten()}
        ],
        bounds=bounds
    )

    # Extract results
    final_radii = result.x
    final_centers = centers
    sum_radii = -result.fun

    return (final_centers, final_radii, sum_radii)
