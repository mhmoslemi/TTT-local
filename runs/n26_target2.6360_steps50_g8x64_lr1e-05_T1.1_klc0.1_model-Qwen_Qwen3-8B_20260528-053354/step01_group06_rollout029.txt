import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with some radius
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: circles must be inside the square
        # Constraint 2: circles must not overlap
        # Constraint 3: radii must be non-negative

        # Convert radii to a numpy array
        radii = np.array(radii).reshape(-1)
        for i in range(num_circles):
            if radii[i] < 0:
                return np.inf
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return np.inf

        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    bounds = [(0, 1 - 0.1) for _ in range(num_circles)]

    result = minimize(objective, radii, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints}, bounds=bounds)

    if result.success:
        radii_opt = result.x
        centers_opt = centers
        sum_radii = np.sum(radii_opt)
        return (centers_opt, radii_opt, sum_radii)
    else:
        warnings.warn("Optimization failed.")
        return (centers, radii, 0.0)
