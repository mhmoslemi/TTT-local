import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = i % 5 / 4.0 + 0.125
        centers[i, 1] = i // 5 / 4.0 + 0.125

    radii = np.full(26, 0.1)
    sum_radii = 0.0
    try:
        # Optimization function: maximize sum of radii
        def objective(radii):
            return -np.sum(radii)

        # Constraints: circles must be inside the square, no overlap between circles
        cons = []
        for i in range(26):
            def inside_constraint(radii):
                r = radii[i]
                return [r - centers[i, 0], 1 - centers[i, 0] - r, r - centers[i, 1], 1 - centers[i, 1] - r]
            cons.append({'type': 'ineq', 'fun': inside_constraint})

        def no_overlap(radii):
            r = radii.reshape(1, -1)
            dist = cdist(centers, centers)
            return dist - (r @ r.T)
        cons.append({'type': 'ineq', 'fun': no_overlap})

        # Bounds: radii must be non-negative
        bounds = [(0.0, 1.0) for _ in range(26)]

        # Run the optimization
        result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

        # Update the result
        radii = result.x
        sum_radii = np.sum(radii)
        centers = np.array(centers)
        return (centers, radii, sum_radii)
    except:
        # Return a default configuration if optimization fails
        default_radii = np.full(26, 0.1)
        return (centers, default_radii, np.sum(default_radii))
