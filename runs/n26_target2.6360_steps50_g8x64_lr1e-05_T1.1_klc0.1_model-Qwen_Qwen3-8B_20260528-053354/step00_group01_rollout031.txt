import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for circle centers and radii (random positions and small radii)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05

    def objective_function(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:].reshape(26)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative sum to maximize sum

    def constraints(x):
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:].reshape(26)

        # Constraint 1: All circles are inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return False
        # Constraint 2: Circles do not overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return False
        return True

    # Combine objective and constraints into a single function
    def objective_and_constraints(x):
        # Objective function
        centers_flat = x[:26*2].reshape(26, 2)
        radii_flat = x[26*2:].reshape(26)
        sum_radii = np.sum(radii_flat)
        # Constraints
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf
        return -sum_radii

    # Flatten initial guess
    x0 = np.concatenate([centers.flatten(), radii.flatten()])
    # Bounds for centers (x and y between 0 and 1) and radii (non-negative)
    bounds = [(0, 1) for _ in range(26*2)] + [(0, 1) for _ in range(26)]

    # Optimization
    result = minimize(objective_and_constraints, x0, bounds=bounds, method='SLSQP', options={'maxiter': 1000})
    if not result.success:
        return (centers, radii, np.sum(radii))

    # Extract results
    centers_opt = result.x[:26*2].reshape(26, 2)
    radii_opt = result.x[26*2:].reshape(26)
    sum_radii = np.sum(radii_opt)
    return (centers_opt, radii_opt, sum_radii)
