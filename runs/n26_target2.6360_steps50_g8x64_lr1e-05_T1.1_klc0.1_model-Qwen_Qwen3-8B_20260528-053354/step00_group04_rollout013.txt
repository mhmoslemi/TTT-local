import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii

    def constraints(radii):
        centers = np.zeros((len(radii), 2))
        for i in range(len(radii)):
            # Initial guess for center positions using grid
            centers[i, 0] = 0.2 + i * 0.1 + 0.05 * np.sin(i * 0.1)
            centers[i, 1] = 0.2 + i * 0.1 + 0.05 * np.cos(i * 0.1)
        constraints = []
        # Constraint 1: circles inside the unit square
        for i in range(len(radii)):
            r = radii[i]
            if r > 0:
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (1 - r - centers[i, 0])})
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - (1 - r - centers[i, 1])})
        # Constraint 2: circles not overlapping
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                r_i = radii[i]
                r_j = radii[j]
                if r_i > 0 and r_j > 0:
                    def dist_con(radii):
                        return (np.linalg.norm(centers[i] - centers[j]) - (radii[i] + radii[j]))
                    constraints.append({'type': 'ineq', 'fun': dist_con})
        return constraints

    # Initial guess for radii (equal for all)
    initial_radii = np.full(26, 0.1)
    # Optimization
    result = minimize(objective, initial_radii, constraints=constraints(initial_radii), method='SLSQP')
    radii = result.x
    centers = np.zeros((26, 2))
    for i in range(26):
        x = 0.1 + i * 0.1 + 0.05 * np.sin(i * 0.1)
        y = 0.1 + i * 0.1 + 0.05 * np.cos(i * 0.1)
        if radii[i] > 0:
            centers[i, 0] = x
            centers[i, 1] = y
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
