```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(6)])
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Bounds: radii >= 0, and max radius such that circle fits in [0,1]x[0,1]
        bounds = []
        for i in range(num_circles):
            bounds.append((0, 1 - np.min(centers[i])))
        return bounds

    def inequality_constraints(radii):
        # Ensure all circles are within the unit square
        in_bounds = np.all((radii <= np.min(centers, axis=1)) & (radii <= np.min(1 - centers, axis=1)), axis=1)
        if not np.all(in_bounds):
            return np.inf
        # Ensure no two circles overlap
        dists = pdist(centers)
        radii_sum = radii[np.newaxis, :] + radii[:, np.newaxis]
        overlaps = np.any(dists < radii_sum)
        if overlaps:
            return np.inf
        return 0

    # Optimizer setup
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=constraints(radii),
        constraints={'type': 'ineq', 'fun': inequality_constraints},
        options={'ftol': 1e-9, 'disp': False}
    )
    if result.success:
        centers = np.array([[i / 5, j / 5] for i in range(5) for j in range(6)])
        radii = result.x
        sum_radii = -result.fun
        return (centers, radii, sum_radii)
    else:
        warnings.warn("Optimization failed to converge")
        return (np.zeros((26, 2)), np.zeros(26), 0.0)
```
