import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess for centers and radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles) * 0.1
    radii[radii < 0.01] = 0.01  # Ensure minimal radii to avoid zero

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraints(radii):
        # Convert radii to a numpy array
        radii = np.array(radii)
        centers = np.random.rand(num_circles, 2)  # Reinitialize centers if constraints not met
        # Check that all circles are within the unit square
        for i in range(num_circles):
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf  # Penalize if any circle is outside the square
        # Check that no two circles overlap
        dists = cdist(centers, centers)
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                if dists[i, j] < radii[i] + radii[j]:
                    return np.inf  # Penalize if circles overlap
        return -np.sum(radii)  # Return negative sum to minimize

    # Optimization settings
    bounds = [(0, 1) for _ in range(num_circles)]  # Radii bounds

    # Run optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract results
    centers = np.random.rand(num_circles, 2)  # Re-initialize centers after optimization
    sum_radii = result.fun
    return (centers, np.abs(result.x), sum_radii)
