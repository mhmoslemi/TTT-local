```python
import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraint_inside(x):
        for i in range(26):
            r = radii[i]
            x_i, y_i = x[2*i], x[2*i + 1]
            if not (r <= x_i <= 1 - r and r <= y_i <= 1 - r):
                return -np.inf
        return 0

    def constraint_overlap(x):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = x[2*i] - x[2*j]
                dy = x[2*i + 1] - x[2*j + 1]
                dist = np.hypot(dx, dy)
                r_i = radii[i]
                r_j = radii[j]
                if dist < r_i + r_j - 1e-8:
                    return -np.inf
        return 0

    def bounds():
        return [(0, 1) for _ in range(52)]

    def initial_guess():
        centers = np.random.rand(26, 2)
        radii = np.random.rand(26)
        radii = np.minimum(radii, 0.1)
        for i in range(26):
            r = radii[i]
            x, y = centers[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                centers[i] = [r + np.random.rand(), r + np.random.rand()]
        return centers.flatten(), radii

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        x0, radii = initial_guess()
        cons = [
            {'type': 'ineq', 'fun': constraint_inside},
            {'type': 'ineq', 'fun': constraint_overlap}
        ]
        result = minimize(objective, x0, bounds=bounds(), constraints=cons, method='SLSQP')
        centers = result.x.reshape(26, 2)
        sum_radii = -result.fun
        return (centers, radii, sum_radii)
```
