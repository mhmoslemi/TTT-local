import numpy as np
from scipy.optimize import minimize
import scipy.optimize as opt

def run_packing():
    # Initial guess: placing circles in a grid pattern
    def initial_centers():
        centers = []
        for i in range(26):
            x = (i % 5 + 0.5) / 5
            y = (i // 5 + 0.5) / 5
            centers.append([x, y])
        return np.array(centers)
    
    # Objective function to minimize negative sum of radii
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraints for each circle to be inside the unit square
    def in_square(r, i):
        return r <= centers[i, 0] <= 1 - r and r <= centers[i, 1] <= 1 - r
    
    # Constraints for non-overlapping circles
    def no_overlap(radii, i, j):
        dx = centers[i, 0] - centers[j, 0]
        dy = centers[i, 1] - centers[j, 1]
        dist = np.sqrt(dx*dx + dy*dy)
        return dist >= radii[i] + radii[j]
    
    # Define initial centers
    centers = initial_centers()
    radii = np.zeros(26)
    
    # Set up bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(26)]
    
    # Constraints: in_square for all circles
    constraints = []
    for i in range(26):
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 0], 'args': (i, )})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - centers[i, 0], 'args': (i, )})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 1], 'args': (i, )})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r[i] - centers[i, 1], 'args': (i, )})
    
    # Constraints: no overlap between circles
    for i in range(26):
        for j in range(i + 1, 26):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2) - r[i] - r[j]})

    # Optimization
    result = minimize(objective, radii, args=(centers,), method='SLSQP', bounds=bounds, constraints=constraints)
    
    # Extract optimal result
    radii = result.x
    sum_radii = np.sum(radii)
    centers = initial_centers()
    
    return (centers, radii, sum_radii)
