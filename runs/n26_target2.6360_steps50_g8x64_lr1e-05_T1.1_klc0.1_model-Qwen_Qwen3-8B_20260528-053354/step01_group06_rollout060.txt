import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimalResult
import warnings

def run_packing():
    # Initial guess: 26 equally spaced circles along a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5) / 4.0 + 0.125
        centers[i, 1] = (i // 5) / 4.0 + 0.125

    # Optimization parameters
    bounds = [(0.0, 1.0) for _ in range(2 * num_circles)]
    constraints = []

    def objective(radii):
        return -np.sum(radii)

    def constraint_center_inside(radii):
        return np.array([r for r in radii])

    def constraint_no_overlap(radii):
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraint_callback(res):
        if res.fun < -1e-6:
            warnings.warn("Optimization did not converge to feasible solution")

    # Initial radii guess: all 0.1
    initial_radii = np.full(num_circles, 0.1)

    # Add constraints for circle placement
    constraints.append({'type': 'ineq', 'fun': lambda r: r})
    constraints.append({'type': 'ineq', 'fun': lambda r: np.array([1 - r[i] - centers[i, 0] for i in range(num_circles)])})
    constraints.append({'type': 'ineq', 'fun': lambda r: np.array([1 - r[i] - centers[i, 1] for i in range(num_circles)])})
    constraints.append({'type': 'ineq', 'fun': lambda r: np.array([centers[i, 0] - r[i] for i in range(num_circles)])})
    constraints.append({'type': 'ineq', 'fun': lambda r: np.array([centers[i, 1] - r[i] for i in range(num_circles)])})
    constraints.append({'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r).flatten()})

    # Optimization
    result = minimize(
        fun=objective,
        x0=initial_radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-6,
        options={'maxiter': 1000, 'ftol': 1e-9, 'disp': False},
        callback=constraint_callback
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
