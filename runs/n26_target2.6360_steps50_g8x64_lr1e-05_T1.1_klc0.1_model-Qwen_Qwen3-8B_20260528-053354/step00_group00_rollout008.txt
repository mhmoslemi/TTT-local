import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial guess for radii: uniform
    radii[:] = 0.1
    
    # Initial centers in a grid pattern with spacing of 0.4
    for i in range(num_circles):
        row = i // 4
        col = i % 4
        centers[i] = [col * 0.4 + 0.2, row * 0.4 + 0.2]
    
    # Objective function: maximize the sum of radii
    def objective(x):
        # x is flattened array of [r1, r2, ..., r26, x1, y1, x2, y2, ...]
        r = x[:num_circles]
        pos = x[num_circles:].reshape((num_circles, 2))
        
        # Enforce non-negative radii
        if np.any(r < 0):
            return np.inf
        
        # Enforce circles inside the unit square
        for i in range(num_circles):
            x_i, y_i = pos[i]
            r_i = r[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        
        # Enforce no overlaps
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = pos[i][0] - pos[j][0]
                dy = pos[i][1] - pos[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return np.inf
        
        # Return negative sum of radii for maximization
        return -np.sum(r)
    
    # Bounds for radii (non-negative) and positions (within [0,1])
    bounds = []
    for i in range(num_circles):
        bounds.append((0, 0.55))  # Radius can't be more than 0.55
        bounds.append((0, 1))
        bounds.append((0, 1))
    
    # Optimization
    result = minimize(objective, x0=np.concatenate((radii, centers.flatten())), bounds=bounds, method='SLSQP', tol=1e-6)
    
    # Extract results
    r = result.x[:num_circles]
    pos = result.x[num_circles:].reshape((num_circles, 2))
    
    # Check constraints
    for i in range(num_circles):
        x_i, y_i = pos[i]
        r_i = r[i]
        if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
            r[i] = max(0, min(r[i], x_i - r_i, 1 - x_i - r_i, y_i - r_i, 1 - y_i - r_i))
    
    # Final check for overlaps (to ensure correctness)
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = pos[i][0] - pos[j][0]
            dy = pos[i][1] - pos[j][1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < r[i] + r[j]:
                r[i] = (dist - r[j]) / 2
                r[j] = (dist - r[i]) / 2
    
    # Final sum of radii
    sum_radii = np.sum(r)
    
    # Return results
    return (np.array(pos), np.array(r), sum_radii)
