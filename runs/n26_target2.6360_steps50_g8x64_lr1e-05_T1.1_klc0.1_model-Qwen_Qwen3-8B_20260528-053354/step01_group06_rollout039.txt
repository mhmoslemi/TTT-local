import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    def objective(radii):
        return -np.sum(radii)

    def constraints(c):
        centers = c[:52].reshape(26, 2)
        radii = c[52:]
        # Constraint 1: All circles are inside the unit square
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Initial guess: 26 random circles inside the square
    centers = np.random.uniform(0.1, 0.9, (26, 2))
    radii = np.random.uniform(0.01, 0.05, 26)
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization
    result = minimize(objective, initial_guess, method='L-BFGS-B', constraints={'type': 'ineq', 'fun': constraints})
    if result.success:
        centers_opt = result.x[:52].reshape(26, 2)
        radii_opt = result.x[52:]
        sum_radii = -result.fun
        return (centers_opt, radii_opt, sum_radii)
    else:
        # Fallback: return initial guess if optimization fails
        return (centers, radii, np.sum(radii))
