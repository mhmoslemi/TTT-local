import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    centers = np.array([[i / 5 + 0.1, j / 5 + 0.1] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [0.5, 0.5]])
    centers = np.vstack([centers, [0.25, 0.75], [0.75, 0.75], [0.25, 0.25], [0.75, 0.25]])
    centers = np.vstack([centers, [0.5, 0.25], [0.5, 0.75], [0.125, 0.5], [0.375, 0.5],
                         [0.625, 0.5], [0.875, 0.5], [0.5, 0.125], [0.5, 0.375],
                         [0.5, 0.625], [0.5, 0.875], [0.25, 0.375], [0.75, 0.375],
                         [0.25, 0.625], [0.75, 0.625], [0.375, 0.25], [0.625, 0.25],
                         [0.375, 0.75], [0.625, 0.75], [0.125, 0.25], [0.125, 0.75],
                         [0.875, 0.25], [0.875, 0.75]])
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.vstack([np.array([0.5, 0.5]), np.array([0.25, 0.75]),
                             np.array([0.75, 0.75]), np.array([0.25, 0.25]),
                             np.array([0.75, 0.25]), np.array([0.5, 0.25]),
                             np.array([0.5, 0.75]), np.array([0.125, 0.5]),
                             np.array([0.375, 0.5]), np.array([0.625, 0.5]),
                             np.array([0.875, 0.5]), np.array([0.5, 0.125]),
                             np.array([0.5, 0.375]), np.array([0.5, 0.625]),
                             np.array([0.5, 0.875]), np.array([0.25, 0.375]),
                             np.array([0.75, 0.375]), np.array([0.25, 0.625]),
                             np.array([0.75, 0.625]), np.array([0.375, 0.25]),
                             np.array([0.625, 0.25]), np.array([0.375, 0.75]),
                             np.array([0.625, 0.75]), np.array([0.125, 0.25]),
                             np.array([0.125, 0.75]), np.array([0.875, 0.25]),
                             np.array([0.875, 0.75])])
        # Ensure circles are inside the unit square
        for i in range(26):
            if radii[i] < 0:
                return np.inf
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and
                    radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        # Ensure circles don't overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=constraints,
            x0=radii,
            method='SLSQP',
            bounds=[(0, 1)] * 26,
            options={'maxiter': 500, 'ftol': 1e-9}
        )

    # Return the result
    return (centers, result.x, -result.fun)
