import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for centers and radii
    initial_centers = np.random.rand(26, 2)
    initial_radii = np.random.rand(26) * 0.1

    def objective_function(x):
        # Unpack variables
        centers = x[:26*2].reshape(26, 2)
        radii = x[26*2:].reshape(26)

        # Objective: maximize sum of radii (minimize negative sum)
        sum_radii = np.sum(radii)
        return -sum_radii

    def constraints(x):
        # Unpack variables
        centers = x[:26*2].reshape(26, 2)
        radii = x[26*2:].reshape(26)

        # Constraint 1: All circles are inside the unit square
        inside = np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                        (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii), axis=0)
        if not np.all(inside):
            return np.inf

        # Constraint 2: No two circles overlap
        dist = np.linalg.norm(centers[:, np.newaxis, :] - centers[np.newaxis, :, :], axis=2)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        overlap = np.any(dist < radii_sum - 1e-8)
        if overlap:
            return np.inf

        return 0

    # Set up optimization problem
    x0 = np.concatenate((initial_centers.flatten(), initial_radii))
    bounds = []
    for _ in range(26):
        bounds.extend([(0, 1), (0, 1)])  # x and y coordinates for each center
        bounds.extend([(0, 0.5)])  # radius, since r <= 0.5 to fit inside unit square

    # Optimization
    result = minimize(objective_function, x0, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Extract results
    centers = result.x[:26*2].reshape(26, 2)
    radii = result.x[26*2:].reshape(26)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
