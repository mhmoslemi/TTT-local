import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    r = 1.0 / np.sqrt(num_circles)
    for i in range(num_circles):
        row = i // np.sqrt(num_circles)
        col = i % np.sqrt(num_circles)
        centers[i] = [col * r, row * r]

    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            centers[i] = [np.random.rand(), np.random.rand()]
        # Ensure circles are inside the unit square
        for i in range(num_circles):
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return np.inf
        # Ensure no overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx * dx + dy * dy)
                if distance < radii[i] + radii[j]:
                    return np.inf
        return -np.sum(radii)

    def callback(x):
        pass

    bounds = [(0, 1) for _ in range(num_circles)]

    # Initial guess for radii
    initial_radii = np.ones(num_circles) * 0.1

    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        callback=callback
    )

    radii = result.x
    sum_radii = -result.fun

    # Generate centers based on radii using a simple greedy approach
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        while True:
            x = np.random.uniform(radii[i], 1 - radii[i])
            y = np.random.uniform(radii[i], 1 - radii[i])
            overlap = False
            for j in range(num_circles):
                if i != j:
                    dx = x - centers[j, 0]
                    dy = y - centers[j, 1]
                    dist = np.sqrt(dx ** 2 + dy ** 2)
                    if dist < radii[i] + radii[j]:
                        overlap = True
                        break
            if not overlap:
                centers[i] = [x, y]
                break

    return centers, radii, sum_radii
