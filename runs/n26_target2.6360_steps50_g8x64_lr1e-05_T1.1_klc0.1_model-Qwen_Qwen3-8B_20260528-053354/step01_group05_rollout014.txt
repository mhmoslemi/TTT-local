import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 circles with small radii
    np.random.seed(42)
    centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
    radii = np.full(26, 0.05)

    def objective(radii_vec):
        return -np.sum(radii_vec)

    def constraints_func(radii_vec):
        # Constraint: each circle is inside the unit square
        for i in range(26):
            if radii_vec[i] > 0:
                if centers[i, 0] - radii_vec[i] < 0 or centers[i, 0] + radii_vec[i] > 1:
                    return np.inf
                if centers[i, 1] - radii_vec[i] < 0 or centers[i, 1] + radii_vec[i] > 1:
                    return np.inf
        # Constraint: no overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_vec[i] + radii_vec[j]:
                    return np.inf
        return -np.sum(radii_vec)

    # Try a few random initial configurations
    best_sum = 0.0
    best_centers = np.zeros((26, 2))
    best_radii = np.zeros(26)

    for _ in range(10):
        # Randomly reset centers and radii for each trial
        centers = np.random.uniform(low=0.1, high=0.9, size=(26, 2))
        radii = np.full(26, 0.05)
        # Perform optimization
        result = minimize(objective, radii, constraints=[{'type': 'ineq', 'fun': constraints_func}])
        if result.success:
            current_sum = np.sum(result.x)
            if current_sum > best_sum:
                best_sum = current_sum
                best_radii = result.x
                best_centers = centers.copy()

    return (best_centers, best_radii, best_sum)
