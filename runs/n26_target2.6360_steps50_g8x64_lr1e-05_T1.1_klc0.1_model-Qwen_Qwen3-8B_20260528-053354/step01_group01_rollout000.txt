import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # Initial grid-based placement
    grid = np.linspace(0.1, 0.9, 5)
    idx = 0
    for i, x in enumerate(grid):
        for j, y in enumerate(grid):
            if idx < num_circles:
                centers[idx] = [x, y]
                radii[idx] = 0.03
            idx += 1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Inequality constraint for circles within the unit square
    def constraint_square(radii_vec):
        constraints = []
        for i in range(num_circles):
            constraints.append(centers[i, 0] - radii_vec[i])
            constraints.append(1 - centers[i, 0] - radii_vec[i])
            constraints.append(centers[i, 1] - radii_vec[i])
            constraints.append(1 - centers[i, 1] - radii_vec[i])
        return np.array(constraints)

    # Inequality constraint for non-overlapping circles
    def constraint_overlap(radii_vec):
        dists = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                dists.append(dist - (radii_vec[i] + radii_vec[j]))
        return np.array(dists)

    # Box constraints: radii >= 0
    bounds = [(0, np.inf) for _ in range(num_circles)]

    # Optimization
    try:
        result = minimize(
            fun=objective,
            x0=radii,
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_square},
                {'type': 'ineq', 'fun': constraint_overlap}
            ],
            method='SLSQP',
            options={'ftol': 1e-9, 'maxiter': 1000}
        )
    except:
        result = minimize(
            fun=objective,
            x0=radii,
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_square},
                {'type': 'ineq', 'fun': constraint_overlap}
            ],
            method='L-BFGS-B',
            options={'ftol': 1e-9, 'maxiter': 1000}
        )

    optimal_radii = result.x
    optimal_centers = centers.copy()
    sum_radii = np.sum(optimal_radii)
    
    return (optimal_centers, optimal_radii, sum_radii)
