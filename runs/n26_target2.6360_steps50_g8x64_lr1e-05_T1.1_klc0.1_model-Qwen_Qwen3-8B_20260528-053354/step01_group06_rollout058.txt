import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles with small radii
    n = 26
    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = i / (n - 1)
        centers[i, 1] = 0.5

    # Initial radii: small positive values
    radii = np.full(n, 0.01)

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Negative for minimization

    # Constraint: circles must be inside the unit square
    def in_square(x):
        return np.array([x[i] <= 1 - x[i + n] for i in range(n)])
    
    # Constraint: circles must not overlap
    def no_overlap(x):
        dists = np.sqrt(np.sum((centers[i] - centers[j])**2 for i in range(n) for j in range(i + 1, n)))
        return dists - (x[i] + x[j]) >= 0

    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': in_square},
        {'type': 'ineq', 'fun': no_overlap}
    ]

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(n)]

    # Optimization
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
