import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced grid with radius 0.05
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    radii[:] = 0.05
    centers = np.array([[x, y] for x in np.linspace(0.1, 0.9, 6) for y in np.linspace(0.1, 0.9, 6)])
    centers = centers[:num_circles]
    
    # Objective function: minimize negative sum of radii to maximize it
    def objective(radii, centers):
        return -np.sum(radii)
    
    # Constraint: circles are inside the unit square
    def inside_box(radii, centers):
        return np.all((centers - radii[:, np.newaxis] >= 0) & (1 - centers - radii[:, np.newaxis] >= 0))
    
    # Constraint: circles do not overlap
    def no_overlap(radii, centers):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Combine constraints
    constraints = [
        {'type': 'ineq', 'fun': lambda x: inside_box(x, centers)},
        {'type': 'ineq', 'fun': lambda x: no_overlap(x, centers)}
    ]
    
    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=lambda x: objective(x, centers),
            x0=radii,
            method='SLSQP',
            bounds=[(0, 0.5) for _ in range(num_circles)],
            constraints=constraints
        )
    
    centers = np.array(result.x).reshape(-1, 2)
    sum_radii = np.sum(result.x)
    return centers, result.x, sum_radii
