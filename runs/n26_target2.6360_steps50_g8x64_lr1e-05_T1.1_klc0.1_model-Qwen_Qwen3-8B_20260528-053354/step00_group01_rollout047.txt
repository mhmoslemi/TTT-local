import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    # Generate initial positions on a grid
    for i in range(num_circles):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]
        radii[i] = 0.01

    def objective(radii, centers):
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint 1: Circles are inside the unit square
        constraints = []
        for i in range(num_circles):
            r_i = radii[i]
            x_i, y_i = centers[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return False
        # Constraint 2: Circles do not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx**2 + dy**2)
                r_i = radii[i]
                r_j = radii[j]
                if dist < r_i + r_j - 1e-8:
                    return False
        return True

    def wrapped_objective(x):
        radii = x
        if not constraints(radii, centers):
            return np.inf
        return -np.sum(radii)

    # Define bounds: radii must be non-negative
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Minimize negative sum of radii (i.e., maximize sum of radii)
    result = minimize(wrapped_objective, radii, bounds=bounds, method='L-BFGS-B', tol=1e-8)

    # Extract results
    radii = result.x
    # Update centers to satisfy constraints (adjust positions if needed)
    # For simplicity, we assume centers are fixed for this example
    # In a full optimization, positions would also be variables
    centers = np.array(centers)
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
