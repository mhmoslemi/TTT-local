import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26)
    radii = np.clip(radii, 0, 0.5)  # Ensure radii are within reasonable bounds

    def objective(x):
        # Reshape x into centers and radii
        centers_flat = x[:26*2].reshape((26, 2))
        radii_flat = x[26*2:].clip(0, 0.5)  # Ensure non-negative and reasonable radii

        # Objective: minimize the negative sum of radii to maximize it
        sum_radii = np.sum(radii_flat)
        return -sum_radii

    def constraints(x):
        # Reshape x into centers and radii
        centers_flat = x[:26*2].reshape((26, 2))
        radii_flat = x[26*2:].clip(0, 0.5)

        # Constraint 1: Each circle is fully inside the unit square
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf

        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf

        return 0

    bounds = [(0, 1) for _ in range(26*2)] + [(0, 0.5) for _ in range(26)]
    result = minimize(objective, np.concatenate([centers.flatten(), radii]), 
                      method='SLSQP', constraints={'type': 'ineq', 'fun': constraints}, 
                      bounds=bounds, tol=1e-6, options={'maxiter': 1000})

    # Extract final centers and radii
    centers_final = result.x[:26*2].reshape((26, 2))
    radii_final = result.x[26*2:].clip(0, 0.5)
    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
