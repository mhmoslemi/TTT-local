import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x = i % 5 / 4
            y = i // 5 / 4
            centers[i] = [x, y]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(26):
            if radii[i] > 1 - centers[i, 0] or radii[i] > 1 - centers[i, 1]:
                return False
        return True

    # Use a genetic algorithm for optimization
    result = minimize(objective, radii, method='SLSQP', constraints=[lambda r: constraints(r)], tol=1e-8)

    # Update centers based on optimal radii
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4
        y = i // 5 / 4
        centers[i] = [x, y]

    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
