import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place all circles in a grid
    centers = np.zeros((26, 2))
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = 0.2 + col * 0.25
        centers[i, 1] = 0.2 + row * 0.25
    radii = np.full(26, 0.1)

    # Objective function to maximize the sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Minimize the negative of sum

    # Bounds for radii: non-negative and fit in square
    bounds = [(0.0, 1.0 - 1e-6) for _ in range(26)]

    # Constraints: circle stays within the square
    def square_constraint(radii_flat):
        r = radii_flat.reshape(-1, 1)
        dists = np.abs(centers - r)
        return np.min(dists, axis=1) - r  # Must be >= 0

    # Constraints: circles do not overlap
    def non_overlap_constraint(radii_flat):
        r = radii_flat.reshape(-1, 1)
        dists = cdist(centers, centers)
        return dists - r - r.T

    # Combine constraints
    cons = []
    cons.append({'type': 'ineq', 'fun': lambda r: square_constraint(r)})
    cons.append({'type': 'ineq', 'fun': lambda r: np.min(non_overlap_constraint(r))})

    # Solve the optimization problem
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP'
    )

    # Extract results
    best_radii = result.x
    sum_radii = np.sum(best_radii)
    best_centers = centers.copy()

    # Ensure all circles fit inside the square and constraints are satisfied (sanity check)
    for i in range(26):
        if best_radii[i] < 1e-8:
            continue
        if best_centers[i, 0] < best_radii[i] or best_centers[i, 0] > 1 - best_radii[i]:
            best_centers[i, 0] = np.clip(best_centers[i, 0], best_radii[i], 1 - best_radii[i])
        if best_centers[i, 1] < best_radii[i] or best_centers[i, 1] > 1 - best_radii[i]:
            best_centers[i, 1] = np.clip(best_centers[i, 1], best_radii[i], 1 - best_radii[i])

    return (best_centers, best_radii, sum_radii)
