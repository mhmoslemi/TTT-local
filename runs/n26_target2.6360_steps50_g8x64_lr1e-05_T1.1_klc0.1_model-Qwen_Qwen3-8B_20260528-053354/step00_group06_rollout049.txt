import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import warnings

def run_packing():
    # Initial guess: evenly spaced grid points
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Objective function: minimize the negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: 1. Circles inside the square, 2. No overlap
    constraints = []

    # Constraint 1: circles inside the square
    def in_square(radii, centers):
        return np.sum((radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) &
                      (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1]))

    constraints.append({'type': 'ineq', 'fun': lambda r: in_square(r, centers)})

    # Constraint 2: no overlap between circles
    def no_overlap(radii, centers):
        N = len(radii)
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.min(dist - radii[:, np.newaxis] - radii[np.newaxis, :])

    constraints.append({'type': 'ineq', 'fun': lambda r: no_overlap(r, centers)})

    # Initial guess for radii
    radii = np.ones(num_circles)

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, radii, method='SLSQP', constraints=constraints)

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii = -result.fun

    return (centers_final, radii_final, sum_radii)
