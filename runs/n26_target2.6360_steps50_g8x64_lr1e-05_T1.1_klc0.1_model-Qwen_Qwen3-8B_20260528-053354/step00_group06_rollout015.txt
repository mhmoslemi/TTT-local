import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: arrange circles in a square grid
    num_circles = 26
    initial_centers = np.array([
        [i / 5 + 0.1, j / 5 + 0.1] for i in range(5) for j in range(6)
    ] + [[0.1, 0.6], [0.1, 0.7], [0.1, 0.8], [0.1, 0.9]]  # additional circles
    )
    initial_radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints: circles must be inside the unit square, and no overlap
    def constraints(radii):
        # Circle centers
        centers = np.zeros((num_circles, 2))
        # Use the initial guess for centers, but allow them to change? No, we fix centers
        # Since we're optimizing radii, we fix centers for simplicity
        centers = initial_centers
        
        # Constraints: radii >= 0
        cons = [{'type': 'ineq', 'fun': lambda r: r}]
        # Constraints: circle inside the unit square
        for i in range(num_circles):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 0]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 0] - r[i]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] - centers[i, 1]})
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i, 1] - r[i]})
        
        # No overlap between circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                def dist_constraint(r):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx ** 2 + dy ** 2)
                    return dist - (r[i] + r[j])
                cons.append({'type': 'ineq', 'fun': dist_constraint})
        
        return cons
    
    # Optimization
    bounds = [(0, 1)] * num_circles
    result = minimize(objective, initial_radii, bounds=bounds, constraints=constraints(initial_radii), method='SLSQP', tol=1e-5)
    
    # Extract result
    centers = initial_centers  # Centers are not optimized, we fix them
    radii = result.x
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
