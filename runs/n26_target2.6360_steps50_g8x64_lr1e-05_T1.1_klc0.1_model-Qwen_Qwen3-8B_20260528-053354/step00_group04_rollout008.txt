```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint1(radii):
        # Each circle is inside the square
        r = radii
        return np.minimum(r, 1 - r - np.min(np.vstack((centers[:, 0], centers[:, 1])), axis=0))

    def constraint2(radii):
        # No two circles overlap
        r = radii
        dists = pdist(centers)
        return np.array([dists[i] - r[i] - r[j] for i, j in enumerate(range(26))])

    # Optimization
    bounds = [(0, 1 - 1e-6) for _ in range(26)]
    cons = [{'type': 'ineq', 'fun': constraint1},
            {'type': 'ineq', 'fun': constraint2}]
    result = minimize(objective, np.full(26, 0.1), method='SLSQP', bounds=bounds, constraints=cons)

    # Extract results
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
```
