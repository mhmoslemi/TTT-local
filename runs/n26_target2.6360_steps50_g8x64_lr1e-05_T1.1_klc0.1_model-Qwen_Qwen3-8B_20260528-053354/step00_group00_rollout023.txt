import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1  # Start with small radii

    # Define objective function to maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraint_circle_in_square(radii, centers):
        # Check if all circles are inside the square
        return np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]))

    def constraint_no_overlap(radii, centers):
        # Check if all pairs of circles are non-overlapping
        dists = cdist(centers, centers)
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Combine all constraints
    constraints = [
        {'type': 'ineq', 'fun': lambda x: constraint_circle_in_square(x, centers)},
        {'type': 'ineq', 'fun': lambda x: constraint_no_overlap(x, centers)}
    ]

    # Bounds on radii
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimize
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract result
    centers = np.array([[0.5, 0.5] for _ in range(num_circles)])  # Use initial guess as centers
    sum_radii = result.fun * -1

    return (centers, result.x, sum_radii)
