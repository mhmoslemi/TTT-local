import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: place 26 circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    # Initial guess for radii: uniform value
    initial_radius = 0.1
    radii[:] = initial_radius

    # Initial placement in a grid
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col / 4.0 + 0.125
        centers[i, 1] = row / 4.0 + 0.125

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(radii, centers):
        # Constraint: each circle must be inside the unit square
        return np.sum((radii - (1 - centers[:, 0]) - (1 - centers[:, 1])) ** 2)

    def constraint_no_overlap(radii):
        # Constraint: distance between centers >= sum of their radii
        dist = cdist(centers, centers)
        sum_radii = radii[:, np.newaxis] + radii[np.newaxis, :]
        return np.sum((dist - sum_radii) ** 2)

    # Set up constraints
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r)}
    ]

    # Set up bounds for radii
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        options={'ftol': 1e-8, 'maxiter': 500}
    )

    # Extract results
    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii = -result.fun

    # Ensure all circles are inside the unit square
    for i in range(num_circles):
        if centers_opt[i, 0] - radii_opt[i] < 0 or centers_opt[i, 0] + radii_opt[i] > 1:
            centers_opt[i, 0] = max(radii_opt[i], min(1 - radii_opt[i], centers_opt[i, 0]))
        if centers_opt[i, 1] - radii_opt[i] < 0 or centers_opt[i, 1] + radii_opt[i] > 1:
            centers_opt[i, 1] = max(radii_opt[i], min(1 - radii_opt[i], centers_opt[i, 1]))

    return (centers_opt, radii_opt, sum_radii)
