import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraints(radii):
        n = len(radii)
        cons = []
        # Boundary constraints: radii <= 0.5 (to fit in [0,1] square)
        for i in range(n):
            cons.append({'type': 'ineq', 'fun': lambda r, i=i: 0.5 - r[i]})
        # Non-overlapping constraints
        for i in range(n):
            for j in range(i + 1, n):
                def cons_ij(r, i=i, j=j):
                    dist = np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2)
                    return dist - (radii[i] + radii[j])
                cons.append({'type': 'ineq', 'fun': cons_ij})
        return cons

    # Initial guess for centers (evenly spaced grid)
    centers = np.array([[(i / 5 + 0.25) if j == 0 else (j / 5 + 0.25) for j in range(5)] for i in range(5)])
    centers = centers.reshape(-1, 2)
    radii = np.full(25, 0.05)  # initial radius for each circle

    # Optimization
    result = minimize(objective, radii, method='SLSQP', constraints=constraints(radii))

    # Extract results
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
