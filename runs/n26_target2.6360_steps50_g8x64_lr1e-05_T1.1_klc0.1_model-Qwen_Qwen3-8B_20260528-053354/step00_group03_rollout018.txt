import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.05)

    # Function to compute the objective (negative sum of radii to maximize)
    def objective(radii):
        return -np.sum(radii)

    # Function to compute constraints (distances between centers >= sum of radii)
    def constraint_distances(centers, radii):
        distances = squareform(pdist(centers))
        sum_r = radii[:, np.newaxis] + radii[np.newaxis, :]
        return distances - sum_r

    # Optimization bounds: radii must be non-negative and fit inside unit square
    bounds = [(0.0, 0.5) for _ in range(num_circles)]

    # Initial centers: grid-based placement
    centers = np.zeros((num_circles, 2))
    idx = 0
    for y in np.linspace(0.1, 0.9, 5):
        for x in np.linspace(0.1, 0.9, 5):
            centers[idx] = [x, y]
            idx += 1

    # Optimize using scipy.optimize.minimize
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={
            'type': 'ineq',
            'fun': lambda r: np.min(constraint_distances(centers, r)),
            'jac': lambda r: np.zeros_like(r)  # Jacobian not used here
        },
        method='SLSQP',
        options={'ftol': 1e-9, 'disp': False}
    )

    # Final centers and radii
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii = np.sum(radii_final)

    # Check if all circles are inside the unit square
    for i in range(num_circles):
        x, y = centers_final[i]
        r = radii_final[i]
        if not (r <= x <= 1 - r and r <= y <= 1 - r):
            warnings.warn(f"Circle {i} is not fully inside the unit square.")

    return (centers_final, radii_final, sum_radii)
