import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 5) for x in np.linspace(0.1, 0.9, 5)])
    radii = np.full(25, 0.05)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add one more circle at the center
    radii = np.append(radii, 0.1)
    
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Minimize negative sum to maximize sum

    def constraint_centers(radii_flat):
        radii = radii_flat.reshape(-1)
        centers = np.vstack((np.array([0.5, 0.5] if np.isnan(radii[-1]) else [0.5, 0.5]), np.array([x, y] for i, (x, y) in enumerate(centers) if i != len(centers)-1)))
        for i in range(len(centers)):
            if radii[i] < 0:
                return np.inf
            if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
                return np.inf
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Initial guess
    initial_guess = np.concatenate((centers[:-1, 0], centers[:-1, 1], radii[:-1]))
    initial_guess = np.append(initial_guess, radii[-1])  # Ensure the last radius is included

    # Setup optimization
    bounds = []
    for i in range(len(centers) - 1):
        bounds.append((0, 1 - 0.001))  # Ensure circle fits inside the square
    bounds.append((0, 0.5))  # Limit center radius to prevent overlap with others

    # Optimization setup
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=objective,
            x0=initial_guess,
            method='L-BFGS-B',
            bounds=bounds,
            constraints={'type': 'ineq', 'fun': constraint_centers}
        )

    if result.success:
        final_radii = result.x[-len(radii):]
        centers = np.array([[0.5, 0.5]] + [[x, y] for i, (x, y) in enumerate(centers) if i != len(centers)-1])
        centers = np.vstack((centers, [0.5, 0.5])) if np.isnan(final_radii[-1]) else centers
        centers = np.vstack((centers[:25], np.array([0.5, 0.5])))
        centers = np.array([[x, y] for x, y in zip(result.x[:2*len(centers)-1:2], result.x[1:2*len(centers)-1:2])])
        if final_radii.shape[0] < len(centers):
            final_radii = np.append(final_radii, [result.x[-1]])
        sum_radii = np.sum(final_radii)
        return (centers, final_radii, sum_radii)
    else:
        return (centers, radii, sum_radii)
