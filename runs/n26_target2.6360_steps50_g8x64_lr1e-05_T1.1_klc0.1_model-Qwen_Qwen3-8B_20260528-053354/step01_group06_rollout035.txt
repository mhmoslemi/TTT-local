import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced points in a grid with some random perturbation
    num_circles = 26
    np.random.seed(42)  # For reproducibility
    centers = np.array([(x, y) for x in np.linspace(0.1, 0.9, 5) for y in np.linspace(0.1, 0.9, 5)])
    centers = np.vstack([centers, np.random.rand(num_circles - len(centers), 2) * 0.8 + 0.1])
    radii = np.full(num_circles, 0.01)

    # Objective function to maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraint 1: Circles must be inside the unit square
    def constraint_inside(radii):
        return np.array([r for r in radii], dtype=np.float64)

    # Constraint 2: No two circles can overlap
    def constraint_overlap(radii, centers):
        distances = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distances[i, j] = math.hypot(dx, dy)
                distances[j, i] = distances[i, j]
        return distances - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Setup optimization
    bounds = [(0.0, 0.5) for _ in range(num_circles)]  # Max radius is 0.5

    def constraint_func(radii):
        dist = constraint_overlap(radii, centers)
        return np.concatenate([dist.flatten(), constraint_inside(radii)])

    def constraint_inside_func(radii):
        return np.array([r for r in radii], dtype=np.float64)

    constraints = [
        {'type': 'ineq', 'fun': lambda x: constraint_inside_func(x)},
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers).flatten()}
    ]

    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
