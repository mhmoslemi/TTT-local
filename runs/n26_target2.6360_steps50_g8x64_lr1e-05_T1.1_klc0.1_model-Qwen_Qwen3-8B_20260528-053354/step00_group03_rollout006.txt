import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced circles with radius 0.1
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i // 5 + 0.1
        y = i % 5 + 0.1
        centers[i] = [x, y]
    radii = np.full(26, 0.1)
    
    # Objective function to minimize: negative sum of radii (maximize)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints
    def constraint_inside(radii):
        # Check if centers are within the unit square
        inside = []
        for i in range(26):
            r = radii[i]
            x, y = centers[i]
            inside.append(r <= x <= 1 - r and r <= y <= 1 - r)
        return np.array(inside)
    
    def constraint_overlap(radii):
        # Check if any two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                r1 = radii[i]
                r2 = radii[j]
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r1 + r2:
                    return False
        return True
    
    # Optimization
    def objective_wrapper(radii):
        return objective(radii)
    
    def constraints_wrapper(radii):
        return [constraint_inside(radii), constraint_overlap(radii)]
    
    bounds = [(0.0, 0.5) for _ in range(26)]
    
    result = minimize(objective_wrapper, radii, method='SLSQP', bounds=bounds, constraints=constraints_wrapper)
    
    # Extract result
    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
