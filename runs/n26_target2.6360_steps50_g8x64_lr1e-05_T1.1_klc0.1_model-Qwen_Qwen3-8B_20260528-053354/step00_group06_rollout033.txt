import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import warnings

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.25, 0.75, 5) for x in np.linspace(0.25, 0.75, 5)])
    radii = np.full(25, 0.1)  # initial guess for radii
    centers = np.vstack((centers, [0.5, 0.5]))  # add the last circle at center
    radii = np.append(radii, 0.1)

    # Optimization objective function
    def objective(x):
        radii = x[:26]
        centers = x[26:]
        centers = centers.reshape((26, 2))
        # Constraint: radii must be non-negative
        if np.any(radii < 0):
            return float('inf')
        # Constraint: circles must be inside the unit square
        inside = np.all((radii[:, np.newaxis] <= centers) & (radii[:, np.newaxis] <= 1 - centers), axis=1)
        if not np.all(inside):
            return float('inf')
        # Constraint: circles must not overlap
        dists = squareform(pdist(centers))
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        overlap = np.any(dists < radii_sum - 1e-8)
        if overlap:
            return float('inf')
        # Objective: maximize sum of radii
        return -np.sum(radii)

    # Define bounds for the optimization variables
    bounds = []
    for i in range(26):
        bounds.append((0, 1))  # radii
    for i in range(26):
        bounds.extend([(0, 1), (0, 1)])  # x and y for each circle

    # Define constraints for the optimization
    constraints = [{
        'type': 'ineq',
        'fun': lambda x: -np.sum((x[:26] <= x[26:])[::2])  # Check if all radii <= x and y
    }, {
        'type': 'ineq',
        'fun': lambda x: -np.sum((x[:26] <= 1 - x[26:])[::2])
    }]

    # Run optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(objective, x0=np.concatenate((radii, centers.flatten())), bounds=bounds, constraints=constraints)

    # Extract the result
    final_radii = result.x[:26]
    final_centers = result.x[26:].reshape((26, 2))
    sum_radii = np.sum(final_radii)

    return final_centers, final_radii, sum_radii
