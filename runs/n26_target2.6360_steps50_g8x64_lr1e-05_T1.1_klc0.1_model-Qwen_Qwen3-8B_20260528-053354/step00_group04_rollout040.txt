from scipy.spatial.distance import cdist

        radii = x
        if np.any(radii < 0):
            return np.inf
        return -np.sum(radii)
    
    def constraint_func(x):
        radii = x
        if np.any(radii < 0):
            return np.inf
        centers = np.random.rand(26, 2)
        for i in range(26):
            for j in range(i + 1, 26):

                return np.inf
        return 0
    
    # Initial guess: random radii
    initial_guess = np.random.rand(26)
    result = minimize(objective_func, initial_guess, method='SLSQP', constraints={'type': 'ineq', 'fun': constraint_func})
    
    radii = result.x
    centers = np.random.rand(26, 2)

    for i in range(26):

    for i in range(26):

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
