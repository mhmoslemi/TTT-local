import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05  # Start with small radii

    # Objective function: minimize negative sum of radii to maximize it
    def objective(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        return -np.sum(radii_flat)

    # Constraints
    def constraint_center_inside(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return 1.0
        return 0.0

    def constraint_no_overlap(x):
        centers_flat = x[:52].reshape(26, 2)
        radii_flat = x[52:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j - 1e-8:
                    return 1.0
        return 0.0

    # Bounds for centers (x and y between 0 and 1)
    bounds = []
    for _ in range(26):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])
    # Bounds for radii (non-negative)
    bounds.extend([(0.0, 0.1) for _ in range(26)])  # Small upper bound as a heuristic

    # Optimization setup
    x0 = np.concatenate([centers.flatten(), radii])
    constraints = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Disable warnings for better optimization
    warnings.filterwarnings("ignore", message=".*Optimization failed.*")

    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-6, options={'maxiter': 1000})

    # Extract results
    centers_final = result.x[:52].reshape(26, 2)
    radii_final = result.x[52:]
    sum_radii_final = -result.fun

    return (centers_final, radii_final, sum_radii_final)
