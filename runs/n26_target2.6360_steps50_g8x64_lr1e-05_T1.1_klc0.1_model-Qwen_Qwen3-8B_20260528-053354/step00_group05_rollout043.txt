import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers (randomly spread within the square)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) / 10.0  # Small initial radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Each circle is inside the unit square
        # r_i <= x_i - r_i and r_i <= y_i - r_i => x_i - r_i >= 0 and y_i - r_i >= 0
        # Also, 1 - r_i - x_i >= 0 and 1 - r_i - y_i >= 0
        # To simplify, use r_i <= 0.5 (since max radius is 0.5 for a circle in the square)
        r = radii
        return np.array([r[i] <= 0.5 for i in range(26)])

    def non_overlap_constraints(radii):
        # Constraint 2: No two circles overlap
        r = radii
        dists = np.zeros(26 * 25 // 2)
        idx = 0
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                dists[idx] = dist - (r[i] + r[j])
                idx += 1
        return dists

    # Pack the radii with constraints
    bounds = [(0, 0.5) for _ in range(26)]
    cons = [
        {'type': 'ineq', 'fun': constraints},
        {'type': 'ineq', 'fun': non_overlap_constraints}
    ]

    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract the optimal result
    optimal_radii = result.x
    sum_radii = -result.fun  # Convert back to positive sum

    return (centers, optimal_radii, sum_radii)
