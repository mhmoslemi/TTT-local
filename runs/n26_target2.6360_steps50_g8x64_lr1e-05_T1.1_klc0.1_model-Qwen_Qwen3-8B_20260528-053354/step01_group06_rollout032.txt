import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize -sum(radii) to maximize sum(radii)
    
    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        # Place centers in a grid-like initial configuration
        for i in range(num_circles):
            x = i % 5 / 4.0
            y = i // 5 / 4.0
            centers[i, :] = [x, y]
        return centers
    
    def bounds(radii):
        return [(0.0, 1.0 - 1e-6) for _ in range(num_circles)]
    
    def constraints_func(radii):
        centers = constraints(radii)
        # Ensure all circles are inside the unit square
        for i in range(num_circles):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf
        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        return -np.sum(radii)
    
    # Run optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            fun=constraints_func,
            x0=radii,
            method='SLSQP',
            bounds=bounds(radii),
            constraints={'type': 'eq', 'fun': lambda x: 0},
            options={'disp': False, 'ftol': 1e-6, 'maxiter': 1000}
        )
    
    # Extract best solution
    best_radii = result.x
    best_centers = constraints(best_radii)
    sum_radii = np.sum(best_radii)
    
    return (best_centers, best_radii, sum_radii)
