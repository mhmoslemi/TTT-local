import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: centers at grid points with small radii
    num_circles = 26
    initial_centers = np.array([[i/num_circles, j/num_circles] for i in range(num_circles) for j in range(num_circles)])
    initial_centers = initial_centers[:num_circles]
    initial_radii = np.full(num_circles, 0.05)
    
    def objective(radii):
        return -sum(radii)  # Minimize negative sum to maximize sum
    
    def constraints(radii):
        centers = np.vstack([initial_centers[i] for i in range(num_circles)])
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Optimization with constraints
    def constraint_func(radii):
        if not constraints(radii):
            return np.inf
        return 0
    
    result = minimize(
        fun=objective,
        x0=initial_radii,
        method='SLSQP',
        constraints=[{'type': 'eq', 'fun': constraint_func}],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )
    
    radii = result.x
    centers = initial_centers[:num_circles]
    
    # Validate constraints
    for i in range(num_circles):
        x, y = centers[i]
        if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
            raise ValueError("Circle is not fully inside the unit square.")
    
    return (centers, radii, -result.fun)
