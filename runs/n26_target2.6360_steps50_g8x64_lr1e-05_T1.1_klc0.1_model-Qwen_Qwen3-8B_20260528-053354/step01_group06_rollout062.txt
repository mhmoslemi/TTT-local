```python
import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: equally spaced grid of 26 points
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
    radii = np.full(num_circles, 0.1)
    
    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        return -np.sum(x)
    
    # Constraints
    def constraint_square(x):
        # Each circle stays within the square
        return np.concatenate([
            x - np.sqrt(1 - x[::2]**2 - x[1::2]**2) > 0,
            x[::2] < 1 - np.sqrt(1 - x[::2]**2 - x[1::2]**2)
        ])
    
    def constraint_overlap(x):
        # Distance between centers >= sum of radii
        n = len(x)
        centers = x.reshape((n//2, 2))
        dists = np.zeros((n//2, n//2))
        for i in range(n//2):
            for j in range(i+1, n//2):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
        return dists - (x[::2] + x[1::2])[:, None] - (x[::2] + x[1::2])[None, :] + np.eye(n//2)
    
    # Flatten radii and centers for the optimization
    x0 = np.concatenate([radii, centers.reshape(-1)])
    
    # Bounds: radii >= 0, centers in [0,1]
    bounds = [(0, 1) for _ in range(len(x0))]
    
    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_square, 'args': (x0[::2], x0[1::2])},
        {'type': 'ineq', 'fun': constraint_overlap, 'args': (x0[::2], x0[1::2])}
    ]
    
    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons)
    
    # Extract results
    radii = result.x[::2]
    centers = result.x[1::2].reshape((num_circles//2, 2))
    
    # Sum of radii
    sum_radii = np.sum(radii)
    
    return (centers, radii, sum_radii)
```
