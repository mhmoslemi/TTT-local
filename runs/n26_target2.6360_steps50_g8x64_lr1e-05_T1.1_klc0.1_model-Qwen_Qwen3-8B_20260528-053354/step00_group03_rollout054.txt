import numpy as np
from scipy.optimize import minimize
import time

def run_packing():
    # Initial guess for center positions (spread out evenly)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Initial small radii

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii, centers):
        # Constraint 1: All circles are within the unit square
        in_square = np.all((radii <= 1 - centers) & (radii <= centers), axis=1)
        # Constraint 2: No two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        no_overlap = np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :], axis=1)
        return in_square & no_overlap

    # Optimization setup
    bounds = [(0, 0.5) for _ in range(26)]  # Radii must be <= 0.5 to fit in unit square
    result = minimize(
        fun=lambda r: -np.sum(r),
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: np.sum(constraints(r, centers))},
        ],
        options={'maxiter': 1000, 'ftol': 1e-8, 'disp': False}
    )

    # Extract results
    radii_opt = result.x
    centers_opt = centers  # Centers are not optimized directly in this simple setup
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
