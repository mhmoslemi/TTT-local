import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess: evenly spaced grid with some spacing
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5
        radii[i] = 0.1  # Initial small radius

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Radii must be non-negative
        for r in radii:
            if r < 0:
                return False
        # Constraint 2: Each circle must be inside the unit square
        for i in range(26):
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Constraint 3: No overlaps
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization
    bounds = [(0.001, 0.5) for _ in range(26)]  # Avoid very small or too large radii

    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=({'type': 'ineq', 'fun': lambda r: 1 - constraints(r)}),
        method='SLSQP',
        tol=1e-6
    )

    # Extract results
    optimal_radii = result.x
    optimal_centers = centers.copy()
    # Ensure no overlaps and inside square due to constraints
    for i in range(26):
        optimal_centers[i, 0] = max(0.001, min(1 - optimal_radii[i], optimal_centers[i, 0]))
        optimal_centers[i, 1] = max(0.001, min(1 - optimal_radii[i], optimal_centers[i, 1]))
    # Final check
    for i in range(26):
        for j in range(i + 1, 26):
            dx = optimal_centers[i, 0] - optimal_centers[j, 0]
            dy = optimal_centers[i, 1] - optimal_centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < optimal_radii[i] + optimal_radii[j]:
                optimal_radii[i] = 0
                optimal_radii[j] = 0
    sum_radii = np.sum(optimal_radii)
    return (optimal_centers, optimal_radii, sum_radii)
