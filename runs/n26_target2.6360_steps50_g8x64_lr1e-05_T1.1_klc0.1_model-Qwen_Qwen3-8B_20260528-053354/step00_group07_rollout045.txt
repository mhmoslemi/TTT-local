import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    num_circles = 26
    grid = np.linspace(0.1, 0.9, 5)  # 5 points in x and y
    centers = np.array([[x, y] for y in grid for x in grid])
    radii = np.full(num_circles, 0.05)

    def objective(radii):
        return -np.sum(radii)  # maximize sum of radii

    def constraint_circle_in_square(radii, centers):
        # Ensure all circles are inside the unit square
        return np.sum((centers[:, 0] - radii) > 0) + np.sum((1 - centers[:, 0] - radii) > 0) + \
               np.sum((centers[:, 1] - radii) > 0) + np.sum((1 - centers[:, 1] - radii) > 0)

    def constraint_no_overlap(radii, centers):
        # Ensure no two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization configuration
    bounds = [(0, 0.5) for _ in range(num_circles)]  # Upper bound for radius
    constraints = [
        {
            'type': 'ineq',
            'fun': lambda r: constraint_circle_in_square(r, centers),
            'jac': lambda r: np.zeros_like(r)
        },
        {
            'type': 'ineq',
            'fun': lambda r: constraint_no_overlap(r, centers),
            'jac': lambda r: np.zeros_like(r)
        }
    ]

    # Optimization setup
    result = minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-6
    )

    # Extract results
    centers_best = copy.deepcopy(centers)
    radii_best = result.x
    sum_radii_best = -result.fun

    # Ensure all circles are within the square
    valid_mask = (radii_best <= centers_best[:, 0]) & (radii_best <= 1 - centers_best[:, 0]) & \
                 (radii_best <= centers_best[:, 1]) & (radii_best <= 1 - centers_best[:, 1])
    centers_best = centers_best[valid_mask]
    radii_best = radii_best[valid_mask]

    # Ensure no circles are overlapping
    dists = np.sqrt(np.sum((centers_best[:, np.newaxis, :] - centers_best[np.newaxis, :, :])**2, axis=2))
    valid_mask = np.triu((dists >= radii_best[:, np.newaxis] + radii_best[np.newaxis, :]), k=1)
    centers_best = centers_best[valid_mask.any(axis=1)]
    radii_best = radii_best[valid_mask.any(axis=1)]

    sum_radii_best = np.sum(radii_best)

    return (centers_best, radii_best, sum_radii_best)
