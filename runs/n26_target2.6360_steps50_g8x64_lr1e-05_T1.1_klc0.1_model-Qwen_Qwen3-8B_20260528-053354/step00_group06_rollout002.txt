import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    # Start with equal radii for all circles
    for i in range(num_circles):
        col = i % 5
        row = i // 5
        centers[i, 0] = col / 4 + 0.25
        centers[i, 1] = row / 4 + 0.25
        radii[i] = 0.1

    # Objective function to maximize sum of radii
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraint: all circles are inside the unit square
    def inside_box(radii, centers):
        return np.all((radii <= centers[:, 0]) & (centers[:, 0] <= 1 - radii) &
                      (radii <= centers[:, 1]) & (centers[:, 1] <= 1 - radii))

    # Constraint: no overlapping between any two circles
    def no_overlap(radii, centers):
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization problem
    bounds = [(0, 0.5) for _ in range(num_circles)]
    cons = (
        {'type': 'ineq', 'fun': lambda r: inside_box(r, centers)},
        {'type': 'ineq', 'fun': lambda r: no_overlap(r, centers)}
    )

    # Optimization with scipy
    result = minimize(objective, radii, bounds=bounds, constraints=cons, method='SLSQP')

    # Extract results
    best_radii = result.x
    best_centers = centers.copy()
    # Update centers based on best radii
    best_centers = np.array([[(col / 4 + 0.25), (row / 4 + 0.25)] for row in range(5) for col in range(5)])
    # Final sum
    best_sum = np.sum(best_radii)

    return (best_centers, best_radii, best_sum)
