import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]
    radii = np.full(26, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x = i % 5 / 4.0
            y = i // 5 / 4.0
            centers[i] = [x, y]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        for i in range(26):
            if radii[i] > 1 - 2 * centers[i, 0] - 1e-8 or radii[i] > 1 - 2 * centers[i, 1] - 1e-8:
                return np.inf
        return 0

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, bounds=bounds, method='SLSQP', constraints={"type": "ineq", "fun": constraints})
    sum_radii = -result.fun
    return (result.x.reshape((-1, 1)), result.x, sum_radii)
