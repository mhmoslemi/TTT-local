import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: uniform
    radii = np.ones(n_circles) * 0.1

    # Objective function: minimize negative sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints:
    # 1. Each circle must be inside the unit square
    # 2. No two circles may overlap
    # 3. Radii must be non-negative

    # Constraint 1: circles in the unit square
    def constraint_inside(radii):
        r = radii.reshape(-1, 1)
        left = r <= centers[:, 0] - r
        bottom = r <= centers[:, 1] - r
        right = r <= 1 - centers[:, 0] - r
        top = r <= 1 - centers[:, 1] - r
        return np.all([left, bottom, right, top])

    # Constraint 2: no overlap between circles
    def constraint_no_overlap(radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.all(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    def constraint_box(radii):
        return radii >= 0

    # Set up bounds
    bounds = [(0, 1) for _ in range(n_circles)]

    # Wrap constraints into a single function
    def constraints(radii):
        result = []
        if not constraint_inside(radii):
            result.append({'type': 'ineq', 'fun': lambda r: constraint_inside(r)})
        if not constraint_no_overlap(radii):
            result.append({'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r)})
        if not np.all(radii >= 0):
            result.append({'type': 'ineq', 'fun': lambda r: constraint_box(r)})
        return result

    # Optimization
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            radii,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            tol=1e-6,
            options={'maxiter': 1000}
        )

    # Extract result
    best_radii = result.x
    sum_radii = np.sum(best_radii)
    return (centers, best_radii, sum_radii)
