import numpy as np
from scipy.optimize import minimize
from scipy.spatial import distance_matrix

def run_packing():
    # Initial guess: place circles spaced evenly in the square
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.1)

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:num_circles*2].reshape(num_circles, 2)
        radii_flat = x[num_circles*2:].reshape(num_circles)
        return -np.sum(radii_flat)  # Negative of sum for minimization

    # Constraints
    constraints = []

    # Constraint 1: Circle must lie within the unit square
    for i in range(num_circles):
        def circle_in_square(i):
            def cons(x):
                centers_flat = x[:num_circles*2].reshape(num_circles, 2)
                radii_flat = x[num_circles*2:].reshape(num_circles)
                r = radii_flat[i]
                x_center, y_center = centers_flat[i]
                return np.array([
                    x_center - r,
                    y_center - r,
                    1 - x_center - r,
                    1 - y_center - r
                ])
        constraints.append({
            'type': 'ineq',
            'fun': circle_in_square(i)
        })

    # Constraint 2: No two circles overlap
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            def no_overlap(i, j):
                def cons(x):
                    centers_flat = x[:num_circles*2].reshape(num_circles, 2)
                    radii_flat = x[num_circles*2:].reshape(num_circles)
                    r_i = radii_flat[i]
                    r_j = radii_flat[j]
                    x_i, y_i = centers_flat[i]
                    x_j, y_j = centers_flat[j]
                    dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
                    return dist - r_i - r_j
            constraints.append({
                'type': 'ineq',
                'fun': no_overlap(i, j)
            })

    # Bounds for x and y coordinates
    bounds = []
    for _ in range(num_circles):
        bounds.extend([(0, 1), (0, 1), (0, 1)])  # x, y, r for each circle

    # Optimization
    result = minimize(
        objective,
        x0=np.concatenate([centers.flatten(), radii.flatten()]),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6
    )

    # Extract the result
    centers_result = result.x[:num_circles*2].reshape(num_circles, 2)
    radii_result = result.x[num_circles*2:].reshape(num_circles)
    sum_radii = np.sum(radii_result)

    return (centers_result, radii_result, sum_radii)
