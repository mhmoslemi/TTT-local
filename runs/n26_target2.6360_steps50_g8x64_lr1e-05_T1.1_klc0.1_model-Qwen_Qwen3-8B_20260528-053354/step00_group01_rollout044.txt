    # Initial guess: place circles in a grid

        radii[i] = 0.05

    def constraint_circle_inside(radii, centers):
        # Constraint: circles are inside the unit square
        return np.array([r - (1 - x - r) for r, x in zip(radii, centers[:, 0])] +
                        [r - (1 - y - r) for r, y in zip(radii, centers[:, 1])])

    def constraint_no_overlap(radii, centers):
        # Constraint: no two circles overlap
        n = len(radii)
        dists = np.zeros(n * (n - 1) // 2)
        k = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[k] = np.sqrt(dx*dx + dy*dy) - radii[i] - radii[j]
                k += 1
        return dists

    # Define bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(num_circles)]

    # Set up optimization problem
    def objective_func(x):
        radii[:] = x
        return -np.sum(x)

    def constraint_func(x):
        # Check if circles are inside the square
        inside = np.array([r <= 1 - x - r for r, x in zip(x, centers[:, 0])] +
                          [r <= 1 - y - r for r, y in zip(x, centers[:, 1])])
        if not np.all(inside):
            return np.inf
        # Check if circles do not overlap
        n = len(x)
        dists = np.zeros(n * (n - 1) // 2)
        k = 0
        for i in range(n):
            for j in range(i + 1, n):

                dy = centers[i, 1] - centers[j, 1]

                k += 1
        if np.any(dists < 0):
            return np.inf
        return 0

    res = minimize(
        objective_func,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraint_func},

    # Extract final result
    final_radii = res.x

    sum_radii = np.sum(final_radii)
