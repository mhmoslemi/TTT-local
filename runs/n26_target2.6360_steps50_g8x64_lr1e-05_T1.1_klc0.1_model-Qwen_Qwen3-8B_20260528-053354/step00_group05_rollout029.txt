import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
import random

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    def initial_centers():
        centers = []
        for i in range(26):
            x = i // 5 + np.random.rand() * 0.2 - 0.1
            y = i % 5 + np.random.rand() * 0.2 - 0.1
            centers.append([x, y])
        return np.array(centers)

    def objective(radii):
        # Objective is to maximize sum of radii, so minimize -sum(radii)
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint 1: All circles are inside the unit square
        # r_i <= x_i - r_i and r_i <= 1 - x_i - r_i => x_i - r_i >= r_i => x_i >= 2r_i => r_i <= x_i / 2
        # Similarly for y_i
        constraints = []
        for i in range(26):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] * 2 - centers[i][0]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i] * 2 - centers[i][1]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i][0] - r[i] * 2})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - centers[i][1] - r[i] * 2})
        # Constraint 2: No two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                def dist_constraint(r):
                    return np.linalg.norm(centers[i] - centers[j]) - r[i] - r[j]
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        return constraints

    # Initial centers
    centers = initial_centers()
    # Initial guess for radii
    radii_initial = np.zeros(26) + 0.05

    # Optimization problem
    result = minimize(
        objective,
        radii_initial,
        method='SLSQP',
        constraints=constraints(radii_initial, centers),
        bounds=[(0, 0.5) for _ in range(26)]
    )

    # Return the result
    return (centers, result.x, -result.fun)
