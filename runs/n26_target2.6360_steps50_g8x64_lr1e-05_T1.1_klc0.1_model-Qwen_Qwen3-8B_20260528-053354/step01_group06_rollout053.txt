import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col / 4 + 0.25
        centers[i, 1] = row / 4 + 0.25
        radii[i] = 0.1  # Initial radius

    def objective_function(radii):
        return -np.sum(radii)

    def constraint_circle_inside(radii):
        return np.sum((radii > 0) * (np.min(centers[:, 0] - radii, axis=0) < 0) + 
                      (radii > 0) * (np.max(centers[:, 0] + radii, axis=0) > 1) + 
                      (radii > 0) * (np.min(centers[:, 1] - radii, axis=0) < 0) + 
                      (radii > 0) * (np.max(centers[:, 1] + radii, axis=0) > 1))

    def constraint_no_overlap(radii):
        dist = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        return np.sum((dist < radii[:, np.newaxis] + radii[np.newaxis, :]) * (radii[:, np.newaxis] + radii[np.newaxis, :] > 0))

    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    result = minimize(objective_function, radii, method='SLSQP', constraints=constraints, tol=1e-8)

    # Extract the result
    centers_final = centers.copy()
    radii_final = result.x
    sum_radii = -result.fun

    # Ensure all radii are non-negative
    radii_final = np.clip(radii_final, 0, None)

    return (centers_final, radii_final, sum_radii)
