```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: equally spaced grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i / np.sqrt(num_circles)
        y = (i % np.sqrt(num_circles)) / np.sqrt(num_circles)
        centers[i] = [x, y]
    
    # Objective function to minimize: negative of sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints: circles within bounds and no overlap
    def constraint_bounds(radii):
        return np.sum((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]))
    
    def constraint_overlap(radii):
        dist = cdist(centers, centers)
        return np.sum((dist >= radii[:, np.newaxis] + radii[np.newaxis, :]) & (dist > 0))
    
    # Optimization
    bounds = [(0, 1) for _ in range(num_circles)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_bounds},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    result = minimize(objective, x0=np.ones(num_circles), bounds=bounds, constraints=constraints, method='SLSQP')
    
    return (centers, result.x, -result.fun)
```
