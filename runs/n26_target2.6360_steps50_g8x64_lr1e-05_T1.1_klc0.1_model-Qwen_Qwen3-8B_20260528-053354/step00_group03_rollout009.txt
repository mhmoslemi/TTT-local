import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        centers[i, 0] = 0.2 + (i % 5) * 0.3
        centers[i, 1] = 0.2 + (i // 5) * 0.3
        radii[i] = 0.04  # Initial radius

    def objective(radii_params):
        return -np.sum(radii_params)  # Maximize sum of radii

    def constraints_func(x):
        # x is a flattened array of [x1, y1, r1, x2, y2, r2, ...]
        # Reconstruct centers and radii
        centers_flat = x[:2*num_circles].reshape(num_circles, 2)
        radii_flat = x[2*num_circles:].copy()

        # Constraints: circles are inside the square
        inside = np.all((radii_flat <= centers_flat[:, 0]) & (centers_flat[:, 0] <= 1 - radii_flat) &
                        (radii_flat <= centers_flat[:, 1]) & (centers_flat[:, 1] <= 1 - radii_flat))
        if not inside:
            return np.inf

        # Constraints: no circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return np.inf

        return -np.sum(radii_flat)  # We minimize -sum(radii) to maximize sum(radii)

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Optimization
    result = minimize(objective, x0, method='SLSQP', constraints={'type': 'ineq', 'fun': constraints_func})
    x_opt = result.x

    centers_opt = x_opt[:2*num_circles].reshape(num_circles, 2)
    radii_opt = x_opt[2*num_circles:]
    sum_radii = -result.fun  # Objective was -sum(radii)

    return (centers_opt, radii_opt, sum_radii)
