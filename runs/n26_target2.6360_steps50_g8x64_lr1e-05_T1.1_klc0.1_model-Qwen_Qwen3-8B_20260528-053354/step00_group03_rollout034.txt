import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: evenly spaced grid
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    radii = np.full(36, 0.05)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
        for i in range(len(radii)):
            if radii[i] < 0:
                return False
            for j in range(len(radii)):
                if i == j:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Define optimization problem
    bounds = [(0, 0.5) for _ in range(36)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints)

    # Extract solution
    centers = np.array([[x, y] for y in np.linspace(0.1, 0.9, 6) for x in np.linspace(0.1, 0.9, 6)])
    radii = result.x
    sum_radii = -result.fun

    # Filter valid circles (those with positive radius and satisfying constraints)
    valid_indices = np.where(radii > 0)[0]
    centers = centers[valid_indices]
    radii = radii[valid_indices]
    sum_radii = np.sum(radii)

    # Ensure no more than 26 circles
    if len(radii) > 26:
        radii = radii[:26]
        centers = centers[:26]
        sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
