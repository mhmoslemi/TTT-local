import numpy as np
from scipy.optimize import minimize
from scipy.optimize.optimize import OptimizeResult
import math

def run_packing():
    # Initial guess for center positions (randomly scattered in the unit square)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.05  # Small random radii initially

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_circle_inside(radii, centers):
        # Ensure each circle is fully inside the unit square
        return np.min(np.vstack([
            radii,
            1 - radii,
        ]), axis=0)

    def constraint_no_overlap(radii, centers):
        # Ensure no two circles overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :])**2, axis=2))
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dists - radii_sum

    # Optimization bounds: radii must be >= 0 and <= 0.5 (since 2r <= 1)
    bounds = [(0, 0.5) for _ in range(26)]

    # Constraints
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)},
    ]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Extract result
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
