import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess for centers and radii
    # We start with uniform spacing of centers and equal radii
    n_circles = 26
    centers = np.random.rand(n_circles, 2)
    radii = np.full(n_circles, 0.05)

    def objective_function(x):
        centers_flat = x[:n_circles*2].reshape(n_circles, 2)
        radii_flat = x[n_circles*2:]
        sum_radii = np.sum(radii_flat)
        # Constraint 1: All circles are inside the unit square
        constraints = []
        for i in range(n_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append((x_i - r_i) >= 0)
            constraints.append((1 - x_i - r_i) >= 0)
            constraints.append((y_i - r_i) >= 0)
            constraints.append((1 - y_i - r_i) >= 0)
        # Constraint 2: No two circles overlap
        for i in range(n_circles):
            for j in range(i+1, n_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                constraints.append(dist >= radii_flat[i] + radii_flat[j])
        return -sum_radii  # Minimize -sum_radii to maximize sum_radii

    # Define bounds for the optimization
    bounds = []
    for _ in range(n_circles*2):
        bounds.append((0.0, 1.0))  # x and y coordinates must be in [0,1]
    for _ in range(n_circles):
        bounds.append((0.0, 1.0 - 0.01))  # Radius is small enough to fit in square

    # Perform optimization
    result = minimize(objective_function, x0=np.concatenate([centers.ravel(), radii]), bounds=bounds)

    # Extract final results
    centers_opt = result.x[:n_circles*2].reshape(n_circles, 2)
    radii_opt = result.x[n_circles*2:]
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
