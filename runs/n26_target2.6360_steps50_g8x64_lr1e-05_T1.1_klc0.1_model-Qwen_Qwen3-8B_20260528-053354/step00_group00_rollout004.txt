from scipy.optimize import minimize
from scipy.spatial.distance import pdist

    centers = np.zeros((26, 2))

        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radius
    radii = np.ones(26) * 0.1

            if dists[i] < r1 + r2:
                return np.inf
        return 0

    def objective_function(x):
        radii[:] = x
        return -np.sum(x)

    # Bounds: All radii must be >= 0 and <= 1
    bounds = [(0, 1) for _ in range(26)]

    # Constraints: ensure circles are inside the square and non-overlapping
    # Note: Using a wrapper to handle constraints in the minimize function
    def constraint_func(x):
        # Constraint 1: Circles must be within the unit square
        for i in range(26):
            if x[i] > 1 - centers[i, 0] or x[i] > 1 - centers[i, 1]:

    result = minimize(objective_function, radii, bounds=bounds, constraints={"fun": constraint_func, "type": "ineq"})
