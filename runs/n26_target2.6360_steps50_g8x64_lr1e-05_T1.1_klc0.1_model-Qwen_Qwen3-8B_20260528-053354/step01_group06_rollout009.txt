import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: equally spaced grid
    centers = np.zeros((26, 2))
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Initial radii: uniform small radius
    radii = np.ones(26) * 0.1

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Negative for minimization

    # Constraints: circles must be inside the square
    def constraint_inside(r):
        return np.array([r <= center[0] for center in centers]) & \
               np.array([r <= 1 - center[0] for center in centers]) & \
               np.array([r <= center[1] for center in centers]) & \
               np.array([r <= 1 - center[1] for center in centers])

    # Constraint: no overlap between circles
    def constraint_no_overlap(r):
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j]:
                    return False
        return True

    # Optimization settings
    bounds = [(0, 1) for _ in range(26)]
    with warnings.catch_warnings():
        warnings.filterwarnings("ignore", message="Optimization failed")
        result = minimize(objective, radii, method='SLSQP', bounds=bounds,
                          constraints=[{'type': 'ineq', 'fun': lambda r: np.min([r[i] for i in range(26)])},  # >=0
                                       {'type': 'ineq', 'fun': lambda r: np.array([1 - centers[i, 0] - r[i] for i in range(26)])},  # x <= 1 - r
                                       {'type': 'ineq', 'fun': lambda r: np.array([centers[i, 0] - r[i] for i in range(26)])},  # x >= r
                                       {'type': 'ineq', 'fun': lambda r: np.array([1 - centers[i, 1] - r[i] for i in range(26)])},  # y <= 1 - r
                                       {'type': 'ineq', 'fun': lambda r: np.array([centers[i, 1] - r[i] for i in range(26)])},  # y >= r
                                       {'type': 'ineq', 'fun': lambda r: np.array([np.sqrt((centers[i, 0] - centers[j, 0])**2 + (centers[i, 1] - centers[j, 1])**2) - (r[i] + r[j]) for i in range(26) for j in range(i+1, 26)])}  # distance >= sum of radii
                                      ])

    radii = result.x
    sum_radii = -result.fun  # Convert back to maximization
    # Ensure all radii are non-negative
    radii = np.maximum(radii, 0)
    # Ensure circles are fully inside
    centers = np.array(centers)
    for i in range(26):
        if radii[i] > centers[i, 0]:
            radii[i] = centers[i, 0]
        if radii[i] > 1 - centers[i, 0]:
            radii[i] = 1 - centers[i, 0]
        if radii[i] > centers[i, 1]:
            radii[i] = centers[i, 1]
        if radii[i] > 1 - centers[i, 1]:
            radii[i] = 1 - centers[i, 1]
    # Ensure no overlap
    for i in range(26):
        for j in range(i + 1, 26):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j]:
                radii[i] = (dist - radii[j]) * 0.9
                if radii[i] < 0:
                    radii[i] = 0
                radii[j] = (dist - radii[i]) * 0.9
                if radii[j] < 0:
                    radii[j] = 0
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
