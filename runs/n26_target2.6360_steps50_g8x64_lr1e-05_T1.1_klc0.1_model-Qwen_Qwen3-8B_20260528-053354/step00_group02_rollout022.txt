import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [[0.25, 0.75], [0.75, 0.25], [0.75, 0.75], [0.5, 0.5]]])
    radii = np.full(26, 0.1)
    
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    def constraint_center_in_square(radii, centers):
        return np.sum((radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) & \
                      (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1]))
    
    def constraint_no_overlap(radii, centers):
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Optimization
    def objective_wrapper(radii_vec):
        radii = radii_vec.reshape(26)
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            result = minimize(objective, radii_vec, method='L-BFGS-B',
                              bounds=[(0, 1)]*26,
                              constraints=[
                                  {'type': 'ineq', 'fun': lambda r: constraint_center_in_square(r, centers)},
                                  {'type': 'ineq', 'fun': lambda r: constraint_no_overlap(r, centers)}
                              ])
        return result
    
    result = objective_wrapper(radii)
    radii = result.x
    centers = np.array([[i/5 + 0.25, j/5 + 0.25] for i in range(5) for j in range(5)])
    centers = np.vstack([centers, [[0.25, 0.75], [0.75, 0.25], [0.75, 0.75], [0.5, 0.5]]])
    
    # Ensure all circles are within the square
    valid_mask = (radii <= centers[:, 0]) & (radii <= 1 - centers[:, 0]) & \
                 (radii <= centers[:, 1]) & (radii <= 1 - centers[:, 1])
    centers = centers[valid_mask]
    radii = radii[valid_mask]
    
    # Ensure no overlaps
    dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                     (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
    valid_mask = (dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
    valid_mask = np.all(valid_mask, axis=1)
    centers = centers[valid_mask]
    radii = radii[valid_mask]
    
    # Ensure there are exactly 26 circles
    if len(radii) < 26:
        added = np.full(26 - len(radii), 0.0)
        radii = np.concatenate([radii, added])
        centers = np.vstack([centers, np.zeros((26 - len(centers), 2))])
    
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
