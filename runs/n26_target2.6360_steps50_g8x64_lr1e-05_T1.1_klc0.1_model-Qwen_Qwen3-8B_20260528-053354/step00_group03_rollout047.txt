import numpy as np
from scipy.optimize import minimize
import itertools

def run_packing():
    # Initial guess for centers (random points in the square)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.random.rand(26) * 0.1  # Initial small radii

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraints: circles must be within the square
        r = radii.copy()
        for i in range(26):
            if r[i] > 1 - 2 * r[i]:
                return -np.inf
        # Constraints: distances between centers must be >= sum of radii
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r[i] + r[j]:
                    return -np.inf
        return 0

    # Define optimization bounds
    bounds = [(0.0, 1.0) for _ in range(26)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},
        options={'disp': False, 'maxiter': 1000}
    )

    # Extract result
    radii_opt = result.x
    sum_radii = -result.fun

    # Ensure circles are within the square
    for i in range(26):
        if radii_opt[i] > 1 - 2 * radii_opt[i]:
            radii_opt[i] = 1 - 2 * radii_opt[i]

    # Final check for overlaps
    for i in range(26):
        for j in range(i + 1, 26):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii_opt[i] + radii_opt[j]:
                radii_opt[i] = (dist - radii_opt[j]) / 2
                radii_opt[j] = (dist - radii_opt[i]) / 2

    return (centers, radii_opt, sum_radii)
