import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced centers with a small radius
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0

    # Initial radius guess
    radii = np.ones(num_circles) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are inside the square
        for i in range(num_circles):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False

        # Ensure no two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False

        return True

    # Optimize the radii
    bounds = [(0.0, 1.0) for _ in range(num_circles)]
    result = minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "custom", "fun": constraints},
        method="SLSQP",
        options={"ftol": 1e-8, "maxiter": 1000}
    )

    radii_opt = result.x
    centers_opt = centers.copy()
    sum_radii_opt = -result.fun

    # Final validation check
    if not constraints(radii_opt):
        raise ValueError("Optimization failed to satisfy constraints.")

    return (centers_opt, radii_opt, sum_radii_opt)
