import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint

def run_packing():
    # Number of circles
    n = 26
    # Initial guess: equally spaced grid
    x = np.linspace(0.1, 0.9, 7)
    y = np.linspace(0.1, 0.9, 7)
    centers = np.array([x[i] + (x[i+1] - x[i]) * j for i in range(7) for j in range(7)]).reshape(-1, 2)
    radii = np.full(n, 0.1)

    # Objective function: minimize negative sum of radii (maximize sum of radii)
    def objective(x):
        return -np.sum(x)

    # Constraints:
    # 1. Each circle is inside the square
    # 2. No two circles overlap
    # 3. All radii are non-negative

    # Convert centers and radii to a single vector for optimization
    x0 = np.concatenate((centers.reshape(-1), radii))

    # Bounds: x and y coordinates of centers must be between 0.1 and 0.9
    # Radius must be non-negative
    bounds = Bounds(np.concatenate((np.full(n*2, 0.1), np.zeros(n))), np.concatenate((np.full(n*2, 0.9), np.full(n, np.inf))))

    # Linear constraints: distance between centers >= sum of radii
    def constraint_ineq(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        dist = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :])**2).sum(axis=2))
        return dist - radii_flat[:, np.newaxis] - radii_flat[np.newaxis, :]

    # Nonlinear constraint: distance >= sum of radii
    cons = ({'type': 'ineq', 'fun': lambda x: constraint_ineq(x).flatten()})

    # Optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6, options={'maxiter': 1000})

    # Extract results
    centers_opt = result.x[:n*2].reshape(n, 2)
    radii_opt = result.x[n*2:]
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
