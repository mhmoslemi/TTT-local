```python
import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: 26 circles arranged in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = np.array([x, y])
    
    # Initial radii (assume all same for initialization)
    radii = np.full(num_circles, 0.1)
    
    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    # Constraints: circles inside the square and no overlap
    def constraint_inside(radii, centers):
        return np.sum((radii <= 1 - centers[:, 0]) * (radii <= centers[:, 0]) * (radii <= 1 - centers[:, 1]) * (radii <= centers[:, 1]))
    
    def constraint_overlap(radii, centers):
        dist = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[i, j] = np.sqrt(dx**2 + dy**2)
        max_dist = np.max(dist)
        return max_dist - (radii[:, np.newaxis] + radii[np.newaxis, :])
    
    # Combine constraints
    def constraint(radii):
        center_constraints = constraint_inside(radii, centers)
        overlap_constraints = constraint_overlap(radii, centers)
        return np.concatenate([center_constraints, overlap_constraints])
    
    # Optimization
    bounds = [(0, 1) for _ in range(num_circles)]
    result = minimize(objective, radii, bounds=bounds, constraints={"type": "ineq", "fun": constraint})
    
    # Extract solution
    sum_radii = result.fun * -1
    return (centers, result.x, sum_radii)
```
