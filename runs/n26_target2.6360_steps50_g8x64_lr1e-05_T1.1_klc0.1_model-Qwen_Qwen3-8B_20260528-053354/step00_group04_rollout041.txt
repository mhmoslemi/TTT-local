import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Initial guess: evenly spaced grid of 26 circles with radius 0.05
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.3
        centers[i, 1] = row * 0.3
        radii[i] = 0.05

    # Objective function: minimize negative sum of radii
    def objective(x):
        return -np.sum(x)

    # Constraints: all circles must be within the unit square
    def bounds():
        # Each circle has x, y, r
        return [(0.0, 1.0 - r) for _ in range(num_circles * 3)]

    # Constraints: no overlap between any two circles
    def non_overlap_cst(x):
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        radii = x[num_circles * 2:]
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=-1))
        return dist - (radii[:, np.newaxis] + radii[np.newaxis, :])

    constraints = (
        {'type': 'ineq', 'fun': non_overlap_cst},
    )

    # Initial guess: flatten centers and radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: each coordinate must be within [r, 1 - r], and radii must be ≥ 0
    bounds = []
    for i in range(num_circles):
        x_low = 0.0
        x_high = 1.0 - x0[num_circles * 2 + i]
        bounds.append((x_low, x_high))
        bounds.append((0.0, 1.0 - x0[num_circles * 2 + i]))
    for i in range(num_circles):
        bounds.append((0.0, 1.0 - x0[num_circles * 2 + i]))
    bounds += [(0.0, 1.0 - x0[num_circles * 2 + i]) for i in range(num_circles)]

    # Optimize
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-6)

    # Extract results
    centers = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii = result.x[num_circles * 2:]
    sum_radii = -result.fun

    # Ensure all circles are within the unit square and radii are non-negative
    for i in range(num_circles):
        if not (radii[i] <= centers[i, 0] <= 1 - radii[i] and radii[i] <= centers[i, 1] <= 1 - radii[i]):
            radii[i] = 0
            sum_radii -= radii[i]

    return (centers, radii, sum_radii)
