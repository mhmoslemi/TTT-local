import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid-like pattern within the unit square
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, 0.15)

    def objective_function(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraint_circle_inside(radii, centers):
        # Constraint: circle i is inside the unit square
        # Ensure x - r_i >= 0, x + r_i <= 1, same for y
        violation = np.zeros_like(radii)
        for i in range(num_circles):
            violation[i] = max(centers[i, 0] - radii[i], centers[i, 1] - radii[i],
                               radii[i] - (1 - centers[i, 0]), radii[i] - (1 - centers[i, 1]))
        return violation

    def constraint_circle_overlap(radii, centers):
        # Constraint: distance between any two centers >= sum of their radii
        violation = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distance = np.sqrt(dx*dx + dy*dy)
                violation[i, j] = distance - (radii[i] + radii[j])
                violation[j, i] = violation[i, j]
        return np.min(violation)

    # Minimization setup
    bounds = [(0, 1) for _ in range(num_circles)]  # All radii non-negative
    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_circle_overlap(r, centers)}
    ]

    result = minimize(
        objective_function,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    # Extract results
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
