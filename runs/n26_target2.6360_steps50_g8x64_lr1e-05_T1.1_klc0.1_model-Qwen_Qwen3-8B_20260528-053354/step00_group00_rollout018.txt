import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: centers on a grid, radii set to max possible
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = 0.2 + col * 0.25
        centers[i, 1] = 0.2 + row * 0.25
        radii[i] = 0.1

    # Objective function: minimize negative sum of radii
    def objective(x):
        return -np.sum(x[26:])  # x[26:] are the radii

    # Bounds: centers in [0, 1], radii >= 0, and radii <= (1 - center) for all directions
    bounds = []
    for i in range(26):
        # Center bounds
        bounds.append((0.0, 1.0))
        bounds.append((0.0, 1.0))
        # Radius bounds
        bounds.append((0.0, 1.0 - max(centers[i, 0], 1 - centers[i, 0], centers[i, 1], 1 - centers[i, 1])))

    # Constraints: no overlapping between circles
    def constraint(x):
        # x[0:26*2] are centers, x[26*2:] are radii
        centers = x[:52].reshape((26, 2))
        radii = x[52:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    return dist - radii[i] - radii[j]
        return 0.0

    cons = {'type': 'ineq', 'fun': constraint}

    # Optimization
    result = minimize(objective, x0=np.concatenate([centers.flatten(), radii]), bounds=bounds, constraints=cons)

    # Extract final result
    centers_final = result.x[:52].reshape((26, 2))
    radii_final = result.x[52:]
    sum_radii_final = np.sum(radii_final)

    return (centers_final, radii_final, sum_radii_final)
