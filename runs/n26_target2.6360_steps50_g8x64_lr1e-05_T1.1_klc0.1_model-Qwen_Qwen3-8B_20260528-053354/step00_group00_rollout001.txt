import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Define the unit square
    square = np.array([[0, 1], [0, 1]])

    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        return -np.sum(radii_flat)  # Negative for minimization

    # Constraints:
    # 1. All circles must be inside the square
    def constraint_inside(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        return np.array([np.all(r <= np.min([c[0], 1 - c[0], c[1], 1 - c[1]]) for c, r in zip(centers_flat, radii_flat))])

    # 2. No two circles overlap
    def constraint_overlap(x):
        centers_flat = x[:2*26].reshape(26, 2)
        radii_flat = x[2*26:]
        dists = np.sqrt(np.sum((centers_flat[:, np.newaxis] - centers_flat[np.newaxis, :])**2, axis=2))
        return np.array([np.all(dists >= radii_flat[:, np.newaxis] + radii_flat[np.newaxis, :])])

    # Bounds: radii must be >= 0 and <= 0.5 (since max radius in unit square is 0.5)
    bounds = [(0, 0.5) for _ in range(26)] + [(0, 1) for _ in range(2*26)]

    # Optimization setup
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective,
            np.concatenate([centers.ravel(), radii]),
            method='SLSQP',
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': constraint_inside},
                {'type': 'ineq', 'fun': constraint_overlap}
            ],
            tol=1e-6,
            options={'maxiter': 500}
        )

    # Extract final result
    centers = result.x[:2*26].reshape(26, 2)
    radii = result.x[2*26:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
