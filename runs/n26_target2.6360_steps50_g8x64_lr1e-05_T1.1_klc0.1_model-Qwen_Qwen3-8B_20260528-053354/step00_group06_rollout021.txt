import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: place circles regularly spaced with small radii
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, 0.05)

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative to maximize sum

    def constraint_circle_inside(radii, centers):
        # Ensure each circle is inside the unit square
        return np.array([r - (1 - r) for r in radii])

    def constraint_no_overlap(radii, centers):
        # Ensure circles do not overlap
        dists = np.sqrt(np.sum((centers[:, np.newaxis] - centers[np.newaxis, :])**2, axis=2))
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Bounds for radii
    bounds = Bounds(0, 1)

    # Initial guess
    initial_guess = np.concatenate((centers.flatten(), radii))

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_circle_inside, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_no_overlap, 'args': (radii, centers)}
    ]

    # Optimization
    result = minimize(
        fun=lambda x: objective(x[num_circles:], x[:2*num_circles].reshape((num_circles, 2))),
        x0=initial_guess,
        bounds=bounds,
        constraints=cons,
        method='SLSQP'
    )

    # Extract results
    final_centers = result.x[:2*num_circles].reshape((num_circles, 2))
    final_radii = result.x[num_circles:]
    sum_radii = -objective(final_radii, final_centers)

    return (final_centers, final_radii, sum_radii)
