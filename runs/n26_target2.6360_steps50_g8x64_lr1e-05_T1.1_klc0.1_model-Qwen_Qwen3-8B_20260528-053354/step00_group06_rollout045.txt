import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: randomly place 26 circles with small radius
    np.random.seed(42)
    centers = np.random.uniform(0.1, 0.9, (26, 2))
    radii = np.full(26, 0.01)
    sum_radii = np.sum(radii)
    
    # Objective function: maximize sum of radii (minimize negative)
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints:
    # 1. Each circle is within the unit square
    # 2. No overlap between circles
    def constraints(radii):
        # Constraint 1: Each circle is within the unit square
        for i in range(26):
            if radii[i] < 0:
                return False
            for j in range(2):
                if centers[i, j] - radii[i] < 0 or centers[i, j] + radii[i] > 1:
                    return False
        
        # Constraint 2: No overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        return True
    
    # Optimization using scipy.optimize.minimize
    result = minimize(
        fun=lambda r: -np.sum(r),
        x0=radii,
        method='L-BFGS-B',
        bounds=[(0, 1.0) for _ in range(26)],
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r)}]
    )
    
    # Extract result
    radii_opt = result.x
    sum_radii_opt = np.sum(radii_opt)
    centers_opt = centers.copy()
    
    return (centers_opt, radii_opt, sum_radii_opt)
