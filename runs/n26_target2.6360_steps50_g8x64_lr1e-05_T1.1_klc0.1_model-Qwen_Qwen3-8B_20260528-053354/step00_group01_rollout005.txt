import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial guess: evenly spaced grid with small radii
    centers = np.array([[i/5 + 0.1, j/5 + 0.1] for i in range(5) for j in range(5)])
    radii = np.zeros(25)
    centers = np.vstack((centers, [0.5, 0.5]))  # Add one more center in the middle
    radii = np.append(radii, 0.02)

    def objective(radii):
        return -np.sum(radii)  # Maximize sum by minimizing negative sum

    def constraint_no_overlap(radii):
        dist = cdist(centers, centers)
        radii_sum = radii[:, np.newaxis] + radii[np.newaxis, :]
        return dist - radii_sum

    def constraint_inside(radii):
        x = centers[:, 0]
        y = centers[:, 1]
        r = radii
        return np.minimum(x - r, 1 - x - r, y - r, 1 - y - r)

    cons = [
        {'type': 'ineq', 'fun': constraint_no_overlap},
        {'type': 'ineq', 'fun': constraint_inside}
    ]

    # Optimize
    result = minimize(objective, radii, constraints=cons, method='SLSQP', bounds=[(0, 1.0) for _ in range(26)])
    centers = np.array(result.x).reshape(-1, 2)
    radii = result.x
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
