import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x, y = np.random.uniform(0.1, 0.9, 2)  # initial random centers
            centers[i] = [x, y]
        radii = np.maximum(radii, 0)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return np.inf  # invalid packing
        for i in range(26):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return np.inf  # circle out of bounds
        return -np.sum(radii)  # maximize the sum

    # Initial guess for radii
    radii_initial = np.random.uniform(0.01, 0.1, 26)

    # Minimize negative sum to maximize sum
    result = minimize(
        fun=objective,
        x0=radii_initial,
        method='L-BFGS-B',
        bounds=[(0, 0.5) for _ in range(26)],
        options={'ftol': 1e-8}
    )

    radii = result.x
    centers = np.zeros((26, 2))
    for i in range(26):
        x, y = np.random.uniform(0.1, 0.9, 2)
        centers[i] = [x, y]
    # Final check after optimization
    for i in range(26):
        x, y = centers[i]
        if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
            radii[i] = 0
    for i in range(26):
        for j in range(i + 1, 26):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j]:
                radii[i] = 0
                radii[j] = 0

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
