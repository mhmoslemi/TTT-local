import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess for circle centers (randomly distributed)
    np.random.seed(42)
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure circles are within the unit square
        r = radii.reshape(-1, 1)
        bounds = np.concatenate([np.ones((26, 1)) * r, np.ones((26, 1)) * (1 - r)], axis=1)
        return bounds

    def non_overlap_constraint(radii):
        r = radii.reshape(-1, 1)
        centers = np.random.rand(26, 2)
        dist = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        for i in range(26):
            for j in range(i + 1, 26):
                if dist[i, j] < r[i] + r[j]:
                    return np.inf
        return 0

    # Optimization setup
    bounds = [(0.0, 0.5) for _ in range(26)]

    result = minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: np.min(constraints(x))},
            {'type': 'ineq', 'fun': non_overlap_constraint}
        ]
    )

    centers = np.random.rand(26, 2)
    radii = result.x
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
