import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: arrange circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.1  # Start with small radii

    # Initial grid layout
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        x = col * 0.4 + 0.1
        y = row * 0.4 + 0.1
        centers[i] = [x, y]

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    def constraint_circle_inside(radii, centers):
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if not (r <= x <= 1 - r and r <= y <= 1 - r):
                return 1.0
        return 0.0

    def constraint_circle_overlap(radii, centers):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return 1.0
        return 0.0

    constraints = [
        {'type': 'ineq', 'fun': lambda r: constraint_circle_inside(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_circle_overlap(r, centers)}
    ]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0.0, 1.0 - 0.1) for _ in range(num_circles)],
        options={'ftol': 1e-9, 'maxiter': 1000}
    )

    centers = np.array(centers)
    radii = result.x
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
