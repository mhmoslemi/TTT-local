import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles at equally spaced grid points
    num_circles = 26
    grid_size = 5
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        centers[i, 0] = (i % grid_size + 0.5) / grid_size
        centers[i, 1] = (i // grid_size + 0.5) / grid_size
        radii[i] = 0.05

    def objective(radii):
        return -np.sum(radii)

    def constraint1(radii):
        # Ensure circles are within the unit square
        r = radii
        return np.min([r, 1 - r - centers[:, 0], 1 - r - centers[:, 1]])

    def constraint2(radii):
        # Ensure no two circles overlap
        r = radii
        dists = np.zeros((num_circles, num_circles))
        for i in range(num_circles):
            for j in range(num_circles):
                if i != j:
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dists[i, j] = np.sqrt(dx**2 + dy**2) - (r[i] + r[j])
        return np.min(dists)

    # Define bounds for radii
    bounds = [(0.0, 1.0 - 0.5) for _ in range(num_circles)]

    # Define constraints
    cons = [
        {'type': 'ineq', 'fun': lambda r: constraint1(r)},
        {'type': 'ineq', 'fun': lambda r: constraint2(r)}
    ]

    # Optimize
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=cons)

    centers = np.array(centers)
    radii = result.x
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
