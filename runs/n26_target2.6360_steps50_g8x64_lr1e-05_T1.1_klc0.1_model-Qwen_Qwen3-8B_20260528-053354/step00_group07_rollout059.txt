import numpy as np
from scipy.optimize import minimize
import copy

def run_packing():
    # Initial guess: 26 circles arranged in a grid, equally spaced
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]
    radii = np.full(26, 0.1)

    def objective(radii):
        return -np.sum(radii)  # We minimize the negative sum to maximize it

    def constraints(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x = i % 5 / 4.0
            y = i // 5 / 4.0
            centers[i] = [x, y]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(26):
            if radii[i] < 0:
                return False
            if radii[i] > 1 - centers[i][0] or radii[i] > 1 - centers[i][1]:
                return False
        return True

    bounds = [(0, 1) for _ in range(26)]

    # Use a local optimization method
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=lambda x: constraints(x))

    # Final centers (same as initial guess but with radii updated)
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.0
        y = i // 5 / 4.0
        centers[i] = [x, y]

    return (centers, result.x, -result.fun)
