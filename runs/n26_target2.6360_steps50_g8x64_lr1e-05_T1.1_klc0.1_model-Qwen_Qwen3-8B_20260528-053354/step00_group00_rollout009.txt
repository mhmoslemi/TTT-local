import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: spread circles evenly in the square
    centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 + 0.1
        y = i // 5 + 0.1
        centers[i] = [x, y]
        
    radii = np.full(26, 0.1)
    
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        # Ensure circles are inside the square
        inside = np.all((radii <= 1 - centers[:, 0]) & (radii <= 1 - centers[:, 1]), axis=0)
        if not np.all(inside):
            return np.inf
        
        # Ensure no overlapping
        dists = np.sqrt(((centers[:, np.newaxis, 0] - centers[np.newaxis, :, 0])**2 +
                         (centers[:, np.newaxis, 1] - centers[np.newaxis, :, 1])**2))
        overlaps = np.sum(dists < radii[np.newaxis, :] + radii[:, np.newaxis], axis=1)
        if np.any(overlaps > 0):
            return np.inf
        
        return 0
    
    # Optimization
    bounds = [(0, 0.5) for _ in range(26)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})
    
    # Final result
    sum_radii = -result.fun
    return (centers, result.x, sum_radii)
