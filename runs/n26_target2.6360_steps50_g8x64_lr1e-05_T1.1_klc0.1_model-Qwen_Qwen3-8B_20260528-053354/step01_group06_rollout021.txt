import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Initial guess: evenly spaced grid of 26 points
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        centers[i, 0] = (i + 1) / (n_circles + 1)
        centers[i, 1] = 0.5

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: each circle must be fully inside the square
    def square_bounds(radii):
        return (radii <= (1 - centers[:, 0]) * np.ones(n_circles)) & (radii <= (1 - centers[:, 1]) * np.ones(n_circles))

    # Constraints: no two circles overlap
    def no_overlap(radii):
        dists = np.sqrt(np.sum((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2, axis=2))
        return dists >= radii[:, np.newaxis] + radii[np.newaxis, :]

    # Pack the circles
    bounds = Bounds([0.0] * n_circles, [1.0] * n_circles)
    res = minimize(objective, np.ones(n_circles) * 0.1, bounds=bounds, constraints=[{'type': 'ineq', 'fun': square_bounds},
                                                                                   {'type': 'ineq', 'fun': no_overlap}])

    radii = res.x
    centers = np.vstack([centers[i] for i in range(n_circles) if radii[i] > 0][:n_circles])
    radii = radii[radii > 0]
    sum_radii = np.sum(radii)

    # Ensure we have exactly 26 circles (pad with zeros if necessary)
    centers = np.vstack([centers, np.zeros((n_circles - len(centers), 2))])
    radii = np.pad(radii, (0, n_circles - len(radii)), 'constant', constant_values=0)

    return (centers, radii, sum_radii)
