import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        centers[i, 0] = (i % 5 + 0.5) / 5.0
        centers[i, 1] = (i // 5 + 0.5) / 5.0
        radii[i] = 0.05

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii, centers):
        return -np.sum(radii)

    # Constraints: circle is inside the unit square
    def inside_box(radii, centers):
        return np.array([r <= 1 - r and r <= 1 - r for r in radii])

    # Constraints: no two circles overlap
    def no_overlap(radii, centers):
        dists = np.linalg.norm(centers[:, np.newaxis] - centers[np.newaxis, :], axis=2)
        return np.array([dists[i, j] >= radii[i] + radii[j] for i in range(26) for j in range(i+1, 26)]).flatten()

    # Combined constraints
    def constraints(radii, centers):
        in_box = inside_box(radii, centers)
        no_overlap = no_overlap(radii, centers)
        return np.concatenate([in_box, no_overlap])

    # Optimization
    def optimize():
        bounds = [(0, 0.5) for _ in range(26)]
        result = minimize(
            fun=lambda r: -np.sum(r),
            x0=radii,
            bounds=bounds,
            constraints={
                'type': 'ineq',
                'fun': lambda r: constraints(r, centers)
            }
        )
        return result.x

    # Run optimization
    radii_opt = optimize()
    sum_radii = np.sum(radii_opt)

    # Update centers based on optimized radii
    # Use a simple algorithm to update centers
    # This is a placeholder for a more sophisticated optimization

    return (centers, radii_opt, sum_radii)
