import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: 26 circles arranged in 2 rows of 13 with equal spacing
    num_circles = 26
    initial_centers = np.zeros((num_circles, 2))
    initial_radii = np.zeros(num_circles)
    spacing = 0.15
    radii_init = 0.05
    for i in range(num_circles):
        row = i // 13
        col = i % 13
        x = col * spacing
        y = row * spacing
        initial_centers[i, 0] = x
        initial_centers[i, 1] = y
        initial_radii[i] = radii_init

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.zeros((num_circles, 2))
        for i in range(num_circles):
            x = initial_centers[i, 0] + np.random.normal(0, 0.01)
            y = initial_centers[i, 1] + np.random.normal(0, 0.01)
            centers[i, 0] = x
            centers[i, 1] = y
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j]:
                    return False
        for i in range(num_circles):
            x, y = centers[i]
            if not (radii[i] <= x <= 1 - radii[i] and radii[i] <= y <= 1 - radii[i]):
                return False
        return True

    # Define bounds for radii
    bounds = [(0, 0.5) for _ in range(num_circles)]

    # Minimize negative sum of radii to maximize it
    result = minimize(objective, initial_radii, method='L-BFGS-B', bounds=bounds,
                      constraints={'type': 'ineq', 'fun': constraints})

    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        x = initial_centers[i, 0] + np.random.normal(0, 0.05)
        y = initial_centers[i, 1] + np.random.normal(0, 0.05)
        centers[i, 0] = x
        centers[i, 1] = y

    return (centers, result.x, -result.fun)
