import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    centers = np.random.rand(26, 2)
    radii = np.full(26, 0.1)

    def objective(radii_and_centers):
        radii = radii_and_centers[:26]
        centers = radii_and_centers[26:]
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_center_inside(radii_and_centers):
        radii = radii_and_centers[:26]
        centers = radii_and_centers[26:]
        return np.array([np.all((centers[i, 0] - radii[i] >= 0) & (centers[i, 1] - radii[i] >= 0) &
                               (1 - centers[i, 0] - radii[i] >= 0) & (1 - centers[i, 1] - radii[i] >= 0))
                         for i in range(26)])

    def constraint_no_overlap(radii_and_centers):
        radii = radii_and_centers[:26]
        centers = radii_and_centers[26:]
        dists = np.zeros((26, 26))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx**2 + dy**2)
                dists[j, i] = dists[i, j]
        return np.array([dists[i, j] - radii[i] - radii[j] for i in range(26) for j in range(26) if i < j])

    # Initial guess
    initial_guess = np.concatenate([radii, centers.ravel()])

    # Optimization settings
    bounds = [(0, 0.5) for _ in range(26)] + [(0, 1) for _ in range(52)]

    # Setup constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_center_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Suppress warnings
    warnings.filterwarnings("ignore", message="Objective function returned a value of nan")

    # Run optimization
    result = minimize(objective, initial_guess, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)

    # Extract results
    radii = result.x[:26]
    centers = result.x[26:].reshape(26, 2)
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
