import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: place circles in a grid
    centers = np.zeros((26, 2))
    radii = np.ones(26) * 0.1  # Initial small radii
    for i in range(26):
        row = i // 5
        col = i % 5
        centers[i, 0] = col * 0.4 + 0.2
        centers[i, 1] = row * 0.4 + 0.2

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        # Constraint 1: Ensure circles are inside the unit square
        for i in range(26):
            r = radii[i]
            if centers[i, 0] - r < 0 or centers[i, 0] + r > 1:
                return False
            if centers[i, 1] - r < 0 or centers[i, 1] + r > 1:
                return False
        # Constraint 2: Ensure no overlap between circles
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    bounds = [(0.0, 0.3) for _ in range(26)]  # Upper bound on radii

    # Objective function to maximize sum of radii
    def objective_function(x):
        return -np.sum(x)

    # Constraint function for scipy.optimize.minimize
    def constraint_func(x):
        # Check if all circles are within the square
        for i in range(26):
            r = x[i]
            if not (centers[i, 0] - r >= 0 and centers[i, 0] + r <= 1 and
                    centers[i, 1] - r >= 0 and centers[i, 1] + r <= 1):
                return -np.inf
        # Check for overlap
        for i in range(26):

                if dist < x[i] + x[j]:
                    return -np.inf
        return np.sum(x)  # Penalize by returning the sum (to be maximized)

    # Optimization
    result = minimize(objective_function, radii, bounds=bounds, method='SLSQP',

    # Check if the optimization was successful
    if not result.success:
        warnings.warn("Optimization failed.")
        return (centers, radii, np.sum(radii))

    radii_opt = result.x
    sum_radii = np.sum(radii_opt)
