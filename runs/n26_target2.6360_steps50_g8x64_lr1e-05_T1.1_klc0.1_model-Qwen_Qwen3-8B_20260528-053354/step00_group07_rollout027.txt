import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles) * 0.1

    def objective(x):
        # x is a flattened array of centers and radii
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:].reshape(num_circles)
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize -sum_radii to maximize sum_radii

    def constraints(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:].reshape(num_circles)
        # Constraint 1: All circles are within the unit square
        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return np.inf
        # Constraint 2: No two circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                r_i = radii_flat[i]
                r_j = radii_flat[j]
                if dist < r_i + r_j:
                    return np.inf
        return 0

    # Prepare initial guess
    x0 = np.concatenate([centers.flatten(), radii.flatten()])

    # Define bounds to ensure radii are non-negative
    bounds = [(0, 0.5) for _ in range(num_circles * 2)]  # Max radius is 0.5 to fit in unit square

    # Set up optimization
    result = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints}, tol=1e-8)

    # Extract results
    centers_result = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii_result = result.x[num_circles * 2:].reshape(num_circles)
    sum_radii_result = -result.fun

    return (centers_result, radii_result, sum_radii_result)
