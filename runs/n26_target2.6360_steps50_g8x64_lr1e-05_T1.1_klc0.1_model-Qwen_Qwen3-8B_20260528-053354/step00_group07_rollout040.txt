import numpy as np
from scipy.optimize import minimize
import sys

def run_packing():
    # Initial guess: evenly spaced circles in a grid
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        centers[i, 0] = (i % 5 + 0.5) / 5
        centers[i, 1] = (i // 5 + 0.5) / 5

    # Optimization bounds for centers and radii
    bounds = [(0.0, 1.0) for _ in range(2 * num_circles)] + [(0.0, 1.0) for _ in range(num_circles)]

    def objective(radii):
        return -np.sum(radii)  # We want to maximize the sum, so minimize the negative

    def constraint_center_inside(radii, centers):
        return np.sum((radii <= centers[:, 0]) & (radii <= 1.0 - centers[:, 0]) &
                      (radii <= centers[:, 1]) & (radii <= 1.0 - centers[:, 1]))

    def constraint_circle_overlap(radii, centers):
        n = len(radii)
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        return np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization
    try:
        result = minimize(objective, x0=np.zeros(3 * num_circles), bounds=bounds,
                          constraints=[
                              {'type': 'ineq', 'fun': constraint_center_inside, 'args': (centers,)},
                              {'type': 'ineq', 'fun': constraint_circle_overlap, 'args': (centers,)}
                          ])
    except:
        # Fallback if optimization fails
        result = minimize(objective, x0=np.zeros(3 * num_circles), bounds=bounds)

    # Extract results
    optimized = result.x
    centers_opt = optimized[:2 * num_circles].reshape((num_circles, 2))
    radii_opt = optimized[2 * num_circles:].clip(min=0)
    sum_radii = np.sum(radii_opt)

    return (centers_opt, radii_opt, sum_radii)
