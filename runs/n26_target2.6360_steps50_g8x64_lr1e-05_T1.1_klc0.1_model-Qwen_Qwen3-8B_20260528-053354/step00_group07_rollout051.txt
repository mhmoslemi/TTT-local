import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place circles in a grid pattern
    centers = np.zeros((26, 2))
    radii = np.zeros(26)
    for i in range(26):
        x = i % 5 / 4.0 + 0.1
        y = i // 5 / 4.0 + 0.1
        centers[i] = [x, y]
        radii[i] = 0.1

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize

    # Constraints
    def constraint_circle_inside(radii):
        return np.all(radii <= 1 - 0.1)  # initial guess radius is 0.1, adjust if needed

    def constraint_no_overlap(radii):
        centers = np.zeros((26, 2))
        for i in range(26):
            x = i % 5 / 4.0 + 0.1
            y = i // 5 / 4.0 + 0.1
            centers[i] = [x, y]
        distances = np.zeros((26, 26))
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                distances[i, j] = np.sqrt(dx**2 + dy**2)
                distances[j, i] = distances[i, j]
        return np.all(distances >= radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization
    bounds = [(0.0, 1.0) for _ in range(26)]
    constraints = [
        {'type': 'ineq', 'fun': constraint_circle_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=constraints, tol=1e-5)

    # Final centers and radii
    final_centers = np.zeros((26, 2))
    for i in range(26):
        x = i % 5 / 4.0 + 0.1
        y = i // 5 / 4.0 + 0.1
        final_centers[i] = [x, y]
    final_radii = result.x
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
