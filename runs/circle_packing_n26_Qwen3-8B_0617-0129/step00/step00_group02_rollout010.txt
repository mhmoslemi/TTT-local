from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Hexagonal grid initialization
    num_circles = 26
    rows = int(np.sqrt(num_circles))
    cols = num_circles // rows + (1 if num_circles % rows else 0)
    
    # Initial positions based on hexagonal packing (unit square scaled)
    centers = np.zeros((num_circles, 2))
    radius = 0.15  # initial radius estimate
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col + (row % 2) * 0.5  # staggered x
        y = row * np.sqrt(3)/2  # vertical spacing for hexagonal packing
        x = x * (1 / (cols - 1)) if cols > 1 else 0.5
        y = y * (1 / (rows - 1)) if rows > 1 else 0.5
        centers[i] = np.array([x, y])
    
    # Initial guess for radii
    radii = np.full(num_circles, radius)
    
    # Objective function to maximize the sum of radii
    def objective(x):
        # x is a flattened array of radii
        return -np.sum(x)  # minimize the negative sum to maximize the sum
    
    # Constraints: non-overlapping and within the unit square
    def constraint_overlap(x):
        x_r = x.reshape((num_circles,))
        dist_matrix = cdist(centers, centers)
        min_dist = np.min(dist_matrix - (x_r[:, np.newaxis] + x_r[np.newaxis, :]))
        return min_dist + 1e-8  # ensure no overlap
    
    def constraint_boundary(x):
        x_r = x.reshape((num_circles,))
        dist_boundary = np.array([

        return np.min(dist_boundary) + 1e-8  # ensure within the unit square
