import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize sum

    def non_overlap_constraints(circles):
        centers = circles[:26, :2]
        radii = circles[:26, 2]
        constraints = []

        # Add boundary constraints
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda circles, i=i: x - r + 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda circles, i=i: 1.0 - x - r + 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda circles, i=i: y - r + 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda circles, i=i: 1.0 - y - r + 1e-12})

        # Add non-overlap constraints
        for i in range(26):
            for j in range(i + 1, 26):
                def constraint_func(circles, i=i, j=j):
                    x1, y1 = circles[i, :2]
                    r1 = circles[i, 2]
                    x2, y2 = circles[j, :2]
                    r2 = circles[j, 2]
                    dx = x1 - x2
                    dy = y1 - y2
                    dist_sq = dx**2 + dy**2
                    return np.sqrt(dist_sq) - r1 - r2 + 1e-12
                constraints.append({'type': 'ineq', 'fun': constraint_func})

        return constraints

    def initial_config():
        # Hexagonal packing with some adjustment to fit into square
        centers = np.zeros((26, 2))
        radii = np.zeros(26)
        spacing = 0.25  # Initial spacing between centers
        radius = 0.05

        for i in range(26):
            row = i // 5
            col = i % 5
            x = col * spacing
            y = row * spacing
            if row % 2 == 1:
                x += spacing / 2
            centers[i, 0] = x
            centers[i, 1] = y
            radii[i] = radius

        return np.column_stack((centers, radii))

    def bounds():
        # Bounds for each circle: (x, y, r)
        return Bounds([0, 0, 0], [1, 1, 0.15])

    # Initialize
    initial_guess = initial_config()
    constraints = non_overlap_constraints(initial_guess)
    result = opt.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        constraints=constraints,
        bounds=bounds(),
        tol=1e-8
    )

    centers = result.x[:, :2]
    radii = result.x[:, 2]
    sum_radii = np.sum(radii)

    # Validate
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Invalid packing: {msg}")

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
