import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def get_optimal_radii(centers):
    n = centers.shape[0]
    radii = np.zeros(n)
    for i in range(n):
        min_dist = np.inf
        for j in range(n):
            if i != j:
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < min_dist:
                    min_dist = dist
        r = min_dist / 2
        r = min(r, centers[i][0], 1 - centers[i][0], centers[i][1], 1 - centers[i][1])
        radii[i] = r
    return radii

def optimize_packing(centers):

        return -np.sum(x)

            r_i = x[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12

        return 0.0

    bounds = [(0, 1) for _ in range(n)]
    result = optimize.minimize(
        objective,
        initial_radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraint},
        method="SLSQP",
        options={"ftol": 1e-9, "maxiter": 500, "disp": False}
    )
    return result.x

def generate_hexagonal_config(n):
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    spacing = 1.0 / (rows + 1)
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(rows):
        for c in range(cols):
            x = c * spacing + spacing / 2
            y = r * spacing + spacing / 2
            if r % 2 == 1:
                y += spacing / 2
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers
