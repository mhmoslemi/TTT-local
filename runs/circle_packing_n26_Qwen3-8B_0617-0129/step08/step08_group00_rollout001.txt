import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_hexagonal_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    rows = int(np.sqrt(n))
    cols = int(np.sqrt(n))
    for r in range(rows + 1):
        for c in range(cols + 1):
            if i < n:
                x = c / (cols + 1) + 0.5 / (cols + 1)
                y = r / (rows + 1) + 0.5 / (rows + 1) + (r % 2) * 0.5 / (rows + 1)
                x = np.clip(x, 0.0, 1.0)
                y = np.clip(y, 0.0, 1.0)
                centers[i] = [x, y]
                i += 1
    return centers

def generate_corner_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    for corner in [(0, 0), (1, 0), (0, 1), (1, 1)]:
        for col in range(5):
            for row in range(5):
                if i < n:
                    x = corner[0] + col * 0.1 + 0.05
                    y = corner[1] + row * 0.1 + 0.05
                    centers[i] = [x, y]
                    i += 1
    return centers

def generate_random_centers(n):
    centers = np.random.rand(n, 2)
    centers = np.clip(centers, 0.0, 1.0)
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < 0.05:
                centers[i] = [np.random.rand(), np.random.rand()]
                centers[i] = np.clip(centers[i], 0.0, 1.0)
    return centers

def generate_clustered_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    for col in range(5):
        for row in range(5):
            if i < n:
                x = col * 0.2 + 0.1
                y = row * 0.2 + 0.1
                centers[i] = [x, y]
                i += 1
    return centers

def run_packing():
    n = 26

    initial_layouts = []
    initial_layouts.append(generate_hexagonal_centers(n))
    initial_layouts.append(generate_corner_centers(n))
    initial_layouts.append(generate_random_centers(n))
    initial_layouts.append(generate_clustered_centers(n))

    best_sum = 0.0
    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)

    for centers in initial_layouts:
        radii = np.zeros(n)
        for i in range(n):
            min_dist_boundary = min(centers[i][0], 1 - centers[i][0], centers[i][1], 1 - centers[i][1])
            min_dist_neighbor = np.inf
            for j in range(n):
                if i != j:
                    dx = centers[i][0] - centers[j][0]
                    dy = centers[i][1] - centers[j][1]
                    dist = np.sqrt(dx**2 + dy**2)
                    if dist < min_dist_neighbor:
                        min_dist_neighbor = dist
            r = min(min_dist_boundary, min_dist_neighbor / 2)
            radii[i] = r

        def objective(x):
            return -np.sum(x)

        def constraints(x):
            for i in range(n):
                x_i, y_i = centers[i]
                r_i = x[i]
                if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or
