import scipy.optimize as optimize
from itertools import combinations

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26
    
    # Generate an initial grid-based layout with flexibility
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    spacing = 1.0 / (rows + 1)
    
    centers = np.zeros((n, 2))
    for i in range(n):
        col = i % cols
        row = i // cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i] = [x, y]
    
    # Initial radii based on the initial layout
    initial_radii = np.full(n, 0.05)
    
    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)
    
    # Constraint function: circles must stay within the unit square
    def constraint_square(x):
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = x[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12

                return -np.inf
        return 0.0
    
    # Constraint function: circles must not overlap
    def constraint_overlap(x):
        for i, j in combinations(range(n), 2):

        'gtol': 1e-9,

    }
    
    # Perform optimization using L-BFGS-B for better handling of bounds
    result = optimize.minimize(
