def run_packing():
    n = 26
    # Generate a hexagonal grid packing
    cols = int(np.ceil(np.sqrt(n)))

        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        if row % 2 == 1:
            x += spacing / 2
        centers[i] = [x, y]

        # Space for radius such that no overlap
        r = min_dist / 2
