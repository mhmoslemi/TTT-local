import scipy.optimize as optimize
from itertools import combinations

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26

    # Custom initial layout with some circles near corners
    centers = np.zeros((n, 2))

    corner_circles = [

        [0.95, 0.05],  # top-right
        [0.05, 0.95],  # bottom-left
        [0.95, 0.95],  # bottom-right
        [0.05, 0.5],   # left-middle
        [0.95, 0.5],   # right-middle
    ]

    # 20 circles in a hexagonal grid in the middle
    cols = int(np.ceil(np.sqrt(20)))
    rows = int(np.ceil(20 / cols))
    spacing = 1.0 / (rows + 1)

    middle_centers = []
    for i in range(20):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2 + (col % 2) * spacing / 2
        middle_centers.append([x, y])

    # Combine corner and middle centers
    centers[:6] = corner_circles
    centers[6:] = middle_centers

    # Initial radii as 0.05 (can be adjusted)
    initial_radii = np.full(n, 0.05)

    # Objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)

    def constraint_square(x):

            r_i = x[i]

                return -np.inf
        return 0.0
