from itertools import combinations

            if dist < radii[i] + radii[j] - 1e-12:

def run_packing():
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    spacing = 1.0 / (rows + 1)

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2 + (col % 2) * spacing / 2
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial radii for optimization

    # Objective: maximize sum of radii
    def objective(x):
        return -np.sum(x)
