import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    spacing = 1 / (rows + 1)
    
    # Create a hexagonal pattern with dynamic row spacing
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        offset = (row % 2) * spacing / 2  # Alternate row offset for hexagonal pattern
        x = col * spacing + spacing / 2 + offset
        y = row * spacing + spacing / 2
        centers[i] = [x, y]
    
    # Initial radii, small to allow optimization
    initial_radii = np.full(n, 0.05)

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)

                return -np.inf
        return 0.0

        return 0.0

    # Bounds for radii

    # Optimization using SLSQP with tight constraints
    result = optimize.minimize(
