import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
import matplotlib.pyplot as plt

def run_packing():
    # Initial placement: Hexagonal close-packed (HCP) pattern
    n = 26
    # Calculate grid dimensions
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    # Create an initial grid with spacing to avoid overlaps
    spacing = 0.3
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing
        y = row * spacing
        centers[i] = [x, y]
    
    # Define objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    # Constraints
    # 1. All circles must be within the unit square
    def constraint_boundary(radii, centers):
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return np.inf
        return 0
    
    # 2. No two circles can overlap
    def constraint_overlap(radii, centers):
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        return 0
    
    # Initial guess for radii (same for all circles)

    bounds = [(0, 1) for _ in range(n)]

        return constraint_boundary(radii, centers) + constraint_overlap(radii, centers)

        return constraint_combined(radii)
    
    # Optimization

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
