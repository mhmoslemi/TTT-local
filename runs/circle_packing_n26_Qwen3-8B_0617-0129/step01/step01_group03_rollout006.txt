import numpy as np
import scipy.optimize as opt

def run_packing():
    # Generate initial hexagonal grid of 26 circles
    n = 26
    max_rows = int(np.sqrt(n))
    remaining = n - max_rows * (max_rows + 1) // 2

    # Create hexagonal grid
    centers = []
    radii = []

    for row in range(max_rows):
        for col in range(max_rows + 1):
            x = col + (row % 2) * 0.5
            y = row * np.sqrt(3) / 2
            x = x / (max_rows + 1)
            y = y / (max_rows + 1)
            centers.append([x, y])
            radii.append(0.01)

    # Convert to numpy arrays
    centers = np.array(centers)
    radii = np.array(radii)

    # Objective function to maximize the sum of radii
    def objective(x):
        return -np.sum(x)

    # Bounds for radii (>= 0)
    bounds = [(0, 1.0) for _ in range(n)]

    # Constraints
    def constraint_center(x, i):
        x_i, y_i = centers[i]
        if x_i - x[i] < -1e-12 or x_i + x[i] > 1 + 1e-12 or y_i - x[i] < -1e-12 or y_i + x[i] > 1 + 1e-12:
            return -1
        return 0

    def constraint_overlap(x, i, j):
        dist = np.linalg.norm(centers[i] - centers[j])
        if dist < x[i] + x[j] - 1e-12:
            return -1
        return 0

    # Convert constraints to a list of nonlinear constraints
    constraints = []
    for i in range(n):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_center(x, i)})
        for j in range(i + 1, n):
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)})

    # Optimization
    res = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Use the optimized radii and same centers
    opt_radii = res.x
    if not res.success:

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
