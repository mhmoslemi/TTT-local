import scipy.optimize as optimize

def run_packing():

    initial_radius = 0.045  # Initial guess for radii (can be refined)
    center_padding = 0.01  # Padding from boundaries for initial grid
    grid = np.zeros((num_circles, 2))

    # Define an initial hexagonal grid layout
    # Place 4 corners
    grid[0] = [center_padding, center_padding]  # Top-left
    grid[1] = [1 - center_padding, center_padding]  # Top-right
    grid[2] = [center_padding, 1 - center_padding]  # Bottom-left
    grid[3] = [1 - center_padding, 1 - center_padding]  # Bottom-right

    # Add hexagonal grid in the center
    row = 4
    col = 0

                x = center_padding + j * (1 - 2 * center_padding) / 4
                y = center_padding + i * (1 - 2 * center_padding) / 4
            else:
                x = center_padding + (j + 0.5) * (1 - 2 * center_padding) / 4
                y = center_padding + i * (1 - 2 * center_padding) / 4
            grid[row + i] = [x, y]
            col += 1

    # Initial guess for radii
    radii = np.full(num_circles, initial_radius)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize the negative sum

    # Constraints: non-overlapping (distance >= sum of radii), boundary, and non-negative
    def constraints(x):
        centers = x[:num_circles * 2].reshape(num_circles, 2)
        radii = x[num_circles * 2:]
        constraints = []
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
