import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess for centers: grid-based placement
    num_circles = 26
    grid_size = int(np.ceil(np.sqrt(num_circles)))
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // grid_size
        col = i % grid_size
        centers[i, 0] = (col + 0.5) / grid_size
        centers[i, 1] = (row + 0.5) / grid_size

    # Initial guess for radii: all equal
    radii = np.full(num_circles, 0.05)

    # Objective function: minimize the negative sum of radii
    def objective(x):
        centers_flat = x[:num_circles * 2].reshape((num_circles, 2))
        radii_flat = x[num_circles * 2:]
        return -np.sum(radii_flat)

    # Constraints: non-overlapping and within bounds
    def constraint_non_overlap(x):
        centers_flat = x[:num_circles * 2].reshape((num_circles, 2))
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist_sq = dx**2 + dy**2
                if dist_sq < (radii_flat[i] + radii_flat[j])**2 - 1e-12:
                    return -1.0
        return 0.0

    def constraint_bounds(x):
        centers_flat = x[:num_circles * 2].reshape((num_circles, 2))
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            x_center, y_center = centers_flat[i]
            r = radii_flat[i]
            if x_center - r < -1e-12 or x_center + r > 1 + 1e-12:
                return -1.0
            if y_center - r < -1e-12 or y_center + r > 1 + 1e-12:
                return -1.0
        return 0.0

    # Initial x vector: centers and radii
    x0 = np.concatenate((centers.flatten(), radii))

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_non_overlap},
        {'type': 'ineq', 'fun': constraint_bounds}
    ]

    bounds = [
        (0.0, 1.0) for _ in range(num_circles * 2)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
