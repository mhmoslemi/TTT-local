import numpy as np
import scipy.optimize as optimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_circles = 26
    grid_spacing = 1.0 / math.sqrt(3)  # Hexagonal grid spacing

    num_cols = 6  # Number of columns in the hexagonal grid

    # Initial centers for hexagonal packing
    centers = np.zeros((num_circles, 2))

            col = j

                col += 0.5  # Offset for staggered rows

            y = row * grid_spacing * 1.5  # Vertical spacing in hexagonal grid
            if y > 1.0 - 1e-6:
                y = 1.0 - 1e-6
            if x > 1.0 - 1e-6:
                x = 1.0 - 1e-6
            centers[i * num_cols + j] = [x, y]

    radii = np.full(num_circles, 0.1)

    def objective(radii):
        return -np.sum(radii)

        # Extract centers and radii
        centers = x[:num_circles * 2].reshape((num_circles, 2))

            if (x_center - r < -1e-12 or x_center + r > 1 + 1e-12 or

                return np.inf

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
