import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        x, y = centers[i]
        r = radii[i]

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

    n = 26
    centers = np.zeros((n, 2))

    # First, arrange circles in a hexagonal grid
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    
    spacing = 1 / (rows + 1)
    
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2

        centers[i] = [x, y]

        # x is a vector of flattened radii (n elements)

        # Objective: maximize sum of radii (equivalent to minimize negative sum)
        return -np.sum(radii)
    
    # Define constraints
    def constraints(x):
        nonlocal centers, radii
        radii = x
        # Check boundary constraints
        valid = True
        for i in range(n):
            x_i, y_i = centers[i]

    # Bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(n)]
    
    # Use SLSQP optimization
    result = optimize.minimize(

        bounds=bounds,
