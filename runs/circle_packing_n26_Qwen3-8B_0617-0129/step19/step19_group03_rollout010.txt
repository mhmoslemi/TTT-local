def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_hexagonal_config(n, spacing=0.2):
    centers = np.zeros((n, 2))
    idx = 0
    for row in range(5):
        for col in range(6):
            if row % 2 == 0:
                x = col * spacing + spacing / 2
            else:
                x = (col + 0.5) * spacing
            y = row * spacing + spacing / 2
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers

        for row in range(level + 1):
            for col in range(level + 1):
                x = col * spacing + spacing / 2
                y = row * spacing + spacing / 2
                if idx < n:
                    centers[idx] = [x, y]
                    idx += 1
    return centers

def generate_random_config(n, density=0.8):

        x = np.random.rand()

        if np.random.rand() < density:
            centers[idx] = [x, y]
            idx += 1
    return centers

def optimize_config(centers):
    n = centers.shape[0]
    radii = np.zeros(n)
    for i in range(n):
        min_dist = np.inf
        for j in range(n):
            if i != j:
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < min_dist:
                    min_dist = dist
        r = min_dist / 2
        r = min(r, centers[i][0])
        r = min(r, 1 - centers[i][0])
        r = min(r, centers[i][1])
        r = min(r, 1 - centers[i][1])
        radii[i] = r

        return -np.sum(x)

                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < x[i] + x[j] - 1e-12:
                    return -np.inf
        return 0.0

    result = optimize.minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
