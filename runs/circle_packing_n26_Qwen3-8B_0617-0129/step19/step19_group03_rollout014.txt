def validate_packing(centers, radii):

def generate_hexagonal_config(n, rows=6, cols=6, spacing=0.2):
    centers = np.zeros((n, 2))
    idx = 0
    for row in range(rows):
        for col in range(cols):
            if row % 2 == 0:
                x = col * spacing + spacing / 2
            else:
                x = (col + 0.5) * spacing
            y = row * spacing + spacing / 2
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers
