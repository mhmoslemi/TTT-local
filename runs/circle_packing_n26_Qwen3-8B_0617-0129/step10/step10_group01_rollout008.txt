import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_hexagonal_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    rows = int(np.sqrt(n)) + 1
    cols = int(np.sqrt(n)) + 1
    for r in range(rows):
        for c in range(cols):
            if i < n:
                x = c / cols + 0.5 / cols
                y = r / rows + 0.5 / rows
                x = np.clip(x, 0.0, 1.0)
                y = np.clip(y, 0.0, 1.0)
                centers[i] = [x, y]
                i += 1
    return centers

def generate_corner_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    corners = [(0, 0), (1, 0), (0, 1), (1, 1)]

        x_start = corner[0] + 0.1

        for row in range(5):
            for col in range(5):
                if i < n:
                    x = x_start + col * 0.15
                    y = y_start + row * 0.15
                    x = np.clip(x, 0.0, 1.0)
                    y = np.clip(y, 0.0, 1.0)
                    centers[i] = [x, y]

def generate_random_centers(n):

    n = 26
    initial_layouts = []
