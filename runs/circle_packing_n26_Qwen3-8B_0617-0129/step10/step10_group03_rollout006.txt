import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]
    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"
    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"
    return True, "ok"

def generate_grid_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(np.sqrt(n)))
    spacing = 1.0 / (cols + 1)
    for row in range(rows):
        for col in range(cols):
            if i < n:
                x = col * spacing + spacing / 2.0
                y = row * spacing + spacing / 2.0
                centers[i] = [x, y]
                i += 1
    return centers

def generate_corner_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    spacing = 0.2
    for corner in [(0, 0), (1, 0), (0, 1), (1, 1)]:

                    i += 1
    return centers

def generate_hybrid_centers(n):

                    i += 1
    return centers

def run_packing():
    n = 26

    # Generate multiple initial layouts
    initial_layouts = []
    initial_layouts.append(generate_grid_centers(n))
    initial_layouts.append(generate_corner_centers(n))
    initial_layouts.append(generate_hybrid_centers(n))

    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)

        radii = np.zeros(n)
        for i in range(n):

            min_dist_boundary = min(x, 1 - x, y, 1 - y)
            # Neighbor distances
            min_dist_neighbor = np.inf
            for j in range(n):
