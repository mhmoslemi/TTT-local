import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_hexagonal_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    rows = int(np.sqrt(n))
    cols = int(np.sqrt(n))
    for r in range(rows + 1):
        for c in range(cols + 1):
            if i < n:
                # Hexagonal grid offset
                offset_r = (r % 2) * (1 / (cols + 1))
                x = c / (cols + 1) + 0.5 / (cols + 1)
                y = r / (rows + 1) + 0.5 / (rows + 1) + offset_r
                x = np.clip(x, 0.0, 1.0)
                y = np.clip(y, 0.0, 1.0)
                centers[i] = [x, y]
                i += 1
    return centers

def generate_corner_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    for corner in [(0, 0), (1, 0), (0, 1), (1, 1)]:
        for r in range(3):
            for c in range(3):
                if i < n:
                    x = corner[0] + c / 3.0
                    y = corner[1] + r / 3.0
                    x = np.clip(x, 0.0, 1.0)
                    y = np.clip(y, 0.0, 1.0)
                    centers[i] = [x, y]
