def validate_packing(centers, radii):

        return False, "NaN values present"

def generate_hexagonal_centers(n):

    i = 0
    for row in range(int(np.sqrt(n)) + 1):
        for col in range(int(np.sqrt(n)) + 1):

                spacing = 1.0 / (np.sqrt(n) + 1.0)
                x = col * spacing + spacing / 2.0

                x = np.clip(x, spacing / 2.0, 1.0 - spacing / 2.0)
                y = np.clip(y, spacing / 2.0, 1.0 - spacing / 2.0)
                centers[i] = [x, y]
                i += 1
    return centers
