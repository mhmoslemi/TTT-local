import scipy.optimize as optimize
import random

def validate_packing(centers, radii):

        return False, "NaN values present"

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"
    return True, "ok"

def generate_corner_cluster_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    for corner in [(0, 0), (1, 0), (0, 1), (1, 1)]:
        for col in range(5):
            for row in range(5):
                if i < n:
                    spacing = 0.2
                    x = corner[0] + col * spacing + spacing / 2.0
                    y = corner[1] + row * spacing + spacing / 2.0
                    x = np.clip(x, spacing / 2.0, 1.0 - spacing / 2.0)
                    y = np.clip(y, spacing / 2.0, 1.0 - spacing / 2.0)
                    centers[i] = [x, y]
                    i += 1
    return centers

def generate_hexagonal_centers(n):
    centers = np.zeros((n, 2))
    i = 0
    spacing = 1.0 / (np.sqrt(n) + 1.0)
    for row in range(int(np.sqrt(n)) + 1):
        for col in range(int(np.sqrt(n)) + 1):
            if i < n:
                x = col * spacing + spacing / 2.0
                y = row * spacing + spacing / 2.0 + (row % 2) * spacing / 2.0
                x = np.clip(x, spacing / 2.0, 1.0 - spacing / 2.0)
                y = np.clip(y, spacing / 2.0, 1.0 - spacing / 2.0)
                centers[i] = [x, y]
                i += 1
    return centers

def generate_random_centers(n):
    centers = np.zeros((n, 2))
    for i in range(n):
        x = np.random.uniform(0.1, 0.9)
        y = np.random.uniform(0.1, 0.9)

        x = radius * np.cos(angle) + 0.5
        y = radius * np.sin(angle) + 0.5
        centers[i] = [x, y]
    return centers
