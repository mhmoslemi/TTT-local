import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]
    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"
    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"
    return True, "ok"

def generate_grid_centers(n, grid_size=5, spacing=0.2):
    centers = np.zeros((n, 2))
    i = 0
    for row in range(grid_size):
        for col in range(grid_size):
            if i < n:
                x = col * spacing + spacing / 2.0
                y = row * spacing + spacing / 2.0
                x = np.clip(x, spacing / 2.0, 1.0 - spacing / 2.0)
                y = np.clip(y, spacing / 2.0, 1.0 - spacing / 2.0)
                centers[i] = [x, y]
                i += 1
    return centers

def generate_hexagonal_centers(n, grid_size=5, spacing=0.18):
    centers = np.zeros((n, 2))
    i = 0
    for row in range(grid_size):
        for col in range(grid_size):
            if i < n:
                x = col * spacing + spacing / 2.0
                y = row * spacing + spacing / 2.0 + (row % 2) * spacing / 2.0
                x = np.clip(x, spacing / 2.0, 1.0 - spacing / 2.0)
                y = np.clip(y, spacing / 2.0, 1.0 - spacing / 2.0)
                centers[i] = [x, y]
                i += 1
    return centers

def generate_corner_centers(n, spacing=0.15):
    centers = np.zeros((n, 2))
    i = 0
    corner_positions = [(0, 0), (1, 0), (0, 1), (1, 1)]
    for corner in corner_positions:
        for col in range(5):
            for row in range(5):
                if i < n:
                    x = corner[0] + col * spacing + spacing / 2.0
                    y = corner[1] + row * spacing + spacing / 2.0
                    x = np.clip(x, spacing / 2.0, 1.0 - spacing / 2.0)
                    y = np.clip(y, spacing / 2.0, 1.0 - spacing / 2.0)
                    centers[i] = [x, y]
                    i += 1
    return centers

def run_packing():
    n = 26

    # Generate multiple initial layouts
    initial_layouts = []
    initial_layouts.append(generate_hexagonal_centers(n))
    initial_layouts.append(generate_corner_centers(n))
    initial_layouts.append(generate_grid_centers(n))
    initial_layouts.append(generate_hexagonal_centers(n, grid_size=6, spacing=0.16))
    initial_layouts.append(generate_corner_centers(n, spacing=0.16))

    best_sum = 0.0
    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)

    for centers in initial_layouts:
        # Initialize radii with better spacing
        radii = np.zeros(n)
        for i in range(n):
            # Boundary distance
            min_dist_boundary = min(centers[i][0], 1 - centers[i][0], centers[i][1], 1 - centers[i][1])
            # Neighbor distance
            min_dist_neighbor = np.inf
            for j in range(n):

                    dy = centers[i][1] - centers[j][1]

        def objective(x):
            return -np.sum(x)
