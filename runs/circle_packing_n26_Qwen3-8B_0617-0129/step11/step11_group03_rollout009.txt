    centers = np.zeros((n, 2))
    i = 0
    cols = int(np.sqrt(n)) + 1
    rows = int(np.ceil(n / cols))
    for r in range(rows):

            if i < n:
                x = c / cols
                y = r / rows
                centers[i] = [x, y]
                i += 1
    return centers
