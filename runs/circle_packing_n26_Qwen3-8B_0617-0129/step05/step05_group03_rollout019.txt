import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26
    
    # Generate a hexagonal grid with some refinement at the edges
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    spacing = 1 / (rows + 0.5)  # Smaller spacing for better packing
    
    centers = np.zeros((n, 2))
    
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i] = [x, y]
    
    # Initial radii based on minimum distance to other circles and boundaries
    radii = np.zeros(n)
    
    for i in range(n):
        min_dist = np.inf
        for j in range(n):
            if i != j:
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < min_dist:
                    min_dist = dist
        r = min_dist / 2
        r = min(r, centers[i][0] - 1e-12)
        r = min(r, 1 - centers[i][0] - 1e-12)
        r = min(r, centers[i][1] - 1e-12)
        r = min(r, 1 - centers[i][1] - 1e-12)
        radii[i] = r
    
    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)
    
    # Constraints: non-overlapping and within the unit square
    def constraints(x):
        for i in range(n):

            r_i = x[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12

                    return -np.inf
        return 0.0
    
    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(n)]
    
    # Run the optimization using SLSQP
    result = optimize.minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
