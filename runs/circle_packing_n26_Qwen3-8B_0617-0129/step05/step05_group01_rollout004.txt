import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26

    # Generate a hexagonal grid layout
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    spacing = 1 / (rows + 1)
    centers = np.zeros((n, 2))
    count = 0

    for i in range(rows):
        for j in range(cols):
            if count >= n:
                break
            x = j * spacing + spacing / 2
            y = i * spacing + spacing / 2
            centers[count] = [x, y]
            count += 1

    # Add extra rows for hexagonal packing
    for i in range(rows - 1):
        for j in range(cols - 1):
            if count >= n:
                break
            x = j * spacing + spacing / 2
            y = (i + 1) * spacing + spacing / 2 + spacing * 0.5
            centers[count] = [x, y]
            count += 1

    # Initialize radii based on local distances and boundaries
    radii = np.zeros(n)
    for i in range(n):
        # Distance to nearest neighbor in the grid
        min_dist = np.inf
        for j in range(n):
            if i != j:
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < min_dist:
                    min_dist = dist
        # Space for radius such that no overlap
        r = min_dist / 2
        # Space for boundary
        r = min(r, centers[i][0])
        r = min(r, 1 - centers[i][0])
        r = min(r, centers[i][1])
        r = min(r, 1 - centers[i][1])
        radii[i] = r

    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Define constraints
    def constraints(x):
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = x[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12
                    or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return -np.inf
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < x[i] + x[j] - 1e-12:
                    return -np.inf
        return 0.0

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(n)]

    # Run the optimization with improved parameters
    result = optimize.minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
        options={"ftol": 1e-9, "maxiter": 1000, "disp": False}
    )

    final_radii = result.x
    final_centers = centers

    # Validate packing
    valid, message = validate_packing(final_centers, final_radii)
    if not valid:
        print(f"Validation Failed: {message}")
        return np.array([[-1, -1]]), np.array([-1]), -1

    sum_radii = np.sum(final_radii)

    return final_centers, final_radii, sum_radii
