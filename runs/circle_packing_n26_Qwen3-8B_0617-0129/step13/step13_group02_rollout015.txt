from scipy.optimize import differential_evolution

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_hex_config(n, rows=5, cols=5, scale=1.0):
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(rows):
        for c in range(cols):
            x = c * (1.0 / (cols)) + 1.0 / (2 * cols)
            if r % 2 == 1:
                x += 0.5 / cols
            y = r * (1.0 / (rows)) + 1.0 / (2 * rows)
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers

def generate_grid_config(n, rows=5, cols=5):
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(rows):
        for c in range(cols):
            x = c * (1.0 / (cols)) + 1.0 / (2 * cols)
            y = r * (1.0 / (rows)) + 1.0 / (2 * rows)
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers

def generate_corner_config(n, spacing=0.2):
    centers = np.zeros((n, 2))
    idx = 0
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    for r in range(rows):
        for c in range(cols):

def run_packing():

    def objective(x):

    def constraints(x, centers):

            r_i = x[i]
