import numpy as np
import scipy.optimize as optimize
from scipy.optimize import differential_evolution

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_grid_config(n, rows=5, cols=9):
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(rows):
        for c in range(cols):
            x = c * (1.0 / (cols)) + 1.0 / (2 * cols)
            y = r * (1.0 / (rows)) + 1.0 / (2 * rows)
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers

def generate_hex_config(n, rows=5, cols=9):
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(rows):
        for c in range(cols):
            x = c * (1.0 / (cols)) + 1.0 / (2 * cols)
            if r % 2 == 1:
                x += 0.5 / cols
            y = r * (1.0 / (rows)) + 1.0 / (2 * rows)
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers

def generate_corner_config(n, spacing=0.18):
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(5):
        for c in range(5):
            x = c * spacing + spacing / 2
            y = r * spacing + spacing / 2
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    for r in range(5):
        for c in range(5):
            x = 1 - (c * spacing + spacing / 2)
            y = r * spacing + spacing / 2
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    for r in range(5):
        for c in range(5):
            x = c * spacing + spacing / 2
            y = 1 - (r * spacing + spacing / 2)
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    for r in range(5):
        for c in range(5):
            x = 1 - (c * spacing + spacing / 2)
            y = 1 - (r * spacing + spacing / 2)
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers

def run_packing():
    n = 26
    best_sum = 0.0
    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)

    initial_configurations = [
        generate_grid_config(n, rows=5, cols=9),
        generate_hex_config(n, rows=5, cols=9),
        generate_corner_config(n),
        generate_grid_config(n, rows=6, cols=5)
    ]

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii, centers):
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return -np.inf
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -np.inf
        return 0.0

    for centers in initial_configurations:
        def constraint_func(radii):
            return constraints(radii, centers)

        bounds = [(0, 1) for _ in range(n)]

        result = differential_evolution(
            objective,
            bounds,
            constraints=constraint_func,
            strategy='best1bin',
            maxiter=100,
            popsize=15,
            tol=1e-9,
            disp=False
        )

        final_radii = result.x

            print(f"Validation Failed: {message}")

        sum_radii = np.sum(final_radii)

            best_sum = sum_radii
            best_centers = final_centers.copy()
            best_radii = final_radii.copy()

            return -np.sum(radii)

            return constraints(radii, perturbed_centers)

        bounds = [(0, 1) for _ in range(n)]

        result = optimize.minimize(
            objective_perturbed,
            best_radii,
            bounds=bounds,
            constraints={"type": "ineq", "fun": constraint_func_perturbed},

            best_sum = sum_radii_perturbed
            best_centers = final_centers_perturbed.copy()
            best_radii = final_radii_perturbed.copy()

    return best_centers, best_radii, best_sum
