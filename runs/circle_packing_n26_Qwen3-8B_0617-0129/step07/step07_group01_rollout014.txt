import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i][0] - centers[j][0]
            dy = centers[i][1] - centers[j][1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n = 26
    # Generate a hexagonal grid inside the unit square
    def hexagonal_grid(num_circles):
        # Create a hexagonal grid of circles within the unit square
        row_count = int(np.sqrt(num_circles))
        col_count = (num_circles + row_count - 1) // row_count
        centers = np.zeros((num_circles, 2))
        spacing = 1.0 / (col_count + 1)
        for row in range(row_count):
            for col in range(col_count):
                if row % 2 == 0:
                    x = col * spacing + spacing / 2
                else:
                    x = (col + 0.5) * spacing
                y = row * spacing + spacing / 2
                if x + spacing / 2 <= 1 and y + spacing / 2 <= 1:
                    centers[row * col_count + col] = [x, y]
        return centers

    # Initial grid using hexagonal pattern
    initial_centers = hexagonal_grid(n)

    # Objective function to minimize (negative of sum of radii)
    def objective(radii):
        return -np.sum(radii)

    # Constraints: no overlap and within unit square
    def constraint(radii, centers):
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12

                    return -np.inf
        return 0.0

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(n)]

    result = optimize.minimize(
