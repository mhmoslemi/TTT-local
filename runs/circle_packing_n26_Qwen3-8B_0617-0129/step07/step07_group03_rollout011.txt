def optimize_circle_packing(centers: np.ndarray, n: int) -> np.ndarray:
    """Optimize the radii of the given centers to maximize the sum of radii."""
    # Initial radii calculation: minimum distance to nearest neighbor and boundary
    radii = np.zeros(n)
    for i in range(n):
        min_dist = np.inf
        for j in range(n):
            if i != j:
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)

                    min_dist = dist
        r = min_dist / 2
        r = min(r, centers[i][0])
        r = min(r, 1 - centers[i][0])
        r = min(r, centers[i][1])
        r = min(r, 1 - centers[i][1])
        radii[i] = r

    # Objective function to minimize (negative of sum of radii)
    def objective(x):
        return -np.sum(x)

    # Inequality constraints: no overlap, within square
    def constraints(x):
        # Boundary constraints
        for i in range(n):
            x_i, y_i = centers[i]

                    return -np.inf
        return 0.0

    )
    return result.x
