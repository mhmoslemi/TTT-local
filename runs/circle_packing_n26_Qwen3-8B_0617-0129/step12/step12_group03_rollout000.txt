def validate_packing(centers, radii):

                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_grid_config(n, rows=5, cols=9):
    centers = np.zeros((n, 2))
    idx = 0
    for r in range(rows):
        for c in range(cols):
            x = c * (1.0 / (cols)) + 1.0 / (2 * cols)
            y = r * (1.0 / (rows)) + 1.0 / (2 * rows)
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers
