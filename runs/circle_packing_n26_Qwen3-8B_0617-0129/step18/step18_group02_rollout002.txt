import numpy as np
import scipy.optimize as optimize
import random

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_grid_config(n, rows=5, cols=6, spacing=0.2):
    centers = np.zeros((n, 2))
    idx = 0
    for row in range(rows):
        for col in range(cols):
            x = col * spacing + spacing / 2
            y = row * spacing + spacing / 2
            if idx < n:
                centers[idx] = [x, y]
                idx += 1
    return centers

def optimize_config(centers):

    # Initialize radii based on minimum distances to neighbors and boundaries
    radii = np.zeros(n)

        min_dist = np.inf

                    min_dist = dist
        r = min_dist / 2
        r = min(r, centers[i][0])
        r = min(r, 1 - centers[i][0])
        r = min(r, centers[i][1])
        r = min(r, 1 - centers[i][1])
        radii[i] = r

        return -np.sum(x)

                    return -np.inf
        return 0.0

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(n)]

    # Run the optimization with SLSQP method (better for constrained optimization)
    result = optimize.minimize(
        objective,
        radii,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
        options={"ftol": 1e-9, "maxiter": 500, "disp": False}
    )

    # Validate packing

        print(f"Validation Failed: {message}")
        return None, None, 0.0
