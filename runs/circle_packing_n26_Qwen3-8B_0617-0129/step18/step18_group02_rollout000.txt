def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            if dist < radii[i] + radii[j] - 1e-12:

    return True, "ok"

    centers = np.zeros((n, 2))
    idx = 0
    for row in range(rows):

            else:
                x = (col + 0.5) * spacing

                idx += 1
    return centers

def optimize_config(centers):
    n = centers.shape[0]
    # Initialize radii based on minimum distances to neighbors and boundaries
    radii = np.zeros(n)
    for i in range(n):
        min_dist = np.inf
        for j in range(n):
            if i != j:
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < min_dist:
                    min_dist = dist
        r = min_dist / 2
        r = min(r, centers[i][0])
        r = min(r, 1 - centers[i][0])
        r = min(r, centers[i][1])
        r = min(r, 1 - centers[i][1])
        radii[i] = r

    # Objective function to minimize (negative of sum of radii)
    def objective(x):
        return -np.sum(x)

    # Constraints: no overlap and within unit square
    def constraints(x):
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = x[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12

                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < x[i] + x[j] - 1e-12:
                    return -np.inf
        return 0.0

    # Bounds for radii: non-negative
    bounds = [(0, 1) for _ in range(n)]

    # Run the optimization with SLSQP method (better for constrained optimization)
    result = optimize.minimize(
        objective,
        radii,
        bounds=bounds,

        method="SLSQP",
