def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

            return False, f"Circle {i} has negative radius {radii[i]}"

        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

    centers = np.zeros((n, 2))

                idx += 1
    return centers

def optimize_config(centers):

    # Initialize radii based on minimum distances to neighbors and boundaries
    radii = np.zeros(n)
    for i in range(n):
        min_dist = np.inf

                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]

                    min_dist = dist
        r = min_dist / 2
        r = min(r, centers[i][0])
        r = min(r, 1 - centers[i][0])

        r = min(r, 1 - centers[i][1])

        return -np.sum(x)

            r_i = x[i]
