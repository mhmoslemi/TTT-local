    def objective(x):

        radii_flat = x[n*2:].reshape(n)
        return -np.sum(radii_flat)

    center_bounds = [(0.0, 1.0) for _ in range(n*2)]

    # Bounds for radii (non-negative)
    radius_bounds = [(0.0, 1.0) for _ in range(n)]

    # Bounds for all variables
    bounds = center_bounds + radius_bounds

    # Constraints: ensure circles stay within the square
    def constraint_centers_within_square(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1.0
        return 0.0

    # Constraint: ensure no overlap between circles
    def constraint_no_overlap(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]

                distance = np.sqrt(dx**2 + dy**2)
                if distance < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1.0
        return 0.0

    # Combine constraints
    constraints = [
