import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial guess for centers using a hexagonal close packing pattern
    # We'll define a grid of 5 rows and 6 columns for 26 circles
    n = 26
    centers = np.zeros((n, 2))
    rows, cols = 5, 6
    radius_initial = 0.1  # Initial guess for the radii

    # Calculate the distance between centers in a hexagonal grid
    hexagon_distance = 2 * radius_initial
    row_offset = np.sqrt(3) * radius_initial

    # Assign positions
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                row = i * hexagon_distance
            else:
                row = i * hexagon_distance + row_offset
            col = j * 2 * radius_initial
            centers[i * cols + j] = [col, row]

    # Initial guess for radii
    radii = np.full(n, radius_initial)

    # Bounds for radius (non-negative)
    bounds = [(0, 1) for _ in range(n)]

    # Constraints: centers must lie within [0,1]^2
    def center_bounds(center):
        return np.array([center[0], center[1]])

    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Negative because we minimize

    # Constraints for center positions
    def center_constraints(x):
        center = x.reshape((n, 2))
        return np.concatenate([center_bounds(center[i]) for i in range(n)])

    # Constraints for circle positions
    def circle_position_constraints(x):
        center = x.reshape((n, 2))
        for i in range(n):
            x_i, y_i = center[i]
            if x_i - radii[i] < 0 or x_i + radii[i] > 1 or y_i - radii[i] < 0 or y_i + radii[i] > 1:
                return np.array([0])
        return np.array([])

    # Constraints for non-overlapping circles
    def non_overlap_constraints(x):
        center = x.reshape((n, 2))
        for i in range(n):
            for j in range(i + 1, n):
                x_i, y_i = center[i]
                x_j, y_j = center[j]
                dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.array([0])
        return np.array([])

    # Combine all constraints
    constraints = [
        {'type': 'ineq', 'fun': center_position_constraints},
        {'type': 'ineq', 'fun': non_overlap_constraints}
    ]

    # Run optimization
    result = optimize.minimize(
        objective,
        x0=radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-6
    )

    # Extract results
    radii_opt = result.x
    if not result.success:
        raise ValueError("Optimization failed with message: " + result.message)

    # Final centers (same as initial, but could be refined further)
    centers_opt = centers  # For now, keep initial centers

    # Validate packing
    valid, message = validate_packing(centers_opt, radii_opt)
    if not valid:
        raise ValueError(message)

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
