from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initialize centers in a grid with some padding to allow space for circles
    n_circles = 26
    grid_size = 5
    spacing = 1 / (grid_size - 1)

    # Generate an initial placement of circles using a grid
    centers = np.zeros((n_circles, 2))
    idx = 0
    for i in range(grid_size):

                centers[idx, 0] = i * spacing
                centers[idx, 1] = j * spacing

        # Ensure radii are non-negative
        new_radii = np.maximum(new_radii, 0)
        # Compute the sum of radii
        return -np.sum(new_radii)  # We minimize the negative to maximize the sum

    # Define constraints for the optimization

        return np.sum((centers[:, 0] - radii) < 0) + np.sum((centers[:, 0] + radii) > 1)

        dist = cdist(centers, centers)
        return np.sum(dist < (x + x[:, np.newaxis] - 1e-12))

    initial_guess = radii.copy()

    result = minimize(
