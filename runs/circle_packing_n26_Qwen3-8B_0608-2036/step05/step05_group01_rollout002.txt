import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((n, 2))

    num_cols = 6  # Columns
    offset = 0.1  # Random offset for spread

        row = i // num_cols
        col = i % num_cols
        if row % 2 == 1:
            col += 1  # Alternate row offset
        if col >= num_cols:
            col = 0
            row += 1
        # Position in hexagonal grid
        x = 0.5 * (1 - (col % 2)) + 0.5 * (col + 1) / (num_cols + 1)

        centers[i] = [x + np.random.uniform(-offset, offset), y + np.random.uniform(-offset, offset)]

    radii = np.ones(n) * initial_radius

    # Objective: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum for maximization

    # Bounds: radii >= 0
    bounds = [(0, np.inf) for _ in range(n)]

    # Constraints
    constraints = []

    # Constraint: All circles must be within the unit square
    def constraint_inside(radii, idx):
        x, y = centers[idx]

        return x - r + 1e-12  # x - r >= -1e-12
    def constraint_outside(radii, idx):

        r = radii[idx]
        return 1 + 1e-12 - x - r  # x + r <= 1 + 1e-12

        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_inside(r, i)})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_outside(r, i)})

    # Constraint: No overlapping circles
    def constraint_no_overlap(radii, i, j):

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for j in range(i + 1, n):

            if dist < radii[i] + radii[j] - 1e-12:
