import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = np.zeros((n, 2))
    
    # Initial placement: carefully place circles near corners and edges, and fill the center with a hexagonal grid
    # Place 2 circles at the corners
    corner_positions = np.array([
        [0.1, 0.1],
        [0.1, 0.9],
        [0.9, 0.1],
        [0.9, 0.9]
    ])
    
    # Place 6 circles along the edges (but not at the corners)
    edge_positions = np.array([
        [0.1, 0.5], [0.5, 0.1], [0.5, 0.9], [0.9, 0.5],
        [0.2, 0.5], [0.8, 0.5]
    ])
    
    # Place 16 circles in a hexagonal grid in the center
    hex_grid_positions = np.array([
        [0.3, 0.3], [0.5, 0.3], [0.7, 0.3],
        [0.2, 0.45], [0.4, 0.45], [0.6, 0.45], [0.8, 0.45],
        [0.2, 0.6], [0.4, 0.6], [0.6, 0.6], [0.8, 0.6],
        [0.3, 0.75], [0.5, 0.75], [0.7, 0.75],
        [0.4, 0.9], [0.6, 0.9]
    ])
    
    # Assign positions to centers
    centers[:4] = corner_positions[:4]
    centers[4:10] = edge_positions[:6]
    centers[10:26] = hex_grid_positions[:16]
    
    # Initialize radii with a more aggressive guess
    max_initial_radius = 0.15
    radii = np.full(n, max_initial_radius)
    
    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    # Constraints for optimization
    def constraints(radii):
        n = len(radii)
        constraints = []
        
        # Non-negativity of radii
        constraints.append({'type': 'ineq', 'fun': lambda r: r})
        
        # Boundary constraints
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 - y - r})
        
        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.hypot(dx, dy)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})
        
        return constraints
    
    # Optimization: use SLSQP for better performance with complex constraints
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, np.inf) for _ in range(n)],
        constraints=constraints(radii),
        options={'ftol': 1e-8, 'maxiter': 1000, 'disp': False}
    )
    
    # Extract results
    radii = result.x
    sum_radii = np.sum(radii)
    centers = np.array(centers)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.hypot(centers[i, 0] - centers[j, 0], centers[i, 1] - centers[j, 1])
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
