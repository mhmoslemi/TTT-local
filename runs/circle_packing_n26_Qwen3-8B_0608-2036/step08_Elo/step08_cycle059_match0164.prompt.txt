import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate a more refined hybrid layout
    # Grid points
    grid_points = 6
    grid_x = np.linspace(0.1, 0.9, grid_points)
    grid_y = np.linspace(0.1, 0.9, grid_points)

    # Corners
    corner_positions = [[0.1, 0.1], [0.1, 0.9], [0.9, 0.1], [0.9, 0.9]]
    # Edges (center of each side)
    edge_positions = [[0.1, 0.5], [0.5, 0.1], [0.5, 0.9], [0.9, 0.5]]
    # Additional edge points for denser packing
    edge_positions += [[0.2, 0.5], [0.5, 0.2], [0.5, 0.8], [0.8, 0.5]]

    # Combine initial positions
    centers = np.zeros((n, 2))
    idx = 0

    for x in grid_x:
        for y in grid_y:
            if idx < n:
                centers[idx] = [x, y]
                idx += 1

    for x, y in corner_positions:
        if idx < n:
            centers[idx] = [x, y]
            idx += 1

    for x, y in edge_positions:
        if idx < n:
            centers[idx] = [x, y]
            idx += 1

    # Add remaining circles in a grid if any are missing
    while idx < n:
        x = (idx % grid_points) * (0.8 / (grid_points - 1)) + 0.1
        y = (idx // grid_points) * (0.8 / (grid_points - 1)) + 0.1
        centers[idx] = [x, y]
        idx += 1

    # Initialize radii
    max_dist = np.inf
    for i in range(n):
        for j in range(i+1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.hypot(dx, dy)
            if dist < max_dist:
                max_dist = dist

    initial_radius = max_dist / 2
    radii = np.ones(n) * initial_radius

    # Objective function: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints for optimization
    def constraints(radii):
        n = len(radii)
        constraints = []

        # Non-negativity of radii
        constraints.append({'type': 'ineq', 'fun': lambda r: r})

        # Boundary constraints
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 - y - r})

        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.hypot(dx, dy)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})

        return constraints

    # Optimization: use SLSQP
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, np.inf) for _ in range(n)],
        constraints=constraints(radii),
        options={'ftol': 1e-8, 'maxiter': 500}

    sum_radii = np.sum(radii)
