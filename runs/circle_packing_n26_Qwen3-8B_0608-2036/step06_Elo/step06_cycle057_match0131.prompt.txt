=== SYSTEM ===
You are an expert evaluator comparing two candidate programs for a scientific discovery search, determining which is the more promising direction to expand next.

=== USER ===
Goal: Discover a program whose output maximizes the metric 'sum of radii' (target 2.636). We search by repeatedly expanding promising programs sampled from a buffer.

Evaluation criteria:
- Likelihood the underlying approach can be pushed to a stronger solution
- Novelty / originality of the search strategy (not just parameter tweaks)
- Exploitation of problem structure
- Robustness, i.e. the chance it keeps producing valid solutions
- Headroom: how much further this direction can plausibly improve

Judge only the merit and headroom of each approach. Do not assume that the
already-measured score decides the comparison.

import numpy as np
import scipy.optimize as opt
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate a more flexible initial configuration
    def generate_initial_centers(n):
        centers = []
        col_count = int(np.ceil(n ** 0.5))
        row_count = int(np.ceil(n / col_count))
        spacing = 0.15
        for row in range(row_count):
            for col in range(col_count):
                x = col * spacing
                if row % 2 == 1:
                    x += spacing / 2
                y = row * spacing
                if y > 1 - spacing:
                    y = 1 - spacing
                if x > 1 - spacing:
                    x = 1 - spacing
                centers.append([x, y])
                if len(centers) >= n:
                    break
            if len(centers) >= n:
                break
        # Introduce some randomness to avoid local optima
        random.shuffle(centers)
        return np.array(centers)

    centers = generate_initial_centers(n)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Function to compute pairwise distances and generate constraints
    def compute_constraints(centers):
        constraints = []
        
        # Non-negativity constraints
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Boundary constraints (x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1)
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: 1 - x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: y - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: 1 - y - r[i]})
        
        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})
        
        return constraints

    # Initial guess for radii
    initial_radii = np.ones(n) * 0.05

    # Run the optimization
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=compute_constraints(centers),
        bounds=[(0, np.inf) for _ in range(n)],
        options={
            'ftol': 1e-10,
            'maxiter': 1500,  # Increased number of iterations
            'eps': 1e-8,
            'disp': False
        }
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial hexagonal packing parameters
    num_circles = 26
    # Estimate an initial radius for a hexagonal packing in a unit square
    initial_radius = 0.12
    initial_centers = []
    row_count = int(np.sqrt(num_circles)) + 1
    col_count = int(np.sqrt(num_circles)) + 1

    for i in range(row_count):
        for j in range(col_count):
            if i % 2 == 0:
                x = j * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            else:
                x = (j + 0.5) * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            if x < 1 + 1e-6 and y < 1 + 1e-6 and x > -1e-6 and y > -1e-6:
                initial_centers.append([x, y])
    
    initial_centers = np.array(initial_centers)
    initial_radii = np.full(len(initial_centers), initial_radius)

    # Function to compute the total sum of radii
    def objective_function(params):
        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Function to check constraints
    def constraints(params):
        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]
        # Constraint: centers within unit square
        for i in range(len(centers)):
            x, y = centers[i]
            if x < -1e-6 or x > 1 + 1e-6 or y < -1e-6 or y > 1 + 1e-6:
                return np.inf
        # Constraint: radii non-negative
        for r in radii:
            if r < -1e-6:
                return np.inf
        # Constraint: no overlaps
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    return np.inf
        return 0

    # Initial guess
    initial_guess = np.concatenate([initial_centers.reshape(-1), initial_radii])

    # Optimization
    result = opt.minimize(
        objective_function,
        initial_guess,
        method='L-BFGS-B',
        bounds=[(-1e-6, 1 + 1e-6) for _ in range(2 * len(initial_centers))] + [(0, 1 + 1e-6) for _ in range(len(initial_centers))],
        constraints={'type': 'ineq', 'fun': constraints}
    )

    centers = result.x[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
    radii = result.x[2 * len(initial_centers):]

    # Final validation
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
        return np.array([[0.5, 0.5]] * 26), np.full(26, 0.0), 0.0

    return centers, radii, np.sum(radii)
