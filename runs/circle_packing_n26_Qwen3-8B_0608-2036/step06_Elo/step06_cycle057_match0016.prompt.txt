=== SYSTEM ===
You are an expert evaluator comparing two candidate programs for a scientific discovery search, determining which is the more promising direction to expand next.

=== USER ===
Goal: Discover a program whose output maximizes the metric 'sum of radii' (target 2.636). We search by repeatedly expanding promising programs sampled from a buffer.

Evaluation criteria:
- Likelihood the underlying approach can be pushed to a stronger solution
- Novelty / originality of the search strategy (not just parameter tweaks)
- Exploitation of problem structure
- Robustness, i.e. the chance it keeps producing valid solutions
- Headroom: how much further this direction can plausibly improve

Judge only the merit and headroom of each approach. Do not assume that the
already-measured score decides the comparison.

import numpy as np
import scipy.optimize as opt
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate a more flexible initial configuration
    def generate_initial_centers(n):
        centers = []
        col_count = int(np.ceil(n ** 0.5))
        row_count = int(np.ceil(n / col_count))
        spacing = 0.15
        for row in range(row_count):
            for col in range(col_count):
                x = col * spacing
                if row % 2 == 1:
                    x += spacing / 2
                y = row * spacing
                if y > 1 - spacing:
                    y = 1 - spacing
                if x > 1 - spacing:
                    x = 1 - spacing
                centers.append([x, y])
                if len(centers) >= n:
                    break
            if len(centers) >= n:
                break
        # Introduce some randomness to avoid local optima
        random.shuffle(centers)
        return np.array(centers)

    centers = generate_initial_centers(n)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Function to compute pairwise distances and generate constraints
    def compute_constraints(centers):
        constraints = []
        
        # Non-negativity constraints
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Boundary constraints (x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1)
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: 1 - x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: y - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: 1 - y - r[i]})
        
        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})
        
        return constraints

    # Initial guess for radii
    initial_radii = np.ones(n) * 0.05

    # Run the optimization
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=compute_constraints(centers),
        bounds=[(0, np.inf) for _ in range(n)],
        options={
            'ftol': 1e-10,
            'maxiter': 1500,  # Increased number of iterations
            'eps': 1e-8,
            'disp': False
        }
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Step 1: Generate initial hexagonal grid
    def hexagonal_grid(n, unit_square=True):
        if not unit_square:
            return np.random.rand(n, 2)
        rows = int(np.sqrt(n))
        cols = int(np.ceil(n / rows))
        grid = np.zeros((n, 2))
        idx = 0
        for i in range(rows):
            for j in range(cols):
                x = j + 0.5 * (i % 2)
                y = i
                if i % 2 == 1:
                    x += 0.5
                if x > 1 - 1e-6 or y > 1 - 1e-6:
                    continue
                grid[idx] = [x, y]
                idx += 1
        return grid[:n]

    # Step 2: Use a basic hexagonal grid
    centers = hexagonal_grid(26)
    radii = np.ones(26) * 0.05

    # Step 3: Define the objective function (maximize sum of radii)
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Step 4: Define constraints
    def constraint_overlap(radii, centers):
        n = centers.shape[0]
        dists = cdist(centers, centers)
        for i in range(n):
            for j in range(i + 1, n):
                if dists[i, j] < radii[i] + radii[j] - 1e-6:
                    return -1  # Violation
        return 0

    def constraint_boundary(radii, centers):
        for i in range(len(centers)):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-6 or x + r > 1 + 1e-6 or y - r < -1e-6 or y + r > 1 + 1e-6:
                return -1  # Violation
        return 0

    def constraint_non_negative(radii):
        for r in radii:
            if r < 0:
                return -1
        return 0

    # Create a helper function to combine constraints
    def constraint_func(radii, centers):
        c1 = constraint_overlap(radii, centers)
        c2 = constraint_boundary(radii, centers)
        c3 = constraint_non_negative(radii)
        return np.array([c1, c2, c3])

    # Initial radii
    initial_radii = np.repeat(0.05, 26)

    # Define optimization bounds
    bounds = [(0, 0.1) for _ in range(26)]

    # Run optimization
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: -constraint_func(x, centers)}
        ]
    )

    # Extract optimal results
    optimal_radii = result.x
    optimal_centers = centers  # Centers are fixed in this approach

    # Validate the packing
    valid, message = validate_packing(optimal_centers, optimal_radii)
    if not valid:
        print(f"Validation error: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    sum_radii = np.sum(optimal_radii)
    return optimal_centers, optimal_radii, sum_radii
