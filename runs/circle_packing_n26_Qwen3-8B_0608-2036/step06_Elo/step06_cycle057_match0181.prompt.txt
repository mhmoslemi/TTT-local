import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Create a hexagonal grid in the unit square
    # Hexagonal grid parameters
    spacing = 0.1
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    # Generate hexagonal grid points (shifted for better packing)
    centers = np.zeros((n, 2))
    idx = 0

    for row in range(rows):
        for col in range(cols):
            if idx >= n:
                break
            if row % 2 == 1:
                x = 0.5 * (col + 1) / cols
            else:
                x = 0.5 * (col + 1) / cols
            y = 0.5 * (row + 1) / rows
            centers[idx] = [x, y]
            idx += 1

    # Calculate initial radius based on spacing
    max_radius = (1 - spacing * (cols + 1)) / 2
    radii = np.ones(n) * max_radius

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    def constraints(radii):
        n = len(radii)
        constraints = []

        # Non-negativity of radii
        constraints.append({'type': 'ineq', 'fun': lambda r: r})

        # All circles within the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 + 1e-12 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 + 1e-12 - y - r})

        # No overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})

        return constraints

    # Optimization with SLSQP
    # Use bounds and tighter tolerances for better convergence
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints(radii),
        bounds=[(0, np.inf) for _ in range(n)],
        options={'ftol': 1e-10, 'maxiter': 1000}
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as opt
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate a more flexible initial configuration
    def generate_initial_centers(n):
        centers = []
        col_count = int(np.ceil(n ** 0.5))
        row_count = int(np.ceil(n / col_count))
        spacing = 0.15
        for row in range(row_count):
            for col in range(col_count):
                x = col * spacing
                if row % 2 == 1:
                    x += spacing / 2
                y = row * spacing
                if y > 1 - spacing:
                    y = 1 - spacing
                if x > 1 - spacing:
                    x = 1 - spacing
                centers.append([x, y])
                if len(centers) >= n:
                    break
            if len(centers) >= n:
                break
        # Introduce some randomness to avoid local optima
        random.shuffle(centers)
        return np.array(centers)

    centers = generate_initial_centers(n)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Function to compute pairwise distances and generate constraints
    def compute_constraints(centers):
        constraints = []
        
        # Non-negativity constraints
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Boundary constraints (x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1)
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: 1 - x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: y - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: 1 - y - r[i]})
        
        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})
        
        return constraints

    # Initial guess for radii
    initial_radii = np.ones(n) * 0.05

    # Run the optimization
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=compute_constraints(centers),
        bounds=[(0, np.inf) for _ in range(n)],
        options={
            'ftol': 1e-10,
            'maxiter': 1500,  # Increased number of iterations
            'eps': 1e-8,
            'disp': False
        }
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii
