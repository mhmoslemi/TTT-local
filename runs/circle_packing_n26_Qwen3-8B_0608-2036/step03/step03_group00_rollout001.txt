import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Dynamically calculate the grid based on the number of circles
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    centers = np.zeros((n, 2))
    idx = 0
    for row in range(rows):
        for col in range(cols):
            if idx >= n:
                break
            x = col / (cols - 1) if cols > 1 else 0.5
            y = row / (rows - 1) if rows > 1 else 0.5
            centers[idx] = [x, y]
            idx += 1

    # Function to compute the sum of radii (negative for minimization)
    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    def constraints(radii):
        n = len(radii)

        constraints.append({'type': 'ineq', 'fun': lambda r: r})

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: y - r})

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
