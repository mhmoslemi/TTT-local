import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Create a hexagonal grid in the unit square
    # Hexagonal grid parameters
    spacing = 0.1
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    # Generate hexagonal grid points (shifted for better packing)
    centers = np.zeros((n, 2))
    idx = 0

        for col in range(cols):

                break

                x = 0.5 * (col + 1) / cols

    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    def constraints(radii):
        n = len(radii)
        constraints = []

        constraints.append({'type': 'ineq', 'fun': lambda r: r})

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
