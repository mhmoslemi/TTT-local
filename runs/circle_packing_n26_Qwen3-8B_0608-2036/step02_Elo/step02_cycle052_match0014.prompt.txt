from scipy.spatial.distance import cdist
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial placement using a hexagonal grid inside the unit square
    n = 26
    row_count = int(np.sqrt(n)) + 1
    col_count = int(np.sqrt(n)) + 1

    radii = np.ones(n)

        row = i // col_count
        col = i % col_count
        x = col * (1 + 1 / (col_count))
        y = row * (1 + np.sqrt(3) / (row_count))
        centers[i] = [x, y]

        centers[i] = np.clip(centers[i], 0, 1)

    # Calculate initial spacing for radii
    spacing = np.min(cdist(centers, centers))
    radii = np.full(n, spacing / 2)  # Initial radius based on spacing

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints: circles do not overlap, and stay within the square
    def constraints(radii, centers):
        constraints = []
        # Circle within the square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - 1e-12})
        # Non-overlapping circles
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: r[i] + r[j] - dist - 1e-12})
        return constraints

    # Initial radii with small perturbation
    initial_radii = radii.copy() + np.random.uniform(-0.01, 0.01, n)

    # Bounds for radii (non-negative)
    bounds = Bounds(0, 1)

    # Optimization
    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints(initial_radii, centers))
    radii = result.x

    # Final centers (no change, since we optimize radii only)
    # Revalidate the packing
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
        # If invalid, fallback to initial packing
        radii = np.full(n, 0.05)  # Fallback to smaller radii
        valid, msg = validate_packing(centers, radii)
        if not valid:
            print("Failed to validate, returning empty packing")
            return np.zeros((n, 2)), np.zeros(n), 0.0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Step 1: Generate initial hexagonal grid
    def hexagonal_grid(n, unit_square=True):
        if not unit_square:
            return np.random.rand(n, 2)
        rows = int(np.sqrt(n))
        cols = int(np.ceil(n / rows))
        grid = np.zeros((n, 2))
        idx = 0
        for i in range(rows):
            for j in range(cols):
                x = j + 0.5 * (i % 2)
                y = i
                if i % 2 == 1:
                    x += 0.5
                if x > 1 - 1e-6 or y > 1 - 1e-6:
                    continue
                grid[idx] = [x, y]
                idx += 1
        return grid[:n]

    # Step 2: Use a basic hexagonal grid
    centers = hexagonal_grid(26)
    radii = np.ones(26) * 0.05

    # Step 3: Define the objective function (maximize sum of radii)
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Step 4: Define constraints
    def constraint_overlap(radii, centers):
        n = centers.shape[0]
        dists = cdist(centers, centers)
        for i in range(n):
            for j in range(i + 1, n):
                if dists[i, j] < radii[i] + radii[j] - 1e-6:
                    return -1  # Violation
        return 0

    def constraint_boundary(radii, centers):
        for i in range(len(centers)):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-6 or x + r > 1 + 1e-6 or y - r < -1e-6 or y + r > 1 + 1e-6:
                return -1  # Violation
        return 0

    def constraint_non_negative(radii):
        for r in radii:
            if r < 0:
                return -1
        return 0

    # Create a helper function to combine constraints
    def constraint_func(radii, centers):
        c1 = constraint_overlap(radii, centers)
        c2 = constraint_boundary(radii, centers)
        c3 = constraint_non_negative(radii)
        return np.array([c1, c2, c3])

    # Initial radii
    initial_radii = np.repeat(0.05, 26)

    # Define optimization bounds
    bounds = [(0, 0.1) for _ in range(26)]

    # Run optimization
    result = minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: -constraint_func(x, centers)}
        ]
    )

    # Extract optimal results
    optimal_radii = result.x
    optimal_centers = centers  # Centers are fixed in this approach

    # Validate the packing
    valid, message = validate_packing(optimal_centers, optimal_radii)
    if not valid:
        print(f"Validation error: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    sum_radii = np.sum(optimal_radii)
    return optimal_centers, optimal_radii, sum_radii
