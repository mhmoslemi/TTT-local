import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial hexagonal packing parameters
    num_circles = 26
    # Estimate an initial radius for a hexagonal packing in a unit square
    initial_radius = 0.12
    initial_centers = []

    col_count = int(np.sqrt(num_circles)) + 1

    for i in range(row_count):

            if i % 2 == 0:
                x = j * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            else:
                x = (j + 0.5) * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            if x < 1 + 1e-6 and y < 1 + 1e-6 and x > -1e-6 and y > -1e-6:
                initial_centers.append([x, y])
    
    initial_centers = np.array(initial_centers)
    initial_radii = np.full(len(initial_centers), initial_radius)

    def objective_function(params):

        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]

        return 0
