```python
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define the number of circles
    n = 26

    # Initial guess for centers
    centers = np.random.rand(n, 2)
    radii = np.full(n, 0.05)  # Initial small radii

    # Objective function: maximize sum of radii
    def objective(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        return -np.sum(radii_flat)  # Minimize negative sum

    # Bounds for centers and radii
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        bounds.append((0.0, 1.0))  # radius

    # Constraints for circle boundaries
    def boundary_constraints(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        return np.array([centers_flat[i, 0] - radii_flat[i] + 1e-12 for i in range(n)] +
                        [centers_flat[i, 1] - radii_flat[i] + 1e-12 for i in range(n)] +
                        [1.0 - centers_flat[i, 0] - radii_flat[i] + 1e-12 for i in range(n)] +
                        [1.0 - centers_flat[i, 1] - radii_flat[i] + 1e-12 for i in range(n)])

    # Constraints for non-overlapping circles
    def non_overlap_constraints(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:]
        constraints = []
        for i in range(n):
            for j in range(i+1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append(dist - radii_flat[i] - radii_flat[j] + 1e-12)
        return np.array(constraints)

    # Combine constraints
    cons = [
        {'type': 'ineq', 'fun': boundary_constraints},
        {'type': 'ineq', 'fun': non_overlap_constraints}
    ]

    # Run optimization
    result = optimize.minimize(
        objective, 
        x0=np.concatenate([centers.flatten(), radii]),
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6
```

```python
def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
