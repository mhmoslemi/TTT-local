import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Generate a more optimized hexagonal grid with spacing between circles
    # Calculate the radius of the largest circle that can fit in a hexagonal grid
    # For a hexagonal packing, the spacing between centers is 2r
    # We aim for a grid that fits within the unit square [0,1]×[0,1]
    spacing = 0.1  # Initial spacing between centers (can be adjusted)

    # Determine the number of rows and columns in hexagonal grid
    rows = int(np.sqrt(n)) + 1
    cols = int(np.ceil(n / rows))
    
    # Calculate the maximum possible radius for the initial grid
    max_radius = (1 - spacing * (cols + 1)) / 2  # Adjust based on spacing
    
    # Create a better initial grid with hexagonal packing
    centers = np.zeros((n, 2))
    radii = np.ones(n) * max_radius  # Initial guess for radii
    idx = 0

    for row in range(rows):
        for col in range(cols):
            if idx >= n:
                break
            x = 0.5 * (1 - (col % 2)) + 0.5 * (col + 1) / (cols + 1)
            y = 0.5 * (1 - (row % 2)) + 0.5 * (row + 1) / (rows + 1)
            centers[idx] = [x, y]
            idx += 1

    # Function to compute the sum of radii (negative for minimization)
    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    def constraints(radii):
        n = len(radii)
        constraints = []

        # Non-negativity of radii
        constraints.append({'type': 'ineq', 'fun': lambda r: r})

        # All circles within the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 + 1e-12 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 + 1e-12 - y - r})

        # No overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                # We want dist >= radii[i] + radii[j]
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})

        return constraints

    # Optimization with SLSQP
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints(radii),
        bounds=[(0, np.inf) for _ in range(n)],
        options={'ftol': 1e-9, 'maxiter': 500}
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as opt

def run_packing():
    num_circles = 26
    initial_radius = 0.15  # Start with a larger radius to allow for optimization

    # Generate an initial grid-based packing with alternating row offsets
    initial_centers = []
    row_count = int(np.sqrt(num_circles)) + 1
    col_count = int(np.sqrt(num_circles)) + 1
    for i in range(row_count):
        for j in range(col_count):
            if i % 2 == 0:
                x = j * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            else:
                x = (j + 0.5) * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            # Clamp to the unit square
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            if x < 1 - 1e-6 and y < 1 - 1e-6 and x > 1e-6 and y > 1e-6:
                initial_centers.append([x, y])

    initial_centers = np.array(initial_centers)
    initial_radii = np.full(len(initial_centers), initial_radius)

    # Function to compute the total sum of radii (maximize sum)
    def objective_function(params):
        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Function to check constraints
    def constraints(params):
        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]
        # Constraint: centers within unit square
        for i in range(len(centers)):
            x, y = centers[i]
            if x < 0 or x > 1 or y < 0 or y > 1:
                return np.inf
        # Constraint: radii non-negative
        for r in radii:
            if r < 0:
                return np.inf
        # Constraint: no overlaps
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        return 0

    # Initial guess
    initial_guess = np.concatenate([initial_centers.reshape(-1), initial_radii])

    # Bounds for positions and radii
    bounds = [(-1e-6, 1 + 1e-6) for _ in range(2 * len(initial_centers))] + [(0, 1 + 1e-6) for _ in range(len(initial_centers))]

    # Optimization using SLSQP method
    result = opt.minimize(
        objective_function,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints}
    )

    # Extract the final centers and radii
    centers = result.x[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
    radii = result.x[2 * len(initial_centers):]

    # Final validation
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
        return np.array([[0.5, 0.5]] * 26), np.full(26, 0.0), 0.0

    return centers, radii, np.sum(radii)
