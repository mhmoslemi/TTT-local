        # x is a flattened array: [c1x, c1y, c2x, c2y, ..., c26x, c26y, r1, r2, ..., r26]
        # Split into centers and radii
        n = 26

        n = 26
        centers = x[:n*2].reshape(n, 2)

        dists = cdist(centers, centers)
        for i in range(n):
            for j in range(i+1, n):
                if dists[i, j] < radii[i] + radii[j] - 1e-6:
                    return -1
        return 0

    def constraint_boundary(x):
        n = 26
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:].reshape(n)
        for i in range(n):
            x_center, y_center = centers[i]
            r = radii[i]
            if x_center - r < -1e-6 or x_center + r > 1 + 1e-6 or \
               y_center - r < -1e-6 or y_center + r > 1 + 1e-6:

        return 0

    def constraint_non_negative(x):
        n = 26
        radii = x[n*2:].reshape(n)
        for r in radii:
            if r < -1e-6:
                return -1
        return 0

    # Step 4: Combine constraints
    def constraint_func(x):
        c1 = constraint_overlap(x)
        c2 = constraint_boundary(x)
        c3 = constraint_non_negative(x)
        return np.array([c1, c2, c3])

    # Step 5: Initialize the configuration with random centers and initial radii

        initial_x,
        method='SLSQP',
        bounds=bounds,
        constraints=[

            'maxiter': 200
        }

    if not valid:

        return np.zeros((26, 2)), np.zeros(26), 0.0
