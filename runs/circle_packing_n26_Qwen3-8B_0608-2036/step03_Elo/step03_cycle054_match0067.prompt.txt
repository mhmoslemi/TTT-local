import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n_circles = 26
    # Generate random initial positions and radii around the square
    centers = np.random.rand(n_circles, 2)
    radii = np.random.rand(n_circles) * 0.1  # Initial random radii
    
    # Flatten centers and radii for optimization
    initial_params = np.concatenate([centers.reshape(-1), radii])
    
    # Set bounds for centers [0, 1]^2
    center_bounds = [(0, 1) for _ in range(2 * n_circles)]
    # Set bounds for radii [0, 0.5]
    radius_bounds = [(0, 0.5) for _ in range(n_circles)]
    # Combine bounds
    bounds = center_bounds + radius_bounds

    # Objective function to maximize sum of radii
    def objective_function(params):
        centers = params[:2 * n_circles].reshape(n_circles, 2)
        radii = params[2 * n_circles:]
        
        # Check for negative radii
        if np.any(radii < 0):
            return np.inf  # Penalize negative radii

        # Check for circles outside the unit square
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            if not (0 - 1e-10 <= x - r <= 1 + 1e-10 and 0 - 1e-10 <= y - r <= 1 + 1e-10):
                return np.inf  # Penalize circles outside the square

        # Compute pairwise distances
        distances = cdist(centers, centers)
        # Check for overlaps
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                if distances[i, j] < radii[i] + radii[j] - 1e-10:
                    return np.inf  # Penalize overlapping circles

        # Return negative of sum of radii to maximize it
        return -np.sum(radii)

    # Perform optimization using SLSQP
    result = minimize(
        objective_function,
        initial_params,
        method='SLSQP',
        bounds=bounds,
        options={'ftol': 1e-12, 'maxiter': 1000}
    )

    # Extract the optimized parameters
    optimized_params = result.x
    centers = optimized_params[:2 * n_circles].reshape(n_circles, 2)
    radii = optimized_params[2 * n_circles:]

    # Check if optimization was successful
    if result.success:
        is_valid, message = validate_packing(centers, radii)
        if is_valid:
            sum_radii = np.sum(radii)
            return centers, radii, sum_radii
        else:
            print(f"Optimization failed: {message}")
            return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
    else:
        print(f"Optimization failed with message: {result.message}")
        return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Step 1: Generate initial random centers within the unit square
    def initial_centers(n):
        # Generate n random points within [0,1]x[0,1]
        return np.random.rand(n, 2)

    # Step 2: Define the objective function to maximize the sum of radii
    def objective(x):
        # x is a flattened array: [c1x, c1y, c2x, c2y, ..., c26x, c26y, r1, r2, ..., r26]
        # Split into centers and radii
        n = 26
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:].reshape(n)
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Step 3: Define the constraints
    def constraint_overlap(x):
        n = 26
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:].reshape(n)
        dists = cdist(centers, centers)
        for i in range(n):
            for j in range(i+1, n):
                if dists[i, j] < radii[i] + radii[j] - 1e-6:
                    return -1
        return 0

    def constraint_boundary(x):
        n = 26
        centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:].reshape(n)
        for i in range(n):
            x_center, y_center = centers[i]
            r = radii[i]
            if x_center - r < -1e-6 or x_center + r > 1 + 1e-6 or \
               y_center - r < -1e-6 or y_center + r > 1 + 1e-6:
                return -1
        return 0

    def constraint_non_negative(x):
        n = 26
        radii = x[n*2:].reshape(n)
        for r in radii:
            if r < -1e-6:
                return -1
        return 0

    # Step 4: Combine constraints
    def constraint_func(x):
        c1 = constraint_overlap(x)
        c2 = constraint_boundary(x)
        c3 = constraint_non_negative(x)
        return np.array([c1, c2, c3])

    # Step 5: Initialize the configuration with random centers and initial radii
    n = 26
    initial_centers_flat = initial_centers(n).flatten()
    initial_radii = np.repeat(0.05, n)
    initial_x = np.concatenate([initial_centers_flat, initial_radii])

    # Step 6: Set bounds for centers and radii
    # Centers must be between 0 and 1
    # Radii must be between 0 and some maximum value
    centers_bounds = [(0, 1) for _ in range(n*2)]
    radii_bounds = [(0, 0.2) for _ in range(n)]
    bounds = centers_bounds + radii_bounds

    # Step 7: Perform optimization
    result = minimize(
        objective,
        initial_x,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_func}
        ],
        options={
            'disp': False,
            'ftol': 1e-8,
            'maxiter': 200
        }
    )

    # Step 8: Extract the optimized centers and radii
    optimal_centers = result.x[:n*2].reshape(n, 2)
    optimal_radii = result.x[n*2:].reshape(n)
    sum_radii = np.sum(optimal_radii)

    # Step 9: Validate the packing
    valid, message = validate_packing(optimal_centers, optimal_radii)
    if not valid:
        print(f"Validation error: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return optimal_centers, optimal_radii, sum_radii
