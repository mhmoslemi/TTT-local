import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_hexagonal_packing(n_circles):
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))
    
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            centers.append([x, y])
    
    while len(centers) > n_circles:
        centers.pop()
    
    return np.array(centers)

def objective_function(params, n_circles):
    centers = params[:2 * n_circles].reshape(n_circles, 2)
    radii = params[2 * n_circles:]
    
    # Ensure all radii are non-negative
    if np.any(radii < 0):
        return np.inf
    
    # Ensure all circles are within the unit square
    for i in range(n_circles):
        x, y = centers[i]
        r = radii[i]
        if not (0 - 1e-12 <= x - r <= 1 + 1e-12 and 0 - 1e-12 <= y - r <= 1 + 1e-12):
            return np.inf
    
    # Compute sum of radii
    sum_radii = np.sum(radii)
    
    # Compute pairwise distances
    distances = cdist(centers, centers)
    
    # Check for overlaps
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            if distances[i, j] < radii[i] + radii[j] - 1e-12:
                return np.inf
    
    return -sum_radii  # Minimize -sum_radii is equivalent to maximizing sum_radii

def run_packing():
    n_circles = 26
    initial_centers = initial_hexagonal_packing(n_circles)
    
    # Flatten centers and radii for optimization
    initial_params = np.concatenate([initial_centers.reshape(-1), np.ones(n_circles)])
    
    # Set bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(n_circles)]
    
    # Set bounds for centers (within [0, 1]^2)
    center_bounds = [(0, 1) for _ in range(2 * n_circles)]
    
    # Combine bounds for all parameters
    all_bounds = center_bounds + bounds
    
    # Perform optimization
    result = minimize(
        objective_function,
        initial_params,
        args=(n_circles,),
        method='L-BFGS-B',
        bounds=all_bounds,
        options={'ftol': 1e-12, 'maxiter': 1000}
    )
    
    optimized_params = result.x
    centers = optimized_params[:2 * n_circles].reshape(n_circles, 2)
    radii = optimized_params[2 * n_circles:]
    
    # Check if the optimization was successful
    if result.success:
        is_valid, message = validate_packing(centers, radii)
        if is_valid:
            sum_radii = np.sum(radii)
            return centers, radii, sum_radii
        else:
            print(f"Optimization failed: {message}")
            return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
    else:
        print(f"Optimization failed with message: {result.message}")
        return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
