import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = np.zeros((n, 2))
    radii = np.ones(n) * 0.05  # Initial guess for radii

    # Initial placement: hexagonal close packing with some randomness for better distribution
    for i in range(n):
        row = i // 5
        col = i % 5
        if row % 2 == 1:
            col += 1  # Offset for odd rows
        if col >= 5:
            col = 0
            row += 1
        # Adjust for corners and edges
        x = 0.5 * (1 - (col % 2)) + 0.5 * (col + 1) / (5 + 1)
        y = 0.5 * (1 - (row % 2)) + 0.5 * (row + 1) / (5 + 1)
        centers[i] = [x, y]

    # Function to compute the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints:
    # 1. Radii must be nonnegative
    # 2. Circles must be within the unit square
    # 3. Circles must not overlap

    # Define bounds and constraints
    bounds = [(0, np.inf) for _ in range(n)]
    constraints = []

    # Constraint 2: All circles must be within the unit square
    for i in range(n):
        def constraint_inside(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return x - r + 1e-12  # x - r >= -1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_inside})

        def constraint_inside_y(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return y - r + 1e-12  # y - r >= -1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_inside_y})

        def constraint_outside_x(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return 1 + 1e-12 - x - r  # x + r <= 1 + 1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_outside_x})

        def constraint_outside_y(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return 1 + 1e-12 - y - r  # y + r <= 1 + 1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_outside_y})

    # Constraint 3: No overlap between circles
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            def constraint_no_overlap(radii, idx=i, jdx=j, dist=dist):
                r_i = radii[idx]
                r_j = radii[jdx]
                return dist - r_i - r_j + 1e-12  # dist >= r_i + r_j - 1e-12
            constraints.append({'type': 'ineq', 'fun': constraint_no_overlap})

    # Optimization: Use SLSQP for constrained optimization
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as optimize

def run_packing():
    num_circles = 26
    radius_initial = 0.1  # Initial guess for radii
    radius_guess = np.ones(num_circles) * radius_initial

    def hex_grid_positions(n):
        # Generate hexagonal grid points inside [0,1]^2
        positions = []
        for i in range(n):
            r = i / (n - 1)
            x = r * (1 - 4 * r * r)  # Adjust for hexagonal spacing
            y = 0.5
            positions.append([x, y])
        return np.array(positions)

    def hex_grid_radii(n):
        # Initial radius for each circle based on grid spacing
        radii = np.zeros(n)
        for i in range(n):
            r = i / (n - 1)
            radii[i] = 0.1  # Simple initial guess
        return radii

    # Initial guess for centers and radii
    centers = hex_grid_positions(num_circles)
    radii = hex_grid_radii(num_circles)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_non_overlap(centers, radii):
        def cons(x):
            # Flatten radii for use in optimization
            x_radii = x[1:]
            x_centers = x[:2*num_circles].reshape(num_circles, 2)
            # Check non-overlapping constraint
            for i in range(num_circles):
                for j in range(i + 1, num_circles):
                    dx = x_centers[i, 0] - x_centers[j, 0]
                    dy = x_centers[i, 1] - x_centers[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    if dist < x_radii[i] + x_radii[j] - 1e-8:
                        return 1.0  # Constraint violated
            return 0.0  # Constraint satisfied

        return {'type': 'ineq', 'fun': cons}

    def constraint_boundary(centers, radii):
        def cons(x):
            # Flatten radii for use in optimization
            x_radii = x[1:]
            x_centers = x[:2*num_circles].reshape(num_circles, 2)
            # Check boundary constraint
            for i in range(num_circles):
                x, y = x_centers[i]
                r = x_radii[i]
                if (x - r < -1e-8 or x + r > 1 + 1e-8 or
                    y - r < -1e-8 or y + r > 1 + 1e-8):
                    return 1.0  # Constraint violated
            return 0.0  # Constraint satisfied

        return {'type': 'ineq', 'fun': cons}

    # Flatten the centers and radii for optimization
    x0 = np.concatenate([centers.flatten(), radius_guess])

    # Define optimization problem
    cons = [constraint_non_overlap(centers, radii), constraint_boundary(centers, radii)]
    bounds = [(0, 1) for _ in range(2*num_circles)] + [(0, 1) for _ in range(num_circles)]

    result = optimize.minimize(
        fun=objective,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Extract results
    centers_result = result.x[:2*num_circles].reshape(num_circles, 2)
    radii_result = result.x[2*num_circles:]
    sum_radii_result = np.sum(radii_result)

    # Validate the packing (optional)
    is_valid, message = validate_packing(centers_result, radii_result)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers_result, radii_result, sum_radii_result
