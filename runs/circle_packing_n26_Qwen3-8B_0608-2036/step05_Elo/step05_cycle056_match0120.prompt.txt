    def generate_initial_centers(n):

        spacing = 0.15

                if len(centers) >= n:
                    break

                break
        return np.array(centers)

    centers = generate_initial_centers(n)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Function to compute pairwise distances and generate constraints
    def compute_constraints(centers):
        constraints = []
        
        # Non-negativity constraints
        for i in range(n):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r[i]})
        
        # Boundary constraints (x - r >= 0, x + r <= 1, y - r >= 0, y + r <= 1)
        for i in range(n):
            x, y = centers[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x: 1 - x - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: y - r[i]})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, y=y: 1 - y - r[i]})

            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})
        
        return constraints

    # Initial guess for radii
    initial_radii = np.ones(n) * 0.05

    # Run the optimization
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        constraints=compute_constraints(centers),

    sum_radii = np.sum(radii)

def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

            if dist < radii[i] + radii[j] - 1e-12:

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n_circles = 26
    # Generate random initial positions and radii around the square
    centers = np.random.rand(n_circles, 2)
    radii = np.random.rand(n_circles) * 0.1  # Initial random radii
    
    # Flatten centers and radii for optimization
    initial_params = np.concatenate([centers.reshape(-1), radii])
    
    # Set bounds for centers [0, 1]^2
    center_bounds = [(0, 1) for _ in range(2 * n_circles)]
    # Set bounds for radii [0, 0.5]
    radius_bounds = [(0, 0.5) for _ in range(n_circles)]
    # Combine bounds
    bounds = center_bounds + radius_bounds

    # Objective function to maximize sum of radii
    def objective_function(params):
        centers = params[:2 * n_circles].reshape(n_circles, 2)
        radii = params[2 * n_circles:]
        
        # Check for negative radii
        if np.any(radii < 0):
            return np.inf  # Penalize negative radii

        # Check for circles outside the unit square
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            if not (0 - 1e-10 <= x - r <= 1 + 1e-10 and 0 - 1e-10 <= y - r <= 1 + 1e-10):
                return np.inf  # Penalize circles outside the square

        # Compute pairwise distances
        distances = cdist(centers, centers)
        # Check for overlaps
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                if distances[i, j] < radii[i] + radii[j] - 1e-10:
                    return np.inf  # Penalize overlapping circles

        # Return negative of sum of radii to maximize it
        return -np.sum(radii)

    # Perform optimization using SLSQP
    result = minimize(
        objective_function,
        initial_params,
        method='SLSQP',
        bounds=bounds,
        options={'ftol': 1e-12, 'maxiter': 1000}
    )

    # Extract the optimized parameters
    optimized_params = result.x
    centers = optimized_params[:2 * n_circles].reshape(n_circles, 2)
    radii = optimized_params[2 * n_circles:]

    # Check if optimization was successful
    if result.success:
        is_valid, message = validate_packing(centers, radii)
        if is_valid:
            sum_radii = np.sum(radii)
            return centers, radii, sum_radii
        else:
            print(f"Optimization failed: {message}")
            return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
    else:
        print(f"Optimization failed with message: {result.message}")
        return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
