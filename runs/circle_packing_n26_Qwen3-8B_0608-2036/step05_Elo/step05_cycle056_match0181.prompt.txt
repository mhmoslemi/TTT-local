from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    n_circles = 26
    # Generate random initial positions and radii around the square

    radii = np.random.rand(n_circles) * 0.1  # Initial random radii

    initial_params = np.concatenate([centers.reshape(-1), radii])

    # Set bounds for radii [0, 0.5]

    # Combine bounds
    bounds = center_bounds + radius_bounds

    # Objective function to maximize sum of radii
    def objective_function(params):
        centers = params[:2 * n_circles].reshape(n_circles, 2)
        radii = params[2 * n_circles:]

        if np.any(radii < 0):
            return np.inf  # Penalize negative radii

        # Check for circles outside the unit square
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            if not (0 - 1e-10 <= x - r <= 1 + 1e-10 and 0 - 1e-10 <= y - r <= 1 + 1e-10):
                return np.inf  # Penalize circles outside the square

        # Compute pairwise distances
        distances = cdist(centers, centers)
        # Check for overlaps
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                if distances[i, j] < radii[i] + radii[j] - 1e-10:
                    return np.inf  # Penalize overlapping circles

        # Return negative of sum of radii to maximize it
        return -np.sum(radii)

        initial_params,

        bounds=bounds,
        options={'ftol': 1e-12, 'maxiter': 1000}

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

    radii[0] = 0.1

        row = i // 5
        col = i % 5

        x = 0.1 + col * dx
        y = 0.1 + row * dy
        if row % 2 == 1:
            x += dx / 2

        centers[i] = [x, y]
        radii[i] = radius_guess

    # Refine the packing with optimization
    def objective(radii, centers):
        return -np.sum(radii)  # Objective is to minimize negative sum (maximize sum)

    def constraints(radii, centers):
        n = len(radii)
        constraints = []
        # Boundary constraint: ensure circles are within the square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r >= -1e-12) & (x + r <= 1 + 1e-12) &
                               (y - r >= -1e-12) & (y + r <= 1 + 1e-12))
        # Non-overlapping constraint
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

    # Optimization
    bounds = Bounds(0, 1.0)
    result = optimize.minimize(
        objective,
        radii,
        args=(centers,),
        bounds=bounds,
        method='L-BFGS-B',
        options={'ftol': 1e-8, 'maxiter': 100}
    )

    # Adjust centers and radii based on optimization
    optimized_radii = result.x
    optimized_centers = centers.copy()

    # Refine centers using local search to fill gaps
    def refine_centers(centers, radii):
        for i in range(len(radii)):
            for j in range(len(radii)):
                if i == j:

                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:

                    diff = np.array([dx, dy]) / dist
                    centers[i] += diff * (radii[i] + radii[j] - dist) * 0.1

    for _ in range(5):
        optimized_centers = refine_centers(optimized_centers, optimized_radii)

    # Validate packing
    is_valid, msg = validate_packing(optimized_centers, optimized_radii)

        return np.zeros((26, 2)), np.zeros(26), 0.0

    # Calculate total sum of radii
    sum_radii = np.sum(optimized_radii)

    return optimized_centers, optimized_radii, sum_radii
