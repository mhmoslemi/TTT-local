from scipy.spatial.distance import cdist

def run_packing():

    def generate_hexagonal_grid(n, unit_square=True):

        cols = int(np.ceil(n / rows))
        grid = np.zeros((n, 2))
        idx = 0
        for i in range(rows):
            for j in range(cols):
                if i % 2 == 0:
                    x = j + 0.5
                else:
                    x = j + 0.5
                y = i
                if i % 2 == 1:
                    x += 0.5
                # Scale and shift to unit square
                x = x / (cols - 1)
                y = y / (rows - 1)
                if x < 1e-6 or x > 1 - 1e-6 or y < 1e-6 or y > 1 - 1e-6:
                    continue
                grid[idx] = [x, y]
                idx += 1
        return grid[:n]

    # Step 2: Generate initial centers and radii

        dists = cdist(centers, centers)
        for i in range(n):

                if dists[i, j] < radii[i] + radii[j] - 1e-8:
                    return -1
        return 0

    def constraint_boundary(radii, centers):
        for i in range(len(centers)):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-8 or x + r > 1 + 1e-8 or y - r < -1e-8 or y + r > 1 + 1e-8:
                return -1
        return 0

    def constraint_non_negative(radii):
        for r in radii:
            if r < 0:
                return -1
        return 0

    def constraint_func(radii, centers):
        c1 = constraint_overlap(radii, centers)
        c2 = constraint_boundary(radii, centers)
        c3 = constraint_non_negative(radii)
        return np.array([c1, c2, c3])
