import numpy as np
from scipy.optimize import minimize, Bounds
from scipy.spatial.distance import cdist
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial placement using a hexagonal grid inside the unit square
    n = 26
    row_count = int(np.sqrt(n)) + 1
    col_count = int(np.sqrt(n)) + 1
    
    # Start with a hexagonal grid
    centers = np.zeros((n, 2))
    radii = np.ones(n)
    
    for i in range(n):
        row = i // col_count
        col = i % col_count
        x = col * (1 + 1 / (col_count))
        y = row * (1 + np.sqrt(3) / (row_count))
        centers[i] = [x, y]
        # Clip to the unit square
        centers[i] = np.clip(centers[i], 0, 1)
    
    # Initial guess for radii: scale based on spacing
    # Calculate initial spacing for radii
    spacing = np.min(cdist(centers, centers))
    radii = np.full(n, spacing / 2)  # Initial radius based on spacing

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints: circles do not overlap, and stay within the square
    def constraints(radii, centers):
        constraints = []
        # Circle within the square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - 1e-12})
        # Non-overlapping circles
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: r[i] + r[j] - dist - 1e-12})
        return constraints

    # Initial radii with small perturbation
    initial_radii = radii.copy() + np.random.uniform(-0.01, 0.01, n)

    # Bounds for radii (non-negative)
    bounds = Bounds(0, 1)

    # Optimization
    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints(initial_radii, centers))
    radii = result.x

    # Final centers (no change, since we optimize radii only)
    # Revalidate the packing
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
        # If invalid, fallback to initial packing
        radii = np.full(n, 0.05)  # Fallback to smaller radii
        valid, msg = validate_packing(centers, radii)
        if not valid:
            print("Failed to validate, returning empty packing")
            return np.zeros((n, 2)), np.zeros(n), 0.0

    sum_radii = np.sum(radii)
    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import scipy.optimize as opt

def run_packing():
    num_circles = 26
    # Initialize with a hexagonal packing that fits within the square
    initial_radius = 0.1
    initial_centers = []
    row_count = int(np.sqrt(num_circles)) + 1
    col_count = int(np.sqrt(num_circles)) + 1

    for i in range(row_count):

            if i % 2 == 0:
                x = j * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            else:
                x = (j + 0.5) * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            if x < 1 + 1e-6 and y < 1 + 1e-6 and x > -1e-6 and y > -1e-6:
                initial_centers.append([x, y])
    
    initial_centers = np.array(initial_centers)
    initial_radii = np.full(len(initial_centers), initial_radius)

    # Function to compute the total sum of radii
    def objective_function(params):

        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]
