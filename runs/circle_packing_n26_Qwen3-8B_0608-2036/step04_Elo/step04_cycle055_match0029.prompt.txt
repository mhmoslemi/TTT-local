def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Generate a more optimized hexagonal grid with spacing between circles
    # Calculate the radius of the largest circle that can fit in a hexagonal grid
    # For a hexagonal packing, the spacing between centers is 2r
    # We aim for a grid that fits within the unit square [0,1]×[0,1]
    spacing = 0.1  # Initial spacing between centers (can be adjusted)

    # Determine the number of rows and columns in hexagonal grid
    rows = int(np.sqrt(n)) + 1
    cols = int(np.ceil(n / rows))
    
    # Calculate the maximum possible radius for the initial grid
    max_radius = (1 - spacing * (cols + 1)) / 2  # Adjust based on spacing

    radii = np.ones(n) * max_radius  # Initial guess for radii
    idx = 0

    for row in range(rows):
        for col in range(cols):
            if idx >= n:
                break
            x = 0.5 * (1 - (col % 2)) + 0.5 * (col + 1) / (cols + 1)
            y = 0.5 * (1 - (row % 2)) + 0.5 * (row + 1) / (rows + 1)
            centers[idx] = [x, y]
            idx += 1

    # Function to compute the sum of radii (negative for minimization)
    def objective(radii):
        return -np.sum(radii)

    # Define constraints
    def constraints(radii):
        n = len(radii)
        constraints = []

        # Non-negativity of radii
        constraints.append({'type': 'ineq', 'fun': lambda r: r})

        # All circles within the unit square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 + 1e-12 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, x=x, y=y: 1 + 1e-12 - y - r})

        # No overlap between circles
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                # We want dist >= radii[i] + radii[j]
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})

        return constraints

    # Optimization with SLSQP
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints(radii),
        bounds=[(0, np.inf) for _ in range(n)],
        options={'ftol': 1e-9, 'maxiter': 500}
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize, Bounds
from scipy.spatial.distance import cdist
import warnings

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial placement using a hexagonal grid inside the unit square
    n = 26
    row_count = int(np.sqrt(n)) + 1
    col_count = int(np.sqrt(n)) + 1
    
    # Start with a hexagonal grid
    centers = np.zeros((n, 2))
    radii = np.ones(n)
    
    for i in range(n):
        row = i // col_count
        col = i % col_count
        x = col * (1 + 1 / (col_count))
        y = row * (1 + np.sqrt(3) / (row_count))
        centers[i] = [x, y]
        # Clip to the unit square
        centers[i] = np.clip(centers[i], 0, 1)
    
    # Initial guess for radii: scale based on spacing
    # Calculate initial spacing for radii
    spacing = np.min(cdist(centers, centers))
    radii = np.full(n, spacing / 2)  # Initial radius based on spacing

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints: circles do not overlap, and stay within the square
    def constraints(radii, centers):
        constraints = []
        # Circle within the square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: r - 1e-12})
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i: 1 - r - 1e-12})
        # Non-overlapping circles
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j: r[i] + r[j] - dist - 1e-12})
        return constraints

    # Initial radii with small perturbation
    initial_radii = radii.copy() + np.random.uniform(-0.01, 0.01, n)

    # Bounds for radii (non-negative)
    bounds = Bounds(0, 1)

    # Optimization
    result = minimize(objective, initial_radii, method='SLSQP', bounds=bounds, constraints=constraints(initial_radii, centers))
    radii = result.x

    # Final centers (no change, since we optimize radii only)
    # Revalidate the packing
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
        # If invalid, fallback to initial packing
        radii = np.full(n, 0.05)  # Fallback to smaller radii
        valid, msg = validate_packing(centers, radii)
        if not valid:
            print("Failed to validate, returning empty packing")
            return np.zeros((n, 2)), np.zeros(n), 0.0

    sum_radii = np.sum(radii)
    return centers, radii, sum_radii
