import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial hexagonal packing parameters
    num_circles = 26
    # Estimate an initial radius for a hexagonal packing in a unit square
    initial_radius = 0.12
    initial_centers = []
    row_count = int(np.sqrt(num_circles)) + 1
    col_count = int(np.sqrt(num_circles)) + 1

    for i in range(row_count):
        for j in range(col_count):
            if i % 2 == 0:
                x = j * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            else:
                x = (j + 0.5) * (2 * initial_radius)
                y = i * (np.sqrt(3) * initial_radius)
            if x < 1 + 1e-6 and y < 1 + 1e-6 and x > -1e-6 and y > -1e-6:
                initial_centers.append([x, y])
    
    initial_centers = np.array(initial_centers)
    initial_radii = np.full(len(initial_centers), initial_radius)

    # Function to compute the total sum of radii
    def objective_function(params):
        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Function to check constraints
    def constraints(params):
        centers = params[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
        radii = params[2 * len(initial_centers):]
        # Constraint: centers within unit square
        for i in range(len(centers)):
            x, y = centers[i]
            if x < -1e-6 or x > 1 + 1e-6 or y < -1e-6 or y > 1 + 1e-6:
                return np.inf
        # Constraint: radii non-negative
        for r in radii:
            if r < -1e-6:
                return np.inf
        # Constraint: no overlaps
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-6:
                    return np.inf
        return 0

    # Initial guess
    initial_guess = np.concatenate([initial_centers.reshape(-1), initial_radii])

    # Optimization
    result = opt.minimize(
        objective_function,
        initial_guess,
        method='L-BFGS-B',
        bounds=[(-1e-6, 1 + 1e-6) for _ in range(2 * len(initial_centers))] + [(0, 1 + 1e-6) for _ in range(len(initial_centers))],
        constraints={'type': 'ineq', 'fun': constraints}
    )

    centers = result.x[:2 * len(initial_centers)].reshape((len(initial_centers), 2))
    radii = result.x[2 * len(initial_centers):]

    # Final validation
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print(f"Validation failed: {msg}")
        return np.array([[0.5, 0.5]] * 26), np.full(26, 0.0), 0.0

    return centers, radii, np.sum(radii)
