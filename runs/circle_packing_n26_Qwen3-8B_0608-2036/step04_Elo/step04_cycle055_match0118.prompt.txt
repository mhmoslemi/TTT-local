=== USER ===
Goal: Discover a program whose output maximizes the metric 'sum of radii' (target 2.636). We search by repeatedly expanding promising programs sampled from a buffer.

Evaluation criteria:
- Likelihood the underlying approach can be pushed to a stronger solution
- Novelty / originality of the search strategy (not just parameter tweaks)
- Exploitation of problem structure
- Robustness, i.e. the chance it keeps producing valid solutions
- Headroom: how much further this direction can plausibly improve

Judge only the merit and headroom of each approach. Do not assume that the
already-measured score decides the comparison.

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
from scipy.optimize import Bounds

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_compact_packing(n_circles):
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))
    
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            centers.append([x, y])
    
    while len(centers) > n_circles:
        centers.pop()
    
    return np.array(centers)

def objective_function(params, n_circles):
    centers = params[:2 * n_circles].reshape(n_circles, 2)
    radii = params[2 * n_circles:]
    
    # Ensure all radii are non-negative
    if np.any(radii < 0):
        return np.inf
    
    # Ensure all circles are within the unit square
    for i in range(n_circles):
        x, y = centers[i]
        r = radii[i]
        if not (0 - 1e-12 <= x - r <= 1 + 1e-12 and 0 - 1e-12 <= y - r <= 1 + 1e-12):
            return np.inf
    
    # Compute sum of radii
    sum_radii = np.sum(radii)
    
    # Compute pairwise distances
    distances = cdist(centers, centers)
    
    # Check for overlaps
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            if distances[i, j] < radii[i] + radii[j] - 1e-12:
                return np.inf
    
    return -sum_radii  # Minimize -sum_radii is equivalent to maximizing sum_radii

def run_packing():
    n_circles = 26
    initial_centers = initial_compact_packing(n_circles)
    
    # Flatten centers and radii for optimization
    initial_params = np.concatenate([initial_centers.reshape(-1), np.ones(n_circles)])
    
    # Set bounds for radii (non-negative)
    bounds = [(0, 1) for _ in range(n_circles)]
    
    # Set bounds for centers (within [0, 1]^2)
    center_bounds = [(0, 1) for _ in range(2 * n_circles)]
    
    # Combine bounds for all parameters
    all_bounds = center_bounds + bounds
    
    # Set up constraints for optimization
    constraints = []

    # Perform optimization
    result = minimize(
        objective_function,
        initial_params,
        args=(n_circles,),
        method='SLSQP',
        bounds=all_bounds,
        constraints=constraints,
        options={'ftol': 1e-12, 'maxiter': 2000}
    )
    
    optimized_params = result.x
    centers = optimized_params[:2 * n_circles].reshape(n_circles, 2)
    radii = optimized_params[2 * n_circles:]
    
    # Check if the optimization was successful
    if result.success:
        is_valid, message = validate_packing(centers, radii)
        if is_valid:
            sum_radii = np.sum(radii)
            return centers, radii, sum_radii
        else:
            print(f"Optimization failed: {message}")
            return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0
    else:
        print(f"Optimization failed with message: {result.message}")
        return np.zeros((n_circles, 2)), np.zeros(n_circles), 0.0

import numpy as np
import scipy.optimize as optimize

def run_packing():
    num_circles = 26
    radius_initial = 0.1  # Initial guess for radii
    radius_guess = np.ones(num_circles) * radius_initial

    def hex_grid_positions(n):
        # Generate hexagonal grid points inside [0,1]^2
        positions = []
        for i in range(n):
            r = i / (n - 1)
            x = r * (1 - 4 * r * r)  # Adjust for hexagonal spacing
            y = 0.5
            positions.append([x, y])
        return np.array(positions)

    def hex_grid_radii(n):
        # Initial radius for each circle based on grid spacing
        radii = np.zeros(n)
        for i in range(n):
            r = i / (n - 1)
            radii[i] = 0.1  # Simple initial guess
        return radii

    # Initial guess for centers and radii
    centers = hex_grid_positions(num_circles)
    radii = hex_grid_radii(num_circles)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_non_overlap(centers, radii):
        def cons(x):
            # Flatten radii for use in optimization
            x_radii = x[1:]
            x_centers = x[:2*num_circles].reshape(num_circles, 2)
            # Check non-overlapping constraint
            for i in range(num_circles):
                for j in range(i + 1, num_circles):
                    dx = x_centers[i, 0] - x_centers[j, 0]
                    dy = x_centers[i, 1] - x_centers[j, 1]
                    dist = np.sqrt(dx*dx + dy*dy)
                    if dist < x_radii[i] + x_radii[j] - 1e-8:
                        return 1.0  # Constraint violated
            return 0.0  # Constraint satisfied

        return {'type': 'ineq', 'fun': cons}

    def constraint_boundary(centers, radii):
        def cons(x):
            # Flatten radii for use in optimization
            x_radii = x[1:]
            x_centers = x[:2*num_circles].reshape(num_circles, 2)
            # Check boundary constraint
            for i in range(num_circles):
                x, y = x_centers[i]
                r = x_radii[i]
                if (x - r < -1e-8 or x + r > 1 + 1e-8 or
                    y - r < -1e-8 or y + r > 1 + 1e-8):
                    return 1.0  # Constraint violated
            return 0.0  # Constraint satisfied

        return {'type': 'ineq', 'fun': cons}

    # Flatten the centers and radii for optimization
    x0 = np.concatenate([centers.flatten(), radius_guess])

    # Define optimization problem
    cons = [constraint_non_overlap(centers, radii), constraint_boundary(centers, radii)]
    bounds = [(0, 1) for _ in range(2*num_circles)] + [(0, 1) for _ in range(num_circles)]

    result = optimize.minimize(
        fun=objective,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Extract results
    centers_result = result.x[:2*num_circles].reshape(num_circles, 2)
    radii_result = result.x[2*num_circles:]
    sum_radii_result = np.sum(radii_result)

    # Validate the packing (optional)
    is_valid, message = validate_packing(centers_result, radii_result)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers_result, radii_result, sum_radii_result
