import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    # Initial guess for radius based on hexagonal packing
    initial_guess = np.full(n, 0.03)  # Start with a conservative estimate

    # Generate better initial positions using hexagonal packing
    centers = np.zeros((n, 2))
    idx = 0

    # Compute optimal row and column sizes for hexagonal grid
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    grid_centers = []

    for row in range(rows):
        for col in range(cols):
            if idx >= n:
                break
            # Hexagonal grid position
            x = 0.5 * (1 - (col % 2)) + 0.5 * (col + 1) / (cols + 1)
            y = 0.5 * (1 - (row % 2)) + 0.5 * (row + 1) / (rows + 1)
            grid_centers.append([x, y])
            idx += 1

    centers = np.array(grid_centers)

    # Objective function to maximize the sum of radii (minimize negative sum)
    def objective(radii):
        return -np.sum(radii)

    # Constraints
    def constraints(radii):
        constraints = []

        # Non-negativity
        constraints.append({'type': 'ineq', 'fun': lambda r: r})

        # Boundary constraints for each circle
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1 + 1e-12 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1 + 1e-12 - y - r})

        # Non-overlap constraints for each pair of circles
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, dist=dist: dist - (r[i] + r[j])})

        return constraints

    # Optimization with SLSQP
    result = opt.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        constraints=constraints(initial_guess),
        bounds=[(0, np.inf) for _ in range(n)],
        options={'ftol': 1e-9, 'maxiter': 500}
    )

    # Extract results
    radii = result.x
    centers = np.array(grid_centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

    # and additional circles near the boundaries and corners.

        col = i % 5

            x += 0.05

        radii[i] = 0.05

        return -np.sum(radii)  # minimize negative sum (maximize sum)

        n = len(radii)
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return False
        return True

    # Define constraints to ensure circles do not overlap
    def non_overlap_constraints(radii, centers):
        n = len(radii)
        for i in range(n):

                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)

                    return False
        return True

    # Define bounds for radii (non-negative and bounded)
    bounds = Bounds(0.0, 1.0)

    for _ in range(5):
        result = optimize.minimize(
            objective,

            bounds=bounds,
