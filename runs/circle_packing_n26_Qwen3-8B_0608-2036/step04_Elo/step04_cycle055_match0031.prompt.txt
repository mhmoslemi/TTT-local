import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = np.zeros((n, 2))
    radii = np.ones(n) * 0.05  # Initial guess for radii

    # Initial placement: hexagonal close packing with some randomness for better distribution
    for i in range(n):
        row = i // 5
        col = i % 5
        if row % 2 == 1:
            col += 1  # Offset for odd rows
        if col >= 5:
            col = 0
            row += 1
        # Adjust for corners and edges
        x = 0.5 * (1 - (col % 2)) + 0.5 * (col + 1) / (5 + 1)
        y = 0.5 * (1 - (row % 2)) + 0.5 * (row + 1) / (5 + 1)
        centers[i] = [x, y]

    # Function to compute the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints:
    # 1. Radii must be nonnegative
    # 2. Circles must be within the unit square
    # 3. Circles must not overlap

    # Define bounds and constraints
    bounds = [(0, np.inf) for _ in range(n)]
    constraints = []

    # Constraint 2: All circles must be within the unit square
    for i in range(n):
        def constraint_inside(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return x - r + 1e-12  # x - r >= -1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_inside})

        def constraint_inside_y(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return y - r + 1e-12  # y - r >= -1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_inside_y})

        def constraint_outside_x(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return 1 + 1e-12 - x - r  # x + r <= 1 + 1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_outside_x})

        def constraint_outside_y(radii, idx=i):
            x, y = centers[idx]
            r = radii[idx]
            return 1 + 1e-12 - y - r  # y + r <= 1 + 1e-12
        constraints.append({'type': 'ineq', 'fun': constraint_outside_y})

    # Constraint 3: No overlap between circles
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            def constraint_no_overlap(radii, idx=i, jdx=j, dist=dist):
                r_i = radii[idx]
                r_j = radii[jdx]
                return dist - r_i - r_j + 1e-12  # dist >= r_i + r_j - 1e-12
            constraints.append({'type': 'ineq', 'fun': constraint_no_overlap})

    # Optimization: Use SLSQP for constrained optimization
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Extract results
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Final validation
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial packing using a hexagonal grid approach
    num_circles = 26
    max_radius = 1.0
    radius_guess = 0.1  # Initial guess for each circle radius

    # Start with a hexagonal packing
    centers = np.zeros((num_circles, 2))
    radii = radius_guess * np.ones(num_circles)

    # First circle in the center
    centers[0] = [0.5, 0.5]
    radii[0] = 0.1

    # Define initial hexagonal grid pattern for the rest
    dx = 0.1  # horizontal spacing
    dy = np.sqrt(3) / 2 * dx  # vertical spacing in hexagonal grid

    for i in range(1, num_circles):
        row = i // 5
        col = i % 5

        x = 0.1 + col * dx
        y = 0.1 + row * dy
        if row % 2 == 1:
            x += dx / 2

        centers[i] = [x, y]
        radii[i] = radius_guess

    # Refine the packing with optimization
    def objective(radii, centers):
        return -np.sum(radii)  # Objective is to minimize negative sum (maximize sum)

    def constraints(radii, centers):
        n = len(radii)
        constraints = []
        # Boundary constraint: ensure circles are within the square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r >= -1e-12) & (x + r <= 1 + 1e-12) &
                               (y - r >= -1e-12) & (y + r <= 1 + 1e-12))
        # Non-overlapping constraint
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

    # Optimization
    bounds = Bounds(0, 1.0)
    result = optimize.minimize(
        objective,
        radii,
        args=(centers,),
        bounds=bounds,
        method='L-BFGS-B',
        options={'ftol': 1e-8, 'maxiter': 100}
    )

    # Adjust centers and radii based on optimization
    optimized_radii = result.x
    optimized_centers = centers.copy()

    # Refine centers using local search to fill gaps
    def refine_centers(centers, radii):
        for i in range(len(radii)):
            for j in range(len(radii)):
                if i == j:
                    continue
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    # Adjust centers to increase separation
                    diff = np.array([dx, dy]) / dist
                    centers[i] += diff * (radii[i] + radii[j] - dist) * 0.1
                    centers[j] -= diff * (radii[i] + radii[j] - dist) * 0.1
        return centers

    # Perform refinement
    for _ in range(5):
        optimized_centers = refine_centers(optimized_centers, optimized_radii)

    # Validate packing
    is_valid, msg = validate_packing(optimized_centers, optimized_radii)
    if not is_valid:
        print("Error:", msg)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    # Calculate total sum of radii
    sum_radii = np.sum(optimized_radii)

    return optimized_centers, optimized_radii, sum_radii
