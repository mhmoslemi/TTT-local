import scipy.optimize as opt

    centers = []
    radii = []

    rows = int(np.ceil(np.sqrt(n_circles)))
    columns = int(np.ceil(n_circles / rows))

    # Calculate the horizontal and vertical distances between centers
    # For a hexagonal grid: horizontal spacing is 2r, vertical spacing is sqrt(3)*r
    # To simplify, assume that the horizontal distance between centers is 1.0 / sqrt(2) and vertical is 1.0 / sqrt(3)
    # This will give a rough hexagonal grid within the unit square
    r_initial = 0.14
    h_spacing = 1.0 / np.sqrt(2)
    v_spacing = 1.0 / np.sqrt(3)

    for i in range(rows):
        for j in range(columns):
            x = j * h_spacing
            y = i * v_spacing
            centers.append([x, y])
            radii.append(r_initial)

    centers = np.array(centers)
    radii = np.array(radii)

    # Define the objective function: maximize the sum of radii
    def objective(x):
        # x contains concatenated values: [x1, y1, r1, x2, y2, r2, ...]
        x_flat = x.reshape((n_circles, 3))
        centers_opt = x_flat[:, :2]
        radii_opt = x_flat[:, 2]

        # Compute the total sum of radii
        total_radius = np.sum(radii_opt)
        return -total_radius  # Minimize the negative of the sum (equivalent to maximizing the sum)

    # Define the constraints
    constraints = []

    # Constraint: All centers must lie within [0, 1]^2 with radius r
    for i in range(n_circles):
        def center_bounds_i(x):
            x_flat = x.reshape((n_circles, 3))
            x_i, y_i, r_i = x_flat[i]
            return [x_i - r_i, x_i + r_i, y_i - r_i, y_i + r_i]
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: center_bounds_i(x)[0]})  # x - r >= 0
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: 1 - center_bounds_i(x)[1]})  # x + r <= 1
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: center_bounds_i(x)[2]})  # y - r >= 0

                dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)

            constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: distance_constraint_i_j(x)})

    bounds = []

    x0 = np.concatenate([centers.reshape(-1), radii])
