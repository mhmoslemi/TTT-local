import scipy.optimize as optimize

def run_packing():

    num_circles = 26
    rows = np.ceil(np.sqrt(num_circles)).astype(int)
    cols = np.ceil(num_circles / rows).astype(int)
    centers = np.zeros((num_circles, 2))
    radius_initial = 0.1  # Initial radius guess

    for i in range(num_circles):

        col = i % cols

        centers[i, 0] = x
        centers[i, 1] = y

    # Scale the initial packing to fit within the unit square
    max_x = np.max(centers[:, 0])
    max_y = np.max(centers[:, 1])
    scale = 1.0 / max(max_x, max_y)

    centers[:, 0] += (1 - max_x) / 2
    centers[:, 1] += (1 - max_y) / 2

    radii = np.full((num_circles,), radius_initial)

        return -np.sum(x)

    # Function to enforce non-overlapping constraint
    def constraint_overlap(x, centers):

                dy = centers[i, 1] - centers[j, 1]

                    return -1e6  # Penalize overlap
        return 0

    def constraint_boundary(x):
