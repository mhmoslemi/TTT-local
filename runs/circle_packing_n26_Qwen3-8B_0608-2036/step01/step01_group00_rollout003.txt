    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    initial_radius = 0.05
    initial_positions = []
    for i in range(rows):

            y = i * (np.sqrt(3) * initial_radius + 1e-6)

                x += initial_radius
            initial_positions.append((x, y))
    
    # Ensure we have 26 circles

    centers = np.array(initial_positions)
    radii = np.full(n, 0.05)
    
    # Optimization function to maximize sum of radii

        return -np.sum(radii)  # Minimize negative sum to maximize

    # Constraints

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
