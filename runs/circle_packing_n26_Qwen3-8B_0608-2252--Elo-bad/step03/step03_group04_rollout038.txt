def _initialize_hexagonal_grid(num_circles):
    # Return a hexagonal grid of circles within the unit square
    centers = []
    rows = int(np.sqrt(num_circles))
    cols = rows
    if rows * cols < num_circles:
        cols += 1
    spacing = 1.0 / (cols + 1)
    for i in range(rows):
        for j in range(cols):
            x = j * spacing + spacing / 2
            y = i * spacing + spacing / 2
            if i % 2 == 1:
                x += spacing / 2
            centers.append([x, y])
    # Shuffle to create randomness
    random.shuffle(centers)
    return np.array(centers)

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):

            return False, f"Circle {i} has negative radius {radii[i]}"
