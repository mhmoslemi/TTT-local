import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.2 + np.random.uniform(-0.05, 0.05)
            y = row * 0.2 + np.random.uniform(-0.05, 0.05)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    def random_positions(n):
        return np.random.rand(n, 2)

    def refine_packing(positions, initial_radii):

            centers = x[:2*n].reshape(n, 2)
            radii = x[2*n:]
            return -np.sum(radii)

        def boundary_constraint(x):
            centers = x[:2*n].reshape(n, 2)
            radii = x[2*n:]
            return np.concatenate([
                centers[:, 0] - radii - 1e-6,
                1 - centers[:, 0] - radii - 1e-6,
                centers[:, 1] - radii - 1e-6,
                1 - centers[:, 1] - radii - 1e-6
            ])

        def distance_constraint(x):
            centers = x[:2*n].reshape(n, 2)
            radii = x[2*n:]

                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                    idx += 1
            return dist

        x0 = np.concatenate([positions.flatten(), initial_radii])
