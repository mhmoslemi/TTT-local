import scipy.optimize as optimize
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    def generate_initial_positions(n):
        positions = []
        for i in range(n):
            # Hexagonal packing in the center
            row = i // 5
            col = i % 5
            if row == 0 and col == 0:
                x = np.random.uniform(0.2, 0.3)
                y = np.random.uniform(0.2, 0.3)
            elif row == 0 and col == 4:
                x = np.random.uniform(0.7, 0.8)
                y = np.random.uniform(0.2, 0.3)
            elif row == 4 and col == 0:
                x = np.random.uniform(0.2, 0.3)
                y = np.random.uniform(0.7, 0.8)
            elif row == 4 and col == 4:
                x = np.random.uniform(0.7, 0.8)
                y = np.random.uniform(0.7, 0.8)
            else:
                # Hexagonal arrangement
                x = col * 0.2 + 0.1
                y = row * 0.2 + 0.1
                # Perturbation
                x += np.random.uniform(-0.01, 0.01)
                y += np.random.uniform(-0.01, 0.01)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    best_sum_radii = 0
    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)

    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)

    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1 - centers[:, 1] - radii - 1e-6
        ])

    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

    initial_radii = np.full(n, 0.05)

    x0 = np.concatenate([initial_positions.flatten(), initial_radii])

    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    cons = [
        {'type': 'ineq', 'fun': boundary_constraint},
        {'type': 'ineq', 'fun': distance_constraint}

    result = optimize.minimize(

    centers = result.x[:2*n].reshape(n, 2)
