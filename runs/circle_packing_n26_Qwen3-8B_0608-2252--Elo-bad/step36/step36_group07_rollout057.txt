import numpy as np
import scipy.optimize as optimize
import random

def generate_grid_positions(n, spacing, offset_x, offset_y):
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * spacing + offset_x
        y = row * spacing + offset_y
        positions.append([x, y])
    return np.array(positions)

def generate_corner_grid_positions(n, spacing):
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * spacing
        y = row * spacing
        positions.append([x, y])
    return np.array(positions)

def generate_corner_weighted_positions(n, spacing):
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * spacing
        y = row * spacing
        # Give more weight to circles near the corners
        if (col == 0 or col == 4) or (row == 0 or row == 4):
            x += np.random.uniform(-0.05, 0.05) if col != 0 and col != 4 else 0
            y += np.random.uniform(-0.05, 0.05) if row != 0 and row != 4 else 0
        positions.append([x, y])
    return np.array(positions)

def generate_hybrid_positions(n, spacing):
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * spacing
        y = row * spacing
        x += np.random.uniform(-0.02, 0.02)
        y += np.random.uniform(-0.02, 0.02)
        positions.append([x, y])
    return np.array(positions)

def generate_random_positions(n):
    positions = np.random.rand(n, 2)
    return positions

def generate_corner_positions(n):
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * 0.2 + (0.3 if col == 0 or col == 4 else 0.0)
        y = row * 0.2 + (0.3 if row == 0 or row == 4 else 0.0)
        positions.append([x, y])
    return np.array(positions)

def generate_zigzag_positions(n):
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        base_x = col * 0.2 + (0.3 if col == 0 or col == 4 else 0.0)
        base_y = row * 0.2 + (0.3 if row == 0 or row == 4 else 0.0)
        x = base_x
        y = base_y + (0.02 if (row + col) % 2 == 1 else 0.0)
        positions.append([x, y])
    return np.array(positions)

def generate_spiral_positions(n):
    positions = []
    for i in range(n):
        angle = 2 * np.pi * i / n
        radius = 0.2 + 0.1 * (i % 5)
        x = 0.5 + radius * np.cos(angle)
        y = 0.5 + radius * np.sin(angle)
        positions.append([x, y])
    return np.array(positions)

def generate_radial_positions(n):
    positions = []
    for i in range(n):
        angle = 2 * np.pi * i / n
        radius = 0.2 + 0.1 * (i % 5)
        x = 0.5 + radius * np.cos(angle)
        y = 0.5 + radius * np.sin(angle)
        positions.append([x, y])
    return np.array(positions)

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    edge_radius = 0.08
    initial_guesses = [
        (generate_grid_positions(n, 0.26, 0.1, 0.1), np.full(n, 0.04)),

        (generate_zigzag_positions(n), np.full(n, 0.04)),
        (generate_random_positions(n), np.full(n, 0.04)),

def objective(x):
    centers = x[:2 * 26].reshape(26, 2)
    radii = x[2 * 26:]
    edge_bias = 0.0
    for j in range(26):
        x_j, y_j = centers[j]
        r_j = radii[j]
        if x_j < 0.08 or x_j > 1 - 0.08 or y_j < 0.08 or y_j > 1 - 0.08:
            edge_bias += r_j
    return -np.sum(radii) - edge_bias * 0.2

def boundary_constraint(x, n):
    centers = x[:2 * n].reshape(n, 2)
    radii = x[2 * n:]
    left = centers[:, 0] - radii - 1e-8
    right = 1.0 - centers[:, 0] - radii - 1e-8
    bottom = centers[:, 1] - radii - 1e-8
    top = 1.0 - centers[:, 1] - radii - 1e-8
    return np.concatenate([left, right, bottom, top])

def distance_constraint(x, n):
    centers = x[:2 * n].reshape(n, 2)
    radii = x[2 * n:]
    dist = np.zeros(n * (n - 1) // 2)
    idx = 0
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-8
            idx += 1
    return dist

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
