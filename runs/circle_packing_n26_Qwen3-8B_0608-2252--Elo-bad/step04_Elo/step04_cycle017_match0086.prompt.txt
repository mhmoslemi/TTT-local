import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate a hybrid initial layout: hexagonal in the center, boundary-filling for corners
    def hybrid_positions(n, spacing):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * spacing
            y = row * spacing + (col % 2) * (spacing * np.sqrt(3) / 2)
            positions.append([x, y])
        return np.array(positions)

    # Initial spacing to fit all circles in the unit square
    initial_spacing = 1.0 / np.sqrt(n)
    initial_positions = hybrid_positions(n, initial_spacing)
    initial_radii = np.full(n, initial_spacing / 2)

    # Normalize positions to fit within the unit square
    min_x, min_y = initial_positions.min(axis=0)
    max_x, max_y = initial_positions.max(axis=0)
    scale = min(1.0 / max_x, 1.0 / max_y)
    initial_positions *= scale
    initial_radii *= scale

    # Define objective function
    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)  # We minimize negative of sum to maximize sum

    # Define constraints
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([centers[:, 0] - radii - 1e-6, 
                              1 - centers[:, 0] - radii - 1e-6,
                              centers[:, 1] - radii - 1e-6, 
                              1 - centers[:, 1] - radii - 1e-6])

    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

    # Initial guess
    x0 = np.concatenate([initial_positions.flatten(), initial_radii])

    # Bounds: [0, 1] for positions and [0, 1] for radii
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': boundary_constraint},
        {'type': 'ineq', 'fun': distance_constraint}
    ]

    # Optimization using a different method for better convergence
    # First use COBYLA for initial refinement
    result = optimize.minimize(
        objective,
        x0,
        method='COBYLA',  # Trying COBYLA for better robustness
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1500}
    )

    # Extract the result
    centers = result.x[:2*n].reshape(n, 2)
    radii = result.x[2*n:]
    sum_radii = np.sum(radii)

    return (centers, radii, sum_radii)
