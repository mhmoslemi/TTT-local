import numpy as np
import scipy.optimize as optimize

def hexagonal_positions(n):
    """Generate a hexagonal grid of circles with the given number of circles."""
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        # Define horizontal and vertical spacing
        h = 1.0 / np.sqrt(3)
        v = h * np.sqrt(3) / 2
        # Adjust for staggered rows
        if row % 2 == 1:
            y_offset = v / 2
        else:
            y_offset = 0
        x = col * h
        y = row * v + y_offset
        positions.append([x, y])
    return np.array(positions)

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate a more compact hexagonal grid with staggered rows
    initial_positions = hexagonal_positions(n)
    # Normalize to fit within the unit square
    initial_positions /= np.max(np.abs(initial_positions))

    # Initial guess for radii
    initial_radii = np.full(n, 1.0 / (2 * np.sqrt(3)))

    # Define objective function
    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)  # We minimize negative of sum to maximize sum

    # Define boundary constraint to ensure circles lie within the unit square
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,  # Left boundary
            1 - centers[:, 0] - radii - 1e-6,  # Right boundary
            centers[:, 1] - radii - 1e-6,  # Bottom boundary
            1 - centers[:, 1] - radii - 1e-6  # Top boundary
        ])

    # Define distance constraint to ensure circles do not overlap
    def distance_constraint(x):

                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

    # Initial guess

    # Bounds: [0, 1] for positions and [0, 1] for radii
