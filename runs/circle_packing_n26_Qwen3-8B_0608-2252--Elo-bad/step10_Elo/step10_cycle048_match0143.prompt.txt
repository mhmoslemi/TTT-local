import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate a variety of initial configurations
    def generate_initial_positions(num_attempts=5):
        best_config = None
        best_sum_radii = 0
        for _ in range(num_attempts):
            # Grid-based with perturbation
            positions = []
            for i in range(n):
                row = i // 5
                col = i % 5
                x = col * 0.25
                y = row * 0.25
                # Add small random perturbation
                x += np.random.uniform(-0.02, 0.02)
                y += np.random.uniform(-0.02, 0.02)
                x = np.clip(x, 0, 1)
                y = np.clip(y, 0, 1)
                positions.append([x, y])
            positions = np.array(positions)
            radii = np.full(n, 0.05)
            yield positions, radii

        # Hybrid grid and random configurations
        for _ in range(num_attempts):
            positions = np.random.rand(n, 2)
            positions = np.clip(positions, 0, 1)
            radii = np.full(n, 0.05)
            yield positions, radii

        # Dense corner packing
        positions = np.zeros((n, 2))
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.25 + np.random.uniform(-0.02, 0.02)
            y = row * 0.25 + np.random.uniform(-0.02, 0.02)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions[i] = [x, y]
        radii = np.full(n, 0.05)
        yield positions, radii

    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)
    best_sum_radii = 0

    # Objective function: maximize sum of radii
    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)

    # Boundary constraint: ensure circles are within the square
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-8,
            1 - centers[:, 0] - radii - 1e-8,
            centers[:, 1] - radii - 1e-8,
            1 - centers[:, 1] - radii - 1e-8
        ])

    # Distance constraint: ensure circles do not overlap
    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx*dx + dy*dy) - radii[i] - radii[j] - 1e-8
                idx += 1
        return dist

    # Define bounds for centers and radii
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Try multiple initial configurations
    for positions, radii in generate_initial_positions(num_attempts=10):
        x0 = np.concatenate([positions.flatten(), radii])
        result = optimize.minimize(
            objective,
            x0,
            method='SLSQP',
            bounds=bounds,
            constraints=[
                {'type': 'ineq', 'fun': boundary_constraint},
                {'type': 'ineq', 'fun': distance_constraint}
            ],
            tol=1e-8,
            options={'maxiter': 5000, 'ftol': 1e-9, 'eps': 1e-8}
        )

        if result.success:
            centers = result.x[:2*n].reshape(n, 2)
            radii = result.x[2*n:]
            sum_radii = np.sum(radii)
            if sum_radii > best_sum_radii:
                best_sum_radii = sum_radii
                best_centers = centers
                best_radii = radii

    return (best_centers, best_radii, best_sum_radii)
