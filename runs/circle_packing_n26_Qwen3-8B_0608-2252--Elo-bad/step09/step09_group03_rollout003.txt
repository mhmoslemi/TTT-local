import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

            offset_y = (row % 2) * 0.15
            x = col * 0.2
            y = row * 0.2 + offset_y

            y += np.random.uniform(-0.02, 0.02)
            # Ensure positions are within square
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    best_sum_radii = 0
    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)

    # Define objective function (maximize sum of radii)
    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)

    # Define boundary constraints: circle must stay within [0, 1]^2
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,  # x - r >= 0
            1 - centers[:, 0] - radii - 1e-6,  # 1 - x - r >= 0
            centers[:, 1] - radii - 1e-6,  # y - r >= 0
            1 - centers[:, 1] - radii - 1e-6  # 1 - y - r >= 0
        ])

    # Define distance constraints: circles must not overlap
    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        constraints = []
        for i in range(n):

                dist = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j]
