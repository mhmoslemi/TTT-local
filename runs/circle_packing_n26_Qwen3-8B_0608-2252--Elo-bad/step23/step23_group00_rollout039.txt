import numpy as np
import scipy.optimize as optimize
import random

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    def generate_hexagonal_positions(n, spacing, offset_x, offset_y):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * spacing + offset_x
            y = row * spacing + (col % 2) * spacing * 0.5 + offset_y
            positions.append([x, y])
        return np.array(positions)

    def generate_corner_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.2 + (0.3 if col == 0 or col == 4 else 0.0)
            y = row * 0.2 + (0.3 if row == 0 or row == 4 else 0.0)
            positions.append([x, y])
        return np.array(positions)

    def generate_random_positions(n):
        positions = []
        for _ in range(n):
            x = np.random.uniform(0.1, 0.9)
            y = np.random.uniform(0.1, 0.9)
            positions.append([x, y])
        return np.array(positions)

    def generate_initial_guesses(n):
        positions = []
        radii = []

        # Hexagonal grid with adjusted spacing
        hex_positions = generate_hexagonal_positions(n, 0.22, 0.1, 0.1)
        positions.append(hex_positions)
        radii.append(np.full(n, 0.05))

        # Corner positions
        corner_positions = generate_corner_positions(n)
        positions.append(corner_positions)
        radii.append(np.full(n, 0.05))

        # Random positions with edge bias
        random_positions = generate_random_positions(n)
        positions.append(random_positions)
        radii.append(np.full(n, 0.05))

        return positions, radii

    def objective(x, n):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        # Objective: maximize sum of radii, so minimize -sum(radii)
        # Add edge bias to guide circles towards edges
        edge_bias = 0.0
        for j in range(n):
            x_j, y_j = centers[j]
            r_j = radii[j]
            if x_j < 0.08 or x_j > 1 - 0.08 or y_j < 0.08 or y_j > 1 - 0.08:
                edge_bias += r_j
        return -np.sum(radii) - edge_bias * 0.2  # Weigh edge bias more strongly

    def boundary_constraint(x, n):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        left = centers[:, 0] - radii - 1e-6
        right = 1.0 - centers[:, 0] - radii - 1e-6
        bottom = centers[:, 1] - radii - 1e-6
        top = 1.0 - centers[:, 1] - radii - 1e-6
        return np.concatenate([left, right, bottom, top])

        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)

        for i in range(n):

                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

            return False, "NaN values present"

            best_centers = centers

    # Final validation of best packing
    valid, msg = validate_packing(best_centers, best_radii)
    if not valid:
        print("Invalid packing:", msg)
        raise ValueError(msg)

    return (best_centers, best_radii, best_sum)
