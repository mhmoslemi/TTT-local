import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    def generate_corner_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.2 + (0.3 if col == 0 or col == 4 else 0.0)
            y = row * 0.2 + (0.3 if row == 0 or row == 4 else 0.0)
            positions.append([x, y])
        return np.array(positions)

    def generate_hexagonal_positions(n, spacing, offset_x, offset_y):

        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * spacing + offset_x
            y = row * spacing + (col % 2) * spacing * 0.5 + offset_y
            positions.append([x, y])
        return np.array(positions)

    def generate_custom_positions(n):
        positions = []
        for i in range(n):
            row = i // 5

            base_x = col * 0.2 + (0.3 if col == 0 or col == 4 else 0.0)

            x = base_x + (0.04 if row < 2 or row > 3 else 0.0)
            y = base_y + (0.04 if col < 2 or col > 3 else 0.0)
            positions.append([x, y])
        return np.array(positions)

    def generate_hexagonal_full_positions(n, spacing, offset_x, offset_y):
        positions = []

            row = i // 5
            col = i % 5
            x = col * spacing + offset_x
            y = row * spacing + (col % 2) * spacing * 0.5 + offset_y
            positions.append([x, y])
        return np.array(positions)

    initial_guesses.append((custom_positions, np.full(n, 0.05)))

        scale = min(1.0 / positions.max(), 1.0 / positions.min())
        scaled_positions = positions * scale

        def objective(x):

            edge_bias = 0.0
            for j in range(n):
