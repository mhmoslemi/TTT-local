import numpy as np

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initial placement (modified hexagonal grid)
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    spacing = 1.0 / (cols + 1)

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = spacing / 2 - 1e-4

    # Define objective function for optimization
    def objective(x):
        centers_opt = x[:2*n].reshape(n, 2)
        radii_opt = x[2*n:]

        # Apply constraints
        for i in range(n):
            x, y = centers_opt[i]
            r = radii_opt[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return float('inf')

        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_opt[i] + radii_opt[j] - 1e-12:
                    return float('inf')

        # Objective is to maximize total radius (negative for minimization)
        return -np.sum(radii_opt)

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: centers in [0,1], radii >= 0
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Use Simulated Annealing for global optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        options={'maxiter': 1000, 'ftol': 1e-6}
    )

    # If optimization failed, use a random perturbation
    if result.success:
        centers_opt = result.x[:2*n].reshape(n, 2)
        radii_opt = result.x[2*n:]
    else:
        # Fallback to perturbed initial guess
        centers_opt = centers + np.random.rand(n, 2) * 0.1
        radii_opt = radii + np.random.rand(n) * 0.1

        # Ensure centers and radii are within bounds
        for i in range(n):
            centers_opt[i, 0] = np.clip(centers_opt[i, 0], 0, 1)
            centers_opt[i, 1] = np.clip(centers_opt[i, 1], 0, 1)
            radii_opt[i] = np.clip(radii_opt[i], 0, 1)

    # Final check and validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Warning: Packing validation failed: {message}")
        return centers, radii, np.sum(radii)

    return centers_opt, radii_opt, np.sum(radii_opt)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
