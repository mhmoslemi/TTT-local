import numpy as np
from scipy.optimize import minimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Step 1: Generate a hexagonal grid inside the unit square
    # Define number of layers and positions
    layers = 5
    cols_per_layer = [1, 2, 3, 4, 5]
    
    # Place the first circle at the center
    centers[0] = [0.5, 0.5]
    radii[0] = 0.1  # Initial guess for radius

    # Place other circles in a hexagonal pattern
    for layer in range(1, layers):
        for col in range(1, cols_per_layer[layer]):
            # Adjust x and y so that the circles do not go out of bounds
            x = 0.5 + (col - 0.5) * (radii[0] * 2) * (1.5)
            y = 0.5 + layer * (radii[0] * 2)
            # Ensure that the circle does not go out of bounds
            if x + radii[0] > 1 or x - radii[0] < 0 or y + radii[0] > 1 or y - radii[0] < 0:
                continue
            centers[layer * cols_per_layer[layer - 1] + col] = [x, y]
    
    # Step 2: Define objective function with better constraints and more aggressive optimization
    def objective(x: np.ndarray) -> float:
        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:].reshape(n)
        
        # Ensure all radii are non-negative
        if np.any(radii_flat < 0):
            return float('inf')
        
        # Check if any circle is outside the unit square
        for i in range(n):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return float('inf')
        
        # Check if circles are overlapping
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return float('inf')
        
        return -np.sum(radii_flat)  # We minimize the negative of total radius to maximize it
    
    # Prepare initial guess
    x_initial = np.concatenate([centers.reshape(n * 2), radii])
    
    # Optimization setup
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # for x
        bounds.append((0.0, 1.0))  # for y
        bounds.append((0.0, 1.0))  # for radius
    
    # Optimization using Nelder-Mead for better convergence
    result = minimize(objective, x_initial, method='Nelder-Mead', bounds=bounds, tol=1e-7, options={'xatol': 1e-7, 'fatol': 1e-7})
    
    # Extract final centers and radii
    centers_final = result.x[:n * 2].reshape(n, 2)
    radii_final = result.x[n * 2:].reshape(n)
    
    # Validate the final packing
    is_valid, message = validate_packing(centers_final, radii_final)
    if not is_valid:
        raise RuntimeError(f"Invalid packing: {message}")
    
    # Calculate total sum
    sum_radii = np.sum(radii_final)
    
    return centers_final, radii_final, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Initial grid positions
    def grid_positions(n, spacing):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * spacing
            y = row * spacing
            positions.append([x, y])
        return np.array(positions)

    # Generate initial grid-based positions
    initial_spacing = 0.35  # Adjust this spacing for a tight initial packing
    initial_positions = grid_positions(n, initial_spacing)
    initial_radii = np.full(n, initial_spacing * 0.5)  # Initial radius as a fraction of spacing

    # Normalize positions to fit within the unit square and scale radii
    min_x, min_y = initial_positions.min(axis=0)
    max_x, max_y = initial_positions.max(axis=0)
    scale = min(1.0 / max_x, 1.0 / max_y)
    initial_positions *= scale
    initial_radii *= scale

    # Boundary-aware adjustment: circles near the edges and corners can grow more
    edge_bonus = 0.2  # Bonus for edge/corner positions
    edge_distance = 0.1
    edge_weights = np.zeros(n)
    for i in range(n):
        x, y = initial_positions[i]
        if x < edge_distance or x > 1 - edge_distance or y < edge_distance or y > 1 - edge_distance:
            edge_weights[i] = 1.0 + edge_bonus
    initial_radii *= edge_weights
    initial_radii = np.clip(initial_radii, 0, 1.0)

    # Define objective function
    def objective(x):
        # x is a flattened array of [pos1_x, pos1_y, r1, pos2_x, pos2_y, r2, ...]
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        sum_r = np.sum(radii)
        return -sum_r  # Minimize the negative sum to maximize sum

    # Define constraints
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        # Ensure that each circle is entirely inside the unit square
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1.0 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1.0 - centers[:, 1] - radii - 1e-6
        ])

    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

    # Initial guess
    x0 = np.concatenate([initial_positions.flatten(), initial_radii])

    # Bounds: [0, 1] for positions and [0, 1] for radii
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': boundary_constraint},
        {'type': 'ineq', 'fun': distance_constraint}
    ]

    # Stage 1: Coarse optimization to get an initial good solution
    result_1 = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-4
    )

    # Refine with tighter constraints
    result_2 = optimize.minimize(
        objective,
        result_1.x,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6
    )

    # Final result
    centers = result_2.x[:2*n].reshape(n, 2)
    radii = result_2.x[2*n:]
    sum_radii = np.sum(radii)

    # Validate result
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print("Invalid packing:", msg)
        raise ValueError(msg)

    return (centers, radii, sum_radii)
