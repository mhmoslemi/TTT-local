def hexagonal_positions(n, spacing):
    positions = []
    for i in range(n):
        row = i // 5
        col = i % 5
        x = col * spacing
        y = row * spacing + (col % 2) * (spacing * np.sqrt(3) / 2)
        positions.append([x, y])
    return np.array(positions)
