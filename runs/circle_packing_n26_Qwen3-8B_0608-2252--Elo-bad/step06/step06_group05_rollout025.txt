import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    def hexagonal_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            # Calculate spacing for hexagonal grid in unit square
            x = col * 0.2
            y = row * 0.2 + (col % 2) * 0.1
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.2
            y = row * 0.2
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    initial_position_strategies = [hexagonal_positions, grid_positions, random_positions, nested_positions, boundary_initial_positions]

    best_sum_radii = 0
    best_centers = np.zeros((n, 2))
    best_radii = np.zeros(n)

    for positions_strategy in initial_position_strategies:

        positions = positions_strategy(n)
        # Initial radii for all circles
        initial_radii = np.full(n, 0.05)

            radii = x[2*n:]
            return -np.sum(radii)

            centers = x[:2*n].reshape(n, 2)
            radii = x[2*n:]

                centers[:, 1] - radii - 1e-6,
