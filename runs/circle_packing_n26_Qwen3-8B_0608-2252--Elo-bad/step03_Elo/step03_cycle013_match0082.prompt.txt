import numpy as np
import random
import scipy.optimize as optimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Initial placement using a hexagonal grid
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    spacing = 1.0 / (cols + 1)

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i, 0] = x
        centers[i, 1] = y
        # Initial radii based on spacing, with some reduction to allow for better packing
        radii[i] = spacing / 2

    # Define objective function for optimization
    def objective(x):
        centers_opt = x[:2*n].reshape(n, 2)
        radii_opt = x[2*n:]

        # Check boundaries
        for i in range(n):
            x, y = centers_opt[i]
            r = radii_opt[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return float('inf')

        # Check for overlaps
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_opt[i] + radii_opt[j] - 1e-12:
                    return float('inf')

        # Objective is to maximize total radius (negative for minimization)
        return -np.sum(radii_opt)

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Bounds: centers in [0,1], radii >= 0
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Use SLSQP for optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        options={'maxiter': 1000, 'ftol': 1e-6}
    )

    # If optimization failed, use a random perturbation
    if result.success:
        centers_opt = result.x[:2*n].reshape(n, 2)
        radii_opt = result.x[2*n:]
    else:
        # Fallback to perturbed initial guess
        centers_opt = centers + np.random.rand(n, 2) * 0.1
        radii_opt = radii + np.random.rand(n) * 0.1

        # Ensure centers and radii are within bounds
        for i in range(n):
            centers_opt[i, 0] = np.clip(centers_opt[i, 0], 0, 1)
            centers_opt[i, 1] = np.clip(centers_opt[i, 1], 0, 1)
            radii_opt[i] = np.clip(radii_opt[i], 0, 1)

    # Final check and validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Warning: Packing validation failed: {message}")
        return centers, radii, np.sum(radii)

    return centers_opt, radii_opt, np.sum(radii_opt)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Define a more flexible initial grid position generator
    def grid_positions(n, spacing, rows, cols):
        positions = []
        for i in range(n):
            row = i // cols
            col = i % cols
            x = col * spacing
            y = row * spacing
            positions.append([x, y])
        return np.array(positions)

    # Generate initial grid-based positions
    initial_spacing = 0.3  # Start with smaller spacing to allow more flexibility
    initial_rows = 5
    initial_cols = 5
    initial_positions = grid_positions(n, initial_spacing, initial_rows, initial_cols)
    
    # Normalize initial positions to fit within the unit square
    initial_positions /= np.max(initial_positions)
    
    # Initial radii as a fraction of spacing
    initial_radii = np.full(n, initial_spacing * 0.5)

    # Boundary-aware adjustment: circles near edges/corners can grow more
    edge_distance = 0.1  # Distance threshold for edge/corner proximity
    edge_bonus = 0.3  # Relative bonus for edge/corner positions
    edge_weights = np.zeros(n)
    for i in range(n):
        x, y = initial_positions[i]
        if x < edge_distance or x > 1 - edge_distance or y < edge_distance or y > 1 - edge_distance:
            edge_weights[i] = 1.0 + edge_bonus
    initial_radii *= edge_weights
    initial_radii = np.clip(initial_radii, 0, 1.0)

    # Define objective function to maximize sum of radii
    def objective(x):
        # x is a flattened array of [pos1_x, pos1_y, r1, pos2_x, pos2_y, r2, ...]
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        sum_r = np.sum(radii)
        return -sum_r  # Minimize the negative sum to maximize sum

    # Define boundary constraint to ensure circles are within the unit square
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        # Ensure that each circle is entirely inside the unit square
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1.0 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1.0 - centers[:, 1] - radii - 1e-6
        ])

    # Define distance constraint to ensure circles do not overlap
    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

    # Initial guess
    x0 = np.concatenate([initial_positions.flatten(), initial_radii])

    # Bounds: [0, 1] for positions and [0, 1] for radii
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': boundary_constraint},
        {'type': 'ineq', 'fun': distance_constraint}
    ]

    # Run optimization with multiple stages
    result_1 = optimize.minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=cons,
        tol=1e-4
    )

    # Refine with tighter constraints using a different method
    result_2 = optimize.minimize(
        objective,
        result_1.x,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6
    )

    # Final result
    centers = result_2.x[:2*n].reshape(n, 2)
    radii = result_2.x[2*n:]
    sum_radii = np.sum(radii)

    # Validate result
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print("Invalid packing:", msg)
        raise ValueError(msg)

    return (centers, radii, sum_radii)
