import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial positions using a hexagonal grid pattern
    n = 26
    rows = int(np.ceil(n / 6))  # Adjusting for hexagonal packing
    cols = 6
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col + (row % 2) * 0.5
        y = row * np.sqrt(3) / 2
        centers[i, 0] = x / 6  # Scale to fit in [0, 1]
        centers[i, 1] = y / (rows * np.sqrt(3) / 2)

    # Initial guess for radii (equal radii for simplicity)
    initial_radius = 0.1
    radii = np.full(n, initial_radius)

    # Define optimization bounds: centers within [0, 1]^2, radii non-negative
    bounds_centers = [(0, 1) for _ in range(2 * n)]
    bounds_radii = [(0, 1) for _ in range(n)]

    # Objective function to maximize the sum of radii
    def objective(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:].reshape(n)
        return -np.sum(radii_flat)  # Minimize negative of sum

    # Constraints: circles within square and no overlap
    def constraints(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:].reshape(n)
        constraints = []

        # Circle within square
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12
                or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return np.inf  # Penalize violations
    
        # Circle-to-circle distances
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return np.inf  # Penalize violations

        return 0  # No constraints violated

    # Optimization setup
    x0 = np.concatenate([centers.flatten(), radii.flatten()])
    result = optimize.minimize(
        objective,
        x0,
        bounds=np.concatenate([bounds_centers, bounds_radii]),
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",
        options={"maxiter": 500, "ftol": 1e-8, "eps": 1e-8}
    )

    # Extract final centers and radii
    centers_opt = result.x[:2 * n].reshape(n, 2)
    radii_opt = result.x[2 * n:].reshape(n)
    sum_radii = np.sum(radii_opt)

    # Validate packing for safety
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        print("Invalid packing:", msg)
        return np.zeros((n, 2)), np.zeros(n), 0.0

    return centers_opt, radii_opt, sum_radii

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial grid-based packing configuration
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Initial positions with spacing
    grid = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        grid[i, 0] = col * (1.0 / (cols - 1))
        grid[i, 1] = row * (1.0 / (rows - 1))
    
    # Initial radii (each circle will have radius such that they touch their neighbors)
    # For adjacent circles in the grid, minimal radius for contact
    min_radius = 1.0 / np.sqrt(2) / (cols - 1) * 0.9  # Adjust for safety margin
    radii = np.full(n, min_radius)
    
    # Objective function
    def objective(x):
        return -np.sum(x)  # Minimize negative sum (maximize sum)
    
    # Equality constraints (each circle must be within the unit square)
    def constraint_center(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, cx + r, cy - r, cy + r]  # cx - r >= 0, cx + r <= 1, cy - r >= 0, cy + r <= 1
    
    # Inequality constraints for circle positions
    def constraint_position(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, 1 - cx - r, cy - r, 1 - cy - r]  # cx - r >= 0, 1 - cx - r >= 0, cy - r >= 0, 1 - cy - r >= 0
    
    # Inequality constraints for non-overlapping between circles
    def constraint_overlap(x, i, j):
        cx_i = x[2 * i]
        cy_i = x[2 * i + 1]
        r_i = x[2 * n + i]
        cx_j = x[2 * j]
        cy_j = x[2 * j + 1]
        r_j = x[2 * n + j]
        dist = np.sqrt((cx_i - cx_j)**2 + (cy_i - cy_j)**2)
        return dist - (r_i + r_j)  # dist >= r_i + r_j
    
    # Define variables: 2n for centers, n for radii
    bounds = [(0.0, 1.0) for _ in range(2 * n)] + [(0.0, 1.0) for _ in range(n)]
    
    # Initial guess for optimization
    x0 = np.concatenate([grid.flatten(), radii])
    
    # Set up constraints
    constraints = []
    
    # Position constraints (each circle within the square)
    for i in range(n):
        c = {'type': 'ineq', 'fun': lambda x, idx=i: constraint_position(x, idx)}
        constraints.append(c)
    
    # Non-overlapping constraints
    for i in range(n):
        for j in range(i + 1, n):
            c = {'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)}
            constraints.append(c)
    
    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-8)
    
    # Post-process result
    centers = result.x[:2 * n].reshape(n, 2)
    radii = result.x[2 * n:]
    
    # Validate the solution
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Optimization failed: {msg}")
    
    return centers, radii, np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
