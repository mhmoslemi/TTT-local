def generate_initial_guess(n_circles, square_size=(1, 1)):
    # Generate initial hexagonal packing guess inside the unit square

    max_col = int(np.ceil(n_circles / max_row))
    centers = []
    radii = []
    radius_initial = 0.05
    for i in range(max_row):
        for j in range(max_col):
            x = j * (2 * radius_initial * np.sqrt(3) / 1) + radius_initial
            y = i * (2 * radius_initial) + radius_initial
            if x > square_size[0] or y > square_size[1]:
                continue
            centers.append([x, y])
            radii.append(radius_initial)
    # If not enough circles, continue with smaller spacing

                y = i * (2 * radius_initial) + radius_initial

                radii.append(radius_initial)
    # Trim to only n_circles and randomize

    radii = np.array(radii[:n_circles])

def run_packing():
    n_circles = 26

    def constraint_boundary(x):
        centers_flat = x[:2 * n_circles].reshape((n_circles, 2))
        radii_flat = x[2 * n_circles:]
        for i in range(n_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
                return -1
            if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    # Constraint: circles must not overlap
    def constraint_overlap(x):
        centers_flat = x[:2 * n_circles].reshape((n_circles, 2))
        radii_flat = x[2 * n_circles:]
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return -1
        return 0

    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up optimization problem
    cons = [
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization settings
    result = optimize.minimize(

        method='SLSQP',
        constraints=cons,
