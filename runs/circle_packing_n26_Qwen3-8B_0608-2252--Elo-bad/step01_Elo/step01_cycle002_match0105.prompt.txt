```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
from itertools import combinations

def run_packing():
    # Initial hexagonal grid packing
    num_circles = 26
    num_rows = 5
    circles_per_row = 5
    spacing = 1.0 / (num_rows + 1)  # vertical spacing
    horizontal_shift = 0.5 / (circles_per_row + 1)  # horizontal shift for hexagonal packing

    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Initial guess for radii
    for i in range(num_rows):
        for j in range(circles_per_row):
            row_idx = i * circles_per_row + j
            x = j * (1.0 / circles_per_row) + horizontal_shift
            y = i * spacing + spacing
            centers[row_idx] = [x, y]
            radii[row_idx] = 0.1  # Initial small radius

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraint functions
    def constraint1(radii):
        return np.sum(radii) - 2.636  # Ensure the sum is at least 2.636

    def constraint2(radii):
        return np.max(radii) - 0.1  # Ensure no radius exceeds 0.1 initially

    # Bounds for radii: non-negative
    bounds = [(0, 0.1) for _ in range(num_circles)]

    # Initial radius guess
    initial_radii = np.full(num_circles, 0.1)

    # Optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(
        objective,
        initial_radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    optimal_radii = result.x
    optimal_centers = centers.copy()

    # Local refinement: move circles slightly to fill gaps
    for _ in range(5):  # Run local refinement multiple times
        for i in range(num_circles):
            # Try moving this circle slightly
            dx = np.random.uniform(-0.01, 0.01)
            dy = np.random.uniform(-0.01, 0.01)
            new_center = optimal_centers[i] + [dx, dy]

            # Check if new center is within bounds
            if (new_center[0] < -1e-12 or new_center[0] > 1 + 1e-12 or
                new_center[1] < -1e-12 or new_center[1] > 1 + 1e-12):
                continue

            # Check for overlap with other circles
            overlap = False
            for j in range(num_circles):
                if i == j:
                    continue
                dist = np.hypot(new_center[0] - optimal_centers[j][0],
                                new_center[1] - optimal_centers[j][1])
                if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                    overlap = True
                    break

            if not overlap:
                optimal_centers[i] = new_center

    # Final check for boundary and overlap
    for i in range(num_circles):
        x, y = optimal_centers[i]
        r = optimal_radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return np.zeros((26, 2)), np.zeros(26), 0.0

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.hypot(optimal_centers[i][0] - optimal_centers[j][0],
                            optimal_centers[i][1] - optimal_centers[j][1])
            if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                return np.zeros((26, 2)), np.zeros(26), 0.0

    return optimal_centers, optimal_radii, np.sum(optimal_radii)
```

```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial grid-based packing configuration
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Initial positions with spacing
    grid = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        grid[i, 0] = col * (1.0 / (cols - 1))
        grid[i, 1] = row * (1.0 / (rows - 1))
    
    # Initial radii (each circle will have radius such that they touch their neighbors)
    # For adjacent circles in the grid, minimal radius for contact
    min_radius = 1.0 / np.sqrt(2) / (cols - 1) * 0.9  # Adjust for safety margin
    radii = np.full(n, min_radius)
    
    # Objective function
    def objective(x):
        return -np.sum(x)  # Minimize negative sum (maximize sum)
    
    # Equality constraints (each circle must be within the unit square)
    def constraint_center(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, cx + r, cy - r, cy + r]  # cx - r >= 0, cx + r <= 1, cy - r >= 0, cy + r <= 1
    
    # Inequality constraints for circle positions
    def constraint_position(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, 1 - cx - r, cy - r, 1 - cy - r]  # cx - r >= 0, 1 - cx - r >= 0, cy - r >= 0, 1 - cy - r >= 0
    
    # Inequality constraints for non-overlapping between circles
    def constraint_overlap(x, i, j):
        cx_i = x[2 * i]
        cy_i = x[2 * i + 1]
        r_i = x[2 * n + i]
        cx_j = x[2 * j]
        cy_j = x[2 * j + 1]
        r_j = x[2 * n + j]
        dist = np.sqrt((cx_i - cx_j)**2 + (cy_i - cy_j)**2)
        return dist - (r_i + r_j)  # dist >= r_i + r_j
    
    # Define variables: 2n for centers, n for radii
    bounds = [(0.0, 1.0) for _ in range(2 * n)] + [(0.0, 1.0) for _ in range(n)]
    
    # Initial guess for optimization
    x0 = np.concatenate([grid.flatten(), radii])
    
    # Set up constraints
    constraints = []
    
    # Position constraints (each circle within the square)
    for i in range(n):
        c = {'type': 'ineq', 'fun': lambda x, idx=i: constraint_position(x, idx)}
        constraints.append(c)
    
    # Non-overlapping constraints
    for i in range(n):
        for j in range(i + 1, n):
            c = {'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)}
            constraints.append(c)
    
    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-8)
    
    # Post-process result
    centers = result.x[:2 * n].reshape(n, 2)
    radii = result.x[2 * n:]
    
    # Validate the solution
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Optimization failed: {msg}")
    
    return centers, radii, np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
