import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Initialize centers in a hexagonal grid
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Define the number of layers and positions
    layers = 5
    cols_per_layer = [1, 2, 3, 4, 5]
    
    # Place the first circle at the center
    centers[0] = [0.5, 0.5]
    radii[0] = 0.1  # Initial guess for radius
    
    # Place other circles in a hexagonal pattern
    for layer in range(1, layers):
        for col in range(1, cols_per_layer[layer]):
            x = 0.5 + (col - 0.5) * (radii[0] * 2) * (1.5)
            y = 0.5 + layer * (radii[0] * 2)
            centers[layer * cols_per_layer[layer - 1] + col] = [x, y]
    
    # Step 2: Optimize
    def objective(x: np.ndarray) -> float:
        # x is a flattened array of centers and radii
        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:].reshape(n)
        
        # Calculate the sum of radii
        sum_radii = np.sum(radii_flat)
        
        # Ensure all radii are non-negative
        if np.any(radii_flat < 0):
            return float('inf')
        
        # Check if circles are inside the square
        for i in range(n):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x - r < -1e-5 or x + r > 1 + 1e-5 or y - r < -1e-5 or y + r > 1 + 1e-5:
                return float('inf')
        
        # Check for circle-circle overlaps
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-5:
                    return float('inf')
        
        return -sum_radii  # We minimize the negative of total radius to maximize it
    
    # Prepare initial guess
    x_initial = np.concatenate([centers.reshape(n * 2), radii])
    
    # Define bounds: centers must be between 0 and 1, radii must be >= 0
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # for x
        bounds.append((0.0, 1.0))  # for y
        bounds.append((0.0, 1.0))  # for radius
    
    # Optimization
    result = minimize(objective, x_initial, method='L-BFGS-B', bounds=bounds, tol=1e-6)
    
    # Extract final centers and radii
    centers_final = result.x[:n * 2].reshape(n, 2)
    radii_final = result.x[n * 2:].reshape(n)
    
    # Validate the final packing
    is_valid, message = validate_packing(centers_final, radii_final)
    if not is_valid:
        raise RuntimeError(f"Invalid packing: {message}")
    
    # Calculate total sum
    sum_radii = np.sum(radii_final)
    
    return centers_final, radii_final, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
from itertools import combinations

def run_packing():
    # Initial hexagonal grid packing
    num_circles = 26
    num_rows = 5
    circles_per_row = 5
    spacing = 1.0 / (num_rows + 1)  # vertical spacing
    horizontal_shift = 0.5 / (circles_per_row + 1)  # horizontal shift for hexagonal packing

    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Initial guess for radii
    for i in range(num_rows):
        for j in range(circles_per_row):
            row_idx = i * circles_per_row + j
            x = j * (1.0 / circles_per_row) + horizontal_shift
            y = i * spacing + spacing
            centers[row_idx] = [x, y]
            radii[row_idx] = 0.1  # Initial small radius

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraint functions
    def constraint1(radii):
        return np.sum(radii) - 2.636  # Ensure the sum is at least 2.636

    def constraint2(radii):
        return np.max(radii) - 0.1  # Ensure no radius exceeds 0.1 initially

    # Bounds for radii: non-negative
    bounds = [(0, 0.1) for _ in range(num_circles)]

    # Initial radius guess
    initial_radii = np.full(num_circles, 0.1)

    # Optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(
        objective,
        initial_radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    optimal_radii = result.x
    optimal_centers = centers.copy()

    # Local refinement: move circles slightly to fill gaps
    for _ in range(5):  # Run local refinement multiple times
        for i in range(num_circles):
            # Try moving this circle slightly
            dx = np.random.uniform(-0.01, 0.01)
            dy = np.random.uniform(-0.01, 0.01)
            new_center = optimal_centers[i] + [dx, dy]

            # Check if new center is within bounds
            if (new_center[0] < -1e-12 or new_center[0] > 1 + 1e-12 or
                new_center[1] < -1e-12 or new_center[1] > 1 + 1e-12):
                continue

            # Check for overlap with other circles
            overlap = False
            for j in range(num_circles):
                if i == j:
                    continue
                dist = np.hypot(new_center[0] - optimal_centers[j][0],
                                new_center[1] - optimal_centers[j][1])
                if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                    overlap = True
                    break

            if not overlap:
                optimal_centers[i] = new_center

    # Final check for boundary and overlap
    for i in range(num_circles):
        x, y = optimal_centers[i]
        r = optimal_radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return np.zeros((26, 2)), np.zeros(26), 0.0

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.hypot(optimal_centers[i][0] - optimal_centers[j][0],
                            optimal_centers[i][1] - optimal_centers[j][1])
            if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                return np.zeros((26, 2)), np.zeros(26), 0.0

    return optimal_centers, optimal_radii, np.sum(optimal_radii)
