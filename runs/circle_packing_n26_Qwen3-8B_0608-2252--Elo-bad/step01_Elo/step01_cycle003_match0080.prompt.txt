import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial placement of 25 circles in a grid
    num_circles = 26
    grid_size = 5
    grid_spacing = 1.0 / (grid_size - 1)
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # First 25 circles in grid
    for i in range(grid_size):
        for j in range(grid_size):
            if i * grid_size + j < 25:
                centers[i * grid_size + j, 0] = j * grid_spacing
                centers[i * grid_size + j, 1] = i * grid_spacing
                radii[i * grid_size + j] = 0.1  # Initial radius

    # Place the 26th circle at a central position
    centers[25, 0] = 0.5
    centers[25, 1] = 0.5
    radii[25] = 0.1

    def objective(radii_vector):
        return -np.sum(radii_vector)

    def constraints_func(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:]
        centers = centers_flat
        radii = radii_flat

        # Boundary constraints
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf  # Violates boundary

        # Non-overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf  # Overlap

        # Return negative sum of radii (since we're minimizing)
        return -np.sum(radii)

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Bounds: center coordinates between 0 and 1, radii non-negative
    bounds = []
    for _ in range(num_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    # Optimization settings
    result = opt.minimize(
        fun=constraints_func,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        options={
            'ftol': 1e-9,
            'maxiter': 1000,
            'disp': False
        }
    )

    # Extract the solution
    optimized_x = result.x
    centers_opt = optimized_x[:2 * num_circles].reshape(num_circles, 2)
    radii_opt = optimized_x[2 * num_circles:]

    # Check validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Optimization failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    # Hexagonal grid parameters
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))

    # Initial grid spacing (adjust accordingly based on hexagonal packing)
    spacing = 1.0 / (cols + 1)
    offset = spacing / 2
    centers = np.zeros((num_circles, 2))
    radius = np.ones(num_circles) * 0.05  # Initial guess for radii
    
    idx = 0
    for row in range(rows):
        for col in range(cols):
            x = col * spacing + offset
            y = row * spacing * np.sqrt(3) / 2 + offset
            if idx < num_circles:
                centers[idx] = [x, y]
                idx += 1

    # Optimization objective: minimize sum of squared radii (equivalent to maximizing sum of radii under constraint sum(r) = constant)
    def optimize_objective(x):
        global centers, radius
        new_centers = x[:2*num_circles].reshape(num_circles, 2)
        new_radii = x[2*num_circles:]
        sum_radii = np.sum(new_radii)
        penalty = 0
        # Constraint: All circles must be within the square
        for i in range(num_circles):
            if new_centers[i, 0] - new_radii[i] < 0 or new_centers[i, 0] + new_radii[i] > 1:
                penalty += 1e6
            if new_centers[i, 1] - new_radii[i] < 0 or new_centers[i, 1] + new_radii[i] > 1:
                penalty += 1e6
        # Constraint: No overlaps
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dist = np.linalg.norm(new_centers[i] - new_centers[j])
                if dist < new_radii[i] + new_radii[j] - 1e-8:
                    penalty += 1e6
        # Objective: Minimize sum of squared radii (equivalent to maximizing sum of radii due to constraint)
        return np.sum(new_radii ** 2) + penalty

    # Bounds on positions and radii
    bounds = []
    for i in range(num_circles):
        # Bounds for (x, y) coordinates
        x_bounds = (0.0, 1.0)
        y_bounds = (0.0, 1.0)
        bounds.extend([x_bounds, y_bounds])
        # Bounds for radius
        radius_bounds = (0.0, 0.5)
        bounds.extend([radius_bounds])

    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radius])

    # Run optimization
    result = opt.minimize(optimize_objective, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-8)

    # Extract result
    optimized_centers = result.x[:2*num_circles].reshape(num_circles, 2)
    optimized_radii = result.x[2*num_circles:]
    sum_radii = np.sum(optimized_radii)

    # Validate the result
    is_valid, msg = validate_packing(optimized_centers, optimized_radii)
    if not is_valid:
        print(msg)
        # If invalid, fall back to initial guess
        optimized_centers = centers
        optimized_radii = radius
        sum_radii = np.sum(optimized_radii)

    return optimized_centers, optimized_radii, sum_radii
