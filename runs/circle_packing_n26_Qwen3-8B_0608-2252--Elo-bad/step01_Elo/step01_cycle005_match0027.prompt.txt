import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial grid-based packing configuration
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Initial positions with spacing
    grid = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        grid[i, 0] = col * (1.0 / (cols - 1))
        grid[i, 1] = row * (1.0 / (rows - 1))
    
    # Initial radii (each circle will have radius such that they touch their neighbors)
    # For adjacent circles in the grid, minimal radius for contact
    min_radius = 1.0 / np.sqrt(2) / (cols - 1) * 0.9  # Adjust for safety margin
    radii = np.full(n, min_radius)
    
    # Objective function
    def objective(x):
        return -np.sum(x)  # Minimize negative sum (maximize sum)
    
    # Equality constraints (each circle must be within the unit square)
    def constraint_center(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, cx + r, cy - r, cy + r]  # cx - r >= 0, cx + r <= 1, cy - r >= 0, cy + r <= 1
    
    # Inequality constraints for circle positions
    def constraint_position(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, 1 - cx - r, cy - r, 1 - cy - r]  # cx - r >= 0, 1 - cx - r >= 0, cy - r >= 0, 1 - cy - r >= 0
    
    # Inequality constraints for non-overlapping between circles
    def constraint_overlap(x, i, j):
        cx_i = x[2 * i]
        cy_i = x[2 * i + 1]
        r_i = x[2 * n + i]
        cx_j = x[2 * j]
        cy_j = x[2 * j + 1]
        r_j = x[2 * n + j]
        dist = np.sqrt((cx_i - cx_j)**2 + (cy_i - cy_j)**2)
        return dist - (r_i + r_j)  # dist >= r_i + r_j
    
    # Define variables: 2n for centers, n for radii
    bounds = [(0.0, 1.0) for _ in range(2 * n)] + [(0.0, 1.0) for _ in range(n)]
    
    # Initial guess for optimization
    x0 = np.concatenate([grid.flatten(), radii])
    
    # Set up constraints
    constraints = []
    
    # Position constraints (each circle within the square)
    for i in range(n):
        c = {'type': 'ineq', 'fun': lambda x, idx=i: constraint_position(x, idx)}
        constraints.append(c)
    
    # Non-overlapping constraints
    for i in range(n):
        for j in range(i + 1, n):
            c = {'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)}
            constraints.append(c)
    
    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-8)
    
    # Post-process result
    centers = result.x[:2 * n].reshape(n, 2)
    radii = result.x[2 * n:]
    
    # Validate the solution
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Optimization failed: {msg}")
    
    return centers, radii, np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Initialize centers in a hexagonal grid
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Define the number of layers and positions
    layers = 5
    cols_per_layer = [1, 2, 3, 4, 5]
    
    # Place the first circle at the center
    centers[0] = [0.5, 0.5]
    radii[0] = 0.1  # Initial guess for radius
    
    # Place other circles in a hexagonal pattern
    for layer in range(1, layers):
        for col in range(1, cols_per_layer[layer]):
            x = 0.5 + (col - 0.5) * (radii[0] * 2) * (1.5)
            y = 0.5 + layer * (radii[0] * 2)
            centers[layer * cols_per_layer[layer - 1] + col] = [x, y]
    
    # Step 2: Optimize
    def objective(x: np.ndarray) -> float:
        # x is a flattened array of centers and radii
        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:].reshape(n)
        
        # Calculate the sum of radii
        sum_radii = np.sum(radii_flat)
        
        # Ensure all radii are non-negative
        if np.any(radii_flat < 0):
            return float('inf')
        
        # Check if circles are inside the square
        for i in range(n):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x - r < -1e-5 or x + r > 1 + 1e-5 or y - r < -1e-5 or y + r > 1 + 1e-5:
                return float('inf')
        
        # Check for circle-circle overlaps
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-5:
                    return float('inf')
        
        return -sum_radii  # We minimize the negative of total radius to maximize it
    
    # Prepare initial guess
    x_initial = np.concatenate([centers.reshape(n * 2), radii])
    
    # Define bounds: centers must be between 0 and 1, radii must be >= 0
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # for x
        bounds.append((0.0, 1.0))  # for y
        bounds.append((0.0, 1.0))  # for radius
    
    # Optimization
    result = minimize(objective, x_initial, method='L-BFGS-B', bounds=bounds, tol=1e-6)
    
    # Extract final centers and radii
    centers_final = result.x[:n * 2].reshape(n, 2)
    radii_final = result.x[n * 2:].reshape(n)
    
    # Validate the final packing
    is_valid, message = validate_packing(centers_final, radii_final)
    if not is_valid:
        raise RuntimeError(f"Invalid packing: {message}")
    
    # Calculate total sum
    sum_radii = np.sum(radii_final)
    
    return centers_final, radii_final, sum_radii
