import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial grid-based packing configuration
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Initial positions with spacing
    grid = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        grid[i, 0] = col * (1.0 / (cols - 1))
        grid[i, 1] = row * (1.0 / (rows - 1))
    
    # Initial radii (each circle will have radius such that they touch their neighbors)
    # For adjacent circles in the grid, minimal radius for contact
    min_radius = 1.0 / np.sqrt(2) / (cols - 1) * 0.9  # Adjust for safety margin
    radii = np.full(n, min_radius)
    
    # Objective function
    def objective(x):
        return -np.sum(x)  # Minimize negative sum (maximize sum)
    
    # Equality constraints (each circle must be within the unit square)
    def constraint_center(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, cx + r, cy - r, cy + r]  # cx - r >= 0, cx + r <= 1, cy - r >= 0, cy + r <= 1
    
    # Inequality constraints for circle positions
    def constraint_position(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, 1 - cx - r, cy - r, 1 - cy - r]  # cx - r >= 0, 1 - cx - r >= 0, cy - r >= 0, 1 - cy - r >= 0
    
    # Inequality constraints for non-overlapping between circles
    def constraint_overlap(x, i, j):
        cx_i = x[2 * i]
        cy_i = x[2 * i + 1]
        r_i = x[2 * n + i]
        cx_j = x[2 * j]
        cy_j = x[2 * j + 1]
        r_j = x[2 * n + j]
        dist = np.sqrt((cx_i - cx_j)**2 + (cy_i - cy_j)**2)
        return dist - (r_i + r_j)  # dist >= r_i + r_j
    
    # Define variables: 2n for centers, n for radii
    bounds = [(0.0, 1.0) for _ in range(2 * n)] + [(0.0, 1.0) for _ in range(n)]
    
    # Initial guess for optimization
    x0 = np.concatenate([grid.flatten(), radii])
    
    # Set up constraints
    constraints = []
    
    # Position constraints (each circle within the square)
    for i in range(n):
        c = {'type': 'ineq', 'fun': lambda x, idx=i: constraint_position(x, idx)}
        constraints.append(c)
    
    # Non-overlapping constraints
    for i in range(n):
        for j in range(i + 1, n):
            c = {'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)}
            constraints.append(c)
    
    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-8)
    
    # Post-process result
    centers = result.x[:2 * n].reshape(n, 2)
    radii = result.x[2 * n:]
    
    # Validate the solution
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Optimization failed: {msg}")
    
    return centers, radii, np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial placement of 25 circles in a grid
    num_circles = 26
    grid_size = 5
    grid_spacing = 1.0 / (grid_size - 1)
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    # First 25 circles in grid
    for i in range(grid_size):
        for j in range(grid_size):
            if i * grid_size + j < 25:
                centers[i * grid_size + j, 0] = j * grid_spacing
                centers[i * grid_size + j, 1] = i * grid_spacing
                radii[i * grid_size + j] = 0.1  # Initial radius

    # Place the 26th circle at a central position
    centers[25, 0] = 0.5
    centers[25, 1] = 0.5
    radii[25] = 0.1

    def objective(radii_vector):
        return -np.sum(radii_vector)

    def constraints_func(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:]
        centers = centers_flat
        radii = radii_flat

        # Boundary constraints
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf  # Violates boundary

        # Non-overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf  # Overlap

        # Return negative sum of radii (since we're minimizing)
        return -np.sum(radii)

    # Initial guess
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Bounds: center coordinates between 0 and 1, radii non-negative
    bounds = []
    for _ in range(num_circles):
        bounds.append((0.0, 1.0))  # x
        bounds.append((0.0, 1.0))  # y
        bounds.append((0.0, 1.0))  # radius

    # Optimization settings
    result = opt.minimize(
        fun=constraints_func,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        options={
            'ftol': 1e-9,
            'maxiter': 1000,
            'disp': False
        }
    )

    # Extract the solution
    optimized_x = result.x
    centers_opt = optimized_x[:2 * num_circles].reshape(num_circles, 2)
    radii_opt = optimized_x[2 * num_circles:]

    # Check validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Optimization failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii
