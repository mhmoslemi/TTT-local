import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Initialize centers in a hexagonal grid
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Define the number of layers and positions
    layers = 5
    cols_per_layer = [1, 2, 3, 4, 5]
    
    # Place the first circle at the center
    centers[0] = [0.5, 0.5]
    radii[0] = 0.1  # Initial guess for radius
    
    # Place other circles in a hexagonal pattern
    for layer in range(1, layers):
        for col in range(1, cols_per_layer[layer]):
            x = 0.5 + (col - 0.5) * (radii[0] * 2) * (1.5)
            y = 0.5 + layer * (radii[0] * 2)
            centers[layer * cols_per_layer[layer - 1] + col] = [x, y]
    
    # Step 2: Optimize
    def objective(x: np.ndarray) -> float:
        # x is a flattened array of centers and radii
        centers_flat = x[:n * 2].reshape(n, 2)
        radii_flat = x[n * 2:].reshape(n)
        
        # Calculate the sum of radii
        sum_radii = np.sum(radii_flat)
        
        # Ensure all radii are non-negative
        if np.any(radii_flat < 0):
            return float('inf')
        
        # Check if circles are inside the square
        for i in range(n):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x - r < -1e-5 or x + r > 1 + 1e-5 or y - r < -1e-5 or y + r > 1 + 1e-5:
                return float('inf')
        
        # Check for circle-circle overlaps
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i][0] - centers_flat[j][0]
                dy = centers_flat[i][1] - centers_flat[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-5:
                    return float('inf')
        
        return -sum_radii  # We minimize the negative of total radius to maximize it
    
    # Prepare initial guess
    x_initial = np.concatenate([centers.reshape(n * 2), radii])
    
    # Define bounds: centers must be between 0 and 1, radii must be >= 0
    bounds = []
    for i in range(n):
        bounds.append((0.0, 1.0))  # for x
        bounds.append((0.0, 1.0))  # for y
        bounds.append((0.0, 1.0))  # for radius
    
    # Optimization
    result = minimize(objective, x_initial, method='L-BFGS-B', bounds=bounds, tol=1e-6)
    
    # Extract final centers and radii
    centers_final = result.x[:n * 2].reshape(n, 2)
    radii_final = result.x[n * 2:].reshape(n)
    
    # Validate the final packing
    is_valid, message = validate_packing(centers_final, radii_final)
    if not is_valid:
        raise RuntimeError(f"Invalid packing: {message}")
    
    # Calculate total sum
    sum_radii = np.sum(radii_final)
    
    return centers_final, radii_final, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Create an initial hexagonal grid arrangement
    def create_initial_grid(n_circles):
        # Define a more compact layout that covers the unit square
        row_count = int(np.ceil(np.sqrt(n_circles)))
        col_count = int(np.ceil(n_circles / row_count))
        radius = 0.05  # Small initial guess to allow for optimization
        centers = []
        for i in range(n_circles):
            row = i // col_count
            col = i % col_count
            x = col * (1.0 / (col_count - 1)) * 0.95 + 0.025
            y = row * (1.0 / (row_count - 1)) * 0.95 + 0.025
            centers.append([x, y])
        centers = np.array(centers)
        return centers, np.full(n_circles, radius)

    # Objective function to maximize the sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    # Constraint for boundary (circle must stay within unit square)
    def boundary_constraint(radii, i, x, y):
        return x - radii[i]  # Must be >= 0
    def boundary_constraint_right(radii, i, x, y):
        return 1.0 - x - radii[i]  # Must be >= 0
    def boundary_constraint_top(radii, i, x, y):
        return 1.0 - y - radii[i]  # Must be >= 0
    def boundary_constraint_bottom(radii, i, x, y):
        return y - radii[i]  # Must be >= 0

    # Constraint for no overlap
    def no_overlap_constraint(radii, i, j, centers):
        dx = centers[i, 0] - centers[j, 0]
        dy = centers[i, 1] - centers[j, 1]
        dist = np.sqrt(dx*dx + dy*dy)
        return dist - radii[i] - radii[j]

    # Initialize centers and radii
    centers, initial_radii = create_initial_grid(26)
    bounds = [(0, 1) for _ in range(26)]  # Radii are non-negative

    # Construct constraints
    constraints = []
    for i in range(26):
        x, y = centers[i]
        constraints.append({'type': 'ineq', 'fun': lambda r, idx=i, x=x, y=y: boundary_constraint(r, idx, x, y)})
        constraints.append({'type': 'ineq', 'fun': lambda r, idx=i, x=x, y=y: boundary_constraint_right(r, idx, x, y)})
        constraints.append({'type': 'ineq', 'fun': lambda r, idx=i, x=x, y=y: boundary_constraint_top(r, idx, x, y)})
        constraints.append({'type': 'ineq', 'fun': lambda r, idx=i, x=x, y=y: boundary_constraint_bottom(r, idx, x, y)})

    for i in range(26):
        for j in range(i + 1, 26):
            constraints.append({'type': 'ineq', 'fun': lambda r, i=i, j=j, centers=centers: no_overlap_constraint(r, i, j, centers)})

    # Run optimization
    result = minimize(
        fun=objective,
        x0=initial_radii,
        args=(centers,),
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Extract the solution
    radii = result.x
    centers = np.array(centers)
    sum_radii = np.sum(radii)

    # Check if the solution is valid
    valid, message = validate_packing(centers, radii)
    if not valid:
        print(f"Optimization failed: {message}")
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, radii, sum_radii
