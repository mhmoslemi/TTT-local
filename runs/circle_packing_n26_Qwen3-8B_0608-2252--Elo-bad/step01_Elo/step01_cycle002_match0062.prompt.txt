```python
import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist
from itertools import combinations

def run_packing():
    # Initial hexagonal grid packing
    num_circles = 26
    num_rows = 5
    circles_per_row = 5
    spacing = 1.0 / (num_rows + 1)  # vertical spacing
    horizontal_shift = 0.5 / (circles_per_row + 1)  # horizontal shift for hexagonal packing

    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Initial guess for radii
    for i in range(num_rows):
        for j in range(circles_per_row):
            row_idx = i * circles_per_row + j
            x = j * (1.0 / circles_per_row) + horizontal_shift
            y = i * spacing + spacing
            centers[row_idx] = [x, y]
            radii[row_idx] = 0.1  # Initial small radius

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraint functions
    def constraint1(radii):
        return np.sum(radii) - 2.636  # Ensure the sum is at least 2.636

    def constraint2(radii):
        return np.max(radii) - 0.1  # Ensure no radius exceeds 0.1 initially

    # Bounds for radii: non-negative
    bounds = [(0, 0.1) for _ in range(num_circles)]

    # Initial radius guess
    initial_radii = np.full(num_circles, 0.1)

    # Optimization
    constraints = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    result = minimize(
        objective,
        initial_radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )

    optimal_radii = result.x
    optimal_centers = centers.copy()

    # Local refinement: move circles slightly to fill gaps
    for _ in range(5):  # Run local refinement multiple times
        for i in range(num_circles):
            # Try moving this circle slightly
            dx = np.random.uniform(-0.01, 0.01)
            dy = np.random.uniform(-0.01, 0.01)
            new_center = optimal_centers[i] + [dx, dy]

            # Check if new center is within bounds
            if (new_center[0] < -1e-12 or new_center[0] > 1 + 1e-12 or
                new_center[1] < -1e-12 or new_center[1] > 1 + 1e-12):
                continue

            # Check for overlap with other circles
            overlap = False
            for j in range(num_circles):
                if i == j:
                    continue
                dist = np.hypot(new_center[0] - optimal_centers[j][0],
                                new_center[1] - optimal_centers[j][1])
                if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                    overlap = True
                    break

            if not overlap:
                optimal_centers[i] = new_center

    # Final check for boundary and overlap
    for i in range(num_circles):
        x, y = optimal_centers[i]
        r = optimal_radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return np.zeros((26, 2)), np.zeros(26), 0.0

    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.hypot(optimal_centers[i][0] - optimal_centers[j][0],
                            optimal_centers[i][1] - optimal_centers[j][1])
            if dist < optimal_radii[i] + optimal_radii[j] - 1e-12:
                return np.zeros((26, 2)), np.zeros(26), 0.0

    return optimal_centers, optimal_radii, np.sum(optimal_radii)
```

```python
import numpy as np
import scipy.optimize as optimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing
    num_circles = 26
    rows = int(np.sqrt(num_circles))
    cols = (num_circles + rows - 1) // rows

    # Hexagonal packing parameters
    hex_radius = 1.0 / (2 * np.sqrt(3))  # Radius to fill the unit square
    spacing = 2 * hex_radius  # Distance between centers in hexagonal pattern

    # Initial positions
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col * spacing
        y = row * spacing * np.sqrt(3) / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = hex_radius

    # Adjust positions to fit in unit square
    scale = 1.0 / (np.max(centers) + np.max(radii))
    centers *= scale
    radii *= scale

    # Define objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)

    # Define constraints: circles within square and non-overlapping
    def constraint_within_square(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return -1
        return 0

    def constraint_no_overlap(x):
        centers_flat = x[:num_circles*2].reshape((num_circles, 2))
        radii_flat = x[num_circles*2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return -1
        return 0

    # Optimization variables:
    # - First 2*num_circles entries: coordinates of centers
    # - Last num_circles entries: radii of circles
    x0 = np.concatenate((centers.flatten(), radii))

    # Optimization bounds: circles are within [0,1]^2, radii >= 0
    bounds = [(0, 1) for _ in range(2 * num_circles)] + [(0, 1) for _ in range(num_circles)]

    # Optimization constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_within_square},
        {'type': 'ineq', 'fun': constraint_no_overlap}
    ]

    # Run optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8,
        options={'maxiter': 500}
    )

    # Extract results
    centers_opt = result.x[:2*num_circles].reshape((num_circles, 2))
    radii_opt = result.x[2*num_circles:]
    sum_radii_opt = np.sum(radii_opt)

    # Validate the packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    return centers_opt, radii_opt, sum_radii_opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
