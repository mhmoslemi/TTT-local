            if col % 2 == 1:
                y += spacing * np.sqrt(3)/2
            positions.append([x, y])

        for i in range(n):
            angle = 2 * np.pi * i / n
            r = radius * (spacing_ratio ** i)
            x = center[0] + r * np.cos(angle)
            y = center[1] + r * np.sin(angle)
            positions.append([x, y])
        return np.array(positions)

    # Helper function to generate grid positions with optimized spacing
    def grid_positions(n, spacing):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * spacing + spacing / 2
            y = row * spacing + spacing / 2
            positions.append([x, y])
        return np.array(positions)

    # Helper function to generate corner-based grid positions
    def generate_corner_grid_positions(n):
        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * 0.25 + 0.05
            y = row * 0.25 + 0.05
            # Add some randomness to avoid perfect overlap

            y += random.uniform(-0.02, 0.02)
