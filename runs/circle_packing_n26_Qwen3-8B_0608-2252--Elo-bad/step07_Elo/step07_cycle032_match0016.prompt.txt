        if len(positions) < n:
            positions.append(pos)
    
    # Ensure the number of circles does not exceed n
    if len(positions) > n:

    # Introduce some randomness to avoid perfect hexagonal symmetry

            positions[i][0] += np.random.uniform(-0.1, 0.1)

import numpy as np
import scipy.optimize as optimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Generate initial positions with refined boundary bias
    def generate_initial_positions(n):
        positions = []
        # Use a hexagonal grid
        for i in range(n):
            row = i // 5
            col = i % 5
            x, y = col * 0.25, row * 0.25
            # Randomly bias edge and corner positions to improve boundary utilization
            if col == 0 or col == 4 or row == 0 or row == 4:
                x += np.random.uniform(-0.05, 0.05)
                y += np.random.uniform(-0.05, 0.05)
            positions.append([x, y])
        return np.array(positions)

    # Generate initial positions and normalize them into the unit square
    initial_positions = generate_initial_positions(n)
    initial_positions /= np.max(initial_positions)
    initial_radii = np.full(n, 0.05)

    # Define objective function to maximize sum of radii
    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        sum_r = np.sum(radii)
        return -sum_r  # Minimize the negative sum to maximize sum

    # Define boundary constraint to ensure circles are within the unit square
    def boundary_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        # Ensure that each circle is entirely inside the unit square
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1.0 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1.0 - centers[:, 1] - radii - 1e-6
        ])

    # Define distance constraint to ensure circles do not overlap
    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx * dx + dy * dy) - radii[i] - radii[j] - 1e-6
                idx += 1
        return dist

    # Initial guess
    x0 = np.concatenate([initial_positions.flatten(), initial_radii])

    # Bounds: [0, 1] for positions and [0, 1] for radii
    bounds = [(0, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Constraints
    cons = [
        {'type': 'ineq', 'fun': boundary_constraint},
        {'type': 'ineq', 'fun': distance_constraint}
    ]

    # Initial optimization with L-BFGS-B
    result_1 = optimize.minimize(
        objective,
        x0,
        method='L-BFGS-B',
        bounds=bounds,
        constraints=cons,
        tol=1e-4
    )

    # Refinement with SLSQP for tighter constraints
    result_2 = optimize.minimize(
        objective,
        result_1.x,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6
    )

    # Final refinement with COBYLA for further precision
    result_3 = optimize.minimize(
        objective,
        result_2.x,
        method='COBYLA',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Final result
    centers = result_3.x[:2*n].reshape(n, 2)
    radii = result_3.x[2*n:]
    sum_radii = np.sum(radii)

    # Validate result
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print("Invalid packing:", msg)
        raise ValueError(msg)

    return (centers, radii, sum_radii)
