import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_circles = 26
    rows = 5
    cols = 6
    spacing = 1.0 / (cols - 1)  # Equal spacing

    # Initial grid layout
    centers = np.zeros((num_circles, 2))
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            centers[idx, 0] = j * spacing
            centers[idx, 1] = i * spacing

    # Initial radii (equal for all)
    radii = np.full(num_circles, 0.05)

    # Define the objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Negative because we use minimization

    # Define constraints
    constraints = []

    # Constraint: centers inside the unit square
    def center_bounds(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        return np.all(centers_flat[:, 0] >= 0 - 1e-12) and np.all(centers_flat[:, 0] <= 1 + 1e-12) and \
               np.all(centers_flat[:, 1] >= 0 - 1e-12) and np.all(centers_flat[:, 1] <= 1 + 1e-12)
    constraints.append({'type': 'ineq', 'fun': center_bounds})

    # Constraint: radii non-negative
    def radii_bounds(x):
        radii_flat = x[num_circles * 2:]
        return np.all(radii_flat >= -1e-12)
    constraints.append({'type': 'ineq', 'fun': radii_bounds})

    # Constraint: no overlap between any two circles
    def no_overlap(x):

    constraints.append({'type': 'ineq', 'fun': no_overlap})
