import scipy.optimize as optimize
import matplotlib.pyplot as plt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing configuration
    num_circles = 26
    radius = 0.05  # initial guess for radius
    centers = np.zeros((num_circles, 2))
    rows = int(np.sqrt(num_circles)) + 1
    cols = int(np.ceil(num_circles / rows))
    
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col * (1 + radius) / cols  # uniform spacing with some padding
        y = row * (1 + radius) / rows  # uniform spacing with some padding
        centers[i] = [x, y]

        # x is a flat array [r1, r2, ... r26]

        return -sum_radii  # minimize negative sum to maximize sum

    # Define constraints
    def constraint_center_bounds(x, centers, num_circles):
        radii = x.reshape((num_circles,))
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
                return -100  # strong penalty for boundary violation
            if y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -100
        return 0

    def constraint_overlap(x, centers, num_circles):
        radii = x.reshape((num_circles,))
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]

                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -100  # strong penalty for overlap
        return 0

    # Set up optimization

    constraints = [
        {'type': 'ineq', 'fun': constraint_center_bounds, 'args': (centers, num_circles)},

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
