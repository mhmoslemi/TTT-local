import numpy as np
import scipy.optimize as opt
from typing import Tuple

def calculate_distance(c1: np.ndarray, c2: np.ndarray) -> float:
    """Calculate Euclidean distance between two points."""
    return np.linalg.norm(c1 - c2)

def create_initial_hcp_grid(num_circles: int, max_radius: float) -> Tuple[np.ndarray, np.ndarray]:
    """Generate an initial hexagonal close packing grid in a unit square."""
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    spacing = 1.0 / (cols + 1)
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, max_radius)
    
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j * spacing + spacing / 2
            y = i * spacing + spacing / 2
            if i % 2 == 1:
                x += spacing / 2
            centers[idx] = [x, y]
    return centers, radii

def objective_function(params: np.ndarray, centers: np.ndarray, radii: np.ndarray) -> float:
    """Objective function to maximize the sum of radii."""
    return -np.sum(radii)  # Minimize the negative sum to maximize the sum

def constraints_function(params: np.ndarray, centers: np.ndarray, radii: np.ndarray) -> np.ndarray:
    """Constraint function to ensure circles are within bounds and do not overlap."""
    total = []
    # Update radii and centers from parameters
    new_centers = params[:2 * len(radii)].reshape((len(radii), 2))
    new_radii = params[2 * len(radii):]
    
    # Boundary constraints

        x, y = new_centers[i]
        r = new_radii[i]
        total.append(x - r + 1e-12)
        total.append(x + r - 1.0 - 1e-12)

        total.append(y + r - 1.0 - 1e-12)

    return total

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
