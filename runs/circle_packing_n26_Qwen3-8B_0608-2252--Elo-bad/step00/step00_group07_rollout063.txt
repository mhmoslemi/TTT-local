import numpy as np
import scipy.optimize as optimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Define the number of circles
    n = 26

    # Initial grid-based hexagonal packing
    def hexagonal_grid(num_circles):
        # Hexagonal packing in a unit square
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        spacing = 1.0 / (cols - 1)
        centers = []
        for i in range(rows):
            for j in range(cols):
                x = j * spacing
                y = i * spacing * np.sqrt(3) / 2
                centers.append([x, y])
        # Trim to exactly num_circles
        return np.array(centers[:num_circles]) / 1.0  # Normalize to [0,1]

    # Initial centers using hexagonal packing
    centers = hexagonal_grid(n)
    radii = np.ones(n) * 0.1  # Initial guess for radii

    # Objective function (negative of sum of radii to maximize)
    def objective(radii_flat):
        return -np.sum(radii_flat)

    # Constraint 1: Each circle is within the unit square
    def square_constraint(radii_flat):
        return np.array([0.0 for _ in range(n)])  # No direct constraint on positions

    # Constraint 2: No overlap between circles
    def no_overlap_constraint(radii_flat):
        # Flatten radii
        radii = radii_flat.reshape((n, 1))
        # Compute pairwise distance
        dists = squareform(pdist(centers))
        # Return distance - (radii[i] + radii[j]) >= 0.0
        return dists - (radii[:, np.newaxis] + radii[np.newaxis, :])

    # Optimization bounds
    bounds = [(0.0, 0.5) for _ in range(n)]  # Radii between 0 and 0.5

    # Optimization setup
    initial_radii = np.ones(n) * 0.1
    result = optimize.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: no_overlap_constraint(x)},
            {'type': 'ineq', 'fun': lambda x: square_constraint(x)}
        ]
    )

    # Extract results
    centers_final = centers
    radii_final = result.x
    sum_radii = np.sum(radii_final)

    # Validate the packing
    valid, msg = validate_packing(centers_final, radii_final)

        centers_final = centers
        radii_final = np.ones(n) * 0.1
        sum_radii = np.sum(radii_final)

    return centers_final, radii_final, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
