import numpy as np
import scipy.optimize as optimize

def run_packing():

    n = 26

    spacing = 0.2
    cols = int(np.ceil(n ** 0.5))
    rows = int(np.ceil(n / cols))
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    idx = 0
    for i in range(rows):
        for j in range(cols):
            if idx >= n:
                break
            x = j * spacing + 0.1
            y = i * spacing + 0.1
            centers[idx] = [x, y]
            radii[idx] = 0.05  # Small initial radius
            idx += 1

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ...]
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:].reshape(n)
        return -np.sum(radii_flat)  # Minimize negative sum to maximize sum

    # Bounds: x and y between 0 and 1, radii non-negative
    bounds = []
    for i in range(n):
        bounds.extend([(0.0, 1.0), (0.0, 1.0), (0.0, 1.0)])

    # Constraint: no overlap between circles
    def constraint_overlap(x):
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:].reshape(n)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return dist - (radii_flat[i] + radii_flat[j] - 1e-8)
        return 0

    # Constraint: circles within the square
    def constraint_boundary(x):
        centers_flat = x[:2*n].reshape(n, 2)
