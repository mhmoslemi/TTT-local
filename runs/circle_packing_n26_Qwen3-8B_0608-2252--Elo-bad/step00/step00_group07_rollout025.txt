import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial hexagonal grid of 26 circles
    def generate_hex_grid(n, spacing=0.08):
        centers = []
        for i in range(n):
            row = i // 6
            col = i % 6
            x = 0.5 + (col - 2.5) * spacing
            y = 0.5 + (row - 2.5) * spacing * np.sqrt(3)/2
            centers.append([x, y])
        return np.array(centers)

    centers = generate_hex_grid(26)
    radii = np.ones(26) * 0.05  # Initial small radii

    # Objective function to maximize sum of radii
    def objective(x):
        # x is a vector of radii
        return -np.sum(x)  # Negative for minimization

    # Constraint functions
    def constraint_boundary(x, centers):
        for i in range(len(centers)):
            r = x[i]
            x_center, y_center = centers[i]
            if x_center - r < -1e-12 or x_center + r > 1 + 1e-12:
                return -1
            if y_center - r < -1e-12 or y_center + r > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x, centers):
        for i in range(len(centers)):
            for j in range(i + 1, len(centers)):
                r_i = x[i]
                r_j = x[j]
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r_i + r_j - 1e-12:
                    return -1
        return 0

    # Constraints in the form of (type, fun, args)
    constraints = [
        {'type': 'ineq', 'fun': lambda x: constraint_boundary(x, centers)},
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers)}
    ]

    # Optimization
    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1) for _ in range(len(radii))],
        tol=1e-8
    )

    # Extract results
    final_centers = centers
    final_radii = result.x
    final_sum = np.sum(final_radii)

    valid, message = validate_packing(final_centers, final_radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
