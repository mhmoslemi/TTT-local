from scipy.optimize import minimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Number of circles
    n = 26
    
    # Initial hexagonal packing
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Fill centers with hexagonal packing pattern
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= n:
                break
            x = j + 0.5 * (i % 2)
            y = i * (np.sqrt(3) / 2)
            centers[idx] = [x, y]
            radii[idx] = 0.05  # Initial guess for radius
    
    # Objective function to maximize sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Negative because we minimize

    # Constraints:
    # 1. All circles within the unit square
    # 2. No overlap between circles
    def constraint_square(radii_flat):
        centers_flat = centers.reshape(-1)
        radii = radii_flat.reshape(-1)
        for i in range(n):
            x = centers_flat[i]
            y = centers_flat[i + n]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(radii_flat):
        centers_flat = centers.reshape(-1)
        radii = radii_flat.reshape(-1)
        for i in range(n):
            for j in range(i + 1, n):
                x1 = centers_flat[i]
                y1 = centers_flat[i + n]
                r1 = radii[i]
                x2 = centers_flat[j]
                y2 = centers_flat[j + n]

                dx = x1 - x2
                dy = y1 - y2
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < r1 + r2 - 1e-12:
                    return -1
        return 0

    # Initial guess for radii
    radii_initial = radii.copy()

    # Optimization
    result = minimize(
        objective,
        radii_initial,
        method='SLSQP',
        constraints=[
