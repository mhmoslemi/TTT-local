import numpy as np
import scipy.optimize as opt
from typing import Tuple, Any

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial placement: a hexagonal grid with some padding
    num_circles = 26
    spacing = 0.25  # Initial spacing between centers
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Generate initial hexagonal grid
    for i in range(num_circles):
        row = i // 5
        col = i % 5
        x = col * spacing + 0.1
        y = row * spacing * np.sqrt(3) + 0.1
        centers[i] = [x, y]
        radii[i] = 0.02  # Initial radius

    # Define objective function: maximize sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Minimize negative sum to maximize sum

    # Define constraints for optimization
    def constraint_overlap(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    return -1
        return 0

    def constraint_boundary(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        for i in range(num_circles):
            x_center, y_center = centers_flat[i]
            r = radii_flat[i]
            if (x_center - r < -1e-12 or x_center + r > 1 + 1e-12 or
                y_center - r < -1e-12 or y_center + r > 1 + 1e-12):
                return -1
        return 0

    def constraint_radii(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        for r in radii_flat:
            if r < 0:
                return -1
        return 0

    # Bounds: centers must be within [0,1]^2, radii must be >= 0
    bounds = []
    for _ in range(num_circles * 2):
        bounds.append((0.0, 1.0))
    for _ in range(num_circles):
        bounds.append((0.0, 1.0))

    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Optimization with constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_radii}
    ]

    result = opt.minimize(
        fun=objective,
        x0=initial_guess,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-9
    )

    # Extract optimized centers and radii
    centers_opt = result.x[:num_circles * 2].reshape(num_circles, 2)
    radii_opt = result.x[num_circles * 2:]
    sum_radii = np.sum(radii_opt)

    # Validate the packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        raise ValueError(f"Packing is invalid: {message}")

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
