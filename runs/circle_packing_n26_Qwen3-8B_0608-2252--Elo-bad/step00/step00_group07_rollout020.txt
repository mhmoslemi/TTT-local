from scipy.spatial.distance import cdist

    # Define objective function

    # Define constraints
    def constraint_center_bounds(x):
        x = x.reshape((n, 4))
        centers = x[:, :2]
        radii = x[:, 2]
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
