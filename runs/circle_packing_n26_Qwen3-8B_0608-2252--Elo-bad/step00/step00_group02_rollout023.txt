import numpy as np
import random
import math
import sys
import time

def run_packing():
    # Initial parameters
    num_circles = 26
    initial_radius = 0.01
    max_iterations = 1000
    cooling_rate = 0.995
    initial_temperature = 1000
    final_temperature = 1e-6
    perturbation = 0.05

    # Function to calculate the total sum of radii
    def total_sum(radii):
        return np.sum(radii)

    # Function to check if a circle is within the unit square
    def within_bounds(center, radius):
        x, y = center
        return (x - radius >= -1e-12) and (x + radius <= 1 + 1e-12) and \
               (y - radius >= -1e-12) and (y + radius <= 1 + 1e-12)

    # Function to check if two circles overlap
    def circles_overlap(c1, r1, c2, r2):
        dx = c1[0] - c2[0]
        dy = c1[1] - c2[1]
        dist = math.sqrt(dx*dx + dy*dy)
        return dist < r1 + r2 - 1e-12

    # Function to compute the total penalty for overlaps and out-of-bounds circles
    def compute_penalty(centers, radii):
        penalty = 0.0
        for i in range(num_circles):
            if not within_bounds(centers[i], radii[i]):
                penalty += 1e6  # Large penalty for out-of-bounds
            for j in range(i + 1, num_circles):
                if circles_overlap(centers[i], radii[i], centers[j], radii[j]):
                    penalty += 1e6  # Large penalty for overlap
        return penalty

    # Initialize centers and radii
    centers = np.random.rand(num_circles, 2)
    radii = np.full(num_circles, initial_radius)

    # Simulated Annealing
    current_temperature = initial_temperature
    best_centers = centers.copy()
    best_radii = radii.copy()
    best_sum = total_sum(best_radii)

    for iteration in range(max_iterations):
        # Perturb one circle
        idx = random.randint(0, num_circles - 1)
        perturb = random.uniform(-perturbation, perturbation)
        new_center = centers[idx] + perturb * np.random.randn(2)
        new_radius = radii[idx] + random.uniform(-0.01, 0.01)

        # Ensure new_radius is non-negative
        new_radius = max(new_radius, 0)

        # Check if new_center is within the unit square
        if not within_bounds(new_center, new_radius):
            continue

        # Create a copy with the perturbed circle
        new_centers = centers.copy()
        new_centers[idx] = new_center
        new_radii = radii.copy()
        new_radii[idx] = new_radius

        # Compute penalty of new configuration
        new_penalty = compute_penalty(new_centers, new_radii)

        # Compute penalty of current configuration
        current_penalty = compute_penalty(centers, radii)

        # Acceptance probability
        if new_penalty < current_penalty:
            centers = new_centers
            radii = new_radii
        else:
            delta = new_penalty - current_penalty
            prob = math.exp(-delta / current_temperature)
            if random.random() < prob:
                centers = new_centers
                radii = new_radii

        # Update best solution
        current_sum = total_sum(radii)
        if current_sum > best_sum:
            best_sum = current_sum
            best_centers = centers.copy()
            best_radii = radii.copy()

        # Cool down

        # Optional: early stopping if temperature is too low

    if not is_valid:

        sys.exit(1)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
