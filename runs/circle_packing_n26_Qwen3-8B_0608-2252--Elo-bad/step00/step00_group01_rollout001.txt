import numpy as np
import cvxpy as cp

def run_packing():
    n = 26

    # Define decision variables
    centers = cp.Variable((n, 2))
    radii = cp.Variable(n, non_negative=True)

    # Objective: maximize the sum of radii
    objective = cp.Maximize(cp.sum(radii))

    # Constraints
    constraints = []

    # All circles must be entirely within the unit square
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        constraints += [
            x - r >= 0,
            x + r <= 1,
            y - r >= 0,
            y + r <= 1
        ]

    # No two circles may overlap
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.sqrt(cp.sum_squares(centers[i] - centers[j]))
            constraints += [dist >= radii[i] + radii[j]]

    # Solve the optimization problem
    problem = cp.Problem(objective, constraints)
    problem.solve(warm_start=True)

    # Extract the optimal centers and radii
    centers_opt = centers.value
    radii_opt = radii.value
    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
