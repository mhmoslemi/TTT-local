import numpy as np
import scipy.optimize as optimize
import math

def generate_hexagonal_grid(n, spacing):
    """
    Generate a hexagonal grid of circles.
    """
    rows = int(math.ceil(math.sqrt(n)))
    cols = int(math.ceil(n / rows))
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j * spacing
            y = i * (math.sqrt(3) / 2) * spacing
            if i % 2 == 1:
                x += spacing / 2
            centers.append([x, y])
    return np.array(centers)

def initial_packing(n=26, spacing=0.2):
    """
    Generate an initial packing of n circles in a hexagonal pattern.
    """
    centers = generate_hexagonal_grid(n, spacing)
    radii = np.full(n, 0.05)
    return centers, radii

def objective_function(x, centers, radii):
    """
    Objective function to maximize the sum of radii.
    """
    # x is the vector of parameters (positions and radii)
    n = len(radii)
    if n != len(x) // 2:
        raise ValueError("Mismatch between number of radii and parameters")
    
    # Reshape x to get positions and radii
    x_flat = x.reshape(-1, 2)
    new_centers = x_flat[:n]
    new_radii = x_flat[n:]
    
    # Constraint: sum of radii
    return -np.sum(new_radii)  # Negative for minimization

def constraint_boundary(x, centers, radii):
    """
    Constraint to ensure circles are within the unit square.
    """
    n = len(radii)
    if n != len(x) // 2:
        raise ValueError("Mismatch between number of radii and parameters")
    
    x_flat = x.reshape(-1, 2)
    new_centers = x_flat[:n]
    new_radii = x_flat[n:]
    
    violations = []
    for i in range(n):
        x, y = new_centers[i]
        r = new_radii[i]
        if x - r < -1e-8 or x + r > 1 + 1e-8 or y - r < -1e-8 or y + r > 1 + 1e-8:
            violations.append(1)
    if np.any(violations):
        return -1  # Constraint violated
    return 0

def constraint_overlap(x, centers, radii):
    """
    Constraint to ensure circles do not overlap.
    """
    n = len(radii)
    if n != len(x) // 2:
        raise ValueError("Mismatch between number of radii and parameters")
    
    x_flat = x.reshape(-1, 2)
    new_centers = x_flat[:n]
    new_radii = x_flat[n:]
    
    violations = []
    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((new_centers[i] - new_centers[j]) ** 2))
            if dist < new_radii[i] + new_radii[j] - 1e-8:
                violations.append(1)
    if np.any(violations):
        return -1  # Constraint violated
    return 0

def run_packing():
    """
    Run the circle packing algorithm to find the optimal configuration.
    """
    n = 26
    centers, radii = initial_packing(n=n)
    
    # Flatten the centers and radii into a vector
    x0 = np.vstack((centers, radii)).ravel()
    
    # Define bounds for optimization
    bounds = []
    for i in range(n):
        # Each circle has 2 position coordinates and 1 radius
        bounds.extend([(0, 1), (0, 1), (0, 1)])
    
    # Define constraints
    constraints = [
        {'type': 'ineq', 'fun': lambda x: constraint_boundary(x, centers, radii)},
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers, radii)}
    ]
    
    # Perform optimization
    result = optimize.minimize(
        fun=objective_function,
        x0=x0,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP'
    )
    
    # Extract the optimal centers and radii
    x_opt = result.x
    centers_opt = x_opt[:n*2].reshape(n, 2)
    radii_opt = x_opt[n*2:]
    
    # Validate the packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        raise ValueError(f"Invalid packing: {message}")
    
    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
