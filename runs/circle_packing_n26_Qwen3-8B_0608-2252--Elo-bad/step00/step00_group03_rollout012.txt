import numpy as np
import scipy.optimize as optimize

def is_valid_circle(center, radius):
    x, y = center
    return (x - radius >= -1e-12 and x + radius <= 1 + 1e-12 and
            y - radius >= -1e-12 and y + radius <= 1 + 1e-12)

def is_valid_overlap(center1, radius1, center2, radius2):
    dx = center1[0] - center2[0]
    dy = center1[1] - center2[1]
    dist_squared = dx*dx + dy*dy
    return dist_squared >= (radius1 + radius2)**2 - 1e-12

def objective_function(params, n):
    centers = params[:2*n].reshape(n, 2)
    radii = params[2*n:]
    return -np.sum(radii)  # Maximize sum of radii

def constraint_boundary(params, n):

    radii = params[2*n:]
    for i in range(n):
        center = centers[i]
        radius = radii[i]
        if not is_valid_circle(center, radius):
            return -1
    return 0

def constraint_overlap(params, n):
    centers = params[:2*n].reshape(n, 2)

    for i in range(n):

    return 0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
