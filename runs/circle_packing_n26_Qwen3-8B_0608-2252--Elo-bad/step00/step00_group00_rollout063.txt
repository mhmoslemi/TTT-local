import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Number of circles
    n = 26

    # Initial guess: grid with 5 rows and 6 columns
    # This ensures some spacing and allows for adjustment
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Create a grid-like arrangement
    for i in range(n):
        row = i // 6
        col = i % 6
        x = col / 5.0 + 0.1  # Add offset to allow for better spacing
        y = row / 5.0 + 0.1
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = 0.1

    # Define objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Define constraints:
    # 1. Each circle must be within the unit square
    # 2. The distance between any two circles must be >= sum of their radii

    def constraint1(radii):
        """Ensure each circle is within unit square."""
        r = radii
        for i in range(n):
            x, y = centers[i]
            if x - r[i] < 0 or x + r[i] > 1 or y - r[i] < 0 or y + r[i] > 1:
                return False
        return True

    def constraint2(radii):
        """Ensure no circles overlap."""
        r = radii
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < r[i] + r[j] - 1e-8:
                    return False
        return True

    # Convert constraints to callable for the optimizer
    def constraint_func(radii):
        return [constraint1(radii), constraint2(radii)]

    # Bounds on radii: non-negative
    bounds = [(0.0, 1.0) for _ in range(n)]

    # Initial guess for radii
    initial_guess = 0.1 * np.ones(n)

    # Optimize the radii
    result = optimize.minimize(
        objective,
        initial_guess,
        bounds=bounds,
        constraints={"type": "callback", "fun": constraint_func},
        method="SLSQP"

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
