import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    def generate_initial_guess(n_circles=26):
        rows = int(np.ceil(np.sqrt(n_circles)))
        cols = int(np.ceil(n_circles / rows))
        centers = np.zeros((n_circles, 2))
        idx = 0

            for c in range(cols):
                if idx >= n_circles:
                    break

                y = r / (rows - 1)
                centers[idx] = [x, y]
                idx += 1
        return centers

    # Objective function to minimize (negative sum of radii)
    def objective_function(radii, centers):
        return -np.sum(radii)

    # Constraint 1: All circles must be within the unit square
    def constraint_square(centers, radii):
        return np.sum((centers[:, 0] - radii) < 0) + np.sum((centers[:, 0] + radii) > 1)

    def constraint_overlap(centers, radii):
        n = centers.shape[0]
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1
        return 0

    centers = generate_initial_guess()
    radii = np.full(centers.shape[0], 0.01)  # Small initial radii

    constraints = [
        {'type': 'ineq', 'fun': constraint_square, 'args': (centers, radii)},
        {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers, radii)}

    bounds = [(0, 1) for _ in range(centers.shape[0])]

    result = optimize.minimize(
        fun=objective_function,
        x0=radii,

        jac=None,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
