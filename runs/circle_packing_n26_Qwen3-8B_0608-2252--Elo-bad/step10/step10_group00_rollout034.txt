import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    def hexagonal_positions(n):

        for i in range(n):

            y = np.clip(y, 0, 1)
            positions.append([x, y])

        return np.random.rand(n, 2)

        positions = []
        for i in range(n):
            row = i // 5
            col = i % 5
            x = col * (1.0 / (5 - 1)) + np.random.uniform(-0.01, 0.01)
            y = row * (1.0 / (5 - 1)) + np.random.uniform(-0.01, 0.01)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

        radii = x[2*n:]
        return -np.sum(radii)

        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1 - centers[:, 1] - radii - 1e-6
        ])

    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)

        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]

        return dist
