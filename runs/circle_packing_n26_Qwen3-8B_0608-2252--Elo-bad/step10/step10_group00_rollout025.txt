import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

            row = i // 6

            y = row * (1.0 / (6)) + np.random.uniform(-0.01, 0.01)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

        positions = []
        for i in range(n):
            row = i // 5

            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)

        return np.array(positions)

    def initial_positions(n):

        for i in range(n):
            row = i // 6
            col = i % 6
            x = col * (1.0 / (6 - 1)) + np.random.uniform(-0.01, 0.01)
            y = row * (1.0 / (6 - 1)) + np.random.uniform(-0.01, 0.01)
            x = np.clip(x, 0, 1)
            y = np.clip(y, 0, 1)
            positions.append([x, y])
        return np.array(positions)

    def objective(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return -np.sum(radii)

        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]
        return np.concatenate([
            centers[:, 0] - radii - 1e-6,
            1 - centers[:, 0] - radii - 1e-6,
            centers[:, 1] - radii - 1e-6,
            1 - centers[:, 1] - radii - 1e-6
        ])

    def distance_constraint(x):
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]

        return dist
