import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():
    # Initial grid-based packing configuration
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Initial positions with spacing
    grid = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        grid[i, 0] = col * (1.0 / (cols - 1))
        grid[i, 1] = row * (1.0 / (rows - 1))
    
    # Initial radii (each circle will have radius such that they touch their neighbors)
    # For adjacent circles in the grid, minimal radius for contact
    min_radius = 1.0 / np.sqrt(2) / (cols - 1) * 0.9  # Adjust for safety margin
    radii = np.full(n, min_radius)
    
    # Objective function
    def objective(x):
        return -np.sum(x)  # Minimize negative sum (maximize sum)
    
    # Equality constraints (each circle must be within the unit square)
    def constraint_center(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, cx + r, cy - r, cy + r]  # cx - r >= 0, cx + r <= 1, cy - r >= 0, cy + r <= 1
    
    # Inequality constraints for circle positions
    def constraint_position(x, idx):
        cx = x[2 * idx]
        cy = x[2 * idx + 1]
        r = x[2 * n + idx]
        return [cx - r, 1 - cx - r, cy - r, 1 - cy - r]  # cx - r >= 0, 1 - cx - r >= 0, cy - r >= 0, 1 - cy - r >= 0
    
    # Inequality constraints for non-overlapping between circles
    def constraint_overlap(x, i, j):
        cx_i = x[2 * i]
        cy_i = x[2 * i + 1]
        r_i = x[2 * n + i]
        cx_j = x[2 * j]
        cy_j = x[2 * j + 1]
        r_j = x[2 * n + j]
        dist = np.sqrt((cx_i - cx_j)**2 + (cy_i - cy_j)**2)
        return dist - (r_i + r_j)  # dist >= r_i + r_j
    
    # Define variables: 2n for centers, n for radii
    bounds = [(0.0, 1.0) for _ in range(2 * n)] + [(0.0, 1.0) for _ in range(n)]
    
    # Initial guess for optimization
    x0 = np.concatenate([grid.flatten(), radii])
    
    # Set up constraints
    constraints = []
    
    # Position constraints (each circle within the square)
    for i in range(n):
        c = {'type': 'ineq', 'fun': lambda x, idx=i: constraint_position(x, idx)}
        constraints.append(c)
    
    # Non-overlapping constraints
    for i in range(n):
        for j in range(i + 1, n):
            c = {'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)}
            constraints.append(c)
    
    # Run optimization
    result = minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-8)
    
    # Post-process result
    centers = result.x[:2 * n].reshape(n, 2)
    radii = result.x[2 * n:]
    
    # Validate the solution
    valid, msg = validate_packing(centers, radii)
    if not valid:
        raise ValueError(f"Optimization failed: {msg}")
    
    return centers, radii, np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: 5 rows with 5 or 6 circles
    num_circles = 26
    rows = 5
    cols = 6
    centers = np.zeros((num_circles, 2))
    
    # Initial placement in a grid
    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if idx >= num_circles:
                break
            x = j / (cols - 1)
            y = i / (rows - 1)
            centers[idx] = [x, y]

    # Initial guess for radii (uniform)
    radii = np.ones(num_circles) * 0.1
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Bounds for radii
    bounds = Bounds(0, 1.0)

    # Define constraints
    def constraint_boundary(x):
        # Check if circle is within unit square
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = x[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        # Check if any circles overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                x_i, y_i = centers[i]
                r_i = x[i]
                x_j, y_j = centers[j]
                r_j = x[j]
                dist_sq = (x_i - x_j) ** 2 + (y_i - y_j) ** 2
                if dist_sq < (r_i + r_j) ** 2 - 1e-12:
                    return -1
        return 0

    # Nonlinear constraints
    cons = [
        NonlinearConstraint(constraint_boundary, -np.inf, 0),
        NonlinearConstraint(constraint_overlap, -np.inf, 0)
    ]

    # Set up optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    if not result.success:
        print("Optimization failed:", result.message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    optimized_radii = result.x
    sum_radii = np.sum(optimized_radii)

    # Validate the solution
    valid, message = validate_packing(centers, optimized_radii)
    if not valid:
        print("Validation failed:", message)
        return np.zeros((26, 2)), np.zeros(26), 0.0

    return centers, optimized_radii, sum_radii
