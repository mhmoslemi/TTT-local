import numpy as np
import scipy.optimize as opt

def run_packing():
    # Hexagonal grid parameters
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))

    # Initial grid spacing (adjust accordingly based on hexagonal packing)
    spacing = 1.0 / (cols + 1)
    offset = spacing / 2
    centers = np.zeros((num_circles, 2))
    radius = np.ones(num_circles) * 0.05  # Initial guess for radii
    
    idx = 0
    for row in range(rows):
        for col in range(cols):
            x = col * spacing + offset
            y = row * spacing * np.sqrt(3) / 2 + offset
            if idx < num_circles:
                centers[idx] = [x, y]
                idx += 1

    # Optimization objective: minimize sum of squared radii (equivalent to maximizing sum of radii under constraint sum(r) = constant)
    def optimize_objective(x):
        global centers, radius
        new_centers = x[:2*num_circles].reshape(num_circles, 2)
        new_radii = x[2*num_circles:]
        sum_radii = np.sum(new_radii)
        penalty = 0
        # Constraint: All circles must be within the square
        for i in range(num_circles):
            if new_centers[i, 0] - new_radii[i] < 0 or new_centers[i, 0] + new_radii[i] > 1:
                penalty += 1e6
            if new_centers[i, 1] - new_radii[i] < 0 or new_centers[i, 1] + new_radii[i] > 1:
                penalty += 1e6
        # Constraint: No overlaps
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dist = np.linalg.norm(new_centers[i] - new_centers[j])
                if dist < new_radii[i] + new_radii[j] - 1e-8:
                    penalty += 1e6
        # Objective: Minimize sum of squared radii (equivalent to maximizing sum of radii due to constraint)
        return np.sum(new_radii ** 2) + penalty

    # Bounds on positions and radii
    bounds = []
    for i in range(num_circles):
        # Bounds for (x, y) coordinates
        x_bounds = (0.0, 1.0)
        y_bounds = (0.0, 1.0)
        bounds.extend([x_bounds, y_bounds])
        # Bounds for radius
        radius_bounds = (0.0, 0.5)
        bounds.extend([radius_bounds])

    # Initial guess
    initial_guess = np.concatenate([centers.flatten(), radius])

    # Run optimization
    result = opt.minimize(optimize_objective, initial_guess, bounds=bounds, method='L-BFGS-B', tol=1e-8)

        optimized_radii = radius
        sum_radii = np.sum(optimized_radii)

    return optimized_centers, optimized_radii, sum_radii
