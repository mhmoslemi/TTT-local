def generate_hexagonal_positions(n, spacing, offset_x, offset_y):
    positions = []
    for i in range(n):

        y = row * spacing + (col % 2) * spacing * 0.5 + offset_y
        positions.append([x, y])
    return np.array(positions)

        level = i // 5

        x = offset * spacing + offset_x
        y = level * spacing + offset_y

    return np.array(positions)
