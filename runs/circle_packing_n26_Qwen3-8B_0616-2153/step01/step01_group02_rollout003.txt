import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial grid layout: 5 rows and 6 columns (for 26 circles)
    num_circles = 26
    rows, cols = 5, 6  # 5x6 grid (30 slots) to leave room for optimization
    spacing = 1.0 / (cols - 1)  # spacing between centers
    grid_x = np.linspace(0.5 * spacing, 1 - 0.5 * spacing, cols)
    grid_y = np.linspace(0.5 * spacing, 1 - 0.5 * spacing, rows)
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.05  # initial guess for radii

    # Assign positions (first 26 of 30 grid points)
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        centers[i] = [grid_x[col], grid_y[row]]

    # Define objective function: minimize negative sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Define constraints
    def constraint1(x):
        # Boundary constraints
        return np.array([x[0] - x[2] >= -1e-8,  # x - r >= 0
                         1 - x[0] - x[2] >= -1e-8,  # 1 - x - r >= 0
                         x[1] - x[2] >= -1e-8,  # y - r >= 0

    # Define non-overlap constraints
    def constraint2(x):
        # x is a flat array of [x1, y1, r1, x2, y2, r2, ..., x26, y26, r26]
        N = num_circles
        constraints = []
        for i in range(N):

                idx_j = 2 * j

                x_j, y_j, r_j = x[idx_j], x[idx_j + 1], x[idx_j + 2]
                dist = np.sqrt((x_i - x_j)**2 + (y_i - y_j)**2)
                constraints.append(dist - r_i - r_j + 1e-8)
        return np.array(constraints)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
