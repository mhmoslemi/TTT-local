import scipy.optimize as opt
from scipy.optimize import Bounds

    # So we will take the first 26 points

    centers = []

            x = j + 0.5 * i  # offset for hexagonal packing
            y = i * np.sqrt(3) / 2  # vertical spacing for hexagonal packing
            centers.append([x, y])

    centers = np.array(centers[:26])  # take the first 26 points

    # Initial radii estimate
    # We estimate the initial radius based on the unit square and the number of circles
    # Using a rough estimate for the area per circle

    # Area of a circle = πr², so r ≈ sqrt(0.0385 / π) ≈ 0.109

        # x is a vector containing flattened centers and radii

        x_radii = x[2 * n_circles:]

        sum_radii = -np.sum(x_radii)

        # For each circle, ensure x_center[i] - r[i] >= 0, x_center[i] + r[i] <= 1

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
