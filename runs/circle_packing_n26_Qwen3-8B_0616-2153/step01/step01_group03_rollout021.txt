import numpy as np
from scipy.optimize import minimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Initial guess using hexagonal packing
    def hexagonal_packing(n_circles):
        # Grid-based hexagonal packing
        rows = int(np.sqrt(n_circles))
        cols = n_circles // rows + 1
        centers = np.zeros((n_circles, 2))
        for i in range(n_circles):
            row = i // cols
            col = i % cols
            x = col + 0.5 * (row % 2)
            y = row * np.sqrt(3) / 2
            if y > 1:
                y -= np.sqrt(3) / 2
            centers[i] = [x, y]
        # Normalize to unit square
        centers /= max(centers.max(), 1)
        return centers

    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_overlap(i, j, centers, radii):
        def constraint(x):
            r_i = radii[i]
            r_j = radii[j]
            dx = centers[i, 0] - x[0]
            dy = centers[i, 1] - x[1]
            dist_sq = dx**2 + dy**2
            return dist_sq - (r_i + r_j)**2
        return constraint

    def constraint_boundary(i, centers, radii):
        def constraint(x):
            r = radii[i]
            dx = x[0] - r
            dy = x[1] - r
            return dx if dx < 0 else dy
        return constraint

    # Initial setup
    n_circles = 26
    centers_init = hexagonal_packing(n_circles)
    radii_init = np.full(n_circles, 0.1)  # Initial guess for radii

    # Setup optimization problem
    bounds = [(0, 1) for _ in range(n_circles * 2)]

    # Optimization
    result = minimize(
        fun=objective,
        x0=np.concatenate([centers_init.flatten(), radii_init]),
        bounds=bounds,
        method='SLSQP',
        constraints=[
            constraint_overlap(i, j, centers_init, radii_init) for i in range(n_circles) for j in range(i + 1, n_circles)
        ] + [
            constraint_boundary(i, centers_init, radii_init) for i in range(n_circles)
        ]
    )

    # Extract results
    optimized = result.x
    centers_opt = optimized[:n_circles * 2].reshape(n_circles, 2)
    radii_opt = optimized[n_circles * 2:]

    if not valid:
        raise ValueError(msg)

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
