from scipy.optimize import minimize
from scipy.optimize import Bounds
import sys

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    return True, "ok"

def objective_function(params, n_circles):
    # params = [x1, y1, r1, x2, y2, r2, ..., xn, yn, rn]
    centers = params[:2*n_circles].reshape(n_circles, 2)
    radii = params[2*n_circles:].reshape(n_circles)
    return -np.sum(radii)  # Minimize negative of sum to maximize sum

def constraint_boundary(params, n_circles):
    centers = params[:2*n_circles].reshape(n_circles, 2)
    radii = params[2*n_circles:].reshape(n_circles)
    for i in range(n_circles):
        x, y = centers[i]
        r = radii[i]
        if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
            return -1e-6  # Constraint violation
    return 0.0

def constraint_overlap(params, n_circles):
    centers = params[:2*n_circles].reshape(n_circles, 2)
    radii = params[2*n_circles:].reshape(n_circles)
    for i in range(n_circles):
