import scipy.optimize as optimize

def run_packing():

    n = 26
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col + 0.5 * (col + 1) / num_cols
        y = row + 0.5 * (row + 1) / num_rows
        centers[i] = [x, y]
        radii[i] = 0.05 * (1.0 / (num_rows + num_cols))  # Initial radii
    
    # Optimization bounds
    bounds = [(0.0, 1.0) for _ in range(2 * n)]  # For centers
    bounds += [(0.0, 1.0) for _ in range(n)]     # For radii
    
    # Objective function to maximize the sum of radii
    def objective(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:]
        return -np.sum(radii_flat)
    
    # Constraints
    def constraint_center_bounds(x):
        return x[:2 * n] - np.array([0.0] * (2 * n))
    
    def constraint_radius_bounds(x):
        return x[2 * n:] - np.array([0.0] * n)
    
    def constraint_overlap(x):
        centers_flat = x[:2 * n].reshape(n, 2)
        radii_flat = x[2 * n:]
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers_flat[i] - centers_flat[j])
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    return dist - (radii_flat[i] + radii_flat[j] - 1e-8)
        return 1.0
    
    # Create constraints for optimization
    constraints = [

        {'type': 'ineq', 'fun': constraint_radius_bounds},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Run optimization
    initial_guess = np.concatenate([centers.flatten(), radii])
    result = optimize.minimize(
