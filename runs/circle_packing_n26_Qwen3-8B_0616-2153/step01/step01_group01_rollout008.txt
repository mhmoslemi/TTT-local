import numpy as np
import scipy.optimize as opt
from scipy.spatial.distance import cdist

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess for hexagonal packing
    def hexagonal_centers(n_circles):
        rows = int(np.ceil(np.sqrt(n_circles)))
        cols = int(np.ceil(np.sqrt(n_circles)))
        centers = []
        for i in range(rows):
            for j in range(cols):
                x = j + 0.5 * i
                y = i
                if x < 1.0 and y < 1.0:
                    centers.append([x, y])
        return np.array(centers)

    # Initial guess
    n_circles = 26
    initial_centers = hexagonal_centers(n_circles)
    initial_radii = np.full(n_circles, 0.05)  # Small initial radii

    def within_bounds(centers, radii):

            centers = x[:2*n_circles].reshape(n_circles, 2)

            for i in range(n_circles):
                x_i, y_i = centers[i]
                r_i = radii[i]
                if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                    return -1
            return 0
        return bounds_constraint

    # Define constraints: no two circles overlap
    def no_overlap(centers, radii):
        def overlap_constraint(x):
            centers = x[:2*n_circles].reshape(n_circles, 2)

            return 0

    def total_radius(x):

        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # - Next n_circles: radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"
