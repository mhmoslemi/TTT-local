import numpy as np
import scipy.optimize as optimize

def run_packing():

    n = 26
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    # Initialize centers for hexagonal packing

    for i in range(n):
        row = i // num_cols
        col = i % num_cols

    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum
    
    # Bounds for radii (non-negative)
    bounds = [(0.0, 1.0) for _ in range(n)]
    
    # Constraints

        return np.array([0.0] * n)  # Ensure radii are within bounds
