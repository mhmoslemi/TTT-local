import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # First row: centered at (0.5, 0.2)
    # Subsequent rows are spaced with vertical spacing of sqrt(3)/2 times the diameter
    radius_initial = 0.1  # Initial radius
    num_rows = 6
    num_circles_per_row = 5

    centers = []
    radii = []

    # Initial hexagonal packing setup
    row_height = np.sqrt(3) / 2 * 2 * radius_initial
    for i in range(num_rows):
        row = []
        for j in range(num_circles_per_row):
            x = 0.5 + (j - (num_circles_per_row - 1) / 2) * 2 * radius_initial
            y = 0.2 + i * row_height
            row.append([x, y])
        centers.append(row)
    
    # Flatten the list of centers
    centers = np.array(centers).flatten().reshape(-1, 2)
    radii = np.full(centers.shape[0], radius_initial)

    # Define the optimization variables
    # We'll optimize the radii and centers simultaneously
    # Variables will be flattened into a single vector

    def objective(x):
        # Reshape x into centers and radii
        size = centers.shape[0]
        centers_flat = x[:2 * size].reshape(size, 2)
        radii_flat = x[2 * size:].reshape(size)
        return -np.sum(radii_flat)  # Minimize the negative sum to maximize sum

    def constraint_inside(x):
        size = centers.shape[0]
        centers_flat = x[:2 * size].reshape(size, 2)
        radii_flat = x[2 * size:].reshape(size)
        for i in range(size):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return 1e6  # Penalize if any circle is outside the square
        return 0

    def constraint_overlap(x):

        centers_flat = x[:2 * size].reshape(size, 2)
        radii_flat = x[2 * size:].reshape(size)

                x_j, y_j = centers_flat[j]

                r_j = radii_flat[j]

                    return 1e6  # Penalize if circles overlap
        return 0

    x0 = np.concatenate([centers.flatten(), radii.flatten()])

        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    for _ in range(centers.shape[0]):
        bounds.append((0, 1))  # radius

    result = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
