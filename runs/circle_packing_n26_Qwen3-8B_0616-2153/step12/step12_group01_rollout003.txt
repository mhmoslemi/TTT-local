from typing import Tuple, List, Any

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def objective_function(params, n_circles):
    centers = params[:n_circles * 2].reshape((n_circles, 2))
    radii = params[n_circles * 2:].reshape((n_circles,))
    return -np.sum(radii)

def constraint_function(params, n_circles):

def generate_hexagonal_centers(n_circles, margin=0.1, spacing=0.25):

                centers.append([x, y])
    return np.array(centers)

def generate_corner_centers(n_circles, margin=0.1):

    centers = []
    for _ in range(n_circles):
        x, y = np.random.choice(corners, 1)[0]
        centers.append([x, y])
    return np.array(centers)

def generate_random_centers(n_circles, margin=0.1):
    centers = []
    for _ in range(n_circles):
        x = np.random.uniform(margin, 1 - margin)
        y = np.random.uniform(margin, 1 - margin)
        centers.append([x, y])
    return np.array(centers)

    combined = np.vstack([hex_centers, random_centers, corner_centers])
    if combined.shape[0] > n_circles:

    return combined
