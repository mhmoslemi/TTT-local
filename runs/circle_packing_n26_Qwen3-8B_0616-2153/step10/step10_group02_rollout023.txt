def generate_hexagonal_centers(n_circles, square_size=(1, 1), seed=None):
    if seed is not None:
        random.seed(seed)

    cols = int(np.ceil(n_circles / rows))
    centers = []
    for i in range(rows):

            x = j / (cols - 1) + 0.1

            if (i % 2 == 0):
                x += 0.5 / (cols - 1)
            centers.append([x, y])
    return np.array(centers)

def compute_initial_radii(centers):

                if dist < min_dist:

        radii[i] = min_dist / 2.0
    return radii
