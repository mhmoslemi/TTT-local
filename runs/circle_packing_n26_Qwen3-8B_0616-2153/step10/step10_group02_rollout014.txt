            random.seed(seed)

                # Corner placement
                x, y = random.choice([(0, 0), (0, 1), (1, 0), (1, 1)])
                centers.append([x + 0.05, y + 0.05])

                    y = random.choice([0.05, 0.95])
                else:
                    y = random.uniform(0.05, 0.95)
                    x = random.choice([0.05, 0.95])
                centers.append([x, y])
            else:
                # Interior placement
                x = random.uniform(0.1, 0.9)
                y = random.uniform(0.1, 0.9)
                centers.append([x, y])
        return np.array(centers)

    def generate_hexagonal_centers(n_circles, square_size=(1, 1), seed=None):

            random.seed(seed)
        rows = int(np.ceil(np.sqrt(n_circles)))
        cols = int(np.ceil(n_circles / rows))

            for j in range(cols):
                x = j / (cols - 1) + 0.1
                y = i / (rows - 1) + 0.1
                if (i % 2 == 0):
                    x += 0.5 / (cols - 1)
                centers.append([x, y])
        return np.array(centers)

            random.seed(seed)

                # Place near the left or right
                y = random.uniform(0.1, 0.9)
                x = random.choice([0.1, 0.9])
            centers.append([x, y])
        return np.array(centers)

        n = len(centers)
        radii = np.zeros(n)
        for i in range(n):
            min_dist = float('inf')
            for j in range(n):
                if i != j:
                    dist = np.hypot(centers[i, 0] - centers[j, 0], centers[i, 1] - centers[j, 1])
                    if dist < min_dist:
                        min_dist = dist
            radii[i] = min_dist / 2.0
        return radii
