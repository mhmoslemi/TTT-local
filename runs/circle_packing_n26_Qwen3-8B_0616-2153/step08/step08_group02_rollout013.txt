import numpy as np
import scipy.optimize as optimize
import random
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    def validate_packing(centers, radii):
        n = centers.shape[0]

        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"

        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

    def objective_function(params, n_circles):
        centers = params[:n_circles * 2].reshape((n_circles, 2))
        radii = params[n_circles * 2:].reshape((n_circles,))
        return -np.sum(radii)

    def constraint_function(params, n_circles):
        centers = params[:n_circles * 2].reshape((n_circles, 2))
        radii = params[n_circles * 2:].reshape((n_circles,))

        constraints = []

        # Boundary constraints
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append((x - r + 1e-12))
            constraints.append((1 - x - r + 1e-12))
            constraints.append((y - r + 1e-12))
            constraints.append((1 - y - r + 1e-12))

        # Circle-circle constraints
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append(dist - (radii[i] + radii[j]) + 1e-12)

        return constraints

    def generate_initial_centers(n_circles):
        np.random.seed(42)
        centers = []

        # Hexagonal grid
        grid_rows = int(math.sqrt(n_circles))
        grid_cols = grid_rows
        spacing = 0.2
        for i in range(grid_rows):
            for j in range(grid_cols):
                x = j * spacing + 0.2
                y = i * spacing * np.sqrt(3) + 0.2
                if (x <= 0.9 and y <= 0.9 and x >= 0.1 and y >= 0.1):
                    centers.append([x, y])

        remaining = n_circles - len(centers)

            x = np.random.uniform(0.2, 0.8)

            centers.append([x, y])

        if len(centers) > n_circles:
            centers = centers[:n_circles]
        return centers

            return centers, radii
        max_radius = 0.15
        radii = np.clip(radii * (max_radius / np.max(radii)), 0, max_radius)
        return centers, radii

    def optimize_with_local_refinement(n_circles, n_iterations=10):
        best_centers = np.array([])
        best_radii = np.array([])

                method="SLSQP"
