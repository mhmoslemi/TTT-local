import numpy as np
import scipy.optimize as optimize
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    def validate_packing(centers, radii):
        n = centers.shape[0]

        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"

        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

    def objective_function(params, n_circles):
        centers = params[:n_circles * 2].reshape((n_circles, 2))
        radii = params[n_circles * 2:].reshape((n_circles,))
        return -np.sum(radii)

    def constraint_function(params, n_circles):
        centers = params[:n_circles * 2].reshape((n_circles, 2))
        radii = params[n_circles * 2:].reshape((n_circles,))

        constraints = []

        # Boundary constraints
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            # x - r >= 0
            constraints.append((x - r) * 1e5)
            # x + r <= 1
            constraints.append((1 - x - r) * 1e5)
            # y - r >= 0
            constraints.append((y - r) * 1e5)
            # y + r <= 1
            constraints.append((1 - y - r) * 1e5)

        # Circle-circle constraints
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                constraints.append((dist - (radii[i] + radii[j])) * 1e5)

        return constraints

        centers = []
        grid_rows = int((1 - 2 * margin) / (0.25 * np.sqrt(3)))
        grid_cols = int((1 - 2 * margin) / 0.25)
        for i in range(grid_rows):
            for j in range(grid_cols):
                x = j * 0.25 + margin
                y = i * 0.25 * np.sqrt(3) + margin
                if (x <= 1 - margin and y <= 1 - margin and x >= margin and y >= margin):
                    centers.append([x, y])
        return np.array(centers)

        centers = []
        for _ in range(n_circles):
            x = np.random.choice([margin, 1 - margin])
            y = np.random.choice([margin, 1 - margin])
            centers.append([x, y])
        return np.array(centers)

        centers = []
        for _ in range(n_circles):
            x = np.random.uniform(margin, 1 - margin)
            y = np.random.uniform(margin, 1 - margin)
            centers.append([x, y])
        return np.array(centers)

            np.random.seed(seed)

        # Generate initial configurations
        initial_centers = generate_hexagonal_centers(n_circles)

            bounds.append((0, 1))

            bounds.append((0, 1))

            options={'maxiter': 1000}

        radii_opt = result.x[n_circles * 2:].reshape((n_circles,))
