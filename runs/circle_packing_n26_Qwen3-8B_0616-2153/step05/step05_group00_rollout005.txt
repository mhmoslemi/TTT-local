import numpy as np
import scipy.optimize as optimize
import random
from typing import Tuple, List, Any
from deap import base, creator, tools, algorithms

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    def validate_packing(centers, radii):

        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"
        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-10 or x + r > 1 + 1e-10 or
                y - r < -1e-10 or y + r > 1 + 1e-10):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-10:
                    return False, f"Circles {i} and {j} overlap"
        return True, "ok"

    def objective_function(circles):
        return -np.sum(circles["radii"])

    def update_circles(individual):
        circles = {
            "centers": individual[:26*2].reshape((26, 2)),
            "radii": individual[26*2:].reshape((26,))
        }
        return circles

    def fitness(individual):
        circles = update_circles(individual)
        if not validate_packing(circles["centers"], circles["radii"])[0]:
            return -np.inf,  # Invalid configuration
        return -np.sum(circles["radii"]),  # Maximize sum of radii

    def gen_initial_guess(n_circles=26):
        centers = []
        radii = []

        rows = int(np.ceil(np.sqrt(n_circles)))
        cols = int(np.ceil(n_circles / rows))

        spacing = 1.0 / (np.sqrt(3) + 1)
        for row in range(rows):
            for col in range(cols):
                x = 0.5 + (col - (cols - 1)/2) * spacing
                y = 0.5 + (row - (rows - 1)/2) * spacing * np.sqrt(3)/2
                if 0 <= x <= 1 and 0 <= y <= 1:
                    centers.append([x, y])
                    radii.append(0.02 + random.uniform(0, 0.1))

        # Add some circles near the corners
        for i in range(4):
            x = [0.1, 0.9][i % 2]
            y = [0.1, 0.9][i // 2]
            centers.append([x, y])
            radii.append(0.05 + random.uniform(0, 0.1))

        # Fill remaining circles in a grid
        while len(centers) < n_circles:
            x = np.random.uniform(0.1, 0.9)
            y = np.random.uniform(0.1, 0.9)
            centers.append([x, y])
            radii.append(0.02)

        return np.array(centers), np.array(radii)

    def genetic_algorithm(n_circles=26):
        creator.create("FitnessMax", base.Fitness, weights=(1.0,))
        creator.create("Individual", list, fitness=creator.FitnessMax)

        toolbox = base.Toolbox()
        toolbox.register("center", random.uniform, 0.1, 0.9)
        toolbox.register("radius", random.uniform, 0.02, 0.12)

        # Create initial population
        toolbox.register("individual", tools.initRepeat, creator.Individual, 
                         lambda: [toolbox.center() for _ in range(2)] + [toolbox.radius() for _ in range(n_circles)])
        toolbox.register("population", tools.initRepeat, list, toolbox.individual)

        # Fitness and selection
        toolbox.register("evaluate", fitness)
        toolbox.register("mate", tools.cxTwoPoint)
        toolbox.register("mutate", tools.mutGaussian, mu=0, sigma=0.05, indpb=0.1)
        toolbox.register("select", tools.selTournament, tournsize=3)

        pop = toolbox.population(n=50)
        stats = tools.Statistics(lambda ind: ind.fitness.values)
        stats.register("avg", np.mean)
        stats.register("std", np.std)
        stats.register("min", np.min)
        stats.register("max", np.max)

        hof = tools.HallOfFame(1)
        hof.update(pop)
        best = hof[0]
        centers = best[:26*2].reshape((26, 2))
        radii = best[26*2:].reshape((26,))
        return centers, radii

        n_positions = n_circles * 2
        n_radii = n_circles

        initial_params = np.concatenate([centers.flatten(), radii])

        for _ in range(n_positions):
            bounds.append((0, 1))
        for _ in range(n_radii):
            bounds.append((0, 1))

        def objective_function(params):
            centers = params[:n_circles * 2].reshape((n_circles, 2))

            return -np.sum(radii)

        def constraint_function(params):
            centers = params[:n_circles * 2].reshape((n_circles, 2))
            radii = params[n_circles * 2:].reshape((n_circles,))

                x, y = centers[i]
                r = radii[i]
                constraints.append(x - r)
                constraints.append(1 - x - r)
                constraints.append(y - r)
                constraints.append(1 - y - r)

                    constraints.append(dist - (radii[i] + radii[j]))

            return constraints

        result = optimize.minimize(
            fun=objective_function,
            x0=initial_params,
            bounds=bounds,
            constraints={"type": "ineq", "fun": constraint_function},
            method="SLSQP",
            options={"maxiter": 1000, "ftol": 1e-10, "gtol": 1e-10, "disp": False}

            return centers, radii

        sum_radii = np.sum(radii_opt)

            print(f"Validation failed: {message}")
            return centers, radii

        return centers_opt, radii_opt, sum_radii

    centers, radii = gen_initial_guess()
    initial_sum = np.sum(radii)
    print("Initial sum of radii:", initial_sum)

    best_centers, best_radii = genetic_algorithm()
    genetic_sum = np.sum(best_radii)
    print("Genetic algorithm sum of radii:", genetic_sum)

    centers_opt, radii_opt, sum_radii = local_optimization(best_centers, best_radii)

    print("Final sum of radii:", sum_radii)
    return centers_opt, radii_opt, sum_radii
