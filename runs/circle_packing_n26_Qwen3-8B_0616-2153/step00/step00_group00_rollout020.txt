import scipy.optimize as optimize

def run_packing():

    grid_size = 5

    for i in range(grid_size):
        for j in range(grid_size):
            x = grid_center[0] - grid_spacing * (grid_size / 2) + j * grid_spacing
            y = grid_center[1] - grid_spacing * (grid_size / 2) + i * grid_spacing
            centers.append([x, y])
            radii.append(0.05)

    centers = np.array(centers)
    radii = np.array(radii)

        return -np.sum(radii_flat)  # Negative for minimization

        for i in range(26):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12:
