import cvxpy as cp

def run_packing():

    n = 26

    def hexagonal_lattice(n):

                # Hexagonal lattice coordinates
                x = j + 0.5 * i

        np.random.shuffle(centers)
        return np.array(centers)

    centers = hexagonal_lattice(n)

    radii = cp.Variable(n)

    # Objective: maximize sum of radii

    # Constraints
    constraints = []

    # Boundary constraint: each circle is inside the unit square
    for i in range(n):

        constraints.append(x - radii[i] >= 0)
