import numpy as np
import scipy.optimize as opt
import cvxpy as cp

def run_packing():
    # Initial guess: place circles in a grid-like pattern
    n = 26
    centers = np.random.rand(n, 2)
    radii = np.full(n, 0.1)  # Initial small radius
    
    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Negative for minimization

    # Decision variables
    centers_var = cp.Variable((n, 2))
    radii_var = cp.Variable(n, non_negative=True)

    # Constraints
    constraints = []
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.norm(centers_var[i] - centers_var[j])
            constraints.append(dist >= radii_var[i] + radii_var[j] - 1e-6)
        # Boundary constraints
        constraints.append(centers_var[i][0] - radii_var[i] >= -1e-6)
        constraints.append(centers_var[i][0] + radii_var[i] <= 1 + 1e-6)
        constraints.append(centers_var[i][1] - radii_var[i] >= -1e-6)
        constraints.append(centers_var[i][1] + radii_var[i] <= 1 + 1e-6)
    
    # Solve the optimization problem
    prob = cp.Problem(cp.Maximize(objective(radii_var)), constraints)
    prob.solve(solver=cp.SCS, max_iter=1000)

    radii_opt = radii_var.value

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            if dist < radii[i] + radii[j] - 1e-12:

    return True, "ok"
