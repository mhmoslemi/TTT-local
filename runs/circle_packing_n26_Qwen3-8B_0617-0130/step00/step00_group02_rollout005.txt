    n = 26
    grid_rows = 5
    grid_cols = 6  # More than 5 to fit 26 circles

        y = row / (grid_rows)
        centers[i] = np.array([x, y])

    # Objective: maximize sum of radii
    # Constraints:
    # - Each circle must be fully within the square
    # - Circles cannot overlap
    # - All radii are non-negative

    radii = np.ones(n) * 0.05  # Start with small radii

    def objective(x):
        return -sum(x)  # Negative because we want to minimize the negative of the sum

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
