import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 10 circles
    centers = np.zeros((10, 2))
    for i in range(10):
        centers[i, 0] = (i % 5) / 4.0 + 0.25
        centers[i, 1] = (i // 5) / 4.0 + 0.25
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # All circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        # No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=[(0, 1)] * 10,
        constraints=[{'type': 'ineq', 'fun': constraints}]
    )

    # Extract results
    centers = np.array([[0.25, 0.25], [0.75, 0.25], [0.25, 0.75], [0.75, 0.75],
                        [0.5, 0.1], [0.5, 0.9], [0.1, 0.5], [0.9, 0.5],
                        [0.3, 0.3], [0.7, 0.7]])
    radii = result.x
    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
