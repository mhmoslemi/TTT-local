import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: place circles at grid points with equal spacing
    centers = np.array([[i * 0.25, j * 0.25] for i in range(10) for j in range(10)])
    centers = centers[:10]  # Keep first 10 circles
    radii = np.full(10, 0.1)  # Initial radius

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Constraint 1: All circles are inside the unit square
        for i in range(10):
            if radii[i] < 0:
                return False
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return False
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return False
        # Constraint 2: No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j]:
                    return False
        return True

    # Optimization setup
    bounds = [(0.0, 0.5) for _ in range(10)]  # Radii can't exceed 0.5
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints=lambda x: {'type': 'ineq', 'fun': lambda x: constraints(x)})

    if not result.success:
        return (centers, radii, np.sum(radii))

    final_centers = centers.copy()
    final_radii = result.x
    sum_radii = np.sum(final_radii)

    return (final_centers, final_radii, sum_radii)
