import numpy as np
from scipy.optimize import minimize
import random

def run_packing():
    # Initial guess: place 10 circles at random positions
    np.random.seed(42)
    centers = np.random.uniform(low=0.1, high=0.9, size=(10, 2))
    radii = np.full(10, 0.1)

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        # Ensure all circles are within boundaries
        for i in range(10):
            if radii[i] < 0:
                return np.inf
            if centers[i, 0] - radii[i] < 0 or centers[i, 0] + radii[i] > 1:
                return np.inf
            if centers[i, 1] - radii[i] < 0 or centers[i, 1] + radii[i] > 1:
                return np.inf
        # Ensure no overlaps
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j]:
                    return np.inf
        return 0

    # Use a bounds-based approach for radius optimization
    bounds = [(0, 0.5) for _ in range(10)]
    result = minimize(objective, radii, method='SLSQP', bounds=bounds, constraints={'type': 'ineq', 'fun': constraints})

    # Use the optimized radii and centers
    optimized_radii = result.x
    # Adjust centers slightly to avoid overlap if needed
    # (simple heuristic to prevent overlaps with small adjustments)
    for i in range(10):
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < optimized_radii[i] + optimized_radii[j]:
                # Move the circles apart slightly
                unit_vec = np.array([dx, dy]) / dist
                move = 0.01 * (optimized_radii[i] + optimized_radii[j] - dist)
                centers[i] += move * unit_vec
                centers[j] -= move * unit_vec

    # Final check to ensure all circles are within bounds
    for i in range(10):
        if centers[i, 0] - optimized_radii[i] < 0 or centers[i, 0] + optimized_radii[i] > 1:
            optimized_radii[i] = min(optimized_radii[i], 1 - centers[i, 0])
            optimized_radii[i] = min(optimized_radii[i], centers[i, 0])
        if centers[i, 1] - optimized_radii[i] < 0 or centers[i, 1] + optimized_radii[i] > 1:
            optimized_radii[i] = min(optimized_radii[i], 1 - centers[i, 1])
            optimized_radii[i] = min(optimized_radii[i], centers[i, 1])

    sum_radii = np.sum(optimized_radii)
    return (centers, optimized_radii, sum_radii)
