import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess: 10 circles at regular grid positions
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    centers = centers[:10]
    radii = np.ones(10) * 0.1

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii):
        centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])[:10]
        centers = np.vstack([centers, np.random.rand(10, 2) * 0.5 + 0.25])
        return np.sum(radii)

    def bounds(radii):
        return [(0, 1) for _ in range(10)]

    def inequality_constraints(radii):
        centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])[:10]
        centers = np.vstack([centers, np.random.rand(10, 2) * 0.5 + 0.25])
        res = []
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                res.append(dist - radii[i] - radii[j])
        return res

    # Optimization
    result = minimize(
        objective,
        radii,
        bounds=bounds(radii),
        constraints={"type": "ineq", "fun": lambda x: inequality_constraints(x)},
        method="SLSQP"
    )

    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])[:10]
    centers = np.vstack([centers, np.random.rand(10, 2) * 0.5 + 0.25])
    radii = result.x

    # Final check for constraints
    for i in range(10):
        if radii[i] > 1:
            radii[i] = 1
        for j in range(i + 1, 10):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx ** 2 + dy ** 2)
            if dist < radii[i] + radii[j]:
                radii[i] = dist - radii[j]
                radii[j] = dist - radii[i]

    sum_radii = np.sum(radii)
    return (centers, radii, sum_radii)
