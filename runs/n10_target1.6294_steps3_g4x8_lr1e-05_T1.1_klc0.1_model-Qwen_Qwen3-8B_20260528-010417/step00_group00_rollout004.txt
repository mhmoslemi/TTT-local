import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess: placing circles in a grid
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.full(10, 0.05)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(radii):
        centers = np.zeros((10, 2))
        for i in range(10):
            centers[i, 0] = 0.1 + (i // 5) * 0.2
            centers[i, 1] = 0.1 + (i % 5) * 0.2
        return np.sum(radii)

    def bounds(radii):
        return [(0, 1 - 0.1) for _ in range(10)]  # Ensure circles fit in square

    # Optimization options
    options = {'disp': False, 'ftol': 1e-8, 'xtol': 1e-8}

    # Initial guess
    initial_guess = np.full(10, 0.05)

    # Define constraints
    def constraint1(x):
        # All circles must be inside the square
        dists = np.abs(x - 0.5)
        return np.min(dists) - 0.5 + 1e-8  # Ensure circle fits

    def constraint2(x):
        # No two circles overlap
        for i in range(10):
            for j in range(i + 1, 10):
                dx = x[2*i] - x[2*j]
                dy = x[2*i+1] - x[2*j+1]
                dist = np.sqrt(dx**2 + dy**2) - (x[i] + x[j])
                if dist < -1e-8:
                    return -dist
        return 0

    # Define optimization problem
    x0 = np.concatenate((centers.flatten(), radii))
    bounds = [(0, 1 - 0.1) for _ in range(20)]

    res = minimize(objective, x0, method='SLSQP', bounds=bounds, constraints=[constraint1, constraint2], options=options)

    # Extract results
    centers_opt = res.x[:20].reshape(10, 2)
    radii_opt = res.x[20:]
    sum_radii = np.sum(radii_opt)

    # Final check for constraints
    for i in range(10):
        if centers_opt[i, 0] < radii_opt[i] or centers_opt[i, 0] > 1 - radii_opt[i] or \
           centers_opt[i, 1] < radii_opt[i] or centers_opt[i, 1] > 1 - radii_opt[i]:
            raise ValueError("Circle out of bounds in final solution")
        for j in range(i + 1, 10):
            dx = centers_opt[i, 0] - centers_opt[j, 0]
            dy = centers_opt[i, 1] - centers_opt[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            if dist < radii_opt[i] + radii_opt[j] - 1e-8:
                raise ValueError("Circles overlap in final solution")

    return (centers_opt, radii_opt, sum_radii)
