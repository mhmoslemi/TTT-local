import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess: evenly spaced grid of 10 points
    centers = np.array([[i / 3, j / 3] for i in range(3) for j in range(3)])
    radii = np.full(10, 0.1)  # Initial small radius

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: all circles fully inside the unit square
    def square_constraint(radii):
        return np.array([r for r in radii])

    # Constraints: no overlapping between circles
    def circle_constraints(centers, radii):
        n = len(radii)
        cons = []
        for i in range(n):
            for j in range(i + 1, n):
                # Distance between centers
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                # Constraint: distance >= sum of radii
                cons.append(dist - (radii[i] + radii[j]))
        return np.array(cons)

    # Bounds for radii
    bounds = [(0, 1) for _ in range(10)]

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: 1 - r},  # r <= 1
            {'type': 'ineq', 'fun': lambda r: r},      # r >= 0
            {'type': 'ineq', 'fun': square_constraint},
            {'type': 'ineq', 'fun': lambda r: circle_constraints(centers, r)}
        ]
    )

    radii_result = result.x
    sum_radii_result = np.sum(radii_result)

    return (centers, radii_result, sum_radii_result)
