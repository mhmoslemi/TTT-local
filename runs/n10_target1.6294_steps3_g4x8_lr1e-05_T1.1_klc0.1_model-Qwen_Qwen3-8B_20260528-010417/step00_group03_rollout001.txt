import numpy as np
from scipy.optimize import minimize
import math

def run_packing():
    # Initial guess for centers and radii
    # Place circles in a grid-like pattern
    centers = np.array([[0.1 + i*0.2, 0.1 + j*0.2] for i in range(5) for j in range(2)])
    radii = np.ones(10) * 0.1

    def objective(radii_and_centers):
        # Flatten the input to extract centers and radii
        flat = radii_and_centers.flatten()
        centers = flat[:20].reshape(10, 2)
        radii = flat[20:]
        
        # Calculate the sum of radii
        sum_radii = np.sum(radii)
        # Constraint: each circle is inside the unit square
        constraint1 = np.sum((centers[:, 0] - radii) >= 0) + np.sum((1 - centers[:, 0] - radii) >= 0)
        constraint2 = np.sum((centers[:, 1] - radii) >= 0) + np.sum((1 - centers[:, 1] - radii) >= 0)
        # Constraint: circles do not overlap
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        constraint3 = np.sum(dists >= radii[:, np.newaxis] + radii[np.newaxis, :])
        # Return negative of sum_radii to maximize it with minimize
        return -sum_radii

    def constraint_func(radii_and_centers):
        flat = radii_and_centers.flatten()
        centers = flat[:20].reshape(10, 2)
        radii = flat[20:]
        # Constraint: circles inside the unit square
        inside = (centers[:, 0] - radii >= 0) & (1 - centers[:, 0] - radii >= 0) & \
                 (centers[:, 1] - radii >= 0) & (1 - centers[:, 1] - radii >= 0)
        # Constraint: circles do not overlap
        dists = np.sqrt(((centers[:, np.newaxis, :] - centers[np.newaxis, :, :]) ** 2).sum(axis=2))
        dists = np.triu(dists)
        overlap = (dists < radii[:, np.newaxis] + radii[np.newaxis, :])
        return np.sum(~inside) + np.sum(overlap)

    # Bounds: radii must be between 0 and 0.5 (to fit in unit square)
    bounds = [(0, 0.5) for _ in range(10)]

    # Initial guess flattened
    initial_guess = np.concatenate([centers.flatten(), radii])

    # Define constraints
    constraints = [{'type': 'ineq', 'fun': constraint_func}]

    # Optimization
    result = minimize(objective, initial_guess, bounds=bounds, constraints=constraints, method='SLSQP')

    # Extract results
    flat = result.x
    centers = flat[:20].reshape(10, 2)
    radii = flat[20:]
    sum_radii = -result.fun

    return (centers, radii, sum_radii)
