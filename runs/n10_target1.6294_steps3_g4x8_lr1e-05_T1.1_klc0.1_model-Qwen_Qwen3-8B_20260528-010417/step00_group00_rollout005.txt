import numpy as np
from scipy.optimize import minimize
import warnings

def run_packing():
    # Initial guess for centers and radii
    np.random.seed(42)
    centers = np.random.rand(10, 2)
    radii = np.random.rand(10) * 0.1

    def objective_function(x):
        # x is a flat array: [x1, y1, r1, x2, y2, r2, ..., x10, y10, r10]
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        # Objective: minimize negative sum of radii
        return -np.sum(radii_flat)

    def constraint_circle_inside(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        for i in range(10):
            x_i, y_i, r_i = centers_flat[i, 0], centers_flat[i, 1], radii_flat[i]
            if not (r_i <= x_i <= 1 - r_i and r_i <= y_i <= 1 - r_i):
                return -1.0
        return 0.0

    def constraint_no_overlap(x):
        centers_flat = x[:60].reshape(10, 2)
        radii_flat = x[60:].reshape(10)
        for i in range(10):
            for j in range(i + 1, 10):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j]:
                    return -1.0
        return 0.0

    def constraint_non_negative(x):
        radii_flat = x[60:].reshape(10)
        if np.any(radii_flat < 0):
            return -1.0
        return 0.0

    # Initial x vector
    x0 = np.zeros(60 + 10)
    x0[:60] = centers.flatten()
    x0[60:] = radii.flatten()

    # Set up optimization problem
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        result = minimize(
            objective_function,
            x0,
            method='SLSQP',
            constraints=[
                {'type': 'ineq', 'fun': constraint_circle_inside},
                {'type': 'ineq', 'fun': constraint_no_overlap},
                {'type': 'ineq', 'fun': constraint_non_negative}
            ],
            options={
                'ftol': 1e-8,
                'maxiter': 1000
            }
        )

    centers_opt = result.x[:60].reshape(10, 2)
    radii_opt = result.x[60:].reshape(10)
    sum_radii_opt = -result.fun

    return (centers_opt, radii_opt, sum_radii_opt)
