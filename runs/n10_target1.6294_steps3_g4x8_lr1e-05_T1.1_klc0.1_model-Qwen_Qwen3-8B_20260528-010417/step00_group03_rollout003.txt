import numpy as np
from scipy.optimize import minimize
import scipy.spatial.distance as distance

def run_packing():
    # Initial guess: place circles evenly spaced in a grid
    initial_centers = np.array([[i/3 + 0.1, j/3 + 0.1] for i in range(3) for j in range(3)])
    initial_centers = np.vstack([initial_centers, [0.5, 0.2], [0.5, 0.8], [0.2, 0.5], [0.8, 0.5]])
    initial_radii = np.full(10, 0.1)

    def objective(x):
        # x is a flat array of centers and radii
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum to maximize it

    def constraints(x):
        centers = x[:20].reshape(10, 2)
        radii = x[20:]
        cons = []
        # Constraint 1: All circles are inside the square
        for i in range(10):
            x_i, y_i = centers[i]
            r_i = radii[i]
            cons.append((x_i - r_i) >= 0)
            cons.append((x_i + r_i) <= 1)
            cons.append((y_i - r_i) >= 0)
            cons.append((y_i + r_i) <= 1)
        # Constraint 2: No overlaps
        for i in range(10):
            for j in range(i + 1, 10):
                d = distance.euclidean(centers[i], centers[j])
                cons.append(d >= radii[i] + radii[j])
        return cons

    bounds = []
    for _ in range(10):
        bounds.extend([(0, 1), (0, 1)])  # x and y for each center
        bounds.extend([(-np.inf, 1 - 1e-6)])  # radius must be small enough to fit in square

    result = minimize(objective, x0=np.concatenate([initial_centers.flatten(), initial_radii]), method='SLSQP', constraints=constraints, bounds=bounds)

    centers = result.x[:20].reshape(10, 2)
    radii = result.x[20:]
    sum_radii = -result.fun
    return (centers, radii, sum_radii)
