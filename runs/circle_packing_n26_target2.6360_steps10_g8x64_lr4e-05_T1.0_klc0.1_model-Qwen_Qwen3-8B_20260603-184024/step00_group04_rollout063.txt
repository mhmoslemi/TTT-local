import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial hexagonal packing
    def hexagonal_packing(n):
        # Calculate rows and columns
        rows = int(np.ceil(np.sqrt(n)))
        cols = int(np.ceil(n / rows))
        
        # Define spacing
        d = 2 * np.sqrt(3) / (2 * (cols - 1))  # Horizontal spacing
        r_initial = d / 2  # Initial radius to fit the grid
        
        centers = np.zeros((n, 2))
        index = 0
        
        for row in range(rows):
            for col in range(cols):
                x = col * d
                y = row * (np.sqrt(3) / 2) * d
                if row % 2 == 1:
                    x += d / 2
                centers[index] = [x, y]
                index += 1
        
        # Normalize the centers to [0,1]^2
        centers /= (np.max(centers, axis=0) - np.min(centers, axis=0))
        return centers, np.ones(n) * r_initial

    # Objective function to maximize sum of radii
    def objective(x, n):
        return -x[0]  # Negative for minimization

    # Constraints for non-overlap and boundary
    def constraints(x, n):
        # x = [x1, y1, r1, x2, y2, r2, ..., xn, yn, rn]
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]

            x, y, r = centers[i][0], centers[i][1], radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return np.inf

            for j in range(i + 1, n):
                dist = np.sqrt((centers[i][0] - centers[j][0])**2 + (centers[i][1] - centers[j][1])**2)

                    return np.inf
        
        return 0

        bounds.extend([(0, 1), (0, 1), (0, 1)])
    
    # Optimization
    result = optimize.minimize(

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
