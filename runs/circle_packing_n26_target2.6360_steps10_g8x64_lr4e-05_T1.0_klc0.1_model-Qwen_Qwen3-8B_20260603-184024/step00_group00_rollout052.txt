import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid layout with some spacing to allow for optimization
    n = 26
    # Create an initial layout with 5 rows and 6 columns for 30 circles, but we only use 26
    grid_x = np.linspace(0.1, 0.9, 6)
    grid_y = np.linspace(0.1, 0.9, 5)
    centers = []
    radii = []
    for i in range(5):
        for j in range(6):
            if i * 6 + j < n:
                centers.append([grid_x[j], grid_y[i]])
                radii.append(0.05)

    centers = np.array(centers)
    radii = np.array(radii)

    # Define the objective: maximize sum of radii = -minimize(-sum(radii))
    def objective(x):
        return -np.sum(x)

    # Define the constraints
    def constraints(x):
        # x has shape (2*n, ), first n*2 elements are centers, last n elements are radii
        centers = x[:2*n].reshape(n, 2)
        radii = x[2*n:]

        # Constraint: each circle is within the unit square
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf

        # Constraint: no overlap between circles

            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf

        # Constraint: radii >= 0

            return np.inf

        return 0.0

    # Initial guess (flattened)
    x0 = np.concatenate([centers.flatten(), radii])

    # Solve the problem
    result = optimize.minimize(

        x0,
        constraints={"type": "ineq", "fun": constraints},
        method="SLSQP",

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
