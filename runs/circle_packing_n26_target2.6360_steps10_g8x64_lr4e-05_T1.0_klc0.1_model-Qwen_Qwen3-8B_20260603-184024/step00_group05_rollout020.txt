import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Hexagonal grid layout for 26 circles
    n_circles = 26
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))

    # Initial fixed radius (will be optimized)
    initial_radius = 0.03

    # Set up centers in a hexagonal grid
    centers = np.zeros((n_circles, 2))
    idx = 0
    for i in range(rows):
        for j in range(cols):
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            if x < 1 and y < 1:
                centers[idx, :] = [x, y]
                idx += 1

    # Initial radii (equal for all)
    radii = np.full(n_circles, initial_radius)

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints
    def constraint_overlap(x, radii):
        centers = x.reshape(-1, 2)
        n = len(radii)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1
        return 0

    def constraint_boundary(x, radii):
        centers = x.reshape(-1, 2)
        for i in range(len(radii)):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-8 or x_i + r_i > 1 + 1e-8 or y_i - r_i < -1e-8 or y_i + r_i > 1 + 1e-8:
                return -1
        return 0

    # Objective and constraints for optimization
    def objective_with_bounds(radii):
        return -np.sum(radii)

    bounds = Bounds(np.zeros(n_circles), np.inf)

    # Setup optimization
    result = minimize(
        objective_with_bounds,
        radii,
        method='Nelder-Mead',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, x)},
            {'type': 'ineq', 'fun': lambda x: constraint_boundary(x, x)}
        ],
        options={'xatol': 1e-8, 'fatol': 1e-8, 'maxiter': 1000}
    )

    # Extract results
    optimal_radii = result.x
    optimal_centers = centers  # Since we are optimizing radii, centers are fixed in this setup
    sum_radii = np.sum(optimal_radii)

    return optimal_centers, optimal_radii, sum_radii
