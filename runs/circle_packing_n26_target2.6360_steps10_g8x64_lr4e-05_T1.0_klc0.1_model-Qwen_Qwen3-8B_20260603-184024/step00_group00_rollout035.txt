from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing():

    n = 26
    num_per_row = int(np.ceil(np.sqrt(n)))
    centers = np.zeros((n, 2))

        row = i // num_per_row
        col = i % num_per_row
        x = col / (num_per_row - 1) if num_per_row > 1 else 0.5

        centers[i] = [x, y]
    
    # Initial radii (same for all)

        min_dist = np.min(dists)

        return min_dist - min_allowed_dist

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:

        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
