from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import InequalityConstraint

def run_packing():
    # Initial grid-based placement of 26 circles in a 6x5 grid
    num_circles = 26
    grid_rows = 6  # 6 rows
    grid_cols = 5  # 5 columns
    spacing = 1.0 / (grid_rows + 1)  # space between circles
    centers = np.zeros((num_circles, 2))
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j * spacing + spacing / 2
            y = i * spacing + spacing / 2
            centers[i * grid_cols + j] = [x, y]
    radii = np.full(num_circles, spacing / 2)  # Initial radius

    # Objective function: Maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Inequality constraints (non-overlapping and within bounds)
    def constraint_func(x):
        # x has shape (2*num_circles,)
        centers = x[:2*num_circles].reshape(num_circles, 2)
        radii = x[2*num_circles:]
        # Check if all circles are within the unit square
        for i in range(num_circles):
            x_center, y_center = centers[i]
            r = radii[i]
            if x_center - r < -1e-12 or x_center + r > 1 + 1e-12 or y_center - r < -1e-12 or y_center + r > 1 + 1e-12:
                return np.inf
        # Check for overlapping circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

    # Prepare bounds for centers and radii
    centers_bounds = Bounds(np.zeros((num_circles, 2)), np.ones((num_circles, 2)))

    initial_guess = np.concatenate([centers.flatten(), radii])

    cons = InequalityConstraint(lambda x: constraint_func(x), lb=0, ub=np.inf)

    result = minimize(
