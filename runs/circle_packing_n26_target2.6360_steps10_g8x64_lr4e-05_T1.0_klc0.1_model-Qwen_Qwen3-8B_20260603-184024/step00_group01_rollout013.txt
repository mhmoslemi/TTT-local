import cvxpy as cp

def run_packing():
    # Initial estimate: hexagonal packing of 26 circles
    num_circles = 26
    total_width = 1.0
    total_height = 1.0
    radius_estimate = 0.05  # Initial estimate for radius

    # Generate initial hexagonal grid within the unit square
    centers = np.zeros((num_circles, 2))
    radii = np.full(num_circles, radius_estimate)

    # Calculate row and column positions based on hexagonal packing
    # We'll use a hexagonal grid pattern and adjust it to fit in the unit square
    col_count = int(np.sqrt(num_circles))
    row_count = (num_circles + col_count - 1) // col_count  # Round up

    # Adjust to fit within the unit square
    col_spacing = total_width / (col_count + 1)
    row_height = total_height / row_count
    row_offset = row_height / 2

    for i in range(num_circles):
        col = i % col_count
        row = i // col_count
        x = (col + 0.5) * col_spacing
        y = row * row_height + row_offset
        centers[i] = [x, y]

    # Define optimization variables
    x = cp.Variable((num_circles, 2))
    r = cp.Variable(num_circles, nonneg=True)

    # Objective: maximize the sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # Ensure each circle is entirely within the unit square
    for i in range(num_circles):
        x_i, y_i = x[i]
        r_i = r[i]
        constraints.append(x_i - r_i >= 0)
        constraints.append(x_i + r_i <= 1)
        constraints.append(y_i - r_i >= 0)
        constraints.append(y_i + r_i <= 1)

    # Ensure no two circles overlap
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dx = x[i][0] - x[j][0]
            dy = x[i][1] - x[j][1]
            dist = cp.sqrt(dx**2 + dy**2)
            constraints.append(dist >= r[i] + r[j])

    # Set up and solve the optimization problem
    prob = cp.Problem(objective, constraints)
    result = prob.solve(solver=cp.SCS, verbose=False)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
