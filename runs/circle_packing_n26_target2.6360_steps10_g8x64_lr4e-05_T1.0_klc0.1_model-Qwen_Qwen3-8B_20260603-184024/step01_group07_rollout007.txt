import scipy.optimize as optimize

def run_packing():

    rows = (n + cols - 1) // cols

    dy = np.sqrt(3) / (rows + 1)

    # Adjust spacing based on square packing to avoid overfitting
    centers = np.zeros((n, 2))

        row = i // cols

        centers[i] = [x, y]
