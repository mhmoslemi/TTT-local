def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26

    # Initial grid-based arrangement (hexagonal-like)
    # Number of rows and columns to fit 26 circles

    rows = int(np.ceil(n / cols))
    
    # Generate initial centers (hexagonal grid)
    centers = np.zeros((n, 2))
    for i in range(n):
        col = i % cols
        row = i // cols
        x = col / (cols - 1) if cols > 1 else 0.5
        y = row / (rows - 1) if rows > 1 else 0.5
        centers[i] = np.array([x, y])
    
    # Define variables to optimize (radii)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize actual sum

    constraints = []

    # Constraints for circles being within the unit square
    for i in range(n):
        def square_constraint(radii):
            return np.array([radii[i] - (1 - centers[i, 0]),  # x - r <= 1
                             centers[i, 0] - radii[i],         # x + r <= 1
                             radii[i] - (1 - centers[i, 1]),  # y - r <= 1
                             centers[i, 1] - radii[i]])       # y + r <= 1
        constraints.append({'type': 'ineq', 'fun': square_constraint})

    # Constraints for circles not overlapping
    for i in range(n):
        for j in range(i + 1, n):
            def circle_overlap_constraint(radii):

                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                return dist - (radii[i] + radii[j])  # dist >= r_i + r_j

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
