import scipy.optimize as optimize

def run_packing():

    radii = np.zeros(num_circles)

    cols = int(np.ceil(num_circles / rows))

    offset = spacing / 2
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        if row % 2 == 1:
            x = (col + 0.5) * spacing + offset

            x = (col + 0.5) * spacing

        centers[i, 1] = y
        radii[i] = 0.08  # Initial more aggressive radius

    bounds = [(0.0, 1.0) for _ in range(2 * num_circles)]  # x and y coordinates
    radii_bounds = [(0.0, 0.2) for _ in range(num_circles)]  # radius bounds
    bounds += radii_bounds

    # Objective function: maximize total radius
    def objective(x):

        radii_flat = x[2 * num_circles:]
        total_radius = np.sum(radii_flat)
        return -total_radius  # Minimize negative total radius

        centers_flat = x[:2 * num_circles].reshape((num_circles, 2))
        radii_flat = x[2 * num_circles:]
