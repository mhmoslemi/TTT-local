from scipy.optimize import minimize
import random

def run_packing():

    n = 26
    grid_size = int(np.ceil(np.sqrt(n)))
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

        x = i % grid_size / (grid_size - 1)

        return -np.sum(radii)

            r = radii[i]
