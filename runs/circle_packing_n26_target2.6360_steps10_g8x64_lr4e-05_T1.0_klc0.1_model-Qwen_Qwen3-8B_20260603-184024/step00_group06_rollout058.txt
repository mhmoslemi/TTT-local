import numpy as np
import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def generate_initial_guess(num_circles):
    """Generate an initial random guess for circle positions and radii."""
    centers = np.random.rand(num_circles, 2)
    radii = np.random.rand(num_circles) * 0.1  # Small random radii
    return centers, radii

def objective_function(params, num_circles):
    """Objective function to maximize the sum of radii."""
    centers = params[:num_circles * 2].reshape(num_circles, 2)
    radii = params[num_circles * 2:].reshape(num_circles)
    return -np.sum(radii)  # Negative for minimization

def constraints(params, num_circles):
    """Constraint function for optimization."""
    centers = params[:num_circles * 2].reshape(num_circles, 2)
    radii = params[num_circles * 2:].reshape(num_circles)

    # Apply boundary constraints
    for i in range(num_circles):
        x, y = centers[i]
        r = radii[i]
        if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
            return np.inf  # Violation, penalize

    # Apply non-overlapping constraints
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return np.inf  # Violation, penalize

    return 0  # No violation

def run_packing():
    num_circles = 26
    initial_centers, initial_radii = generate_initial_guess(num_circles)
    initial_params = np.concatenate([initial_centers.reshape(-1), initial_radii])

    # Optimization settings
    bounds = []
    for i in range(num_circles):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y coordinates
        bounds.extend([(-1e-6, 1.0)])  # Radius must be non-negative

    result = opt.minimize(
        fun=objective_function,
        x0=initial_params,
        args=(num_circles,),
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraints, 'args': (num_circles,)}],

        options={'maxiter': 1000, 'disp': False}

    if result.success:
        centers = result.x[:num_circles * 2].reshape(num_circles, 2)
        radii = result.x[num_circles * 2:].reshape(num_circles)
        sum_radii = np.sum(radii)
    else:
        centers, radii = generate_initial_guess(num_circles)
        sum_radii = np.sum(radii)

    return centers, radii, sum_radii
