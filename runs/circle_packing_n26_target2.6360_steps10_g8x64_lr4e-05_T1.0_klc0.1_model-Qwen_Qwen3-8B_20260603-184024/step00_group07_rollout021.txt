import numpy as np
import scipy.optimize as optimize

def run_packing():

    n = 26
    initial_radius = 0.05  # Initial uniform radius
    grid_rows = 6
    grid_cols = 5

    # Generate an initial grid configuration
    centers = np.zeros((n, 2))
    radii = np.full(n, initial_radius)

    # Use a hexagonal grid pattern
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            index = i * grid_cols + j
            centers[index] = [x, y]
            if i % 2 == 1:
                radii[index] = 0.05 / 2  # Adjust radii for interleaved rows

    # Define a function to compute the total sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Minimize negative sum to maximize sum

    # Flatten centers and radii for optimization
    centers_flat = centers.reshape(-1)
    radii_flat = radii.copy()

    # Define constraints for optimization
    def constraint1(radii_flat):
        # Ensure circles are within the unit square
        for i in range(n):
            x = centers_flat[i * 2]
            y = centers_flat[i * 2 + 1]
            r = radii_flat[i]
            if not (0 - r <= x <= 1 + r and 0 - r <= y <= 1 + r):
                return -1
        return 0

    def constraint2(radii_flat):
        # Ensure no two circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers_flat[i * 2] - centers_flat[j * 2]
                dy = centers_flat[i * 2 + 1] - centers_flat[j * 2 + 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:

        return 0

    cons = [

        {'type': 'ineq', 'fun': constraint2}
    ]

        method='SLSQP',
        constraints=cons,
        bounds=[(0, 1) for _ in range(n)]
    )

    # Extract optimized radii
    optimized_radii = result.x

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
