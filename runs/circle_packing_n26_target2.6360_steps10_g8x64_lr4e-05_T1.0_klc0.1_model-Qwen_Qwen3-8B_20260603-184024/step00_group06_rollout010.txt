import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import NonlinearConstraint

def run_packing():
    # Initialize centers in a grid pattern
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j / (cols - 1) if cols > 1 else 0.5
            y = i / (rows - 1) if rows > 1 else 0.5
            centers.append([x, y])
    
    centers = np.array(centers)
    radii = np.ones(num_circles) * 0.1  # Initial small radius

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum

    def constraint_overlap(i, j, radii, centers):
        if i >= j:
            return 0
        dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
        return dist - (radii[i] + radii[j]) + 1e-8

    def constraint_boundary(i, radii, centers):
        x, y = centers[i]
        r = radii[i]
        return (x - r) - 0 + 1e-8  # Enforce x - r >= 0
    def constraint_boundary_right(i, radii, centers):
        x, y = centers[i]
        r = radii[i]
        return (1 + 1e-8) - (x + r)  # Enforce x + r <= 1

    def constraint_boundary_top(i, radii, centers):
        x, y = centers[i]
        r = radii[i]
        return (y - r) - 0 + 1e-8  # Enforce y - r >= 0
    def constraint_boundary_bottom(i, radii, centers):
        x, y = centers[i]
        r = radii[i]
        return (1 + 1e-8) - (y + r)  # Enforce y + r <= 1

    constraints = []
    for i in range(num_circles):
        constraints.append(NonlinearConstraint(
            lambda radii: constraint_boundary(i, radii, centers),
            lb=-np.inf,
            ub=0
        ))
        constraints.append(NonlinearConstraint(
            lambda radii: constraint_boundary_right(i, radii, centers),
            lb=-np.inf,
            ub=0
        ))

        method='SLSQP',
