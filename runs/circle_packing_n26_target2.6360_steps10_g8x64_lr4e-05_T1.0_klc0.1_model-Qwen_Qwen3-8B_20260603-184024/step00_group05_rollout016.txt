import scipy.optimize as opt
import scipy.spatial.distance as dist

def validate_packing(centers, radii):

        return False, "NaN values present"

            return False, f"Circle {i} has negative radius {radii[i]}"

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            dist_ij = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist_ij < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_grid_config(n_circles):
    """Initial grid-based configuration to start with."""
    n_rows = int(np.ceil(np.sqrt(n_circles)))
    n_cols = int(np.ceil(n_circles / n_rows))
    spacing = 1.0 / (n_cols - 1) if n_cols > 1 else 1.0
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        row = i // n_cols
        col = i % n_cols
        x = col * spacing
        y = row * spacing
        centers[i] = [x, y]
    return centers

def run_packing():
