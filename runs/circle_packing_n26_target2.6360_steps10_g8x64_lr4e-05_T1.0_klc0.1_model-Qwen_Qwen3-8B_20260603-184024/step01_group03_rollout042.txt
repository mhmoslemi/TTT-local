def hexagonal_initial(n):
    max_rows = int(np.sqrt(2 * n))
    max_cols = n // max_rows if max_rows else n
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // max_cols
        col = i % max_cols
        x = col + 0.5 * (row % 2)
        y = row * np.sqrt(3) / 2
        centers[i] = [x, y]
    return centers

def square_initial(n):
    max_rows = int(np.sqrt(n))
    max_cols = n // max_rows if max_rows else n
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // max_cols
        col = i % max_cols
        x = col + 0.5
        y = row + 0.5
        centers[i] = [x, y]
    return centers

def nested_initial(n):
    row = int(np.sqrt(n))
    col = n // row if row else n
    centers = np.zeros((n, 2))
    for i in range(n):
        x = (i % col) / (col - 1) if col > 1 else 0.5
        y = (i // col) / (row - 1) if row > 1 else 0.5
        centers[i] = [x, y]
    return centers

def hybrid_initial(n):
