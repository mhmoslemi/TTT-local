import numpy as np
import scipy.optimize as optimize
import warnings

def run_packing():
    # Initial placement: grid or hexagonal pattern
    # Let's use a grid pattern with some spacing to start
    rows = 5
    cols = 5
    spacing = 0.4  # Adjust spacing for better initial packing
    centers = np.zeros((26, 2))
    radii = np.zeros(26)

    # Initialize in a grid pattern
    for i in range(26):
        row = i // cols
        col = i % cols
        x = col * spacing + 0.5 * spacing
        y = row * spacing + 0.5 * spacing
        centers[i] = [x, y]
        radii[i] = 0.15  # Initial radius based on spacing

    # Define the objective function
    def objective(radii):
        return -np.sum(radii)  # Negative because we use minimization

    # Define constraints for non-overlapping and boundary
    def constraint_overlap(i, j, radii):
        def cons(radii):
            return np.sqrt(np.sum((centers[i] - centers[j])**2)) - (radii[i] + radii[j])
        return cons

    def constraint_boundary(i, radii):
        def cons(radii):
            x, y = centers[i]
            return np.min([x - radii[i], 1 - x - radii[i], y - radii[i], 1 - y - radii[i]])
        return cons

    constraints = []

    # Add boundary constraints
    for i in range(26):
        constraints.append({'type': 'ineq', 'fun': constraint_boundary(i)})

    # Add non-overlap constraints
    for i in range(26):
        for j in range(i + 1, 26):
            constraints.append({'type': 'ineq', 'fun': constraint_overlap(i, j)})

    # Optimization options
    options = {
        'method': 'SLSQP',
        'ftol': 1e-9,
        'maxiter': 1000,
        'disp': False
    }

    # Use initial radii as starting point
    result = optimize.minimize(
        objective,
        radii,
        constraints=constraints,
        bounds=[(0, 0.5) for _ in range(26)],
        options=options
    )

        return np.zeros((26, 2)), np.zeros(26), 0.0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
