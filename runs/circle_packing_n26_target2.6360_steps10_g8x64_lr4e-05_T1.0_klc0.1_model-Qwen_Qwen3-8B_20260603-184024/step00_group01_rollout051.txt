import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():
    # Initial hexagonal grid placement with some rows
    num_circles = 26
    rows = 6  # Number of rows in the hexagonal grid
    cols = 5  # Number of columns per row
    grid_spacing = 0.3  # Initial spacing between centers
    radius_initial = 0.15  # Initial radius for all circles

    # Create initial hexagonal grid
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles) + radius_initial
    idx = 0
    for row in range(rows):
        for col in range(cols):
            x = col * grid_spacing
            y = row * (np.sqrt(3) / 2) * grid_spacing
            if row % 2 == 1:
                x += grid_spacing / 2
            centers[idx] = [x, y]
            idx += 1

    # Define objective function: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum to maximize sum

    # Define constraints: circle boundary and non-overlap
    def constraint_boundary(x):
        # x is a vector of all parameters: [x_0, y_0, r_0, x_1, y_1, r_1, ..., x_n-1, y_n-1, r_n-1]
        n = len(x) // 3
        x_centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:].reshape(n)
        for i in range(n):
            x_i, y_i = x_centers[i]
            r_i = radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf  # Penalize out-of-bound circles
        return 0.0

    def constraint_overlap(x):
        n = len(x) // 3
        x_centers = x[:n*2].reshape(n, 2)
        radii = x[n*2:].reshape(n)
        for i in range(n):

                dy = x_centers[i, 1] - x_centers[j, 1]
                dist_sq = dx ** 2 + dy ** 2

                if dist_sq < (r_sum ** 2 - 1e-12):
                    return np.inf  # Penalize overlapping circles
        return 0.0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
