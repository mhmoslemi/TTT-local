def run_packing():
    # Define initial positions for 5 rows in a hexagonal packing
    # Adjust to fit within unit square
    n_circles = 26
    rows = 5
    cols = 5
    centers = np.zeros((26, 2))
    radii = np.zeros((26,))

    # Hexagonal grid with decreasing size
    for i in range(rows):
        for j in range(cols - i):
            x = j + 0.5 * i

    initial_radius = 0.05
    radii.fill(initial_radius)

    # Optimization setup
    # Objective: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraints: non-overlapping and boundary
    def constraints(radii):

            x = centers[i, 0]

            center_positions[i, 0] = x
            center_positions[i, 1] = y

        # Non-overlapping constraint

            return 0

        return non_overlapping(radii) + boundary(radii)
