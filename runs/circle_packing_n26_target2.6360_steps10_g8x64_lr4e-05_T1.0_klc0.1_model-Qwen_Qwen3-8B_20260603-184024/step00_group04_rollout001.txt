import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial guess for centers: a grid-like arrangement
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // 4
        col = i % 4
        centers[i, 0] = col / 3.0  # Adjusted for better spacing
        centers[i, 1] = row / 3.0  # Adjusted for better spacing

    # Initial radii guess: uniform
    radii = np.full(num_circles, 0.1)

    # Objective function to maximize the sum of radii
    def objective(x):
        # x is a flattened array with shape (2*num_circles + num_circles,) = (3*num_circles,)
        # Extract centers and radii
        centers = x[:2*num_circles].reshape(num_circles, 2)
        radii = x[2*num_circles:].reshape(num_circles)
        return -np.sum(radii)  # Minimize the negative sum to maximize

    # Constraints: 
    constraints = []

    # Boundary constraints
    for i in range(num_circles):
        def boundary_constraint(x):
            centers = x[:2*num_circles].reshape(num_circles, 2)
            radii = x[2*num_circles:].reshape(num_circles)
            x_center, y_center = centers[i]
            r = radii[i]
            return [x_center - r + 1e-12, 1 - x_center - r + 1e-12, y_center - r + 1e-12, 1 - y_center - r + 1e-12]
        constraints.append({'type': 'ineq', 'fun': boundary_constraint})

    # Non-overlapping constraints
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            def non_overlap_constraint(x):
                centers = x[:2*num_circles].reshape(num_circles, 2)
                radii = x[2*num_circles:].reshape(num_circles)
                x1, y1 = centers[i]
                x2, y2 = centers[j]
                r1 = radii[i]
                r2 = radii[j]
                dist = np.sqrt((x1 - x2)**2 + (y1 - y2)**2)
                return dist - r1 - r2 + 1e-12
            constraints.append({'type': 'ineq', 'fun': non_overlap_constraint})

    # Initial x: a flattened array of centers and radii
    x0 = np.concatenate((centers.flatten(), radii))

    # Bounds: centers must be in [0,1], radii must be ≥ 0
    bounds = []
    for i in range(num_circles):
        bounds += [(0.0, 1.0), (0.0, 1.0)]  # For x and y coordinates of each center
        bounds += [(0.0, 1.0)]  # For radius of each circle

    # Run optimization
    result = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-8,
        options={'maxiter': 500}
    )

    # Extract results
    centers_final = result.x[:2*num_circles].reshape(num_circles, 2)
    radii_final = result.x[2*num_circles:].reshape(num_circles)
    sum_radii = np.sum(radii_final)

    # Validate the packing (for safety)
    valid, msg = validate_packing(centers_final, radii_final)
    if not valid:
        raise ValueError(f"Packing is invalid: {msg}")

    return centers_final, radii_final, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
