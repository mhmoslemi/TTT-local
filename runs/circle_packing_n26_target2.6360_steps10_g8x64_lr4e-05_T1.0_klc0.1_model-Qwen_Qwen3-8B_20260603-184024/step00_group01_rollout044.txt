import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess for centers in a staggered hexagonal pattern
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    spacing = 1.0 / (cols - 1)
    centers = np.zeros((num_circles, 2))
    count = 0

    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                x = j * spacing
            else:
                x = (j + 0.5) * spacing
            y = i * spacing
            centers[count, :] = [x, y]
            count += 1

    # Initial guess for radii (uniform with some variance)
    radii = np.full(num_circles, 1.0 / (np.sqrt(2) * cols))  # Start with small radii to avoid overlap

    def objective_function(radii):
        # Objective is to maximize the sum of radii, so minimize negative sum
        return -np.sum(radii)

    def constraint_positions(centers, radii):
        # Constraint: All circles must be within the unit square
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return False
        return True

    def constraint_distances(centers, radii):
        # Constraint: No overlapping between circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

        # Combine constraints

            return np.array([-np.inf])
        if not constraint_distances(centers, radii):
            return np.array([-np.inf])
        return np.array([0])

    # Set up the optimization
    bounds = [(0, 1.0) for _ in range(num_circles)]
    result = optimize.minimize(
        fun=objective_function,
        x0=radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': lambda r: constraint_positions(centers, r)},
            {'type': 'ineq', 'fun': lambda r: constraint_distances(centers, r)}
        ],
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
