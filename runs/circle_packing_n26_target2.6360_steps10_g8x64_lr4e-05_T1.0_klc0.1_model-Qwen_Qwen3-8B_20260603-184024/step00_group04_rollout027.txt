import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Step 1: Initialize centers using a hexagonal packing pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Generate hexagonal grid
    rows = int(np.sqrt(num_circles))
    cols = (num_circles + rows - 1) // rows
    spacing = 1.0 / (cols - 1)
    x_offset = 0.5
    y_offset = 0.5

    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = x_offset + col * spacing
        y = y_offset + row * spacing * np.sqrt(3) / 2
        centers[i, 0] = x
        centers[i, 1] = y

    # Step 2: Define the objective function to minimize (negative of sum of radii)
    def objective(x):

        centers_flat = x[:2*num_circles].reshape(num_circles, 2)
        radii_flat = x[2*num_circles:]
        # Check if any radius is negative
        if np.any(radii_flat < 0):
            return np.inf
        # Check if any circle is outside the unit square
        for i in range(num_circles):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return np.inf

                    return np.inf

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
