import numpy as np
import scipy.optimize as opt
from scipy.spatial.distance import cdist

def run_packing():

    n_circles = 26
    max_radius = 0.1  # Initial guess for radius

    # Compute the approximate number of layers needed
    layers = int(np.ceil(np.sqrt(n_circles)))
    centers = np.zeros((n_circles, 2))
    radius = np.zeros(n_circles)

    # Place circles in a hexagonal packing
    for i in range(n_circles):
        layer = i // (layers)
        pos_in_layer = i % (layers)
        x = pos_in_layer + layer / 2
        y = layer
        # Adjust to stay within the square
        x = np.clip(x, 0, 1)
        y = np.clip(y, 0, 1)
        centers[i] = np.array([x, y])
        radius[i] = max_radius * (1 - 0.1 * i)

    # Create an initial guess for radii
    initial_radius_guess = np.full(n_circles, 0.1)

    # Objective function to maximize the sum of radii

        # To optimize, we minimize the negative of the objective
        return -np.sum(radii)

    # Constraints
    def constraint_overlap(radii):
        # Ensure no two circles overlap
        dist_matrix = cdist(centers, centers)
        return np.sum(dist_matrix < radii[:, np.newaxis] + radii[np.newaxis, :])

        bounds=bounds,
        constraints=constraints

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
