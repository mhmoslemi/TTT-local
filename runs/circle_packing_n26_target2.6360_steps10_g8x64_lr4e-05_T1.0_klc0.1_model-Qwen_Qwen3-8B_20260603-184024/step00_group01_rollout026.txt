import numpy as np
import scipy.optimize as opt
import math

def generate_initial_centers(n=26):
    """
    Generate a grid-like initial placement of centers for n circles in a unit square.
    """
    # Split the square into rows and columns to form a grid
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    
    centers = []
    for i in range(rows):
        for j in range(cols):
            if i * cols + j < n:
                x = (j + 0.5) / cols
                y = (i + 0.5) / rows
                centers.append([x, y])
    return np.array(centers)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):

            return False, f"Circle {i} has negative radius {radii[i]}"
