import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
from scipy.optimize import Bounds

def run_packing():

    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Initial hexagonal placement
    rows = int(np.sqrt(num_circles))
    cols = num_circles // rows + 1
    for i in range(rows):

            if idx >= num_circles:
                break
            x = j + 0.5 * (i % 2)

    # Optimization: Maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative because we minimize

    def constraints(centers, radii):
        constraints = []
        # Boundary constraints
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda x, y=r: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda x, y=r: 1 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda x, y=r: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda x, y=r: 1 - y - r})

                    r2 = radii[j]
                    return np.sqrt((x1 - x2)**2 + (y1 - y2)**2) - r1 - r2

        return constraints

    # Set up optimization problem
    bounds = Bounds(np.zeros(num_circles), np.inf)  # Radii must be non-negative

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
