import numpy as np
import scipy.optimize as opt
import matplotlib.pyplot as plt
from scipy.spatial import distance

def run_packing():
    # Define number of circles and initial grid size
    n_circles = 26
    grid_size = np.sqrt(n_circles)
    
    # Generate initial hexagonal grid in a square of size (grid_size, grid_size)
    # Map hexagonal grid to unit square [0,1]x[0,1]
    centers = np.zeros((n_circles, 2))
    r = np.full(n_circles, 0.05)  # Initial radius
    
    # Create hexagonal grid
    for i in range(n_circles):
        row = i // 5
        col = i % 5
        x = (col + 0.5 * (row % 2)) / (grid_size)
        y = (row) / (grid_size)
        centers[i] = [x, y]
    
    # Objective function: maximize total radius sum
    def objective(x):
        # x is a flattened array of (x1, y1, r1, x2, y2, r2, ..., xn, yn, rn)
        x_flat = x.reshape((-1, 3))
        centers_opt = x_flat[:, :2]
        radii_opt = x_flat[:, 2]
        return -np.sum(radii_opt)  # Minimize negative sum (i.e., maximize sum)

    # Constraints
    def constraint_centers(x):
        x_flat = x.reshape((-1, 3))
        centers_opt = x_flat[:, :2]
        return np.concatenate([centers_opt[:, 0], centers_opt[:, 1]])

    def constraint_radii(x):
        x_flat = x.reshape((-1, 3))
        return x_flat[:, 2]

    def constraint_boundary(x):
        x_flat = x.reshape((-1, 3))
        centers_opt = x_flat[:, :2]
        radii_opt = x_flat[:, 2]
        # For each circle, ensure x - r >= 0 and x + r <= 1
        # Similarly for y
        bounds_x = np.concatenate([centers_opt[:, 0] - radii_opt, 1 - centers_opt[:, 0] - radii_opt])
        bounds_y = np.concatenate([centers_opt[:, 1] - radii_opt, 1 - centers_opt[:, 1] - radii_opt])
        return np.concatenate([bounds_x, bounds_y])

        centers_opt = x_flat[:, :2]
        radii_opt = x_flat[:, 2]
        dists = distance.cdist(centers_opt, centers_opt)
        # For each pair, ensure distance >= r_i + r_j

        return np.array(constraints)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
