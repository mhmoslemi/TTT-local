import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_rows = np.ceil(np.sqrt(n))

    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        
        # Base grid coordinates
        x = col / (num_cols - 1)

    bounds = []

                return x[i, 0] - r[i]
