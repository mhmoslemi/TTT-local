import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial guess: Grid layout with 5 rows and 6 columns for 26 circles
    n = 26
    grid_rows = 5
    grid_cols = 6
    initial_centers = np.zeros((n, 2))
    initial_radii = np.zeros(n)

    # Create a grid layout
    for i in range(n):
        row = i // grid_cols
        col = i % grid_cols
        x = col / (grid_cols - 1)
        y = row / (grid_rows - 1)
        initial_centers[i, 0] = x
        initial_centers[i, 1] = y
        initial_radii[i] = 0.1  # Initial radius

    # Objective function: minimize negative sum of radii
    def objective(x):
        # x has shape (n * 2 + n) = 52
        centers = x[:n * 2].reshape(n, 2)
        radii = x[n * 2:]

        # Compute total sum of radii
        sum_radii = np.sum(radii)
        return -sum_radii  # Minimize negative sum

    # Constraints for optimization

    # Constraint: All circles must be within the unit square
    def constraint_inside(x):
        centers = x[:n * 2].reshape(n, 2)
        radii = x[n * 2:]
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return -1
        return 0

    # Constraint: No overlapping circles
    def constraint_no_overlap(x):
        centers = x[:n * 2].reshape(n, 2)
        radii = x[n * 2:]
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -1
        return 0

    # Constraint: All radii must be non-negative
    def constraint_radii_non_negative(x):
        radii = x[n * 2:]
        if np.any(radii < 0 - 1e-12):
            return -1
        return 0

    # Constraint functions
    cons = [
        {'type': 'ineq', 'fun': constraint_inside},
        {'type': 'ineq', 'fun': constraint_no_overlap},
        {'type': 'ineq', 'fun': constraint_radii_non_negative}
    ]

    # Bounds for positions: [0, 1]
    bounds = [(0, 1) for _ in range(n * 2)] + [(0, 1) for _ in range(n)]

    # Run optimization
    initial_guess = np.concatenate([initial_centers.flatten(), initial_radii])
    result = opt.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

    # Extract centers and radii from result
    centers = result.x[:n * 2].reshape(n, 2)
    radii = result.x[n * 2:]
    sum_radii = -result.fun

    # Validate the packing
    is_valid, msg = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(msg)

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12 or
            y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
