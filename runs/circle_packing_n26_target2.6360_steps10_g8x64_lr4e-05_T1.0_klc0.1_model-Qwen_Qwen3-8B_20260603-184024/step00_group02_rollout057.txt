import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Number of circles
    n = 26

    # Initial grid spacing for hexagonal packing
    # Adjust these parameters for better fit
    rows = 6
    cols = 5
    spacing = 0.25  # Initial spacing between centers

    # Hexagonal packing: shift alternate rows by half the spacing
    centers = []
    for i in range(rows):
        for j in range(cols):
            y = i * spacing
            x = j * spacing
            if i % 2 == 1:
                x += spacing / 2
            centers.append([x, y])

    centers = np.array(centers)

    # Initial guess for radii using minimal spacing
    radii = np.full(n, spacing / 2)

    # Define the objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Negative because we're minimizing

    # Define constraints: non-overlapping and within the square
    def constraint_overlap(x, centers, radii):
        # x is the flattened array of radii
        radii = x.reshape(-1)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return dist - (radii[i] + radii[j] - 1e-12)
        return 0

    def constraint_bounds(x, centers):
        # x is the flattened array of radii
        radii = x.reshape(-1)
        for i in range(n):
            x_center, y_center = centers[i]
            r = radii[i]
            if x_center - r < -1e-12 or x_center + r > 1 + 1e-12:
                return x_center - r - (-1e-12)
            if y_center - r < -1e-12 or y_center + r > 1 + 1e-12:

        return 0

    bounds = [(0, 1) for _ in range(n)]

    # Set up optimization problem
    x0 = radii.copy()
    cons = [
        {'type': 'ineq', 'fun': lambda x: constraint_overlap(x, centers, radii)},
        {'type': 'ineq', 'fun': lambda x: constraint_bounds(x, centers)}
    ]

    # Run the optimization
    result = optimize.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
