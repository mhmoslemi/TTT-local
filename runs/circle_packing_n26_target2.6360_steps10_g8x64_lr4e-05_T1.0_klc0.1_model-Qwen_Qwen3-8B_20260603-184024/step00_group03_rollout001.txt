import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import cdist

def run_packing():
    # Initial hexagonal packing in the unit square
    num_circles = 26
    # Number of rows in hexagonal packing
    rows = 6
    # Number of circles per row
    cols = 5
    # Height of hexagonal cell
    cell_height = np.sqrt(3) / 2
    # Width of hexagonal cell
    cell_width = 1

    # Initial grid positions
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    for i in range(rows):
        for j in range(cols):
            row_index = i * cols + j
            if row_index >= num_circles:
                break
            y = i * cell_height
            x = 0.5 + j * cell_width
            if i % 2 == 1:
                x += cell_width / 2
            centers[row_index] = np.array([x, y])
            radii[row_index] = 0.05

    # Objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Constraint: each circle must be within the unit square
    def boundary_constraint(radii):
        return np.sum((centers[:, 0] - radii) < 0) + np.sum((centers[:, 0] + radii) > 1) + \
               np.sum((centers[:, 1] - radii) < 0) + np.sum((centers[:, 1] + radii) > 1)

    def non_overlap_constraint(radii):
        distances = cdist(centers, centers)
        min_distances = np.min(distances[np.triu_indices(num_circles, 1)])
        return min_distances - np.sum(radii) + 1e-12

    # Define the optimization problem
    bounds = [(0, 1) for _ in range(num_circles)]

    constraints = [
        {'type': 'ineq', 'fun': lambda radii: non_overlap_constraint(radii)},
        {'type': 'ineq', 'fun': lambda radii: 1 - boundary_constraint(radii)}
    ]

    # Initial guess
    initial_guess = radii.copy()

    # Perform optimization
    result = minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-10
    )

    # Extract results

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
