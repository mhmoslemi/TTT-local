import numpy as np
import cvxpy as cp

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    n = 26

    # Define decision variables
    centers = cp.Variable((n, 2))
    radii = cp.Variable(n, nonneg=True)

    # Objective: maximize the sum of the radii
    objective = cp.Maximize(cp.sum(radii))

    # Constraints
    # 1. Each circle must be within the unit square
    constraints = []
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        constraints.append(x - r >= -1e-8)
        constraints.append(x + r <= 1 + 1e-8)
        constraints.append(y - r >= -1e-8)
        constraints.append(y + r <= 1 + 1e-8)

    # 2. No two circles may overlap
    for i in range(n):
        for j in range(i + 1, n):
            dist = cp.norm(centers[i] - centers[j])
            constraints.append(dist >= radii[i] + radii[j] - 1e-8)

    # Solve the optimization problem
    prob = cp.Problem(objective, constraints)
    prob.solve(solver=cp.SCS, eps=1e-6)

    # Extract the results
    centers_values = np.round(centers.value, 8)

    sum_radii = np.sum(radii_values)
