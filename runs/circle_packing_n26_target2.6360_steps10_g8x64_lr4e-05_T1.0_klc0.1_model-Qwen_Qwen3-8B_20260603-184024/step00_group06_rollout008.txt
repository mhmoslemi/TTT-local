import numpy as np
import scipy.optimize as optimize
import cvxpy as cp

def run_packing():
    n = 26

    # Initial guess: hexagonal packing in a grid
    # Determine number of rows and columns
    cols = np.ceil(np.sqrt(n))
    rows = np.ceil(n / cols)

    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Place circles in hexagonal grid
    for i in range(n):
        col = i % cols
        row = i // cols
        x = col + (0.5 * (row % 2))
        y = row
        centers[i] = [x / (cols + 1), y / (rows + 1)]
        radii[i] = 0.05  # Initial small radius

    # Define optimization variables
    centers_var = cp.Variable((n, 2))
    radii_var = cp.Variable(n, non_negative=True)

    # Objective: maximize sum of radii
    objective = cp.Maximize(cp.sum(radii_var))

    # Constraints
    constraints = []

    # Boundary constraints
    for i in range(n):
        x, y = centers_var[i]
        r = radii_var[i]
        constraints.append(x - r >= 0)
        constraints.append(x + r <= 1)
        constraints.append(y - r >= 0)
        constraints.append(y + r <= 1)

    # Non-overlap constraints
    for i in range(n):
        for j in range(i + 1, n):
            x1, y1 = centers_var[i]
            x2, y2 = centers_var[j]
            r1 = radii_var[i]
            r2 = radii_var[j]
            dist = cp.sqrt((x1 - x2)**2 + (y1 - y2)**2)
            constraints.append(dist >= r1 + r2 - 1e-6)

    # Solve the problem
    prob = cp.Problem(objective, constraints)
    prob.solve(max_iter=10000, eps=1e-6, verbose=True)

    # Extract results

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):

        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
