import numpy as np
import cvxpy as cp
import math

def run_packing():

    num_circles = 26
    grid_rows = math.ceil(math.sqrt(num_circles))
    grid_cols = math.ceil(num_circles / grid_rows)

    radius_base = 0.1  # Initial estimation for radius
    centers = np.zeros((num_circles, 2))
    count = 0

            if count >= num_circles:
                break

            x = col + 0.5 * (row % 2)

    centers *= scale
    radii *= scale

    # Define variables
    x = cp.Variable((num_circles, 2))
    r = cp.Variable(num_circles, nonneg=True)

    objective = cp.Maximize(cp.sum(r))

    # Solve the optimization problem
    problem = cp.Problem(objective, constraints)
    result = problem.solve()

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
