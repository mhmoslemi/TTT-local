import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # To fit in a square of size 1, we can set the maximum side to 1

    grid_width = 2 * cols

    scale = 1 / max(grid_width, grid_height)
    r = 1 * scale / 2  # Maximum possible radius before touching boundaries

    centers = np.zeros((n, 2))

            centers[idx] = [x, y]
