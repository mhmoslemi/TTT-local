import random
import math

def run_packing():

    cooling_rate = 0.995
    max_iterations = 5000
    perturbation_strength = 0.1

    radii = np.random.rand(num_circles) * 0.2  # Initial guess for radii

    def adjust_radii(centers, radii):
        new_radii = np.copy(radii)
        for i in range(len(centers)):
            x, y = centers[i]
            new_r = min(radii[i], 1 - x, 1 - y, x, y)
            new_radii[i] = new_r
        return new_radii

            for j in range(i + 1, len(centers)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = math.hypot(dx, dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

        new_centers = np.copy(centers)
        for i in range(len(centers)):

                dx = centers[i, 0] - centers[j, 0]

                dist = math.hypot(dx, dy)

                    move_amount = overlap / 2
                    dx_normalized = dx / dist
                    dy_normalized = dy / dist
                    new_centers[i, 0] += dx_normalized * move_amount
                    new_centers[i, 1] += dy_normalized * move_amount
                    new_centers[j, 0] -= dx_normalized * move_amount
