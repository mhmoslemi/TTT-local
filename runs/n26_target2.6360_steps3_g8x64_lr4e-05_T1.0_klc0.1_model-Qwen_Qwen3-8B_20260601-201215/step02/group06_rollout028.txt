import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Improved initial arrangement with better spacing
    # Hexagonal grid with 5 rows, 6 circles per row (with slight offset)
    # Using a more compact arrangement
    spacing = 0.18
    row_height = np.sqrt(3) * spacing / 2
    y_start = 0.1
    y_end = 1 - 0.1
    
    # Determine rows and columns
    rows = int((y_end - y_start) / row_height) + 1
    cols = int(1 / spacing) + 1
    
    # Calculate row positions and column positions
    y_positions = np.linspace(y_start, y_end, rows)
    x_positions = np.linspace(0.1, 1 - 0.1, cols)
    
    # Fill the centers in a hexagonal pattern
    for i, y in enumerate(y_positions):
        for j, x in enumerate(x_positions):
            # Alternate row offset for hexagonal packing
            if i % 2 == 1:
                x += spacing / 2
            if x + spacing > 1:
                x = 1 - spacing
            centers[i * cols + j, 0] = x
            centers[i * cols + j, 1] = y

    # Trim to 26 circles if needed
    if num_circles < len(centers):
        centers = centers[:num_circles]
        radii = radii[:num_circles]

    # Initial radii: start with smaller radius and gradually increase
    initial_radius = 0.04
    radii[:] = initial_radius

    def objective(radii):
        return -np.sum(radii)

    def constraint_overlap(i, j, centers, radii):
        dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
        return dist - radii[i] - radii[j]

    def constraint_boundary(i, centers, radii):
        x, y = centers[i]
        r = radii[i]
        return x - r  # Left boundary

    def constraint_boundary_right(i, centers, radii):
        x, y = centers[i]
        r = radii[i]
        return 1.0 - x - r  # Right boundary

    def constraint_boundary_top(i, centers, radii):
        x, y = centers[i]
        r = radii[i]
        return 1.0 - y - r  # Top boundary

    def constraint_boundary_bottom(i, centers, radii):
        x, y = centers[i]
        r = radii[i]
        return y - r  # Bottom boundary

    # Optimization
    bounds = [(0, 1) for _ in range(num_circles)]
    constraints = []

    for i in range(num_circles):
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_boundary(i, centers, r)})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_boundary_right(i, centers, r)})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_boundary_top(i, centers, r)})
        constraints.append({'type': 'ineq', 'fun': lambda r, i=i: constraint_boundary_bottom(i, centers, r)})

    # Optimize
    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        options={'ftol': 1e-9, 'maxiter': 500}
    )

    # Return the result
    return (centers, result.x, -result.fun)
