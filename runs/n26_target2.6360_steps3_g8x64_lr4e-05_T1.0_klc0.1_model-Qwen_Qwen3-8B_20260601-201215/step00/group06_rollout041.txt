```python
def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.zeros((n, 2))
    idx = 0
    for i in range(grid_size):
        for j in range(grid_size):
            if idx < n:
                centers[idx, 0] = i / (grid_size - 1)
                centers[idx, 1] = j / (grid_size - 1)
                idx += 1

    # Initial guess for radii: all equal and small
    radii = np.full(n, 0.05)

    # Objective: maximize the sum of radii
    def objective(x):
        return -x[0]  # Negative because we use minimization

    constraints = []

    # Inequality constraints: boundary and non-overlap
    def boundary_constraints(x):
        # x contains [r0, r1, ..., r25] for radii, then [x0, y0, x1, y1, ..., x25, y25] for centers
        # Extract radii and centers

        centers = x[n:]
        centers = centers.reshape((n, 2))
        
        # Boundary constraints
        for i in range(n):
            r = radii[i]
            x_center = centers[i, 0]
            y_center = centers[i, 1]
            if x_center - r < -1e-12 or x_center + r > 1 + 1e-12:
                return np.inf
            if y_center - r < -1e-12 or y_center + r > 1 + 1e-12:
                return np.inf
        
        # Non-overlap constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        
        return 0  # No constraints violated

    def boundary_func(x):
        return boundary_constraints(x)

    bounds = []

        bounds.append((0.0, 1.0))  # Radii between 0 and 1

    x0 = np.concatenate((radii, centers.flatten()))

    # Setup optimization
    result = opt.minimize(
```

```python
def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
