import scipy.optimize as optimize
import scipy.spatial.distance as dist

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26

    # Generate initial centers in a grid-like pattern
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # For simplicity, we will place circles in a hexagonal-like grid, adjusted for the square
    # This is a simple initial guess to be refined
    
    # Define grid parameters
    cols = 5
    rows = 6
    offset = 1 / (cols + 1)
    vertical_offset = np.sqrt(3) / 2 * offset  # Hexagonal offset

    # Populate initial centers
    for i in range(rows):
        for j in range(cols):
            x = j * offset + offset / 2
            y = i * vertical_offset + offset / 2
            centers[i * cols + j] = [x, y]

    # Initialize all radii to a small value (start with uniform)
    radii[:] = 0.025

    # Define objective: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraints: boundary and non-overlapping
    def constraint_boundary(radii, centers):
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return np.inf
        return 0

    def constraint_overlap(radii, centers):
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

    # Define a wrapper to pass centers as a fixed parameter
    def optimize_objective(radii_vec):
        radii = radii_vec
        return objective(radii)

    def optimize_constraints(radii_vec):
        radii = radii_vec
        return constraint_boundary(radii, centers) + constraint_overlap(radii, centers)

    # Combine constraints
    cons = [

    bounds = [(0, 1) for _ in range(n)]

    result = optimize.minimize(optimize_objective, radii, method='SLSQP', bounds=bounds, constraints=cons, tol=1e-6)
