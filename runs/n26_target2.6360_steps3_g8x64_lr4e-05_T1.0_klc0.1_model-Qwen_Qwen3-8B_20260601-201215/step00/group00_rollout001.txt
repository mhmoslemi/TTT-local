import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Total number of circles
    n = 26
    
    # Parameters
    rows = int(np.sqrt(n))  # number of rows in the hexagonal packing
    cols = (n + rows - 1) // rows  # number of columns
    
    # Radius of the largest circle that can fit without overlap
    r_max = 0.135  # This is a rough estimate based on hexagonal packing in a square
    
    # Initialize centers and radii
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    vertical_spacing = (np.sqrt(3)/2) * r_max * 2  # For hexagonal packing
    
    # Distribute circles in a hexagonal grid
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * horizontal_spacing
        y = row * vertical_spacing
        # Adjust y for even rows to create hexagonal spacing
        if row % 2 == 1:
            y += vertical_spacing / 2
        centers[i] = [x, y]
        radii[i] = r_max
    
    # Function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
