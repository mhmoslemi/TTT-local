import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define number of circles
    n_circles = 26

    # Initial hexagonal grid layout
    rows = 5
    cols = 5
    spacing = 0.25
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    for i in range(rows):
        for j in range(cols):
            idx = i * cols + j
            if i % 2 == 0:
                x = j * spacing
            else:
                x = (j + 0.5) * spacing
            y = i * spacing * np.sqrt(3) / 2
            centers[idx] = [x, y]
            radii[idx] = 0.05  # Initial guess for radius

    # Define objective function for optimization
    def objective(radii_flat):
        radii = radii_flat.reshape(n_circles)
        total_sum = np.sum(radii)
        return -total_sum  # Minimize negative to maximize

    # Define constraints for optimization
    def constraint1(radii_flat):
        radii = radii_flat.reshape(n_circles)
        for i in range(n_circles):
            x, y = centers[i]
            if x - radii[i] < -1e-12 or x + radii[i] > 1 + 1e-12 or y - radii[i] < -1e-12 or y + radii[i] > 1 + 1e-12:
                return -1
        return 0

    def constraint2(radii_flat):
        radii = radii_flat.reshape(n_circles)
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -1
        return 0

    # Optimization bounds
    bounds = [(0, 0.2) for _ in range(n_circles)]

    options = {
        'maxiter': 1000,
        'ftol': 1e-8,
        'disp': False
    }

    # Initial guess for radii
    initial_guess = radii.copy()

    # Run optimization
    result = optimize.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint1},

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
