import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial layout: hexagonal packing
    n = 26
    radius = 1 / np.sqrt(2) * 0.8  # Starting radius based on hexagonal packing in unit square
    initial_centers = []

    # Generate hexagonal grid
    for row in range(5):
        for col in range(6):
            if row % 2 == 0:
                x = col * (1.5 * radius)
                y = row * (np.sqrt(3) * radius)
            else:
                x = (col + 0.5) * (1.5 * radius)
                y = row * (np.sqrt(3) * radius)
            initial_centers.append([x, y])

    # Adjust to fit into unit square and scale properly
    initial_centers = np.array(initial_centers)
    scale = 1.0 / np.max(initial_centers, axis=0)
    initial_centers *= scale

    # Add some randomness to initial positions to explore better arrangements
    np.random.seed(42)
    initial_centers += np.random.uniform(-0.05, 0.05, size=initial_centers.shape)

    # Use a constrained optimization to maximize the radii
    def objective_function(radii):
        return -np.sum(radii)  # Minimize the negative sum to maximize sum

    # Constraints: circles do not overlap, and fit within the unit square
    def constraints_func(radii, centers):
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j])**2))
                constraints.append(dist - (radii[i] + radii[j]) >= 1e-6)
        for i in range(n):
            x, y = centers[i]
            constraints.append(x - radii[i] >= -1e-6)
            constraints.append(x + radii[i] <= 1 + 1e-6)
            constraints.append(y - radii[i] >= -1e-6)
            constraints.append(y + radii[i] <= 1 + 1e-6)
        return constraints

    # Use Sequential Least Squares Programming (SLSQP) to optimize
    result = opt.minimize(
        objective_function,
        initial_radii := np.full(n, radius),
        method='SLSQP',
        bounds=[(0, 1.0) for _ in range(n)],
        constraints={
            'type': 'ineq',
            'fun': lambda radii: constraints_func(radii, initial_centers)
        }
    )

    # Extract results
    centers = initial_centers
    radii = result.x
    sum_radii = np.sum(radii)

    # Validate the packing
    is_valid, message = validate_packing(centers, radii)
    if not is_valid:
        print(f"Validation failed: {message}")
    else:
        print(f"Packing validated. Total sum of radii = {sum_radii}")

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
