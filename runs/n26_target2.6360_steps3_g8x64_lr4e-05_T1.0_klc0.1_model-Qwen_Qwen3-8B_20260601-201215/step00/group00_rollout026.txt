import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: 26 circles placed in a grid-like pattern
    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # First place all circles in a grid pattern
    grid_rows = int(np.ceil(np.sqrt(num_circles)))
    grid_cols = int(np.ceil(num_circles / grid_rows))
    
    # Distribute the circles in the square
    for i in range(num_circles):
        x = i % grid_cols
        y = i // grid_cols
        radius = 0.1  # Start with a small uniform radius
        centers[i, 0] = x / (grid_cols - 1) if grid_cols > 1 else 0.5
        centers[i, 1] = y / (grid_rows - 1) if grid_rows > 1 else 0.5
        radii[i] = radius

    # Define the objective function (sum of radii)
    def objective(x):
        return -np.sum(x)

    # Define the constraints
    def constraint_boundary(x):
        centers_vec = x[:num_circles*2].reshape((num_circles, 2))
        radii_vec = x[num_circles*2:]
        for i in range(num_circles):
            x_i, y_i = centers_vec[i]
            r_i = radii_vec[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf
        return 0

    def constraint_overlap(x):
        centers_vec = x[:num_circles*2].reshape((num_circles, 2))
        radii_vec = x[num_circles*2:]
        for i in range(num_circles):
            for j in range(i+1, num_circles):
                dx = centers_vec[i, 0] - centers_vec[j, 0]
                dy = centers_vec[i, 1] - centers_vec[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_vec[i] + radii_vec[j] - 1e-12:
                    return np.inf
        return 0

    # Set up the optimization problem
    bounds = []
    for _ in range(num_circles):
        bounds.append((0, 1))  # x-coordinate constraint
        bounds.append((0, 1))  # y-coordinate constraint
    for _ in range(num_circles):
        bounds.append((0, 1))  # radius constraint

    # Initial guess for the optimization
    x0 = np.concatenate([centers.flatten(), radii])

    # Run optimization
    result = opt.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_boundary},
            {'type': 'ineq', 'fun': constraint_overlap}
        ],
        method='SLSQP',
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    # Extract the final solution
    centers_opt = result.x[:num_circles*2].reshape((num_circles, 2))
    radii_opt = result.x[num_circles*2:]
    sum_radii = np.sum(radii_opt)

    return centers_opt, radii_opt, sum_radii
