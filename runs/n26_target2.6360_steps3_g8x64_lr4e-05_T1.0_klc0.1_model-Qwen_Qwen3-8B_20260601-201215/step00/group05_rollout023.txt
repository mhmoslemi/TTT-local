import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    max_iterations = 1000
    tolerance = 1e-6

    # Initial hexagonal packing (regular grid)
    # Calculate the number of rows and columns
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))

    centers = np.zeros((n_circles, 2))

        col = i % cols

        y = row / (rows - 1) if rows > 1 else 0.5
        centers[i] = [x, y]

    spacing = 1.0 / (cols - 1)

        radii[i] = r

    x0 = np.concatenate([centers.flatten(), radii])

    # Define the objective function
    def objective(x):
        centers_flatten = x[:n_circles*2].reshape(n_circles, 2)
        radii_flatten = x[n_circles*2:]
        return -np.sum(radii_flatten)  # Minimize negative sum to maximize sum

    # Define constraints
    def constraints(x):
        centers_flatten = x[:n_circles*2].reshape(n_circles, 2)
        radii_flatten = x[n_circles*2:]
        constraints = []

        # Bound constraints
        for i in range(n_circles):
            # Ensure all circles are within the unit square
            if centers_flatten[i, 0] - radii_flatten[i] < -1e-12:

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
