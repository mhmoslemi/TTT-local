import scipy.optimize as opt

def generate_hexagonal_centers(n, spacing):
    centers = []
    row = 0
    col = 0
    while len(centers) < n:
        x = col * spacing
        y = row * (np.sqrt(3) / 2) * spacing
        centers.append((x, y))
        col += 1
        if col % 2 == 0:
            row += 1
    return centers

def normalize_centers(centers, square_size=1.0):

    max_x = max(c[0] for c in centers)
    max_y = max(c[1] for c in centers)

    height = max_y - min_y
    
    normalized = []
    for x, y in centers:
        nx = (x - min_x) / width
        ny = (y - min_y) / height
        normalized.append((nx, ny))
    return normalized
