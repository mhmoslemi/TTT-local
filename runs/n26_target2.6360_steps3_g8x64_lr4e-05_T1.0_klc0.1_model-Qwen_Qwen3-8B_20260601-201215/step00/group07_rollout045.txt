import scipy.optimize as optimize

def run_packing():

    grid = np.array([[i / 5, j / 5] for i in range(6) for j in range(6)])
    centers = grid[:26]
    radii = np.full(26, 0.05)  # Initial guess

    # Objective function to maximize the sum of radii
    def objective(x):
        return -x[0]  # Minimize negative of sum of radii

    # Variables: x[0:52] = centers of 26 circles (26 * 2)
    #             x[52:78] = radii of 26 circles
    #             x[78:] = auxiliary variables for constraint handling
    n = 26
    bounds = []
    for i in range(n):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
        bounds.append((0, 1))  # radius
    bounds.extend([(0, 1) for _ in range(n)])  # Auxiliary variables

    # Constraints
    def constraints(x):
        centers = x[:n * 2].reshape(n, 2)
        radii = x[n * 2: n * 2 + n]
        constraints = []

        # Boundary constraints
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            constraints.append([x_i - r_i, 1 + r_i - x_i, y_i - r_i, 1 + r_i - y_i])

        # Non-overlapping constraints
        for i in range(n):

                dx = centers[i, 0] - centers[j, 0]

                dist = np.sqrt(dx ** 2 + dy ** 2)
                constraints.append([dist - (radii[i] + radii[j])])

        return np.array(constraints)

    result = optimize.minimize(

        x0=np.concatenate([centers.flatten(), radii, np.zeros(n)]),
        bounds=bounds,

        method='SLSQP',
        tol=1e-8
