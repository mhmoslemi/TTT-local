import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial packing: Hexagonal grid
    num_circles = 26
    grid_rows = 5
    grid_cols = 6
    spacing = 1.0 / (grid_rows - 1)  # Spacing between rows
    spacing_col = 1.0 / (grid_cols - 1)  # Spacing between columns

    # Create initial positions
    centers = np.zeros((num_circles, 2))
    for i in range(grid_rows):
        for j in range(grid_cols):
            row = i
            col = j
            centers[i * grid_cols + j] = [col * spacing_col, row * spacing]

    # Radius starts from small value, but we will optimize
    radii = np.full(num_circles, 0.01)

    # Define objective function to maximize sum of radii
    def objective(x):
        # x is flattened array of centers and radii
        centers_flat = x[:num_circles * 2].reshape((num_circles, 2))
        radii_flat = x[num_circles * 2:].reshape((num_circles,))
        sum_radii = np.sum(radii_flat)
        return -sum_radii  # Minimize negative to maximize

    # Define constraints: non-overlapping, staying within the unit square
    def constraint_overlap(x):
        centers_flat = x[:num_circles * 2].reshape((num_circles, 2))
        radii_flat = x[num_circles * 2:].reshape((num_circles,))
        violations = []
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-8:
                    violations.append(1)
        return np.sum(violations)

    def constraint_boundary(x):

            r = radii_flat[i]

                violations.append(1)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
