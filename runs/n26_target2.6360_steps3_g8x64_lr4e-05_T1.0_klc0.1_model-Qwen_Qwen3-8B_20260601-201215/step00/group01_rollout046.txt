import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: evenly spaced grid of centers with a certain radius
    n = 26
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // num_cols
        col = i % num_cols

        centers[i, 1] = row / (num_rows - 1) if num_rows > 1 else 0.5

    # Initial radii: all circles start with the same radius, adjusted to fit the square
    initial_radius = 0.05

    def objective(x):
        return -np.sum(x)  # Negative because we use minimization

    def constraint_bounds(x):
