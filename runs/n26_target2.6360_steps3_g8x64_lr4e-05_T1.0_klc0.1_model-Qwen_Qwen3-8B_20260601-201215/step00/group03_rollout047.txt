import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    
    radii_guess = np.full(n, 0.03)  # Initial guess for radii
    centers = np.zeros((n, 2))
    
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col * (1 + 0.1) / num_cols
        y = row * (1 + 0.1) / num_rows
        centers[i] = [x, y]
    
    # Define the objective function to maximize sum of radii
    def objective(x):
        r = x[:n]
        return -np.sum(r)  # Negative for minimization
    
    # Define constraints
    def constraint(x):
        x_centers = x[n:]
        r = x[:n]
        constraints = []
        # Boundary constraint
        for i in range(n):
            x_i, y_i = x_centers[i*2], x_centers[i*2+1]
            r_i = r[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return np.inf
            constraints.append(x_i - r_i)
            constraints.append(x_i + r_i)
            constraints.append(y_i - r_i)
            constraints.append(y_i + r_i)
        # Non-overlap constraint
        for i in range(n):
            for j in range(i+1, n):
                dx = x_centers[i*2] - x_centers[j*2]
                dy = x_centers[i*2+1] - x_centers[j*2+1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < r[i] + r[j] - 1e-12:
                    return np.inf
        return 0
    
    # Optimization bounds
    bounds = []
    for i in range(n):
        bounds.append((0.01, 0.1))  # Radii constraint
    for i in range(n):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
    
    # Optimization
    initial_guess = np.concatenate((radii_guess, centers.flatten()))
    result = optimize.minimize(
        objective,
        initial_guess,
        bounds=bounds,
        constraints={"type": "ineq", "fun": constraint},
