from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based arrangement
    n = 26
    rows = 5
    cols = 6
    spacing = 1.0 / (cols - 1) * 0.9  # 0.9 to leave space for radii
    
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * spacing
        y = row * spacing
        centers[i] = [x, y]
        radii[i] = 0.03  # Initial guess for radii
    
    # Define the objective function: maximize sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Negative for minimization
    
    # Define the constraints
    def constraint_overlap(radii_flat):
        centers_flat = centers.flatten()
        n = len(radii_flat)
        constraints = []
        for i in range(n):

                idx_i = i
                idx_j = j

                xj = centers_flat[2 * j]
                yj = centers_flat[2 * j + 1]
                ri = radii_flat[i]
                rj = radii_flat[j]
                dist = np.sqrt((xi - xj)**2 + (yi - yj)**2)
                constraints.append(dist - (ri + rj) + 1e-12)
        return np.array(constraints)
    
    def constraint_boundary(radii_flat):
        constraints = []

            ri = radii_flat[i]

        return np.array(constraints)
