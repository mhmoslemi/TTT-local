import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid of 26 points in the square [0, 1] x [0, 1]
    n_circles = 26
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        row = i // 5
        col = i % 5
        centers[i, 0] = col / 4 + 0.125  # Offset to avoid touching the edge
        centers[i, 1] = row / 4 + 0.125  # Offset to avoid touching the edge

    # Initial radius: assume all circles are the same size
    radii = np.full(n_circles, 0.05)

    # Objective function: maximize the sum of radii
    def objective(x):
        return -np.sum(x)  # Negative for minimization

    # Constraints
    # - Non-overlap: distance between centers i and j >= radii[i] + radii[j]
    # - Radii >= 0
    # - Circle must be inside the square: radius <= 1 - max(center_x, 1 - center_x) and similar for y

    # Flatten centers and radii
    x = np.concatenate((centers.flatten(), radii))

    # Define constraints
    constraints = []

    # Radii >= 0
    for i in range(n_circles):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: x[len(x) - n_circles + i]})

    # Boundary constraints: radius <= 1 - center_x (left/right) and radius <= 1 - center_y (top/bottom)
    for i in range(n_circles):
        # Left and right boundary
        idx_center_x = len(x) - n_circles + i
        idx_radius = len(x) - n_circles + i
        constraints.append({'type': 'ineq', 'fun': lambda x, idx_center_x=idx_center_x, idx_radius=idx_radius:
                           x[idx_radius] - (1 - abs(x[idx_center_x] - 0.5))})

        # Top and bottom boundary
        idx_center_y = len(x) - n_circles + i
        idx_radius = len(x) - n_circles + i
        constraints.append({'type': 'ineq', 'fun': lambda x, idx_center_y=idx_center_y, idx_radius=idx_radius:
                           x[idx_radius] - (1 - abs(x[idx_center_y] - 0.5))})

    # Non-overlap constraints (min distance between centers >= sum of radii)
    for i in range(n_circles):
        for j in range(i + 1, n_circles):
            idx_center_i = len(x) - n_circles + i
            idx_center_j = len(x) - n_circles + j
            idx_radius_i = len(x) - n_circles + i
            idx_radius_j = len(x) - n_circles + j

            def constraint(x):
                center_i = x[:len(x) - n_circles]
                center_j = x[len(x) - n_circles: len(x) - n_circles + n_circles]
                radius_i = x[len(x) - n_circles:]
                return np.sqrt((center_i[2*i] - center_j[2*j]) ** 2 + (center_i[2*i + 1] - center_j[2*j + 1]) ** 2) - (radius_i[i] + radius_i[j])

            constraints.append({'type': 'ineq', 'fun': constraint})

    # Bounds: centers in [0, 1], radii >= 0
    bounds = []
    for i in range(n_circles):
        bounds.append((0, 1))  # x-coordinate
        bounds.append((0, 1))  # y-coordinate
        bounds.append((0, 1))  # radius

    # Run the optimization
    result = optimize.minimize(
        fun=objective,
        x0=x,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
        tol=1e-6,
        options={'maxiter': 1000}
    )

    # Extract final centers and radii
    final_centers = result.x[:2 * n_circles].reshape(n_circles, 2)
    final_radii = result.x[2 * n_circles:]
    final_sum = np.sum(final_radii)

    # Validate the packing
    valid, msg = validate_packing(final_centers, final_radii)
    if not valid:

    return final_centers, final_radii, final_sum

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
