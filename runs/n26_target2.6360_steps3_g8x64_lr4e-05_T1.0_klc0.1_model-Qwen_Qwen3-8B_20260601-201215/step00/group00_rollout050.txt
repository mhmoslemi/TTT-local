```python
    radii = []

    # We will start with an approximate spacing to fit all 26 circles

    rows = int(np.ceil(np.sqrt(total_circles)))
    cols = int(np.ceil(np.sqrt(total_circles)))
    
    # Start with a spacing of 0.25
    spacing = 0.25
    
    for i in range(rows):
        for j in range(cols):
            x = j * spacing + 0.5 * (spacing / 2)

            centers.append([x, y])
            radii.append(0.05)
    
    # Adjust the grid to fit the square

    radii = np.array(radii)

    # For now, use the initial radii as a starting point

        return -np.sum(radii)  # Minimize negative sum to maximize sum

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return -1
        return 0
    
    def constraint_overlap(radii):

            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)

                    return -1
        return 0
```

```python
def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
