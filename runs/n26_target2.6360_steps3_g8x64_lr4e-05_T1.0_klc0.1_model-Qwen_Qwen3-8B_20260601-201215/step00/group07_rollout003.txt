import scipy.optimize as optimize

def run_packing():

    n_circles = 26

    # Each row alternates between horizontal offsets to simulate hexagonal packing
    # We choose 5 rows, each with 5 or 6 circles (adjust for the unit square)
    rows = 5
    cols = 5
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    # Generate the positions of the centers for the initial hexagonal grid
    # Define hexagonal packing parameters
    horizontal_spacing = 1.0 / (cols - 1)

        for j in range(cols):
            if i % 2 == 1:
                x = j * horizontal_spacing + offset
            else:
                x = j * horizontal_spacing
            y = i * vertical_spacing
            centers[i * cols + j] = [x, y]
    
    # Step 2: Initial radii estimation
    # Each radius initially assigned as minimum distance to the boundary
    for i in range(n_circles):
        x, y = centers[i]
        # Minimum distance to the boundary is the minimum of x, 1-x, y, 1-y
        rad = min(x, 1 - x, y, 1 - y)
        radii[i] = rad
