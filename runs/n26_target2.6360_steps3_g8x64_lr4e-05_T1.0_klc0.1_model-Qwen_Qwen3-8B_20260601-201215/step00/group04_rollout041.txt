import scipy.optimize as opt

def run_packing():

    n = 26

    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    centers = np.zeros((n, 2))

    radii = np.zeros(n) + r_initial

        centers[i, 0] = x

    def objective(x):

        return -np.sum(x)  # Negative for minimization
    
    # Constraints: boundary containment and non-overlap
    def constraint_boundary(x, centers):
        # x contains radii
        for i in range(n):
            x_i = x[i]

                return False
            if centers[i, 1] - x_i < 0 or centers[i, 1] + x_i > 1:
                return False
        return True

        # x contains radii
        for i in range(n):

    ]

        method='SLSQP',
        bounds=bounds,
        constraints=constraints,
