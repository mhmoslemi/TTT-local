import numpy as np
import random
import math
import cvxpy as cp

def run_packing():

    num_circles = 26
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Start with a regular grid pattern
    rows = math.ceil(math.sqrt(num_circles))
    cols = math.ceil(num_circles / rows)

    for i in range(num_circles):
        row = i // cols
        col = i % cols
        centers[i, 0] = col / (cols - 1) if cols > 1 else 0.5
        centers[i, 1] = row / (rows - 1) if rows > 1 else 0.5
        radii[i] = 0.01  # Small initial radius

            best_radii = radii_perturbed.copy()

            temp *= cooling_rate

                best_radii = radii_perturbed.copy()

    # Final refinement with CVXPY
    x = cp.Variable((num_circles, 2))
    r = cp.Variable(num_circles)
    obj = cp.Maximize(cp.sum(r))

    constraints = []
    for i in range(num_circles):
        constraints.append(cp.potential(x[i, 0], x[i, 1], r[i]) <= 1)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
