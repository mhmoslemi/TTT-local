import numpy as np
from scipy.optimize import minimize
from scipy.spatial import distance

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial configuration: a grid-like arrangement of 26 circles
    # Try a 5x5 grid with 1 extra circle
    grid_size = 5
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    
    # Initial grid placement
    index = 0
    for i in range(grid_size):
        for j in range(grid_size):
            if index < n:
                centers[index, 0] = (i + 0.5) / (grid_size - 1)
                centers[index, 1] = (j + 0.5) / (grid_size - 1)
                radii[index] = 0.1  # initial guess for radius
                index += 1
            else:
                break

    # Function to calculate total radius sum and constraints
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Constraints to ensure no overlaps and within the square
    def constraint_overlap(i, j, radii):
        def cons(radii):
            r1 = radii[i]
            r2 = radii[j]
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx**2 + dy**2)
            return dist - (r1 + r2) + 1e-8  # Ensure non-overlapping
        return cons

    # Set up optimization problem
    bounds = [(0, 1.0) for _ in range(n)]  # Radii must be non-negative
    cons = []
    for i in range(n):
        for j in range(i + 1, n):
            cons.append({'type': 'ineq', 'fun': constraint_overlap(i, j, radii)})

    # Run optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6
    )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
