import numpy as np
import scipy.optimize as optimize
from typing import Tuple, Any

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initialize grid of centers for square packing
    n = 26
    grid_size = int(np.sqrt(n))
    grid = np.linspace(0.1, 0.9, grid_size)  # Avoid edges
    centers = np.array([x for x in np.meshgrid(grid, grid) for x in x.flatten()])
    centers = centers.reshape((grid_size, grid_size, 2))
    centers = centers[centers.shape[0] // 2 - 4:centers.shape[0] // 2 + 4, :, :]
    centers = centers.reshape(-1, 2)

    # Initial guess for radii
    radii = np.full(n, 0.05)

    # Objective function: maximize sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    # Constraint function: ensure no overlap and within bounds
    def constraint(radii, centers):
        # Check boundaries
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return np.inf
        # Check pairwise distances
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

    # Optimization
    def objective_func(radii):
        return objective(radii, centers)

    def constraint_func(radii):
        return constraint(radii, centers)

    bounds = [(0.0, 0.5) for _ in range(n)]
    res = optimize.minimize(
        objective_func,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraint_func}],
        bounds=bounds,
        tol=1e-6
    )

    # Return the best packing
    best_centers = centers
    best_radii = res.x
    sum_radii = np.sum(best_radii)

    # Final check for validity
    valid, msg = validate_packing(best_centers, best_radii)
    if not valid:
        raise ValueError(f"Invalid packing: {msg}")

    return best_centers, best_radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
