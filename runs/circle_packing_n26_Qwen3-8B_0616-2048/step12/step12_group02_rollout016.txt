import numpy as np
import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    num_circles = 26

    # Initial refined hexagonal grid filling the square
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    row_spacing = 1.0 / rows
    col_spacing = 1.0 / cols

    centers = np.zeros((num_circles, 2))
    idx = 0

    for i in range(rows):
        for j in range(cols):
            if idx >= num_circles:
                break
            if i % 2 == 0:
                x = j * col_spacing + 0.5 * col_spacing
            else:
                x = (j + 0.5) * col_spacing
            y = i * row_spacing + 0.5 * row_spacing
            centers[idx] = [x, y]
            idx += 1

    # Add remaining circles with random offset near the center
    if idx < num_circles:
        remaining = num_circles - idx
        random_centers = np.random.rand(remaining, 2)
        random_centers = random_centers * 0.8 + 0.1  # Adjusted to keep within unit square
        centers[idx:] = random_centers

    # Function to compute and validate the objective
    def objective(x):
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:].reshape(num_circles)

        is_valid, _ = validate_packing(centers_flat, radii_flat)

            return float('inf')

    center_bounds = [(0.0, 1.0) for _ in range(2 * num_circles)]

    bounds = center_bounds + radius_bounds

    # Initial guess
    x0 = np.concatenate((centers.flatten(), np.full(num_circles, 0.1)))

    # Optimization steps
    # Step 1: Use SLSQP to globally search
    result_slsqp = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        tol=1e-8,
        options={'maxiter': 200, 'disp': False}
    )

    # Step 2: Use Nelder-Mead for local optimization
    result_nm = opt.minimize(
        objective,
        result_slsqp.x,
        method='Nelder-Mead',
        bounds=bounds,
        tol=1e-10,
        options={'maxiter': 100, 'disp': False}
    )

    # Step 3: Use L-BFGS-B for local refinement
    result_bfgs = opt.minimize(
        objective,
        result_nm.x,
        method='L-BFGS-B',
        bounds=bounds,
        tol=1e-12,

    sum_radii = np.sum(radii_final)

    return centers_final, radii_final, sum_radii
