    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    bounds = [(0, 1) for _ in range(2 * n)]
    x_bounds = [(0, 1) for _ in range(n)]
    y_bounds = [(0, 1) for _ in range(n)]
    bounds = [(x_bounds[i], y_bounds[i]) for i in range(n)] + [(0, 1) for _ in range(n)]

    constraints = []

    def boundary_constraint(radii):

            if x - radii[n + i] < -1e-10 or x + radii[n + i] > 1 + 1e-10:
