import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Each row is offset by a half circle diameter horizontally
    # Radius is initially set so the circles fit roughly within the square

    # Number of circles
    n = 26

    # Initial guess for center positions (hexagonal grid)
    centers = np.zeros((n, 2))
    radii = np.ones(n) * 0.1  # Initial radius

    # Generate hexagonal grid
    num_rows = 5
    row_spacing = 1.0
    col_spacing = np.sqrt(3)/2
    for i in range(num_rows):
        for j in range((n + num_rows) // 2):
            if i % 2 == 0:
                x = j * col_spacing
            else:
                x = (j + 0.5) * col_spacing
            y = i * row_spacing
            if i < num_rows and j < n // num_rows:
                centers[i * n // num_rows + j, 0] = x
                centers[i * n // num_rows + j, 1] = y

    # Set initial radii so circles fit the square
    max_radius = 0.1
    radii = max_radius * np.ones(n)

    # Define objective function to maximize total radius sum
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraints to ensure circles are within the square and do not overlap
    def constraint_center(x, i):
        x_pos, y_pos = x[2*i], x[2*i+1]
        r = x[n*2 + i]
        return (x_pos - r >= 0) and (x_pos + r <= 1) and (y_pos - r >= 0) and (y_pos + r <= 1)

    def constraint_overlap(x, i, j):
        x1, y1 = x[2*i], x[2*i+1]
        x2, y2 = x[2*j], x[2*j+1]
        r1, r2 = x[n*2 + i], x[n*2 + j]
        dist = np.sqrt((x1 - x2)**2 + (y1 - y2)**2)
        return dist >= r1 + r2 - 1e-8  # Allow small overlap due to numerical precision

    # Create flattened variables for optimization
    x0 = np.concatenate((centers.flatten(), radii))
    bounds = [(-1, 1) for _ in range(2*n)] + [(0, 1) for _ in range(n)]

    # Define constraints in the form required by scipy.optimize
    constraints = []

    for i in range(n):
        constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_center(x, i)})
    
    for i in range(n):
        for j in range(i+1, n):
            constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)})

    # Optimize
    result = opt.minimize(

        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints,

    )

        radii_opt = radii

    total_radius = np.sum(radii_opt)
    
    return centers_opt, radii_opt, total_radius

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
