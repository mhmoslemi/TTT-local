import numpy as np
import scipy.optimize as opt

def run_packing():
    n = 26  # Number of circles

    # Initial guess for positions and radii
    # We'll use a hexagonal grid arrangement
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # First, we generate a hexagonal-like grid inside the unit square
    # We'll adjust the coordinates to fit within [0,1]²
    # Let's place 5 rows with 5, 6, 6, 6, 5 circles respectively

    # Initial radii (starting small, to be optimized)
    initial_radii = np.full(n, 0.04)

    # First, let's generate an initial layout (hexagonal grid)
    row_index = 0
    col_index = 0
    for i in range(n):
        # Hexagonal grid parameters
        # Horizontal spacing between circle centers
        h = np.sqrt(3) * initial_radii[i]  # Horizontal distance between circles
        # Vertical spacing
        v = 1.5 * initial_radii[i]  # Vertical distance between rows

        # Calculate x and y coordinates
        x = col_index * h
        y = row_index * v

        # Normalize to fit in [0,1]²
        x = x / (n * h)
        y = y / (n * v)

        # Adjust to fit within the square
        x = np.clip(x, 0, 1)
        y = np.clip(y, 0, 1)

        centers[i] = [x, y]

        # Increment column and row indices
        col_index += 1
        if col_index >= 5:
            row_index += 1
            col_index = 0

    # Define the objective function
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum (maximize sum)

    # Define the constraints
    def non_overlap_constraint(radii):
        centers_with_radii = np.column_stack((centers, radii))
        for i in range(n):
            for j in range(i + 1, n):
                x1, y1, r1 = centers_with_radii[i]
                x2, y2, r2 = centers_with_radii[j]
                distance = np.sqrt((x1 - x2)**2 + (y1 - y2)**2)
                if distance < r1 + r2 - 1e-12:
                    return -distance + r1 + r2 - 1e-12  # Enforce non-overlap
        return 0  # No overlap

    def boundary_constraint(radii):
        centers_with_radii = np.column_stack((centers, radii))
        for i in range(n):
            x, y, r = centers_with_radii[i]
            if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                return -min(x - r, 1 - x - r, y - r, 1 - y - r)  # Enforce boundary
        return 0  # Within boundary

    # Bounds for radii (non-negative)
    bounds = [(0, None) for _ in range(n)]

    # Optimization
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': non_overlap_constraint},
            {'type': 'ineq', 'fun': boundary_constraint}
        ]
    )

    # Extract the final solution
    centers_final = centers
    radii_final = result.x
    sum_radii = np.sum(radii_final)

    return centers_final, radii_final, sum_radii
