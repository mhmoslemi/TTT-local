import numpy as np
import scipy.optimize as opt
import matplotlib.pyplot as plt

def run_packing():

    n = 26

    # Grid dimensions: we will use a 5x5 grid which has 25 points, so we will add an extra point at the center
    grid = np.array([
        [i/4, j/4] for i in range(5) for j in range(5)
    ])

    grid = np.vstack([grid, [0.5, 0.5]])

    def objective(radii):
        return -np.sum(radii)  # Minimize the negative sum to maximize sum

    def constraints(centers, radii):

        def in_square(centers, radii):
            for i in range(n):
                x, y = centers[i]
                r = radii[i]
                if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                    return False
            return True

        # Constraint 2: No two circles can overlap
        def no_overlap(centers, radii):
            for i in range(n):
                for j in range(i + 1, n):
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx**2 + dy**2)
                    if dist < radii[i] + radii[j] - 1e-12:
                        return False
            return True

        return in_square(centers, radii) and no_overlap(centers, radii)

    # Initial guesses for centers and radii
    centers_guess = grid
    radii_guess = np.full(n, 0.05)  # Initial guess for radius

    bounds = [(0, 1) for _ in range(2 * n)] + [(0, 1) for _ in range(n)]

        fun=objective,
        x0=np.concatenate([centers_guess.flatten(), radii_guess]),

        constraints=opt.Constraint(fun=lambda x: constraints(x[:2*n].reshape(n, 2), x[2*n:]), type='ineq')

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
