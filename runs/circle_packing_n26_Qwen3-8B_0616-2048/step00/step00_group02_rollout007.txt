import numpy as np
import scipy.optimize as optimize
import matplotlib.pyplot as plt

def run_packing():
    # Number of circles
    N = 26
    
    # Initial placement: hexagonal pattern in the unit square
    num_rows = 6
    num_cols = 6
    # Place circles in hexagonal pattern
    centers = np.zeros((N, 2))
    radii = np.zeros(N)

    for i in range(N):
        row = i // num_cols
        col = i % num_cols
        x = col * (1.0 / num_cols) + 0.5 / num_cols
        y = row * (1.0 / num_rows) + 0.5 / num_rows
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = 0.1  # Initial radius

    # Objective function to maximize total sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum for maximization

    # Constraints: 1. Each circle must be within the unit square
    def constraint_boundary(radii, centers):
        for i in range(N):
            x, y = centers[i]
            r = radii[i]
            if (x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1):
                return np.inf
        return 0

    # Constraints: 2. No two circles may overlap
    def constraint_overlap(radii, centers):
        for i in range(N):
            for j in range(i + 1, N):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

    # Combine all constraints
    def constraint_all(radii, centers):
        boundary_error = constraint_boundary(radii, centers)
        overlap_error = constraint_overlap(radii, centers)
        return np.max([boundary_error, overlap_error])

    # Define optimization variables
    bounds = [(0, 1.0) for _ in range(N)]  # Radii must be between 0 and 1

    # Initial guess for radii
    initial_radii = np.full(N, 0.1)

    # Set up optimization problem
    def objective_wrapper(x):
        return objective(x, centers)

    def constraint_wrapper(x):
        return constraint_all(x, centers)

    # Solve using Sequential Least Squares Programming (SLSQP)
    result = optimize.minimize(
        objective_wrapper,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[{'type': 'ineq', 'fun': constraint_wrapper}],
        tol=1e-6,
        options={'maxiter': 500}
    )

    # Extract final solution
    final_radii = result.x
    final_centers = centers.copy()

    # Compute total sum of radii
    sum_radii = np.sum(final_radii)

    # Plot the packing (optional)
    plt.figure(figsize=(6, 6))
    for i in range(N):

        r = final_radii[i]
        circle = plt.Circle((x, y), r, edgecolor='black', facecolor='blue', alpha=0.5)
        plt.gca().add_patch(circle)
    plt.xlim(0, 1)
    plt.ylim(0, 1)
    plt.axis('equal')
    plt.show()

    return final_centers, final_radii, sum_radii
