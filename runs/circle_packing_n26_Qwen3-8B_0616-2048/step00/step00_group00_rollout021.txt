import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Step 1: Initial hexagonal grid placement to start with a good base packing
    num_circles = 26
    num_rows = int(np.ceil(np.sqrt(num_circles)))
    num_cols = int(np.ceil(num_circles / num_rows))
    
    # Adjust for hexagonal grid (each row offset by half a circle radius in x-direction)
    # Initial radius is small to fit in square
    initial_radius = 0.03
    radii = np.full(num_circles, initial_radius)
    
    centers = np.zeros((num_circles, 2))
    for i in range(num_rows):
        for j in range(num_cols):
            col = j
            row = i
            x = col * (2 * initial_radius) + initial_radius
            y = row * (np.sqrt(3) * initial_radius) + initial_radius
            if i % 2 == 1:
                x += initial_radius
            centers[i * num_cols + j] = [x, y]
    
    # Step 2: Define objective function to maximize sum of radii
    def objective(x):
        # x is a flat array of centers and radii
        centers_flat = x[:2 * num_circles].reshape((num_circles, 2))
        radii_flat = x[2 * num_circles:]
        
        # Check if centers are within bounds and radii are >= 0
        if any(r < 0 for r in radii_flat):
            return -np.inf
        
        total_radius = np.sum(radii_flat)
        return -total_radius  # Minimize the negative of total radius

    # Step 3: Define constraints for the optimization
    def constraint_boundary(x):
        centers_flat = x[:2 * num_circles].reshape((num_circles, 2))
        radii_flat = x[2 * num_circles:]
        
        # Check boundary constraints

                return -np.inf
        return 0  # Satisfied

    def constraint_non_overlap(x):
        centers_flat = x[:2 * num_circles].reshape((num_circles, 2))
        radii_flat = x[2 * num_circles:]
        
        # Check non-overlap between all pairs

        return 0  # Satisfied

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
