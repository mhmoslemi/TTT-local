import scipy.optimize as optimize

def run_packing():

    cols = int(np.ceil(num_circles / rows))

    initial_radius = 1 / (2 * np.sqrt(3))  # For hexagonal packing in square

    centers /= max(np.max(centers, axis=0))

    def objective(x):

        radii = x[:, 2]

        total_radius = np.sum(radii)

        return -total_radius
