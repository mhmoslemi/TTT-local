import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial hexagonal grid placement
    # We generate a hexagonal grid of 26 circles with spacing adjusted for unit square
    def hexagonal_grid(n, spacing, width, height):
        grid = []
        for row in range(int(np.sqrt(n)) + 1):
            for col in range(int(np.sqrt(n)) + 1):
                x = col * spacing
                y = row * spacing * np.sqrt(3) / 2
                if x < width and y < height:
                    grid.append((x, y))
        return np.array(grid)
    
    # Create initial hexagonal grid of 26 circles
    initial_centers = hexagonal_grid(26, 0.2, 1.0, 1.0)
    
    # We will try a few starting points to find the best one
    # Initial radii are computed as half the spacing to avoid overlaps
    initial_radii = np.full(26, 0.1)
    
    def objective_function(params):
        # params[0:52] = centers, params[52:] = radii
        centers = params[:52].reshape(26, 2)
        radii = params[52:]
        
        # Check if radii are non-negative
        if np.any(radii < 0):
            return np.inf
        
        # Check if all circles are within the unit square
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return np.inf
        
        # Check for non-overlapping
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        
        # Compute sum of radii

    # Initial guess

    # Perform optimization
    result = optimize.minimize(objective_function, initial_guess, method='L-BFGS-B', bounds=bounds)

    sum_radii = np.sum(radii_opt)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
