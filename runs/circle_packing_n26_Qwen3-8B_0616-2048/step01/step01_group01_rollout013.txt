import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint
from scipy.optimize import NonlinearConstraint

def run_packing():
    # Initial configuration: hexagonal packing
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # First, generate a hexagonal arrangement
    # We will use a staggered grid with 5 rows
    rows = 5
    cols = 6
    spacing = 1.0 / (cols - 1)
    stagger = spacing / 2

    for i in range(rows):
        for j in range(cols):
            x = j * spacing
            y = i * spacing
            if i % 2 == 1:
                y += stagger
            centers[i * cols + j] = [x, y]

    radii_initial = np.full(n, 0.1)  # Initial guess for radii

    # Objective function: maximize sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints: non-overlapping circles and within bounds
    def constraint_bounds(radii, centers):
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if not (0 <= x - r and x + r <= 1 and 0 <= y - r and y + r <= 1):
                return np.inf
        return 0

    def constraint_overlap(radii, centers):
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

    # Define bounds for radii (non-negative)
    bounds = Bounds(0, np.inf)

    # Define constraints
    nonlinear_constraints = [
        NonlinearConstraint(constraint_bounds, 0, 0, jac=None),
        NonlinearConstraint(constraint_overlap, 0, 0, jac=None)
    ]

    # Initial guess
    initial_guess = radii_initial

    # Run optimization
    result = minimize(
        fun=objective,
        x0=initial_guess,
        method='SLSQP',
        bounds=bounds,
        constraints=nonlinear_constraints,
        args=(centers,)
    )

    # Extract the optimized results
    centers_optimal = centers

        return np.zeros((n, 2)), np.zeros(n), 0.0

    return centers_optimal, radii_optimal, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
