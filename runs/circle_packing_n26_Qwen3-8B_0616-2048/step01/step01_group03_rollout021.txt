from scipy.optimize import minimize
import numpy.typing as npt

def run_packing() -> tuple[npt.NDArray[np.float64], npt.NDArray[np.float64], float]:
    # Initial layout: approximative hexagonal grid in the unit square
    
    # Number of circles
    n = 26

    # Create initial hexagonal grid layout
    def hexagonal_grid(n, side_length, margin):
        grid = []
        for row in range(int(np.sqrt(n)) + 1):
            for col in range(int(np.sqrt(n)) + 1):
                if row * (int(np.sqrt(n)) + 1) + col >= n:
                    break
                x = col * (1 + side_length) + margin
                y = row * (np.sqrt(3) * side_length) + margin
                grid.append((x, y))
        return np.array(grid)

    # Initial grid with some spacing
    initial_centers = hexagonal_grid(n, 0.25, 0.05)
    initial_radii = np.full(n, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        # x is a vector of [x1, y1, r1, x2, y2, r2, ...]
        centers = x[:2 * n].reshape(n, 2)
        radii = x[2 * n:]
        
        # Check if any radius is negative
        if np.any(radii < 0):
            return -np.inf

        # Check if any circle is outside the square
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if (x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or
                y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12):
                return -np.inf

        # Check for circle-circle collisions
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -np.inf

        # Return the negative of the sum of radii to minimize
        return -np.sum(radii)

    # Initial guess
    x0 = np.concatenate([initial_centers.flatten(), initial_radii])

    # Bounds: centers must be within [0,1]^2, radii must be non-negative
    bounds = []

    result = minimize(

    centers = result.x[:2 * n].reshape(n, 2)
