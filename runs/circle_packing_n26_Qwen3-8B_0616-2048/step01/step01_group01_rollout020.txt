import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Number of circles
    n = 26
    
    # Initial grid-like arrangement
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Initial radius
    r0 = 0.05
    
    # Initialize centers in a grid-like hexagonal pattern
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col + 0.5 * (row % 2)
        y = row + 0.5
        centers[i] = [x / cols, y / rows]  # Scale to unit square
    
    # Initial radii
    radii = np.full(n, r0)
    
    # Objective: maximize the sum of radii (minimize -sum(radii))
    def objective(x):
        # x[0:(n*2)] are center coordinates (x0, y0, x1, y1, ...)
        # x[(n*2):] are radii
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        return -np.sum(radii_flat)
    
    # Constraints
    # 1. Radii must be non-negative
    def constraint_radii(x):
        return x[n*2:]
    
    # 2. Circle centers must stay within [0,1]x[0,1]
    def constraint_centers(x):
        centers_flat = x[:n*2].reshape(n, 2)
        return np.concatenate([centers_flat[:, 0] - 1, 1 - centers_flat[:, 0], 
                              centers_flat[:, 1] - 1, 1 - centers_flat[:, 1]])
    
    # 3. Non-overlapping constraint
    def constraint_non_overlap(x):
        centers_flat = x[:n*2].reshape(n, 2)
        radii_flat = x[n*2:].reshape(n)
        dists = np.sqrt(((centers_flat[:, np.newaxis, :] - centers_flat[np.newaxis, :, :])**2).sum(axis=2))
        return np.concatenate([dists[i, j] - radii_flat[i] - radii_flat[j] for i in range(n) for j in range(i+1, n)])
    
    # Define bounds
    bounds = []
    for i in range(n):
        bounds.extend([(0.0, 1.0), (0.0, 1.0)])  # x and y coordinates
    bounds.extend([(0.0, 1.0) for _ in range(n)])  # radii
    
    # Initial guess
    x0 = np.concatenate([centers.flatten(), radii])
    
    # Constraints
    cons = [
        {'type': 'ineq', 'fun': constraint_radii},
        {'type': 'ineq', 'fun': constraint_centers},
        {'type': 'ineq', 'fun': constraint_non_overlap}
    ]
    
    # Optimization
    result = optimize.minimize(
        fun=objective,
        x0=x0,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-6,
        maxiter=1000,
        options={'disp': False}
    )
    
    centers_opt = result.x[:n*2].reshape(n, 2)
    radii_opt = result.x[n*2:].reshape(n)
    
    # Validate the result
    valid, message = validate_packing(centers_opt, radii_opt)
    if not valid:
        raise ValueError(f"Packing invalid: {message}")
    
    sum_radii = np.sum(radii_opt)
    
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
