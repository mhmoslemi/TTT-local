import scipy.optimize as opt

def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():

    # Generate an initial hexagonal grid pattern
    centers = np.zeros((num_circles, 2))
    idx = 0
    row_count = 5
    col_count = 5
    spacing = 0.5
    offset = spacing / 2
    for row in range(row_count):
        for col in range(col_count):
            if idx >= num_circles:
                break
            if row % 2 == 0:
                x = col * spacing + offset
            else:
                x = (col + 0.5) * spacing
            y = row * spacing
            centers[idx] = [x, y]
            idx += 1

    # Fill remaining positions with random points
    if idx < num_circles:
        remaining = num_circles - idx
        random_centers = np.random.rand(remaining, 2)
        random_centers = (random_centers - 0.5) * 0.5 + 0.5  # Scale to [0, 1]
        centers[idx:] = random_centers

    # Compute initial radius estimates based on minimum distance
    min_dist = np.inf
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < min_dist:

    # Set initial radii based on minimum distance and boundary constraints
    radii = np.full(num_circles, min_dist / 2.0)
    for i in range(num_circles):
