import numpy as np
import scipy.optimize as opt
import math

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"


def run_packing():
    num_circles = 26

    # Create a hexagonal grid arrangement in the unit square
    # Compute the number of rows and columns needed
    cols = math.ceil(math.sqrt(num_circles))
    rows = math.ceil(num_circles / cols)

    # Adjust spacing to make the grid fit within the square
    spacing_x = 1.0 / cols
    spacing_y = 1.0 / rows

    centers = np.zeros((num_circles, 2))
    idx = 0

    for col in range(cols):
        for row in range(rows):
            if idx >= num_circles:
                break
            if row % 2 == 0:
                x = col * spacing_x + 0.5 * spacing_x
            else:
                x = col * spacing_x + 0.5 * spacing_x + 0.5 * spacing_x
            y = row * spacing_y + 0.5 * spacing_y
            # Shift slightly to avoid being too near the boundary and allow for more radius
            x += np.random.uniform(-0.02, 0.02)
            y += np.random.uniform(-0.02, 0.02)
            centers[idx] = [x, y]
            idx += 1

    # Define objective function to maximize the sum of radii
    def objective(x):
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:].reshape(num_circles)

        # Check if configuration is valid
        is_valid, msg = validate_packing(centers_flat, radii_flat)
        if not is_valid:
            return float('inf')  # Penalize invalid configurations

        # Return negative sum for minimization
        return -np.sum(radii_flat)

    # Bounds for centers: [0,1] for x and y coordinates
    center_bounds = []
    for _ in range(num_circles):
        center_bounds.append((0.0, 1.0))
        center_bounds.append((0.0, 1.0))

    # Bounds for radii: [0, 0.5] (since the square is of size 1)
    radius_bounds = []
    for _ in range(num_circles):
        radius_bounds.append((0.0, 0.5))

    # Combine bounds
    bounds = center_bounds + radius_bounds

    # Initial guess: use the initial center positions and small radii
    x0 = np.concatenate((centers.flatten(), np.full(num_circles, 0.02)))

    # Use a combination of optimization methods for better convergence
    # SLSQP for initial exploration, then Nelder-Mead for fast refinement
    result = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        tol=1e-6,
        options={'maxiter': 1000, 'ftol': 1e-9, 'eps': 1e-9}

    # Refine with another round of Nelder-Mead for fine-tuning
    result = opt.minimize(
