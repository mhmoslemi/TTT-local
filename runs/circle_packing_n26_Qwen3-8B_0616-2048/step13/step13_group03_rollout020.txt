import numpy as np
import scipy.optimize as opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def run_packing():
    num_circles = 26

    # Initial hexagonal grid configuration
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))

    # Calculate spacing for rows and columns
    dx = 1.0 / cols
    dy = 1.0 / rows

    centers = np.zeros((num_circles, 2))
    idx = 0

    for i in range(rows):
        for j in range(cols):
            if idx >= num_circles:
                break
            x = j * dx + dx / 2
            y = i * dy + dy / 2
            if i % 2 == 1:
                x += dx / 2
            centers[idx] = [x, y]
            idx += 1

    # Add extra circles near the edges
    if idx < num_circles:
        extra_positions = []
        for i in range(num_circles - idx):
            if i % 2 == 0:
                x = np.random.uniform(0.1, 0.9)
                y = 0.5
            else:
                x = 0.5
                y = np.random.uniform(0.1, 0.9)
            extra_positions.append([x, y])
        centers[idx:] = extra_positions

    # Compute initial minimum distance between all pairs
    min_dist = np.inf
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < min_dist:

    def objective(x):

    bounds = center_bounds + radius_bounds

    # Initial guess
    x0 = np.concatenate((centers.flatten(), radii))

    # Step 1: Global search with COBYLA
    result_co = opt.minimize(
        objective,
        x0,
        method='COBYLA',
        bounds=bounds,
        tol=1e-8,
        options={'maxiter': 500, 'disp': False}
    )

    # Step 2: Use L-BFGS-B for local refinement
    result_lbgfs = opt.minimize(
