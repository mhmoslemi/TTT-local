import scipy.optimize as optimize
import cvxpy as cp
from typing import Tuple, List, Any

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n_circles = 26

    def validate_packing(centers, radii):
        n = centers.shape[0]

        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"

            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12

                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))

                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

    def objective_function(radii, centers):

        return -np.sum(radii)

        initial_centers = []

            centers = []

            np.random.seed(42 + _)  # Different seed for each configuration
            centers = np.array(centers[:n_circles])

        return initial_centers

    def optimize_config(centers):
        # Define variables
        radii = cp.Variable(n_circles)
        constraints = []

        # Constraint: circles must not overlap
        for i in range(n_circles):

        prob = cp.Problem(cp.Maximize(objective_function(radii, centers)), constraints)
