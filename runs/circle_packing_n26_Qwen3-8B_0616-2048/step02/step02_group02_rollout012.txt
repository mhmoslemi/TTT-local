import numpy as np
import scipy.optimize as optimize

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_hexagonal_packing(n_circles):
    # First, determine the number of rows and columns in the hexagonal grid
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))

    # Generate a hexagonal grid of centers
    centers = []
    for i in range(rows):
        for j in range(cols):
            # Hexagonal grid: alternate row offset
            x = j + 0.5 * i
            y = i * np.sqrt(3) / 2
            if i % 2 == 1:
                x += 0.5
            centers.append([x, y])

    centers = np.array(centers[:n_circles])
    radii = np.ones(n_circles) * 0.05

    # Scale to fit in the unit square
    if n_circles == 1:
        centers = np.array([[0.5, 0.5]])
        radii = np.array([0.25])
    else:
        # Scale x and y to fit in [0,1]
        x_min, x_max = centers[:, 0].min(), centers[:, 0].max()
        y_min, y_max = centers[:, 1].min(), centers[:, 1].max()
        scale_x = (1 - 1e-6) / (x_max - x_min)

        centers = np.array([[x * scale_x, y * scale_y] for x, y in centers])
