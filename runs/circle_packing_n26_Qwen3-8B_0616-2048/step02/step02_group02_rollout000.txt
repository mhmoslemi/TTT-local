import scipy.optimize as optimize
import matplotlib.pyplot as plt

def run_packing():

    # Adjusted to fit within the unit square

    rows = 5
    cols = 5
    offset_x = 0.5 / (cols + 1)  # Horizontal spacing
    offset_y = 0.5 / (rows + 1)  # Vertical spacing

    # Initial circle centers
    centers = np.zeros((26, 2))
    radii = np.zeros(26)

    # Place circles in a hexagonal close packing
    for i in range(26):
        row_idx = i // cols
        col_idx = i % cols

        if row_idx % 2 == 1:
            col_idx += 0.5  # Offset for staggered packing

        x = col_idx * offset_x + offset_x
        y = row_idx * offset_y + offset_y

        # Adjust for boundary (within unit square)
        # Scale to fit into [0,1]^2

        x = x / (cols + 1)
        y = y / (rows + 1)

        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = 0.03  # Initial small radius

    # Optimization problem: maximize sum of radii
    # Define cost function
    def objective(radii):
        return -np.sum(radii)  # Minimize negative of sum to maximize sum

    # Define constraints
    def constraint_overlap(radii):
