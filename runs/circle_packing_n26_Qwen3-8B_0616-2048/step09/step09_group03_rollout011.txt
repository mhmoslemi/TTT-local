import numpy as np
import scipy.optimize as opt
import math
import random

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"


def generate_initial_placement(num_circles):
    # Hexagonal packing in a grid
    grid_cols = math.ceil(math.sqrt(num_circles))
    grid_rows = math.ceil(num_circles / grid_cols)
    spacing = 1.0 / (grid_cols + 1)

    centers = np.zeros((num_circles, 2))
    idx = 0
    for row in range(grid_rows):
        for col in range(grid_cols):
            if idx >= num_circles:
                break
            x = col * spacing + spacing
            y = row * spacing + spacing
            if row % 2 == 1:
                x += spacing / 2
            centers[idx] = [x, y]
            idx += 1

    # Add extra circles for corners and edges
    corner_positions = [
        [0.1, 0.1], [0.1, 0.9], [0.9, 0.1], [0.9, 0.9],
        [0.1, 0.5], [0.5, 0.1], [0.9, 0.5], [0.5, 0.9]
    ]
    for pos in corner_positions:
        if idx < num_circles:
            centers[idx] = pos
            idx += 1

    # Fill in remaining gaps with random placement
    remaining = num_circles - idx
    for _ in range(remaining):
        x = np.random.uniform(0.1, 0.9)
        y = np.random.uniform(0.1, 0.9)
        centers[idx] = [x, y]
        idx += 1

    return centers
