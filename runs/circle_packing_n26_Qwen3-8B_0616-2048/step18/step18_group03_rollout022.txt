def _generate_hex_grid(n):
    # Generate a hexagonal grid with staggered rows
    row_count = 6
    col_count = 5
    centers = np.zeros((n, 2))
    idx = 0
    for row in range(row_count):
        for col in range(col_count):
            if idx >= n:
                break
            if row % 2 == 0:
                x = col * 0.2 + 0.1
            else:
                x = (col + 0.5) * 0.2
            y = row * 0.2 + 0.1
            centers[idx] = [x, y]
            idx += 1

    # Add additional circles if needed
    if idx < n:
        remaining = n - idx
        random_centers = np.random.rand(remaining, 2)
        random_centers = (random_centers - 0.5) * 0.5 + 0.5  # Scale to [0, 1]
        centers[idx:] = random_centers
    return centers
