import numpy as np
import scipy.optimize as opt
import math

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"


def run_packing():
    num_circles = 26

    # Generate a hexagonal grid that densely fills the unit square
    # We'll use a hexagonal grid with varying spacing to fit as many circles as possible

    # First, estimate how many rows and columns we can fit
    # Use a hexagonal packing pattern where x spacing is sqrt(3)/2 * spacing_y
    spacing_y = 1.0 / math.sqrt(2)  # This spacing allows for dense packing

    def place_hexagonal_grid(num_circles, spacing_y):
        grid = []
        row = 0
        col = 0
        while len(grid) < num_circles:
            # Alternate row offset for hexagonal pattern
            x_offset = 0.5 * spacing_y * math.sqrt(3) if row % 2 == 1 else 0.0
            x = col * spacing_y + x_offset
            y = row * spacing_y
            if x < 1.0 - 1e-6 and y < 1.0 - 1e-6:
                grid.append([x, y])
            col += 1
            if col * spacing_y > 1.0 - 1e-6:
                col = 0
                row += 1
        return np.array(grid)

    # Fill the grid with initial centers
    initial_centers = place_hexagonal_grid(num_circles, spacing_y)

    # Function to compute the negative of the objective (to maximize sum)
    def objective(x):
        # x contains both centers and radii
        # Split x into centers and radii
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:].reshape(num_circles)

        # Check if this configuration is valid
        is_valid, _ = validate_packing(centers_flat, radii_flat)
        if not is_valid:
            return float('inf')  # Penalize invalid configurations

        # Compute the sum of radii (we want to maximize this)
        return -np.sum(radii_flat)  # Negative for minimization

    # Bounds for centers: [0,1] for x and y coordinates
    center_bounds = []
    for _ in range(num_circles):
        center_bounds.append((0.0, 1.0))
        center_bounds.append((0.0, 1.0))

    # Bounds for radii: [0, 0.5] (since the square is of size 1)
    radius_bounds = []
    for _ in range(num_circles):
        radius_bounds.append((0.0, 0.5))

    # Combine bounds
    bounds = center_bounds + radius_bounds

    # Initial guess: use the initial center positions and small radii
    x0 = np.concatenate((initial_centers.flatten(), np.full(num_circles, 0.05)))

    # Run the optimization with SLSQP method for constrained optimization
    result = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        tol=1e-8,
        options={'maxiter': 1000, 'ftol': 1e-12}
    )

    # Extract the final centers and radii
    centers_final = result.x[:2 * num_circles].reshape(num_circles, 2)

    sum_radii = np.sum(radii_final)
