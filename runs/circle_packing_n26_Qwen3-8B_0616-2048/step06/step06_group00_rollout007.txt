import numpy as np
import scipy.optimize as opt
import math

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"


def run_packing():
    num_circles = 26

    # Create a hexagonal grid with a more flexible and efficient layout
    grid_size = int(math.ceil(math.sqrt(num_circles)))
    spacing = 1.0 / (grid_size + 1)
    centers = np.zeros((num_circles, 2))
    idx = 0

    # Hexagonal grid with offset for better packing
    for row in range(grid_size):
        for col in range(grid_size):
            if idx >= num_circles:
                break
            x = col * spacing + spacing * 0.5
            y = row * spacing + spacing * 0.5
            # For even rows, offset the x by spacing * 0.5 to stagger
            if row % 2 == 0:
                x += spacing * 0.5
            centers[idx] = [x, y]
            idx += 1

    # Add edge and corner clusters for better utilization of space
    edge_positions = [

            idx += 1

    radii = np.full(num_circles, 0.05)

    # Function to compute the negative of the objective (to maximize sum)
    def objective(x):

            return float('inf')  # Penalize invalid configurations

        # Compute the sum of radii (we want to maximize this)
        return -np.sum(radii_flat)  # Negative for minimization

    # Bounds for centers: [0,1] for x and y coordinates
    center_bounds = []
    for _ in range(num_circles):
        center_bounds.append((0.0, 1.0))

    x0 = np.concatenate((centers.flatten(), radii.flatten()))
