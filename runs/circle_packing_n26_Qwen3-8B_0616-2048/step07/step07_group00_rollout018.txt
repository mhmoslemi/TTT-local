import numpy as np
import scipy.optimize as opt
import math

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"


def run_packing():

    # Hybrid initial placement with some circles near corners and edges
    grid_cols = math.ceil(math.sqrt(num_circles))
    grid_rows = math.ceil(num_circles / grid_cols)
    spacing_x = 1.0 / grid_cols
    spacing_y = 1.0 / grid_rows

    centers = np.zeros((num_circles, 2))
    idx = 0

    # Fill the grid
    for row in range(grid_rows):
        for col in range(grid_cols):
            if idx >= num_circles:
                break
            x = col * spacing_x + 0.5 * spacing_x
            y = row * spacing_y + 0.5 * spacing_y
            centers[idx] = [x, y]

    # Add extra circles near corners and edges
    positions = [
        [0.1, 0.1], [0.1, 0.9], [0.9, 0.1], [0.9, 0.9],

    ]

            centers[idx] = pos
            idx += 1

    # Define objective function to maximize the sum of radii
    def objective(x):

    center_bounds = []
    for _ in range(num_circles):
        center_bounds.append((0.0, 1.0))

    # Use a combination of optimization methods for better convergence
    # First use Nelder-Mead for exploration, then SLSQP for refinement
