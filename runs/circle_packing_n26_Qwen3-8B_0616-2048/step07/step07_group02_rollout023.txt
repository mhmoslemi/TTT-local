import numpy as np
import scipy.optimize as opt
import math

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"


def run_packing():
    num_circles = 26

    # Generate initial centers using a hybrid hexagonal + cluster approach
    centers = np.zeros((num_circles, 2))
    idx = 0

    # Hexagonal grid: 6 rows of 4 circles (approximately)
    # Each row has 4 circles, spaced by 1.0 / (sqrt(3) * 2)
    spacing = 1.0 / (2 * math.sqrt(3))
    vertical_spacing = spacing * math.sqrt(3)

    # Add 4 circles per row in a hexagonal grid
    for row in range(4):
        for col in range(4):
            if idx >= num_circles:
                break
            x = col * spacing
            y = row * vertical_spacing
            centers[idx] = [x, y]
            idx += 1

    # Add edge and corner clusters for extra density
    # We'll place extra circles near the borders and in corners
    edge_positions = [
        [0.1, 0.1], [0.1, 0.9], [0.9, 0.1], [0.9, 0.9],
        [0.1, 0.5], [0.5, 0.1], [0.5, 0.9], [0.9, 0.5],
        [0.2, 0.2], [0.2, 0.8], [0.8, 0.2], [0.8, 0.8],
        [0.25, 0.25], [0.75, 0.25], [0.25, 0.75], [0.75, 0.75]
    ]
    for pos in edge_positions:
        if idx < num_circles:
            centers[idx] = pos
            idx += 1

    # Initial radii for all circles
    radii = np.full(num_circles, 0.05)

    # Function to compute the negative of the objective (to maximize sum)
    def objective(x):
        # x contains both centers and radii
        # Split x into centers and radii
        centers_flat = x[:2 * num_circles].reshape(num_circles, 2)
        radii_flat = x[2 * num_circles:].reshape(num_circles)

        # Check if this configuration is valid
        is_valid, _ = validate_packing(centers_flat, radii_flat)
        if not is_valid:
            return float('inf')  # Penalize invalid configurations

        # Compute the sum of radii (we want to maximize this)
        return -np.sum(radii_flat)  # Negative for minimization

    # Bounds for centers: [0,1] for x and y coordinates
    center_bounds = []
    for _ in range(num_circles):
        center_bounds.append((0.0, 1.0))
        center_bounds.append((0.0, 1.0))

    # Bounds for radii: [0, 0.5] (since the square is of size 1)
    radius_bounds = []
    for _ in range(num_circles):
        radius_bounds.append((0.0, 0.5))

    # Combine bounds
    bounds = center_bounds + radius_bounds

    # Initial guess: use the initial center positions and small radii
    x0 = np.concatenate((centers.flatten(), radii.flatten()))

    # Run the optimization with SLSQP method for constrained optimization
    result = opt.minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        tol=1e-8,
