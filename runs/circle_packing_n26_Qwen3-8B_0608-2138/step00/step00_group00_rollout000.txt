import numpy as np
from scipy.optimize import minimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing in a unit square
    num_circles = 26
    row_count = int(np.sqrt(num_circles))
    col_count = num_circles // row_count
    # Adjust for square packing
    # Start with hexagonal grid, clip if needed

    # Create an initial grid of circle centers
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.05  # Initial small radius

    # Hexagonal grid generation
    for i in range(row_count):
        for j in range(col_count):
            x = j + 0.5 * (i % 2)
            y = i * np.sqrt(3) / 2
            if x < 1 and y < 1 and x > 0 and y > 0:
                centers[i * col_count + j] = [x, y]
                radii[i * col_count + j] = 0.05

    # Optimization: Maximize sum of radii under constraints
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum

    def constraint_overlap(i, j, centers, radii):
        dist = np.linalg.norm(centers[i] - centers[j])
        return dist - (radii[i] + radii[j]) + 1e-12

    constraints = []
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            constraints.append({
                'type': 'ineq',
                'fun': lambda x, i=i, j=j, centers=centers, radii=radii: constraint_overlap(i, j, centers, x)
            })

    for i in range(num_circles):
        x, y = centers[i]
        r = radii[i]
        if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
            # Adjust radius to fit in square
            r = min(r, min(x, 1 - x, y, 1 - y))
            radii[i] = r

    # Optimization
    result = minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1) for _ in range(num_circles)]
    )

    # Apply the optimized radii to the centers

    sum_radii = np.sum(radii)

    # Final validation

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
