import scipy.optimize as opt

def validate_packing(centers, radii):

        return False, "NaN values present"

    return True, "ok"

def run_packing():
    # Initial guess: 5 rows and 5 columns of circles with some spacing
    n_circles = 26
    initial_centers = np.zeros((n_circles, 2))
    initial_radii = np.zeros(n_circles)

    for i in range(n_circles):

        col = i % 5
        x = col / 4.0

        initial_radii[i] = 0.05

    # Optimization
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

        return x

        return np.array([x[0] - x[5*i + 0] >= -1e-12,

                dy = x[2*i + 1] - x[2*j + 1]
