import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial parameters for hexagonal packing
    num_circles = 26
    grid_rows = 5
    grid_cols = 6
    row_spacing = 1.0 / (grid_rows + 1)
    col_spacing = 1.0 / (grid_cols + 1)
    
    # Generate initial hexagonal grid
    centers = np.zeros((num_circles, 2))
    for i in range(grid_rows):
        for j in range(grid_cols):
            x = j * col_spacing
            y = i * row_spacing
            centers[i * grid_cols + j] = [x, y]
    
    # Initial radii based on spacing between points
    initial_radius = 1.0 / (2 * grid_rows)
    radii = np.full(num_circles, initial_radius)
    
    # Define objective function
    def objective(radii, centers):
        return -np.sum(radii)  # Minimize negative sum to maximize sum
    
    # Define constraints
    def constraints(radii, centers):
        constraints = []
        
        # Center within bounds
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r: 1.0 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r: 1.0 - y - r})
        
        # No overlap between circles
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                def dist_constraint(radii):
                    return np.sqrt(np.sum((centers[i] - centers[j])**2)) - (radii[i] + radii[j])
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        
        return constraints
    
    # Optimize
    result = optimize.minimize(
        fun=lambda x: objective(x, centers),
        x0=radii,
        bounds=[(0, 1.0) for _ in range(num_circles)],
        constraints=constraints(radii, centers),
        method='SLSQP'
    )
    
    # Extract final centers and radii
    final_centers = centers
    final_radii = result.x
    final_sum_radii = np.sum(final_radii)
    
    # Validate
    is_valid, message = validate_packing(final_centers, final_radii)
    if not is_valid:
        raise Exception(f"Invalid packing: {message}")
    
    return final_centers, final_radii, final_sum_radii
