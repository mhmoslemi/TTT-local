import scipy.optimize as opt

            y_center = 0.5 + y - 0.5 * spacing * rows * np.sqrt(3) / 2
            centers[i * cols + j] = [x_center, y_center]

        centers[i] = [x, y]
        radii[i] = boundary_radius

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii

    # Define constraints
    def constraint_overlap(c):

        radii_c = c[26:]

                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_c[i] + radii_c[j] - 1e-8:
                    return -1
