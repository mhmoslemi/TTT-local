import numpy as np
import random
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Parameters
    n_circles = 26
    max_iterations = 5000
    initial_temperature = 100.0
    cooling_rate = 0.995
    tolerance = 1e-5
    max_radius = 0.5

    def random_centers(n):
        # Generate random centers within [0,1]^2
        return np.random.rand(n, 2)

    def random_radii(n, max_radius):
        # Generate random radii within [0, max_radius]
        return np.random.rand(n) * max_radius

    # Helper function to check distances
    def circle_distance(c1, c2):
        return np.sqrt(np.sum((c1 - c2) ** 2))

    # Initial configuration
    initial_centers = random_centers(n_circles)
    initial_radii = random_radii(n_circles, max_radius)
    initial_sum_radii = np.sum(initial_radii)

    # Function to check validity of packing
    def is_valid(centers, radii):
        for i in range(len(centers)):
            x, y = centers[i]
            r = radii[i]
            if x - r < -tolerance or x + r > 1 + tolerance or y - r < -tolerance or y + r > 1 + tolerance:
                return False
            for j in range(i + 1, len(centers)):
                dist = circle_distance(centers[i], centers[j])
                if dist < radii[i] + radii[j] - tolerance:
                    return False
        return True

    # Helper function to compute energy (negative of sum of radii)
    def energy(centers, radii):
        return -np.sum(radii)

    # Optimization using simulated annealing
    centers = initial_centers.copy()
    radii = initial_radii.copy()
    best_centers = centers.copy()
    best_radii = radii.copy()
    best_energy = energy(centers, radii)
    current_energy = best_energy
    temperature = initial_temperature

    for iteration in range(max_iterations):
        # Perturb a random circle
        idx = np.random.randint(n_circles)
        x, y = centers[idx]
        r = radii[idx]
        # Perturb position
        dx = np.random.uniform(-0.05, 0.05)
        dy = np.random.uniform(-0.05, 0.05)
        new_x = x + dx
        new_y = y + dy
        new_r = r + np.random.uniform(-0.01, 0.01)
        new_r = max(0.0, min(new_r, max_radius))

        # Check if perturbation is valid
        new_centers = centers.copy()
        new_radii = radii.copy()
        new_centers[idx] = [new_x, new_y]
        new_radii[idx] = new_r
        if is_valid(new_centers, new_radii):
            new_energy = energy(new_centers, new_radii)
            delta_energy = new_energy - current_energy
            if delta_energy < 0 or np.random.rand() < np.exp(-delta_energy / temperature):
                centers = new_centers
                radii = new_radii
                current_energy = new_energy
                if current_energy < best_energy:
                    best_centers = centers.copy()
                    best_radii = radii.copy()
                    best_energy = current_energy

        temperature *= cooling_rate

    # Validate final configuration
    assert is_valid(best_centers, best_radii), "Final configuration is invalid"

    sum_radii = np.sum(best_radii)
    return best_centers, best_radii, sum_radii
