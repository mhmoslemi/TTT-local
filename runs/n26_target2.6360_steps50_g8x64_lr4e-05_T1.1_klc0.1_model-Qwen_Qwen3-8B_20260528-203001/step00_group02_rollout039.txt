import numpy as np
import cvxpy as cp

def run_packing():
    # Define variables
    n = 26
    centers = cp.Variable((n, 2))
    radii = cp.Variable(n)

    # Optimization objective: maximize the sum of radii
    objective = cp.Maximize(cp.sum(radii))

    # Constraints
    constraints = []

    # Center coordinates must be within [0, 1] x [0, 1]
    for i in range(n):
        constraints.append(centers[i, 0] >= 0)
        constraints.append(centers[i, 0] <= 1)
        constraints.append(centers[i, 1] >= 0)
        constraints.append(centers[i, 1] <= 1)

    # Radii must be non-negative
    for i in range(n):
        constraints.append(radii[i] >= 0)

    # Ensure circles do not exceed the square's boundaries
    for i in range(n):
        constraints.append(centers[i, 0] - radii[i] >= -1e-8)
        constraints.append(centers[i, 0] + radii[i] <= 1 + 1e-8)
        constraints.append(centers[i, 1] - radii[i] >= -1e-8)
        constraints.append(centers[i, 1] + radii[i] <= 1 + 1e-8)

    # Non-overlapping circles
    for i in range(n):
        for j in range(i + 1, n):
            # Distance between centers
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = cp.sqrt(dx * dx + dy * dy)
            # Constraint: distance >= sum of radii
            constraints.append(dist >= radii[i] + radii[j] - 1e-8)

    # Solve the optimization problem
    prob = cp.Problem(objective, constraints)
    prob.solve()

    # Convert variables to numpy arrays
    centers_np = centers.value
    radii_np = radii.value

    # Check if the solution was successful
    if prob.status != "optimal":
        raise ValueError("Optimization did not converge.")

    if not is_valid:
        raise ValueError(f"Packing is invalid: {message}")

    sum_radii = np.sum(radii_np)
    return centers_np, radii_np, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
