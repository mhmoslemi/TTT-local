import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Radius initialization: All circles have the same initial radius
    initial_radius = 0.06

    max_ring = 5
    positions = []
    for ring in range(max_ring + 1):
        for angle in np.linspace(0, 2 * np.pi, 6, endpoint=False):
            x = 0.5 + (ring + 1) * np.cos(angle)
            y = 0.5 + (ring + 1) * np.sin(angle)
            positions.append([x, y])

    # Shuffle to get a randomized initial configuration
    np.random.shuffle(positions)

    # Truncate to 26 circles
    centers = np.array(positions[:26])
    radii = np.array([initial_radius] * 26)

    # Define the objective function to maximize the sum of radii
    def objective(x):
        # x is a flat array: [r0, r1, ..., r25]
        return -np.sum(x)  # To minimize the negative sum (maximize the sum)

    # Define the constraints
    def constraint_circle_in_square(x):
        # Ensure each circle is within the unit square
        for i in range(26):
            r = x[i]
            if (centers[i, 0] - r < -1e-12 or centers[i, 0] + r > 1 + 1e-12 or
                centers[i, 1] - r < -1e-12 or centers[i, 1] + r > 1 + 1e-12):
                return -1
        return 0

    def constraint_no_overlap(x):
        # Ensure no two circles overlap
        for i in range(26):
            for j in range(i + 1, 26):
                ri = x[i]
                rj = x[j]

                if dist < ri + rj - 1e-12:
                    return -1
        return 0

    # Bounds (all radii must be >= 0)
    bounds = [(0.0, 0.15) for _ in range(26)]
