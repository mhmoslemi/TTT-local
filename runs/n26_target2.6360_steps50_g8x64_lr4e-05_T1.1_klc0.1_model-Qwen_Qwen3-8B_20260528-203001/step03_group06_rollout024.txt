import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    rows, cols = 5, 6  # 5 rows, 6 columns
    grid = np.zeros((n, 2))
    for i in range(n):
        grid[i, 0] = i % cols / (cols - 1)
        grid[i, 1] = i // cols / (rows - 1)

    # Step 2: Define initial radii (uniform, close to maximum allowable)
    initial_radii = np.full(n, 0.05)

    def objective(x):
        # x is a vector of radii
        return -np.sum(x)  # minimize the negative sum to maximize the sum

    # Step 4: Variable bounds (radii must be non-negative)
    bounds = [(0.0, 0.1) for _ in range(n)]

    # Step 5: Define constraints
    def constraint_overlap(x, centers):
        # x is a vector of radii
        # centers is a vector of center positions
        # Check distance between each pair of circles
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i*2 + 0] - centers[j*2 + 0]
                dy = centers[i*2 + 1] - centers[j*2 + 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < x[i] + x[j] - 1e-12:
                    return -1  # violation of non-overlap constraint
        return 0  # no violation of non-overlap constraint

    def constraint_boundary(x, centers):

        # centers is a vector of center positions
        for i in range(n):

            y_pos = centers[i*2 + 1]

            if x_pos - r < -1e-12 or x_pos + r > 1 + 1e-12 or y_pos - r < -1e-12 or y_pos + r > 1 + 1e-12:

    res = optimize.minimize(
