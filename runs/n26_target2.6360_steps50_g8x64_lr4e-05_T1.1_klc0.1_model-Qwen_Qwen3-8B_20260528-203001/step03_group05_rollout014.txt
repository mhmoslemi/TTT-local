import scipy.optimize as optimize
import math

def run_packing():

    cols = 5
    spacing = 0.25

            x = j * spacing + spacing / 2
            y = i * spacing * math.sqrt(3) / 2

            centers[idx] = [x, y]

            idx += 1

        radii_new = x[num_circles * 2:]
