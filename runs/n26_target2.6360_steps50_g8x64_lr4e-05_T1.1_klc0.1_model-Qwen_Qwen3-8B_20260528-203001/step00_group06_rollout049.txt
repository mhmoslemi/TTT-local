import numpy as np
import cvxpy as cp

def run_packing():
    # Step 1: Hexagonal grid of 26 circles
    # Determine the number of rows and columns in the grid
    # 26 = 5 rows (each row has 5, 6, 5, 6, 5 circles)
    num_rows = 5
    num_cols_per_row = [5, 6, 5, 6, 5]
    row_height = 1.0 / num_rows  # vertical spacing
    col_width = 1.0 / max(num_cols_per_row)  # horizontal spacing

    centers = []
    radii = []

    # Place circles in a hexagonal grid
    for row_idx in range(num_rows):
        row_offset = row_idx * row_height
        num_cols = num_cols_per_row[row_idx]
        for col_idx in range(num_cols):
            col_offset = col_idx * col_width if row_idx % 2 == 0 else (col_idx + 0.5) * col_width
            x = col_offset
            y = row_offset
            # Use the horizontal spacing as the base radius
            r = col_width / 2.0
            centers.append([x, y])
            radii.append(r)

    # Convert to numpy arrays
    centers = np.array(centers)
    radii = np.array(radii)

    # Step 2: Set up optimization problem
    # Objective: Maximize the sum of radii
    # Constraints: Non-overlapping and boundary conditions

    n = len(centers)
    x = cp.Variable((n, 2))
    r = cp.Variable(n)

    # Objective function
    obj = cp.Maximize(cp.sum(r))

    # Constraints: boundary conditions
    constraints = []
    for i in range(n):
        x_i, y_i = x[i]
        r_i = r[i]
        constraints.append(x_i - r_i >= 0)
        constraints.append(x_i + r_i <= 1)
        constraints.append(y_i - r_i >= 0)
        constraints.append(y_i + r_i <= 1)

    # Constraints: non-overlapping
    for i in range(n):
        for j in range(i + 1, n):
            dx = x[i][0] - x[j][0]
            dy = x[i][1] - x[j][1]
            dist = cp.sqrt(dx**2 + dy**2)
            constraints.append(dist >= r[i] + r[j])

    # Solve the optimization problem
    prob = cp.Problem(obj, constraints)
    result = prob.solve(solver=cp.SLSQP, eps=1e-6)

    if result is None:
        return centers, radii, np.sum(radii)

    # Extract optimized centers and radii
    centers = np.vstack([x.value[:, 0], x.value[:, 1]]).T
    radii = r.value

    # Validate the packing
    valid, msg = validate_packing(centers, radii)
    if not valid:
        print("Validation failed:", msg)
        return centers, radii, np.sum(radii)

    # Return the result
    return centers, radii, np.sum(radii)


def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
