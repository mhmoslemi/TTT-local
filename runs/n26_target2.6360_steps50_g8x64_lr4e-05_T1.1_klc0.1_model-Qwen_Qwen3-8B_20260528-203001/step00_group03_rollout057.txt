import scipy.optimize as opt
import math

def run_packing():

    num_circles = 26
    max_radius_guess = target_sum / num_circles

    # We'll use a hexagonal packing with 5 rows (rows 0 to 4)
    # The vertical spacing between rows is sqrt(3)/2 * radius
    # The horizontal spacing between centers is 2 * radius

    def initial_guess(radius):

        row_width = 2 * radius
        centers = []
        for row in range(5):  # 5 rows
            for col in range(6):  # 6 columns
                col_offset = col * row_width
                row_offset = row * row_height
                x = col_offset + 0.5 * row_width * (row % 2)
                y = row_offset + 0.5 * row_height
                centers.append((x, y))
        # Normalize to fit in [0, 1] square
        centers = np.array(centers)
        centers = (centers - centers.min(axis=0)) / (centers.max(axis=0) - centers.min(axis=0))
        return centers, np.full(len(centers), radius)

    # Optimization
    def objective(radii):
        return -np.sum(radii)

    def constraint_boundary(centers, radii):
        return np.all(centers.min(axis=0) + radii >= 1e-12) and np.all(centers.max(axis=0) - radii <= 1 - 1e-12)

    def constraint_overlap(centers, radii):

            for j in range(i + 1, num):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
