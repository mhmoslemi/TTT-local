import scipy.optimize as opt
import random

def run_packing():

    def randomized_initial_layout(num_circles):

        centers = np.zeros((num_circles, 2))
        radii = np.zeros(num_circles)

            x = np.random.uniform(0.1, 0.9)
            y = np.random.uniform(0.1, 0.9)

            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            radii[i] = dist_to_boundary

            r = radii[i]
            dist_to_left = x - r
            dist_to_right = 1 - x - r
            dist_to_bottom = y - r
            dist_to_top = 1 - y - r
            if dist_to_left < -1e-6 or dist_to_right < -1e-6 or dist_to_bottom < -1e-6 or dist_to_top < -1e-6:

                dist_to_boundary = min(x - r, 1 - x - r, y - r, 1 - y - r)

                    radii[i] = r_new
        return centers, radii

            return False, "NaN values present"

        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:

        dist = np.sqrt(dx**2 + dy**2)
