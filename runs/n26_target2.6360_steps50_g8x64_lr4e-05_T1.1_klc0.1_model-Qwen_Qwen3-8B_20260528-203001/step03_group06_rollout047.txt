import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based placement of 26 circles
    # We will pack 5 rows of 6 circles and 1 row of 4 circles
    # Grid spacing will be slightly less than 1/2 to allow refinement
    
    # Create center points in a grid-like pattern
    grid_width = 1
    grid_height = 1
    cols = 6
    rows = 5
    spacing_x = grid_width / (cols - 1)
    spacing_y = grid_height / (rows - 1)

    # Generate centers
    centers = np.zeros((26, 2))
    index = 0
    for y in range(rows):
        for x in range(cols):
            if index < 26:
                centers[index, 0] = x * spacing_x
                centers[index, 1] = y * spacing_y
                index += 1
            else:
                break

    # Initial radii (same for all)
    radii = np.full(26, 0.1)

    # Objective function: maximize sum of radii
    def objective(x):
        # x is a flattened array of all variables (centers and radii)
        x_flat = x.reshape(-1)
        centers = x_flat[:26 * 2].reshape(26, 2)
        radii = x_flat[26 * 2:].reshape(26)
        return -np.sum(radii)  # Negative for minimization

    # Constraints
    def constraint_non_overlap(x):
        x_flat = x.reshape(-1)
        centers = x_flat[:26 * 2].reshape(26, 2)
        radii = x_flat[26 * 2:].reshape(26)
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    return dist - (radii[i] + radii[j] - 1e-8)
        return 0

    def constraint_boundary(x):
        x_flat = x.reshape(-1)
        centers = x_flat[:26 * 2].reshape(26, 2)
        radii = x_flat[26 * 2:].reshape(26)
        for i in range(26):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-8 or x + r > 1 + 1e-8 or y - r < -1e-8 or y + r > 1 + 1e-8:
                return x - r + 1e-8 if x - r < -1e-8 else (x + r - 1 - 1e-8 if x + r > 1 + 1e-8 else (y - r + 1e-8 if y - r < -1e-8 else (y + r - 1 - 1e-8)))
        return 0

    # Bounds for x and y coordinates: each circle's center must lie in [0, 1]
    # Bounds for radii: each radius must be >= 0
    bounds = []
    for _ in range(26):
        bounds.extend([(0, 1), (0, 1)])  # x, y for each circle
    for _ in range(26):
        bounds.extend([(0, 0.3)])  # radii (assuming initial guess is 0.1)

    # Initial guess
    initial_guess = np.zeros(26 * 2 + 26)
    initial_guess[:26 * 2] = centers.flatten()
    initial_guess[26 * 2:] = radii

    # Run optimization
    result = optimize.minimize(
        objective,
        initial_guess,
        method='SLSQP',
        constraints=[
            {'type': 'ineq', 'fun': constraint_non_overlap},
            {'type': 'ineq', 'fun': constraint_boundary}
        ],
        bounds=bounds,
        options={'ftol': 1e-8, 'maxiter': 1000}
    )

    radii_opt = result.x[26 * 2:].reshape(26)
    sum_radii = np.sum(radii_opt)

    # Validate the packing
    valid, message = validate_packing(centers_opt, radii_opt)
    if not valid:
        raise ValueError(message)

    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
