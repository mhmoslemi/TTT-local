import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_circles = 26
    x = np.linspace(0, 1, 6)
    y = np.linspace(0, 1, 6)
    grid = np.array([(xi, yi) for xi in x for yi in y])
    centers = grid[:num_circles]
    radii = np.full(num_circles, 0.05)  # Initial guess for radii

    # Define the cost function to maximize the sum of radii
    def objective_function(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        radii_flat = x[num_circles * 2:]
        return -np.sum(radii_flat)  # Objective is to maximize, so we return -sum

    # Constraint 1: Centers inside unit square
    def center_inside_box(x):
        centers_flat = x[:num_circles * 2].reshape(num_circles, 2)
        return np.abs(centers_flat[:, 0] - 0.5) < 0.5 + 1e-6  # Simplified to ensure within [0,1]
    center_box_con = opt.LinearConstraint(np.eye(num_circles * 2), 0, 1)

    # Constraint 2: Radii non-negative

        radii_flat = x[num_circles * 2:]
        return radii_flat >= 0
    radii_non_neg_con = opt.LinearConstraint(np.eye(num_circles), 0, np.inf)

    # Constraint 3: Non-overlapping between circles

        radii_flat = x[num_circles * 2:]

                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii_flat[i] + radii_flat[j] - 1e-6:
                    return -np.inf  # Penalize overlap
        return 0

    bounds = []

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
