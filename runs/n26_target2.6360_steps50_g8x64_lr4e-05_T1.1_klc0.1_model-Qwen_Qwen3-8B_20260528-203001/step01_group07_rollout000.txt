import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26
    cols = np.ceil(np.sqrt(n)).astype(int)
    rows = np.ceil(n / cols).astype(int)

    centers = np.zeros((n, 2))
    for i in range(n):
        x = i % cols
        y = i // cols
        centers[i, 0] = x / (cols - 1)  # Normalize to [0, 1]
        centers[i, 1] = y / (rows - 1)

    radii = np.ones(n) * 0.05

    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    def constraint_overlap(radii, centers):
        n = len(radii)

                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-8:
                    return -1
        return 0

                return -1
        return 0

def validate_packing(centers, radii):
