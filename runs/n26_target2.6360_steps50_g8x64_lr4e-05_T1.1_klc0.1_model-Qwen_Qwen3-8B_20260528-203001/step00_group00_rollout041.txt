import numpy as np
import scipy.optimize as opt
import matplotlib.pyplot as plt

def run_packing():
    # Number of circles
    n = 26
    
    # Initial grid-based placement on a staggered grid (hexagonal close packing)
    cols = 6  # 6 columns
    rows = 5  # 5 rows (including some offset)
    
    # Generate staggered grid positions
    centers = np.zeros((n, 2))
    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:
                x = j * 0.45
            else:
                x = (j + 0.5) * 0.45
            y = i * 0.45
            if i == 0 and j == 0:
                centers[0] = [x, y]
            else:
                centers[i * cols + j] = [x, y]
    
    # Initial radius estimate based on spacing
    spacing = 0.45
    initial_radius = spacing / 2.0 - 0.01  # slightly smaller to avoid overlapping
    
    # Radius vector
    radii = np.full(n, initial_radius)
    
    # Define the optimization objective
    def objective(radii):
        # Return the negative of the sum of radii (to maximize)
        return -np.sum(radii)
    
    # Define the constraints
    def constraints(radii, centers):
        constraints = []
        
        # Constraint 1: All circles must be within the unit square
        def within_bounds(radii):
            for i in range(n):
                x, y = centers[i]
                r = radii[i]
                if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                    return False
            return True
        
        constraints.append({"type": "ineq", "fun": lambda r: within_bounds(r)})
        
        # Constraint 2: No circles can overlap
        def no_overlap(radii):
            for i in range(n):
                for j in range(i + 1, n):
                    dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                    if dist < radii[i] + radii[j] - 1e-12:
                        return False
            return True
        
        constraints.append({"type": "ineq", "fun": lambda r: no_overlap(r)})
        
        return constraints
    
    # Optimize the radii
    result = opt.minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=constraints(radii, centers),
        bounds=[(0, 1) for _ in range(n)]
    )
    
    if not result.success:

        return (centers, radii, np.sum(radii))
