from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform
from typing import Tuple, List

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based arrangement
    rows = 5
    cols = 5
    gaps = 0.15  # Small gap to allow for optimization

    centers = np.zeros((rows * cols, 2))
    radii = np.zeros(rows * cols)

    for i in range(rows):
        for j in range(cols):
            x = j * (1 + gaps) / (cols - 1)
            y = i * (1 + gaps) / (rows - 1)
            centers[i * cols + j] = [x, y]
            radii[i * cols + j] = 0.05  # Initial small radius

    # Convert to array format
    centers = np.array(centers)
    radii = np.array(radii)

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum = maximize sum

    def constraint_circle_in_square(radii):
        def cons(x):
            for i in range(len(x)):
                if x[i] < 0:
                    return np.inf
            return 0
        return {'type': 'ineq', 'fun': cons}

    def constraint_circle_not_outside_square(radii):
        def cons(x):
            for i in range(len(x)):
                center = centers[i]
                if center[0] - x[i] < -1e-12 or center[0] + x[i] > 1 + 1e-12:
                    return np.inf
                if center[1] - x[i] < -1e-12 or center[1] + x[i] > 1 + 1e-12:
                    return np.inf
            return 0
        return {'type': 'ineq', 'fun': cons}

    def constraint_circles_not_overlapping(radii):

            dist = squareform(pdist(centers))

            return 0
        return {'type': 'ineq', 'fun': cons}

    # Initial guess for radii
    radii_guess = radii.copy()

    bounds = [(0, 0.1) for _ in range(len(radii_guess))]  # Reasonable upper bound for radius

    constraints = [

        method='SLSQP',
        bounds=bounds,
