import numpy as np
from scipy.optimize import minimize

def run_packing():
    # Problem size
    n = 26
    np.random.seed(0)  # For reproducibility

    # Step 1: Generate initial random positions within the unit square
    centers = np.random.rand(n, 2)  # Initial guess for circle centers
    radii = np.full(n, 0.05)  # Initial guess for radii

    # Step 2: Objective function to maximize sum of radii
    def objective(radii_flat):
        return -np.sum(radii_flat)  # Negative for minimization

    # Step 3: Constraint: each circle lies within the square [0,1]^2
    def square_constraint(radii_flat):
        r = radii_flat.reshape(n)
        return np.array([
            centers[i, 0] - r[i] >= -1e-8 for i in range(n)
        ]) + np.array([
            centers[i, 0] + r[i] <= 1 + 1e-8 for i in range(n)
        ]) + np.array([
            centers[i, 1] - r[i] >= -1e-8 for i in range(n)
        ]) + np.array([
            centers[i, 1] + r[i] <= 1 + 1e-8 for i in range(n)
        ])

    # Step 4: Constraint: no overlapping between circles
    def no_overlap(radii_flat):
        r = radii_flat.reshape(n)
        dist = np.zeros(n * (n - 1) // 2)
        idx = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist[idx] = np.sqrt(dx ** 2 + dy ** 2)
                idx += 1
        return dist - (r[:n] + r[n:]) + 1e-8

    # Step 5: Bounds for radii (must be non-negative)
    bounds = [(0, 1) for _ in range(n)]

    # Step 6: Run the optimization
    result = minimize(

        radii.flatten(),
        method="SLSQP",
        constraints=[
            {"type": "ineq", "fun": square_constraint},
            {"type": "ineq", "fun": no_overlap}
        ],
        bounds=bounds
    )

    # Step 7: Extract the final solution
    final_radii = result.x
    final_centers = centers.copy()
    sum_radii = np.sum(final_radii)

    # Step 8: Return the result
    return final_centers, final_radii, sum_radii
