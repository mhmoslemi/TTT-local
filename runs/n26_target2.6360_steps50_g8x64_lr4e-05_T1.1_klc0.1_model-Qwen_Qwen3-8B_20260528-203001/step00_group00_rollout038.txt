import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Hexagonal grid parameters
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    
    # Hexagonal grid (with spacing adjusted to fit in unit square)
    spacing_row = 0.9 / (rows - 1)
    spacing_col = 0.9 / (cols - 1)
    
    # Generate initial hexagonal grid
    centers = np.zeros((num_circles, 2))
    idx = 0
    for row in range(rows):
        for col in range(cols):
            x = col * spacing_col
            y = row * spacing_row + (col % 2) * spacing_row / 2
            centers[idx] = [x, y]
            idx += 1
    
    # Initial guess for radii
    radii = np.ones(num_circles) * 0.05
    
    # Objective function (maximize sum of radii)
    def objective(x):
        return -np.sum(x[26:])  # Minimize negative sum
    
    # Constraints
    def constraint_centers(x):
        centers_guess = x[:26].reshape(-1, 2)
        return np.array([np.max(centers_guess[:, 0]) - 1.0, 1.0 - np.min(centers_guess[:, 0]),
                         np.max(centers_guess[:, 1]) - 1.0, 1.0 - np.min(centers_guess[:, 1])])
    
    def constraint_radii(x):
        return x[26:]
    
    def constraint_overlap(x):
        centers_guess = x[:26].reshape(-1, 2)
        radii_guess = x[26:]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_guess[i, 0] - centers_guess[j, 0]
                dy = centers_guess[i, 1] - centers_guess[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii_guess[i] + radii_guess[j] - 1e-8:
                    return -1.0
        return 0.0
    
    # Setup optimization
    x0 = np.concatenate((centers.reshape(-1), radii))
    bounds = [(-1.0, 1.0) for _ in range(2 * num_circles)] + [(0.0, 1.0) for _ in range(num_circles)]
    
    constraints = [
        {'type': 'ineq', 'fun': constraint_centers},
        {'type': 'ineq', 'fun': constraint_radii},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]
    
    # Optimization
    result = optimize.minimize(objective, x0, bounds=bounds, constraints=constraints, method='SLSQP')
    
    # Extract results
    centers_opt = result.x[:26].reshape(-1, 2)
    radii_opt = result.x[26:]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
