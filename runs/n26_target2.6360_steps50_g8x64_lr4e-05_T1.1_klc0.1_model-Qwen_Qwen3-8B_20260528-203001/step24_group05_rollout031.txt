import numpy as np
import scipy.optimize as opt
import random

def run_packing():
    num_circles = 26
    best_sum = 0.0
    best_centers = np.zeros((num_circles, 2))
    best_radii = np.zeros(num_circles)

    def refine_layout(num_circles):
        max_radius = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.15
        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        
        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break
                x = j * (2 * max_radius) + max_radius
                y = i * (np.sqrt(3) * max_radius) + max_radius
                if i % 2 == 1:
                    x += max_radius
                centers[idx] = [x, y]
                idx += 1
        
        # Add corner circles
        corner_positions = [
            (0.1, 0.1), (0.1, 0.9), (0.9, 0.1), (0.9, 0.9)
        ]
        for pos in corner_positions:
            if idx < num_circles:
                centers[idx] = pos
                idx += 1
        
        # Add boundary circles
        boundary_positions = [
            (0.1, 0.5), (0.9, 0.5), (0.5, 0.1), (0.5, 0.9)
        ]
        for pos in boundary_positions:
            if idx < num_circles:
                centers[idx] = pos
                idx += 1
        
        # Adjust circles to be closer to boundaries and corners
        for i in range(num_circles):
            x, y = centers[i]
            dist_to_left = x
            dist_to_right = 1 - x
            dist_to_bottom = y
            dist_to_top = 1 - y
            min_dist = min(dist_to_left, dist_to_right, dist_to_bottom, dist_to_top)
            if min_dist < 1e-6:
                if dist_to_left < 1e-6:
                    x = 0.0
                elif dist_to_right < 1e-6:
                    x = 1.0
                elif dist_to_bottom < 1e-6:
                    y = 0.0
                elif dist_to_top < 1e-6:
                    y = 1.0
                centers[i] = [x, y]
        
        # Assign initial radii based on proximity to boundaries
        initial_radii = np.full(num_circles, max_radius)
        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            if dist_to_boundary < max_radius:
                initial_radii[i] = dist_to_boundary
        
        return centers, initial_radii
    
    def post_refinement(centers, radii):
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            dist_to_left = x - r
            dist_to_right = 1 - x - r
            dist_to_bottom = y - r
            dist_to_top = 1 - y - r
            if dist_to_left < -1e-6 or dist_to_right < -1e-6 or dist_to_bottom < -1e-6 or dist_to_top < -1e-6:
                dist_to_boundary = min(x - r, 1 - x - r, y - r, 1 - y - r)
                r_new = r + dist_to_boundary
                if r_new > r:
                    radii[i] = r_new
        return centers, radii
    
    def objective(params):
        flat_params = params.reshape(-1)
        centers_opt = flat_params[:num_circles * 2].reshape((num_circles, 2))
        radii_opt = flat_params[num_circles * 2:].reshape((num_circles,))
        return -np.sum(radii_opt)  # Minimization of negative sum
    
    def validate_packing(centers, radii):
        n = centers.shape[0]
        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"
        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"
        for i in range(n):
            x, y = centers[i]
            r = radii[i]

                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"
        return True, "ok"
    
    for _ in range(10):
        centers, radii = refine_layout(num_circles)
        centers, radii = post_refinement(centers, radii)
        
        bounds = []
        for _ in range(num_circles):
            bounds.append((0.0, 1.0))  # Radius
            bounds.append((0.0, 1.0))  # x
            bounds.append((0.0, 1.0))  # y

        constraints = []

                circle_data = x[var_indices]
                x_val, y_val, r_val = circle_data
                if x_val - r_val < -1e-12 or x_val + r_val > 1 + 1e-12 or y_val - r_val < -1e-12 or y_val + r_val > 1 + 1e-12:

                return 0.0

                    dx = x1 - x2
                    dy = y1 - y2
                    dist = np.sqrt(dx**2 + dy**2)

                        return -1.0
                    return 0.0
                
                constraints.append({'type': 'ineq', 'fun': constraint_overlap_func})
        
        result = opt.minimize(
            objective,
            initial_guess,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            tol=1e-10,
            options={'maxiter': 2000, 'ftol': 1e-10, 'eps': 1e-8}
        )
        
        optimized = result.x

        if not is_valid:
            print(f"Validation failed: {message}")
            centers_opt = centers
            radii_opt = radii
            is_valid, message = validate_packing(centers_opt, radii_opt)
            if not is_valid:
                print(f"Initial layout also failed: {message}")
                continue
        
        sum_radii = np.sum(radii_opt)
        if sum_radii > best_sum:
            best_sum = sum_radii
            best_centers = centers_opt
            best_radii = radii_opt
    
    return best_centers, best_radii, best_sum
