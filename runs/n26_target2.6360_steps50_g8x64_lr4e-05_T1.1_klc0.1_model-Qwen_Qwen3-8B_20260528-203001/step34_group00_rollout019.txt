import numpy as np
import scipy.optimize as opt
import math
import random

def run_packing():

        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))

        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break

                else:
                    x = j * (2 * max_radius_initial) + max_radius_initial + max_radius_initial
                y = i * (np.sqrt(3) * max_radius_initial) + max_radius_initial

                idx += 1

        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)

            return False, "NaN values present"

        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or

                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))

            return centers, radii
