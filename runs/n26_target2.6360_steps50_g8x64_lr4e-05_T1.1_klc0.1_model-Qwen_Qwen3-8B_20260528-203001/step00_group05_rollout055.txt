import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Define the number of circles
    n_circles = 26
    
    # Initialize circle centers in a hexagonal grid pattern
    # Parameters for hexagonal grid
    hex_row = int(np.ceil(np.sqrt(n_circles)))
    hex_col = int(np.ceil(n_circles / hex_row))
    
    # Radius of circles (initialize to a small value)
    initial_radii = np.full(n_circles, 0.02)
    
    # Place circles in a hexagonal grid within the unit square
    centers = np.zeros((n_circles, 2))
    for i in range(n_circles):
        row = i // hex_col
        col = i % hex_col
        x = col + 0.5 * row
        y = row * np.sqrt(3) / 2
        # Scale to fit within the unit square
        x = x / max(1, hex_col + 1)
        y = y / max(1, hex_row)
        centers[i] = np.array([x, y])
    
    # Optimize the radii and positions to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)
    
    def constraints(radii):
        # Check if all circles are within the square
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return np.inf
        # Check for overlaps between circles

            for j in range(i + 1, n_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0
    
    result = optimize.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=[(0, 1)] * n_circles,
        constraints=[{'type': 'ineq', 'fun': constraints}]
    )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
