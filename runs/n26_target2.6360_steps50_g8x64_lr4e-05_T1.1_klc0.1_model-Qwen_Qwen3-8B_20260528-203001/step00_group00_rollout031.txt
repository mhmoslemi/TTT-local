from scipy.optimize import minimize
import math

    # Hexagonal packing parameters
    radius_initial = 0.1  # Initial radius for hexagonal arrangement

    cols = int(math.ceil(n_circles / rows))

    while rows * cols < n_circles:
        rows += 1
        cols = int(math.ceil(n_circles / rows))
