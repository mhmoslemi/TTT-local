import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26

    # 5 rows by 5 columns = 25 circles, with one extra circle added near the center
    grid_points = np.array([
        [i / 4, j / 4] for i in range(5) for j in range(5)
    ])
    centers = np.vstack((grid_points, [0.5, 0.5]))  # add an extra circle in the center
    radii = np.full(n, 0.05)  # initial radii (can be optimized)
    
    # Define the objective function (maximize total radii)
    def objective(x):
        # x is a vector of all radii and positions
        # Reshape x
        x = x.reshape((n, 4))  # [x1, y1, r1, x2, y2, r2, ...]
        centers = x[:, :2]
        radii = x[:, 2]
        
        # Check if any radius is negative
        if np.any(radii < 0):
            return -np.inf  # Invalid configuration
        
        # Check if circles go out of bounds
        for i in range(n):
            x, y, r = x[i, 0], x[i, 1], x[i, 2]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return -np.inf  # Invalid configuration
        
        # Check if circles overlap
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return -np.inf  # Invalid configuration
        
        # Objective: maximize the sum of radii (minimize negative sum)
        return -np.sum(radii)
    
    # Define bounds for x, y, and r
