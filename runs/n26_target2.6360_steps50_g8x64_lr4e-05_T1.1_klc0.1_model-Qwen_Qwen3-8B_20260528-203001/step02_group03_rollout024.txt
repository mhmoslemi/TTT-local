import numpy as np
import random
from scipy.optimize import minimize
from typing import Tuple, List
from dataclasses import dataclass

@dataclass
class Circle:
    x: float
    y: float
    r: float

def get_hexagonal_layout(n: int, max_x: float, max_y: float, spacing: float) -> List[Circle]:
    """
    Generate an initial hexagonal-like layout of circles.
    """
    circles = []
    row = 0
    while True:
        cols = int(np.ceil((max_x - 1e-6) / spacing)) + 1
        for col in range(cols):
            x = col * spacing
            y = row * spacing * np.sqrt(3) / 2
            if y > max_y:
                break
            circles.append(Circle(x, y, spacing / 2))
        if len(circles) >= n:
            break
        row += 1
    return circles[:n]

def compute_distances(circles: List[Circle]) -> np.ndarray:
    """
    Compute pairwise distances between circles.
    """
    dists = np.zeros((len(circles), len(circles)))
    for i in range(len(circles)):
        for j in range(i + 1, len(circles)):
            dx = circles[i].x - circles[j].x
            dy = circles[i].y - circles[j].y
            dists[i, j] = np.sqrt(dx*dx + dy*dy)
    return dists

def objective_function(x: np.ndarray, circles: List[Circle], max_x: float, max_y: float, spacing: float) -> float:
    """
    Objective function to minimize (negative of sum of radii).
    """
    total_radius = 0.0
    for i in range(len(circles)):
        x_i, y_i = x[2*i], x[2*i + 1]
        r_i = spacing / 2
        total_radius += r_i
    return -total_radius

def constraint_function(x: np.ndarray, circles: List[Circle], max_x: float, max_y: float, spacing: float) -> float:
    """
    Ensure all circles are within the unit square.
    """
    for i in range(len(circles)):
        x_i, y_i = x[2*i], x[2*i + 1]
        if not (0 - 1e-6 <= x_i <= max_x + 1e-6 and 0 - 1e-6 <= y_i <= max_y + 1e-6):
            return -1e6
    return 0.0

def constraint_distance(x: np.ndarray, circles: List[Circle], max_x: float, max_y: float, spacing: float, tolerance: float) -> float:
    """
    Ensure circles do not overlap (distance >= sum of radii - tolerance).
    """
    for i in range(len(circles)):
        x_i, y_i = x[2*i], x[2*i + 1]
        r_i = spacing / 2
        for j in range(i + 1, len(circles)):
            x_j, y_j = x[2*j], x[2*j + 1]
            r_j = spacing / 2
            dx = x_i - x_j
            dy = y_i - y_j
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < r_i + r_j - tolerance:
                return -1e6
    return 0.0

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    max_x, max_y = 1.0, 1.0
    spacing = 0.15
    n = 26
    initial_circles = get_hexagonal_layout(n, max_x, max_y, spacing)
    
    # Convert to numpy arrays
    x = np.zeros(2 * n)
    for i in range(n):
        x[2*i] = initial_circles[i].x
        x[2*i + 1] = initial_circles[i].y
    
    # Optimization
    bounds = []
    for i in range(n):
        bounds.extend([(0.0, max_x + 1e-6), (0.0, max_y + 1e-6)])
    
    constraints = [
        {'type': 'ineq', 'fun': constraint_function, 'args': (initial_circles, max_x, max_y, spacing)},
        {'type': 'ineq', 'fun': constraint_distance, 'args': (initial_circles, max_x, max_y, spacing, 1e-6)}
    ]
    
    result = minimize(objective_function, x, bounds=bounds, constraints=constraints, method='SLSQP', tol=1e-6)
    x_opt = result.x
    
    # Extract centers and radii
    centers = np.zeros((n, 2))
    for i in range(n):
        centers[i, 0] = x_opt[2*i]
        centers[i, 1] = x_opt[2*i + 1]
    radii = np.full(n, spacing / 2)
    
    # Validate packing
    valid, message = validate_packing(centers, radii)
    if not valid:
        print(f"Packing validation failed: {message}")
        raise ValueError(message)
    
    sum_radii = np.sum(radii)
    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
