import numpy as np
import scipy.optimize as opt
import math
import random

def run_packing():

        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        
        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break
                x = j * (2 * max_radius_initial) + max_radius_initial
                y = i * (2 * max_radius_initial) + max_radius_initial
                centers[idx] = [x, y]
                idx += 1

        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)

                dy = (np.random.rand() - 0.5) * 2 * (max_radius_initial - dist_to_boundary)
                new_x = max(0, min(1, x + dx))

            return False, "NaN values present"

            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

        flat_params = params.reshape(-1)

            return centers, radii
