import numpy as np
import scipy.optimize as opt
import math
import random

def run_packing():
    num_circles = 26

    def improved_initial_layout(num_circles):
        max_radius = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.15
        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        
        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break
                if i % 2 == 0:
                    x = j * (2 * max_radius) + max_radius
                else:
                    x = j * (2 * max_radius) + max_radius + max_radius
                y = i * (np.sqrt(3) * max_radius) + max_radius
                centers[idx] = [x, y]
                idx += 1

        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            if dist_to_boundary < max_radius:
                new_x = max(0, min(1, x + (dist_to_boundary - max_radius) * np.random.uniform(-0.5, 0.5)))
                new_y = max(0, min(1, y + (dist_to_boundary - max_radius) * np.random.uniform(-0.5, 0.5)))
                centers[i] = [new_x, new_y]

        initial_radii = np.full(num_circles, max_radius)
        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            if dist_to_boundary < max_radius:
                initial_radii[i] = dist_to_boundary

        return centers, initial_radii

    def validate_packing(centers, radii):
        n = centers.shape[0]
        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"
        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"
        return True, "ok"

    def objective(params):
        flat_params = params.reshape(-1)
        centers_opt = flat_params[:num_circles * 2].reshape((num_circles, 2))
        radii_opt = flat_params[num_circles * 2:].reshape((num_circles,))
        return -np.sum(radii_opt)  # Minimization of negative sum

    def genetic_algorithm():
        population_size = 200
        mutation_rate = 0.1
        generations = 500
        elite_fraction = 0.2

        population = []
        for _ in range(population_size):
            centers, radii = improved_initial_layout(num_circles)
            params = np.concatenate([centers.flatten(), radii])
            population.append(params)

        for generation in range(generations):
            fitness = []
            for params in population:
                centers_opt = params[:num_circles * 2].reshape((num_circles, 2))
                radii_opt = params[num_circles * 2:].reshape((num_circles,))
                is_valid, message = validate_packing(centers_opt, radii_opt)
                if not is_valid:
                    fitness.append(-np.inf)
                else:
                    fitness.append(np.sum(radii_opt))

            parents = np.argsort(fitness)[-int(population_size * elite_fraction):]
            parents = [population[i] for i in parents]

            offspring = []
            for _ in range(population_size - len(parents)):
                parent1 = random.choice(parents)
                parent2 = random.choice(parents)

                for i in range(len(child)):
                    if random.random() < mutation_rate:
                        child[i] += np.random.uniform(-0.01, 0.01)
                offspring.append(child)

            population = parents + offspring

        best_params = None
        best_fitness = -np.inf
        for params in population:
            centers_opt = params[:num_circles * 2].reshape((num_circles, 2))
            radii_opt = params[num_circles * 2:].reshape((num_circles,))

                continue

                best_params = params

            flat_x = x.reshape(-1)

            return centers, radii

    centers, radii = improved_initial_layout(num_circles)

    best_params = genetic_algorithm()

    radii_opt = best_params[num_circles * 2:].reshape((num_circles,))

    # Validate best parameters

    centers_opt, radii_opt = refine_optimization(centers_opt, radii_opt)
