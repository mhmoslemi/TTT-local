def generate_hexagonal_grid(n_circles):
    # Generate a hexagonal grid pattern with a certain number of rows and columns
    rows = int(math.ceil(math.sqrt(n_circles)))
    cols = int(math.ceil(n_circles / rows))
    centers = np.zeros((n_circles, 2))
    index = 0
    
    for row in range(rows):
        for col in range(cols):
            x = col + 0.5 * (row % 2)
            y = row
            centers[index] = [x, y]
            index += 1
    
    return centers

def scale_to_unit_square(centers):
    # Compute the bounding box of the initial grid
    min_x, max_x = np.min(centers[:, 0]), np.max(centers[:, 0])
    min_y, max_y = np.min(centers[:, 1]), np.max(centers[:, 1])
    width = max_x - min_x
    height = max_y - min_y

    # Scale the positions to fit within [0,1]^2
    scaled_centers = (centers - [min_x, min_y]) / [width, height]
    return scaled_centers

def refine_packing(centers, radii):
    n = centers.shape[0]
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        
        # Place circles near corners if they overflow
        if x - r < 0:
            centers[i, 0] = r
        elif x + r > 1:
            centers[i, 0] = 1 - r
        if y - r < 0:
            centers[i, 1] = r
        elif y + r > 1:
            centers[i, 1] = 1 - r
        
        # Ensure the radius fits within bounds
        radii[i] = np.minimum(r, np.minimum(1 - centers[i, 0], 1 - centers[i, 1]))
    
    return centers, radii

def objective_function(x):
    n = 26
    centers = x[:2 * n].reshape(n, 2)

    return energy - np.sum(radii)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):

            if dist < radii[i] + radii[j] - 1e-12:
