import numpy as np
import scipy.optimize as opt
import math
import random

def run_packing():

        # Generate a hexagonal grid layout adjusted for the unit square
        max_radius_initial = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.2
        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        
        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break
                x = j * (2 * max_radius_initial) + max_radius_initial
                y = i * (2 * max_radius_initial) + max_radius_initial
                centers[idx] = [x, y]
                idx += 1

        # Adjust positions for boundary constraints by clustering near corners
        adjustment_factor = 0.5
        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            if dist_to_boundary < max_radius_initial:
                # Place near corner if within boundary
                corner = np.random.choice(['top-left', 'top-right', 'bottom-left', 'bottom-right'])
                if corner == 'top-left':
                    new_x = max(0, x - (dist_to_boundary - max_radius_initial) * adjustment_factor)
                    new_y = max(0, y - (dist_to_boundary - max_radius_initial) * adjustment_factor)
                elif corner == 'top-right':
                    new_x = min(1, x + (dist_to_boundary - max_radius_initial) * adjustment_factor)
                    new_y = max(0, y - (dist_to_boundary - max_radius_initial) * adjustment_factor)
                elif corner == 'bottom-left':
                    new_x = max(0, x - (dist_to_boundary - max_radius_initial) * adjustment_factor)

                    new_y = min(1, y + (dist_to_boundary - max_radius_initial) * adjustment_factor)
                centers[i] = [new_x, new_y]

        initial_radii = np.full(num_circles, max_radius_initial)
        for i in range(num_circles):
            x, y = centers[i]

    def validate_packing(centers, radii):

        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"

        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

    def objective(params):
        flat_params = params.reshape(-1)
        centers_opt = flat_params[:num_circles * 2].reshape((num_circles, 2))

        return -np.sum(radii_opt)  # Minimization of negative sum

    def gradient_descent_refinement(centers, radii):
        bounds = []
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            x_low = max(0, x - r)
            x_high = min(1, x + r)
            y_low = max(0, y - r)
            y_high = min(1, y + r)
            r_low = max(0, r - 0.01)
            r_high = min(1, r + 0.01)
            bounds.extend([(x_low, x_high), (y_low, y_high), (r_low, r_high)])

        def constraints(x):
            flat_x = x.reshape(-1)
            centers_opt = flat_x[:num_circles * 2].reshape((num_circles, 2))

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii
