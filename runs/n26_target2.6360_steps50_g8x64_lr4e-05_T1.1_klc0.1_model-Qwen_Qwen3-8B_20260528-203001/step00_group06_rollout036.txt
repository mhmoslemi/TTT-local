from scipy.optimize import minimize
from scipy.optimize import Bounds

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"

def initial_guess():
    """Generate an initial guess for centers and radii."""
    # Place circles in a regular grid to start with
    num_circles = 26
    centers = np.random.rand(num_circles, 2)
    # Adjust positions to be more uniform
    centers = np.sort(centers, axis=0)
    # Assign radii as small as possible
    radii = np.full(num_circles, 0.01)
    return centers, radii

def objective_function(params, num_circles):
    """Objective function to maximize the sum of radii."""
    centers = params[:num_circles*2].reshape(num_circles, 2)
    radii = params[num_circles*2:].reshape(num_circles)
    return -np.sum(radii)  # Minimize negative sum to maximize sum

def constraint_non_overlap(params, num_circles):
    """Constraint to ensure circles do not overlap."""
    centers = params[:num_circles*2].reshape(num_circles, 2)
    radii = params[num_circles*2:].reshape(num_circles)
    constraints = []
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = np.linalg.norm(centers[i] - centers[j])
            constraint = dist - (radii[i] + radii[j])
            constraints.append(constraint)
    return np.array(constraints)

def constraint_boundary(params, num_circles):
    """Constraint to ensure all circles are within the unit square."""
    centers = params[:num_circles*2].reshape(num_circles, 2)
    radii = params[num_circles*2:].reshape(num_circles)
    constraints = []
    for i in range(num_circles):

        r = radii[i]
        constraints.append(x - r)
        constraints.append(1 - x - r)
        constraints.append(y - r)
        constraints.append(1 - y - r)
    return np.array(constraints)

def run_packing():
    num_circles = 26
    centers, radii = initial_guess()
    params = np.concatenate([centers.flatten(), radii])
    
    # Define bounds for centers and radii
    bounds = []
    for _ in range(num_circles):

    # Set up optimization
    result = minimize(
        fun=objective_function,
        x0=params,
        args=(num_circles,),

        bounds=bounds,
        constraints=[
