import numpy as np
import scipy.optimize as optimize

def run_packing():

    n = 26

    centers = np.zeros((n, 2))

    num_rows = 6
    num_cols = 5
    spacing = 0.22  # spacing between centers (adjustable for hexagonal packing)
    for i in range(n):

        x = col * spacing
        y = row * spacing * np.sqrt(3)/2
        centers[i] = [x, y]

    # Optimization function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize -sum(radii) to maximize sum(radii)

        for i in range(n):
            if radii[i] < 0:
                return False
