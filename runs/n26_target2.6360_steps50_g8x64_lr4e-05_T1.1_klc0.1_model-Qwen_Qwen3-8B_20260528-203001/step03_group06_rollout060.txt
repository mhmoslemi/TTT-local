import cvxpy as cp

    center_vars = cp.Variable((n, 2))
    radius_vars = cp.Variable(n, non_negative=True)

    objective = cp.Maximize(cp.sum(radius_vars))

    # Constraints
    # Each circle must be within the unit square
    for i in range(n):
        x, y = center_vars[i]
        r = radius_vars[i]
        constraints.append(x - r >= 0)

        constraints.append(y - r >= 0)
        constraints.append(y + r <= 1)
