import numpy as np
import scipy.optimize as opt
from typing import Tuple

def validate_packing(centers, radii):

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    return True, "ok"

def generate_initial_config():

    radii = []

    radii = np.array(radii)
    return centers, radii

def objective_function(x, num_circles, centers, radii):

    new_centers = np.reshape(x[:2*num_circles], (num_circles, 2))

def constraint_circle_overlap(x, num_circles, centers, radii):
