import cvxpy as cp

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    r = 0.08
    num_circles = 26

    rows = 3
    cols = 8

            x = j * (2 * r) + r

            if i % 2 == 1:
                x += r
            if idx < num_circles:
                centers[idx] = [x, y]
                idx += 1
    centers = centers[:num_circles]

    # Create variables for radii and centers
    radii = cp.Variable(num_circles, nonneg=True)
    centers_var = cp.Variable((num_circles, 2), nonneg=True)

    # Set the bounds for the center coordinates
    bounds_center = cp.vstack([
        cp.SOC(centers_var[i, 0] - radii[i] <= 1 - 1e-12),

        cp.SOC(centers_var[i, 1] - radii[i] <= 1 - 1e-12),

        for i in range(num_circles)
    ])

    bounds_radius = [radii[i] <= 1 - 1e-12 for i in range(num_circles)]

    objective = cp.Maximize(cp.sum(radii))

    # Solve the optimization problem
    prob = cp.Problem(objective, bounds_center + bounds_radius + constraints_overlap)
    prob.solve(solver=cp.SCS)
