import numpy as np
import cvxpy as cp
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # We'll place 6 rows with 5, 6, 6, 6, 6, 5 circles respectively

    centers = []
    radii = []

    # Generate hexagonal grid
    row = 0
    col = 0
    for i in range(num_circles):
        x = col * (1 + math.sqrt(3)/2) / 2
        y = row * (math.sqrt(3)/2)

        radii.append(0.1)
        # Move to next column
        col += 1
        # If we reached the end of a row, move to next row
        if col >= 6:
            row += 1
            col = 0

    # Convert to numpy arrays
    centers = np.array(centers)
    radii = np.array(radii)

    # Optimization to maximize sum of radii
    # Variable for centers and radii
    x = cp.Variable((num_circles, 2))
    r = cp.Variable(num_circles, nonneg=True)

    # Objective: maximize sum of radii
    objective = cp.Maximize(cp.sum(r))

    # Constraints
    constraints = []

    # Circles must stay within unit square
    for i in range(num_circles):
        constraints.append(cp.constraints.InequalityConstraint(
            cp.vstack([
                x[i, 0] - r[i] >= -1e-12,
                x[i, 1] - r[i] >= -1e-12,
                1 + 1e-12 - x[i, 0] >= -1e-12,
                1 + 1e-12 - x[i, 1] >= -1e-12

            -np.inf, np.inf
        ))

    # Circles must not overlap
    for i in range(num_circles):
        for j in range(i + 1, num_circles):
            dist = cp.norm(x[i] - x[j])
            constraints.append(dist >= r[i] + r[j] - 1e-12)

    # Initial feasible point (from base grid)
    initial_centers = np.array(centers)
    initial_radii = np.array(radii)
    problem = cp.Problem(objective, constraints)
    result = problem.solve(solver=cp.OSQP, max_iter=10000, eps_abs=1e-6, eps_rel=1e-6, verbose=False)

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
