import numpy as np
import random
import math
import scipy.optimize as optimize

def run_packing():

    num_circles = 26
    target_sum_radii = 2.635983

    # Define parameters for the genetic algorithm
    population_size = 100
    generations = 500
    mutation_rate = 0.1
    crossover_rate = 0.8
    elite_ratio = 0.1

    # Function to evaluate fitness of a population member
    def evaluate_fitness(centers, radii):
        # Check boundary constraints
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if not (0 - r <= x <= 1 + r and 0 - r <= y <= 1 + r):
                return 0.0

        # Check overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                distance = math.sqrt(dx * dx + dy * dy)
                if distance < radii[i] + radii[j] - 1e-8:
                    return 0.0

        # Compute total sum of radii as fitness
        return sum(radii)

    def generate_population(population_size):
        population = []

            centers = np.random.uniform(0, 1, (num_circles, 2))
            radii = np.random.uniform(0.01, 0.1, num_circles)
            population.append((centers, radii))
        return population

    def mutate(individual):

            if random.random() < mutation_rate:
                # Mutate center
                centers[i] += np.random.uniform(-0.05, 0.05, 2)
                # Clamp to [0,1] range

                # Mutate radius

                radii[i] = np.clip(radii[i], 0, 0.1)
        return (centers, radii)

    # Function to crossover two individuals
    def crossover(parent1, parent2):
        centers1, radii1 = parent1
        centers2, radii2 = parent2
        child_centers = np.copy(centers1)
        child_radii = np.copy(radii1)
        for i in range(num_circles):

                child_centers[i] = centers2[i]
