import numpy as np
import scipy.optimize as optimize

def compute_distances(centers):
    n = centers.shape[0]
    dists = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dists[i, j] = np.sqrt(dx * dx + dy * dy)
            dists[j, i] = dists[i, j]
    return dists

def objective_function(radii):
    return -np.sum(radii)  # Negative for minimization

def constraint_boundary(centers, radii):
    n = centers.shape[0]
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
            return -1
    return 0

def constraint_overlap(centers, radii):
    n = centers.shape[0]
    dists = compute_distances(centers)
    for i in range(n):
        for j in range(i + 1, n):
            if dists[i, j] < radii[i] + radii[j] - 1e-6:
                return -1
    return 0

def run_packing():

    n = 26
