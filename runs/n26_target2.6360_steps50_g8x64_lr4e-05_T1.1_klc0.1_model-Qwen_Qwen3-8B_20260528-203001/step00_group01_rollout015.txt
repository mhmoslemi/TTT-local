import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial estimate for number of rows and columns
    n = 26

    # Initial guess for centers in a hexagonal grid-like pattern
    centers = np.zeros((n, 2))
    r_initial = 0.1  # initial estimate for radius
    row_idx = 0
    col_idx = 0
    for i in range(n):
        # Adjust row and column indices to fit hexagonal pattern
        if i % 2 == 0:
            centers[i, 0] = col_idx * (1 + r_initial * np.sqrt(3))
            centers[i, 1] = row_idx * (2 * r_initial)
        else:
            centers[i, 0] = col_idx * (1 + r_initial * np.sqrt(3))
            centers[i, 1] = row_idx * (2 * r_initial) + r_initial
        col_idx += 1
        if col_idx >= 10:
            col_idx = 0
            row_idx += 1

    # Initial estimate for radii
    radii = np.full(n, r_initial)

    def objective(x):
        # x is a 1D array of length n, where x[i] is the radius of circle i
        return -np.sum(x)  # maximizing the sum of radii

    def constraint_centers(x):
        # Ensure centers are within the square
        for i in range(n):
            x_i, y_i = centers[i]
            r_i = x[i]
            if (x_i - r_i < -1e-8 or x_i + r_i > 1 + 1e-8 or
                y_i - r_i < -1e-8 or y_i + r_i > 1 + 1e-8):
                return np.inf
        return 0

    def constraint_overlap(x):
        # Ensure centers are at least as far apart as the sum of their radii
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j])**2))
                if dist < x[i] + x[j] - 1e-8:
                    return np.inf
        return 0

    # Optimization bounds: radii must be non-negative
    bounds = [(0, 1) for _ in range(n)]

    # Optimization constraints
    constraints = [
        {'type': 'ineq', 'fun': constraint_centers},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Optimization options
    options = {
        'maxiter': 1000,
        'ftol': 1e-10,
        'disp': False

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
