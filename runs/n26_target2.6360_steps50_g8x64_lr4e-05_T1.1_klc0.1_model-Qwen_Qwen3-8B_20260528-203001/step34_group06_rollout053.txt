import numpy as np
import scipy.optimize as opt
import math
import random

def run_packing():

    def improved_initial_layout(num_circles):

        max_radius = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.2
        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))

        # Generate primary hexagonal grid for most circles
        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break
                if i % 2 == 0:
                    x = j * (2 * max_radius) + max_radius
                else:
                    x = j * (2 * max_radius) + max_radius + max_radius
                y = i * (np.sqrt(3) * max_radius) + max_radius
                centers[idx] = [x, y]
                idx += 1

        # Place extra circles near boundaries for better utilization
        remaining = num_circles - idx
        for i in range(remaining):
            # Add circles near corners and along edges
            if i % 4 == 0:
                # Top-left corner
                centers[idx] = [max_radius, max_radius]
            elif i % 4 == 1:
                # Top-right corner
                centers[idx] = [1 - max_radius, max_radius]
            elif i % 4 == 2:
                # Bottom-left corner
                centers[idx] = [max_radius, 1 - max_radius]
            else:
                # Bottom-right corner
                centers[idx] = [1 - max_radius, 1 - max_radius]
            idx += 1

        # Adjust radii based on distance to boundaries
        initial_radii = np.full(num_circles, max_radius)
        for i in range(num_circles):
            x, y = centers[i]

                initial_radii[i] = dist_to_boundary

        return centers, initial_radii

            return False, "NaN values present"

                return False, f"Circle {i} has negative radius {radii[i]}"

            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

        flat_params = params.reshape(-1)

        return -np.sum(radii_opt)  # Minimization of negative sum

    def genetic_algorithm():
        population_size = 200
        mutation_rate = 0.1
        generations = 300

        # Initial population
        population = []
        for _ in range(population_size):

        for generation in range(generations):

            fitness = []
            for params in population:

            offspring = []

                parent1 = random.choice(parents)

                    if random.random() < mutation_rate:
                        child[i] += np.random.uniform(-0.01, 0.01)
                offspring.append(child)

            # Replace population
            population = parents + offspring

        # Evaluate final population
        best_params = None
        best_fitness = -np.inf
        for params in population:
            centers_opt = params[:num_circles * 2].reshape((num_circles, 2))

        result = opt.minimize(

            constraints={'type': 'ineq', 'fun': constraints}
        )

        if result.success:
            return result.x[:num_circles * 2].reshape((num_circles, 2)), result.x[num_circles * 2:].reshape((num_circles,))

            return centers, radii

    best_params = genetic_algorithm()

    # Validate the packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Validation failed: {message}")
        # Fallback to initial layout if optimization fails
        centers_opt = centers

    centers_opt, radii_opt = refine_optimization(centers_opt, radii_opt)

    is_valid, message = validate_packing(centers_opt, radii_opt)
