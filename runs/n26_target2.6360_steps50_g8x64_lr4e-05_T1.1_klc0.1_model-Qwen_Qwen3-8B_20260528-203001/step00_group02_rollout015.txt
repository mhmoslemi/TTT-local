import numpy as np
import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial guess for the centers (randomly placed)
    n = 26
    centers = np.random.rand(n, 2) * 1.0
    radii = np.random.rand(n) * 0.1  # Initial radii to avoid overlap

    # Objective function: Maximize sum of radii (negative because scipy minimizes)
    def objective(x):
        return -np.sum(x[n:])

    # Flatten centers and radii for optimization
    x0 = np.concatenate((centers.ravel(), radii))

    # Constraints
    def constraint_boundary(x):
        centers_opt = x[:n*2].reshape(n, 2)
        radii_opt = x[n*2:]
        for i in range(n):
            x_i, y_i = centers_opt[i]
            r_i = radii_opt[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return np.inf
        return 0

    def constraint_overlap(x):
        centers_opt = x[:n*2].reshape(n, 2)
        radii_opt = x[n*2:]
        for i in range(n):
            for j in range(i+1, n):
                dx = centers_opt[i, 0] - centers_opt[j, 0]
                dy = centers_opt[i, 1] - centers_opt[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_opt[i] + radii_opt[j] - 1e-6:
                    return np.inf
        return 0

    # Optimization options
    bounds = []
    for _ in range(n*2):
        bounds.append((0.0, 1.0))  # Bounds for centers [0,1]x[0,1]
    for _ in range(n):
        bounds.append((0.0, 1.0))  # Bounds for radii [0,1]

    cons = [
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_overlap}
    ]

    # Run optimization
    result = opt.minimize(
        objective,
        x0,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        tol=1e-6
    )

    # Extract results
    centers_opt = result.x[:n*2].reshape(n, 2)
    radii_opt = result.x[n*2:]

    # Validate packing
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Validation failed: {message}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
