import numpy as np
import scipy.optimize as opt

def run_packing():
    num_circles = 26

    def hexagonal_initial_layout(num_circles):
        # Compute initial spacing to fit in square
        max_radius = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.15
        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))

        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break
                x = j * (2 * max_radius)
                y = i * (np.sqrt(3) * max_radius)
                if i % 2 == 1:
                    x += max_radius
                centers[idx] = [x, y]
                idx += 1

        # Adjust positions to bring circles near boundaries
        for i in range(num_circles):
            x, y = centers[i]
            dist_left = x
            dist_right = 1 - x
            dist_bottom = y
            dist_top = 1 - y
            min_dist = min(dist_left, dist_right, dist_bottom, dist_top)
            if min_dist < 1e-6:
                if dist_left < 1e-6:
                    x = 0.0 + max_radius
                elif dist_right < 1e-6:
                    x = 1.0 - max_radius
                elif dist_bottom < 1e-6:
                    y = 0.0 + max_radius
                elif dist_top < 1e-6:
                    y = 1.0 - max_radius
                centers[i] = [x, y]

        return centers

    def initial_radii(centers):
        max_radius = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.15
        radii = np.full(num_circles, max_radius)
        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            if dist_to_boundary < max_radius:
                radii[i] = dist_to_boundary
        return radii

    def validate_layout(centers, radii):
        n = centers.shape[0]
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"
        return True, "ok"

    # Generate initial random layout with hexagonal packing
    centers = hexagonal_initial_layout(num_circles)
    radii = initial_radii(centers)
    is_valid, message = validate_layout(centers, radii)
    if not is_valid:
        print(f"Initial layout also failed: {message}")
        return centers, radii, np.sum(radii)

    best_sum = np.sum(radii)
    best_centers = centers.copy()
    best_radii = radii.copy()

    # Multi-start optimization
    def optimize_layout(centers_initial, radii_initial):
        bounds = []
        for i in range(num_circles):
            bounds.append((0.0, 1.0))  # Radius
            bounds.append((0.0, 1.0))  # x
            bounds.append((0.0, 1.0))  # y

        constraints = []

        # Add boundary constraints
        for i in range(num_circles):
            def constraint_boundary(x):
                r_idx = i * 3
                x_idx = i * 3 + 1
                y_idx = i * 3 + 2
                r_val = x[r_idx]
                x_val = x[x_idx]
                y_val = x[y_idx]
                if x_val - r_val < -1e-12 or x_val + r_val > 1 + 1e-12 or \
                   y_val - r_val < -1e-12 or y_val + r_val > 1 + 1e-12:
                    return -1.0
                return 0.0
            constraints.append({'type': 'ineq', 'fun': constraint_boundary})

        # Add overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                def constraint_overlap(x):
                    r1_idx = i * 3
                    x1_idx = i * 3 + 1
                    y1_idx = i * 3 + 2
                    r2_idx = j * 3
                    x2_idx = j * 3 + 1

            best_centers = centers.copy()
            best_radii = radii.copy()

    return best_centers, best_radii, best_sum
