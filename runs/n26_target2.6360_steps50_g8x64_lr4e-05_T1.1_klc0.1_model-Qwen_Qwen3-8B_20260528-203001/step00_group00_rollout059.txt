import numpy as np
import scipy.optimize as opt

def run_packing():
    # Initial grid: Hexagonal packing
    N = 26
    num_rows = int(np.ceil(np.sqrt(N)))
    num_cols = int(np.ceil(N / num_rows))
    
    # Generate initial centers using hexagonal grid
    centers = np.zeros((N, 2))
    for i in range(N):
        row = i // num_cols
        col = i % num_cols
        x = col + 0.5 * (row % 2)
        y = row * np.sqrt(3) / 2
        centers[i] = [x, y]
    
    # Scale to unit square
    centers = (centers - centers.min(axis=0)) / (centers.max(axis=0) - centers.min(axis=0))
    
    # Generate initial radii
    # Start with uniform radii scaled based on total area allowed
    max_radius = 1 / (2 * np.sqrt(3))  # Hexagonal packing
    radii = max_radius * np.ones(N)
    
    # Define the objective: maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Define constraints
    def constraints(x):
        # x is a flattened array of radii and centers
        # Reshape into centers and radii
        centers_flat = x[:N*2].reshape(N, 2)
        radii_flat = x[N*2:].reshape(N)
        constraint_values = []
        
        # Check boundary constraint: circles must be inside the unit square
        for i in range(N):
            x, y = centers_flat[i]
            r = radii_flat[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                constraint_values.append(1e6)  # Penalize boundary violations heavily
        constraint_values = np.sum(constraint_values)

        # Check non-overlap: circles must not overlap
        for i in range(N):
            for j in range(i+1, N):
                dx = centers_flat[i, 0] - centers_flat[j, 0]
                dy = centers_flat[i, 1] - centers_flat[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii_flat[i] + radii_flat[j] - 1e-12:
                    constraint_values += 1e6  # Penalize overlaps
        return constraint_values

    # Combine all parameters: centers + radii
    x0 = np.concatenate([centers.flatten(), radii])

    # Set up optimization
    res = opt.minimize(
        fun=objective,
        x0=x0,
        method='L-BFGS-B',
        bounds=[(-1e-6, 1 - 1e-6) for _ in range(N*2 + N)],  # Bound centers and radii
        constraints={'type': 'eq', 'fun': constraints}
    )

    # Extract results
    centers = res.x[:N*2].reshape(N, 2)

    sum_radii = np.sum(radii)

        return np.array([]), np.array([]), 0.0

    print(f"Optimized packing with total sum of radii: {sum_radii:.6f}")
    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
