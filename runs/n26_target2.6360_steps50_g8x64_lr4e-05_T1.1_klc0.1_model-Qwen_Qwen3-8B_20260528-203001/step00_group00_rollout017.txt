import scipy.optimize as optimize
from typing import Tuple

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

        r = radii[i]

            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            if dist < radii[i] + radii[j] - 1e-12:

    return True, "ok"

        radii[i] = 0.15

    def objective(x):
        # x has shape (26*2 + 26,) -> 52 elements

        radii_flat = x[26*2:]
        return -np.sum(radii_flat)  # We minimize the negative sum

    # Constraints:
    # 1. Each circle must be within the unit square
    def in_bounds(x):
        centers_flat = x[:26*2].reshape((26, 2))
        radii_flat = x[26*2:]
        for i in range(26):
            x_c, y_c = centers_flat[i]
            r = radii_flat[i]
            if (x_c - r < -1e-12 or x_c + r > 1 + 1e-12

    # 2. No two circles may overlap

            for j in range(i + 1, 26):
                dist = np.sqrt(np.sum((centers_flat[i] - centers_flat[j]) ** 2))
