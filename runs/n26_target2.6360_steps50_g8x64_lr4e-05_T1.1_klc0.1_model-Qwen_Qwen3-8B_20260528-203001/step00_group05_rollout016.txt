import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_circles = 26
    n_rows = int(np.sqrt(num_circles))  # Number of rows in hexagonal packing
    n_cols = (num_circles + n_rows - 1) // n_rows  # Number of columns

    centers = np.zeros((num_circles, 2))

            centers[idx] = [x, y]
            idx += 1

    x = np.concatenate([centers.flatten(), r])  # Flatten for optimization

    # Objective: maximize sum of radii

        centers = x[:2*num_circles].reshape((num_circles, 2))
        radii = x[2*num_circles:]
        return -np.sum(radii)  # Minimize negative sum (equivalent to maximizing sum)

    def constraints(x):

        radii = x[2*num_circles:]
        constraints = []
        # Boundary constraints

            constraints.append((x_i - r_i + 1e-6) >= 0)  # Ensure x_i - r_i >= 0
            constraints.append((1.0 - x_i - r_i + 1e-6) >= 0)  # Ensure x_i + r_i <= 1

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
