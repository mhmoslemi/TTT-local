import numpy as np
from scipy.optimize import minimize
from scipy.spatial.distance import pdist, squareform

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # Hexagonal grid in the unit square
    rows = 7  # Number of rows in hexagonal grid
    cols = 4  # Number of columns in hexagonal grid
    spacing = 1.0 / (cols + 1)  # Adjust spacing to fit the grid
    centers = np.zeros((rows * cols, 2))

    for i in range(rows):
        for j in range(cols):
            if i % 2 == 0:  # Even rows: horizontal shift
                x = j * spacing + spacing / 2
            else:  # Odd rows: vertical shift
                x = (j + 0.5) * spacing
            y = i * spacing * np.sqrt(3) / 2
            idx = i * cols + j
            centers[idx, 0] = x
            centers[idx, 1] = y

    # Initial radii: based on the spacing
    radii = np.ones(rows * cols) * spacing / 2

    # Define the objective function to maximize the sum of radii
    def objective(x):
        centers_flat = x[:2 * len(radii)].reshape(len(radii), 2)
        radii_flat = x[2 * len(radii):]
        return -np.sum(radii_flat)  # Negative for minimization

    # Define constraints
    def constraint1(x):
        centers_flat = x[:2 * len(radii)].reshape(len(radii), 2)
        radii_flat = x[2 * len(radii):]
        return np.concatenate([centers_flat[:, 0] - radii_flat, 1 - centers_flat[:, 0] - radii_flat,
                               centers_flat[:, 1] - radii_flat, 1 - centers_flat[:, 1] - radii_flat])

    def constraint2(x):
        centers_flat = x[:2 * len(radii)].reshape(len(radii), 2)
        radii_flat = x[2 * len(radii):]
        dists = squareform(pdist(centers_flat))
        return np.concatenate([dists[i, j] - radii_flat[i] - radii_flat[j] for i in range(len(radii)) for j in range(i + 1, len(radii))])

    # Bounds for centers (x, y) in [0,1] and radii >= 0
    bounds = []
    for _ in range(len(radii)):
        bounds.append((-1, 1))  # for x
        bounds.append((-1, 1))  # for y
        bounds.append((0, 1))   # for radius

    # Initial feasible point with adjusted spacing
    x0 = np.concatenate([centers.flatten(), radii])

    # Optimization with constraints
    cons = [
        {'type': 'ineq', 'fun': constraint1},
        {'type': 'ineq', 'fun': constraint2}
    ]

    centers_final = result.x[:2 * len(radii)].reshape(len(radii), 2)

    if not valid:
        print(f"Warning: Packing failed to satisfy constraints - {msg}")

    return centers_final, radii_final, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
