from scipy.optimize import minimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:

        return -np.sum(radii)  # Minimize negative of sum to maximize it

    def constraint_overlap(centers, radii):
        n = centers.shape[0]
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.linalg.norm(centers[i] - centers[j])
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf
        return 0

    def constraint_boundary(centers, radii):
        n = centers.shape[0]
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-8 or x + r > 1 + 1e-8
                or y - r < -1e-8 or y + r > 1 + 1e-8):

        return 0

    def generate_hexagonal_grid(n_circles):
        rows = int(np.sqrt(n_circles))
        cols = int(np.ceil(n_circles / rows))
        centers = np.zeros((n_circles, 2))
        for i in range(rows):

                    x = j * np.sqrt(3) / 2
                else:
                    x = (j + 0.5) * np.sqrt(3) / 2
                y = i * 1.5
                # Rescale x and y to fit in unit square

                y = y / (1.5 * rows)
                centers[i * cols + j] = [x, y]
        return centers

    # Initialize centers in hexagonal packing
    centers = generate_hexagonal_grid(26)

    radii = np.array([0.05] * 26)

    x0 = np.concatenate([centers.flatten(), radii])

    bounds = []
    for _ in range(26 * 2):
        bounds.append((0.0, 1.0))
    for _ in range(26):
        bounds.append((0.0, 1.0))  # radii ≥ 0

    # Constraints for overlap and boundary violation

    result = minimize(
