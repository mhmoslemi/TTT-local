from scipy.optimize import minimize
import numpy.linalg as la
from itertools import combinations

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid layout: 5x5 grid with 25 circles
    # Add one more circle in an optimized position
    num_circles = 26
    grid_size = 5
    spacing = 1.0 / (grid_size - 1)
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    # Place 25 circles in a regular 5x5 grid
    for i in range(5):
        for j in range(5):
            x = i * spacing
            y = j * spacing
            centers[i*5 + j] = [x, y]
            radii[i*5 + j] = spacing / 2

    # Add one more circle in the center
    centers[25] = [0.5, 0.5]
    radii[25] = 0.25

    # Define the objective function to maximize the sum of radii
    def objective(x):
        # x is a flattened vector of centers and radii
        x_flat = x.reshape(-1, 2)
        radii_flat = x_flat[:, 1]
        return -np.sum(radii_flat)  # Negative for minimization

    # Define the constraints
    def constraint_bounds(x):
        x_flat = x.reshape(-1, 2)
        for i in range(num_circles):
            x_i, y_i = x_flat[i]
            r_i = x_flat[i, 1]
            if x_i - r_i < -1e-6 or x_i + r_i > 1 + 1e-6 or y_i - r_i < -1e-6 or y_i + r_i > 1 + 1e-6:
                return np.inf  # Penalize boundary violations
        return 0

    def constraint_overlap(x):
        x_flat = x.reshape(-1, 2)
        radii_flat = x_flat[:, 1]
        for i, j in combinations(range(num_circles), 2):
            dx = x_flat[i, 0] - x_flat[j, 0]
            dy = x_flat[i, 1] - x_flat[j, 1]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < radii_flat[i] + radii_flat[j] - 1e-6:
                return np.inf  # Penalize overlaps
        return 0

    # Initial guess based on the grid
    initial_guess = np.zeros(num_circles * 2)

        initial_guess[2*i] = centers[i, 0]

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
