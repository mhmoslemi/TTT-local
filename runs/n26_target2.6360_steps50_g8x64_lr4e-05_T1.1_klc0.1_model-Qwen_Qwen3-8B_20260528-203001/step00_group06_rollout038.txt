import numpy as np
from scipy.optimize import minimize
import matplotlib.pyplot as plt

def run_packing():
    # Define initial guess: grid-like placement with some spacing
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    spacing = 0.2
    centers = np.zeros((num_circles, 2))
    radii = np.ones(num_circles) * 0.05  # Initial radius

    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col * (1.0 / (cols - 1)) + spacing
        y = row * (1.0 / (rows - 1)) + spacing
        centers[i] = [x, y]

    def objective(radii):
        return -np.sum(radii)

    def constraints(radii, centers):
        # Constraint 1: All circles must be within the square
        for i in range(num_circles):
            r = radii[i]
            x, y = centers[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return np.inf
        # Constraint 2: Circles must not overlap
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

    # Optimization
    bounds = [(0.0, 0.5) for _ in range(num_circles)]
    result = minimize(
        objective,
        radii,
        args=(centers,),
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': lambda r: constraints(r, centers)}],
        bounds=bounds,
        tol=1e-6
    )

    # Final centers and radii
    centers_final = centers

    sum_radii = np.sum(radii_final)

    return centers_final, radii_final, sum_radii
