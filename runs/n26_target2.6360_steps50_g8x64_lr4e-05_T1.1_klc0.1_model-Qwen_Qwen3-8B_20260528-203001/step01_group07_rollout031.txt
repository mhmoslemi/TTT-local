def calculate_sum_radii(radii):
    return np.sum(radii)

def objective_function(x, centers, radii):

    return -calculate_sum_radii(radii)

def constraint_boundary(x, centers, radii, idx):

    return x_i - r_i + 1e-12  # Ensure x - r >= 0
