import scipy.optimize as opt
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial grid parameters
    n = 26
    rows = 5
    cols = 6
    row_height = np.sqrt(3) / 2
    col_width = 1.0 / cols
    
    # Define initial centers in a hexagonal grid
    def generate_hex_grid(num_circles):
        centers = np.zeros((num_circles, 2))
        for i in range(rows):
            for j in range(cols):
                x = j * col_width
                y = i * row_height
                if i % 2 == 1:
                    x += col_width / 2
                centers[i * cols + j] = [x, y]
        return centers

    # Initial radii for all circles
    initial_radii = np.full(n, 0.1)

    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
