import numpy as np
from scipy.optimize import minimize
from scipy.optimize import Bounds

def run_packing():
    # Number of circles
    n = 26

    # Define the number of rows and columns in the hexagonal grid
    rows = 6
    cols = 5  # Adjust for better fit

    # Compute the maximum distance between centers to ensure non-overlapping (diameter = 2*r)
    max_distance = 2 * 1.0  # Placeholder for initial radius

    # Initial grid parameters
    row_spacing = np.sqrt(3) / 2 * max_distance  # Hexagonal grid spacing
    col_spacing = max_distance

    # Generate initial centers in a hexagonal grid
    centers = []
    for i in range(rows):
        for j in range(cols):
            x = j * col_spacing
            y = i * row_spacing
            if i % 2 == 1:
                x += col_spacing / 2  # Adjust for offset in hexagonal grid
            centers.append([x, y])
    centers = np.array(centers)

    # Initial radii (will be refined during optimization)
    radii = np.ones(n) * 0.05

    # Function to compute the objective: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Function to compute constraints:
    # 1. All circles are within the unit square
    # 2. No two circles overlap (distance between centers >= sum of radii)

    def constraints(radii):
        # Constraint 1: All circles are within the unit square
        # Check if any circle is outside the square
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return np.inf  # Violation, return large value

                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf  # Violation, return large value

        return 0  # All constraints satisfied

    def constraint_func(radii):

    bounds = Bounds([0.0] * n, [1.0] * n)

    initial_guess = radii

    result = minimize(

        bounds=bounds,
        method='SLSQP',
