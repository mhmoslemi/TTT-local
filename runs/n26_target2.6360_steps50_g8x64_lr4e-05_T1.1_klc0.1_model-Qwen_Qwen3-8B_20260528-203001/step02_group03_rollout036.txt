import scipy.optimize as optimize

def run_packing():

    n_circles = 26
    centers = np.zeros((n_circles, 2))
    radii = np.zeros(n_circles)

    # Initial radius guess
    initial_radius = 0.08

    # Hexagonal grid parameters
    rows = int(np.ceil(np.sqrt(n_circles)))
    cols = int(np.ceil(n_circles / rows))
    spacing = 1.0 / (cols + 1)

    for i in range(n_circles):
        row = i // cols
        col = i % cols
        x = col * spacing
        y = row * spacing
        centers[i] = np.array([x, y])

    radii = np.full(n_circles, initial_radius)

    # Objective function: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # minimize negative sum to maximize sum

    # Constraints: no overlaps and within bounds
    def constraint_overlap(radii):
        # Check distances between centers
        n = len(radii)

            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return dist - (radii[i] + radii[j] - 1e-12)
        return 0

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
