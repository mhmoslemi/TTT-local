import scipy.optimize as opt

def run_packing():

    def improved_initial_layout(num_circles):
        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        radius = 1.0 / (2 * (np.sqrt(num_circles) + 1))

            for j in range(cols):
                if idx >= num_circles:
                    break
                x = j * (2 * radius) + radius
                y = i * (np.sqrt(3) * radius) + radius
                if i % 2 == 1:
                    x += radius
                centers[idx] = [x, y]
                idx += 1
        
        # Adjust for unit square
        for i in range(num_circles):
            x, y = centers[i]
            r = radius
            x = np.clip(x, r, 1 - r)
            y = np.clip(y, r, 1 - r)
            centers[i] = [x, y]
        
        return centers, np.full(num_circles, radius)

    # Generate multiple initial layouts
    def generate_initial_guesses(num_circles, num_guesses):
        guesses = []

        return -np.sum(radii_opt)

    def validate_packing(centers, radii):
        n = centers.shape[0]
        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"
        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"

            x, y = centers[i]
            r = radii[i]

                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
