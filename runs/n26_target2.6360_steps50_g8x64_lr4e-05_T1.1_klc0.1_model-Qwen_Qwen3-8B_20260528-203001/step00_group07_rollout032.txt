import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n_circles = 26

    centers = np.zeros((n_circles, 2))

    centers[0] = [0.1, 0.1]

    centers[2] = [0.1, 0.9]
    # 4. Bottom-right corner
    centers[3] = [0.9, 0.9]

    # Hexagonal grid in the middle (remaining 22 circles)

    grid_rows = 5
    grid_cols = 5
    grid_spacing = 0.25  # Adjust spacing to fit in the square

    # Start from the center
    x_start = 0.5
    y_start = 0.5

    # Fill the hexagonal grid with circles
    for i in range(4, n_circles):
        row = (i - 4) // grid_cols
        col = (i - 4) % grid_cols
        # Hexagonal grid
        x = x_start + col * grid_spacing

        if row % 2 == 1:

        centers[i] = [x, y]

    def objective(radii):
        return -np.sum(radii)  # Minimize negative total radius
