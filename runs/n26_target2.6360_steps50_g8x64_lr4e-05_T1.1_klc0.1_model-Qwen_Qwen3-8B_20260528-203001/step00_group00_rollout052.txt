from scipy.optimize import minimize
from scipy.optimize import Bounds

def get_initial_hexagonal_centers(num_circles):
    """Generate initial hexagonal grid placement of num_circles in a square."""
    # Determine approximate row and column counts
    cols = int(np.ceil(np.sqrt(num_circles)))
    rows = int(np.ceil(num_circles / cols))

    initial_r = 0.05
    radii = np.full(num_circles, initial_r)

            if x + initial_r > 1:

            if y + initial_r > 1:
                y = 1 - initial_r
                
            centers.append([x, y])
    
    # Ensure we only take the first num_circles circles (may be fewer than cols*rows)
    centers = np.array(centers)[:num_circles]
    
    return centers, radii
