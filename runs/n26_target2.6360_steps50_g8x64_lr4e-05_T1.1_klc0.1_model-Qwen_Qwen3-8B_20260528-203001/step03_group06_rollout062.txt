import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds

def run_packing():
    # Initial grid-based arrangement (can be adjusted)
    num_circles = 26
    grid_rows = 5
    grid_cols = 6
    grid_spacing = 1.0 / grid_rows  # spacing between rows
    grid_spacing_col = 1.0 / grid_cols  # spacing between columns
    
    # Generate initial centers
    centers = np.zeros((num_circles, 2))
    for i in range(grid_rows):
        for j in range(grid_cols):
            idx = i * grid_cols + j
            centers[idx, 0] = j * grid_spacing_col
            centers[idx, 1] = i * grid_spacing
    
    # Initial radii (same for all)
    radii = np.full(num_circles, 0.1)
    
    # Optimization objective: maximize sum of radii
    def objective(x):
        # x contains all variables (radii)
        # Flatten the radii array for optimization
        radii = x.reshape(num_circles)
        return -np.sum(radii)  # We minimize, so we negate to maximize
    
    # Optimization constraints
    def constraint_overlap(x):
        radii = x.reshape(num_circles)
        centers_current = centers.copy()
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers_current[i, 0] - centers_current[j, 0]
                dy = centers_current[i, 1] - centers_current[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return dist - (radii[i] + radii[j] - 1e-12)
        return 0.0
    
    def constraint_boundary(x):
        radii = x.reshape(num_circles)
        for i in range(num_circles):
            x_center, y_center = centers[i]
            r = radii[i]
            if x_center - r < -1e-12 or x_center + r > 1 + 1e-12 or y_center - r < -1e-12 or y_center + r > 1 + 1e-12:
                return -1e-12
        return 0.0
    
    constraints = [
        {'type': 'ineq', 'fun': constraint_overlap},
        {'type': 'ineq', 'fun': constraint_boundary},
    ]
    
    # Bounds for radii (non-negative)
    bounds = Bounds(0, 1.0)
    
    # Initial guess (flat vector of radii)
    initial_guess = np.full(num_circles, 0.1)
    
    # Run optimization
    result = optimize.minimize(
        fun=objective,
        x0=initial_guess,
        bounds=bounds,
        constraints=constraints,
        options={'disp': False, 'maxiter': 500}
    )
    
    # Extract optimized radii and centers
    optimized_radii = result.x
    sum_radii = np.sum(optimized_radii)
    
    # Final validation
    valid, message = validate_packing(centers, optimized_radii)
    if not valid:
        raise ValueError(f"Invalid packing: {message}")
    
    return centers, optimized_radii, sum_radii
