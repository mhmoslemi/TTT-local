from scipy.optimize import minimize
from scipy.optimize import Bounds
from scipy.optimize import LinearConstraint

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # ----------------------
    # 1. Generate initial points
    # ----------------------
    # Grid-based initial points
    num_circles = 26
    x_coords = np.linspace(0.1, 0.9, 6)
    y_coords = np.linspace(0.1, 0.9, 6)

    def objective(radii):
        return -np.sum(radii)  # minimize -sum(radii) => maximize sum(radii)

    # Constraint: each radius >= 0
    bounds = Bounds(np.zeros(num_circles), np.inf)

    # We'll treat this as a hard boundary constraint

    # Constraint: non-overlapping between circles
    def non_overlap_constraints(radii, centers):
        n = len(radii)
        A = np.zeros((n * (n - 1) // 2, 2 * n))
        b = np.zeros(n * (n - 1) // 2)
        for i in range(n):
            for j in range(i + 1, n):

                dist = np.sqrt(dx**2 + dy**2)

        return LinearConstraint(A, -np.inf, b)
