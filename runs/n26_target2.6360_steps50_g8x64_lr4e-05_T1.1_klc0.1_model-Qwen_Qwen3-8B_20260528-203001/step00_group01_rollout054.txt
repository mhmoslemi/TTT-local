import numpy as np
from scipy.optimize import minimize
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based packing with 5 rows and 5 columns, leaving a small gap for spacing
    n = 26
    grid_size = 5
    initial_centers = np.zeros((n, 2))
    initial_radii = np.full(n, 0.1)

    # Fill the grid centers and assign radii initially
    for i in range(n):
        row = i // grid_size
        col = i % grid_size
        x = col + (0.2 * row)
        y = row + (0.2 * col)
        initial_centers[i] = [x, y]

    # Optimization problem to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraints(centers, radii):
        # Define non-overlap and boundary constraints
        constraints = []
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                constraints.append(dist >= radii[i] + radii[j] - 1e-8)
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            constraints.append(x - r >= -1e-8)
            constraints.append(x + r <= 1 + 1e-8)
            constraints.append(y - r >= -1e-8)
            constraints.append(y + r <= 1 + 1e-8)
        return constraints

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = np.sqrt(dx * dx + dy * dy)
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
