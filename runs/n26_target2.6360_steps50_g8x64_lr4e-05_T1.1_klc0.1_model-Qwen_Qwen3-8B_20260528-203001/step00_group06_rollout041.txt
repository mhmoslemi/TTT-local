    def objective(x):

    bounds = []
    for _ in range(n):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate

    def square_bounds(x):
        return np.concatenate([
            np.maximum(x[::3] - x[2::3], 0.0),  # x - r >= 0
            np.maximum(x[1::3] - x[2::3], 0.0),  # y - r >= 0
            np.maximum(1.0 - x[::3] - x[2::3], 0.0),  # 1 - x - r >= 0
