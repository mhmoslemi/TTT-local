import scipy.optimize as opt
import matplotlib.pyplot as plt

def run_packing():

        # Generate multiple initial layouts to find a better starting point
        layouts = []

            max_radius = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.15
            centers = np.zeros((num_circles, 2))
            idx = 0
            rows = int(np.ceil(np.sqrt(num_circles)))
            cols = int(np.ceil(num_circles / rows))
            
            # Generate primary hexagonal grid for most circles
            for i in range(rows):
                for j in range(cols):
                    if idx >= num_circles:
                        break
                    if i % 2 == 0:
                        x = j * (2 * max_radius) + max_radius
                    else:
                        x = j * (2 * max_radius) + max_radius + max_radius
                    y = i * (np.sqrt(3) * max_radius) + max_radius
                    centers[idx] = [x, y]
                    idx += 1
            
            # Adjust remaining circles to fill gaps
            for i in range(num_circles):
                x, y = centers[i]
                dist_to_boundary = min(x, 1 - x, y, 1 - y)
                if dist_to_boundary < max_radius:
                    centers[i] = [
                        max(0, min(1, x + (dist_to_boundary - max_radius) * np.random.uniform(-0.5, 0.5))),
                        max(0, min(1, y + (dist_to_boundary - max_radius) * np.random.uniform(-0.5, 0.5)))
                    ]
            
            # Adjust radii based on distance to boundaries
            initial_radii = np.full(num_circles, max_radius)

                x, y = centers[i]
                dist_to_boundary = min(x, 1 - x, y, 1 - y)
                if dist_to_boundary < max_radius:
                    initial_radii[i] = dist_to_boundary
            
            layouts.append((centers, initial_radii))
        
        # Choose the layout with the largest sum of radii
        best_layout = max(layouts, key=lambda layout: np.sum(layout[1]))
        return best_layout[0], best_layout[1]

    def validate_packing(centers, radii):

            return False, "NaN values present"

                return False, f"Circle {i} has negative radius {radii[i]}"

                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"

        return True, "ok"

        flat_params = params.reshape(-1)

        return -np.sum(radii_opt)  # Minimization of negative sum
