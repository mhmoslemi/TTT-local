import numpy as np
from scipy.optimize import minimize
from typing import Tuple, Any

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Initial guess: evenly spaced grid of 26 circles within the square
    n = 26
    grid = np.indices((5, 5)).reshape(2, -1)
    grid -= np.array([2.5, 2.5])  # center at (0.5, 0.5)
    grid *= 0.4  # scale to fit within the square [0,1]
    grid += np.array([0.5, 0.5])  # shift to center the grid
    centers = grid.T
    radii = np.full(n, 0.06)  # initial guess for radius

    # Objective function: maximize the sum of radii
    def objective(x: np.ndarray) -> float:
        return -np.sum(x[26:])  # negative of the sum of radii

    # Constraints
    def constraint_center(x: np.ndarray, i: int) -> float:
        x_i = x[i*2:(i+1)*2]
        r_i = x[26 + i]
        return (x_i[0] - r_i) - 0.0  # x_i - r_i >= 0

    def constraint_right(x: np.ndarray, i: int) -> float:
        x_i = x[i*2:(i+1)*2]
        r_i = x[26 + i]
        return (1.0 - x_i[0] - r_i)  # 1.0 - x_i - r_i >= 0

    def constraint_top(x: np.ndarray, i: int) -> float:
        x_i = x[i*2:(i+1)*2]
        r_i = x[26 + i]
        return (1.0 - x_i[1] - r_i)  # 1.0 - x_i - r_i >= 0

    def constraint_bottom(x: np.ndarray, i: int) -> float:
        x_i = x[i*2:(i+1)*2]
        r_i = x[26 + i]
        return (x_i[1] - r_i)  # x_i - r_i >= 0

    def constraint_overlap(x: np.ndarray, i: int, j: int) -> float:
        x_i = x[i*2:(i+1)*2]
        x_j = x[j*2:(j+1)*2]
        r_i = x[26 + i]
        r_j = x[26 + j]
        dist = np.sqrt((x_i[0] - x_j[0])**2 + (x_i[1] - x_j[1])**2)
        return dist - r_i - r_j  # dist >= r_i + r_j

    # Create bounds for each variable
    # First 52 variables are for centers (x, y) of 26 circles
    # Last 26 variables are radii of the circles
    bounds = []
    for _ in range(n):
        bounds.append((0.0, 1.0))  # x-coordinate
        bounds.append((0.0, 1.0))  # y-coordinate
        bounds.append((0.0, 1.0))  # radius

    # Initialize the variables
    x0 = np.concatenate((centers.flatten(), radii))

    # Add center constraints
    for i in range(n):
        bounds[2*i] = (0.0, 1.0)
        bounds[2*i + 1] = (0.0, 1.0)
        bounds[2*i + 2] = (0.0, 1.0)

    # Set constraints for boundaries
    nonlin_constraints = []
    for i in range(n):
        nonlin_constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_center(x, i)})
        nonlin_constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_right(x, i)})
        nonlin_constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_top(x, i)})
        nonlin_constraints.append({'type': 'ineq', 'fun': lambda x, i=i: constraint_bottom(x, i)})

    # Set constraints for non-overlap
    for i in range(n):
        for j in range(i + 1, n):
            nonlin_constraints.append({'type': 'ineq', 'fun': lambda x, i=i, j=j: constraint_overlap(x, i, j)})

    # Perform the optimization
    result = minimize(
        objective,
        x0,
        method='SLSQP',
        bounds=bounds,
        constraints=nonlin_constraints,
        tol=1e-8
    )

    # Extract results
    centers = result.x[:2*n].reshape(n, 2)
    radii = result.x[2*n:]
    sum_radii = -result.fun

    # Validate the packing
    is_valid, message = validate_packing(centers, radii)
    if not is_valid:
        raise ValueError(f"Packing invalid: {message}")

    return centers, radii, sum_radii

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
