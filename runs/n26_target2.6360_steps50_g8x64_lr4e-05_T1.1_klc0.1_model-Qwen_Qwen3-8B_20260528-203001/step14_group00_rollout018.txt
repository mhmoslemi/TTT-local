import scipy.optimize as opt
import random

def run_packing():

                centers[idx] = [x, y]
                idx += 1

            # Move circle closer to the nearest boundary
            dist_to_left = x
            dist_to_right = 1 - x
            dist_to_bottom = y
            dist_to_top = 1 - y
            min_dist = min(dist_to_left, dist_to_right, dist_to_bottom, dist_to_top)
            if min_dist < 1e-6:
                if dist_to_left < 1e-6:
                    x = 0.0 + max_radius
                elif dist_to_right < 1e-6:
                    x = 1.0 - max_radius
                elif dist_to_bottom < 1e-6:
                    y = 0.0 + max_radius
                elif dist_to_top < 1e-6:
                    y = 1.0 - max_radius
                centers[i] = [x, y]

            x, y = centers[i]

                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

        for i in range(n):
