import numpy as np
import scipy.optimize as opt
import math
import random

def run_packing():
    num_circles = 26

    def generate_hexagonal_layout(num_circles):
        layouts = []
        max_radius = 1.0 / (2 * (np.sqrt(num_circles) + 1)) * 1.2
        centers = np.zeros((num_circles, 2))
        idx = 0
        rows = int(np.ceil(np.sqrt(num_circles)))
        cols = int(np.ceil(num_circles / rows))
        
        for i in range(rows):
            for j in range(cols):
                if idx >= num_circles:
                    break
                if i % 2 == 0:
                    x = j * (2 * max_radius) + max_radius
                else:
                    x = j * (2 * max_radius) + max_radius + max_radius
                y = i * (np.sqrt(3) * max_radius) + max_radius
                centers[idx] = [x, y]
                idx += 1

        # Adjust for boundaries
        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            if dist_to_boundary < max_radius:
                new_x = max(0, min(1, x + (dist_to_boundary - max_radius) * np.random.uniform(-0.5, 0.5)))
                new_y = max(0, min(1, y + (dist_to_boundary - max_radius) * np.random.uniform(-0.5, 0.5)))
                centers[i] = [new_x, new_y]

        radii = np.full(num_circles, max_radius)
        for i in range(num_circles):
            x, y = centers[i]
            dist_to_boundary = min(x, 1 - x, y, 1 - y)
            if dist_to_boundary < max_radius:
                radii[i] = dist_to_boundary

        layouts.append((centers, radii))
        
        return layouts

    def validate_packing(centers, radii):
        n = centers.shape[0]
        if np.isnan(centers).any() or np.isnan(radii).any():
            return False, "NaN values present"
        for i in range(n):
            if radii[i] < 0:
                return False, f"Circle {i} has negative radius {radii[i]}"
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if (x - r < -1e-12 or x + r > 1 + 1e-12 or
                y - r < -1e-12 or y + r > 1 + 1e-12):
                return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"
        for i in range(n):
            for j in range(i + 1, n):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return False, f"Circles {i} and {j} overlap"
        return True, "ok"

    def objective(params):
        flat_params = params.reshape(-1)
        centers_opt = flat_params[:num_circles * 2].reshape((num_circles, 2))
        radii_opt = flat_params[num_circles * 2:].reshape((num_circles,))
        return -np.sum(radii_opt)  # Minimization of negative sum

    def genetic_algorithm(initial_layouts):
        population_size = 200
        mutation_rate = 0.1
        generations = 500
        elite_fraction = 0.2

        population = []
        for _ in range(population_size):
            centers, radii = random.choice(initial_layouts)
            params = np.concatenate([centers.flatten(), radii])
            population.append(params)

        for generation in range(generations):
            fitness = []
            for params in population:
                centers_opt = params[:num_circles * 2].reshape((num_circles, 2))
                radii_opt = params[num_circles * 2:].reshape((num_circles,))
                is_valid, message = validate_packing(centers_opt, radii_opt)
                if not is_valid:
                    fitness.append(-np.inf)
                else:
                    fitness.append(np.sum(radii_opt))

            parents = np.argsort(fitness)[-int(population_size * elite_fraction):]
            parents = [population[i] for i in parents]

            offspring = []
            for _ in range(population_size - len(parents)):
                parent1 = random.choice(parents)
                parent2 = random.choice(parents)
                child = np.copy(parent1)
                for i in range(len(child)):
                    if random.random() < mutation_rate:
                        child[i] += np.random.uniform(-0.01, 0.01)
                offspring.append(child)

            population = parents + offspring

        best_params = None
        best_fitness = -np.inf
        for params in population:
            centers_opt = params[:num_circles * 2].reshape((num_circles, 2))
            radii_opt = params[num_circles * 2:].reshape((num_circles,))
            is_valid, message = validate_packing(centers_opt, radii_opt)
            if not is_valid:
                continue
            current_fitness = np.sum(radii_opt)
            if current_fitness > best_fitness:
                best_fitness = current_fitness
                best_params = params

        if best_params is None:
            return random.choice(initial_layouts)

        return best_params

    def refine_optimization(centers, radii):

        for i in range(num_circles):

            r = radii[i]
            x_low = max(0, x - r)
            x_high = min(1, x + r)
            y_low = max(0, y - r)
            y_high = min(1, y + r)
            r_low = max(0, r - 0.01)
            r_high = min(1, r + 0.01)
            bounds.extend([(x_low, x_high), (y_low, y_high), (r_low, r_high)])

            flat_x = x.reshape(-1)
            centers_opt = flat_x[:num_circles * 2].reshape((num_circles, 2))
            radii_opt = flat_x[num_circles * 2:].reshape((num_circles,))

                    return -1.0

                        return -1.0

            return 0.0

        result = opt.minimize(
            objective,
            initial_guess,
            method='SLSQP',
            bounds=bounds,
            constraints={'type': 'ineq', 'fun': constraints}
        )

        if result.success:
            return result.x[:num_circles * 2].reshape((num_circles, 2)), result.x[num_circles * 2:].reshape((num_circles,))
        else:
            return centers, radii

    # Generate initial layouts
    initial_layouts = generate_hexagonal_layout(num_circles)

    # Genetic algorithm optimization
    best_params = genetic_algorithm(initial_layouts)
    centers_opt = best_params[:num_circles * 2].reshape((num_circles, 2))
    radii_opt = best_params[num_circles * 2:].reshape((num_circles,))

    # Validate best parameters
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Validation failed: {message}")
        centers_opt = np.zeros((num_circles, 2))
        radii_opt = np.zeros(num_circles)
        is_valid, message = validate_packing(centers_opt, radii_opt)
        if not is_valid:
            print(f"Initial layout also failed: {message}")
            return centers_opt, radii_opt, 0.0

    # Refinement
    centers_opt, radii_opt = refine_optimization(centers_opt, radii_opt)

    # Final validation
    is_valid, message = validate_packing(centers_opt, radii_opt)
    if not is_valid:
        print(f"Refined layout failed: {message}")
        return np.zeros((num_circles, 2)), np.zeros(num_circles), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii
