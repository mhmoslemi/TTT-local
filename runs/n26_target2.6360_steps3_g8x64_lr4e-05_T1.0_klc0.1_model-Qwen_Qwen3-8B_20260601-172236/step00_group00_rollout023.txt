def run_packing():

    num_rows = 5
    num_cols = 6
    side_length = 1.0
    initial_radius = 0.05  # Initial radius to start optimization
    grid_step = 0.2  # Spacing between centers

    # Initial centers and radii

    radii = np.zeros(n)

    # Place the first row of circles

            radii[i * num_cols + j] = initial_radius

    # Define objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Negative for minimization

    # Define constraints
    constraints = []

    def constraint_center_in_bounds(_, centers, radii):
        return np.vstack((centers[:, 0] - radii, centers[:, 1] - radii, 1 - centers[:, 0] - radii, 1 - centers[:, 1] - radii)).flatten()

    def constraint_no_overlap(_, centers, radii):
        n = centers.shape[0]
        dist_matrix = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                dist_matrix[i, j] = dist - radii[i] - radii[j]
        return dist_matrix.flatten()

    # Add constraints to the optimization problem
    for i in range(n):
        constraints.append({
            'type': 'ineq',
            'fun': lambda r, i=i: r[i]  # Ensure non-negative radii
        })

        objective,
        radii,
        method='SLSQP',
        constraints=[{
            'type': 'ineq',
            'fun': constraint_center_in_bounds,

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
