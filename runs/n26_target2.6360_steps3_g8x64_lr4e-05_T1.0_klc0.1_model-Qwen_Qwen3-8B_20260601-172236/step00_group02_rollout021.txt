```python
import numpy as np
import scipy.optimize as optimize
import matplotlib.pyplot as plt

def run_packing():
    # Initial guess for centers and radii
    # Hexagonal packing in the center
    # 19 circles in a hexagonal arrangement
    n_hex = 19
    hex_centers = np.zeros((n_hex, 2))
    hex_radii = np.zeros(n_hex)

    r_initial = 0.05
    offset = 0.2

    for i in range(n_hex):
        row = i // 6
        col = i % 6
        x = offset + col * (2 * r_initial)
        y = offset + row * (np.sqrt(3) * r_initial) + (row % 2) * (np.sqrt(3) * r_initial / 2)
        hex_centers[i] = [x, y]
        hex_radii[i] = r_initial

    # Outer 7 circles near corners and edges
    outer_centers = np.zeros((7, 2))
    outer_radii = np.zeros(7)

    # Corners
    outer_centers[0] = [0.1, 0.1]
    outer_centers[1] = [0.1, 0.9]
    outer_centers[2] = [0.9, 0.9]
    outer_centers[3] = [0.9, 0.1]
    outer_centers[4] = [0.5, 0.1]
    outer_centers[5] = [0.1, 0.5]
    outer_centers[6] = [0.5, 0.9]

    outer_radii[:] = 0.15

    # Combine centers and radii
    centers = np.vstack((hex_centers, outer_centers))
    radii = np.concatenate((hex_radii, outer_radii))

    # Function to calculate sum of radii
    def objective_function(x):
        nonlocal centers, radii
        centers = x[:26*2].reshape((26, 2))
        radii = x[26*2:]
        return -np.sum(radii)  # Negative for minimization

    # Constraints
    def constraint_boundary(x):
        nonlocal centers, radii
        centers = x[:26*2].reshape((26, 2))
        radii = x[26*2:]
        for i in range(26):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1
        return 0

    def constraint_overlap(x):
        nonlocal centers, radii
        centers = x[:26*2].reshape((26, 2))
        radii = x[26*2:]
        for i in range(26):
            for j in range(i + 1, 26):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -1
        return 0

    # Initial guess
    initial_centers = centers.reshape(26*2)
    initial_radii = radii
    initial_guess = np.concatenate((initial_centers, initial_radii))

    # Optimization constraints
    cons = (
        {'type': 'ineq', 'fun': constraint_boundary},
        {'type': 'ineq', 'fun': constraint_overlap},
    )

    # Optimization
    result = optimize.minimize(
        objective_function,
        initial_guess,
        method='SLSQP',
        constraints=cons,
        bounds=[(0, 1) for _ in range(26*2)] + [(0, 1) for _ in range(26)]
    )

    # Extract the optimal centers and radii
    optimal_centers = result.x[:26*2].reshape((26, 2))
    optimal_radii = result.x[26*2:]
    sum_radii = np.sum(optimal_radii)

    # Validate the packing
```

```python
def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
