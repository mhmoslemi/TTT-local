def generate_initial_hexagonal_grid(n, spacing):
    """
    Generate initial positions using a hexagonal grid pattern.
    """
    positions = []
    for i in range(n):
        row = i // 6
        col = i % 6
        y = (row + 0.5 * col) * spacing
        x = col * spacing
        positions.append((x, y))
    return np.array(positions)

def objective_function(radii, centers):
    return -np.sum(radii)  # Minimize negative of sum of radii

def constraints(centers, radii):
    def constraint1(radii):
        return radii >= 0

    def constraint2(centers, radii):
        for i in range(len(centers)):
            x, y = centers[i]
            r = radii[i]
            if (x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1):
                return False
        return True

    def constraint3(centers, radii):
        for i in range(len(centers)):

                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx * dx + dy * dy)
                if dist < radii[i] + radii[j] - 1e-12:
                    return False
        return True

    return constraint1(radii) and constraint2(centers, radii) and constraint3(centers, radii)

def run_packing():
