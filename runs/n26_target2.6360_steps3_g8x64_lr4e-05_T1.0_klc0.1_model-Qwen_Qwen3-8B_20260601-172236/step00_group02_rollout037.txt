import numpy as np
import cvxpy as cp
from typing import Tuple

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    n = 26
    centers = cp.Variable((n, 2))
    radii = cp.Variable(n)

    # Objective: Maximize the sum of the radii
    objective = cp.Maximize(cp.sum(radii))

    # Constraints
    constraints = []

    # 1. All radii must be non-negative
    constraints.append(radii >= 0)

    # 2. All circles must fit within the unit square [0,1]×[0,1]
    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        constraints.append(x - r >= -1e-12)
        constraints.append(x + r <= 1 + 1e-12)
        constraints.append(y - r >= -1e-12)
        constraints.append(y + r <= 1 + 1e-12)

    # 3. No two circles can overlap
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = cp.sqrt(dx**2 + dy**2)
            constraints.append(dist >= radii[i] + radii[j] - 1e-12)

    # Solve the optimization problem
    problem = cp.Problem(objective, constraints)
    problem.solve(solver=cp.SCS, eps=1e-5)

    # Extract the results
    centers_np = np.array(centers.value)
    radii_np = np.array(radii.value)
    sum_radii = np.sum(radii_np)
