import numpy as np
import scipy.optimize as opt

def run_packing():

    n = 26

    # Initial grid of centers for hexagonal packing
    grid_size = 5  # Grid size for initial placement
    centers = np.zeros((n, 2))
    idx = 0
    for i in range(grid_size):
        for j in range(grid_size):
            x = i / (grid_size - 1)
            y = j / (grid_size - 1)
            centers[idx] = [x, y]
            idx += 1

    # Initial radii based on spacing between centers
    radii = np.zeros(n)

                if dist < min_dist:
                    min_dist = dist
        radii[i] = min_dist / 2

    # Define objective function to maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Define constraints for square boundaries
    def boundary_constraints(radii):
        for i in range(n):
            x, y = centers[i]
            if x - radii[i] < -1e-12 or x + radii[i] > 1 + 1e-12:
                return np.inf
            if y - radii[i] < -1e-12 or y + radii[i] > 1 + 1e-12:
                return np.inf
        return 0

    # Define constraints for circle overlaps
    def overlap_constraints(radii):

                if dist < radii[i] + radii[j] - 1e-12:
                    return np.inf
        return 0

    def constraints(radii):

        oc = overlap_constraints(radii)
        return bc + oc

    bounds = [(0, np.inf) for _ in range(n)]

    initial_guess = radii

    # Optimization
    result = opt.minimize(

        initial_guess,
        method="SLSQP",

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
