import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    # We'll place circles in a hexagonal grid
    # Define the number of rows and columns based on hexagonal packing
    # For 26 circles, a 5x6 grid (with one row having one fewer circle) is suitable

    centers = np.zeros((n, 2))
