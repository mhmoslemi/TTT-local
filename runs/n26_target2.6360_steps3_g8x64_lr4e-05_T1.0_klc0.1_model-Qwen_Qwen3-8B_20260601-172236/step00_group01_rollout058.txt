import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Starting positions for 26 circles in a hexagonal pattern
    def initial_hexagonal_packing(n):
        # Generate initial positions for n circles in a hexagonal packing
        centers = []
        for i in range(n):
            row = i // 6
            col = i % 6
            x = col - 2.5 + 0.5
            y = row - 2.5 + 0.5
            centers.append([x, y])
        return np.array(centers)

    def objective_function(radii, centers):
        return -np.sum(radii)

    def constraint_boundary(radii, centers):
        for i in range(len(radii)):
            x, y = centers[i]
            if not (0 - 1e-12 <= x - radii[i] and x + radii[i] <= 1 + 1e-12 and
                    0 - 1e-12 <= y - radii[i] and y + radii[i] <= 1 + 1e-12):
                return -np.inf
        return 0

    def constraint_overlap(radii, centers):
        for i in range(len(radii)):
            for j in range(i + 1, len(radii)):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return -np.inf
        return 0

    # Initial centers in a hexagonal pattern
    centers = initial_hexagonal_packing(26)
    radii = np.full(26, 0.1)  # Initial guess for radii

    bounds = [(0, 1) for _ in range(26)]

    # Constraints
    constraints = [

        {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers,)}
    ]

    # Optimization problem
    result = optimize.minimize(
        fun=objective_function,
        x0=radii,
        bounds=bounds,
        constraints=constraints,
        method='SLSQP',
        tol=1e-8
