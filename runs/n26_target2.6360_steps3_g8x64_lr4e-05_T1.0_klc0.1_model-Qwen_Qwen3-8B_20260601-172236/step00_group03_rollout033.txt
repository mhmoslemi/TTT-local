import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid-based arrangement
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    
    # Grid spacing
    space_x = 1.0 / (cols + 1)
    space_y = 1.0 / (rows + 1)
    
    # Initialize centers and radii
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)
    
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = (col + 1) * space_x
        y = (row + 1) * space_y
        centers[i] = [x, y]
        radii[i] = 0.05  # Initial guess for radius
    
    # Objective function to maximize total area (sum of radii squared)
    def objective(x):
        radii = x[:num_circles]
        return -np.sum(radii ** 2)  # Minimize negative of sum of radii squared
    
    # Constraint for boundary
    def boundary_constraint(x):
        radii = x[:num_circles]
        for i in range(num_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < 0 or x_i + r_i > 1 or y_i - r_i < 0 or y_i + r_i > 1:
                return r_i  # Penalize if constraint is violated
        return 0
    
    # Constraint for non-overlap
    def non_overlap_constraint(x):
        radii = x[:num_circles]
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx ** 2 + dy ** 2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return radii[i] + radii[j] - dist  # Penalize if constraint is violated
        return 0
    
    # Optimization with constraints
    bounds = [(0.0, 1.0) for _ in range(num_circles)]  # Bounds for radii
    
    # Use SLSQP method for constrained optimization
    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': boundary_constraint},
            {'type': 'ineq', 'fun': non_overlap_constraint}

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
