```python
import numpy as np
import scipy.optimize as optimize
import math

def run_packing():
    # Initial guess for hexagonal packing with 26 circles
    n = 26

    # Estimate number of rows and columns in hexagonal packing
    rows = int(np.sqrt(n))
    cols = (n + rows - 1) // rows

    # Generate initial hexagonal grid configuration
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col + 0.5 * (row % 2)
        y = row * np.sqrt(3) / 2
        centers[i] = [x, y]

    # Normalize to unit square
    centers /= max(np.max(centers, axis=0))

    # Initial guess for radii (equal for all)
    radii = np.ones(n) * 0.05

    # Objective function to maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum

    # Constraints
    def constraint_overlap(x):
        # x is a flattened vector of radii
        radii = x.reshape(-1)
        dists = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dists[i][j] = np.sqrt(dx*dx + dy*dy)
        # Ensure no overlap
        overlap = np.sum(radii[i] + radii[j] - dists[i][j] > 1e-8 for i in range(n) for j in range(i+1, n))
        return overlap

    def constraint_boundary(x):
        radii = x.reshape(-1)
        for i in range(n):
            x_center, y_center = centers[i]
            r = radii[i]
            if x_center - r < 0 or x_center + r > 1 or y_center - r < 0 or y_center + r > 1:
                return 1
        return 0

    # Bounds for radii
    bounds = [(0, 0.2) for _ in range(n)]

    # Optimization: maximize sum of radii
    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=[
            {'type': 'ineq', 'fun': constraint_overlap},
            {'type': 'ineq', 'fun': constraint_boundary}
        ]
    )

    radii_opt = result.x
    centers_opt = centers

    # Final validation
    valid, msg = validate_packing(centers_opt, radii_opt)
    if not valid:
        print(f"Optimization failed: {msg}")
        return np.zeros((n, 2)), np.zeros(n), 0.0

    sum_radii = np.sum(radii_opt)
    return centers_opt, radii_opt, sum_radii
```

```python
def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
