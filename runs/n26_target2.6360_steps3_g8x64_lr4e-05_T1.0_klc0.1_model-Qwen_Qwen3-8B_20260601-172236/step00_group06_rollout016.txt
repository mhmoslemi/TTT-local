from scipy.optimize import minimize
from scipy.optimize import Bounds

        r_i = x[i]
        return x_i - r_i >= -1e-12  # x - r >= 0
    def constraint_square_r(x, i):

        r_i = x[i]
        return x_i + r_i <= 1 + 1e-12  # x + r <= 1

        return y_i - r_i >= -1e-12  # y - r >= 0
