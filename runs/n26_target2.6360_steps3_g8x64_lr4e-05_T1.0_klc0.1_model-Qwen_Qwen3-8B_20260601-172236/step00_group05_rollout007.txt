import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds, NonlinearConstraint

def run_packing():
    # Initial rough hexagonal grid for 26 circles
    num_circles = 26

    # Define a hexagonal grid with rows and columns
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    
    # Define the initial grid positions (hexagonal lattice)
    initial_centers = []
    for i in range(rows):
        for j in range(cols):
            x = 0.5 + j * (1 / cols) + (j % 2) * (1 / (2 * cols))
            y = 0.5 + i * (1 / rows)
            initial_centers.append([x, y])
    initial_centers = np.array(initial_centers)

    # Initial estimate for radii (based on spacing)
    initial_radii = np.full(num_circles, 0.05)

    # Objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    # Constraints: non-overlapping circles and within square
    def constraint_overlap(radii, centers):
        n = len(radii)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return 1
        return 0

    def constraint_boundary(radii, centers):
        n = len(radii)
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return 1
        return 0

    # Define bounds for radii (non-negative)
    bounds = Bounds(lb=np.zeros(num_circles), ub=np.inf)

    # Define constraints
    def nonlinear_constraint(radii):
        return constraint_overlap(radii, initial_centers), constraint_boundary(radii, initial_centers)

    constraints = [NonlinearConstraint(nonlinear_constraint, lb=0, ub=0)]

    # Optimize
    result = opt.minimize(
        objective,
        initial_radii,
        method='SLSQP',
        bounds=bounds,
        constraints=constraints
    )

    # Extract results
    centers = initial_centers
    radii = result.x
    sum_radii = np.sum(radii)

    return centers, radii, sum_radii
