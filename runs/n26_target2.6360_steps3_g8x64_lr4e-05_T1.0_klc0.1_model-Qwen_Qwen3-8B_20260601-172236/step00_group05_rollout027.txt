import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial guess for centers
    n_circles = 26
    np.random.seed(42)
    centers = np.random.rand(n_circles, 2)
    radii = np.full(n_circles, 0.01)  # Initial small radii

    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum of radii to maximize it

    def constraint_boundary(radii):
        # Constraint: Each circle must be inside the unit square
        for i in range(n_circles):
            x, y = centers[i]
            r = radii[i]
            x_min, x_max = x - r, x + r
            y_min, y_max = y - r, y + r
            if x_min < -1e-12 or x_max > 1 + 1e-12 or y_min < -1e-12 or y_max > 1 + 1e-12:
                return float('inf')
        return 0

    def constraint_overlap(radii):
        # Constraint: Circles must not overlap
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return float('inf')
        return 0

    # Combine all constraints
    constraints = [
        {'type': 'eq', 'fun': constraint_boundary},
        {'type': 'eq', 'fun': constraint_overlap}
    ]

    # Perform optimization
    result = optimize.minimize(
        fun=objective,
        x0=radii,
        method='SLSQP',
        constraints=constraints,
        bounds=[(0, 1)] * n_circles
    )

    # Extract results
    centers_opt = centers.copy()
    radii_opt = result.x
    sum_radii = -result.fun

    return (centers_opt, radii_opt, sum_radii)
