import scipy.optimize as optimize

def run_packing():

    n_circles = 26

    # Define the number of rows and columns in a hexagonal packing
    # We choose 7 rows to allow for tight packing in a unit square
    n_rows = 7
    n_cols = 4  # 4 columns in the first row, 5 in the second, etc.

    # Define the hexagonal grid layout
    centers = np.zeros((n_circles, 2))
    idx = 0
    for row in range(n_rows):
        for col in range(n_cols):

                x = col * 2 / (n_cols + 1)
                y = row * np.sqrt(3) / (n_rows + 1)
            else:

            centers[idx] = [x, y]
            idx += 1

    # Initial guess for radii (all same)
    radii = np.ones(n_circles) * 0.1

    # Objective: maximize the sum of radii
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum

    # Constraints: circle must fit in [0,1]^2 and not overlap
    def constraints(radii):

        for i in range(n_circles):
            if radii[i] < 0:
                return False

            if x - radii[i] < -1e-12 or x + radii[i] > 1 + 1e-12:
                return False

        for i in range(n_circles):

        return True

    # Use a constrained optimization method
    result = optimize.minimize(
        objective,
        radii,
        method='SLSQP',
        constraints=[{'type': 'ineq', 'fun': constraints}],
        bounds=[(0, 1) for _ in range(n_circles)]

    optimized_radii = result.x

    # Compute the sum of radii
    sum_radii = np.sum(optimized_radii)
