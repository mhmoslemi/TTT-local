import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing():

    num_circles = 26
    rows = 6
    columns = 5
    spacing = 1.0 / (rows + 1)
    row_height = spacing * np.sqrt(3)
    col_width = spacing
    centers = np.zeros((num_circles, 2))
    radii = np.zeros(num_circles)

    for i in range(rows):

            col_index = j
            x = col_index * col_width + col_width / 2
            y = row_index * row_height + spacing
            centers[i * columns + j] = [x, y]
            radii[i * columns + j] = spacing / 2
            # Adjust for the first row to reduce vertical spacing
            if i == 0:
                y = spacing
                radii[i * columns + j] = spacing / 2
            # Adjust for the last row to reduce vertical spacing
            if i == rows - 1:
                y = 1 - spacing
                radii[i * columns + j] = spacing / 2

    # Define the objective function
    def objective(radii_vec):
        return -np.sum(radii_vec)  # minimize negative of sum to maximize

    # Define the constraints
    def constraint_center(x, y, r, index):
        return np.array([x - r, x + r, y - r, y + r])

    def constraint_overlap(x1, y1, r1, x2, y2, r2):
        return np.sqrt((x1 - x2) ** 2 + (y1 - y2) ** 2) - (r1 + r2)

    def constraints_func(x):
        # x is a flat array of [x0, y0, r0, x1, y1, r1, ..., xn, yn, rn]
        n = num_circles
        x_flat = x.reshape(n, 3)

                constraints.append(constraint_overlap(xi, yi, ri, xj, yj, rj))

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
