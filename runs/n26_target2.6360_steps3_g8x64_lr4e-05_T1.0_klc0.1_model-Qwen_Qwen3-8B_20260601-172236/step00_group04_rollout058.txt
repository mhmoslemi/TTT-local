import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    centers = np.random.uniform(0.1, 0.9, (num_circles, 2))
    radii = np.random.uniform(0.01, 0.05, num_circles)

    # Define the optimization problem
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize sum

        return x_i - r_i >= 0.0 and x_i + r_i <= 1.0 and y_i - r_i >= 0.0 and y_i + r_i <= 1.0

        x_i, y_i = x[2*i], x[2*i + 1]

        r_i = radii[i]
        r_j = radii[j]
        dx = x_i - x_j
        dy = y_i - y_j
        dist = np.sqrt(dx*dx + dy*dy)
        return dist - r_i - r_j >= -1e-8  # Allow small error due to optimization

    # Flatten centers and radii into a single vector
    x = np.concatenate((centers.reshape(-1), radii))

    # Define constraints
    constraints = []
    for i in range(num_circles):
