import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Step 1: Define initial grid of circle centers
    grid_size = 5
    grid_spacing = 1.0 / (grid_size - 1)
    centers = np.zeros((26, 2))
    idx = 0
    for i in range(grid_size):
        for j in range(grid_size):
            if idx < 26:
                centers[idx, 0] = i * grid_spacing
                centers[idx, 1] = j * grid_spacing
                idx += 1

    # Step 2: Initialize radii assuming equal spacing
    radii = np.full(26, 0.05)

    # Step 3: Define objective function to maximize sum of radii
    def objective(radii):
        return -np.sum(radii)

    # Step 4: Define constraints
    def constraints(radii):
        # Ensure non-overlapping circles
        n = len(radii)
        for i in range(n):
            for j in range(i + 1, n):
                d = np.sqrt(np.sum((centers[i] - centers[j])**2))
                if d < radii[i] + radii[j]:
                    return False
        return True

    # Step 5: Define optimization bounds
    bounds = Bounds(lb=0.0, ub=1.0)

    # Step 6: Perform optimization
    result = opt.minimize(

        radii,
        method='SLSQP',
        bounds=bounds,
        constraints={'type': 'ineq', 'fun': constraints},

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
