import numpy as np
import scipy.optimize as opt

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Define initial packing: hexagonal grid
    num_circles = 26
    rows = int(np.ceil(np.sqrt(num_circles)))
    cols = int(np.ceil(num_circles / rows))
    
    # Initial radius
    initial_radius = 0.18
    # Initial positions
    centers = np.zeros((num_circles, 2))
    for i in range(num_circles):
        row = i // cols
        col = i % cols
        x = col * (1 + initial_radius * 2) / cols
        y = row * (1 + initial_radius * 2) / rows
        centers[i] = [x, y]
    
    # Optimized radii
    radii = np.full(num_circles, initial_radius)
    
    # Define objective function
    def objective(x):
        # x contains all radii
        return -np.sum(x)

    def constraint_overlap(x):
        # x contains all radii

                dy = centers[i][1] - centers[j][1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < x[i] + x[j] - 1e-12:
                    return -1.0
        return 0.0
    
    def constraint_boundary(x):
        # Check if all circles are within the square

        {'type': 'ineq', 'fun': constraint_boundary}
    ]

        radii,
        method='SLSQP',

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
