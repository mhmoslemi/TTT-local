import numpy as np
import scipy.optimize as optimize

def run_packing():
    # Initial guess: grid layout inside the unit square
    n = 26
    centers = np.zeros((n, 2))
    radii = np.zeros(n)
    # Initial grid with some random offset for diversity
    grid_size = int(np.sqrt(n))
    for i in range(n):
        row = i // grid_size
        col = i % grid_size
        x = col / (grid_size - 1) + np.random.uniform(-0.1, 0.1)
        y = row / (grid_size - 1) + np.random.uniform(-0.1, 0.1)
        centers[i] = [x, y]
        # Initial radius based on spacing
        radii[i] = 0.05

    def objective(radii, centers):
        return -np.sum(radii)

    def constraint_overlap(radii, centers):
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist_sq = dx * dx + dy * dy
                if dist_sq < (radii[i] + radii[j]) ** 2 - 1e-8:
                    return -1.0
        return 0.0

    def constraint_boundary(radii, centers):
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-8 or x + r > 1 + 1e-8 or y - r < -1e-8 or y + r > 1 + 1e-8:
                return -1.0
        return 0.0

    # Optimization problem setup
    bounds = [(0., 1.) for _ in range(n)]
    cons = (
        {'type': 'ineq', 'fun': constraint_overlap, 'args': (centers,)},
        {'type': 'ineq', 'fun': constraint_boundary, 'args': (centers,)}
    )

    # Initial guess for radii
    initial_radii = np.full(n, 0.05)

    # Run optimization
    result = optimize.minimize(
        fun=lambda r: objective(r, centers),
        x0=initial_radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        tol=1e-6
    )

    # Extract results
    optimized_radii = result.x
    final_centers = centers
    sum_radii = np.sum(optimized_radii)

    return (final_centers, optimized_radii, sum_radii)
