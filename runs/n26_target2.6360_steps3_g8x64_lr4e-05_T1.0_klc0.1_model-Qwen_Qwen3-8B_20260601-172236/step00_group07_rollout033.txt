import numpy as np
import scipy.optimize as opt
from scipy.optimize import Bounds, LinearConstraint, NonlinearConstraint
from scipy.spatial.distance import pdist, squareform

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial square grid arrangement
    n = 26
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))
    
    # Initialize centers in a grid
    centers = np.zeros((n, 2))
    for i in range(n):
        row = i // cols
        col = i % cols
        x = col / (cols - 1)
        y = row / (rows - 1)
        centers[i] = [x, y]
    
    # Initial radii (equal for all)
    radii = np.full(n, 1 / (2 * np.sqrt(2)))  # Approximate initial radius for square packing
    
    # Objective function to maximize sum of radii
    def objective(x):
        # x is flattened array of centers and radii
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        return -np.sum(radii_flat)  # Negative for minimization
    
    # Bounds for centers and radii
    bounds = []
    for _ in range(n):
        bounds.extend([ (0, 1), (0, 1) ])  # for x, y of each center
        bounds.extend([ (0, 1) ])         # for radius of each circle
    
    # Nonlinear constraint: non-overlapping circles
    def non_overlap_constraints(x):
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        distances = squareform(pdist(centers_flat))
        constraints = []
        for i in range(n):
            for j in range(i+1, n):
                dist = distances[i, j]
                constraints.append(dist - (radii_flat[i] + radii_flat[j]) + 1e-6)
        return constraints
    
    # Bound constraint: circles within the unit square
    def boundary_constraints(x):
        centers_flat = x[:2*n].reshape(n, 2)
        radii_flat = x[2*n:]
        constraints = []
        for i in range(n):
            x_i, y_i = centers_flat[i]
            r_i = radii_flat[i]
            constraints.append(x_i - r_i + 1e-6)
            constraints.append(x_i + r_i - 1)
            constraints.append(y_i - r_i + 1e-6)
            constraints.append(y_i + r_i - 1)
        return constraints
    
    # Optimization
    x0 = np.concatenate([centers.ravel(), radii])
    result = opt.minimize(

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
