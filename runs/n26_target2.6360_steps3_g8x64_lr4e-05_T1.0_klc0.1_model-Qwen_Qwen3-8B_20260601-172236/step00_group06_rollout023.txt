import scipy.optimize as opt
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    n = 26

    # Initial guess: hexagonal close packing within the square
    # Start with a grid of rows and columns of circles
    # The number of rows and columns is determined by the square root of n
    row_count = int(np.sqrt(n))
    col_count = int(np.ceil(n / row_count))

    centers = np.zeros((n, 2))
    radius_initial = 0.1  # Initial guess radius

    for i in range(n):
        row = i // col_count
        col = i % col_count
        x = col + 0.5 * (row % 2)  # Offset every other row to create hexagonal packing
        y = row
        centers[i] = [x, y]

    # Scale centers to fit within the unit square
    centers /= max(np.max(centers, axis=0))
    centers = centers * 0.95  # Add some margin to prevent touching the square edges

    radii = np.full(n, radius_initial)

    # Objective function to minimize (negative sum of radii for maximization)

        x = np.reshape(x, (n,))
        return -np.sum(x)

    # Constraints
    constraints = []

    # Constraint: each circle must lie inside the unit square

            if x[i] < 0 or x[i] > 1:
                return -1
        return 0

                dx = xi - xj

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
