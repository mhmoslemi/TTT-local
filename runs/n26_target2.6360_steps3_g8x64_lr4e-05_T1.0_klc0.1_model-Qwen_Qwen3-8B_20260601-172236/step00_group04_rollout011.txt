import numpy as np
import scipy.optimize as optimize
import matplotlib.pyplot as plt
import math

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial hexagonal packing
    num_circles = 26
    radius_initial = 0.1  # Initial guess for radius
    centers = []
    for i in range(num_circles):
        col = i % 5
        row = i // 5
        x = col * 0.3 + 0.15
        y = row * math.sqrt(3) * 0.2 + 0.1
        centers.append([x, y])
    centers = np.array(centers)

    # Initial radii guess
    radii = np.full(num_circles, radius_initial)

    # Objective function: maximize sum of radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative because we minimize

    # Constraints
    def constraint_boundary(radii, centers):
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return -1  # Constraint violation
        return 0

    def constraint_overlap(radii, centers):
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                dx = centers[i][0] - centers[j][0]
                dy = centers[i][1] - centers[j][1]
                dist_sq = dx * dx + dy * dy
                if dist_sq < (radii[i] + radii[j]) ** 2 - 1e-12:
                    return -1  # Constraint violation
        return 0

    # Define the constraints
    cons = (
        {'type': 'ineq', 'fun': lambda r: constraint_boundary(r, centers)},
        {'type': 'ineq', 'fun': lambda r: constraint_overlap(r, centers)},
    )

    # Bounds for radii
    bounds = [(0, 1) for _ in range(num_circles)]

    # Optimization
    result = optimize.minimize(
        fun=objective,
        x0=radii,
        bounds=bounds,
        constraints=cons,
        method='SLSQP',
        options={'maxiter': 100, 'ftol': 1e-10}

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
