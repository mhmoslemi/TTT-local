import scipy.optimize as optimize

def run_packing():

    def objective_function(x, n_circles):
        # Flatten x into centers and radii
        centers = x[:n_circles*2].reshape(n_circles, 2)
        radii = x[n_circles*2:].reshape(n_circles)
        return -np.sum(radii)  # Minimize negative sum (equivalent to maximizing sum)

    def constraint_boundary(x, n_circles):
        centers = x[:n_circles*2].reshape(n_circles, 2)
        radii = x[n_circles*2:].reshape(n_circles)
        for i in range(n_circles):
            x_i, y_i = centers[i]
            r_i = radii[i]
            if x_i - r_i < -1e-12 or x_i + r_i > 1 + 1e-12 or y_i - r_i < -1e-12 or y_i + r_i > 1 + 1e-12:
                return -1000  # Penalize for going out of bounds
        return 0

    def constraint_overlap(x, n_circles):
        centers = x[:n_circles*2].reshape(n_circles, 2)
        radii = x[n_circles*2:].reshape(n_circles)
        for i in range(n_circles):
            for j in range(i+1, n_circles):
                dist = np.sqrt(np.sum((centers[i] - centers[j])**2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return -1000  # Penalize for overlap
        return 0

    # --- Optimization Setup ---
    n_circles = 26
    x_initial = np.random.rand(n_circles*2 + n_circles)  # Random initial guess

    # Optimize
    bounds = []
    for _ in range(n_circles):
        bounds.extend([(0, 1), (0, 1), (0, 1)])  # x, y, r for each circle

        objective_function,
        x_initial,
        args=(n_circles,),
        method='SLSQP',
        bounds=bounds,
        constraints=[
