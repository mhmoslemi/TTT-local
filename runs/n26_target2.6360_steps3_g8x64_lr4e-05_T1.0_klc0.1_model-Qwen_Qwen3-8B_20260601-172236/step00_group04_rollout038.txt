import numpy as np
import scipy.optimize as opt
from typing import Tuple, List

def run_packing() -> Tuple[np.ndarray, np.ndarray, float]:
    # Define parameters
    n_circles = 26
    n_dim = 2
    radius_initial = 0.04  # Initial radius for all circles
    
    # Generate initial hexagonal grid
    centers = np.zeros((n_circles, n_dim))
    radii = np.full(n_circles, radius_initial)
    
    # Generate a hexagonal grid with 26 circles
    # Row-wise arrangement with spacing based on hexagonal packing
    for i in range(n_circles):
        row = i // 6  # 6 circles per row (except last row)
        col = i % 6
        x = col * (1.5 * radius_initial) + 0.5 * radius_initial
        y = row * (np.sqrt(3) * radius_initial) + 0.5 * radius_initial
        centers[i, 0] = x
        centers[i, 1] = y

    # Define objective function to maximize sum of radii
    def objective(radii_vec):
        return -np.sum(radii_vec)  # Minimize negative sum to maximize

    # Define constraints
    def constraint_center_bounds(x):
        # Center bounds: x must be within [0, 1]
        return np.concatenate([
            x[:n_circles, 0] - 1e-8,  # x - r >= 0
            1 - x[:n_circles, 0] - 1e-8,  # x + r <= 1
            x[:n_circles, 1] - 1e-8,  # y - r >= 0
            1 - x[:n_circles, 1] - 1e-8  # y + r <= 1
        ])

    def constraint_radius_bounds(r):
        # Radius must be non-negative
        return r

    def constraint_overlap(x, radii):
        # Overlap constraint between all pairs of circles
        centers = x[:n_circles, :]
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-8:
                    return np.inf  # Penalize overlap
        return 0

    # Define bounds for centers and radii
    bounds_centers = [
        (0, 1) for _ in range(n_circles * 2)  # (x1, y1, x2, y2, ..., xn, yn)
    ]
    bounds_radii = [(0, 1) for _ in range(n_circles)]

    # Combine all variables into one vector
    x0 = np.concatenate([centers.reshape(-1), radii])

    # Optimization with constraints
    constraints = [
        opt.Constraint(constraint_center_bounds, bounds=bounds_centers, name="center_bounds"),
        opt.Constraint(constraint_radius_bounds, bounds=bounds_radii, name="radius_bounds"),
        opt.Constraint(lambda x: constraint_overlap(x[:n_circles * 2], x[n_circles * 2:]), name="overlap")
    ]

    result = opt.minimize(
        objective,
        x0,
        method="SLSQP",
        bounds=constraints[0].bounds + constraints[1].bounds,
        constraints=constraints
    )

    # Extract final centers and radii
    centers_opt = result.x[:n_circles * 2].reshape(n_circles, 2)
    radii_opt = result.x[n_circles * 2:]
    sum_radii_opt = np.sum(radii_opt)

    # Validate the packing
    success, msg = validate_packing(centers_opt, radii_opt)
    if not success:
        raise ValueError(f"Validation failed: {msg}")

    return centers_opt, radii_opt, sum_radii_opt

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
