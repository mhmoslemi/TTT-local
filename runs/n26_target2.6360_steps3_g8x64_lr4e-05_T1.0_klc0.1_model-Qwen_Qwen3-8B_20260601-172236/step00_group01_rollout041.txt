```python
import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:
    # Initial grid layout (5x5 grid with 25 circles plus one extra at the corner)
    num_circles = 26
    grid = np.indices((5, 5)).reshape(2, -1)
    grid = grid.T
    grid = grid / 4.0  # Scale to fit the unit square [0, 1] × [0, 1]
    grid = np.vstack((grid, [0.0, 0.0]))  # Add one extra circle at the corner (0,0)
    centers = grid
    radii = np.full(num_circles, 0.04)  # Initial guess for radii

    # Optimization function to maximize total radii
    def objective(radii, centers):
        return -np.sum(radii)  # Negative for minimization

    def constraints(radii, centers):
        constraints = []
        # Boundary constraints
        for i in range(num_circles):
            x, y = centers[i]
            r = radii[i]
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1.0 - x - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: y - r})
            constraints.append({'type': 'ineq', 'fun': lambda r, x=x, y=y: 1.0 - y - r})
        # Non-overlap constraints
        for i in range(num_circles):
            for j in range(i + 1, num_circles):
                def dist_constraint(radii):
                    r1 = radii[i]
                    r2 = radii[j]
                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]
                    dist = np.sqrt(dx ** 2 + dy ** 2)
                    return dist - r1 - r2
                constraints.append({'type': 'ineq', 'fun': dist_constraint})
        return constraints

    # Initial guess
    initial_guess = radii

    # Set up optimization
    result = optimize.minimize(
        fun=objective,
        x0=initial_guess,
        args=(centers,),
        method='SLSQP',
        constraints=constraints(initial_guess, centers),
        bounds=[(0.0, 0.2) for _ in range(num_circles)]
    )

    # Extract optimized radii and centers
    optimized_radii = result.x
    optimized_centers = centers
    sum_radii = np.sum(optimized_radii)

    # Validate the packing
    is_valid, message = validate_packing(optimized_centers, optimized_radii)
    if not is_valid:
        raise ValueError(f"Invalid packing: {message}")

    return optimized_centers, optimized_radii, sum_radii
```

```python
def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
```
