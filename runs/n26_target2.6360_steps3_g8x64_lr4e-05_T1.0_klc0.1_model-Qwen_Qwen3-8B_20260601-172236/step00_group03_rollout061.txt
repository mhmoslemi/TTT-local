import numpy as np
import scipy.optimize as optimize

def run_packing() -> tuple[np.ndarray, np.ndarray, float]:

    num_circles = 26
    row_size = 6  # First row has 6 circles, then alternating
    radii = np.full(num_circles, 0.1)  # Initial guess for radii
    centers = np.zeros((num_circles, 2))

    # Define hexagonal packing grid
    radius = 0.1
    for i in range(num_circles):
        row = i // row_size
        col = i % row_size
        x = col * (2 * radius + 1e-6)  # Spacing between centers
        y = row * (np.sqrt(3) * radius + 1e-6)
        centers[i, 0] = x
        centers[i, 1] = y

    def objective(radii):
        return -np.sum(radii)  # Maximize sum of radii by minimizing negative sum

    def constraint_center_bounds(radii):

            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12 or y - r < -1e-12 or y + r > 1 + 1e-12:
                return -np.inf
        return 0

    def constraint_circle_distances(radii):

                dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
                if dist < radii[i] + radii[j] - 1e-12:
                    return -np.inf
        return 0

        {'type': 'ineq', 'fun': constraint_circle_distances}
    ]

        objective,
        radii,
        method='SLSQP',
        bounds=bounds,
        constraints=cons,
        tol=1e-8
    )

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
