import numpy as np
import scipy.optimize as optimize
from scipy.optimize import Bounds, LinearConstraint

def run_packing():
    # Initial grid-based packing
    n = 26
    num_rows = int(np.ceil(np.sqrt(n)))
    num_cols = int(np.ceil(n / num_rows))
    centers = np.zeros((n, 2))
    radii = np.zeros(n)

    # Uniform spacing
    spacing = 1.0 / max(num_rows, num_cols)
    for i in range(n):
        row = i // num_cols
        col = i % num_cols
        x = col * spacing + spacing / 2
        y = row * spacing + spacing / 2
        centers[i, 0] = x
        centers[i, 1] = y
        radii[i] = spacing / 2 - 1e-5  # Adjust to avoid overflow

    # Optimization bounds and constraints
    def objective(radii):
        return -np.sum(radii)  # Minimize negative sum to maximize sum

    def constraint_overlap(i, j, radii):
        dx = centers[i, 0] - centers[j, 0]
        dy = centers[i, 1] - centers[j, 1]
        dist = np.sqrt(dx ** 2 + dy ** 2)
        return dist - (radii[i] + radii[j])  # >= 0

    def constraint_boundary(i, radii):
        x, y = centers[i]
        return x - radii[i] >= -1e-12 and x + radii[i] <= 1 + 1e-12 and y - radii[i] >= -1e-12 and y + radii[i] <= 1 + 1e-12

    # Initial guess
    initial_radii = np.ones(n) * (1.0 / max(num_rows, num_cols) / 2)

    # Bounds: radii >= 0
    bounds = Bounds(0, np.inf)

    # Optimization
    result = optimize.minimize(
        fun=objective,
        x0=initial_radii,
        bounds=bounds,
        method='SLSQP',
        constraints=[
            LinearConstraint(lambda x: np.array([constraint_overlap(i, j, x) for j in range(i+1, n) for i in range(n)]), 
                             lb=0, ub=np.inf),
            LinearConstraint(lambda x: np.array([constraint_boundary(i, x) for i in range(n)]), 
                             lb=0, ub=np.inf)
        ]
    )

    # Extract final results
    final_radii = result.x
    final_centers = centers
    final_sum_radii = np.sum(final_radii)
