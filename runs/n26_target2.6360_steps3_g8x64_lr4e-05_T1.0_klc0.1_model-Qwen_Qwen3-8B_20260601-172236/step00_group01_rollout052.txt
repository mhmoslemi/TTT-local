import numpy as np
import scipy.optimize as optimize
import cvxpy as cp

def run_packing():
    # Define target number of circles and packing density
    n_circles = 26
    
    # Generate an initial hexagonal-like grid
    def generate_hexagonal_grid(num_circles, max_x=1.0, max_y=1.0):
        circles = []
        # Approximate number of rows based on hexagonal packing
        num_rows = int(np.ceil(np.sqrt(num_circles)))
        num_cols = int(np.ceil(num_circles / num_rows))
        spacing = 1.0 / (num_cols + 1)
        
        for i in range(num_rows):
            for j in range(num_cols):
                x = j * spacing + spacing * 0.5
                y = i * spacing * np.sqrt(3) + spacing * 0.5
                if x <= max_x and y <= max_y:
                    circles.append((x, y))
        
        # If we have fewer than num_circles, add more (in a grid pattern)
        while len(circles) < num_circles:
            x = np.random.uniform(0, 1)
            y = np.random.uniform(0, 1)
            circles.append((x, y))
        
        return np.array(circles)

    # Initial grid
    centers = generate_hexagonal_grid(n_circles)
    radii = np.full(n_circles, 0.02)  # Start with small radii

    # Objective: maximize total radius sum
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints: no overlap, within bounds
    def constraints(centers, radii_vec):
        constraints = []
        # Bounds: all circles must be within [0,1]x[0,1]
        for i in range(n_circles):
            x, y = centers[i]
            r = radii_vec[i]
            constraints.append(r <= 1 - x)
            constraints.append(r <= 1 - y)
            constraints.append(r <= x)
            constraints.append(r <= y)
        
        # No overlap between circles
        for i in range(n_circles):
            for j in range(i + 1, n_circles):
                x1, y1 = centers[i]
                x2, y2 = centers[j]
                r1 = radii_vec[i]
                r2 = radii_vec[j]
                dist = np.sqrt((x1 - x2)**2 + (y1 - y2)**2)
                constraints.append(dist >= r1 + r2 - 1e-6)
        
        return constraints

    # Define variables
    radii_var = cp.Variable(n_circles)
    constraints_list = constraints(centers, radii_var)

    # Solve with CVXPY
    prob = cp.Problem(cp.Maximize(objective(radii_var)), constraints_list)
    result = prob.solve()

    # Extract results
    optimal_radii = radii_var.value
    optimal_centers = centers

    # Check constraints (just to be safe)

    if not valid:
        raise ValueError("Optimization failed to satisfy constraints")

    total_sum = np.sum(optimal_radii)
    return optimal_centers, optimal_radii, total_sum

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
