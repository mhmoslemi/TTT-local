import scipy.optimize as optimize

def run_packing():

    # Grid size for hexagonal packing
    n_rows = 6
    n_cols = 5
    circles_per_row = n_cols
    hex_grid = np.zeros((26, 2))
    spacing = 1.0 / (n_rows + 1)

        for j in range(circles_per_row):
            x = j * spacing * 1.5 + spacing / 2
            y = i * spacing * np.sqrt(3) + spacing / 2
            if i % 2 == 1:
                x += spacing / 2
            hex_grid[i * circles_per_row + j] = [x, y]

    # Initial guess
    centers = hex_grid
    radii = np.full(26, 0.04)  # Initial radius value

    # Objective function: maximize sum of radii (minimize negative sum)
    def objective(radii_vec):
        return -np.sum(radii_vec)

    # Constraints
    def constraint_overlap(centers, radii):
        n = len(radii)
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx**2 + dy**2)
                if dist < radii[i] + radii[j] - 1e-12:
                    return dist - (radii[i] + radii[j] - 1e-12)
        return 0

    def constraint_boundary(centers, radii):
        for i in range(len(radii)):
            x, y = centers[i]
            r = radii[i]
            if x - r < -1e-12 or x + r > 1 + 1e-12:
                return x - r - (-1e-12)
            if y - r < -1e-12 or y + r > 1 + 1e-12:

    result = optimize.minimize(

        x0=radii,
        bounds=bounds,
        constraints=constraints,
