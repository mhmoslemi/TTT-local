def run_packing():
    # Number of circles
    n = 26

    # Start with a hexagonal grid configuration
    rows = int(np.ceil(np.sqrt(n)))
    cols = int(np.ceil(n / rows))

    # Initial grid positions
    centers = np.zeros((n, 2))
    radii = np.ones(n) * 0.1  # Start with small radii

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col + 0.5 * (row % 2)
        y = row * np.sqrt(3) / 2
        centers[i, 0] = x
        centers[i, 1] = y

    # Normalize to unit square
    centers[:, 0] /= max(centers[:, 0])
    centers[:, 1] /= max(centers[:, 1])

    # Optimization variables: radii
    # Define radii as a vector
    x = radii.copy()

    # Objective: maximize sum of radii
    def objective(x):
        return -np.sum(x)  # Minimize negative sum

    # Constraints: circles do not overlap, and are within unit square

    def constraints(x):
        # Boundary constraints: x[i] must be within [0, 1] and y[i] must be within [0, 1]
        # Check if circle i is within the square
        for i in range(n):
            x_center, y_center = centers[i]
            r = x[i]
            if x_center - r < 0 or x_center + r > 1 or y_center - r < 0 or y_center + r > 1:
                return False

                if dist < x[i] + x[j] - 1e-12:
                    return False
        return True

        'maxiter': 1000
    }

        x,
        method='SLSQP',
        bounds=bounds,
        constraints=({'type': 'ineq', 'fun': constraints}),

def validate_packing(centers, radii):
    n = centers.shape[0]

    if np.isnan(centers).any() or np.isnan(radii).any():
        return False, "NaN values present"

    for i in range(n):
        if radii[i] < 0:
            return False, f"Circle {i} has negative radius {radii[i]}"

    for i in range(n):
        x, y = centers[i]
        r = radii[i]
        if (x - r < -1e-12 or x + r > 1 + 1e-12
                or y - r < -1e-12 or y + r > 1 + 1e-12):
            return False, f"Circle {i} at ({x},{y}) r={r} outside unit square"

    for i in range(n):
        for j in range(i + 1, n):
            dist = np.sqrt(np.sum((centers[i] - centers[j]) ** 2))
            if dist < radii[i] + radii[j] - 1e-12:
                return False, f"Circles {i} and {j} overlap"

    return True, "ok"
