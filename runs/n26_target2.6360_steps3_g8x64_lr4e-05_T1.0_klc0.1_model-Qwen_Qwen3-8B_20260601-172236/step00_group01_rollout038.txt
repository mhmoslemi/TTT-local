import math

def generate_initial_packing(num_circles):

def compute_overlap_energy(centers, radii, eps=1e-8):
    n = centers.shape[0]
    energy = 0.0
    for i in range(n):
        for j in range(i + 1, n):
            dx = centers[i, 0] - centers[j, 0]
            dy = centers[i, 1] - centers[j, 1]
            dist = math.sqrt(dx * dx + dy * dy)
            if dist < radii[i] + radii[j] - eps:
                overlap = radii[i] + radii[j] - dist
                energy += overlap
    return energy
